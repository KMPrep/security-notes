##  Validate Email Format: Write a function to check if a string is a valid email (basic: contains '@' and '.').

import re

str_regex = r"\b[0-9A-Za-z]+@[0-9A-Za-z]+\.[A-Za-z]{2,4}\b"

def get_emailaddr(input):
      regex_compiled = re.compile(str_regex)
      if regex_compiled.fullmatch(input) is not None:
            return True
      return False

print(f"Is Valid Email: { get_emailaddr("test@test.com")}")


# --- Manual / no-built-in solution ---
# No re module — scan character by character.

_ALPHANUM = "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789"


def get_emailaddr_manual(email: str) -> bool:
    at_count, at_pos = 0, -1
    for i in range(len(email)):
        if email[i] == '@':
            at_count += 1
            at_pos = i
    if at_count != 1 or at_pos == 0 or at_pos == len(email) - 1:
        return False
    local  = email[:at_pos]
    domain = email[at_pos + 1:]
    for ch in local:
        if ch not in _ALPHANUM:
            return False
    dot_pos = -1
    for i in range(len(domain)):
        if domain[i] == '.':
            dot_pos = i
        elif domain[i] not in _ALPHANUM:
            return False
    if dot_pos <= 0 or dot_pos == len(domain) - 1:
        return False
    tld_len = len(domain) - dot_pos - 1
    return 2 <= tld_len <= 4


print(f"Is Valid Email (manual): {get_emailaddr_manual('test@test.com')}")   # True
print(f"Is Valid Email (manual): {get_emailaddr_manual('bad@')}")             # False
