# Q10 - Find Intersection
# Find common elements between two lists, preserving order from the first list.
#
# Approach:
#   - Convert the second list to a set for O(1) lookups.
#   - Iterate the first list and keep elements present in the set.
#   - Time: O(m+n)  |  Space: O(n)  (n = len of second list)
#
# Edge cases:
#   - No common elements   -> []
#   - Duplicates in first  -> each duplicate preserved if it's also in second
#   - Either list empty    -> []


def intersection(first: list, second: list) -> list:
    second_set = set(second)
    return [x for x in first if x in second_set]


# --- tests ---
assert intersection([1, 2, 3, 4, 5], [3, 4, 5, 6, 7])   == [3, 4, 5]
assert intersection([1, 2, 2, 3], [2, 3, 4])             == [2, 2, 3]
assert intersection([], [1, 2, 3])                       == []
assert intersection([1, 2, 3], [])                       == []
assert intersection([1, 2, 3], [4, 5, 6])                == []

print(intersection([1, 2, 3, 4, 5], [3, 4, 5, 6, 7]))   # [3, 4, 5]
print(intersection([1, 2, 2, 3], [2, 3, 4]))             # [2, 2, 3]


# --- Manual / no-built-in solution ---
# No set() — use a dict as a hash-lookup table for the second list.


def intersection_manual(first: list, second: list) -> list:
    lookup = {}
    for x in second:
        lookup[x] = True
    result = []
    for x in first:
        if x in lookup:
            result.append(x)
    return result


assert intersection_manual([1, 2, 3, 4, 5], [3, 4, 5, 6, 7])   == [3, 4, 5]
assert intersection_manual([1, 2, 2, 3], [2, 3, 4])             == [2, 2, 3]
assert intersection_manual([], [1, 2, 3])                       == []
assert intersection_manual([1, 2, 3], [])                       == []
print(intersection_manual([1, 2, 3, 4, 5], [3, 4, 5, 6, 7]))   # [3, 4, 5]
print(intersection_manual([1, 2, 2, 3], [2, 3, 4]))             # [2, 2, 3]
