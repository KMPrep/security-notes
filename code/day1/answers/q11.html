# Q11 - Word Frequency
# Count the frequency of each word in a sentence.
#
# Approach:
#   - Normalize to lowercase, strip punctuation, split on whitespace.
#   - Use collections.Counter for clean O(n) counting.
#
# Security note: always sanitize/normalize input before processing to avoid
# frequency-analysis inconsistencies (e.g. "Password" vs "password").
#
# Edge cases:
#   - Empty string       -> {}
#   - Punctuation attached to words  -> stripped


import re
from collections import Counter


def word_frequency(sentence: str) -> dict[str, int]:
    # Remove non-alpha characters (keep spaces), lowercase, split
    words = re.findall(r"[a-z]+", sentence.lower())
    return dict(Counter(words))


# --- tests ---
assert word_frequency("the cat sat on the mat") == {
    "the": 2, "cat": 1, "sat": 1, "on": 1, "mat": 1
}
assert word_frequency("Hello, hello! HELLO.")   == {"hello": 3}
assert word_frequency("")                        == {}

result = word_frequency("the cat sat on the mat")
print(result)   # {'the': 2, 'cat': 1, 'sat': 1, 'on': 1, 'mat': 1}
print(word_frequency("Hello, hello! HELLO."))   # {'hello': 3}


# --- Manual / no-built-in solution ---
# No re, Counter, or lower() — manual char scan with ord/chr for case conversion.


def word_frequency_manual(sentence: str) -> dict:
    counts, word = {}, ""
    for ch in sentence:
        lch = chr(ord(ch) + 32) if 'A' <= ch <= 'Z' else ch
        if 'a' <= lch <= 'z':
            word += lch
        elif word:
            counts[word] = counts[word] + 1 if word in counts else 1
            word = ""
    if word:
        counts[word] = counts[word] + 1 if word in counts else 1
    return counts


assert word_frequency_manual("the cat sat on the mat") == {
    "the": 2, "cat": 1, "sat": 1, "on": 1, "mat": 1
}
assert word_frequency_manual("Hello, hello! HELLO.") == {"hello": 3}
assert word_frequency_manual("") == {}
print(word_frequency_manual("the cat sat on the mat"))  # {'the': 2, ...}
print(word_frequency_manual("Hello, hello! HELLO."))    # {'hello': 3}
