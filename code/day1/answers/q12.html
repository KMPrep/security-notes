# Q12 - Group by Key
# Given a list of (key, value) tuples, group values by key.
# e.g.  [("a",1),("b",2),("a",3)]  ->  {"a":[1,3], "b":[2]}
#
# Approach:
#   - collections.defaultdict(list) avoids manual key-existence checks.
#   - Time: O(n)  |  Space: O(n)
#
# Edge cases:
#   - Empty list         -> {}
#   - Single tuple       -> {key: [value]}


from collections import defaultdict


def group_by_key(pairs: list[tuple]) -> dict:
    groups: dict = defaultdict(list)
    for key, value in pairs:
        groups[key].append(value)
    return dict(groups)


# --- tests ---
data = [("a", 1), ("b", 2), ("a", 3), ("c", 4), ("b", 5)]
assert group_by_key(data)             == {"a": [1, 3], "b": [2, 5], "c": [4]}
assert group_by_key([])               == {}
assert group_by_key([("x", 10)])      == {"x": [10]}

print(group_by_key(data))             # {'a': [1, 3], 'b': [2, 5], 'c': [4]}


# --- Manual / no-built-in solution ---
# No defaultdict — manual key-existence check before appending.


def group_by_key_manual(pairs: list) -> dict:
    groups = {}
    for key, value in pairs:
        if key not in groups:
            groups[key] = []
        groups[key].append(value)
    return groups


assert group_by_key_manual(data)         == {"a": [1, 3], "b": [2, 5], "c": [4]}
assert group_by_key_manual([])           == {}
assert group_by_key_manual([("x", 10)]) == {"x": [10]}
print(group_by_key_manual(data))         # {'a': [1, 3], 'b': [2, 5], 'c': [4]}
