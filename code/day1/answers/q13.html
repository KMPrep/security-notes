# Q13 - Unique Elements
# Use a set to find unique elements in a list (return as sorted list).
#
# Approach:
#   - set() deduplicates in O(n); sorted() for deterministic output.
#   - If order-preservation matters, use dict.fromkeys() instead.
#
# Edge cases:
#   - Empty list      -> []
#   - All same        -> [single element]


def unique_elements(items: list) -> list:
    return sorted(set(items))


def unique_elements_ordered(items: list) -> list:
    """Preserves first-seen insertion order."""
    return list(dict.fromkeys(items))


# --- tests ---
assert unique_elements([3, 1, 2, 1, 3, 4])         == [1, 2, 3, 4]
assert unique_elements([])                          == []
assert unique_elements([5, 5, 5])                   == [5]

assert unique_elements_ordered([3, 1, 2, 1, 3, 4]) == [3, 1, 2, 4]

print(unique_elements([3, 1, 2, 1, 3, 4]))           # [1, 2, 3, 4]
print(unique_elements_ordered([3, 1, 2, 1, 3, 4]))   # [3, 1, 2, 4]


# --- Manual / no-built-in solution ---
# No sorted(), set(), or dict.fromkeys() — seen-dict dedup + insertion sort.


def _insertion_sort(lst: list) -> list:
    arr = []
    for x in lst:          # manual copy
        arr.append(x)
    for i in range(1, len(arr)):
        key, j = arr[i], i - 1
        while j >= 0 and arr[j] > key:
            arr[j + 1] = arr[j]
            j -= 1
        arr[j + 1] = key
    return arr


def unique_elements_manual(items: list) -> list:
    seen, uniq = {}, []
    for item in items:
        if item not in seen:
            seen[item] = True
            uniq.append(item)
    return _insertion_sort(uniq)


def unique_elements_ordered_manual(items: list) -> list:
    seen, result = {}, []
    for item in items:
        if item not in seen:
            seen[item] = True
            result.append(item)
    return result


assert unique_elements_manual([3, 1, 2, 1, 3, 4])          == [1, 2, 3, 4]
assert unique_elements_manual([])                          == []
assert unique_elements_manual([5, 5, 5])                   == [5]
assert unique_elements_ordered_manual([3, 1, 2, 1, 3, 4]) == [3, 1, 2, 4]
print(unique_elements_manual([3, 1, 2, 1, 3, 4]))           # [1, 2, 3, 4]
print(unique_elements_ordered_manual([3, 1, 2, 1, 3, 4]))   # [3, 1, 2, 4]
