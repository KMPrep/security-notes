# Q14 - Phone Book Lookup
# Implement a simple phone book using a dict: add and lookup operations.
#
# Approach:
#   - Encapsulate in a class for clean interface (clean OOP is expected).
#   - add()    -> O(1) amortized
#   - lookup() -> O(1)
#
# Security note: In a real system, phone numbers should be stored hashed or
# access-controlled; this is a simplified in-memory demo.
#
# Edge cases:
#   - Lookup non-existent name  -> return None / explicit "not found"
#   - Overwrite existing entry  -> latest number wins


class PhoneBook:
    def __init__(self):
        self._book: dict[str, str] = {}

    def add(self, name: str, number: str) -> None:
        """Add or update a contact."""
        self._book[name.strip().lower()] = number

    def lookup(self, name: str) -> str | None:
        """Return phone number or None if not found."""
        return self._book.get(name.strip().lower())

    def remove(self, name: str) -> bool:
        """Remove a contact; returns True if it existed."""
        return self._book.pop(name.strip().lower(), None) is not None

    def __len__(self) -> int:
        return len(self._book)


# --- tests ---
pb = PhoneBook()
pb.add("Alice", "555-1234")
pb.add("Bob",   "555-5678")

assert pb.lookup("Alice")   == "555-1234"
assert pb.lookup("alice")   == "555-1234"   # case-insensitive
assert pb.lookup("Charlie") is None
assert len(pb)              == 2

pb.add("Alice", "555-9999")               # overwrite
assert pb.lookup("Alice")   == "555-9999"

assert pb.remove("Bob")     == True
assert pb.lookup("Bob")     is None
assert pb.remove("Bob")     == False       # already removed

print(pb.lookup("Alice"))   # 555-9999
print(pb.lookup("Bob"))     # None


# --- Manual / no-built-in solution ---
# No dict.get(), dict.pop(), str.strip(), or str.lower() — all implemented manually.


def _manual_strip(s: str) -> str:
    start = 0
    while start < len(s) and s[start] == ' ':
        start += 1
    end = len(s) - 1
    while end >= start and s[end] == ' ':
        end -= 1
    result = ""
    for i in range(start, end + 1):
        result += s[i]
    return result


def _manual_lower(s: str) -> str:
    result = ""
    for ch in s:
        result += chr(ord(ch) + 32) if 'A' <= ch <= 'Z' else ch
    return result


def _normalize(name: str) -> str:
    return _manual_lower(_manual_strip(name))


class PhoneBookManual:
    def __init__(self):
        self._book = {}

    def add(self, name: str, number: str) -> None:
        self._book[_normalize(name)] = number

    def lookup(self, name: str):
        key = _normalize(name)
        return self._book[key] if key in self._book else None

    def remove(self, name: str) -> bool:
        key = _normalize(name)
        if key in self._book:
            del self._book[key]
            return True
        return False

    def __len__(self) -> int:
        count = 0
        for _ in self._book:
            count += 1
        return count

pb2.add("Alice", "555-1234")
