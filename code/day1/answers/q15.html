# Q15 - Set Operations
# Find the union, intersection, and difference of two sets.
#
# Approach:
#   - Use Python's built-in set operators: |, &, -
#   - Also show the equivalent method calls (good to know both).
#   - All operations: O(m+n)
#
# Edge cases:
#   - Empty sets
#   - Disjoint sets (intersection is empty)
#   - Identical sets (difference is empty)


def set_operations(a: set, b: set) -> dict[str, set]:
    return {
        "union":        a | b,           # all elements from both
        "intersection": a & b,           # only elements in both
        "diff_a_b":     a - b,           # in a but not b
        "diff_b_a":     b - a,           # in b but not a
        "sym_diff":     a ^ b,           # in one but not both
    }


# --- tests ---
A = {1, 2, 3, 4}
B = {3, 4, 5, 6}

ops = set_operations(A, B)
assert ops["union"]        == {1, 2, 3, 4, 5, 6}
assert ops["intersection"] == {3, 4}
assert ops["diff_a_b"]     == {1, 2}
assert ops["diff_b_a"]     == {5, 6}
assert ops["sym_diff"]     == {1, 2, 5, 6}

# Empty set edge cases
empty_ops = set_operations(set(), {1, 2})
assert empty_ops["union"]        == {1, 2}
assert empty_ops["intersection"] == set()
assert empty_ops["diff_a_b"]     == set()

for name, result in set_operations(A, B).items():
    print(f"{name:15}: {sorted(result)}")


# --- Manual / no-built-in solution ---
# No set operators (|, &, -, ^) — iterate and build result dicts, then wrap in set().


def set_operations_manual(a: set, b: set) -> dict:
    union_d, inter_d, diff_ab, diff_ba, sym_d = {}, {}, {}, {}, {}
    for x in a:
        union_d[x] = True
    for x in b:
        union_d[x] = True
    for x in a:
        if x in b:
            inter_d[x] = True
    for x in a:
        if x not in b:
            diff_ab[x] = True
    for x in b:
        if x not in a:
            diff_ba[x] = True
    for x in a:
        if x not in b:
            sym_d[x] = True
    for x in b:
        if x not in a:
            sym_d[x] = True
    return {
        "union":        set(union_d),
        "intersection": set(inter_d),
        "diff_a_b":     set(diff_ab),
        "diff_b_a":     set(diff_ba),
        "sym_diff":     set(sym_d),
    }


ops2 = set_operations_manual(A, B)
assert ops2["union"]        == {1, 2, 3, 4, 5, 6}
assert ops2["intersection"] == {3, 4}
assert ops2["diff_a_b"]     == {1, 2}
assert ops2["diff_b_a"]     == {5, 6}
assert ops2["sym_diff"]     == {1, 2, 5, 6}
for name, result in set_operations_manual(A, B).items():
    print(f"{name:15}: {result}")
