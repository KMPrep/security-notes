# Q16 - FizzBuzz
# Print 1 to 100: "Fizz" for multiples of 3, "Buzz" for 5, "FizzBuzz" for both.
#
# Approach:
#   - Check divisibility by 15 first to catch FizzBuzz without string concat.
#   - Alternative (idiomatic): build result by concatenation — scales to more
#     divisors without adding elif branches.
#
# Edge cases:
#   - n=1  -> just "1"
#   - Extensible to arbitrary divisor/label pairs (shown in fizzbuzz_general)


def fizzbuzz(n: int = 100) -> list[str]:
    result = []
    for i in range(1, n + 1):
        if i % 15 == 0:
            result.append("FizzBuzz")
        elif i % 3 == 0:
            result.append("Fizz")
        elif i % 5 == 0:
            result.append("Buzz")
        else:
            result.append(str(i))
    return result


# Extensible version — note this to show design thinking
def fizzbuzz_general(n: int, rules: list[tuple[int, str]]) -> list[str]:
    result = []
    for i in range(1, n + 1):
        label = "".join(word for div, word in rules if i % div == 0)
        result.append(label or str(i))
    return result


# --- tests ---
fb = fizzbuzz(15)
assert fb[2]   == "Fizz"       # 3
assert fb[4]   == "Buzz"       # 5
assert fb[14]  == "FizzBuzz"   # 15
assert fb[0]   == "1"
assert fb[1]   == "2"

gen = fizzbuzz_general(15, [(3, "Fizz"), (5, "Buzz")])
assert gen == fb

# Print first 20
print(fizzbuzz(20))


# --- Manual / no-built-in solution ---
# No str.join() in the general version — use string concatenation.


def fizzbuzz_general_manual(n: int, rules: list) -> list:
    result = []
    for i in range(1, n + 1):
        label = ""
        for div, word in rules:
            if i % div == 0:
                label += word
        result.append(label if label else str(i))
    return result


gen_manual = fizzbuzz_general_manual(15, [(3, "Fizz"), (5, "Buzz")])
assert gen_manual == fizzbuzz(15)
print(fizzbuzz_general_manual(20, [(3, "Fizz"), (5, "Buzz")]))
