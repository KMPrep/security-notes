# Q18 - Sum of Evens
# Sum all even numbers in a list.
#
# Approach:
#   - Generator expression inside sum() — single pass, O(n), no extra list.
#   - Equivalent list-comprehension version shown for comparison.
#
# Edge cases:
#   - Empty list        -> 0
#   - No even numbers   -> 0
#   - Negative evens    -> included (even means divisible by 2)


def sum_evens(nums: list[int]) -> int:
    return sum(x for x in nums if x % 2 == 0)


# --- tests ---
assert sum_evens([1, 2, 3, 4, 5, 6])   == 12   # 2+4+6
assert sum_evens([1, 3, 5, 7])         == 0
assert sum_evens([])                   == 0
assert sum_evens([-4, -2, 1, 3])       == -6    # negatives count

print(sum_evens([1, 2, 3, 4, 5, 6]))   # 12
print(sum_evens([-4, -2, 1, 3]))        # -6


# --- Manual / no-built-in solution ---
# No sum() — manual accumulation loop.


def sum_evens_manual(nums: list) -> int:
    total = 0
    for x in nums:
        if x % 2 == 0:
            total += x
    return total


assert sum_evens_manual([1, 2, 3, 4, 5, 6]) == 12
assert sum_evens_manual([1, 3, 5, 7])       == 0
assert sum_evens_manual([])                 == 0
assert sum_evens_manual([-4, -2, 1, 3])     == -6
print(sum_evens_manual([1, 2, 3, 4, 5, 6])) # 12
print(sum_evens_manual([-4, -2, 1, 3]))      # -6
