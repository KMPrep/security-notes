# Q19 - Max in List
# Find the maximum value in a list without using the built-in max().
#
# Approach:
#   - Single pass, track running maximum.
#   - Time: O(n)  |  Space: O(1)
#
# Edge cases:
#   - Empty list         -> raise ValueError (mirrors built-in behaviour)
#   - Single element     -> that element
#   - All negatives      -> works correctly (seed with first element, not 0)
#   - Duplicate max      -> first occurrence tracked, value still correct


def find_max(nums: list) -> int:
    if not nums:
        raise ValueError("find_max() arg is an empty sequence")
    current_max = nums[0]
    for num in nums[1:]:
        if num > current_max:
            current_max = num
    return current_max


# --- tests ---
assert find_max([3, 1, 4, 1, 5, 9, 2, 6])  == 9
assert find_max([42])                       == 42
assert find_max([-10, -3, -7])              == -3
assert find_max([0, 0, 0])                  == 0

try:
    find_max([])
    assert False, "should raise"
except ValueError:
    pass

print(find_max([3, 1, 4, 1, 5, 9, 2, 6]))  # 9
print(find_max([-10, -3, -7]))              # -3

# Note: this solution is already the manual / no-built-in implementation.
# It tracks the running maximum with a single pass — no max() used.
