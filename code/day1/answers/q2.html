# Count Special Characters: Given a string, count how many non-alphanumeric characters it contains.
# Note: this solution is already manual (no built-in functions used).

_ALPHA_NUMERIC = "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789"

def count_non_alphanumeric_chars(input: str):
    count = 0
    for char in input:
        if char not in _ALPHA_NUMERIC:
            count += 1
    return count
        
print(f"Here is count for the non alphanumeric count: {count_non_alphanumeric_chars("yyy@#qqq")}" )
