# Q20 - Palindrome List
# Check if a list reads the same forwards and backwards.
#
# Approach:
#   - Two-pointer technique: compare from both ends moving inward.
#   - O(n) time, O(1) extra space (no copy of the list needed).
#   - Simpler slice check  lst == lst[::-1]  is also O(n) but O(n) space.
#
# Edge cases:
#   - Empty list      -> True  (vacuously palindrome)
#   - Single element  -> True
#   - Even length     -> works (pointers meet in the middle)


def is_palindrome_list(lst: list) -> bool:
    left, right = 0, len(lst) - 1
    while left < right:
        if lst[left] != lst[right]:
            return False
        left  += 1
        right -= 1
    return True


# --- tests ---
assert is_palindrome_list([1, 2, 3, 2, 1])  == True
assert is_palindrome_list([1, 2, 2, 1])     == True
assert is_palindrome_list([1, 2, 3])        == False
assert is_palindrome_list([])               == True
assert is_palindrome_list([42])             == True
assert is_palindrome_list(["a", "b", "a"]) == True

print(is_palindrome_list([1, 2, 3, 2, 1]))  # True
print(is_palindrome_list([1, 2, 3]))         # False

# Note: this solution is already the manual / no-built-in implementation.
# It uses two-pointer comparison — no slice reverse (lst[::-1]) used.
