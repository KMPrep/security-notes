# Q21 - Square Numbers
# Use list comprehension to square all numbers in a list.
#
# Approach:
#   - One-liner list comprehension: clearest Pythonic form.
#   - Also show map() equivalent — worth knowing.
#
# Edge cases:
#   - Empty list      -> []
#   - Negative nums   -> still squared (positive result)
#   - Floats          -> works (x**2 generalizes)


def square_numbers(nums: list[int | float]) -> list[int | float]:
    return [x ** 2 for x in nums]


# map() alternative (lazy iterator, convert to list)
def square_numbers_map(nums: list[int | float]) -> list[int | float]:
    return list(map(lambda x: x ** 2, nums))


# --- tests ---
assert square_numbers([1, 2, 3, 4, 5])     == [1, 4, 9, 16, 25]
assert square_numbers([-3, -2, 0, 2, 3])   == [9, 4, 0, 4, 9]
assert square_numbers([])                  == []

assert square_numbers_map([1, 2, 3])       == [1, 4, 9]

print(square_numbers([1, 2, 3, 4, 5]))     # [1, 4, 9, 16, 25]
print(square_numbers([-3, -2, 0, 2, 3]))   # [9, 4, 0, 4, 9]


# --- Manual / no-built-in solution ---
# No map() — explicit loop with multiplication operator.


def square_numbers_manual(nums: list) -> list:
    result = []
    for x in nums:
        result.append(x * x)
    return result


assert square_numbers_manual([1, 2, 3, 4, 5])     == [1, 4, 9, 16, 25]
assert square_numbers_manual([-3, -2, 0, 2, 3])   == [9, 4, 0, 4, 9]
assert square_numbers_manual([])                  == []
print(square_numbers_manual([1, 2, 3, 4, 5]))     # [1, 4, 9, 16, 25]
print(square_numbers_manual([-3, -2, 0, 2, 3]))   # [9, 4, 0, 4, 9]
