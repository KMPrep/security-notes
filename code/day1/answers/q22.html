# Q22 - Filter Strings
# Filter a list of strings to keep only those starting with a vowel.
#
# Approach:
#   - Set membership check for O(1) vowel lookup per string.
#   - List comprehension for readability.
#   - Case-insensitive: check first char lowercased.
#
# Edge cases:
#   - Empty list          -> []
#   - Empty strings       -> skipped (no first character)
#   - Mixed case first    -> handled by .lower()


VOWELS = frozenset("aeiou")


def filter_vowel_start(words: list[str]) -> list[str]:
    return [w for w in words if w and w[0].lower() in VOWELS]


# --- tests ---
assert filter_vowel_start(["apple", "banana", "orange", "grape", "avocado"]) \
    == ["apple", "orange", "avocado"]
assert filter_vowel_start(["Echo", "India", "Alpha", "Bravo"])               \
    == ["Echo", "India", "Alpha"]
assert filter_vowel_start([])                                                 \
    == []
assert filter_vowel_start(["", "apple", "  egg"])                            \
    == ["apple"]   # empty string skipped; " egg" starts with space

print(filter_vowel_start(["apple", "banana", "orange", "grape", "avocado"]))
# ['apple', 'orange', 'avocado']


# --- Manual / no-built-in solution ---
# No frozenset — check first char against a plain string of vowels.


_VOWELS_STR = "aeiou"


def _is_vowel(ch: str) -> bool:
    lch = chr(ord(ch) + 32) if 'A' <= ch <= 'Z' else ch
    for v in _VOWELS_STR:
        if lch == v:
            return True
    return False


def filter_vowel_start_manual(words: list) -> list:
    result = []
    for w in words:
        if w and _is_vowel(w[0]):
            result.append(w)
    return result


assert filter_vowel_start_manual(["apple", "banana", "orange", "grape", "avocado"]) \
    == ["apple", "orange", "avocado"]
assert filter_vowel_start_manual(["Echo", "India", "Alpha", "Bravo"]) \
    == ["Echo", "India", "Alpha"]
assert filter_vowel_start_manual([]) == []
print(filter_vowel_start_manual(["apple", "banana", "orange", "grape", "avocado"]))
# ['apple', 'orange', 'avocado']
