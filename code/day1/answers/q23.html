# Q23 - Function Composition
# Write a function that takes another function and applies it twice.
#
# Approach:
#   - apply_twice(f, x) = f(f(x))
#   - Also show a general compose() that chains any number of functions —
#     demonstrates higher-order function thinking valued in production code.
#
# Edge cases:
#   - Identity function  -> same value returned
#   - Side-effectful f   -> called exactly twice


from functools import reduce


def apply_twice(f, x):
    """Apply f twice: f(f(x))."""
    return f(f(x))


def compose(*funcs):
    """Right-to-left composition: compose(f, g)(x) = f(g(x))."""
    return reduce(lambda f, g: lambda x: f(g(x)), funcs)


# --- tests ---
double      = lambda x: x * 2
add_one     = lambda x: x + 1
to_upper    = lambda s: s.upper()

assert apply_twice(double, 3)         == 12       # double(double(3)) = 12
assert apply_twice(add_one, 5)        == 7        # add_one(add_one(5)) = 7
assert apply_twice(to_upper, "hello") == "HELLO"  # idempotent

double_then_add = compose(add_one, double)        # add_one(double(x))
assert double_then_add(4)             == 9        # add_one(8) = 9

print(apply_twice(double, 3))         # 12
print(apply_twice(add_one, 5))        # 7
print(double_then_add(4))             # 9


# --- Manual / no-built-in solution ---
# No functools.reduce — right-to-left iteration through the funcs tuple.


def compose_manual(*funcs):
    """Right-to-left composition without reduce."""
    def composed(x):
        i = len(funcs) - 1
        result = x
        while i >= 0:
            result = funcs[i](result)
            i -= 1
        return result
    return composed


double_then_add_manual = compose_manual(add_one, double)
assert double_then_add_manual(4) == 9   # add_one(double(4)) = 9
assert double_then_add_manual(0) == 1   # add_one(double(0)) = 1
print(double_then_add_manual(4))        # 9
