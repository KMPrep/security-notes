# Q24 - Default Arguments
# Write a function with default arguments for logging levels.
#
# Approach:
#   - Default mutable arguments are a classic Python trap: never use
#     def log(msg, tags=[]) — use None and assign inside the body.
#   - Use an Enum for log levels to avoid magic strings.
#
# Security note: log functions should sanitize output to prevent log injection
# (e.g., a newline in `msg` could fake extra log entries).


import datetime
from enum import Enum


class LogLevel(Enum):
    DEBUG   = 10
    INFO    = 20
    WARNING = 30
    ERROR   = 40
    CRITICAL= 50


def log(
    message: str,
    level:  LogLevel = LogLevel.INFO,
    prefix: str      = "APP",
    tags:   list | None = None,        # NOT [] — avoids mutable default trap
) -> str:
    """Format a log entry; returns the formatted string (and prints it)."""
    if tags is None:
        tags = []

    # Sanitize: strip newlines to prevent log injection
    safe_msg = message.replace("\n", "\\n").replace("\r", "\\r")

    timestamp = datetime.datetime.now(datetime.timezone.utc).strftime("%Y-%m-%dT%H:%M:%SZ")
    tag_str   = f" [{', '.join(tags)}]" if tags else ""
    entry     = f"[{timestamp}] [{prefix}] [{level.name}]{tag_str} {safe_msg}"
    print(entry)
    return entry


# --- tests ---
entry = log("Service started")
assert "INFO"    in entry
assert "Service started" in entry

entry = log("Disk full", level=LogLevel.ERROR, tags=["storage", "alert"])
assert "ERROR"   in entry
assert "storage" in entry

# Log injection attempt should be neutralized
entry = log("user\nfake log line")
assert "\n" not in entry

print("---")
log("Debug detail",    level=LogLevel.DEBUG)
log("Disk full",       level=LogLevel.ERROR,   tags=["storage"])
log("Config loaded",   prefix="CONFIG")


# --- Manual / no-built-in solution ---
# No datetime, Enum, or str.join() — string concat and plain dict for levels.


_LOG_LEVELS = {"DEBUG": 10, "INFO": 20, "WARNING": 30, "ERROR": 40, "CRITICAL": 50}


def log_manual(
    message: str,
    level:   str = "INFO",
    prefix:  str = "APP",
    tags:    list | None = None,
) -> str:
    if tags is None:
        tags = []
    # sanitize: strip newlines to prevent log injection
    safe_msg = ""
    for ch in message:
        if ch == '\n':
            safe_msg += "\\n"
        elif ch == '\r':
            safe_msg += "\\r"
        else:
            safe_msg += ch
    # build tag string without join()
    tag_str = ""
    if tags:
        tag_str = " ["
        for i in range(len(tags)):
            if i > 0:
                tag_str += ", "
            tag_str += tags[i]
        tag_str += "]"
    entry = "[" + prefix + "] [" + level + "]" + tag_str + " " + safe_msg
    print(entry)
    return entry


entry = log_manual("Service started")
assert "INFO"            in entry
assert "Service started" in entry
entry = log_manual("Disk full", level="ERROR", tags=["storage", "alert"])
assert "ERROR"   in entry
assert "storage" in entry
entry = log_manual("user\nfake log line")
assert "\n" not in entry
log_manual("Debug detail", level="DEBUG")
log_manual("Disk full",    level="ERROR", tags=["storage"])
log_manual("Config loaded", prefix="CONFIG")
