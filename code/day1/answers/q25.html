# Q25 - Security-Flavored: Parse Log Levels
# Given a list of log lines, count occurrences of "INFO", "WARN", "ERROR".
#
# Approach:
#   - Single pass with Counter; use regex word-boundary match to avoid
#     false positives like "INFORMATION" counting as "INFO".
#   - Return counts for all three levels even if zero (explicit is better).
#
# Security note: log parsing is a core SIEM/detection-engineering task.
# Always normalize level tokens before aggregating to handle case variants
# (info, INFO, Info) from different log producers.


import re
from collections import Counter


LOG_LEVELS = ("INFO", "WARN", "ERROR")
_LEVEL_PATTERN = re.compile(r"\b(INFO|WARN(?:ING)?|ERROR)\b", re.IGNORECASE)

_NORMALIZE = {
    "INFO":    "INFO",
    "WARN":    "WARN",
    "WARNING": "WARN",     # treat WARNING as WARN
    "ERROR":   "ERROR",
}


def parse_log_levels(lines: list[str]) -> dict[str, int]:
    counts: Counter = Counter({level: 0 for level in LOG_LEVELS})
    for line in lines:
        for match in _LEVEL_PATTERN.finditer(line):
            token = _NORMALIZE.get(match.group(1).upper())
            if token:
                counts[token] += 1
    return dict(counts)


# --- tests ---
sample_logs = [
    "2024-01-01 INFO  User logged in",
    "2024-01-01 ERROR Failed to connect to DB",
    "2024-01-01 WARN  Disk usage at 85%",
    "2024-01-01 INFO  Request handled",
    "2024-01-01 ERROR Timeout after 30s",
    "2024-01-01 WARNING High memory usage",   # WARNING -> WARN
    "2024-01-01 DEBUG Verbose output",        # not counted
]

result = parse_log_levels(sample_logs)
assert result["INFO"]  == 2
assert result["WARN"]  == 2   # WARN + WARNING
assert result["ERROR"] == 2

print(parse_log_levels(sample_logs))
# {'INFO': 2, 'WARN': 2, 'ERROR': 2}

# Empty input
assert parse_log_levels([]) == {"INFO": 0, "WARN": 0, "ERROR": 0}


# --- Manual / no-built-in solution ---
# No re or Counter — manual uppercase conversion and substring scanning.


# Ordered so that "WARNING" is checked before "WARN" to avoid partial match.
_LEVEL_TOKENS = [("WARNING", "WARN"), ("INFO", "INFO"), ("WARN", "WARN"), ("ERROR", "ERROR")]


def _is_alpha(ch: str) -> bool:
    return ('A' <= ch <= 'Z') or ('a' <= ch <= 'z')


def _to_upper_str(s: str) -> str:
    result = ""
    for ch in s:
        result += chr(ord(ch) - 32) if 'a' <= ch <= 'z' else ch
    return result


def _find_levels(line: str) -> list:
    upper  = _to_upper_str(line)
    found  = []
    i      = 0
    while i < len(upper):
        matched = False
        for token, norm in _LEVEL_TOKENS:
            tlen = len(token)
            if upper[i:i + tlen] == token:
                before_ok = (i == 0) or not _is_alpha(upper[i - 1])
                after_pos = i + tlen
                after_ok  = (after_pos >= len(upper)) or not _is_alpha(upper[after_pos])
                if before_ok and after_ok:
                    found.append(norm)
                    i      += tlen
                    matched = True
                    break
        if not matched:
            i += 1
    return found


def parse_log_levels_manual(lines: list) -> dict:
    counts = {"INFO": 0, "WARN": 0, "ERROR": 0}
    for line in lines:
        for level in _find_levels(line):
            counts[level] += 1
    return counts


assert parse_log_levels_manual(sample_logs)["INFO"]  == 2
assert parse_log_levels_manual(sample_logs)["WARN"]  == 2
assert parse_log_levels_manual(sample_logs)["ERROR"] == 2
assert parse_log_levels_manual([]) == {"INFO": 0, "WARN": 0, "ERROR": 0}
print(parse_log_levels_manual(sample_logs))  # {'INFO': 2, 'WARN': 2, 'ERROR': 2}
