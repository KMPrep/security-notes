# Q3 - Reverse String Words
# Reverse the order of words in a sentence without reversing the words themselves.
#
# Approach:
#   - Split on whitespace (handles multiple spaces), reverse the word list, rejoin.
#   - Time:  O(n)  |  Space: O(n)
#
# Edge cases:
#   - Empty string       -> ""
#   - Single word        -> same word
#   - Leading/trailing spaces -> stripped cleanly via split()


def reverse_words(sentence: str) -> str:
    return " ".join(sentence.split()[::-1])


# --- tests ---
assert reverse_words("Hello World")            == "World Hello"
assert reverse_words("the quick brown fox")    == "fox brown quick the"
assert reverse_words("  spaces   everywhere ") == "everywhere spaces"
assert reverse_words("")                        == ""
assert reverse_words("single")                 == "single"

print(reverse_words("Hello World"))            # "World Hello"
print(reverse_words("the quick brown fox"))    # "fox brown quick the"


# --- Manual / no-built-in solution ---
# No split(), join(), or reverse slicing — manual word extraction, swap, and concat.


def reverse_words_manual(sentence: str) -> str:
    words, word = [], ""
    for ch in sentence:
        if ch == ' ':
            if word:
                words.append(word)
                word = ""
        else:
            word += ch
    if word:
        words.append(word)
    # reverse words list in-place
    left, right = 0, len(words) - 1
    while left < right:
        words[left], words[right] = words[right], words[left]
        left  += 1
        right -= 1
    # join with spaces
    result = ""
    for i in range(len(words)):
        if i > 0:
            result += " "
        result += words[i]
    return result


assert reverse_words_manual("Hello World")            == "World Hello"
assert reverse_words_manual("the quick brown fox")    == "fox brown quick the"
assert reverse_words_manual("  spaces   everywhere ") == "everywhere spaces"
assert reverse_words_manual("")                        == ""
print(reverse_words_manual("Hello World"))            # World Hello
print(reverse_words_manual("the quick brown fox"))    # fox brown quick the
