# Q4 - Extract Domain from URL
# Given a URL string extract the domain.
# e.g.  "https://www.example.com/path"  ->  "www.example.com"
#
# Approach:
#   - Use urllib.parse.urlparse which correctly handles all URL schemes,
#     ports, and query strings without manual string splitting.
#   - Time:  O(n)  |  Space: O(1) extra
#
# Edge cases:
#   - No scheme (bare domain)
#   - URL with port  -> strip the port
#   - Empty string   -> return ""


from urllib.parse import urlparse


def extract_domain(url: str) -> str:
    if not url:
        return ""
    parsed = urlparse(url)
    # netloc includes host + optional port; split on ':' to drop port
    netloc = parsed.netloc or parsed.path          # handle scheme-less input
    return netloc.split(":")[0]                    # strip port if present


# --- tests ---
assert extract_domain("https://www.example.com/path")          == "www.example.com"
assert extract_domain("http://api.github.com:8080/v3/repos")   == "api.github.com"
assert extract_domain("https://accounts.google.com")           == "accounts.google.com"
assert extract_domain("")                                       == ""

print(extract_domain("https://www.example.com/path"))           # www.example.com
print(extract_domain("http://api.github.com:8080/v3/repos"))    # api.github.com


# --- Manual / no-built-in solution ---
# No urllib — scan string manually for scheme, netloc, and port.


def extract_domain_manual(url: str) -> str:
    if not url:
        return ""
    # skip past "://" if present
    start = 0
    for i in range(len(url) - 2):
        if url[i] == ':' and url[i + 1] == '/' and url[i + 2] == '/':
            start = i + 3
            break
    # read until first '/' (end of netloc)
    netloc = ""
    for i in range(start, len(url)):
        if url[i] == '/':
            break
        netloc += url[i]
    # strip port
    domain = ""
    for ch in netloc:
        if ch == ':':
            break
        domain += ch
    return domain


assert extract_domain_manual("https://www.example.com/path")          == "www.example.com"
assert extract_domain_manual("http://api.github.com:8080/v3/repos")   == "api.github.com"
assert extract_domain_manual("https://accounts.google.com")           == "accounts.google.com"
assert extract_domain_manual("")                                       == ""
print(extract_domain_manual("https://www.example.com/path"))           # www.example.com
print(extract_domain_manual("http://api.github.com:8080/v3/repos"))    # api.github.com
