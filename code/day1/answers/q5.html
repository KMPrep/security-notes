# Q5 - Anagram Check
# Check if two strings are anagrams (ignore case and spaces).
#
# Approach:
#   - Normalize: lowercase + remove spaces, then compare sorted character lists.
#   - Sorting approach: O(n log n) — clean and readable.
#   - Counter approach (commented below): O(n) — preferred for very large strings.
#
# Edge cases:
#   - Both empty         -> True
#   - Different lengths after normalization -> False (short-circuit)
#   - Unicode characters -> str.lower() handles standard Unicode


def is_anagram(s1: str, s2: str) -> bool:
    normalize = lambda s: sorted(s.lower().replace(" ", ""))
    return normalize(s1) == normalize(s2)


# O(n) alternative using Counter
from collections import Counter

def is_anagram_linear(s1: str, s2: str) -> bool:
    clean = lambda s: s.lower().replace(" ", "")
    return Counter(clean(s1)) == Counter(clean(s2))


# --- tests ---
assert is_anagram("listen", "silent")          == True
assert is_anagram("Astronomer", "Moon starer") == True
assert is_anagram("hello", "world")            == False
assert is_anagram("", "")                      == True
assert is_anagram("abc", "ab")                 == False

assert is_anagram_linear("listen", "silent")          == True
assert is_anagram_linear("Astronomer", "Moon starer") == True

print(is_anagram("listen", "silent"))           # True
print(is_anagram("Astronomer", "Moon starer"))  # True
print(is_anagram("hello", "world"))             # False


# --- Manual / no-built-in solution ---
# No sorted(), Counter, lower(), or replace() — manual char-frequency map.


def _to_lower(ch: str) -> str:
    return chr(ord(ch) + 32) if 'A' <= ch <= 'Z' else ch


def _char_freq(s: str) -> dict:
    freq = {}
    for ch in s:
        if ch == ' ':
            continue
        lch = _to_lower(ch)
        freq[lch] = freq[lch] + 1 if lch in freq else 1
    return freq


def is_anagram_manual(s1: str, s2: str) -> bool:
    return _char_freq(s1) == _char_freq(s2)


assert is_anagram_manual("listen", "silent")          == True
assert is_anagram_manual("Astronomer", "Moon starer") == True
assert is_anagram_manual("hello", "world")            == False
assert is_anagram_manual("", "")                      == True
print(is_anagram_manual("listen", "silent"))           # True
print(is_anagram_manual("Astronomer", "Moon starer"))  # True
print(is_anagram_manual("hello", "world"))             # False
