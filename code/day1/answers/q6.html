# Q6 - Remove Duplicates
# Remove duplicates from a list while preserving insertion order.
#
# Approach:
#   - dict.fromkeys() preserves order (CPython 3.7+ guarantee) and deduplicates
#     in a single O(n) pass — more idiomatic than a manual seen-set loop.
#
# Edge cases:
#   - Empty list         -> []
#   - All duplicates     -> list with one element
#   - Mixed types        -> works as long as elements are hashable


def remove_duplicates(items: list) -> list:
    return list(dict.fromkeys(items))


# Explicit seen-set version (same complexity, clearer intent)
def remove_duplicates_explicit(items: list) -> list:
    seen = set()
    result = []
    for item in items:
        if item not in seen:
            seen.add(item)
            result.append(item)
    return result


# --- tests ---
assert remove_duplicates([1, 2, 3, 2, 1, 4])          == [1, 2, 3, 4]
assert remove_duplicates(["a", "b", "a", "c", "b"])   == ["a", "b", "c"]
assert remove_duplicates([])                           == []
assert remove_duplicates([1])                          == [1]

print(remove_duplicates([1, 2, 3, 2, 1, 4]))           # [1, 2, 3, 4]
print(remove_duplicates(["a", "b", "a", "c", "b"]))    # ['a', 'b', 'c']


# --- Manual / no-built-in solution ---
# No dict.fromkeys() or set() — explicit seen-dict tracks visited items.


def remove_duplicates_manual(items: list) -> list:
    seen, result = {}, []
    for item in items:
        if item not in seen:
            seen[item] = True
            result.append(item)
    return result


assert remove_duplicates_manual([1, 2, 3, 2, 1, 4])          == [1, 2, 3, 4]
assert remove_duplicates_manual(["a", "b", "a", "c", "b"])   == ["a", "b", "c"]
assert remove_duplicates_manual([])                           == []
print(remove_duplicates_manual([1, 2, 3, 2, 1, 4]))           # [1, 2, 3, 4]
print(remove_duplicates_manual(["a", "b", "a", "c", "b"]))    # ['a', 'b', 'c']
