# Q7 - Find Missing Number
# In a list of numbers from 1 to n with exactly one missing, find it.
#
# Approach:
#   - Gauss formula: expected sum = n*(n+1)//2
#   - Subtract actual sum -> O(n) time, O(1) space.
#   - XOR approach (commented) also O(n) / O(1) and avoids integer overflow
#     concern in languages without arbitrary-precision ints.
#
# Edge cases:
#   - n=1, missing 1  -> list is empty
#   - Missing last element


def find_missing(nums: list[int]) -> int:
    n = len(nums) + 1                  # total count including the missing number
    expected = n * (n + 1) // 2
    return expected - sum(nums)


# XOR alternative (worth mentioning)
def find_missing_xor(nums: list[int]) -> int:
    n = len(nums) + 1
    xor = 0
    for i in range(1, n + 1):
        xor ^= i
    for num in nums:
        xor ^= num
    return xor


# --- tests ---
assert find_missing([1, 2, 4, 5, 6])  == 3
assert find_missing([2, 3, 4, 5])     == 1
assert find_missing([1, 2, 3, 4])     == 5
assert find_missing([])               == 1   # n=1, list empty -> missing 1

assert find_missing_xor([1, 2, 4, 5, 6]) == 3

print(find_missing([1, 2, 4, 5, 6]))   # 3
print(find_missing([2, 3, 4, 5]))       # 1


# --- Manual / no-built-in solution ---
# No sum() — manual accumulation loop.


def find_missing_manual(nums: list) -> int:
    n        = len(nums) + 1
    expected = n * (n + 1) // 2
    actual   = 0
    for num in nums:
        actual += num
    return expected - actual


assert find_missing_manual([1, 2, 4, 5, 6]) == 3
assert find_missing_manual([2, 3, 4, 5])    == 1
assert find_missing_manual([1, 2, 3, 4])    == 5
assert find_missing_manual([])              == 1
print(find_missing_manual([1, 2, 4, 5, 6])) # 3
print(find_missing_manual([2, 3, 4, 5]))     # 1
