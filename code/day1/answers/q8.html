# Q8 - Merge Two Sorted Lists
# Merge two sorted lists into one sorted list.
#
# Approach:
#   - Two-pointer merge (classic merge-sort merge step).
#   - Time: O(m+n)  |  Space: O(m+n)
#
# Edge cases:
#   - Either list empty  -> return the other
#   - Lists of unequal length


def merge_sorted(a: list[int], b: list[int]) -> list[int]:
    result = []
    i = j = 0
    while i < len(a) and j < len(b):
        if a[i] <= b[j]:
            result.append(a[i])
            i += 1
        else:
            result.append(b[j])
            j += 1
    # Append any remaining elements (at most one tail will be non-empty)
    result.extend(a[i:])
    result.extend(b[j:])
    return result


# --- tests ---
assert merge_sorted([1, 3, 5], [2, 4, 6])     == [1, 2, 3, 4, 5, 6]
assert merge_sorted([1, 2, 3], [4, 5, 6])     == [1, 2, 3, 4, 5, 6]
assert merge_sorted([], [1, 2])               == [1, 2]
assert merge_sorted([1, 2], [])               == [1, 2]
assert merge_sorted([1, 1, 2], [1, 3])        == [1, 1, 1, 2, 3]

print(merge_sorted([1, 3, 5], [2, 4, 6]))     # [1, 2, 3, 4, 5, 6]
print(merge_sorted([1, 2, 3], [4, 5, 6]))     # [1, 2, 3, 4, 5, 6]


# --- Manual / no-built-in solution ---
# No extend() — append remaining elements with while loops.


def merge_sorted_manual(a: list, b: list) -> list:
    result = []
    i = j = 0
    while i < len(a) and j < len(b):
        if a[i] <= b[j]:
            result.append(a[i]); i += 1
        else:
            result.append(b[j]); j += 1
    while i < len(a):
        result.append(a[i]); i += 1
    while j < len(b):
        result.append(b[j]); j += 1
    return result


assert merge_sorted_manual([1, 3, 5], [2, 4, 6])     == [1, 2, 3, 4, 5, 6]
assert merge_sorted_manual([1, 2, 3], [4, 5, 6])     == [1, 2, 3, 4, 5, 6]
assert merge_sorted_manual([], [1, 2])               == [1, 2]
assert merge_sorted_manual([1, 2], [])               == [1, 2]
print(merge_sorted_manual([1, 3, 5], [2, 4, 6]))     # [1, 2, 3, 4, 5, 6]
