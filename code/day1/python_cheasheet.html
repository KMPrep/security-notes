Python Quick Reference Cheat Sheet

Conditionals

Use if, elif, else for decision-making.

x = 10
if x > 5:
    print("x is greater than 5")
elif x == 5:
    print("x is 5")
else:
    print("x is less than 5")

Loops

For Loop

Iterate over sequences.

for i in range(5):
    print(i)  # 0, 1, 2, 3, 4

fruits = ["apple", "banana", "cherry"]
for fruit in fruits:
    print(fruit)

While Loop

Loop while condition is true.

i = 0
while i < 5:
    print(i)
    i += 1

Lists

Ordered, mutable sequences.

# Create
my_list = [1, 2, 3, "hello"]

# Access
print(my_list[0])  # 1

# Modify
my_list.append(4)  # [1, 2, 3, "hello", 4]
my_list[1] = "two"  # [1, "two", 3, "hello", 4]

# Common methods: append(), insert(), remove(), pop(), sort(), reverse()

Dictionaries

Key-value pairs, unordered (Python 3.7+ ordered).

# Create
my_dict = {"name": "Alice", "age": 25}

# Access
print(my_dict["name"])  # Alice

# Modify
my_dict["age"] = 26
my_dict["city"] = "NYC"  # {"name": "Alice", "age": 26, "city": "NYC"}

# Common methods: keys(), values(), items(), get(), pop()

Sets

Unordered, unique elements.

# Create
my_set = {1, 2, 3, 3}  # {1, 2, 3}

# Add/Remove
my_set.add(4)  # {1, 2, 3, 4}
my_set.remove(2)  # {1, 3, 4}

# Operations
set1 = {1, 2, 3}
set2 = {3, 4, 5}
print(set1 & set2)  # Intersection: {3}
print(set1 | set2)  # Union: {1, 2, 3, 4, 5}

String Methods

Strings are immutable.

s = "hello world"

# Common methods
print(s.upper())  # "HELLO WORLD"
print(s.lower())  # "hello world"
print(s.split())  # ["hello", "world"]
print(s.replace("world", "Python"))  # "hello Python"
print(s.startswith("he"))  # True
print(s.endswith("ld"))  # True
print(s.find("o"))  # 4 (index of first 'o')

Slicing

Extract parts of sequences (lists, strings, tuples).

my_list = [0, 1, 2, 3, 4, 5]
print(my_list[1:4])  # [1, 2, 3] (start:end, exclusive end)
print(my_list[:3])   # [0, 1, 2] (from start to index 3)
print(my_list[2:])   # [2, 3, 4, 5] (from index 2 to end)
print(my_list[::2])  # [0, 2, 4] (every 2nd element)
print(my_list[::-1]) # [5, 4, 3, 2, 1, 0] (reverse)

s = "python"
print(s[1:4])  # "yth"

List Comprehensions

Concise way to create lists.

# Basic
squares = [x**2 for x in range(5)]  # [0, 1, 4, 9, 16]

# With condition
evens = [x for x in range(10) if x % 2 == 0]  # [0, 2, 4, 6, 8]

# Nested
matrix = [[i*j for j in range(3)] for i in range(3)]  # [[0, 0, 0], [0, 1, 2], [0, 2, 4]]

Functions

Define reusable code blocks.

# Basic function
def greet(name):
    return f"Hello, {name}!"

print(greet("Alice"))  # "Hello, Alice!"

# Function with default argument
def power(base, exp=2):
    return base ** exp

print(power(3))    # 9
print(power(3, 3)) # 27

# Lambda (anonymous function)
add = lambda x, y: x + y
print(add(2, 3))  # 5

Additional Concepts

Tuples

Immutable lists, useful for fixed data (e.g., coordinates or log fields).

t = (1, 2, 3)
print(t[0])  # 1
print(t[1:3])  # (2, 3) - slicing works

File I/O

Reading/writing files, essential for log parsing.

# Read
with open('file.txt', 'r') as f:
    content = f.read()
    lines = f.readlines()  # List of lines

# Write
with open('output.txt', 'w') as f:
    f.write("Hello, world!")

Exception Handling

Handle errors gracefully.

try:
    x = int("abc")
except ValueError as e:
    print(f"Error: {e}")
finally:
    print("Always runs")

Built-in Functions

Common utilities: len(), sum(), max(), min(), sorted(), enumerate(), zip().

nums = [1, 2, 3]
print(len(nums))  # 3
print(sum(nums))  # 6
print(max(nums))  # 3
print(sorted(nums, reverse=True))  # [3, 2, 1]

for i, val in enumerate(nums):
    print(i, val)  # 0 1, 1 2, 2 3

zipped = list(zip([1, 2], ['a', 'b']))  # [(1, 'a'), (2, 'b')]

Modules and Imports

Using libraries like re, json, datetime.

import json
data = json.loads('{"key": "value"}')
print(data["key"])  # "value"

from datetime import datetime
now = datetime.now()
print(now.strftime("%Y-%m-%d"))  # Current date

Day 1 Problem Solutions

Problem 1: Count Vowels

# Day 1 Problem 1: Count the number of vowels in a string
# Vowels: a, e, i, o, u (case insensitive)

def count_vowels(s):
    vowels = 'aeiouAEIOU'
    count = 0
    for char in s:
        if char in vowels:
            count += 1
    return count

# Test cases
print(count_vowels("hello"))  # Expected: 2
print(count_vowels("Python"))  # Expected: 1
print(count_vowels("AEIOU"))  # Expected: 5
print(count_vowels("bcdfg"))  # Expected: 0

Problem 2: Reverse Words

# Day 1 Problem 2: Reverse the words in a sentence
# Keep the order of words reversed, but words themselves intact.

def reverse_words(sentence):
    words = sentence.split()
    reversed_words = words[::-1]
    return ' '.join(reversed_words)

# Test cases
print(reverse_words("Hello world"))  # Expected: "world Hello"
print(reverse_words("Python is fun"))  # Expected: "fun is Python"
print(reverse_words("a b c"))  # Expected: "c b a"
print(reverse_words("single"))  # Expected: "single"

Problem 3: Remove Duplicates

# Day 1 Problem 3: Remove duplicates from a list while preserving order

def remove_duplicates(lst):
    seen = set()
    result = []
    for item in lst:
        if item not in seen:
            seen.add(item)
            result.append(item)
    return result

# Test cases
print(remove_duplicates([1, 2, 2, 3, 4, 4, 5]))  # Expected: [1, 2, 3, 4, 5]
print(remove_duplicates(["a", "b", "a", "c"]))  # Expected: ["a", "b", "c"]
print(remove_duplicates([1, 1, 1]))  # Expected: [1]
print(remove_duplicates([]))  # Expected: []

Problem 4: Frequency Map

# Day 1 Problem 4: Create a frequency map of characters in a string

def frequency_map(s):
    freq = {}
    for char in s:
        if char in freq:
            freq[char] += 1
        else:
            freq[char] = 1
    return freq

# Test cases
print(frequency_map("hello"))  # Expected: {'h': 1, 'e': 1, 'l': 2, 'o': 1}
print(frequency_map("aaa"))  # Expected: {'a': 3}
print(frequency_map(""))  # Expected: {}
print(frequency_map("a b c"))  # Expected: {'a': 1, ' ': 2, 'b': 1, 'c': 1}

Problem 5: Parse CSV Ints

# Day 1 Problem 5: Parse a comma-separated string into a list of integers

def parse_csv_ints(csv_string):
    parts = csv_string.split(',')
    result = []
    for part in parts:
        try:
            result.append(int(part.strip()))
        except ValueError:
            continue  # Skip invalid parts
    return result

# Test cases
print(parse_csv_ints("1,2,3,4"))  # Expected: [1, 2, 3, 4]
print(parse_csv_ints("10, 20 , 30"))  # Expected: [10, 20, 30]
print(parse_csv_ints("1,a,3"))  # Expected: [1, 3]
print(parse_csv_ints(""))  # Expected: []

Problem 6: Most Frequent

# Day 1 Problem 6: Find the most frequent element in a list

def most_frequent(lst):
    if not lst:
        return None
    freq = {}
    for item in lst:
        freq[item] = freq.get(item, 0) + 1
    max_freq = max(freq.values())
    for item, count in freq.items():
        if count == max_freq:
            return item

# Test cases
print(most_frequent([1, 2, 2, 3, 3, 3]))  # Expected: 3
print(most_frequent(["a", "b", "a", "c", "a"]))  # Expected: "a"
print(most_frequent([1]))  # Expected: 1
print(most_frequent([]))  # Expected: None

Problem 7: Is Palindrome

# Day 1 Problem 7: Check if a string is a palindrome
# A palindrome reads the same forwards and backwards, ignoring case and non-alphanumeric.

import re

def is_palindrome(s):
    cleaned = re.sub(r'[^a-zA-Z0-9]', '', s).lower()
    return cleaned == cleaned[::-1]

# Test cases
print(is_palindrome("A man a plan a canal Panama"))  # Expected: True
print(is_palindrome("racecar"))  # Expected: True
print(is_palindrome("hello"))  # Expected: False
print(is_palindrome("12321"))  # Expected: True

Problem 8: To Uppercase

# Day 1 Problem 8: Convert a list of strings to uppercase

def to_uppercase(lst):
    return [s.upper() for s in lst]

# Test cases
print(to_uppercase(["hello", "world"]))  # Expected: ["HELLO", "WORLD"]
print(to_uppercase(["Python", "is", "fun"]))  # Expected: ["PYTHON", "IS", "FUN"]
print(to_uppercase([]))  # Expected: []
print(to_uppercase(["a", "B", "c"]))  # Expected: ["A", "B", "C"]

Problem 9: Intersection

# Day 1 Problem 9: Find the intersection of two lists
# Return elements common to both lists, preserving order from first list, no duplicates.

def intersection(lst1, lst2):
    set2 = set(lst2)
    result = []
    seen = set()
    for item in lst1:
        if item in set2 and item not in seen:
            seen.add(item)
            result.append(item)
    return result

# Test cases
print(intersection([1, 2, 3], [2, 3, 4]))  # Expected: [2, 3]
print(intersection(["a", "b", "c"], ["b", "c", "d"]))  # Expected: ["b", "c"]
print(intersection([1, 2], [3, 4]))  # Expected: []
print(intersection([], [1, 2]))  # Expected: []

Problem 10: Split by Delimiters

# Day 1 Problem 10: Split a string by multiple delimiters
# Delimiters: space, comma, semicolon

import re

def split_by_delimiters(s):
    return re.split(r'[ ,;]+', s)

# Test cases
print(split_by_delimiters("hello world"))  # Expected: ["hello", "world"]
print(split_by_delimiters("a,b;c d"))  # Expected: ["a", "b", "c", "d"]
print(split_by_delimiters("single"))  # Expected: ["single"]
print(split_by_delimiters(""))  # Expected: [""]
