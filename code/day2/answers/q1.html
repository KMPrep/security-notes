# Q1 - Find All Duplicate IPs
# Given a list of IP addresses from a firewall log, return all IPs that appear more than once.
#
# Approach:
#   - Naive:            O(n^2) — for each IP, rescan the rest to check for duplicates.
#   - No built-ins:     O(n)   — manual frequency dict, two-pass.
#   - With built-ins:   O(n)   — Counter + list comprehension.
#
# Edge cases:
#   - Empty list             -> []
#   - All unique IPs         -> []
#   - All the same IP        -> [that IP]
#   - None / non-list input  -> TypeError
#   - Non-string entries     -> ValueError


# === APPROACH 1: NAIVE — O(n^2) ===
# "Let me start brute force: for each IP, scan the rest for a match."

def find_duplicate_ips_naive(ip_list):
    if ip_list is None:
        raise TypeError("Input cannot be None")
    if not isinstance(ip_list, list):
        raise TypeError(f"Expected list, got {type(ip_list).__name__}")

    duplicates = []
    for i in range(len(ip_list)):
        if not isinstance(ip_list[i], str):
            raise ValueError(f"Expected string at index {i}, got {type(ip_list[i]).__name__}")
        for j in range(i + 1, len(ip_list)):
            if ip_list[i] == ip_list[j] and ip_list[i] not in duplicates:
                duplicates.append(ip_list[i])
    return duplicates


# === APPROACH 2: OPTIMIZED — NO BUILT-INS — O(n) ===
# "Use a manual dict to count, then a second pass to collect those with count > 1."

def find_duplicate_ips_no_builtins(ip_list):
    if ip_list is None:
        raise TypeError("Input cannot be None")
    if not isinstance(ip_list, list):
        raise TypeError(f"Expected list, got {type(ip_list).__name__}")

    count = {}
    for ip in ip_list:
        if not isinstance(ip, str):
            raise ValueError(f"Non-string entry: {ip!r}")
        count[ip] = count[ip] + 1 if ip in count else 1

    result = []
    seen_in_result = {}
    for ip in ip_list:
        if count[ip] > 1 and ip not in seen_in_result:
            result.append(ip)
            seen_in_result[ip] = True
    return result


# === APPROACH 3: WITH BUILT-INS — O(n) ===
# "Counter + list comprehension — idiomatic and production-ready."

from collections import Counter

def find_duplicate_ips(ip_list):
    if ip_list is None:
        raise TypeError("Input cannot be None")
    if not isinstance(ip_list, list):
        raise TypeError(f"Expected list, got {type(ip_list).__name__}")
    if not all(isinstance(ip, str) for ip in ip_list):
        raise ValueError("All entries must be strings")

    freq = Counter(ip_list)
    return [ip for ip, cnt in freq.items() if cnt > 1]


# --- tests ---
sample = ["10.0.0.1", "192.168.1.1", "10.0.0.1", "10.0.0.2", "192.168.1.1"]

assert find_duplicate_ips_naive(sample) == ["10.0.0.1", "192.168.1.1"]
assert find_duplicate_ips_no_builtins(sample) == ["10.0.0.1", "192.168.1.1"]
assert find_duplicate_ips(sample) == ["10.0.0.1", "192.168.1.1"]

assert find_duplicate_ips([]) == []
assert find_duplicate_ips(["10.0.0.1"]) == []
assert find_duplicate_ips(["10.0.0.1", "10.0.0.1"]) == ["10.0.0.1"]

print(find_duplicate_ips(sample))                   # ['10.0.0.1', '192.168.1.1']
print(find_duplicate_ips([]))                        # []
print(find_duplicate_ips(["10.0.0.1", "10.0.0.1"])) # ['10.0.0.1']
