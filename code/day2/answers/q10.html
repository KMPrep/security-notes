# Q10 - Normalize Log Paths
# Given a list of file paths that may contain "." and "..", normalize each path.
#
# Approach:
#   - Naive / Optimized:  O(n*m) — stack-based processing per path.
#                         This is already optimal; there's no O(n) shortcut.
#   - No built-ins:       O(n*m) — manual tokeniser + stack + manual join.
#   - With built-ins:     O(n*m) — posixpath.normpath (cross-platform safe).
#
# Stack rules:
#   "."  -> skip (current directory)
#   ".." -> pop stack if non-empty (go up one level)
#   else -> push onto stack
#
# Edge cases:
#   - "/"           -> "/"
#   - "/../etc"     -> "/etc"  (can't go above root)
#   - "/a/b/../../" -> "/"
#   - Empty string  -> "/"
#   - None input    -> TypeError


# === APPROACH 1: SINGLE PATH — NO BUILT-INS — O(m) ===
# "Tokenise on '/', apply stack rules, rebuild with '/'."

def normalize_path_no_builtins(path):
    if path is None:
        raise TypeError("Input cannot be None")
    if not isinstance(path, str):
        raise TypeError(f"Expected string, got {type(path).__name__}")
    if not path:
        return "/"

    # Tokenise manually
    parts = []
    current = ""
    for ch in path + "/":
        if ch == "/":
            if current:
                parts.append(current)
                current = ""
        else:
            current += ch

    stack = []
    for part in parts:
        if part == ".":
            pass
        elif part == "..":
            if stack:
                stack.pop()
        elif part:
            stack.append(part)

    # Manual join
    if not stack:
        return "/"
    result = ""
    for i in range(len(stack)):
        result += "/" + stack[i]
    return result


# === APPROACH 2: LIST OF PATHS — NO BUILT-INS — O(n*m) ===
# "Apply normalize_path_no_builtins to each path in the list."

def normalize_paths_no_builtins(paths):
    if paths is None:
        raise TypeError("Input cannot be None")
    if not isinstance(paths, list):
        raise TypeError(f"Expected list, got {type(paths).__name__}")

    result = []
    for path in paths:
        if not isinstance(path, str):
            raise ValueError(f"Expected string path, got {type(path).__name__}")
        result.append(normalize_path_no_builtins(path))
    return result


# === APPROACH 3: WITH BUILT-INS — O(n*m) ===
# "posixpath.normpath forces POSIX semantics regardless of OS.
#  Important: use posixpath (not os.path) so Windows paths don't mangle results."

import posixpath

def normalize_paths(paths):
    if paths is None:
        raise TypeError("Input cannot be None")
    if not isinstance(paths, list):
        raise TypeError(f"Expected list, got {type(paths).__name__}")

    result = []
    for i, path in enumerate(paths):
        if not isinstance(path, str):
            raise ValueError(f"Expected string at index {i}, got {type(path).__name__}")
        result.append(posixpath.normpath(path))
    return result


# --- tests ---
paths = ["/var/log/../log/./syslog", "/etc/hosts/../hosts"]
expected = ["/var/log/syslog", "/etc/hosts"]

assert normalize_paths_no_builtins(paths) == expected
assert normalize_paths(paths) == expected

assert normalize_path_no_builtins("/") == "/"
assert normalize_path_no_builtins("/a/b/../../c") == "/c"
assert normalize_path_no_builtins("/../etc") == "/etc"
assert normalize_path_no_builtins("/a/./b/./c") == "/a/b/c"
assert normalize_path_no_builtins("") == "/"

print(normalize_paths(paths))                     # ['/var/log/syslog', '/etc/hosts']
print(normalize_path_no_builtins("/a/b/../../c")) # /c
print(normalize_path_no_builtins("/../etc"))      # /etc
