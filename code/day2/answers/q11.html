# Q11 - Two Sum — Threat Pair Detection
# Two threat scores in a list sum to a critical threshold. Return their indices.
#
# Approach:
#   - Naive:          O(n^2) — check every pair.
#   - No built-ins:   O(n)   — single pass, complement dict (value -> index).
#   - With built-ins: O(n)   — same, using enumerate().
#
# Key insight: for element x, its complement is (target - x).
#              If complement was seen before, we found the pair.
#              Store value -> index AFTER the check to handle x+x=target.
#
# Edge cases:
#   - No valid pair         -> []
#   - Same element twice    -> [0, 1] if arr[0] + arr[0] == target
#   - Empty list            -> []
#   - Single element        -> []
#   - None input            -> TypeError
#   - Non-numeric entries   -> ValueError


# === APPROACH 1: NAIVE — O(n^2) ===
# "Check every pair. Works. Say it first, then optimise."

def two_sum_naive(scores, target):
    if scores is None:
        raise TypeError("Input cannot be None")
    if not isinstance(scores, list):
        raise TypeError(f"Expected list, got {type(scores).__name__}")
    if not isinstance(target, (int, float)):
        raise TypeError(f"Target must be numeric, got {type(target).__name__}")

    n = len(scores)
    for i in range(n):
        for j in range(i + 1, n):
            if not isinstance(scores[i], (int, float)) or not isinstance(scores[j], (int, float)):
                raise ValueError("All scores must be numeric")
            if scores[i] + scores[j] == target:
                return [i, j]
    return []


# === APPROACH 2: COMPLEMENT DICT — NO BUILT-INS — O(n) ===
# "For each element, check if its complement was stored in the dict already."

def two_sum_no_builtins(scores, target):
    if scores is None:
        raise TypeError("Input cannot be None")
    if not isinstance(scores, list):
        raise TypeError(f"Expected list, got {type(scores).__name__}")
    if not isinstance(target, (int, float)):
        raise TypeError(f"Target must be numeric, got {type(target).__name__}")

    seen = {}  # value -> index

    for i in range(len(scores)):
        if not isinstance(scores[i], (int, float)):
            raise ValueError(f"Non-numeric at index {i}: {scores[i]!r}")
        complement = target - scores[i]
        if complement in seen:
            return [seen[complement], i]
        seen[scores[i]] = i  # store AFTER check

    return []


# === APPROACH 3: WITH BUILT-INS — O(n) ===
# "enumerate() for cleaner indexing. Logic is identical."

def two_sum(scores, target):
    if scores is None:
        raise TypeError("Input cannot be None")
    if not isinstance(scores, list):
        raise TypeError(f"Expected list, got {type(scores).__name__}")
    if not all(isinstance(x, (int, float)) for x in scores):
        raise ValueError("All scores must be numeric")
    if not isinstance(target, (int, float)):
        raise TypeError("Target must be numeric")

    seen = {}
    for i, score in enumerate(scores):
        complement = target - score
        if complement in seen:
            return [seen[complement], i]
        seen[score] = i
    return []


# --- tests ---
assert two_sum_naive([2, 7, 11, 15], 9) == [0, 1]
assert two_sum_no_builtins([2, 7, 11, 15], 9) == [0, 1]
assert two_sum([2, 7, 11, 15], 9) == [0, 1]

assert two_sum([3, 3], 6) == [0, 1]       # same value twice
assert two_sum([1, 2, 3], 10) == []        # no pair
assert two_sum([], 9) == []
assert two_sum([5], 5) == []               # can't use same index twice

print(two_sum([2, 7, 11, 15], 9))   # [0, 1]
print(two_sum([3, 3], 6))           # [0, 1]
print(two_sum([1, 2, 3], 10))       # []
