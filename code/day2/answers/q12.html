# Q12 - Group Log Lines by Severity
# Given a list of log lines each starting with a severity level, group them by severity.
#
# Approach:
#   - Naive / No built-ins:  O(n) — manual first-space split, manual dict grouping.
#   - With built-ins:        O(n) — str.split(maxsplit=1) + defaultdict.
#
# Important: use split(maxsplit=1) so only the first space is the split point —
# the message body may contain spaces.
#
# Edge cases:
#   - Empty list          -> {}
#   - Line with no space  -> severity=entire line, message=""
#   - Unknown severity    -> grouped under "UNKNOWN" (don't crash)
#   - Blank lines         -> skipped
#   - None input          -> TypeError
#   - Non-string entries  -> ValueError

VALID_SEVERITIES = {"ERROR", "WARN", "WARNING", "INFO", "DEBUG", "CRITICAL"}


# === APPROACH 1: NO BUILT-INS — O(n) ===
# "Manual first-space find, manual dict grouping."

def group_by_severity_no_builtins(log_lines):
    if log_lines is None:
        raise TypeError("Input cannot be None")
    if not isinstance(log_lines, list):
        raise TypeError(f"Expected list, got {type(log_lines).__name__}")

    groups = {}

    for line in log_lines:
        if not isinstance(line, str):
            raise ValueError(f"Expected string log line, got {type(line).__name__}")

        line = line.strip()
        if not line:
            continue

        space_idx = -1
        for i in range(len(line)):
            if line[i] == " ":
                space_idx = i
                break

        if space_idx == -1:
            severity, message = line, ""
        else:
            severity, message = line[:space_idx], line[space_idx + 1:]

        if severity not in VALID_SEVERITIES:
            severity = "UNKNOWN"

        if severity not in groups:
            groups[severity] = []
        groups[severity].append(message)

    return groups


# === APPROACH 2: WITH BUILT-INS — O(n) ===
# "defaultdict + split(maxsplit=1) — clean and production-ready."

from collections import defaultdict

def group_by_severity(log_lines):
    if log_lines is None:
        raise TypeError("Input cannot be None")
    if not isinstance(log_lines, list):
        raise TypeError(f"Expected list, got {type(log_lines).__name__}")

    groups = defaultdict(list)

    for i, line in enumerate(log_lines):
        if not isinstance(line, str):
            raise ValueError(f"Non-string at index {i}: {line!r}")
        line = line.strip()
        if not line:
            continue

        parts = line.split(maxsplit=1)
        severity = parts[0]
        message  = parts[1] if len(parts) > 1 else ""

        if severity not in VALID_SEVERITIES:
            severity = "UNKNOWN"

        groups[severity].append(message)

    return dict(groups)


# --- tests ---
sample = ["ERROR disk full", "INFO user login", "ERROR timeout", "WARN high mem"]
expected = {
    "ERROR": ["disk full", "timeout"],
    "INFO":  ["user login"],
    "WARN":  ["high mem"],
}

assert group_by_severity_no_builtins(sample) == expected
assert group_by_severity(sample) == expected

assert group_by_severity([]) == {}
assert group_by_severity(["INFO"]) == {"INFO": [""]}
assert group_by_severity(["   "]) == {}                          # blank line skipped
assert group_by_severity(["HACKER bad stuff"]) == {"UNKNOWN": ["bad stuff"]}
assert group_by_severity(["ERROR a b c"]) == {"ERROR": ["a b c"]}  # message has spaces

print(group_by_severity(sample))
# {'ERROR': ['disk full', 'timeout'], 'INFO': ['user login'], 'WARN': ['high mem']}
print(group_by_severity([]))                    # {}
print(group_by_severity(["INFO"]))              # {'INFO': ['']}
