# Q13 - Longest Substring Without Repeating Characters
# Find the length of the longest substring with no repeated characters.
# Security context: scan a token stream for repeated auth patterns.
#
# Approach:
#   - Naive:          O(n^2) — expand from each start index until a repeat is found.
#   - No built-ins:   O(n)   — sliding window with last-seen index dict.
#   - With built-ins: O(n)   — same, using enumerate().
#
# Sliding window key:
#   - left pointer moves to max(left, last_seen[ch] + 1) on duplicate.
#   - This avoids re-scanning by skipping past the previous occurrence.
#
# Edge cases:
#   - Empty string         -> 0
#   - Single character     -> 1
#   - All same character   -> 1
#   - All unique           -> len(s)
#   - None input           -> TypeError


# === APPROACH 1: NAIVE — O(n^2) ===
# "For each start, expand right until a repeat. O(n^2) due to nested loop."

def longest_no_repeat_naive(s):
    if s is None:
        raise TypeError("Input cannot be None")
    if not isinstance(s, str):
        raise TypeError(f"Expected string, got {type(s).__name__}")
    if not s:
        return 0

    max_len = 0
    n = len(s)

    for i in range(n):
        seen = {}
        for j in range(i, n):
            ch = s[j]
            if ch in seen:
                break
            seen[ch] = j
            window_len = j - i + 1
            if window_len > max_len:
                max_len = window_len

    return max_len


# === APPROACH 2: SLIDING WINDOW — NO BUILT-INS — O(n) ===
# "Track last-seen index. On duplicate, jump left pointer past it."

def longest_no_repeat_no_builtins(s):
    if s is None:
        raise TypeError("Input cannot be None")
    if not isinstance(s, str):
        raise TypeError(f"Expected string, got {type(s).__name__}")
    if not s:
        return 0

    last_seen = {}
    left = 0
    max_len = 0

    for right in range(len(s)):
        ch = s[right]
        if ch in last_seen and last_seen[ch] >= left:
            left = last_seen[ch] + 1
        last_seen[ch] = right
        window_len = right - left + 1
        if window_len > max_len:
            max_len = window_len

    return max_len


# === APPROACH 3: WITH BUILT-INS — O(n) ===
# "Same sliding window, enumerate() for cleaner indexing."

def longest_no_repeat(s):
    if s is None:
        raise TypeError("Input cannot be None")
    if not isinstance(s, str):
        raise TypeError(f"Expected string, got {type(s).__name__}")

    last_seen = {}
    left = 0
    max_len = 0

    for right, ch in enumerate(s):
        if ch in last_seen and last_seen[ch] >= left:
            left = last_seen[ch] + 1
        last_seen[ch] = right
        max_len = max(max_len, right - left + 1)

    return max_len


# --- tests ---
assert longest_no_repeat_naive("abcabcbb") == 3
assert longest_no_repeat_no_builtins("abcabcbb") == 3
assert longest_no_repeat("abcabcbb") == 3

assert longest_no_repeat("") == 0
assert longest_no_repeat("a") == 1
assert longest_no_repeat("bbbbb") == 1
assert longest_no_repeat("pwwkew") == 3     # "wke"
assert longest_no_repeat("dvdf") == 3       # "vdf"
assert longest_no_repeat("abcdef") == 6     # all unique

print(longest_no_repeat("abcabcbb"))  # 3
print(longest_no_repeat("bbbbb"))     # 1
print(longest_no_repeat("pwwkew"))    # 3
print(longest_no_repeat("dvdf"))      # 3
