# Q14 - Anagram Detection in Alert Names
# Given a list of alert names, group those that are anagrams of each other.
#
# Approach:
#   - Naive:                O(n^2 * m) — compare every pair, count chars manually.
#   - No built-ins:         O(n * m log m) — manual insertion sort as canonical key.
#   - With built-ins:       O(n * m log m) — sorted() as key.
#   - Bonus (linear key):   O(n * m)       — 26-char frequency tuple as key (no sort).
#
# Edge cases:
#   - Empty list              -> []
#   - Single word             -> [[word]]
#   - No anagram groups       -> each word in its own group
#   - Case insensitive        -> "Listen" anagram of "Silent"
#   - Non-string elements     -> ValueError
#   - None input              -> TypeError


# === APPROACH 1: NAIVE — O(n^2 * m) ===
# "Compare every pair. Slow but shows understanding of anagram definition."

def group_anagrams_naive(words):
    if words is None:
        raise TypeError("Input cannot be None")
    if not isinstance(words, list):
        raise TypeError(f"Expected list, got {type(words).__name__}")

    def are_anagrams(w1, w2):
        if len(w1) != len(w2):
            return False
        count = {}
        for ch in w1:
            count[ch] = count.get(ch, 0) + 1
        for ch in w2:
            if count.get(ch, 0) == 0:
                return False
            count[ch] -= 1
        return True

    used = {}
    groups = []
    for i in range(len(words)):
        if i in used:
            continue
        group = [words[i]]
        used[i] = True
        for j in range(i + 1, len(words)):
            if j not in used and are_anagrams(words[i].lower(), words[j].lower()):
                group.append(words[j])
                used[j] = True
        groups.append(group)
    return groups


# === APPROACH 2: SORTED KEY — NO BUILT-INS — O(n * m log m) ===
# "Manual insertion sort on characters gives canonical key."

def group_anagrams_no_builtins(words):
    if words is None:
        raise TypeError("Input cannot be None")
    if not isinstance(words, list):
        raise TypeError(f"Expected list, got {type(words).__name__}")

    def sort_chars(word):
        chars = list(word.lower())
        for i in range(1, len(chars)):
            key = chars[i]
            j = i - 1
            while j >= 0 and chars[j] > key:
                chars[j + 1] = chars[j]
                j -= 1
            chars[j + 1] = key
        result = ""
        for ch in chars:
            result += ch
        return result

    groups = {}
    for word in words:
        if not isinstance(word, str):
            raise ValueError(f"Expected string, got {type(word).__name__}")
        key = sort_chars(word)
        if key not in groups:
            groups[key] = []
        groups[key].append(word)

    result = []
    for key in groups:
        result.append(groups[key])
    return result


# === APPROACH 3: WITH BUILT-INS — O(n * m log m) ===
# "sorted() as canonical key — clean and readable."

from collections import defaultdict

def group_anagrams(words):
    if words is None:
        raise TypeError("Input cannot be None")
    if not isinstance(words, list):
        raise TypeError(f"Expected list, got {type(words).__name__}")
    if not all(isinstance(w, str) for w in words):
        raise ValueError("All elements must be strings")

    groups = defaultdict(list)
    for word in words:
        key = tuple(sorted(word.lower()))
        groups[key].append(word)
    return list(groups.values())


# === BONUS: LINEAR KEY — O(n * m) ===
# "Google follow-up: 'Can you avoid sorting?'
#  Use 26-char frequency tuple — no sort, still O(m) per word."

def group_anagrams_linear_key(words):
    if words is None:
        raise TypeError("Input cannot be None")
    if not isinstance(words, list):
        raise TypeError(f"Expected list, got {type(words).__name__}")

    groups = defaultdict(list)
    for word in words:
        counts = [0] * 26
        for ch in word.lower():
            idx = ord(ch) - ord('a')
            if 0 <= idx < 26:
                counts[idx] += 1
        groups[tuple(counts)].append(word)
    return list(groups.values())

sample = ["listen", "silent", "enlist", "google", "gooles", "inlets"]

# Verify grouping (order within groups may vary)
result = group_anagrams(sample)
assert len(result) == 2
group_sets = [set(g) for g in result]
assert {"listen", "silent", "enlist", "inlets"} in group_sets
assert {"google", "gooles"} in group_sets

result2 = group_anagrams_linear_key(sample)
group_sets2 = [set(g) for g in result2]
assert {"listen", "silent", "enlist", "inlets"} in group_sets2

assert group_anagrams([]) == []
assert group_anagrams(["a"]) == [["a"]]
assert len(group_anagrams(["ab", "ba"])) == 1
