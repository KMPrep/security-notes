# Q15 - Count Users with Multiple Consecutive Failed Attempts
# From a list of (username, result) tuples, return users with >= threshold consecutive failures.
# Default threshold = 3.
#
# Approach:
#   - Naive / Optimized:  O(n) — per-user streak dict. This is already linear.
#   - No built-ins:       O(n) — manual dict, no dict.get(), no enumerate().
#   - With built-ins:     O(n) — defaultdict, enumerate(), parameterised threshold.
#
# Security context: streak >= threshold triggers account lockout or CAPTCHA.
#
# Edge cases:
#   - Empty list                  -> []
#   - streak resets on OK         -> user not flagged if they recover
#   - user flagged even if they   -> flag is sticky once threshold is hit
#     later succeed
#   - Unknown result value        -> ValueError
#   - Malformed tuple             -> ValueError
#   - None input                  -> TypeError


# === APPROACH 1: NAIVE — O(n) ===
# "Running per-user counter. Straightforward — most people write this first."

def users_with_consecutive_failures_naive(events):
    if events is None:
        raise TypeError("Input cannot be None")
    if not isinstance(events, list):
        raise TypeError(f"Expected list, got {type(events).__name__}")

    streak  = {}
    flagged = {}

    for entry in events:
        if not isinstance(entry, (tuple, list)) or len(entry) != 2:
            raise ValueError(f"Each entry must be (username, result), got {entry!r}")
        username, result = entry[0], entry[1]
        if not isinstance(username, str):
            raise ValueError(f"Username must be a string, got {type(username).__name__}")
        if result not in ("OK", "FAIL"):
            raise ValueError(f"Result must be 'OK' or 'FAIL', got {result!r}")

        if username not in streak:
            streak[username] = 0

        if result == "FAIL":
            streak[username] += 1
            if streak[username] >= 3:
                flagged[username] = True
        else:
            streak[username] = 0

    result_list = []
    for user in flagged:
        result_list.append(user)
    return result_list


# === APPROACH 2: NO BUILT-INS — O(n) ===
# "Explicitly manual: no dict.get(), no enumerate(), no in-place augmented assign."

def users_with_consecutive_failures_no_builtins(events, threshold=3):
    if events is None:
        raise TypeError("Input cannot be None")
    if not isinstance(events, list):
        raise TypeError(f"Expected list, got {type(events).__name__}")
    if not isinstance(threshold, int) or threshold < 1:
        raise ValueError(f"Threshold must be a positive integer, got {threshold!r}")

    streak  = {}
    flagged = {}

    for i in range(len(events)):
        entry = events[i]
        if not isinstance(entry, (tuple, list)) or len(entry) != 2:
            raise ValueError(f"Entry {i} must be a 2-element tuple/list")

        username = entry[0]
        result   = entry[1]

        if username not in streak:
            streak[username] = 0

        if result == "FAIL":
            streak[username] = streak[username] + 1
            if streak[username] >= threshold and username not in flagged:
                flagged[username] = True
        elif result == "OK":
            streak[username] = 0
        else:
            raise ValueError(f"Unknown result '{result}' at index {i}")

    result_list = []
    for user in flagged:
        result_list.append(user)
    return result_list


# === APPROACH 3: WITH BUILT-INS — O(n) ===
# "defaultdict + set. sorted() for deterministic output in tests."

from collections import defaultdict

def users_with_consecutive_failures(events, threshold=3):
    if events is None:
        raise TypeError("Input cannot be None")
    if not isinstance(events, list):
        raise TypeError(f"Expected list, got {type(events).__name__}")
    if not isinstance(threshold, int) or threshold < 1:
        raise ValueError(f"Threshold must be a positive integer, got {threshold!r}")

    streak  = defaultdict(int)
    flagged = set()

    for i, entry in enumerate(events):
        if not isinstance(entry, (tuple, list)) or len(entry) != 2:
            raise ValueError(f"Entry at index {i} must be a 2-element tuple")
        username, result = entry
        if result == "FAIL":
            streak[username] += 1
            if streak[username] >= threshold:
                flagged.add(username)
        elif result == "OK":
            streak[username] = 0
        else:
            raise ValueError(f"Unknown result '{result}' at index {i}")

    return sorted(flagged)


# --- tests ---
sample = [
    ("alice", "FAIL"), ("alice", "FAIL"), ("alice", "FAIL"),
    ("bob", "FAIL"), ("bob", "OK"),
]

assert users_with_consecutive_failures_naive(sample) == ["alice"]
assert users_with_consecutive_failures_no_builtins(sample) == ["alice"]
assert users_with_consecutive_failures(sample) == ["alice"]

assert users_with_consecutive_failures([]) == []
assert users_with_consecutive_failures([("alice", "FAIL"), ("alice", "FAIL")]) == []  # only 2
assert users_with_consecutive_failures([("alice", "FAIL")] * 3) == ["alice"]

# flag is sticky
sticky = [("alice", "FAIL"), ("alice", "FAIL"), ("alice", "FAIL"), ("alice", "OK")]
assert users_with_consecutive_failures(sticky) == ["alice"]

# custom threshold
assert users_with_consecutive_failures(
    [("bob", "FAIL"), ("bob", "FAIL")], threshold=2
) == ["bob"]

print(users_with_consecutive_failures(sample))         # ['alice']
print(users_with_consecutive_failures([]))             # []
