# Q16 - Prefix Sum for Suspicious Activity Windows
# Build a prefix sum array in O(n); then answer any range [i, j] query in O(1).
#
# Approach:
#   - Naive:          O(n) per query  — sum the range directly each time.
#   - No built-ins:   O(n) setup + O(1) query — manual prefix array as a class.
#   - With built-ins: O(n) setup + O(1) query — itertools.accumulate.
#
# Formula:
#   prefix[0] = 0
#   prefix[k] = prefix[k-1] + counts[k-1]
#   range_sum(i, j) = prefix[j+1] - prefix[i]
#
# Edge cases:
#   - i == j             -> single element
#   - i == 0             -> sum from start
#   - j == n-1           -> sum to end
#   - i > j              -> IndexError
#   - Out-of-bounds i, j -> IndexError
#   - None / empty input -> TypeError / handled gracefully


# === APPROACH 1: NAIVE — O(n) PER QUERY ===
# "Recompute the range sum every time. Fine for one query; bad for many."

def range_sum_naive(counts, i, j):
    if counts is None:
        raise TypeError("Input cannot be None")
    if not isinstance(counts, list):
        raise TypeError(f"Expected list, got {type(counts).__name__}")
    if not isinstance(i, int) or not isinstance(j, int):
        raise TypeError("Indices must be integers")
    if i < 0 or j >= len(counts) or i > j:
        raise IndexError(f"Invalid range [{i}, {j}] for length {len(counts)}")

    total = 0
    for k in range(i, j + 1):
        total += counts[k]
    return total


# === APPROACH 2: PREFIX SUM CLASS — NO BUILT-INS — O(n) SETUP + O(1) QUERY ===
# "Build once, answer in O(1). Shows understanding of the precomputation tradeoff."

class PrefixSumChecker:
    def __init__(self, counts):
        if counts is None:
            raise TypeError("Input cannot be None")
        if not isinstance(counts, list):
            raise TypeError(f"Expected list, got {type(counts).__name__}")
        for idx in range(len(counts)):
            if not isinstance(counts[idx], (int, float)):
                raise ValueError(f"Non-numeric at index {idx}: {counts[idx]!r}")

        n = len(counts)
        self._prefix = [0] * (n + 1)
        for idx in range(n):
            self._prefix[idx + 1] = self._prefix[idx] + counts[idx]
        self._n = n

    def query(self, i, j):
        if not isinstance(i, int) or not isinstance(j, int):
            raise TypeError("Indices must be integers")
        if i < 0 or j >= self._n or i > j:
            raise IndexError(f"Invalid range [{i}, {j}] for length {self._n}")
        return self._prefix[j + 1] - self._prefix[i]


# === APPROACH 3: WITH BUILT-INS — O(n) SETUP + O(1) QUERY ===
# "itertools.accumulate builds the cumulative sum list cleanly."

from itertools import accumulate

def build_prefix_sum(counts):
    if counts is None:
        raise TypeError("Input cannot be None")
    if not isinstance(counts, list):
        raise TypeError(f"Expected list, got {type(counts).__name__}")
    if not all(isinstance(x, (int, float)) for x in counts):
        raise ValueError("All counts must be numeric")
    return [0] + list(accumulate(counts))

def range_sum(prefix, i, j):
    n = len(prefix) - 1
    if not isinstance(i, int) or not isinstance(j, int):
        raise TypeError("Indices must be integers")
    if i < 0 or j >= n or i > j:
        raise IndexError(f"Invalid range [{i}, {j}] for length {n}")
    return prefix[j + 1] - prefix[i]


# --- tests ---
counts = [1, 3, 2, 5, 4]

assert range_sum_naive(counts, 1, 3) == 10   # 3+2+5
assert range_sum_naive(counts, 0, 4) == 15

checker = PrefixSumChecker(counts)
assert checker.query(1, 3) == 10
assert checker.query(0, 4) == 15
assert checker.query(2, 2) == 2

prefix = build_prefix_sum(counts)
assert range_sum(prefix, 1, 3) == 10
assert range_sum(prefix, 0, 4) == 15
assert range_sum(prefix, 2, 2) == 2

print(checker.query(1, 3))               # 10
print(range_sum(build_prefix_sum(counts), 1, 3))  # 10
print(range_sum(build_prefix_sum(counts), 0, 4))  # 15
