# Q17 - Longest Common Prefix in Hostname List
# Given a list of hostnames, find the longest common prefix.
#
# Approach:
#   - Naive:          O(n * m) — compare first hostname against all others, shrink prefix.
#   - No built-ins:   O(n * m) — find shortest first, then scan column-by-column manually.
#   - With built-ins: O(n * m) — zip(*hostnames) transposes to character columns.
#
# Key insight: the common prefix can be at most as long as the shortest string.
#
# Edge cases:
#   - Empty list          -> ""
#   - Single hostname     -> that hostname
#   - No common prefix    -> ""
#   - Identical hostnames -> any one of them
#   - None input          -> TypeError
#   - Non-string entries  -> ValueError


# === APPROACH 1: NAIVE — O(n * m) ===
# "Compare first hostname to each other, shrink prefix on mismatch."

def longest_common_prefix_naive(hostnames):
    if hostnames is None:
        raise TypeError("Input cannot be None")
    if not isinstance(hostnames, list):
        raise TypeError(f"Expected list, got {type(hostnames).__name__}")
    if not hostnames:
        return ""
    for h in hostnames:
        if not isinstance(h, str):
            raise ValueError(f"Expected string, got {type(h).__name__}")

    prefix = hostnames[0]

    for i in range(1, len(hostnames)):
        new_prefix = ""
        for j in range(min(len(prefix), len(hostnames[i]))):
            if prefix[j] == hostnames[i][j]:
                new_prefix += prefix[j]
            else:
                break
        prefix = new_prefix
        if not prefix:
            return ""

    return prefix


# === APPROACH 2: SHORTEST-FIRST — NO BUILT-INS — O(n * m) ===
# "Find shortest string first to limit inner loop depth. Column-by-column compare."

def longest_common_prefix_no_builtins(hostnames):
    if hostnames is None:
        raise TypeError("Input cannot be None")
    if not isinstance(hostnames, list):
        raise TypeError(f"Expected list, got {type(hostnames).__name__}")
    if not hostnames:
        return ""

    shortest = hostnames[0]
    for h in hostnames:
        if not isinstance(h, str):
            raise ValueError(f"Expected string, got {type(h).__name__}")
        if len(h) < len(shortest):
            shortest = h

    for pos in range(len(shortest)):
        ch = shortest[pos]
        for h in hostnames:
            if h[pos] != ch:
                result = ""
                for k in range(pos):
                    result += shortest[k]
                return result

    return shortest


# === APPROACH 3: WITH BUILT-INS — O(n * m) ===
# "zip(*hostnames) gives character columns. Stop at first non-uniform column."

def longest_common_prefix(hostnames):
    if hostnames is None:
        raise TypeError("Input cannot be None")
    if not isinstance(hostnames, list):
        raise TypeError(f"Expected list, got {type(hostnames).__name__}")
    if not hostnames:
        return ""
    if not all(isinstance(h, str) for h in hostnames):
        raise ValueError("All hostnames must be strings")

    prefix = []
    for chars in zip(*hostnames):
        if len(set(chars)) == 1:
            prefix.append(chars[0])
        else:
            break
    return "".join(prefix)


# --- tests ---
sample = ["auth-server-01", "auth-server-02", "auth-db-01"]

assert longest_common_prefix_naive(sample) == "auth-"
assert longest_common_prefix_no_builtins(sample) == "auth-"
assert longest_common_prefix(sample) == "auth-"

assert longest_common_prefix([]) == ""
assert longest_common_prefix(["abc"]) == "abc"
assert longest_common_prefix(["abc", "abd"]) == "ab"
assert longest_common_prefix(["abc", "xyz"]) == ""
assert longest_common_prefix(["aa", "aa"]) == "aa"

print(longest_common_prefix(sample))            # auth-
print(longest_common_prefix(["abc", "abd"]))    # ab
print(longest_common_prefix(["abc", "xyz"]))    # (empty string)
print(longest_common_prefix([]))                # (empty string)
