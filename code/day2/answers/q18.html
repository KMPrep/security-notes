# Q18 - Zero-Sum Subarray in Anomaly Scores
# Given a list of anomaly delta scores (can be negative), find if any subarray sums to zero.
# Bonus: return the actual subarray if found.
#
# Approach:
#   - Naive:          O(n^2) — compute every subarray sum.
#   - No built-ins:   O(n)   — prefix sums + dict-as-set.
#   - With built-ins: O(n)   — same, also returns the actual subarray.
#
# Key insight:
#   prefix_sum[j] - prefix_sum[i] == 0  =>  prefix_sum[i] == prefix_sum[j]
#   So: if the same prefix sum appears twice, the subarray between them sums to zero.
#   Also: if prefix_sum[j] == 0, the subarray scores[0..j] sums to zero.
#   Handle this by seeding seen with {0: -1}.
#
# Edge cases:
#   - Empty list              -> False / None
#   - All positive            -> False
#   - Single zero element     -> True
#   - Subarray starting at 0  -> handled by seed {0: -1}
#   - None input              -> TypeError


# === APPROACH 1: NAIVE — O(n^2) ===
# "Sum every subarray. Works; say it first."

def has_zero_sum_subarray_naive(scores):
    if scores is None:
        raise TypeError("Input cannot be None")
    if not isinstance(scores, list):
        raise TypeError(f"Expected list, got {type(scores).__name__}")

    n = len(scores)
    for i in range(n):
        running = 0
        for j in range(i, n):
            if not isinstance(scores[j], (int, float)):
                raise ValueError(f"Non-numeric at index {j}: {scores[j]!r}")
            running += scores[j]
            if running == 0:
                return True
    return False


# === APPROACH 2: PREFIX SUM — NO BUILT-INS — O(n) ===
# "Track seen prefix sums in a dict. Duplicate => zero-sum subarray exists."

def has_zero_sum_subarray_no_builtins(scores):
    if scores is None:
        raise TypeError("Input cannot be None")
    if not isinstance(scores, list):
        raise TypeError(f"Expected list, got {type(scores).__name__}")

    seen = {}
    seen[0] = True  # seed: prefix sum of 0 before any element
    running = 0

    for i in range(len(scores)):
        if not isinstance(scores[i], (int, float)):
            raise ValueError(f"Non-numeric at index {i}: {scores[i]!r}")
        running = running + scores[i]
        if running in seen:
            return True
        seen[running] = True

    return False


# === APPROACH 3: WITH BUILT-INS + RETURN SUBARRAY — O(n) ===
# "Google follow-up: 'Can you return the actual subarray?'
#  Store seen[prefix_sum] = index to reconstruct the slice."

def find_zero_sum_subarray(scores):
    if scores is None:
        raise TypeError("Input cannot be None")
    if not isinstance(scores, list):
        raise TypeError(f"Expected list, got {type(scores).__name__}")
    if not all(isinstance(x, (int, float)) for x in scores):
        raise ValueError("All entries must be numeric")

    seen = {0: -1}   # prefix_sum -> index where it was first seen
    running = 0

    for i, val in enumerate(scores):
        running += val
        if running in seen:
            start = seen[running] + 1
            return scores[start:i + 1]
        seen[running] = i

    return None


# --- tests ---
sample = [4, -1, -3, 1, -2, 2]
# prefix sums: 0, 4, 3, 0  <- 0 repeats at index 2, subarray [4,-1,-3] sums to 0

assert has_zero_sum_subarray_naive(sample) is True
assert has_zero_sum_subarray_no_builtins(sample) is True
assert find_zero_sum_subarray(sample) is not None
assert sum(find_zero_sum_subarray(sample)) == 0

assert has_zero_sum_subarray_naive([1, 2, 3]) is False
assert has_zero_sum_subarray_no_builtins([1, 2, 3]) is False
assert find_zero_sum_subarray([1, 2, 3]) is None

assert has_zero_sum_subarray_no_builtins([1, -1]) is True     # [1,-1]
assert has_zero_sum_subarray_no_builtins([0]) is True         # single zero
assert has_zero_sum_subarray_no_builtins([3, 4, -7, 2]) is True  # [3,4,-7]

print(find_zero_sum_subarray([4, -1, -3, 1, -2, 2]))  # [4, -1, -3]
print(find_zero_sum_subarray([3, 4, -7, 2]))           # [3, 4, -7]
print(find_zero_sum_subarray([1, 2, 3]))               # None
print(find_zero_sum_subarray([1, -1]))                 # [1, -1]
