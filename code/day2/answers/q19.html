# Q19 - IP Access Frequency Report
# Given raw log lines in the format "IP METHOD PATH STATUS",
# count total requests per IP and flag any IP with more than 5 requests.
#
# Approach:
#   - No built-ins:   O(n) — manual first-token extraction, manual dict.
#   - With built-ins: O(n) — str.split() + Counter + dict comprehension.
#
# Edge cases:
#   - Empty list           -> {}
#   - Malformed lines      -> skipped gracefully (logged internally)
#   - Blank lines          -> skipped
#   - Custom threshold     -> parameterised
#   - None input           -> TypeError
#   - Non-string entries   -> ValueError


# === APPROACH 1: NO BUILT-INS — O(n) ===
# "Manual first-space extraction. Shows awareness of string parsing."

def ip_access_report_no_builtins(log_lines, threshold=5):
    if log_lines is None:
        raise TypeError("Input cannot be None")
    if not isinstance(log_lines, list):
        raise TypeError(f"Expected list, got {type(log_lines).__name__}")
    if not isinstance(threshold, int) or threshold < 1:
        raise ValueError(f"Threshold must be a positive integer, got {threshold!r}")

    counts = {}

    for i in range(len(log_lines)):
        line = log_lines[i]
        if not isinstance(line, str):
            raise ValueError(f"Non-string at index {i}: {line!r}")
        line = line.strip()
        if not line:
            continue

        space_idx = -1
        for j in range(len(line)):
            if line[j] == " ":
                space_idx = j
                break

        if space_idx == -1:
            continue  # malformed — no space found

        ip = line[:space_idx]
        if not ip:
            continue

        if ip not in counts:
            counts[ip] = 0
        counts[ip] = counts[ip] + 1

    report = {}
    for ip in counts:
        report[ip] = {"count": counts[ip], "flagged": counts[ip] > threshold}
    return report


# === APPROACH 2: WITH BUILT-INS — O(n) ===
# "Counter + dict comprehension. split() is safe with extra whitespace."

from collections import Counter

def ip_access_report(log_lines, threshold=5):
    if log_lines is None:
        raise TypeError("Input cannot be None")
    if not isinstance(log_lines, list):
        raise TypeError(f"Expected list, got {type(log_lines).__name__}")
    if not isinstance(threshold, int) or threshold < 1:
        raise ValueError(f"Threshold must be a positive integer, got {threshold!r}")

    ips = []
    for i, line in enumerate(log_lines):
        if not isinstance(line, str):
            raise ValueError(f"Non-string at index {i}: {line!r}")
        line = line.strip()
        if not line:
            continue
        parts = line.split()
        if len(parts) < 4:
            continue  # malformed — skip gracefully
        ips.append(parts[0])

    counts = Counter(ips)
    return {
        ip: {"count": cnt, "flagged": cnt > threshold}
        for ip, cnt in counts.items()
    }

logs = [
    "10.0.0.1 GET /login 200",
    "10.0.0.1 GET /login 401",
    "10.0.0.1 GET /login 401",
    "10.0.0.1 GET /login 401",
    "10.0.0.1 GET /login 401",
    "10.0.0.1 GET /login 401",  # 6 total -> flagged
    "192.168.1.1 GET /home 200",
]

result = ip_access_report(logs)
assert result["10.0.0.1"]["count"] == 6
assert result["10.0.0.1"]["flagged"] is True
assert result["192.168.1.1"]["count"] == 1
assert result["192.168.1.1"]["flagged"] is False

result_nb = ip_access_report_no_builtins(logs)
assert result_nb["10.0.0.1"]["count"] == 6
assert result_nb["10.0.0.1"]["flagged"] is True

assert ip_access_report([]) == {}
assert ip_access_report(["   "]) == {}
assert ip_access_report(["malformed"]) == {}  # no space, skipped
