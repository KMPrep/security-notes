# Q2 - Merge Sorted Event Timestamps
# Given two sorted lists of Unix timestamps from two log sources,
# merge them into one sorted list without using sorted().
#
# Approach:
#   - Naive:            O((n+m) log(n+m)) — concatenate + insertion sort.
#   - No built-ins:     O(n+m)            — two-pointer merge (merge sort step).
#   - With built-ins:   O(n+m)            — heapq.merge (handles k sources too).
#
# Edge cases:
#   - Either list empty          -> returns the other
#   - Lists of unequal length    -> handled naturally
#   - None input                 -> TypeError
#   - Non-numeric timestamps     -> ValueError


# === APPROACH 1: NAIVE — O((n+m) log(n+m)) ===
# "First instinct: combine then sort. Violates the constraint but useful baseline."

def merge_timestamps_naive(list_a, list_b):
    if list_a is None or list_b is None:
        raise TypeError("Inputs cannot be None")

    combined = list(list_a) + list(list_b)
    # Manual insertion sort (avoids sorted() builtin)
    for i in range(1, len(combined)):
        key = combined[i]
        j = i - 1
        while j >= 0 and combined[j] > key:
            combined[j + 1] = combined[j]
            j -= 1
        combined[j + 1] = key
    return combined


# === APPROACH 2: TWO-POINTER MERGE — NO BUILT-INS — O(n+m) ===
# "Classic merge-sort merge step. Each element visited exactly once."

def merge_timestamps_no_builtins(list_a, list_b):
    if list_a is None or list_b is None:
        raise TypeError("Inputs cannot be None")
    if not isinstance(list_a, list) or not isinstance(list_b, list):
        raise TypeError("Both inputs must be lists")

    result = []
    i, j = 0, 0

    while i < len(list_a) and j < len(list_b):
        if not isinstance(list_a[i], (int, float)):
            raise ValueError(f"Non-numeric at list_a[{i}]: {list_a[i]!r}")
        if not isinstance(list_b[j], (int, float)):
            raise ValueError(f"Non-numeric at list_b[{j}]: {list_b[j]!r}")
        if list_a[i] <= list_b[j]:
            result.append(list_a[i])
            i += 1
        else:
            result.append(list_b[j])
            j += 1

    while i < len(list_a):
        result.append(list_a[i])
        i += 1
    while j < len(list_b):
        result.append(list_b[j])
        j += 1

    return result


# === APPROACH 3: WITH BUILT-INS — O(n+m) ===
# "heapq.merge scales to k sorted sources — O(n log k)."

import heapq

def merge_timestamps(list_a, list_b):
    if list_a is None or list_b is None:
        raise TypeError("Inputs cannot be None")
    if not isinstance(list_a, list) or not isinstance(list_b, list):
        raise TypeError("Both inputs must be lists")

    return list(heapq.merge(list_a, list_b))


# --- tests ---
a = [1000, 1005, 1010]
b = [1002, 1007, 1020]
expected = [1000, 1002, 1005, 1007, 1010, 1020]

assert merge_timestamps_naive(a, b) == expected
assert merge_timestamps_no_builtins(a, b) == expected
assert merge_timestamps(a, b) == expected

assert merge_timestamps([], [1, 2, 3]) == [1, 2, 3]
assert merge_timestamps([5], [1, 2, 3]) == [1, 2, 3, 5]
assert merge_timestamps([], []) == []

print(merge_timestamps(a, b))            # [1000, 1002, 1005, 1007, 1010, 1020]
print(merge_timestamps([], [1, 2, 3]))   # [1, 2, 3]
print(merge_timestamps([5], [1, 2, 3]))  # [1, 2, 3, 5]
