# Q20 - Session Token Validation
# Given a list of session tokens (should be alphanumeric, length 32),
# identify any that are: wrong length, contain non-alphanumeric chars, or are duplicates.
# Return {"invalid": [...], "duplicates": [...]}.
#
# Approach:
#   - No built-ins:   O(n) — manual ord-based alnum check + dict dedup.
#   - With built-ins: O(n) — re.compile for pattern + Counter for duplicates.
#
# Edge cases:
#   - Empty list           -> {"invalid": [], "duplicates": []}
#   - All valid, no dups   -> {"invalid": [], "duplicates": []}
#   - Token that is both   -> appears in both lists
#   - Non-string entry     -> flagged in invalid
#   - None input           -> TypeError

VALID_LENGTH = 32


# === APPROACH 1: NO BUILT-INS — O(n) ===
# "Manual ASCII range check. No re, no str.isalnum()."

def validate_tokens_no_builtins(tokens):
    if tokens is None:
        raise TypeError("Input cannot be None")
    if not isinstance(tokens, list):
        raise TypeError(f"Expected list, got {type(tokens).__name__}")

    def is_alnum(ch):
        c = ord(ch)
        return (48 <= c <= 57) or (65 <= c <= 90) or (97 <= c <= 122)

    invalid    = []
    seen       = {}
    duplicates = []
    dup_set    = {}

    for i in range(len(tokens)):
        token = tokens[i]

        if not isinstance(token, str):
            invalid.append({"index": i, "token": repr(token), "reasons": ["not a string"]})
            continue

        reasons = []
        if len(token) != VALID_LENGTH:
            reasons.append(f"length {len(token)} != {VALID_LENGTH}")

        non_alnum = False
        for j in range(len(token)):
            if not is_alnum(token[j]):
                reasons.append(f"non-alphanumeric char '{token[j]}' at pos {j}")
                non_alnum = True
                break  # report first offending char only

        if reasons:
            invalid.append({"index": i, "token": token, "reasons": reasons})

        # Duplicate check independent of validity
        if token in seen:
            if token not in dup_set:
                duplicates.append(token)
                dup_set[token] = True
        else:
            seen[token] = i

    return {"invalid": invalid, "duplicates": duplicates}


# === APPROACH 2: WITH BUILT-INS — O(n) ===
# "re.compile for token format + Counter for duplicates."

import re
from collections import Counter

_VALID_TOKEN = re.compile(r'^[a-zA-Z0-9]{32}$')

def validate_tokens(tokens):
    if tokens is None:
        raise TypeError("Input cannot be None")
    if not isinstance(tokens, list):
        raise TypeError(f"Expected list, got {type(tokens).__name__}")

    invalid    = []
    counts     = Counter(t for t in tokens if isinstance(t, str))
    duplicates = [t for t, c in counts.items() if c > 1]

    for i, token in enumerate(tokens):
        if not isinstance(token, str):
            invalid.append({"index": i, "reason": f"not a string: {token!r}"})
            continue
        if not _VALID_TOKEN.match(token):
            reasons = []
            if len(token) != VALID_LENGTH:
                reasons.append(f"length {len(token)} != {VALID_LENGTH}")
            if re.search(r'[^a-zA-Z0-9]', token):
                reasons.append("non-alphanumeric characters")
            invalid.append({"index": i, "token": token, "reasons": reasons})

    return {"invalid": invalid, "duplicates": duplicates}


# --- tests ---
valid_token = "a" * VALID_LENGTH

result = validate_tokens([valid_token])
assert result["invalid"] == []
assert result["duplicates"] == []

# Duplicate
result = validate_tokens([valid_token, valid_token])
assert result["duplicates"] == [valid_token]

# Wrong length
result = validate_tokens(["a" * 31])
assert len(result["invalid"]) == 1

# Non-alphanumeric
result = validate_tokens(["a!" + "b" * 30])
assert len(result["invalid"]) == 1

# Non-string
result = validate_tokens([123])
assert len(result["invalid"]) == 1

# Empty list
assert validate_tokens([]) == {"invalid": [], "duplicates": []}

print(validate_tokens([valid_token, valid_token, "a" * 31, "a!" + "b" * 30]))
