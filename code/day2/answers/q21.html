# Q21 - Rate-Limit Checker
# Given a sorted list of (user_id, timestamp) pairs,
# detect any user who has more than `threshold` requests within any `window_sec`-second window.
# Return the set of flagged user IDs.
#
# Approach:
#   - Naive:          O(n^2) — for each request check all earlier ones in the same window.
#   - No built-ins:   O(n)   — per-user list as a manual queue (pop front with index pointer).
#   - With built-ins: O(n)   — defaultdict(deque), popleft() to slide window.
#
# Key insight: sliding window per user — add current timestamp, evict timestamps
# older than (current - window_sec), if window size > threshold then flag.
#
# Edge cases:
#   - Empty list             -> set()
#   - Single user            -> fine
#   - Timestamps equal       -> multiple in same second count
#   - threshold = 1          -> any two events within window flags the user
#   - None input             -> TypeError
#   - Invalid tuple format   -> ValueError


# === APPROACH 1: NAIVE — O(n^2) ===
# "For each event, scan backwards to see how many prior events fit the window."

def rate_limit_naive(events, threshold=5, window_sec=60):
    if events is None:
        raise TypeError("Input cannot be None")
    if not isinstance(events, list):
        raise TypeError(f"Expected list, got {type(events).__name__}")
    if not isinstance(threshold, int) or threshold < 1:
        raise ValueError("threshold must be a positive integer")
    if not isinstance(window_sec, (int, float)) or window_sec <= 0:
        raise ValueError("window_sec must be a positive number")

    flagged = set()
    for i in range(len(events)):
        uid, ts = events[i]
        count = 1
        for j in range(i - 1, -1, -1):
            uid2, ts2 = events[j]
            if uid2 == uid:
                if ts - ts2 <= window_sec:
                    count += 1
                else:
                    break   # sorted input — no earlier one can fit
        if count > threshold:
            flagged.add(uid)
    return flagged


# === APPROACH 2: NO BUILT-INS — O(n) ===
# "Manual per-user list with a sliding pointer. No deque, no defaultdict."

def rate_limit_no_builtins(events, threshold=5, window_sec=60):
    if events is None:
        raise TypeError("Input cannot be None")
    if not isinstance(events, list):
        raise TypeError(f"Expected list, got {type(events).__name__}")
    if not isinstance(threshold, int) or threshold < 1:
        raise ValueError("threshold must be a positive integer")
    if not isinstance(window_sec, (int, float)) or window_sec <= 0:
        raise ValueError("window_sec must be a positive number")

    windows  = {}   # user_id -> list of timestamps in current window
    pointers = {}   # user_id -> oldest valid index in windows
    flagged  = set()

    for item in events:
        if not (isinstance(item, (tuple, list)) and len(item) == 2):
            raise ValueError(f"Expected (user_id, timestamp) pair, got {item!r}")
        uid, ts = item

        if uid not in windows:
            windows[uid]  = []
            pointers[uid] = 0

        windows[uid].append(ts)

        while windows[uid][pointers[uid]] < ts - window_sec:
            pointers[uid] += 1

        window_size = len(windows[uid]) - pointers[uid]
        if window_size > threshold:
            flagged.add(uid)

    return flagged


# === APPROACH 3: WITH BUILT-INS — O(n) ===
# "defaultdict(deque). popleft() is O(1) unlike list.pop(0)."

from collections import defaultdict, deque

def rate_limit(events, threshold=5, window_sec=60):
    if events is None:
        raise TypeError("Input cannot be None")
    if not isinstance(events, list):
        raise TypeError(f"Expected list, got {type(events).__name__}")
    if not isinstance(threshold, int) or threshold < 1:
        raise ValueError("threshold must be a positive integer")
    if not isinstance(window_sec, (int, float)) or window_sec <= 0:
        raise ValueError("window_sec must be a positive number")

    windows = defaultdict(deque)
    flagged = set()

    for item in events:
        if not (isinstance(item, (tuple, list)) and len(item) == 2):
            raise ValueError(f"Expected (user_id, timestamp) pair, got {item!r}")
        uid, ts = item
        window = windows[uid]
        window.append(ts)
        while window[0] < ts - window_sec:
            window.popleft()
        if len(window) > threshold:
            flagged.add(uid)

    return flagged

events = [
    ("alice", 0),
    ("alice", 10),
    ("alice", 20),
    ("alice", 30),
    ("alice", 40),
    ("alice", 50),   # 6 in [0,60] window -> flagged
    ("bob",   0),
    ("bob",   100),
]

assert rate_limit_naive(events)       == {"alice"}
assert rate_limit_no_builtins(events) == {"alice"}
assert rate_limit(events)             == {"alice"}

# Bob only 2 events; alice eviction — events at t=0 fall outside [50-60]=-10
# so all 6 alice events fit in 50s window -> flagged
assert "bob" not in rate_limit(events)

# Threshold 6 -> alice no longer flagged
assert rate_limit(events, threshold=6) == set()

# Empty
assert rate_limit([]) == set()

print(rate_limit(events))             # {'alice'}
print(rate_limit(events, threshold=6))# set()
