# Q22 - Reconstruct Access Order
# Given a path_map {current_node: next_node} and a start node,
# reconstruct the full traversal order by following the chain.
# Raise ValueError if a cycle is detected.
#
# Approach:
#   - Naive:          O(n) — follow the chain; use a list to check for cycles (O(n) per lookup).
#   - No built-ins:   O(n) — same chain-follow; use dict as visited set for O(1) lookup.
#   - With built-ins: O(n) — clean with path_map.get(); explicit cycle detection with a set.
#
# Edge cases:
#   - Single node, no mapping  -> [start]
#   - Chain of length 1        -> [start, next]
#   - Cycle                    -> raises ValueError("Cycle detected")
#   - Start not in map         -> [start]
#   - None input               -> TypeError
#   - start not a string       -> TypeError
#   - path_map values can be None (but are not expected in practice)


# === APPROACH 1: NAIVE — O(n^2) CYCLE CHECK ===
# "Follow the chain, use a list to detect cycles. Simple to explain."

def reconstruct_order_naive(path_map, start):
    if path_map is None:
        raise TypeError("path_map cannot be None")
    if not isinstance(path_map, dict):
        raise TypeError(f"Expected dict, got {type(path_map).__name__}")
    if start is None:
        raise TypeError("start cannot be None")
    if not isinstance(start, str):
        raise TypeError(f"start must be a string, got {type(start).__name__}")

    path    = [start]
    current = start

    while current in path_map:
        nxt = path_map[current]

        # Cycle check — O(n) scan
        found = False
        for node in path:
            if node == nxt:
                found = True
                break
        if found:
            raise ValueError(f"Cycle detected at node '{nxt}'")

        path.append(nxt)
        current = nxt

    return path


# === APPROACH 2: NO BUILT-INS — O(n) ===
# "Dict-as-visited-set gives O(1) lookup. This is what you'd say second."

def reconstruct_order_no_builtins(path_map, start):
    if path_map is None:
        raise TypeError("path_map cannot be None")
    if not isinstance(path_map, dict):
        raise TypeError(f"Expected dict, got {type(path_map).__name__}")
    if start is None:
        raise TypeError("start cannot be None")
    if not isinstance(start, str):
        raise TypeError(f"start must be a string, got {type(start).__name__}")

    path    = []
    visited = {}
    current = start

    while True:
        if current in visited:
            raise ValueError(f"Cycle detected at node '{current}'")
        visited[current] = True
        path.append(current)

        # Check for next node
        found_next = False
        for key in path_map:
            if key == current:
                found_next = True
                nxt = path_map[key]
                break
        if not found_next:
            break
        current = nxt

    return path


# === APPROACH 3: WITH BUILT-INS — O(n) ===
# "dict.get() is clean; set for visited is idiomatic Python."

def reconstruct_order(path_map, start):
    if path_map is None:
        raise TypeError("path_map cannot be None")
    if not isinstance(path_map, dict):
        raise TypeError(f"Expected dict, got {type(path_map).__name__}")
    if start is None:
        raise TypeError("start cannot be None")
    if not isinstance(start, str):
        raise TypeError(f"start must be a string, got {type(start).__name__}")

    path    = []
    visited = set()
    current = start

    while current is not None:
        if current in visited:
            raise ValueError(f"Cycle detected at node '{current}'")
        visited.add(current)
        path.append(current)
        current = path_map.get(current)

    return path


# --- tests ---
path_map = {"b": "c", "a": "b", "c": "d"}

assert reconstruct_order_naive(path_map, "a")       == ["a", "b", "c", "d"]
assert reconstruct_order_no_builtins(path_map, "a") == ["a", "b", "c", "d"]
assert reconstruct_order(path_map, "a")             == ["a", "b", "c", "d"]

# Start not in map
assert reconstruct_order(path_map, "d") == ["d"]

# Single node
assert reconstruct_order({}, "x") == ["x"]

# Cycle detection
cyclic = {"a": "b", "b": "c", "c": "a"}
try:
    reconstruct_order(cyclic, "a")
    assert False, "Should have raised ValueError"
except ValueError:
    pass

print(reconstruct_order(path_map, "a"))    # ['a', 'b', 'c', 'd']
print(reconstruct_order(path_map, "d"))    # ['d']
try:
    reconstruct_order(cyclic, "a")
except ValueError as e:
    print(e)                               # Cycle detected at node 'a'
