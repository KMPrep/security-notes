# Q23 - Find Symmetric Pairs in Firewall Rules
# Given a list of (src, dst) tuples, find all pairs where both (A,B) and (B,A) exist.
# Return each symmetric pair once, in a canonical form sorted((A,B)).
#
# Approach:
#   - Naive:          O(n^2) — for each rule, scan rest of list for its reverse.
#   - No built-ins:   O(n)   — dict-as-set; check (b,a) before recording (a,b).
#   - With built-ins: O(n)   — frozenset({a,b}) as key to unify both directions.
#
# Edge cases:
#   - Empty list           -> []
#   - (A,A) self-loop      -> (A,A) and (A,A) trivially symmetric; included.
#   - No symmetric pairs   -> []
#   - Duplicate rules      -> dedup; same pair appears at most once in output.
#   - None input           -> TypeError
#   - Non-tuple entries    -> ValueError


# === APPROACH 1: NAIVE — O(n^2) ===
# "For each rule scan the list for its reverse. Clean but slow."

def find_symmetric_pairs_naive(rules):
    if rules is None:
        raise TypeError("Input cannot be None")
    if not isinstance(rules, list):
        raise TypeError(f"Expected list, got {type(rules).__name__}")

    result     = []
    result_set = set()

    for i in range(len(rules)):
        if not (isinstance(rules[i], (tuple, list)) and len(rules[i]) == 2):
            raise ValueError(f"Expected (src, dst) pair at index {i}, got {rules[i]!r}")
        a, b = rules[i]
        canonical = (min(a, b), max(a, b))

        if canonical in result_set:
            continue  # already recorded

        for j in range(len(rules)):
            if i == j:
                continue
            if not (isinstance(rules[j], (tuple, list)) and len(rules[j]) == 2):
                raise ValueError(f"Expected (src, dst) pair at index {j}, got {rules[j]!r}")
            c, d = rules[j]
            if c == b and d == a:
                result_set.add(canonical)
                result.append(canonical)
                break

    return sorted(result)


# === APPROACH 2: NO BUILT-INS — O(n) ===
# "Store seen rules in a dict. Before storing (a,b) check if (b,a) exists."

def find_symmetric_pairs_no_builtins(rules):
    if rules is None:
        raise TypeError("Input cannot be None")
    if not isinstance(rules, list):
        raise TypeError(f"Expected list, got {type(rules).__name__}")

    seen   = {}
    result = {}  # canonical -> True, acts as ordered dedup set

    for i in range(len(rules)):
        item = rules[i]
        if not (isinstance(item, (tuple, list)) and len(item) == 2):
            raise ValueError(f"Expected (src, dst) pair at index {i}, got {item!r}")
        a, b = item

        if (b, a) in seen:
            canonical = (min(a, b), max(a, b))
            result[canonical] = True
        else:
            seen[(a, b)] = True

    out = []
    for pair in result:
        out.append(pair)

    # manual sort by first element, then second
    for i in range(len(out)):
        for j in range(i + 1, len(out)):
            if out[i] > out[j]:
                out[i], out[j] = out[j], out[i]

    return out


# === APPROACH 3: WITH BUILT-INS — O(n) ===
# "frozenset({a,b}) collapses both directions into one key."

def find_symmetric_pairs(rules):
    if rules is None:
        raise TypeError("Input cannot be None")
    if not isinstance(rules, list):
        raise TypeError(f"Expected list, got {type(rules).__name__}")
    if not all(isinstance(r, (tuple, list)) and len(r) == 2 for r in rules):
        raise ValueError("All entries must be (src, dst) pairs")

    rule_set = set()
    for r in rules:
        rule_set.add(tuple(r))

    found = set()
    for a, b in rule_set:
        if (b, a) in rule_set:
            found.add(frozenset({a, b}) if a != b else (a, b))

    return sorted(tuple(sorted(pair)) for pair in found)


# --- tests ---
rules = [
    ("A", "B"),
    ("B", "A"),
    ("C", "D"),
    ("D", "C"),
    ("E", "F"),
]

assert find_symmetric_pairs_naive(rules)       == [("A", "B"), ("C", "D")]
assert find_symmetric_pairs_no_builtins(rules) == [("A", "B"), ("C", "D")]
assert find_symmetric_pairs(rules)             == [("A", "B"), ("C", "D")]

# No symmetric pairs
assert find_symmetric_pairs([("A", "B"), ("C", "D")]) == []

# Empty
assert find_symmetric_pairs([]) == []

# Self-loop
assert find_symmetric_pairs([("X", "X")]) == [("X", "X")]

print(find_symmetric_pairs(rules))             # [('A', 'B'), ('C', 'D')]
print(find_symmetric_pairs([("A","B")]))       # []
