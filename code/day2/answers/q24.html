# Q24 - Count Distinct Alert Types in Rolling Window
# Given a list of alert type strings and a window size k,
# return the number of distinct alert types in each window of size k.
#
# n = len(alerts), windows = n-k+1.
# e.g. ["XSS","SQLi","XSS","CSRF","SQLi"] k=3 -> windows: [0..2],[1..3],[2..4] -> [2,3,3]
#
# Approach:
#   - Naive:          O(n * k) — recompute distinct count for each window from scratch.
#   - No built-ins:   O(n)     — maintain freq dict + distinct counter; add/remove manually.
#   - With built-ins: O(n)     — same with Counter.
#
# Key insight for O(n):
#   When right pointer advances: freq[new]++; if it was 0 before, distinct++.
#   When left pointer advances:  freq[old]--; if it becomes 0 now, distinct--.
#
# Edge cases:
#   - k > n      -> [] (no full window)
#   - k == n     -> [distinct count of entire list]
#   - k == 1     -> [1, 1, ..., 1]  (each window is a single element)
#   - None input -> TypeError
#   - k <= 0     -> ValueError


# === APPROACH 1: NAIVE — O(n * k) ===
# "Rebuild a set for every window. Says you understand the problem; mention it's slow."

def distinct_in_windows_naive(alerts, k):
    if alerts is None:
        raise TypeError("Input cannot be None")
    if not isinstance(alerts, list):
        raise TypeError(f"Expected list, got {type(alerts).__name__}")
    if not isinstance(k, int) or k <= 0:
        raise ValueError(f"k must be a positive integer, got {k!r}")

    n      = len(alerts)
    result = []

    for i in range(n - k + 1):
        window_set = {}
        for j in range(i, i + k):
            window_set[alerts[j]] = True
        result.append(len(window_set))

    return result


# === APPROACH 2: SLIDING WINDOW — NO BUILT-INS — O(n) ===
# "Maintain freq dict + distinct counter. Show the add/remove logic clearly."

def distinct_in_windows_no_builtins(alerts, k):
    if alerts is None:
        raise TypeError("Input cannot be None")
    if not isinstance(alerts, list):
        raise TypeError(f"Expected list, got {type(alerts).__name__}")
    if not isinstance(k, int) or k <= 0:
        raise ValueError(f"k must be a positive integer, got {k!r}")

    n = len(alerts)
    if n < k:
        return []

    freq     = {}
    distinct = 0
    result   = []

    # Build initial window
    for i in range(k):
        elem = alerts[i]
        if elem not in freq:
            freq[elem] = 0
        if freq[elem] == 0:
            distinct += 1
        freq[elem] = freq[elem] + 1

    result.append(distinct)

    # Slide the window
    for right in range(k, n):
        # Add incoming element (right)
        incoming = alerts[right]
        if incoming not in freq:
            freq[incoming] = 0
        if freq[incoming] == 0:
            distinct += 1
        freq[incoming] = freq[incoming] + 1

        # Remove outgoing element (right - k)
        outgoing = alerts[right - k]
        freq[outgoing] = freq[outgoing] - 1
        if freq[outgoing] == 0:
            distinct -= 1

        result.append(distinct)

    return result


# === APPROACH 3: WITH BUILT-INS — O(n) ===
# "Counter-based. Same logic; more Pythonic."

from collections import Counter

def distinct_in_windows(alerts, k):
    if alerts is None:
        raise TypeError("Input cannot be None")
    if not isinstance(alerts, list):
        raise TypeError(f"Expected list, got {type(alerts).__name__}")
    if not isinstance(k, int) or k <= 0:
        raise ValueError(f"k must be a positive integer, got {k!r}")

    n = len(alerts)
    if n < k:
        return []

    freq     = Counter(alerts[:k])
    distinct = len(freq)
    result   = [distinct]

    for right in range(k, n):
        incoming = alerts[right]
        outgoing = alerts[right - k]

        freq[incoming] += 1
        if freq[incoming] == 1:
            distinct += 1

        freq[outgoing] -= 1
        if freq[outgoing] == 0:
            distinct -= 1
            del freq[outgoing]

        result.append(distinct)

    return result


# --- tests ---
alerts = ["XSS", "SQLi", "XSS", "CSRF", "SQLi"]
k = 3
# Window [0..2]: {XSS, SQLi}       -> 2
# Window [1..3]: {SQLi, XSS, CSRF} -> 3
# Window [2..4]: {XSS, CSRF, SQLi} -> 3

assert distinct_in_windows_naive(alerts, k)       == [2, 3, 3]
assert distinct_in_windows_no_builtins(alerts, k) == [2, 3, 3]
assert distinct_in_windows(alerts, k)             == [2, 3, 3]

assert distinct_in_windows([], 3)                        == []
assert distinct_in_windows(["A", "B"], 5)                == []
assert distinct_in_windows(["A", "A", "A"], 2)           == [1, 1]
assert distinct_in_windows(["A", "B", "C", "D"], 1)      == [1, 1, 1, 1]
assert distinct_in_windows(["A", "B", "C", "D"], 4)      == [4]

print(distinct_in_windows(alerts, k))             # [2, 3, 3]
print(distinct_in_windows(["A","A","B","A"], 2))  # [1, 2, 2]
