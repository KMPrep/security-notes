# Q25 - Minimum Window Containing All Severity Levels
# Given a list of log severity strings and a required set of severity levels,
# find the smallest contiguous subarray (window) that contains at least one of each required level.
# Return the actual subarray, or None if impossible.
#
# Approach:
#   - Naive:          O(n^2) — try all subarrays, track which required levels are covered.
#   - No built-ins:   O(n)   — two-pointer / sliding window; manual freq dict + covered counter.
#   - With built-ins: O(n)   — same with Counter; clean conditionals.
#
# Key insight (classic minimum window substring):
#   Expand right until all required levels are covered.
#   Then shrink left as far as possible while still covered.
#   Track minimum window length seen.
#
# Edge cases:
#   - Empty log or empty required -> None (or [] if required is empty)
#   - Required level missing entirely -> None
#   - Exact match                  -> that one-element window (if one element covers all)
#   - None input                   -> TypeError
#   - Custom required set          -> parameterised


DEFAULT_REQUIRED = {"INFO", "WARN", "ERROR"}


# === APPROACH 1: NAIVE — O(n^2) ===
# "Check all subarrays. Correct; mention complexity before optimizing."

def min_window_naive(logs, required=None):
    if logs is None:
        raise TypeError("Input cannot be None")
    if not isinstance(logs, list):
        raise TypeError(f"Expected list, got {type(logs).__name__}")
    if required is None:
        required = DEFAULT_REQUIRED
    if not required:
        return []

    n        = len(logs)
    best_len = n + 1
    best_start = -1

    for i in range(n):
        coverage = {}
        for j in range(i, n):
            elem = logs[j]
            if elem in required:
                if elem not in coverage:
                    coverage[elem] = 0
                coverage[elem] += 1

            if len(coverage) == len(required):
                window_len = j - i + 1
                if window_len < best_len:
                    best_len   = window_len
                    best_start = i
                break   # no need to go further right from this i

    if best_start == -1:
        return None
    return logs[best_start: best_start + best_len]


# === APPROACH 2: SLIDING WINDOW — NO BUILT-INS — O(n) ===
# "Expand right; shrink left. Classic two-pointer pattern."

def min_window_no_builtins(logs, required=None):
    if logs is None:
        raise TypeError("Input cannot be None")
    if not isinstance(logs, list):
        raise TypeError(f"Expected list, got {type(logs).__name__}")
    if required is None:
        required = DEFAULT_REQUIRED
    if not required:
        return []

    n        = len(logs)
    need     = len(required)
    freq     = {}       # count of required elements in current window
    covered  = 0        # how many distinct required levels have freq >= 1

    best_len   = n + 1
    best_start = -1
    left       = 0

    for right in range(n):
        elem = logs[right]
        if elem in required:
            if elem not in freq:
                freq[elem] = 0
            freq[elem] += 1
            if freq[elem] == 1:
                covered += 1

        while covered == need:
            window_len = right - left + 1
            if window_len < best_len:
                best_len   = window_len
                best_start = left

            outgoing = logs[left]
            left += 1
            if outgoing in required:
                freq[outgoing] -= 1
                if freq[outgoing] == 0:
                    covered -= 1

    if best_start == -1:
        return None
    return logs[best_start: best_start + best_len]


# === APPROACH 3: WITH BUILT-INS — O(n) ===
# "Counter + any() check. Clean and easy to explain."

from collections import Counter

def min_window(logs, required=None):
    if logs is None:
        raise TypeError("Input cannot be None")
    if not isinstance(logs, list):
        raise TypeError(f"Expected list, got {type(logs).__name__}")
    if required is None:
        required = DEFAULT_REQUIRED
    if not required:
        return []

    n        = len(logs)
    need     = len(required)
    freq     = Counter()
    covered  = 0

    best_len   = n + 1
    best_start = -1
    left       = 0

    for right, elem in enumerate(logs):
        if elem in required:
            freq[elem] += 1
            if freq[elem] == 1:
                covered += 1

        while covered == need:
            if right - left + 1 < best_len:
                best_len   = right - left + 1
                best_start = left

            outgoing = logs[left]
            left += 1
            if outgoing in required:
                freq[outgoing] -= 1
                if freq[outgoing] == 0:
                    covered -= 1

    if best_start == -1:
        return None
    return logs[best_start: best_start + best_len]


# --- tests ---
logs1 = ["INFO", "DEBUG", "WARN", "ERROR", "INFO", "WARN"]
# Minimum window is ["WARN","ERROR"] at [2..3] length 2? No:
# [2..3] = ["WARN","ERROR"] missing INFO -> not covered.
# Smallest covering all 3: [0..3] = ["INFO","DEBUG","WARN","ERROR"] len 4
# OR [2..4] = ["WARN","ERROR","INFO"] len 3  <- that is the minimum

result = min_window(logs1)
assert result is not None
assert set(result) >= DEFAULT_REQUIRED
assert len(result) == 3

logs2 = ["INFO", "WARN", "ERROR"]
assert min_window(logs2) == ["INFO", "WARN", "ERROR"]

logs3 = ["INFO", "DEBUG", "INFO"]
assert min_window(logs3) is None   # no ERROR or WARN

assert min_window([]) is None
assert min_window(["INFO", "WARN", "ERROR", "INFO"]) is not None

# Check naive and no-builtins agree
for logs in [logs1, logs2, logs3, []]:
    r1 = min_window_naive(logs)
    r2 = min_window_no_builtins(logs)
    r3 = min_window(logs)
    if r1 is None:
        assert r2 is None and r3 is None
    else:
        assert len(r1) == len(r2) == len(r3)
        assert set(r1) >= DEFAULT_REQUIRED

print(min_window(logs1))           # e.g. ['WARN', 'ERROR', 'INFO']
print(min_window(logs2))           # ['INFO', 'WARN', 'ERROR']
print(min_window(logs3))           # None
print(min_window([]))              # None
