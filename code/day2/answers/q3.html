# Q3 - Longest Consecutive Sequence of Failed Logins
# Given a list of login results ("OK" or "FAIL"), find the longest consecutive run of "FAIL".
#
# Approach:
#   - Naive / Optimized:  O(n) — linear scan with running counter.
#                         This problem is already O(n); the value is
#                         in adding position tracking and edge case handling.
#   - With built-ins:     O(n) — itertools.groupby for elegant grouping.
#
# Security context: Detects brute-force login bursts. If run >= threshold, trigger lockout.
#
# Edge cases:
#   - Empty list               -> 0
#   - All OK                   -> 0
#   - All FAIL                 -> len(list)
#   - None input               -> TypeError
#   - Non-string entries       -> ValueError


# === APPROACH 1: NAIVE — O(n) ===
# "Running counter — what most people write first."

def longest_fail_run_naive(results):
    if results is None:
        raise TypeError("Input cannot be None")
    if not isinstance(results, list):
        raise TypeError(f"Expected list, got {type(results).__name__}")

    max_run = 0
    current_run = 0

    for i in range(len(results)):
        if not isinstance(results[i], str):
            raise ValueError(f"Expected string at index {i}, got {type(results[i]).__name__}")
        if results[i] == "FAIL":
            current_run += 1
            if current_run > max_run:
                max_run = current_run
        else:
            current_run = 0

    return max_run


# === APPROACH 2: WITH POSITION TRACKING — NO BUILT-INS — O(n) ===
# "Google follow-up: 'Where does the longest run start and end?'"

def longest_fail_run_with_position(results):
    if results is None:
        raise TypeError("Input cannot be None")
    if not isinstance(results, list):
        raise TypeError(f"Expected list, got {type(results).__name__}")

    max_run, max_start, max_end = 0, -1, -1
    current_run, current_start = 0, 0

    for i in range(len(results)):
        if not isinstance(results[i], str):
            raise ValueError(f"Non-string at index {i}: {results[i]!r}")
        if results[i] == "FAIL":
            if current_run == 0:
                current_start = i
            current_run += 1
            if current_run > max_run:
                max_run = current_run
                max_start = current_start
                max_end = i
        else:
            current_run = 0

    return max_run, max_start, max_end


# === APPROACH 3: WITH BUILT-INS — O(n) ===
# "itertools.groupby groups consecutive equal elements cleanly."

from itertools import groupby

def longest_fail_run(results):
    if results is None:
        raise TypeError("Input cannot be None")
    if not isinstance(results, list):
        raise TypeError(f"Expected list, got {type(results).__name__}")
    if not all(isinstance(r, str) for r in results):
        raise ValueError("All entries must be strings")

    max_run = 0
    for key, group in groupby(results):
        if key == "FAIL":
            run_length = sum(1 for _ in group)
            if run_length > max_run:
                max_run = run_length
    return max_run


# --- tests ---
sample = ["OK", "FAIL", "FAIL", "FAIL", "OK", "FAIL", "FAIL"]

assert longest_fail_run_naive(sample) == 3
assert longest_fail_run(sample) == 3

run, start, end = longest_fail_run_with_position(sample)
assert run == 3 and start == 1 and end == 3

assert longest_fail_run([]) == 0
assert longest_fail_run(["OK", "OK", "OK"]) == 0
assert longest_fail_run(["FAIL", "FAIL", "FAIL"]) == 3
assert longest_fail_run(["FAIL"]) == 1

print(longest_fail_run(sample))                                    # 3
print(longest_fail_run_with_position(sample))                      # (3, 1, 3)
print(longest_fail_run(["FAIL", "FAIL", "OK", "FAIL", "FAIL"]))   # 2
