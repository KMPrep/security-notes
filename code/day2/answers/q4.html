# Q4 - Rotate Access Log Window
# Given a list of log lines and an integer k, rotate the list LEFT by k positions.
#
# Approach:
#   - Naive:            O(n*k) — rotate one position at a time, k times.
#   - No built-ins:     O(n)   — reverse-based in-place rotation (3-reversal trick).
#   - With built-ins:   O(n)   — slicing (logs[k:] + logs[:k]).
#
# The 3-reversal trick:
#   Left-rotate by k == reverse all -> reverse first (n-k) -> reverse last k
#   Intuition: rotating [a,b,c,d,e] left by 2 == [c,d,e,a,b]
#   Step 1 reverse all:      [e,d,c,b,a]
#   Step 2 reverse [0..n-k): [c,d,e,b,a]   (n-k = 3)
#   Step 3 reverse [n-k..n): [c,d,e,a,b]   ✓
#
# Edge cases:
#   - k == 0 or k == n      -> original list
#   - k > n                 -> k % n (normalised)
#   - Empty list            -> []
#   - None / non-list input -> TypeError
#   - k < 0                 -> ValueError


# === APPROACH 1: NAIVE — O(n*k) ===
# "Shift one element at a time, k times. Works but slow for large k."

def rotate_left_naive(logs, k):
    if logs is None:
        raise TypeError("Input cannot be None")
    if not isinstance(logs, list):
        raise TypeError(f"Expected list, got {type(logs).__name__}")
    if not isinstance(k, int):
        raise TypeError(f"k must be an integer, got {type(k).__name__}")
    if k < 0:
        raise ValueError(f"k must be non-negative, got {k}")

    n = len(logs)
    if n == 0:
        return []

    k = k % n
    result = logs[:]

    for _ in range(k):
        first = result[0]
        for i in range(1, n):
            result[i - 1] = result[i]
        result[n - 1] = first

    return result


# === APPROACH 2: 3-REVERSAL TRICK — NO BUILT-INS — O(n) ===
# "Reverse trick: no extra memory beyond the copy, O(n) time."

def rotate_left_no_builtins(logs, k):
    if logs is None:
        raise TypeError("Input cannot be None")
    if not isinstance(logs, list):
        raise TypeError(f"Expected list, got {type(logs).__name__}")
    if not isinstance(k, int):
        raise TypeError(f"k must be an integer, got {type(k).__name__}")
    if k < 0:
        raise ValueError(f"k must be non-negative, got {k}")

    n = len(logs)
    if n == 0:
        return []

    k = k % n
    if k == 0:
        return logs[:]

    result = logs[:]

    def reverse_segment(arr, left, right):
        while left < right:
            arr[left], arr[right] = arr[right], arr[left]
            left += 1
            right -= 1

    reverse_segment(result, 0, n - 1)
    reverse_segment(result, 0, n - k - 1)
    reverse_segment(result, n - k, n - 1)

    return result


# === APPROACH 3: WITH SLICING — O(n) ===
# "Most readable Python. logs[k:] + logs[:k] creates a new list."

def rotate_left(logs, k):
    if logs is None:
        raise TypeError("Input cannot be None")
    if not isinstance(logs, list):
        raise TypeError(f"Expected list, got {type(logs).__name__}")
    if not isinstance(k, int):
        raise TypeError(f"k must be an integer, got {type(k).__name__}")
    if k < 0:
        raise ValueError(f"k must be non-negative, got {k}")

    n = len(logs)
    if n == 0:
        return []

    k = k % n
    return logs[k:] + logs[:k]


# --- tests ---
logs = ["a", "b", "c", "d", "e"]

assert rotate_left_naive(logs, 2)      == ["c", "d", "e", "a", "b"]
assert rotate_left_no_builtins(logs, 2) == ["c", "d", "e", "a", "b"]
assert rotate_left(logs, 2)            == ["c", "d", "e", "a", "b"]

assert rotate_left(logs, 0) == ["a", "b", "c", "d", "e"]   # no rotation
assert rotate_left(logs, 5) == ["a", "b", "c", "d", "e"]   # full rotation
assert rotate_left(logs, 7) == ["c", "d", "e", "a", "b"]   # k=7%5=2
assert rotate_left([], 3)   == []

print(rotate_left(logs, 2))  # ['c', 'd', 'e', 'a', 'b']
print(rotate_left(logs, 0))  # ['a', 'b', 'c', 'd', 'e']
print(rotate_left(logs, 7))  # ['c', 'd', 'e', 'a', 'b']
