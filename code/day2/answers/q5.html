# Q5 - Sliding Window: Peak Traffic Detection
# Given a list of request counts per second and window size k,
# find the max total requests in any window of size k.
#
# Approach:
#   - Naive:          O(n*k) — recompute sum from scratch for each window.
#   - No built-ins:   O(n)   — sliding window: add incoming, subtract outgoing.
#   - With built-ins: O(n)   — same logic, using sum() + max().
#
# Key insight: each element enters the window once and leaves once -> O(n).
#
# Edge cases:
#   - k == len(list)         -> sum of entire list
#   - k > len(list)          -> ValueError
#   - Negative request count -> ValueError (counts can't be negative)
#   - Empty list             -> 0
#   - None input             -> TypeError


# === APPROACH 1: NAIVE — O(n*k) ===
# "Recompute the window sum from scratch every time."

def max_window_sum_naive(requests, k):
    if requests is None:
        raise TypeError("Input cannot be None")
    if not isinstance(requests, list):
        raise TypeError(f"Expected list, got {type(requests).__name__}")
    if not isinstance(k, int) or k <= 0:
        raise ValueError(f"k must be a positive integer, got {k!r}")
    if len(requests) == 0:
        return 0
    if len(requests) < k:
        raise ValueError(f"List length {len(requests)} is less than window size {k}")

    max_sum = 0
    n = len(requests)

    for i in range(n - k + 1):
        window_sum = 0
        for j in range(i, i + k):
            if not isinstance(requests[j], (int, float)):
                raise ValueError(f"Non-numeric at index {j}: {requests[j]!r}")
            window_sum += requests[j]
        if window_sum > max_sum:
            max_sum = window_sum

    return max_sum


# === APPROACH 2: SLIDING WINDOW — NO BUILT-INS — O(n) ===
# "Build first window sum, then slide: add new element, drop leftmost."

def max_window_sum_no_builtins(requests, k):
    if requests is None:
        raise TypeError("Input cannot be None")
    if not isinstance(requests, list):
        raise TypeError(f"Expected list, got {type(requests).__name__}")
    if not isinstance(k, int) or k <= 0:
        raise ValueError(f"k must be a positive integer, got {k!r}")

    n = len(requests)
    if n == 0:
        return 0
    if n < k:
        raise ValueError(f"List length {n} is less than window size {k}")

    for i in range(n):
        if not isinstance(requests[i], (int, float)):
            raise ValueError(f"Non-numeric at index {i}: {requests[i]!r}")
        if requests[i] < 0:
            raise ValueError(f"Negative request count at index {i}: {requests[i]}")

    # Build first window
    window_sum = 0
    for i in range(k):
        window_sum += requests[i]

    max_sum = window_sum

    # Slide
    for i in range(k, n):
        window_sum += requests[i] - requests[i - k]
        if window_sum > max_sum:
            max_sum = window_sum

    return max_sum


# === APPROACH 3: WITH BUILT-INS — O(n) ===
# "Same sliding window, using sum() for the initial window and max()."

def max_window_sum(requests, k):
    if requests is None:
        raise TypeError("Input cannot be None")
    if not isinstance(requests, list):
        raise TypeError(f"Expected list, got {type(requests).__name__}")
    if not isinstance(k, int) or k <= 0:
        raise ValueError(f"k must be a positive integer, got {k!r}")

    n = len(requests)
    if n == 0:
        return 0
    if n < k:
        raise ValueError(f"List length {n} is less than window size {k}")
    if not all(isinstance(x, (int, float)) for x in requests):
        raise ValueError("All request counts must be numeric")
    if any(x < 0 for x in requests):
        raise ValueError("Request counts cannot be negative")

    window_sum = sum(requests[:k])
    max_sum = window_sum

    for i in range(k, n):
        window_sum += requests[i] - requests[i - k]
        max_sum = max(max_sum, window_sum)

    return max_sum

assert max_window_sum([1, 2, 3, 4, 5], 5) == 15
