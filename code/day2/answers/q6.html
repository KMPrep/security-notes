# Q6 - First Non-Repeated Error Code
# Given a string of error codes separated by spaces, find the first error code
# that appears only once.
#
# Approach:
#   - Naive:          O(n^2) — for each code, rescan entire list for its count.
#   - No built-ins:   O(n)   — two-pass with manual freq dict + manual tokenizer.
#   - With built-ins: O(n)   — dict.get() two-pass, or OrderedDict single-pass.
#
# Edge cases:
#   - Empty string          -> None
#   - All codes repeated    -> None
#   - Single code           -> that code
#   - None input            -> TypeError
#   - Extra whitespace      -> handled by strip + split


# === APPROACH 1: NAIVE — O(n^2) ===
# "For each code, count it again from scratch. Slow but clear."

def first_non_repeated_naive(error_string):
    if error_string is None:
        raise TypeError("Input cannot be None")
    if not isinstance(error_string, str):
        raise TypeError(f"Expected string, got {type(error_string).__name__}")

    error_string = error_string.strip()
    if not error_string:
        return None

    # Manual tokenise (no .split())
    codes = []
    current = ""
    for ch in error_string + " ":
        if ch == " ":
            if current:
                codes.append(current)
                current = ""
        else:
            current += ch

    for i in range(len(codes)):
        count = 0
        for j in range(len(codes)):
            if codes[i] == codes[j]:
                count += 1
        if count == 1:
            return codes[i]

    return None


# === APPROACH 2: TWO-PASS FREQ DICT — NO BUILT-INS — O(n) ===
# "Count all, then return the first with count 1. Preserves order."

def first_non_repeated_no_builtins(error_string):
    if error_string is None:
        raise TypeError("Input cannot be None")
    if not isinstance(error_string, str):
        raise TypeError(f"Expected string, got {type(error_string).__name__}")

    error_string = error_string.strip()
    if not error_string:
        return None

    codes = []
    current = ""
    for ch in error_string + " ":
        if ch == " ":
            if current:
                codes.append(current)
                current = ""
        else:
            current += ch

    freq = {}
    for code in codes:
        freq[code] = freq[code] + 1 if code in freq else 1

    for code in codes:
        if freq[code] == 1:
            return code

    return None


# === APPROACH 3: WITH BUILT-INS — O(n) ===
# "dict.get() two-pass. In Python 3.7+ dicts are insertion-ordered."

def first_non_repeated(error_string):
    if error_string is None:
        raise TypeError("Input cannot be None")
    if not isinstance(error_string, str):
        raise TypeError(f"Expected string, got {type(error_string).__name__}")

    error_string = error_string.strip()
    if not error_string:
        return None

    codes = error_string.split()
    freq = {}
    for code in codes:
        freq[code] = freq.get(code, 0) + 1

    for code in codes:
        if freq[code] == 1:
            return code

    return None


# --- tests ---
assert first_non_repeated_naive("E404 E200 E404 E500 E200 E403") == "E500"
assert first_non_repeated_no_builtins("E404 E200 E404 E500 E200 E403") == "E500"
assert first_non_repeated("E404 E200 E404 E500 E200 E403") == "E500"

assert first_non_repeated("") is None
assert first_non_repeated("E404 E404") is None
assert first_non_repeated("E200") == "E200"
assert first_non_repeated("  E100  E200  E100  ") == "E200"

print(first_non_repeated("E404 E200 E404 E500 E200 E403"))  # E500
print(first_non_repeated(""))                               # None
print(first_non_repeated("E200"))                           # E200
