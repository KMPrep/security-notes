# Q7 - Valid Bracket Depth in Config String
# Given a config string with nested brackets, return the maximum nesting depth.
# Return -1 if brackets are unmatched or mismatched.
#
# Approach:
#   - Naive / Optimized:  O(n) — single scan with depth counter (curly only).
#   - No built-ins:       O(n) — stack-based, handles mixed {}, [], ().
#   - With built-ins:     O(n) — dict for matching map + list as stack.
#
# Edge cases:
#   - Empty string          -> 0
#   - No brackets           -> 0
#   - Extra closing bracket -> -1
#   - Unclosed opening      -> -1
#   - Mismatched types      -> -1  e.g. "{]"
#   - None input            -> TypeError


# === APPROACH 1: NAIVE — O(n) — CURLY BRACES ONLY ===
# "Counter approach. Works for single bracket type — most people start here."

def max_bracket_depth_naive(config):
    if config is None:
        raise TypeError("Input cannot be None")
    if not isinstance(config, str):
        raise TypeError(f"Expected string, got {type(config).__name__}")

    depth = 0
    max_depth = 0

    for ch in config:
        if ch == "{":
            depth += 1
            if depth > max_depth:
                max_depth = depth
        elif ch == "}":
            depth -= 1
            if depth < 0:
                return -1  # more closing than opening

    return -1 if depth != 0 else max_depth


# === APPROACH 2: STACK-BASED — NO BUILT-INS — O(n) ===
# "Google follow-up: 'What if there are {}, [], and () brackets?'
#  Stack handles all types and detects mismatches."

def max_bracket_depth_no_builtins(config):
    if config is None:
        raise TypeError("Input cannot be None")
    if not isinstance(config, str):
        raise TypeError(f"Expected string, got {type(config).__name__}")

    open_brackets  = ["{", "[", "("]
    close_brackets = ["}", "]", ")"]

    # Build match map manually
    match = {}
    for i in range(len(open_brackets)):
        match[close_brackets[i]] = open_brackets[i]

    stack = []
    max_depth = 0

    for ch in config:
        if ch in open_brackets:
            stack.append(ch)
            if len(stack) > max_depth:
                max_depth = len(stack)
        elif ch in close_brackets:
            if len(stack) == 0 or stack[-1] != match[ch]:
                return -1
            stack.pop()

    return -1 if len(stack) != 0 else max_depth


# === APPROACH 3: WITH BUILT-INS — O(n) ===
# "Clean dict-based matcher. Same O(n), more readable."

def max_bracket_depth(config):
    if config is None:
        raise TypeError("Input cannot be None")
    if not isinstance(config, str):
        raise TypeError(f"Expected string, got {type(config).__name__}")

    matching = {"}": "{", "]": "[", ")": "("}
    openers  = set(matching.values())
    stack    = []
    max_depth = 0

    for ch in config:
        if ch in openers:
            stack.append(ch)
            max_depth = max(max_depth, len(stack))
        elif ch in matching:
            if not stack or stack[-1] != matching[ch]:
                return -1
            stack.pop()

    return -1 if stack else max_depth


# --- tests ---
assert max_bracket_depth_naive("{host: {ip: {range: 10}}}") == 3
assert max_bracket_depth_no_builtins("{host: {ip: {range: 10}}}") == 3
assert max_bracket_depth("{host: {ip: {range: 10}}}") == 3

assert max_bracket_depth("") == 0
assert max_bracket_depth("{}") == 1
assert max_bracket_depth("}{") == -1
assert max_bracket_depth("{host: {ip: {range: 10}}") == -1   # unclosed
assert max_bracket_depth("{host: [ip: (range: 10)]}") == 3
assert max_bracket_depth("{host: [ip: (range: 10}]}") == -1  # mismatch

print(max_bracket_depth("{host: {ip: {range: 10}}}"))   # 3
print(max_bracket_depth("{host: {ip: {range: 10}}"))    # -1
print(max_bracket_depth("{host: [ip: (range: 10)]}"))   # 3
print(max_bracket_depth("{host: [ip: (range: 10}]}"))   # -1
