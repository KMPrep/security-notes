# Q8 - Caesar Cipher Detect and Decode
# Given ciphertext and a known plaintext word that must appear in the decoded string,
# find the shift (1-25) and return the decoded string.
#
# Approach:
#   - Naive / Optimized:  O(25 * n) = O(n) — try all 25 shifts (alphabet is bounded).
#   - No built-ins:       O(n) — manual lowercase, manual alpha check, manual substr search.
#   - With built-ins:     O(n) — str.isalpha(), chr/ord arithmetic.
#
# Security note: Caesar cipher is trivially broken by frequency analysis or brute force.
# Real encryption uses AES-GCM or ChaCha20-Poly1305. This tests modular arithmetic.
#
# Edge cases:
#   - known_word not found in any shift -> (None, ciphertext)
#   - Empty ciphertext                  -> (None, "")
#   - Empty known_word                  -> ValueError
#   - None inputs                       -> TypeError
#   - Uppercase / mixed case            -> normalised to lowercase


# === APPROACH 1: BRUTE FORCE — O(25*n) ===
# "Try all 25 shifts. With a bounded alphabet this is effectively O(n)."

def caesar_decode_naive(ciphertext, known_word):
    if ciphertext is None or known_word is None:
        raise TypeError("Inputs cannot be None")
    if not isinstance(ciphertext, str) or not isinstance(known_word, str):
        raise TypeError("Both inputs must be strings")
    if not known_word.strip():
        raise ValueError("known_word cannot be empty or whitespace")

    ct = ciphertext.lower()
    kw = known_word.lower()

    for shift in range(1, 26):
        decoded = ""
        for ch in ct:
            if ch.isalpha():
                decoded += chr((ord(ch) - ord('a') - shift) % 26 + ord('a'))
            else:
                decoded += ch
        if kw in decoded:
            return shift, decoded

    return None, ciphertext


# === APPROACH 2: NO BUILT-INS — O(25*n) ===
# "Manual lowercase, alpha check, and substring search — no str methods."

def caesar_decode_no_builtins(ciphertext, known_word):
    if ciphertext is None or known_word is None:
        raise TypeError("Inputs cannot be None")
    if not isinstance(ciphertext, str) or not isinstance(known_word, str):
        raise TypeError("Both inputs must be strings")

    def to_lower(s):
        result = ""
        for ch in s:
            code = ord(ch)
            result += chr(code + 32) if 65 <= code <= 90 else ch
        return result

    def is_alpha(ch):
        c = ord(ch)
        return (97 <= c <= 122) or (65 <= c <= 90)

    def contains(text, pattern):
        plen = len(pattern)
        if plen > len(text):
            return False
        for i in range(len(text) - plen + 1):
            match = True
            for j in range(plen):
                if text[i + j] != pattern[j]:
                    match = False
                    break
            if match:
                return True
        return False

    ct = to_lower(ciphertext)
    kw = to_lower(known_word)

    for shift in range(1, 26):
        decoded = ""
        for ch in ct:
            if is_alpha(ch):
                decoded += chr((ord(ch) - 97 - shift) % 26 + 97)
            else:
                decoded += ch
        if contains(decoded, kw):
            return shift, decoded

    return None, ciphertext


# === APPROACH 3: WITH BUILT-INS — O(25*n) ===
# "Same logic, idiomatic Python. str.maketrans alternative shown as bonus."

def caesar_decode(ciphertext, known_word):
    if ciphertext is None or known_word is None:
        raise TypeError("Inputs cannot be None")
    if not isinstance(ciphertext, str) or not isinstance(known_word, str):
        raise TypeError("Both inputs must be strings")
    if not known_word.strip():
        raise ValueError("known_word cannot be empty or whitespace")

    ct = ciphertext.lower()
    kw = known_word.lower()

    for shift in range(1, 26):
        decoded = "".join(
            chr((ord(ch) - ord('a') - shift) % 26 + ord('a')) if ch.isalpha() else ch
            for ch in ct
        )
        if kw in decoded:
            return shift, decoded

    return None, ciphertext


# --- tests ---
assert caesar_decode_naive("khoor zruog", "hello") == (3, "hello world")
assert caesar_decode_no_builtins("khoor zruog", "hello") == (3, "hello world")
assert caesar_decode("khoor zruog", "hello") == (3, "hello world")

shift, decoded = caesar_decode("khoor zruog", "hello")
assert shift == 3
assert decoded == "hello world"

assert caesar_decode("abc", "xyz") == (None, "abc")  # no shift produces "xyz" in "abc"

print(caesar_decode("khoor zruog", "hello"))  # (3, 'hello world')
print(caesar_decode("abc", "xyz"))            # (None, 'abc')
