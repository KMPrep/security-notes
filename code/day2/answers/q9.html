# Q9 - Count Unique Substrings of Length K
# Given a string and length k, return how many distinct substrings of length k exist.
#
# Approach:
#   - Naive:          O(n*k) — extract substrings into a list, deduplicate with a dict.
#   - No built-ins:   O(n)   — dict as a manual set (no set() builtin).
#   - With built-ins: O(n)   — set comprehension (sliding window of slices).
#
# Edge cases:
#   - k > len(s)      -> 0
#   - Empty string    -> 0
#   - k == len(s)     -> 1 (the string itself is the only substring)
#   - All same chars  -> 1  e.g. "aaaa", k=2 -> only "aa"
#   - None input      -> TypeError
#   - k <= 0          -> ValueError


# === APPROACH 1: NAIVE — O(n*k) ===
# "Extract substrings into a list, deduplicate manually — O(n*k) due to dict key hashing."

def count_unique_substrings_naive(s, k):
    if s is None:
        raise TypeError("Input cannot be None")
    if not isinstance(s, str):
        raise TypeError(f"Expected string, got {type(s).__name__}")
    if not isinstance(k, int) or k <= 0:
        raise ValueError(f"k must be a positive integer, got {k!r}")
    if k > len(s):
        return 0

    substrings = []
    for i in range(len(s) - k + 1):
        substrings.append(s[i:i + k])

    seen = {}
    for sub in substrings:
        seen[sub] = True

    count = 0
    for _ in seen:
        count += 1
    return count


# === APPROACH 2: DICT AS SET — NO BUILT-INS — O(n) ===
# "Single pass using dict keys. No set() builtin."

def count_unique_substrings_no_builtins(s, k):
    if s is None:
        raise TypeError("Input cannot be None")
    if not isinstance(s, str):
        raise TypeError(f"Expected string, got {type(s).__name__}")
    if not isinstance(k, int) or k <= 0:
        raise ValueError(f"k must be a positive integer, got {k!r}")
    if k > len(s):
        return 0

    seen = {}
    for i in range(len(s) - k + 1):
        sub = s[i:i + k]
        seen[sub] = 1

    count = 0
    for _ in seen:
        count += 1
    return count


# === APPROACH 3: WITH BUILT-INS — O(n) ===
# "Set comprehension: extract and deduplicate in one expression."

def count_unique_substrings(s, k):
    if s is None:
        raise TypeError("Input cannot be None")
    if not isinstance(s, str):
        raise TypeError(f"Expected string, got {type(s).__name__}")
    if not isinstance(k, int) or k <= 0:
        raise ValueError(f"k must be a positive integer, got {k!r}")
    if k > len(s):
        return 0

    return len({s[i:i + k] for i in range(len(s) - k + 1)})


# --- tests ---
assert count_unique_substrings_naive("abcabc", 3) == 3     # "abc", "bca", "cab"
assert count_unique_substrings_no_builtins("abcabc", 3) == 3
assert count_unique_substrings("abcabc", 3) == 3

assert count_unique_substrings("aaaa", 2) == 1             # only "aa"
assert count_unique_substrings("abc", 5) == 0              # k > len
assert count_unique_substrings("", 3) == 0
assert count_unique_substrings("abcd", 4) == 1             # k == len

print(count_unique_substrings("abcabc", 3))  # 3
print(count_unique_substrings("aaaa", 2))    # 1
print(count_unique_substrings("abc", 5))     # 0
