Q1. Find All Duplicate IPs

The setup:

"Given a list of IP addresses from a firewall log, return all IPs that appear more than once."

Step 1 — Naive (O(n²)) — "Let me start with brute force"

def find_duplicate_ips_naive(ip_list):
    """
    Naive: For each IP, scan the rest of the list to check if it appears again.
    Time: O(n^2)  Space: O(n) for result
    In practice: 'I know this is slow — I'm going to improve it next.'
    """
    if ip_list is None:
        raise TypeError("Input cannot be None")
    if not isinstance(ip_list, list):
        raise TypeError(f"Expected list, got {type(ip_list).__name__}")

    duplicates = []
    for i in range(len(ip_list)):
        if not isinstance(ip_list[i], str):
            raise ValueError(f"Expected string at index {i}, got {type(ip_list[i]).__name__}")
        for j in range(i + 1, len(ip_list)):
            if ip_list[i] == ip_list[j] and ip_list[i] not in duplicates:
                duplicates.append(ip_list[i])
    return duplicates


# Quick test
print(find_duplicate_ips_naive(["10.0.0.1", "192.168.1.1", "10.0.0.1", "10.0.0.2", "192.168.1.1"]))
# ["10.0.0.1", "192.168.1.1"]

Step 2 — Optimized (O(n)) — Without Built-ins

def find_duplicate_ips_no_builtins(ip_list):
    """
    Optimized: single-pass counting with a manual dict.
    We deliberately avoid Counter() to show we understand the mechanics.
    Time: O(n)  Space: O(n)
    """
    if ip_list is None:
        raise TypeError("Input cannot be None")
    if not isinstance(ip_list, list):
        raise TypeError(f"Expected list, got {type(ip_list).__name__}")

    count = {}
    for ip in ip_list:
        if not isinstance(ip, str):
            raise ValueError(f"Non-string entry found: {ip!r}")
        if ip in count:
            count[ip] = count[ip] + 1
        else:
            count[ip] = 1

    # Second pass: collect those with count > 1
    result = []
    seen_in_result = {}  # avoid duplicates in output without using 'set()'
    for ip in ip_list:
        if count[ip] > 1 and ip not in seen_in_result:
            result.append(ip)
            seen_in_result[ip] = True

    return result


print(find_duplicate_ips_no_builtins(["10.0.0.1", "192.168.1.1", "10.0.0.1", "10.0.0.2", "192.168.1.1"]))

Step 3 — Optimized (O(n)) — With Built-ins

from collections import Counter

def find_duplicate_ips(ip_list):
    """
    Production-quality: Counter + set comprehension.
    Time: O(n)  Space: O(n)
    """
    if ip_list is None:
        raise TypeError("Input cannot be None")
    if not isinstance(ip_list, list):
        raise TypeError(f"Expected list, got {type(ip_list).__name__}")
    if not all(isinstance(ip, str) for ip in ip_list):
        raise ValueError("All entries must be strings")

    freq = Counter(ip_list)
    return [ip for ip, count in freq.items() if count > 1]

print(find_duplicate_ips(["10.0.0.1"]))               # []
print(find_duplicate_ips(["10.0.0.1", "10.0.0.1"]))  # ["10.0.0.1"]

Q2. Merge Sorted Event Timestamps

The setup:

"Two sorted lists of Unix timestamps from two log sources — merge into one sorted list without using sorted()."

Step 1 — Naive (O((n+m) log(n+m))) — "First instinct"

def merge_timestamps_naive(list_a, list_b):
    """
    Naive: concatenate and sort.
    Violates the constraint but say it first to show you understand the baseline.
    Time: O((n+m) log(n+m))  — sorting dominates
    'I know the constraint says no sorted(), let me do two-pointer next.'
    """
    if list_a is None or list_b is None:
        raise TypeError("Inputs cannot be None")
    combined = []
    for t in list_a:
        combined.append(t)
    for t in list_b:
        combined.append(t)
    # Manual insertion sort to avoid sorted() builtin
    for i in range(1, len(combined)):
        key = combined[i]
        j = i - 1
        while j >= 0 and combined[j] > key:
            combined[j + 1] = combined[j]
            j -= 1
        combined[j + 1] = key
    return combined

Step 2 — Optimized Two-Pointer — Without Built-ins

def merge_timestamps_no_builtins(list_a, list_b):
    """
    Two-pointer merge. This is the standard merge step from merge sort.
    Time: O(n + m)  Space: O(n + m)
    No built-in sort. No zip. No heapq.
    """
    if list_a is None or list_b is None:
        raise TypeError("Inputs cannot be None")
    if not isinstance(list_a, list) or not isinstance(list_b, list):
        raise TypeError("Both inputs must be lists")

    result = []
    i = 0
    j = 0

    while i < len(list_a) and j < len(list_b):
        if not isinstance(list_a[i], (int, float)):
            raise ValueError(f"Non-numeric timestamp at list_a[{i}]: {list_a[i]!r}")
        if not isinstance(list_b[j], (int, float)):
            raise ValueError(f"Non-numeric timestamp at list_b[{j}]: {list_b[j]!r}")

        if list_a[i] <= list_b[j]:
            result.append(list_a[i])
            i += 1
        else:
            result.append(list_b[j])
            j += 1

    # Drain remaining elements
    while i < len(list_a):
        result.append(list_a[i])
        i += 1
    while j < len(list_b):
        result.append(list_b[j])
        j += 1

    return result


print(merge_timestamps_no_builtins([1000, 1005, 1010], [1002, 1007, 1020]))
# [1000, 1002, 1005, 1007, 1010, 1020]

Step 3 — With Built-ins (heapq for generality)

import heapq

def merge_timestamps(list_a, list_b):
    """
    Uses heapq.merge — O(n + m), works on any number of sorted iterables.
    Production preference when merging many sources.
    """
    if list_a is None or list_b is None:
        raise TypeError("Inputs cannot be None")
    if not isinstance(list_a, list) or not isinstance(list_b, list):
        raise TypeError("Both inputs must be lists")

    return list(heapq.merge(list_a, list_b))


# Edge cases
print(merge_timestamps([], [1, 2, 3]))        # [1, 2, 3]
print(merge_timestamps([5], [1, 2, 3]))       # [1, 2, 3, 5]
print(merge_timestamps([], []))               # []

Complexity note: The two-pointer approach and heapq.merge are both O(n+m). For k sources, heapq.merge degrades to O(n log k) which is better than naive merge of all pairs.

Q3. Longest Consecutive Sequence of Failed Logins

The setup:

"Find the longest consecutive run of 'FAIL' in a list of login results."

Step 1 — Naive (O(n)) — First attempt is already linear, just not clean

def longest_fail_run_naive(results):
    """
    Simple counter — most people write this first.
    Time: O(n)  Space: O(1)
    This is already optimal but often written with nested logic.
    """
    if results is None:
        raise TypeError("Input cannot be None")
    if not isinstance(results, list):
        raise TypeError(f"Expected list, got {type(results).__name__}")

    max_run = 0
    current_run = 0

    for i in range(len(results)):
        entry = results[i]
        if not isinstance(entry, str):
            raise ValueError(f"Expected string at index {i}, got {type(entry).__name__}")
        if entry == "FAIL":
            current_run = current_run + 1
            if current_run > max_run:
                max_run = current_run
        else:
            current_run = 0

    return max_run


print(longest_fail_run_naive(["OK", "FAIL", "FAIL", "FAIL", "OK", "FAIL", "FAIL"]))
# 3

Step 3 — With Built-ins (itertools groupby style)

from itertools import groupby

def longest_fail_run(results):
    """
    itertools.groupby groups consecutive equal elements.
    Elegant but less transparent to a reviewer — explain it.
    Time: O(n)  Space: O(n) for intermediate groups
    """
    if results is None:
        raise TypeError("Input cannot be None")
    if not isinstance(results, list):
        raise TypeError(f"Expected list, got {type(results).__name__}")
    if not all(isinstance(r, str) for r in results):
        raise ValueError("All entries must be strings")

    max_run = 0
    for key, group in groupby(results):
        if key == "FAIL":
            run_length = sum(1 for _ in group)
            if run_length > max_run:
                max_run = run_length
    return max_run


# Edge cases
print(longest_fail_run([]))                        # 0
print(longest_fail_run(["OK", "OK", "OK"]))        # 0
print(longest_fail_run(["FAIL", "FAIL", "FAIL"]))  # 3

Security context: This directly maps to brute-force detection thresholds. If FAIL run ≥ threshold (e.g., 5), trigger account lock or CAPTCHA challenge.

Q4. Rotate Access Log Window

The setup:

"Rotate a list left by k positions."

Step 1 — Naive (O(n*k)) — "One rotation at a time"

def rotate_left_naive(logs, k):
    """
    Naive: rotate one step at a time, k times.
    Time: O(n * k)  Space: O(1) extra
    'This works but is O(n*k). I can do better.'
    """
    if logs is None:
        raise TypeError("Input cannot be None")
    if not isinstance(logs, list):
        raise TypeError(f"Expected list, got {type(logs).__name__}")
    if not isinstance(k, int):
        raise TypeError(f"k must be an integer, got {type(k).__name__}")
    if k < 0:
        raise ValueError(f"k must be non-negative, got {k}")

    n = len(logs)
    if n == 0:
        return []

    k = k % n  # handle k >= n
    result = logs[:]  # copy to avoid mutating input

    for _ in range(k):
        first = result[0]
        for i in range(1, n):
            result[i - 1] = result[i]
        result[n - 1] = first

    return result

Step 2 — Optimized O(n) — Without Slicing (Without Built-ins)

def rotate_left_no_builtins(logs, k):
    """
    Reverse-based rotation: reverse whole -> reverse first k -> reverse rest.
    Time: O(n)  Space: O(1) extra (in-place on a copy)

    Intuition: rotating left by k is equivalent to:
      1. Reverse entire array
      2. Reverse last (n-k) elements
      3. Reverse first k elements
    """
    if logs is None:
        raise TypeError("Input cannot be None")
    if not isinstance(logs, list):
        raise TypeError(f"Expected list, got {type(logs).__name__}")
    if not isinstance(k, int):
        raise TypeError(f"k must be an integer, got {type(k).__name__}")
    if k < 0:
        raise ValueError(f"k must be non-negative, got {k}")

    n = len(logs)
    if n == 0:
        return []

    k = k % n
    if k == 0:
        return logs[:]

    result = logs[:]  # work on a copy, don't mutate caller's data

    def reverse_segment(arr, left, right):
        while left < right:
            arr[left], arr[right] = arr[right], arr[left]
            left += 1
            right -= 1

    reverse_segment(result, 0, n - 1)
    reverse_segment(result, 0, n - k - 1)
    reverse_segment(result, n - k, n - 1)

    return result


print(rotate_left_no_builtins(["a", "b", "c", "d", "e"], 2))
# ["c", "d", "e", "a", "b"]

Step 3 — With Slicing (With Built-ins)

def rotate_left(logs, k):
    """
    Slicing version — most readable, acceptable in production Python.
    Time: O(n)  Space: O(n) — slicing creates new lists
    """
    if logs is None:
        raise TypeError("Input cannot be None")
    if not isinstance(logs, list):
        raise TypeError(f"Expected list, got {type(logs).__name__}")
    if not isinstance(k, int):
        raise TypeError(f"k must be an integer, got {type(k).__name__}")
    if k < 0:
        raise ValueError(f"k must be non-negative, got {k}")

    n = len(logs)
    if n == 0:
        return []

    k = k % n
    return logs[k:] + logs[:k]


# Edge cases
print(rotate_left(["a", "b", "c", "d", "e"], 0))   # ["a","b","c","d","e"]
print(rotate_left(["a", "b", "c", "d", "e"], 5))   # ["a","b","c","d","e"] (full rotation)
print(rotate_left(["a", "b", "c", "d", "e"], 7))   # ["c","d","e","a","b"] (k=7%5=2)
print(rotate_left([], 3))                           # []

Q5. Sliding Window — Peak Traffic Detection

The setup:

"Find the max total requests in any window of size k."

Step 1 — Naive (O(n*k)) — "Nested loop"

def max_window_sum_naive(requests, k):
    """
    Naive: recompute sum from scratch for each window.
    Time: O(n * k)  Space: O(1)
    'People almost always start here. Let me improve it.'
    """
    if requests is None:
        raise TypeError("Input cannot be None")
    if not isinstance(requests, list):
        raise TypeError(f"Expected list, got {type(requests).__name__}")
    if not isinstance(k, int) or k <= 0:
        raise ValueError(f"k must be a positive integer, got {k!r}")
    if len(requests) < k:
        raise ValueError(f"List length {len(requests)} is less than window size {k}")

    n = len(requests)
    max_sum = 0

    for i in range(n - k + 1):
        window_sum = 0
        for j in range(i, i + k):
            if not isinstance(requests[j], (int, float)):
                raise ValueError(f"Non-numeric value at index {j}: {requests[j]!r}")
            window_sum = window_sum + requests[j]
        if window_sum > max_sum:
            max_sum = window_sum

    return max_sum

Step 2 — Optimized Sliding Window (O(n)) — Without Built-ins

def max_window_sum_no_builtins(requests, k):
    """
    Sliding window: compute first window, then slide right by
    adding the new element and dropping the leftmost.
    Time: O(n)  Space: O(1)
    Key insight: each element is added once and removed once.
    """
    if requests is None:
        raise TypeError("Input cannot be None")
    if not isinstance(requests, list):
        raise TypeError(f"Expected list, got {type(requests).__name__}")
    if not isinstance(k, int) or k <= 0:
        raise ValueError(f"k must be a positive integer, got {k!r}")

    n = len(requests)
    if n == 0:
        return 0
    if n < k:
        raise ValueError(f"List length {n} is less than window size {k}")

    for i in range(n):
        if not isinstance(requests[i], (int, float)):
            raise ValueError(f"Non-numeric at index {i}: {requests[i]!r}")
        if requests[i] < 0:
            raise ValueError(f"Negative request count at index {i}: {requests[i]}")

    # Build first window
    window_sum = 0
    for i in range(k):
        window_sum = window_sum + requests[i]

    max_sum = window_sum

    # Slide the window
    for i in range(k, n):
        window_sum = window_sum + requests[i] - requests[i - k]
        if window_sum > max_sum:
            max_sum = window_sum

    return max_sum

def max_window_sum(requests, k):
    """
    Same sliding window, using sum() for initial window and slicing.
    Time: O(n)  Space: O(1)
    """
    if requests is None:
        raise TypeError("Input cannot be None")
    if not isinstance(requests, list):
        raise TypeError(f"Expected list, got {type(requests).__name__}")
    if not isinstance(k, int) or k <= 0:
        raise ValueError(f"k must be a positive integer, got {k!r}")

    n = len(requests)
    if n < k:
        raise ValueError(f"List length {n} is less than window size {k}")
    if not all(isinstance(x, (int, float)) for x in requests):
        raise ValueError("All request counts must be numeric")
    if any(x < 0 for x in requests):
        raise ValueError("Request counts cannot be negative")

    window_sum = sum(requests[:k])
    max_sum = window_sum

    for i in range(k, n):
        window_sum += requests[i] - requests[i - k]
        max_sum = max(max_sum, window_sum)

    return max_sum


# Edge cases
print(max_window_sum([100, 200, 150, 400, 300, 250], 3))  # 950
print(max_window_sum([5], 1))                             # 5
print(max_window_sum([1, 2, 3, 4, 5], 5))                # 15

Complexity recap for Q5:

  • Naive: O(n*k) time, O(1) space
  • Sliding window: O(n) time, O(1) space
  • Key insight: "each element enters and leaves the window exactly once"

Q6. First Non-Repeated Error Code

The setup:

"Find the first error code in a space-separated string that appears only once."

Step 1 — Naive (O(n²)) — "Count with scan each time"

def first_non_repeated_naive(error_string):
    """
    Naive: for each code, count its occurrences by rescanning.
    Time: O(n^2)  Space: O(n) for split result
    """
    if error_string is None:
        raise TypeError("Input cannot be None")
    if not isinstance(error_string, str):
        raise TypeError(f"Expected string, got {type(error_string).__name__}")

    error_string = error_string.strip()
    if not error_string:
        return None

    # Manual split without .split()
    codes = []
    current = ""
    for ch in error_string:
        if ch == " ":
            if current:
                codes.append(current)
                current = ""
        else:
            current = current + ch
    if current:
        codes.append(current)

    for i in range(len(codes)):
        count = 0
        for j in range(len(codes)):
            if codes[i] == codes[j]:
                count = count + 1
        if count == 1:
            return codes[i]

    return None  # All codes are repeated


print(first_non_repeated_naive("E404 E200 E404 E500 E200 E403"))
# "E500"

Step 2 — Optimized Two-Pass (O(n)) — Without Built-ins

def first_non_repeated_no_builtins(error_string):
    """
    Two-pass with manual dict.
    Pass 1: count frequencies.
    Pass 2: return first with count == 1.
    Time: O(n)  Space: O(n)
    """
    if error_string is None:
        raise TypeError("Input cannot be None")
    if not isinstance(error_string, str):
        raise TypeError(f"Expected string, got {type(error_string).__name__}")

    error_string = error_string.strip()
    if not error_string:
        return None

    # Manual tokenization
    codes = []
    current = ""
    for ch in error_string + " ":  # sentinel space to flush last token
        if ch == " ":
            if current:
                codes.append(current)
                current = ""
        else:
            current = current + ch

    # Count
    freq = {}
    for code in codes:
        if code in freq:
            freq[code] = freq[code] + 1
        else:
            freq[code] = 1

    # First with freq == 1 (preserve order)
    for code in codes:
        if freq[code] == 1:
            return code

    return None


print(first_non_repeated_no_builtins("E404 E200 E404 E500 E200 E403"))
# "E500"

from collections import OrderedDict

def first_non_repeated(error_string):
    """
    OrderedDict preserves insertion order and lets us do one pass.
    In Python 3.7+ regular dict also preserves insertion order.
    Time: O(n)  Space: O(n)
    """
    if error_string is None:
        raise TypeError("Input cannot be None")
    if not isinstance(error_string, str):
        raise TypeError(f"Expected string, got {type(error_string).__name__}")

    error_string = error_string.strip()
    if not error_string:
        return None

    codes = error_string.split()
    freq = {}
    for code in codes:
        freq[code] = freq.get(code, 0) + 1

    for code in codes:
        if freq[code] == 1:
            return code

    return None


# Edge cases
print(first_non_repeated(""))             # None
print(first_non_repeated("E404 E404"))   # None (all repeated)
print(first_non_repeated("E200"))        # "E200"

Q7. Valid Bracket Depth in Config String

The setup:

"Return the maximum nesting depth of brackets. Return -1 if unmatched."

Step 1 — Naive — "Linear scan with counter"

def max_bracket_depth_naive(config):
    """
    Single scan with depth counter.
    This is already O(n) — the naive version just lacks edge case handling.
    Time: O(n)  Space: O(1)
    """
    if config is None:
        raise TypeError("Input cannot be None")
    if not isinstance(config, str):
        raise TypeError(f"Expected string, got {type(config).__name__}")

    depth = 0
    max_depth = 0

    for ch in config:
        if ch == "{":
            depth = depth + 1
            if depth > max_depth:
                max_depth = depth
        elif ch == "}":
            depth = depth - 1
            if depth < 0:
                return -1  # More closing than opening

    if depth != 0:
        return -1  # Unmatched opening bracket

    return max_depth


print(max_bracket_depth_naive("{host: {ip: {range: 10}}}"))  # 3

Step 2 — Stack-Based (No Built-ins) — "Handles mixed bracket types"

In structured practice, the follow-up is always: "What about (), [], {}?"

def max_bracket_depth_no_builtins(config):
    """
    Stack-based: handles mixed bracket types {}, [], ().
    Returns max depth, or -1 on mismatch.
    Time: O(n)  Space: O(n) for stack
    """
    if config is None:
        raise TypeError("Input cannot be None")
    if not isinstance(config, str):
        raise TypeError(f"Expected string, got {type(config).__name__}")

    open_brackets = ["{", "[", "("]
    close_brackets = ["}", "]", ")"]

    # Build a matching map without dict literal — manual
    match = {}
    for i in range(len(open_brackets)):
        match[close_brackets[i]] = open_brackets[i]

    stack = []
    max_depth = 0

    for ch in config:
        if ch in open_brackets:
            stack.append(ch)
            if len(stack) > max_depth:
                max_depth = len(stack)
        elif ch in close_brackets:
            if len(stack) == 0 or stack[-1] != match[ch]:
                return -1
            stack.pop()

    if len(stack) != 0:
        return -1

    return max_depth

def normalize_path_naive(path):
    """
    Stack-based path normalization.
    This is already the optimal approach — the 'naive' is just without
    os.path.normpath or split() built-ins.
    Time: O(n)  Space: O(n)
    """
    if path is None:
        raise TypeError("Input cannot be None")
    if not isinstance(path, str):
        raise TypeError(f"Expected string, got {type(path).__name__}")
    if not path:
        return "/"

            if current:
                parts.append(current)

            pass  # current dir — skip
        elif part == "..":
            if stack:
                stack.pop()  # go up one level
            # If stack is empty and path is absolute, stay at root
        else:
            stack.append(part)

    return "/" + "/".join(stack)

        raise TypeError("Input cannot be None")

    Time: O(n^2 * m log m) where m = avg word length

                j -= 1

        groups[key].append(word)

    for key in groups:
        result.append(groups[key])
    return result

class PrefixSumChecker:
    """
    Build prefix sums once in O(n), answer any range query in O(1).
    Space: O(n) for prefix array.

    prefix[i] = sum of counts[0..i-1]
    range_sum(i, j) = prefix[j+1] - prefix[i]
    """

    def __init__(self, counts):
        if counts is None:
            raise TypeError("Input cannot be None")
        if not isinstance(counts, list):
            raise TypeError(f"Expected list, got {type(counts).__name__}")
        for idx in range(len(counts)):
            if not isinstance(counts[idx], (int, float)):
                raise ValueError(f"Non-numeric at index {idx}: {counts[idx]!r}")

        n = len(counts)
        self._prefix = [0] * (n + 1)
        for idx in range(n):
            self._prefix[idx + 1] = self._prefix[idx] + counts[idx]
        self._n = n

    def query(self, i, j):
        if not isinstance(i, int) or not isinstance(j, int):
            raise TypeError("Indices must be integers")
        if i < 0 or j >= self._n or i > j:
            raise IndexError(f"Invalid range [{i}, {j}] for length {self._n}")
        return self._prefix[j + 1] - self._prefix[i]

def longest_common_prefix_naive(hostnames):
    """
    Compare first hostname against all others, shrink prefix on mismatch.
    Time: O(n * m) where n=count, m=min length  Space: O(m)
    """
    if hostnames is None:
        raise TypeError("Input cannot be None")
    if not isinstance(hostnames, list):
        raise TypeError(f"Expected list, got {type(hostnames).__name__}")
    if not hostnames:
        return ""

    for h in hostnames:
        if not isinstance(h, str):
            raise ValueError(f"Expected string, got {type(h).__name__}")

    prefix = hostnames[0]

    for i in range(1, len(hostnames)):
        # Shrink prefix until it matches the start of hostnames[i]
        new_prefix = ""
        for j in range(min(len(prefix), len(hostnames[i]))):
            if prefix[j] == hostnames[i][j]:
                new_prefix = new_prefix + prefix[j]
            else:
                break
        prefix = new_prefix
        if not prefix:
            return ""

    return prefix

def rate_limit_no_builtins(events, window_sec=60, max_requests=5):
    """
    Per-user sliding window using a list as a queue.
    Since events are sorted by timestamp, we can maintain a
    per-user queue of recent timestamps and evict stale ones.
    Time: O(n)  Space: O(n) for queues
    """
    if events is None:
        raise TypeError("Input cannot be None")
    if not isinstance(events, list):
        raise TypeError(f"Expected list, got {type(events).__name__}")
    if not isinstance(window_sec, (int, float)) or window_sec <= 0:
        raise ValueError(f"window_sec must be positive, got {window_sec!r}")
    if not isinstance(max_requests, int) or max_requests < 1:
        raise ValueError(f"max_requests must be positive int, got {max_requests!r}")

    user_timestamps = {}  # user -> list of recent timestamps (manual queue)
    offenders = {}

    for i in range(len(events)):
        entry = events[i]
        if not isinstance(entry, (tuple, list)) or len(entry) != 2:
            raise ValueError(f"Entry {i} must be (user_id, timestamp)")

        user, ts = entry[0], entry[1]

        if not isinstance(ts, (int, float)):
            raise ValueError(f"Timestamp at index {i} must be numeric, got {type(ts).__name__}")

        if user not in user_timestamps:
            user_timestamps[user] = []

        queue = user_timestamps[user]
        queue.append(ts)

        # Evict timestamps outside the window
        cutoff = ts - window_sec
        new_queue = []
        for t in queue:
            if t >= cutoff:
                new_queue.append(t)
        user_timestamps[user] = new_queue

        if len(user_timestamps[user]) > max_requests:

from collections import defaultdict, deque

def rate_limit(events, window_sec=60, max_requests=5):
    """
    deque gives O(1) popleft vs O(n) list.pop(0).
    Time: O(n)  Space: O(n)
    """
    if events is None:
        raise TypeError("Input cannot be None")
    if not isinstance(events, list):
        raise TypeError(f"Expected list, got {type(events).__name__}")

    user_window = defaultdict(deque)
    offenders = set()

    for i, (user, ts) in enumerate(events):
        if not isinstance(ts, (int, float)):
            raise ValueError(f"Non-numeric timestamp at index {i}")

        window = user_window[user]
        window.append(ts)

        # Evict timestamps outside the rolling window
        while window and window[0] < ts - window_sec:
            window.popleft()

        if len(window) > max_requests:
            offenders.add(user)

    return sorted(offenders)

Q23. Find Symmetric Pairs in Firewall Rules

The setup:

"Find all (A, B) where both (A, B) and (B, A) exist."

Step 1 — Naive O(n²) — "Compare each pair"

def symmetric_pairs_naive(rules):
    """
    Naive: for each rule, scan for its reverse.
    Time: O(n^2)  Space: O(1) extra
    """
    if rules is None:
        raise TypeError("Input cannot be None")
    if not isinstance(rules, list):
        raise TypeError(f"Expected list, got {type(rules).__name__}")

    result = []
    for i in range(len(rules)):
        a, b = rules[i][0], rules[i][1]
        for j in range(len(rules)):
            if i != j and rules[j][0] == b and rules[j][1] == a:
                # Found symmetric — add only once per pair
                pair = (a, b) if a <= b else (b, a)
                if pair not in result:
                    result.append(pair)
    return result

Step 2 — O(n) with Set — Without Built-ins

def symmetric_pairs_no_builtins(rules):
    """
    Use a dict as a set. For each (a, b), check if (b, a) was seen.
    Time: O(n)  Space: O(n)
    """
    if rules is None:
        raise TypeError("Input cannot be None")
    if not isinstance(rules, list):
        raise TypeError(f"Expected list, got {type(rules).__name__}")

    seen = {}
    result = []
    result_set = {}

    for i in range(len(rules)):
        rule = rules[i]
        if not isinstance(rule, (tuple, list)) or len(rule) != 2:
            raise ValueError(f"Rule at index {i} must be a 2-element pair")
        a, b = rule[0], rule[1]

        reverse_key = (b, a)
        if reverse_key in seen:
            # Canonical form: smaller first to avoid (1,2) and (2,1) as separate
            canonical = (a, b) if a <= b else (b, a)
            if canonical not in result_set:
                result.append(canonical)
                result_set[canonical] = True

        seen[(a, b)] = True

    return result

def symmetric_pairs(rules):
    """
    Use frozenset({a, b}) as key — handles (a,b) and (b,a) as same pair.
    Time: O(n)  Space: O(n)
    """
    if rules is None:
        raise TypeError("Input cannot be None")
    if not isinstance(rules, list):
        raise TypeError(f"Expected list, got {type(rules).__name__}")
    if not all(isinstance(r, (tuple, list)) and len(r) == 2 for r in rules):
        raise ValueError("All rules must be 2-element tuples or lists")

    seen = set()
    found_symmetric = set()

    for a, b in rules:
        reverse = (b, a)
        if reverse in seen:
            found_symmetric.add(frozenset({a, b}))
        seen.add((a, b))

    # Convert back to sorted tuple pairs for deterministic output
    return sorted(tuple(sorted(pair)) for pair in found_symmetric)

    """

    if n < k:
        raise ValueError(f"List length {n} < window size {k}")

    distinct_count = 0

    # Build first window
    for i in range(k):
        val = alerts[i]
        if val not in freq:
            freq[val] = 0
        if freq[val] == 0:
            distinct_count += 1
        freq[val] += 1

        if freq[outgoing] == 0:
            distinct_count -= 1

        results.append(distinct_count)

    return results

The setup:

"Find the smallest subarray containing at least one each of INFO, WARN, ERROR."

Step 1 — Naive O(n²) — "Check every subarray"

def min_window_naive(logs, required=None):
    """
    Naive: check every subarray to see if it contains all required severities.
    Time: O(n^2)  Space: O(n)
    """
    if required is None:
        required = {"INFO", "WARN", "ERROR"}
    if logs is None:
        raise TypeError("Input cannot be None")
    if not isinstance(logs, list):
        raise TypeError(f"Expected list, got {type(logs).__name__}")
    if not required:
        raise ValueError("Required set cannot be empty")

    n = len(logs)
    min_len = n + 1
    result = None

    for i in range(n):
        present = {}
        for j in range(i, n):
            if logs[j] in required:
                present[logs[j]] = True
            # Check if all required are present
            all_present = True
            for r in required:
                if r not in present:
                    all_present = False
                    break
            if all_present:
                window_len = j - i + 1
                if window_len < min_len:
                    min_len = window_len
                    result = logs[i:j + 1]
                break  # No need to extend further for this start

    return result

Step 2 — Sliding Window O(n) — Without Built-ins

def min_window_no_builtins(logs, required=None):
    """
    Sliding window:
    - Expand right until all required severities are covered.
    - Shrink left while still covered.
    - Track minimum window seen.
    Time: O(n)  Space: O(required_count + alphabet_size)
    """
    if required is None:
        required = {"INFO", "WARN", "ERROR"}
    if logs is None:
        raise TypeError("Input cannot be None")
    if not isinstance(logs, list):
        raise TypeError(f"Expected list, got {type(logs).__name__}")

    n = len(logs)
    if n == 0:
        return None

    # Count how many distinct required types we need
    required_count = 0
    required_map = {}
    for r in required:
        required_map[r] = True
        required_count += 1

    window_freq = {}   # frequency of required types in current window
    formed = 0         # how many required types are covered (count >= 1)
    left = 0
    min_start = 0
    min_len = n + 1

    for right in range(n):
        ch = logs[right]
        if ch in required_map:
            if ch not in window_freq:
                window_freq[ch] = 0
            window_freq[ch] += 1
            if window_freq[ch] == 1:
                formed += 1

        # Shrink from left while window still covers all required
        while formed == required_count and left <= right:
            window_len = right - left + 1
            if window_len < min_len:
                min_len = window_len
                min_start = left

            leaving = logs[left]
            left += 1
            if leaving in required_map:
                window_freq[leaving] -= 1
                if window_freq[leaving] == 0:
                    formed -= 1

    if min_len == n + 1:
        return None  # No valid window found

    result = []
    for k in range(min_start, min_start + min_len):
        result.append(logs[k])
    return result

from collections import Counter

def min_window(logs, required=None):
    """
    Minimum window substring — classic sliding window pattern.
    Time: O(n)  Space: O(required + n)
    """
    if required is None:
        required = {"INFO", "WARN", "ERROR"}
    if logs is None:
        raise TypeError("Input cannot be None")
    if not isinstance(logs, list):
        raise TypeError(f"Expected list, got {type(logs).__name__}")
    if not required:
        raise ValueError("Required set cannot be empty")

    n = len(logs)
    if n == 0:
        return None

    need = {r: 1 for r in required}  # each must appear at least once
    have = Counter()
    formed = 0
    required_count = len(required)
    left = 0
    best = None

    for right, token in enumerate(logs):
        if token in need:
            have[token] += 1
            if have[token] == need[token]:
                formed += 1

        while formed == required_count:
            # Update best
            if best is None or (right - left + 1) < len(best):
                best = logs[left:right + 1]

            # Shrink from left
            leaving = logs[left]
            left += 1
            if leaving in need:
                have[leaving] -= 1
                if have[leaving] < need[leaving]:
                    formed -= 1

    return best
