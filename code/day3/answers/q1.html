# Q1 - Balanced Brackets in Firewall Rule Config
# Determine if all brackets ({}, [], ()) in a config string are balanced.
#
# Approach:
#   - Naive:          O(n) — counter-based, curly braces only.
#   - No built-ins:   O(n) — explicit stack using a list; manual open/close dict.
#   - With built-ins: O(n) — dict-matcher + list as stack; cleanest.
#
# Edge cases:
#   - Empty string           -> True
#   - Only non-bracket chars -> True
#   - Single bracket         -> False
#   - Mismatched close type  -> False  e.g. "(]"
#   - None input             -> TypeError


# === APPROACH 1: NAIVE — O(n), CURLY BRACES ONLY ===
# "First instinct: track open/close count. But this doesn't catch mismatches like ({)}."

def is_balanced_naive(s):
    if s is None:
        raise TypeError("Input cannot be None")
    if not isinstance(s, str):
        raise TypeError(f"Expected str, got {type(s).__name__}")

    depth = 0
    for ch in s:
        if ch == "{":
            depth += 1
        elif ch == "}":
            depth -= 1
            if depth < 0:
                return False
    return depth == 0


# === APPROACH 2: STACK — NO BUILT-INS — O(n) ===
# "Use a stack. For every close bracket, check the top of the stack matches."

def is_balanced_no_builtins(s):
    if s is None:
        raise TypeError("Input cannot be None")
    if not isinstance(s, str):
        raise TypeError(f"Expected str, got {type(s).__name__}")

    opens  = "({["
    closes = ")}]"
    match  = {")": "(", "}": "{", "]": "["}
    stack  = []

    for ch in s:
        is_open = False
        for o in opens:
            if ch == o:
                is_open = True
                break

        if is_open:
            stack.append(ch)
        else:
            is_close = False
            for c in closes:
                if ch == c:
                    is_close = True
                    break
            if is_close:
                if len(stack) == 0:
                    return False
                top = stack[len(stack) - 1]
                stack = stack[:len(stack) - 1]
                if top != match[ch]:
                    return False

    return len(stack) == 0


# === APPROACH 3: WITH BUILT-INS — O(n) ===
# "Dict lookup + list.append/pop. The standard solution."

def is_balanced(s):
    if s is None:
        raise TypeError("Input cannot be None")
    if not isinstance(s, str):
        raise TypeError(f"Expected str, got {type(s).__name__}")

    match = {")": "(", "}": "{", "]": "["}
    stack = []

    for ch in s:
        if ch in "({[":
            stack.append(ch)
        elif ch in ")}]":
            if not stack or stack[-1] != match[ch]:
                return False
            stack.pop()

    return len(stack) == 0


# --- tests ---
assert is_balanced("{allow: [tcp(80), tcp(443)]}") is True
assert is_balanced("{allow: [tcp(80)]}") is True
assert is_balanced("{allow: [tcp(80)]") is False   # missing close
assert is_balanced("") is True
assert is_balanced("(]") is False                  # mismatch type
assert is_balanced("{[()]}") is True
assert is_balanced("{[(])}") is False

assert is_balanced_no_builtins("{allow: [tcp(80), tcp(443)]}") is True
assert is_balanced_no_builtins("(]") is False

print(is_balanced("{allow: [tcp(80), tcp(443)]}"))   # True
print(is_balanced("{allow: [tcp(80)]"))              # False
print(is_balanced("(]"))                             # False
print(is_balanced(""))                               # True
