# Q10 - Generate All Subsets of Permission Flags
# Given a list of permission flags, generate all possible subsets (power set).
# 2^n subsets total.
#
# Approach:
#   - Naive/Iterative:  O(2^n * n) — start with [[]], add each flag to all existing subsets.
#   - Recursive:        O(2^n * n) — choose/don't-choose pattern; classic backtracking.
#   - With built-ins:   O(2^n * n) — itertools.combinations for each size 0..n.
#
# Choose/don't-choose explanation (key concept):
#   At each step, either include the current item or skip it.
#   This binary decision tree produces exactly 2^n leaves.
#
# Edge cases:
#   - Empty list   -> [[]]  (one subset: the empty set)
#   - n=1          -> [[], [item]]
#   - Duplicates   -> treated as distinct positions (not deduplicated)
#   - None input   -> TypeError


# === APPROACH 1: ITERATIVE EXPANSION — O(2^n * n) ===
# "Start with [[]], for each element extend every existing subset."

def all_subsets_iterative(flags):
    if flags is None:
        raise TypeError("Input cannot be None")
    if not isinstance(flags, list):
        raise TypeError(f"Expected list, got {type(flags).__name__}")

    result = [[]]   # start with empty subset

    for flag in flags:
        new_subsets = []
        for existing in result:
            new_subsets.append(existing + [flag])
        result = result + new_subsets

    return result


# === APPROACH 2: RECURSIVE CHOOSE/DON'T-CHOOSE — O(2^n * n) ===
# "Classic backtracking. Show clearly at senior level."

def _subsets_helper(flags, index, current, result):
    if index == len(flags):
        result.append(list(current))
        return
    # Don't choose flags[index]
    _subsets_helper(flags, index + 1, current, result)
    # Choose flags[index]
    current.append(flags[index])
    _subsets_helper(flags, index + 1, current, result)
    current.pop()

def all_subsets_recursive(flags):
    if flags is None:
        raise TypeError("Input cannot be None")
    if not isinstance(flags, list):
        raise TypeError(f"Expected list, got {type(flags).__name__}")

    result = []
    _subsets_helper(flags, 0, [], result)
    return result


# === APPROACH 3: WITH BUILT-INS — O(2^n * n) ===
# "itertools.combinations for each size 0..n. Explicit and readable."

from itertools import combinations

def all_subsets(flags):
    if flags is None:
        raise TypeError("Input cannot be None")
    if not isinstance(flags, list):
        raise TypeError(f"Expected list, got {type(flags).__name__}")

    result = []
    for size in range(len(flags) + 1):
        for combo in combinations(flags, size):
            result.append(list(combo))
    return result


# --- tests ---
flags = ["read", "write", "exec"]

it_result  = all_subsets_iterative(flags)
rec_result = all_subsets_recursive(flags)
bi_result  = all_subsets(flags)

# All must have 2^3 = 8 subsets
assert len(it_result)  == 8
assert len(rec_result) == 8
assert len(bi_result)  == 8

# Empty subset must be in all
assert [] in it_result
assert [] in rec_result
assert [] in bi_result

# Full set must be in all
assert sorted(flags) in [sorted(s) for s in it_result]

# Empty input -> [[]]
assert all_subsets([]) == [[]]

print(f"Total subsets for {flags}: {len(all_subsets(flags))}")  # 8
for s in all_subsets(["read", "write"]):
    print(s)
