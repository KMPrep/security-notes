# Q11 - Power Set of Alert Tags (Size <= 2)
# Given a list of alert tags, return all subsets of size 0, 1, or 2.
# Practical: models alert correlation pairs in SIEM rules.
#
# Approach:
#   - Naive:          O(n^2) — generate all size-0, size-1, size-2 explicitly.
#   - Recursive:      O(n^2) — prune choose/don't-choose tree at depth > 2.
#   - With built-ins: O(n^2) — itertools.combinations for sizes 0, 1, 2.
#
# Edge cases:
#   - Empty list    -> [[]]
#   - Single tag    -> [[], [tag]]
#   - n >= 2        -> C(n,0) + C(n,1) + C(n,2) = 1 + n + n*(n-1)/2 subsets
#   - None input    -> TypeError


# === APPROACH 1: NAIVE EXPLICIT LOOPS — O(n^2) ===
# "Enumerate size-0, size-1, size-2 separately. Simple and clear."

def small_subsets_naive(tags):
    if tags is None:
        raise TypeError("Input cannot be None")
    if not isinstance(tags, list):
        raise TypeError(f"Expected list, got {type(tags).__name__}")

    result = [[]]  # size-0

    for i in range(len(tags)):
        result.append([tags[i]])  # size-1

    for i in range(len(tags)):
        for j in range(i + 1, len(tags)):
            result.append([tags[i], tags[j]])  # size-2

    return result


# === APPROACH 2: RECURSIVE WITH DEPTH PRUNING — O(n^2) ===
# "Standard backtracking; prune at depth 2."

def _small_helper(tags, index, current, result, max_size):
    result.append(list(current))
    if len(current) == max_size:
        return
    for i in range(index, len(tags)):
        current.append(tags[i])
        _small_helper(tags, i + 1, current, result, max_size)
        current.pop()

def small_subsets_recursive(tags, max_size=2):
    if tags is None:
        raise TypeError("Input cannot be None")
    if not isinstance(tags, list):
        raise TypeError(f"Expected list, got {type(tags).__name__}")
    if not isinstance(max_size, int) or max_size < 0:
        raise ValueError("max_size must be a non-negative integer")

    result = []
    _small_helper(tags, 0, [], result, max_size)
    return result


# === APPROACH 3: WITH BUILT-INS — O(n^2) ===
# "itertools.combinations for sizes 0..max_size. Clean."

from itertools import combinations

def small_subsets(tags, max_size=2):
    if tags is None:
        raise TypeError("Input cannot be None")
    if not isinstance(tags, list):
        raise TypeError(f"Expected list, got {type(tags).__name__}")
    if not isinstance(max_size, int) or max_size < 0:
        raise ValueError("max_size must be a non-negative integer")

    result = []
    for size in range(min(max_size, len(tags)) + 1):
        for combo in combinations(tags, size):
            result.append(list(combo))
    return result


# --- tests ---
tags = ["XSS", "SQLi", "RCE", "LFI"]
# Expected: C(4,0)+C(4,1)+C(4,2) = 1 + 4 + 6 = 11 subsets

n  = small_subsets_naive(tags)
r  = small_subsets_recursive(tags)
bi = small_subsets(tags)

assert len(n)  == 11
assert len(r)  == 11
assert len(bi) == 11

assert [] in bi
assert ["XSS"] in bi
assert ["XSS", "SQLi"] in bi
# No size-3 subset
assert ["XSS", "SQLi", "RCE"] not in bi

assert small_subsets([]) == [[]]
assert small_subsets(["A"]) == [[], ["A"]]

print(f"Subsets of size <= 2 for {tags}: {len(small_subsets(tags))}")  # 11
for s in small_subsets(["XSS", "SQLi", "RCE"]):
    print(s)
