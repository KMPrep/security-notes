# Q13 - Extract Timestamps from Mixed Logs
# Given a multi-line log string, extract all timestamps matching YYYY-MM-DD HH:MM:SS.
# Return them in the order they appear.
#
# Approach:
#   - Manual:         O(n) — scan char-by-char for digit sequences fitting the pattern.
#   - With re:        O(n) — compiled regex with re.findall().
#   - Strict re:      O(n) — validate actual date/time ranges in regex.
#
# Pattern: YYYY-MM-DD HH:MM:SS
#   YYYY: 4 digits
#   MM:   01-12
#   DD:   01-31
#   HH:   00-23
#   MM:   00-59
#   SS:   00-59
#
# Edge cases:
#   - No timestamps           -> []
#   - Multiple on same line   -> all extracted
#   - None input              -> TypeError

import re


# === APPROACH 1: SIMPLE REGEX — O(n) ===
# "Captures the format; doesn't validate ranges. Say this and then improve."

_TS_SIMPLE = re.compile(r'\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}')

def extract_timestamps_simple(log):
    if log is None:
        raise TypeError("Input cannot be None")
    if not isinstance(log, str):
        raise TypeError(f"Expected str, got {type(log).__name__}")
    return _TS_SIMPLE.findall(log)


# === APPROACH 2: STRICT RANGE-VALIDATING REGEX — O(n) ===
# "Encode valid ranges. Catches 2023-99-01 as invalid."

_TS_STRICT = re.compile(
    r'\b'
    r'(\d{4})'                  # year
    r'-(0[1-9]|1[0-2])'        # month 01-12
    r'-(0[1-9]|[12]\d|3[01])'  # day   01-31
    r' '
    r'([01]\d|2[0-3])'         # hour  00-23
    r':([0-5]\d)'               # min   00-59
    r':([0-5]\d)'               # sec   00-59
    r'\b'
)

def extract_timestamps(log):
    if log is None:
        raise TypeError("Input cannot be None")
    if not isinstance(log, str):
        raise TypeError(f"Expected str, got {type(log).__name__}")
    return [m.group(0) for m in _TS_STRICT.finditer(log)]


# === APPROACH 3: WITH GROUPS — PARSED RESULT — O(n) ===
# "Return structured dicts. Useful when caller needs to compare or sort timestamps."

def extract_timestamps_parsed(log):
    if log is None:
        raise TypeError("Input cannot be None")
    if not isinstance(log, str):
        raise TypeError(f"Expected str, got {type(log).__name__}")

    results = []
    for m in _TS_STRICT.finditer(log):
        results.append({
            "raw":    m.group(0),
            "year":   m.group(1),
            "month":  m.group(2),
            "day":    m.group(3),
            "hour":   m.group(4),
            "minute": m.group(5),
            "second": m.group(6),
        })
    return results


# --- tests ---
log = """
INFO 2023-10-01 08:00:00 Server started
WARN 2023-10-01 08:05:32 High CPU
ERROR 2023-10-02 23:59:59 Disk full
bad timestamp 2023-99-01 00:00:00 (invalid month)
another bad 2023-10-32 00:00:00 (invalid day)
"""

expected = ["2023-10-01 08:00:00", "2023-10-01 08:05:32", "2023-10-02 23:59:59"]

assert extract_timestamps_simple(log) == expected + ["2023-99-01 00:00:00", "2023-10-32 00:00:00"]
assert extract_timestamps(log) == expected          # strict: rejects bad ranges
assert [r["raw"] for r in extract_timestamps_parsed(log)] == expected

assert extract_timestamps("no timestamps here") == []

print(extract_timestamps(log))
# ['2023-10-01 08:00:00', '2023-10-01 08:05:32', '2023-10-02 23:59:59']
