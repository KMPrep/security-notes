# Q14 - Validate Email-Like Strings
# Validate strings of the form user@domain.tld where:
#   user:   [a-zA-Z0-9._-]+
#   domain: [a-zA-Z0-9-]+
#   tld:    [a-zA-Z]{2,6}
# Return True/False for each input.
#
# Approach:
#   - Manual:     O(n) — scan char-by-char; check each section.
#   - With re:    O(n) — re.fullmatch() with compiled pattern.
#   - Strict re:  O(n) — additional constraints (no leading/trailing dots in user, etc.)
#
# Important: This is NOT RFC 5321 compliance — it's a simplified rule for this context.
#            Say that upfront.
#
# Edge cases:
#   - No @                   -> False
#   - Multiple @             -> False
#   - Empty user/domain/tld  -> False
#   - TLD too short/long     -> False (< 2 or > 6 alpha chars)
#   - None input             -> TypeError

import re


# === APPROACH 1: MANUAL — O(n) ===
# "Split on @, validate each part. Verbose but crystal clear."

_VALID_USER_CHARS   = set("abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789._-")
_VALID_DOMAIN_CHARS = set("abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789-")
_VALID_TLD_CHARS    = set("abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ")

def is_valid_email_manual(addr):
    if addr is None:
        raise TypeError("Input cannot be None")
    if not isinstance(addr, str):
        raise TypeError(f"Expected str, got {type(addr).__name__}")

    if addr.count("@") != 1:
        return False

    user, rest = addr.split("@")
    if not user:
        return False
    for ch in user:
        if ch not in _VALID_USER_CHARS:
            return False

    dot_idx = rest.rfind(".")
    if dot_idx <= 0:   # no dot, or dot at start
        return False

    domain = rest[:dot_idx]
    tld    = rest[dot_idx + 1:]

    if not domain:
        return False
    for ch in domain:
        if ch not in _VALID_DOMAIN_CHARS:
            return False

    if not (2 <= len(tld) <= 6):
        return False
    for ch in tld:
        if ch not in _VALID_TLD_CHARS:
            return False

    return True


# === APPROACH 2: COMPILED REGEX — O(n) ===
# "re.fullmatch anchors both ends. Cleaner for clarity."

_EMAIL_RE = re.compile(
    r'^[a-zA-Z0-9._-]+'    # user
    r'@'
    r'[a-zA-Z0-9-]+'       # domain
    r'\.'
    r'[a-zA-Z]{2,6}$'      # tld
)

def is_valid_email(addr):
    if addr is None:
        raise TypeError("Input cannot be None")
    if not isinstance(addr, str):
        raise TypeError(f"Expected str, got {type(addr).__name__}")
    return bool(_EMAIL_RE.fullmatch(addr))


# === APPROACH 3: BATCH VALIDATION — O(m * n) ===
# "Validate a list; return list of bools. Useful in practice."

def validate_emails(addresses):
    if addresses is None:
        raise TypeError("Input cannot be None")
    if not isinstance(addresses, list):
        raise TypeError(f"Expected list, got {type(addresses).__name__}")
    return [is_valid_email(addr) for addr in addresses]


# --- tests ---
valid_cases = [
    "user@example.com",
    "first.last@sub-domain.org",
    "user_123@domain.io",
    "a@b.co",
]
invalid_cases = [
    "notanemail",
    "@domain.com",
    "user@",
    "user@domain",
    "user@domain.toolongtld",
    "user@@domain.com",
    "user@dom ain.com",  # space
]

for addr in valid_cases:
    assert is_valid_email_manual(addr) is True,  f"Manual failed for valid: {addr}"
    assert is_valid_email(addr) is True,         f"Regex failed for valid: {addr}"

for addr in invalid_cases:
    assert is_valid_email_manual(addr) is False, f"Manual failed for invalid: {addr}"
    assert is_valid_email(addr) is False,        f"Regex failed for invalid: {addr}"

results = validate_emails(valid_cases + invalid_cases)
print("Valid:", results[:len(valid_cases)])    # all True
print("Invalid:", results[len(valid_cases):]) # all False
