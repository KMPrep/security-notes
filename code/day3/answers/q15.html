# Q15 - Detect Suspicious URL Patterns
# Flag URLs containing:
#   - SQL injection:    ' OR, %27, --
#   - Path traversal:  ../, ..%2F
#   - Double encoding: %25
# Return list of (url, reason) tuples for flagged URLs.
#
# Approach:
#   - Naive:        O(m * n) — check each URL against each pattern with 'in'.
#   - With re:      O(m * n) — compiled patterns per rule; findall for evidence.
#   - Combined re:  O(m)     — single alternation regex; group name gives reason.
#
# Edge cases:
#   - URL with multiple issues -> one entry per distinct matched reason
#   - Empty list               -> []
#   - Non-string entries       -> ValueError
#   - None input               -> TypeError
#   - Case-insensitive SQL kw  -> use re.IGNORECASE

import re


# === APPROACH 1: NAIVE STRING SEARCH — O(m * n) ===
# "Check each pattern with 'in'. Fast to write; misses variants."

_NAIVE_PATTERNS = [
    ("sql_injection",  ["' OR", "%27", "--"]),
    ("path_traversal", ["../", "..%2F", "..%2f"]),
    ("double_encoding",["%%25", "%25"]),
]

def flag_urls_naive(urls):
    if urls is None:
        raise TypeError("Input cannot be None")
    if not isinstance(urls, list):
        raise TypeError(f"Expected list, got {type(urls).__name__}")

    flagged = []
    for i, url in enumerate(urls):
        if not isinstance(url, str):
            raise ValueError(f"Non-string URL at index {i}: {url!r}")
        reasons = []
        url_upper = url.upper()
        for reason, patterns in _NAIVE_PATTERNS:
            for pat in patterns:
                if pat.upper() in url_upper:
                    if reason not in reasons:
                        reasons.append(reason)
                    break
        if reasons:
            for r in reasons:
                flagged.append((url, r))
    return flagged


# === APPROACH 2: COMPILED REGEX PER RULE — O(m * n) ===
# "Compile once; run many. Use re.IGNORECASE for case variants."

_RULES = [
    ("sql_injection",   re.compile(r"(%27|'|\bOR\b|--)", re.IGNORECASE)),
    ("path_traversal",  re.compile(r"\.\./|\.\.%2[Ff]",  re.IGNORECASE)),
    ("double_encoding", re.compile(r"%25",                re.IGNORECASE)),
]

def flag_urls(urls):
    if urls is None:
        raise TypeError("Input cannot be None")
    if not isinstance(urls, list):
        raise TypeError(f"Expected list, got {type(urls).__name__}")

    flagged = []
    for i, url in enumerate(urls):
        if not isinstance(url, str):
            raise ValueError(f"Non-string URL at index {i}: {url!r}")
        for reason, pattern in _RULES:
            if pattern.search(url):
                flagged.append((url, reason))
    return flagged


# === APPROACH 3: NAMED-GROUP COMBINED REGEX — O(m) PER URL ===
# "One regex with named groups. Single pass per URL."

_COMBINED = re.compile(
    r"(?P<sql_injection>%27|'|--|\bOR\b)"
    r"|(?P<path_traversal>\.\./|\.\.%2[Ff])"
    r"|(?P<double_encoding>%25)",
    re.IGNORECASE
)

def flag_urls_combined(urls):
    if urls is None:
        raise TypeError("Input cannot be None")
    if not isinstance(urls, list):
        raise TypeError(f"Expected list, got {type(urls).__name__}")

    flagged = []
    for i, url in enumerate(urls):
        if not isinstance(url, str):
            raise ValueError(f"Non-string URL at index {i}: {url!r}")
        seen_reasons = set()
        for m in _COMBINED.finditer(url):
            for group_name, val in m.groupdict().items():
                if val is not None and group_name not in seen_reasons:
                    seen_reasons.add(group_name)
                    flagged.append((url, group_name))
    return flagged


# --- tests ---
urls = [
    "https://example.com/search?q=hello",
    "https://example.com/page?id=1' OR '1'='1",
    "https://example.com/file?path=../../etc/passwd",
    "https://example.com/data?id=%27union%20select",
    "https://example.com/x?a=%2527",   # double-encoded %
    "https://example.com/ok",
]

for fn in [flag_urls, flag_urls_combined]:
    result = fn(urls)
    result_urls = [r[0] for r in result]
    assert "https://example.com/search?q=hello" not in result_urls
    assert "https://example.com/ok" not in result_urls

    # SQL injection detected
    assert any("sql" in r[1] for r in result if "OR" in r[0] or "%27" in r[0])

    # Path traversal detected
    assert any("path_traversal" in r[1] for r in result if "../" in r[0])

print(flag_urls(urls))
