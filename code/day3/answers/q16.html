# Q16 - Extract HTTP Log Fields (Combined Log Format)
# Parse a line in Combined Log Format and extract: IP, user, timestamp,
# method, path, status code, bytes.
#
# Format:
#   IP - user [timestamp] "METHOD path HTTP/ver" status bytes
#
# Approach:
#   - Manual:     O(n) — split and index; fragile with spaces in path.
#   - With re:    O(n) — single compiled regex with named capture groups.
#   - Structured: O(n) — return parsed dict with type coercion.
#
# Edge cases:
#   - User is "-"        -> include as-is (anonymous)
#   - Bytes is "-"       -> treat as 0 or None
#   - Malformed line     -> raises ValueError
#   - None input         -> TypeError

import re


# === APPROACH 1: NAIVE SPLIT — O(n) ===
# "Split on space, then pull fields by index. Breaks if path has spaces."

def parse_log_naive(line):
    if line is None:
        raise TypeError("Input cannot be None")
    if not isinstance(line, str):
        raise TypeError(f"Expected str, got {type(line).__name__}")

    parts = line.strip().split()
    if len(parts) < 9:
        raise ValueError(f"Too few fields in log line: {line!r}")

    ip        = parts[0]
    user      = parts[2]
    timestamp = parts[3].lstrip("[")   # crude; doesn't capture timezone
    method    = parts[5].lstrip('"')
    path      = parts[6]
    status    = parts[8]

    return {"ip": ip, "user": user, "timestamp": timestamp,
            "method": method, "path": path, "status": status}


# === APPROACH 2: SINGLE COMPILED REGEX — O(n) ===
# "Named groups. Robust against spaces in path because we match quoted string."

_CLF = re.compile(
    r'^(?P<ip>\S+)'                            # IP
    r'\s+-\s+'                                 # ident (always -)
    r'(?P<user>\S+)'                           # user
    r'\s+\[(?P<timestamp>[^\]]+)\]'            # [timestamp]
    r'\s+"(?P<method>\S+)'                     # "METHOD
    r'\s+(?P<path>\S+)'                        # path
    r'\s+\S+"'                                 # HTTP/ver"
    r'\s+(?P<status>\d{3})'                    # status code
    r'\s+(?P<bytes>\S+)'                       # bytes (may be -)
)

def parse_log(line):
    if line is None:
        raise TypeError("Input cannot be None")
    if not isinstance(line, str):
        raise TypeError(f"Expected str, got {type(line).__name__}")

    m = _CLF.match(line.strip())
    if not m:
        raise ValueError(f"Line does not match Combined Log Format: {line!r}")
    return m.groupdict()


# === APPROACH 3: STRUCTURED WITH TYPE COERCION — O(n) ===
# "Return typed dict: status as int, bytes as int or None."

def parse_log_typed(line):
    if line is None:
        raise TypeError("Input cannot be None")
    if not isinstance(line, str):
        raise TypeError(f"Expected str, got {type(line).__name__}")

    raw = parse_log(line)
    raw["status"] = int(raw["status"])
    raw["bytes"]  = None if raw["bytes"] == "-" else int(raw["bytes"])
    return raw


# --- tests ---
sample = '192.168.1.1 - frank [10/Oct/2023:13:55:36 -0700] "GET /index.html HTTP/1.1" 200 2326'

r = parse_log(sample)
assert r["ip"]        == "192.168.1.1"
assert r["user"]      == "frank"
assert r["timestamp"] == "10/Oct/2023:13:55:36 -0700"
assert r["method"]    == "GET"
assert r["path"]      == "/index.html"
assert r["status"]    == "200"
assert r["bytes"]     == "2326"

rt = parse_log_typed(sample)
assert rt["status"] == 200
assert rt["bytes"]  == 2326

# Anonymous user, no bytes
anon = '10.0.0.1 - - [01/Jan/2024:00:00:00 +0000] "POST /login HTTP/1.1" 401 -'
r2   = parse_log_typed(anon)
assert r2["user"]   == "-"
assert r2["bytes"]  is None
assert r2["status"] == 401

print(parse_log_typed(sample))
print(parse_log_typed(anon))
