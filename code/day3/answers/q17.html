# Q17 - Match Base64-Looking Strings
# In a log blob, identify tokens that look like Base64-encoded data:
#   - Chars: [A-Za-z0-9+/=]
#   - Length >= 20 characters
# Flag them for further analysis. Return list of matched tokens.
#
# Approach:
#   - Naive:    O(n * m) — split on whitespace; check each token manually.
#   - With re:  O(n)     — compiled regex; findall for all matches.
#   - Strict:   O(n)     — validate padding (= only at end; length divisible by 4).
#
# Notes:
#   - Base64 strings in logs are often suspicious (encoded payloads, exfil).
#   - True Base64 has length divisible by 4 (with padding). Mention this.
#   - Without word boundaries, partial matches may occur — use \b or whitespace split.
#
# Edge cases:
#   - Short Base64 (< 20 chars)   -> excluded
#   - URL-safe Base64 (-_)        -> not matched (standard variant only)
#   - Padding = chars mid-string  -> invalid Base64 but included in naive version
#   - None input                  -> TypeError

import re


_B64_CHARS = set("ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/=")


# === APPROACH 1: MANUAL TOKEN CHECK — O(n * m) ===
# "Split on whitespace/common delimiters, validate char-by-char."

def find_base64_manual(log):
    if log is None:
        raise TypeError("Input cannot be None")
    if not isinstance(log, str):
        raise TypeError(f"Expected str, got {type(log).__name__}")

    import re as _re
    tokens  = _re.split(r'[\s,;|"\']', log)
    results = []

    for tok in tokens:
        if len(tok) < 20:
            continue
        valid = True
        for ch in tok:
            if ch not in _B64_CHARS:
                valid = False
                break
        if valid:
            results.append(tok)

    return results


# === APPROACH 2: COMPILED REGEX — O(n) ===
# "Find all substrings of length >= 20 using only Base64 chars."

_B64_RE = re.compile(r'[A-Za-z0-9+/=]{20,}')

def find_base64(log):
    if log is None:
        raise TypeError("Input cannot be None")
    if not isinstance(log, str):
        raise TypeError(f"Expected str, got {type(log).__name__}")
    return _B64_RE.findall(log)


# === APPROACH 3: STRICT PADDING VALIDATION — O(n) ===
# "Valid Base64: total length divisible by 4; = only at end (0, 1, or 2)."

_B64_STRICT_RE = re.compile(r'[A-Za-z0-9+/]{20,}={0,2}')

def find_base64_strict(log):
    if log is None:
        raise TypeError("Input cannot be None")
    if not isinstance(log, str):
        raise TypeError(f"Expected str, got {type(log).__name__}")

    results = []
    for m in _B64_STRICT_RE.finditer(log):
        token = m.group(0)
        # Padding check: length must be divisible by 4
        if len(token) % 4 == 0:
            results.append(token)
    return results


# --- tests ---
log = """
Normal log entry at 2023-10-01
Encoded payload: SGVsbG8gV29ybGQhISEhISEhISEhISEhISE=
Short b64: SGVs
Another suspicious: dGhpcyBpcyBhIHRlc3Q=
IP: 192.168.1.1
"""

# SGVsbG8gV29ybGQhISEhISEhISEhISEhISE= -> length > 20, all base64 chars
results = find_base64(log)
assert any("SGVsbG8" in r for r in results)
assert all(len(r) >= 20 for r in results)
assert "SGVs" not in results  # too short

# Manual agrees
results_manual = find_base64_manual(log)
assert any("SGVsbG8" in r for r in results_manual)

print(find_base64(log))
print(find_base64_strict(log))
