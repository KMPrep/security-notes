# Q19 - Policy String Parser
# Parse a policy string like "DENY ip=10.0.0.1 port=443 proto=TCP" into a dict:
#   {"action": "DENY", "ip": "10.0.0.1", "port": "443", "proto": "TCP"}
#
# Approach:
#   - Naive/split:  O(n) — split on whitespace; first token is action; rest are key=value.
#   - With re:      O(n) — regex to extract key=value pairs; first word for action.
#   - Strict:       O(n) — validate action is in allowed set; validate key names.
#
# Edge cases:
#   - Extra whitespace         -> strip/split handles it
#   - No key=value pairs       -> {"action": action}
#   - Duplicate keys           -> last wins (or raise, depending on spec)
#   - Malformed key=value      -> raises ValueError
#   - None input               -> TypeError

import re

_VALID_ACTIONS = {"ALLOW", "DENY", "LOG", "DROP", "ACCEPT"}


# === APPROACH 1: SPLIT-BASED — O(n) ===
# "Simple split. Doesn't validate keys. Good starting point."

def parse_policy_naive(policy_str):
    if policy_str is None:
        raise TypeError("Input cannot be None")
    if not isinstance(policy_str, str):
        raise TypeError(f"Expected str, got {type(policy_str).__name__}")

    parts = policy_str.strip().split()
    if not parts:
        raise ValueError("Empty policy string")

    result = {"action": parts[0]}
    for part in parts[1:]:
        if "=" not in part:
            raise ValueError(f"Expected key=value, got: {part!r}")
        key, _, value = part.partition("=")
        result[key] = value
    return result


# === APPROACH 2: REGEX KEY=VALUE EXTRACTION — O(n) ===
# "Named groups or findall for k=v pairs. Handles quoted values too."

_KV_RE = re.compile(r'(\w+)=([^\s]+)')

def parse_policy(policy_str):
    if policy_str is None:
        raise TypeError("Input cannot be None")
    if not isinstance(policy_str, str):
        raise TypeError(f"Expected str, got {type(policy_str).__name__}")

    stripped = policy_str.strip()
    if not stripped:
        raise ValueError("Empty policy string")

    # First token is the action
    first_space = stripped.find(" ")
    if first_space == -1:
        return {"action": stripped}

    action = stripped[:first_space]
    rest   = stripped[first_space:]

    result = {"action": action}
    for key, value in _KV_RE.findall(rest):
        result[key] = value
    return result


# === APPROACH 3: STRICT VALIDATED PARSER — O(n) ===
# "Validate action in allowed set. Mention this in security context."

def parse_policy_strict(policy_str, valid_actions=None):
    if policy_str is None:
        raise TypeError("Input cannot be None")
    if not isinstance(policy_str, str):
        raise TypeError(f"Expected str, got {type(policy_str).__name__}")
    if valid_actions is None:
        valid_actions = _VALID_ACTIONS

    result = parse_policy(policy_str)

    if result["action"].upper() not in valid_actions:
        raise ValueError(f"Unknown action '{result['action']}'. "
                         f"Expected one of: {sorted(valid_actions)}")
    result["action"] = result["action"].upper()
    return result


# --- tests ---
s = "DENY ip=10.0.0.1 port=443 proto=TCP"

r1 = parse_policy_naive(s)
r2 = parse_policy(s)
r3 = parse_policy_strict(s)

for r in [r1, r2, r3]:
    assert r["action"] == "DENY"
    assert r["ip"]     == "10.0.0.1"
    assert r["port"]   == "443"
    assert r["proto"]  == "TCP"

# Action only
assert parse_policy("ALLOW") == {"action": "ALLOW"}

# Invalid action
try:
    parse_policy_strict("HACK ip=1.2.3.4")
    assert False, "Should raise ValueError"
except ValueError:
    pass

print(parse_policy(s))
# {'action': 'DENY', 'ip': '10.0.0.1', 'port': '443', 'proto': 'TCP'}
