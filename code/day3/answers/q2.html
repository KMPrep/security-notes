# Q2 - Simplify Filesystem Path (Directory Traversal Detection)
# Given an absolute POSIX path possibly containing . and .., return the canonical path.
#
# Approach:
#   - Naive:          O(n) — split on "/" and process with a list; no edge-case handling.
#   - No built-ins:   O(n) — manual split by scanning chars; stack-based resolution.
#   - With built-ins: O(n) — str.split("/") + stack; idiomatic.
#
# Security context:
#   Path traversal attacks use sequences like ../../etc/passwd.
#   Canonical resolution is the first step in detecting or blocking them.
#
# Edge cases:
#   - "/"                -> "/"
#   - "/../"             -> "/"  (can't go above root)
#   - Trailing slash     -> stripped
#   - Multiple slashes   -> treated as one
#   - "."                -> current dir, ignored
#   - None input         -> TypeError


# === APPROACH 1: NAIVE — SPLIT + LIST — O(n) ===
# "Split, process each token. Good first answer; misses some edge cases at first."

def simplify_path_naive(path):
    if path is None:
        raise TypeError("Input cannot be None")
    if not isinstance(path, str):
        raise TypeError(f"Expected str, got {type(path).__name__}")

    parts = path.split("/")
    stack = []
    for part in parts:
        if part == "..":
            if stack:
                stack.pop()
        elif part and part != ".":
            stack.append(part)
    return "/" + "/".join(stack)


# === APPROACH 2: NO BUILT-INS — MANUAL SPLIT — O(n) ===
# "Parse the string char-by-char to build tokens. Shows depth when asked 'without split'."

def simplify_path_no_builtins(path):
    if path is None:
        raise TypeError("Input cannot be None")
    if not isinstance(path, str):
        raise TypeError(f"Expected str, got {type(path).__name__}")

    # Manual split on '/'
    tokens = []
    token  = ""
    for ch in path:
        if ch == "/":
            if token:
                tokens.append(token)
                token = ""
        else:
            token += ch
    if token:
        tokens.append(token)

    # Stack-based resolution
    stack = []
    for t in tokens:
        if t == "..":
            if stack:
                stack.pop()
        elif t != ".":
            stack.append(t)

    # Manual join
    if not stack:
        return "/"
    result = ""
    for part in stack:
        result += "/" + part
    return result


# === APPROACH 3: WITH BUILT-INS — O(n) ===
# "Clean, standard. Use posixpath.normpath as a bonus comparison."

def simplify_path(path):
    if path is None:
        raise TypeError("Input cannot be None")
    if not isinstance(path, str):
        raise TypeError(f"Expected str, got {type(path).__name__}")

    stack = []
    for part in path.split("/"):
        if part == "..":
            if stack:
                stack.pop()
        elif part and part != ".":
            stack.append(part)
    return "/" + "/".join(stack)


# --- tests ---
assert simplify_path("/var/log/../log/./syslog")      == "/var/log/syslog"
assert simplify_path("/etc/passwd/../../etc/shadow")  == "/etc/shadow"
assert simplify_path("/")                             == "/"
assert simplify_path("/../")                          == "/"
assert simplify_path("/a//b///c")                     == "/a/b/c"
assert simplify_path("/a/./b/./c")                    == "/a/b/c"
assert simplify_path("/a/b/c/")                       == "/a/b/c"

assert simplify_path_no_builtins("/var/log/../log/./syslog") == "/var/log/syslog"
assert simplify_path_no_builtins("/../")                    == "/"
assert simplify_path_naive("/etc/passwd/../../etc/shadow")  == "/etc/shadow"

print(simplify_path("/var/log/../log/./syslog"))      # /var/log/syslog
print(simplify_path("/etc/passwd/../../etc/shadow"))  # /etc/shadow
print(simplify_path("/../"))                          # /
print(simplify_path("/a//b///c"))                     # /a/b/c
