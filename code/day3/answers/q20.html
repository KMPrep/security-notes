# Q20 - Validate Config Nesting Depth
# Given a JSON-like config string, verify:
#   1. All brackets are balanced (using {, }, [, ])
#   2. Nesting depth never exceeds max_depth (default 5)
# Return (valid: bool, max_depth_seen: int).
#
# Approach:
#   - Naive:        O(n) — counter; no type mismatch detection.
#   - With stack:   O(n) — full balanced-bracket check + depth tracking.
#   - Parameterised: same but max_depth is a parameter.
#
# Edge cases:
#   - Empty string           -> (True, 0)
#   - Just whitespace        -> (True, 0)
#   - Exceeds depth limit    -> (False, actual_max_depth)
#   - Mismatched brackets    -> (False, max_depth_seen_so_far)
#   - None input             -> TypeError


# === APPROACH 1: COUNTER-BASED NAIVE — O(n) ===
# "Track depth with a counter. Can't detect mismatch type ({]). Say this."

def validate_config_naive(config, max_depth=5):
    if config is None:
        raise TypeError("Input cannot be None")
    if not isinstance(config, str):
        raise TypeError(f"Expected str, got {type(config).__name__}")
    if not isinstance(max_depth, int) or max_depth < 1:
        raise ValueError("max_depth must be a positive integer")

    depth     = 0
    max_seen  = 0
    depth_ok  = True

    for ch in config:
        if ch in "{[":
            depth   += 1
            max_seen = max(max_seen, depth)
            if depth > max_depth:
                depth_ok = False
        elif ch in "}]":
            depth -= 1
            if depth < 0:
                return (False, max_seen)

    valid = depth_ok and depth == 0
    return (valid, max_seen)


# === APPROACH 2: STACK-BASED — O(n) ===
# "Full mismatch detection + depth tracking. Standard approach."

def validate_config(config, max_depth=5):
    if config is None:
        raise TypeError("Input cannot be None")
    if not isinstance(config, str):
        raise TypeError(f"Expected str, got {type(config).__name__}")
    if not isinstance(max_depth, int) or max_depth < 1:
        raise ValueError("max_depth must be a positive integer")

    match    = {"}": "{", "]": "[", ")": "("}
    stack    = []
    max_seen = 0
    depth_ok = True

    for ch in config:
        if ch in "{[(":
            stack.append(ch)
            depth = len(stack)
            if depth > max_seen:
                max_seen = depth
            if depth > max_depth:
                depth_ok = False
        elif ch in "}])":
            if not stack or stack[-1] != match[ch]:
                return (False, max_seen)
            stack.pop()

    valid = depth_ok and len(stack) == 0
    return (valid, max_seen)


# === APPROACH 3: ALSO TRACK WHERE VIOLATION OCCURS — O(n) ===
# "Return position of first depth violation for debugging."

def validate_config_detailed(config, max_depth=5):
    if config is None:
        raise TypeError("Input cannot be None")
    if not isinstance(config, str):
        raise TypeError(f"Expected str, got {type(config).__name__}")
    if not isinstance(max_depth, int) or max_depth < 1:
        raise ValueError("max_depth must be a positive integer")

    match        = {"}": "{", "]": "[", ")": "("}
    stack        = []
    max_seen     = 0
    violation_at = None

    for i, ch in enumerate(config):
        if ch in "{[(":
            stack.append(ch)
            depth = len(stack)
            if depth > max_seen:
                max_seen = depth
            if depth > max_depth and violation_at is None:
                violation_at = i
        elif ch in "}])":
            if not stack or stack[-1] != match[ch]:
                return {"valid": False, "max_depth_seen": max_seen,
                        "error": f"Mismatch at position {i}: '{ch}'"}
            stack.pop()

    valid = violation_at is None and len(stack) == 0
    result = {"valid": valid, "max_depth_seen": max_seen}
    if violation_at is not None:
        result["error"] = f"Depth exceeded at position {violation_at}"
    if stack:
        result["error"] = f"Unclosed brackets: {stack}"
    return result


# --- tests ---
cfg1 = '{"a": {"b": {"c": 1}}}'        # depth 3 -> valid with default limit 5
cfg2 = '{"a": {"b": {"c": {"d": {"e": {"f": 1}}}}}}'  # depth 6 -> invalid
cfg3 = '{"a": [1, 2]}'                  # valid
cfg4 = '{"a": [1, 2}'                   # missing ]
cfg5 = '{a: [1]}'                       # depth 2
cfg6 = ''                               # empty

assert validate_config(cfg1) == (True, 3)
assert validate_config(cfg2) == (False, 6)
assert validate_config(cfg3) == (True, 2)
assert validate_config(cfg4) == (False, 2)
assert validate_config(cfg5) == (True, 2)
assert validate_config(cfg6) == (True, 0)
assert validate_config("{[(])}") == (False, 3)   # mismatch

# Custom limit
assert validate_config(cfg1, max_depth=2) == (False, 3)
assert validate_config(cfg1, max_depth=3) == (True, 3)

print(validate_config(cfg1))            # (True, 3)
print(validate_config(cfg2))            # (False, 6)
print(validate_config_detailed(cfg2))   # depth exceeded
