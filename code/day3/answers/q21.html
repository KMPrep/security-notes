# Q21 - Stack-Based Undo Log Replay
# Simulate a key-value store supporting SET and UNDO operations.
#   SET key value  -> set key to value; push old value to undo stack.
#   UNDO           -> revert the most recent SET.
# Return the final state of all keys.
#
# Approach:
#   - Naive:        O(n) — stack of (key, old_value); pop on UNDO.
#   - No built-ins: O(n) — manual stack via list slicing.
#   - With built-ins: O(n) — same with list.append/pop; clean.
#
# Key insight:
#   Each SET pushes (key, previous_value) onto the undo stack.
#   UNDO pops the stack and restores that (key, previous_value).
#   If previous_value is a sentinel (key didn't exist), delete the key.
#
# Edge cases:
#   - UNDO on empty history  -> no-op (or raise, depending on spec); we silently ignore
#   - SET same key twice     -> each SET is independently undoable
#   - UNDO past beginning    -> no-op
#   - None input             -> TypeError

_UNSET = object()   # sentinel for "key did not exist"


# === APPROACH 1: NAIVE — O(n) ===
# "Dict as store; list as undo stack. Classic."

def replay_undo_naive(ops):
    if ops is None:
        raise TypeError("Input cannot be None")
    if not isinstance(ops, list):
        raise TypeError(f"Expected list, got {type(ops).__name__}")

    store      = {}
    undo_stack = []

    for op in ops:
        if not isinstance(op, tuple):
            raise ValueError(f"Expected tuple operation, got {op!r}")

        if op[0] == "SET":
            if len(op) != 3:
                raise ValueError(f"SET requires (SET, key, value), got {op!r}")
            _, key, value = op
            old = store.get(key, _UNSET)
            undo_stack.append((key, old))
            store[key] = value

        elif op[0] == "UNDO":
            if undo_stack:
                key, old = undo_stack.pop()
                if old is _UNSET:
                    del store[key]
                else:
                    store[key] = old
        else:
            raise ValueError(f"Unknown operation: {op[0]!r}")

    return store


# === APPROACH 2: NO BUILT-INS — O(n) ===
# "Manual stack and dict access. Same logic."

def replay_undo_no_builtins(ops):
    if ops is None:
        raise TypeError("Input cannot be None")
    if not isinstance(ops, list):
        raise TypeError(f"Expected list, got {type(ops).__name__}")

    store      = {}
    undo_stack = []

    for op in ops:
        if not isinstance(op, tuple):
            raise ValueError(f"Expected tuple, got {op!r}")

        action = op[0]

        if action == "SET":
            key   = op[1]
            value = op[2]
            found = False
            for k in store:
                if k == key:
                    found = True
                    break
            old = store[key] if found else _UNSET
            undo_stack = undo_stack + [(key, old)]
            store[key] = value

        elif action == "UNDO":
            if len(undo_stack) > 0:
                key, old  = undo_stack[len(undo_stack) - 1]
                undo_stack = undo_stack[:len(undo_stack) - 1]
                if old is _UNSET:
                    del store[key]
                else:
                    store[key] = old
        else:
            raise ValueError(f"Unknown operation: {action!r}")

    return store


# === APPROACH 3: WITH BUILT-INS — O(n) ===
# "Same as naive; written cleanly."

def replay_undo(ops):
    if ops is None:
        raise TypeError("Input cannot be None")
    if not isinstance(ops, list):
        raise TypeError(f"Expected list, got {type(ops).__name__}")

    store      = {}
    undo_stack = []

    for op in ops:
        if not isinstance(op, tuple):
            raise ValueError(f"Expected tuple, got {op!r}")
        action = op[0]
        if action == "SET":
            _, key, value = op
            undo_stack.append((key, store.get(key, _UNSET)))
            store[key] = value
        elif action == "UNDO":
            if undo_stack:
                key, old = undo_stack.pop()
                if old is _UNSET:
                    store.pop(key, None)
                else:
                    store[key] = old
        else:
            raise ValueError(f"Unknown operation: {action!r}")

    return store


# --- tests ---
# 1 UNDO: reverts SET x=3  -> {x:1, y:2}
ops = [
    ("SET", "x", 1),
    ("SET", "y", 2),
    ("SET", "x", 3),
    ("UNDO",),
]

assert replay_undo_naive(ops)       == {"x": 1, "y": 2}
assert replay_undo_no_builtins(ops) == {"x": 1, "y": 2}
assert replay_undo(ops)             == {"x": 1, "y": 2}

# 2 UNDOs: also reverts SET y=2 -> {x:1}
ops_two = [
    ("SET", "x", 1),
    ("SET", "y", 2),
    ("SET", "x", 3),
    ("UNDO",),
    ("UNDO",),
]
assert replay_undo(ops_two) == {"x": 1}

# Undo past beginning -> no-op
ops2 = [("SET","a",1), ("UNDO",), ("UNDO",), ("UNDO",)]
assert replay_undo(ops2) == {}

# SET same key multiple times
ops3 = [("SET","k",1),("SET","k",2),("UNDO",)]
assert replay_undo(ops3) == {"k": 1}

print(replay_undo(ops))      # {'x': 1, 'y': 2}
print(replay_undo(ops_two))  # {'x': 1}
print(replay_undo(ops2))     # {}
