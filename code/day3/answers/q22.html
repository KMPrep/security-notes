# Q22 - Recursive Permission Inheritance
# Given a role structure {role: {"parents": [...], "perms": [...]}},
# collect all permissions for a role by recursively merging parent perms.
# Return deduplicated merged permission list (order: own first, then inherited).
#
# Approach:
#   - Naive/Recursive:  O(n * p) — recurse through parents; accumulate; dedup at end.
#   - Memoised:         O(n * p) — cache computed perm sets to avoid re-traversal.
#   - Iterative DFS:    O(n * p) — explicit stack; avoid recursion limit for deep hierarchies.
#
# Edge cases:
#   - No parents                -> own perms only
#   - Role not in structure     -> treated as no perms, no parents
#   - Cycle in role hierarchy   -> visited set prevents infinite recursion
#   - Empty structure           -> []
#   - None input                -> TypeError

from collections import OrderedDict


# === APPROACH 1: NAIVE RECURSIVE — O(n * p) ===
# "Recurse into parents, collect all perms, dedup preserving first-seen order."

def collect_perms_recursive(roles, role, _visited=None):
    if roles is None:
        raise TypeError("roles cannot be None")
    if not isinstance(roles, dict):
        raise TypeError(f"Expected dict, got {type(roles).__name__}")
    if role is None:
        raise TypeError("role cannot be None")

    if _visited is None:
        _visited = set()
    if role in _visited:
        return []
    _visited.add(role)

    entry   = roles.get(role, {})
    own     = list(entry.get("perms", []))
    parents = entry.get("parents", [])

    for parent in parents:
        own += collect_perms_recursive(roles, parent, _visited)

    # Dedup preserving order
    seen   = {}
    result = []
    for p in own:
        if p not in seen:
            seen[p] = True
            result.append(p)
    return result


# === APPROACH 2: MEMOISED RECURSIVE — O(n * p) ===
# "Cache results. Mention this if there are many roles sharing parents."

def collect_perms_memoised(roles, role, _cache=None, _visited=None):
    if roles is None:
        raise TypeError("roles cannot be None")
    if not isinstance(roles, dict):
        raise TypeError(f"Expected dict, got {type(roles).__name__}")

    if _cache   is None: _cache   = {}
    if _visited is None: _visited = set()

    if role in _cache:
        return _cache[role]
    if role in _visited:
        return []
    _visited.add(role)

    entry   = roles.get(role, {})
    own     = list(entry.get("perms", []))
    parents = entry.get("parents", [])

    for parent in parents:
        own += collect_perms_memoised(roles, parent, _cache, _visited)

    seen = {}
    result = []
    for p in own:
        if p not in seen:
            seen[p] = True
            result.append(p)

    _cache[role] = result
    return result


# === APPROACH 3: ITERATIVE DFS — O(n * p) ===
# "Explicit stack with visited set. Safe for large hierarchies."

def collect_perms(roles, role):
    if roles is None:
        raise TypeError("roles cannot be None")
    if not isinstance(roles, dict):
        raise TypeError(f"Expected dict, got {type(roles).__name__}")
    if role is None:
        raise TypeError("role cannot be None")

    stack   = [role]
    visited = set()
    all_perms = []

    while stack:
        current = stack.pop()
        if current in visited:
            continue
        visited.add(current)

        entry = roles.get(current, {})
        all_perms.extend(entry.get("perms", []))

        for parent in reversed(entry.get("parents", [])):
            if parent not in visited:
                stack.append(parent)

    seen   = {}
    result = []
    for p in all_perms:
        if p not in seen:
            seen[p] = True
            result.append(p)
    return result


# --- tests ---
roles = {
    "admin":    {"parents": ["user"],  "perms": ["delete", "write"]},
    "user":     {"parents": ["guest"], "perms": ["read"]},
    "guest":    {"parents": [],        "perms": ["view"]},
    "superuser":{"parents": ["admin"], "perms": ["sudo"]},
}

admin_perms = collect_perms(roles, "admin")
assert "delete" in admin_perms
assert "write"  in admin_perms
assert "read"   in admin_perms   # inherited from user
assert "view"   in admin_perms   # inherited from guest

super_perms = collect_perms(roles, "superuser")
assert "sudo" in super_perms
assert "delete" in super_perms

# Role not in structure
assert collect_perms(roles, "unknown") == []

# Cycle guard
cyclic = {
    "a": {"parents": ["b"], "perms": ["perm_a"]},
    "b": {"parents": ["a"], "perms": ["perm_b"]},
}
result = collect_perms(cyclic, "a")
assert "perm_a" in result
assert "perm_b" in result

print(collect_perms(roles, "superuser"))
# ['sudo', 'delete', 'write', 'read', 'view']
