# Q23 - Regex-Based Log Anomaly Tagger
# Given a list of log lines and a dict {tag: regex_pattern},
# annotate each log line with all matching tags.
# Return [(line, [tags])] — include all lines, even untagged ones (empty tag list).
#
# Approach:
#   - Naive:        O(m * n) — for each line check each pattern with re.search().
#   - Compiled:     O(m * n) — pre-compile patterns; faster for repeated use.
#   - With groups:  O(m * n) — compile once; report match details too.
#
# Edge cases:
#   - No lines              -> []
#   - No matching patterns  -> (line, []) for each line
#   - Overlapping patterns  -> line may have multiple tags
#   - Invalid regex pattern -> raises ValueError at compile time
#   - None input            -> TypeError

import re


# === APPROACH 1: NAIVE — COMPILE EACH TIME — O(m * n) ===
# "Correct but recompiles on every call. Mention it as a performance concern."

def tag_logs_naive(log_lines, tag_patterns):
    if log_lines is None or tag_patterns is None:
        raise TypeError("Inputs cannot be None")
    if not isinstance(log_lines, list):
        raise TypeError(f"Expected list for log_lines, got {type(log_lines).__name__}")
    if not isinstance(tag_patterns, dict):
        raise TypeError(f"Expected dict for tag_patterns, got {type(tag_patterns).__name__}")

    result = []
    for line in log_lines:
        tags = []
        for tag, pattern in tag_patterns.items():
            if re.search(pattern, line):
                tags.append(tag)
        result.append((line, tags))
    return result


# === APPROACH 2: PRE-COMPILED PATTERNS — O(m * n) ===
# "Compile once, use many times. This is the right way in production."

def compile_rules(tag_patterns):
    if not isinstance(tag_patterns, dict):
        raise TypeError(f"Expected dict, got {type(tag_patterns).__name__}")
    compiled = {}
    for tag, pattern in tag_patterns.items():
        try:
            compiled[tag] = re.compile(pattern)
        except re.error as e:
            raise ValueError(f"Invalid regex for tag '{tag}': {e}") from e
    return compiled

def tag_logs(log_lines, tag_patterns):
    if log_lines is None or tag_patterns is None:
        raise TypeError("Inputs cannot be None")
    if not isinstance(log_lines, list):
        raise TypeError(f"Expected list, got {type(log_lines).__name__}")

    compiled = compile_rules(tag_patterns)
    result   = []

    for line in log_lines:
        if not isinstance(line, str):
            raise ValueError(f"Non-string log line: {line!r}")
        tags = [tag for tag, rx in compiled.items() if rx.search(line)]
        result.append((line, tags))

    return result


# === APPROACH 3: DETAILED — INCLUDE MATCH EVIDENCE — O(m * n) ===
# "Return match snippets alongside tags. Useful for SIEM alert context."

def tag_logs_detailed(log_lines, tag_patterns):
    if log_lines is None or tag_patterns is None:
        raise TypeError("Inputs cannot be None")
    if not isinstance(log_lines, list):
        raise TypeError(f"Expected list, got {type(log_lines).__name__}")

    compiled = compile_rules(tag_patterns)
    result   = []

    for line in log_lines:
        matches = {}
        for tag, rx in compiled.items():
            m = rx.search(line)
            if m:
                matches[tag] = m.group(0)   # first match snippet
        result.append({"line": line, "tags": list(matches.keys()), "evidence": matches})

    return result


# --- tests ---
logs = [
    "2023-10-01 FAIL login attempt for user admin",
    "2023-10-01 SYN packet from 10.0.0.1 to multiple ports",
    "2023-10-01 INFO normal operation",
    "2023-10-01 FAIL login and SYN to multiple ports detected",
]

patterns = {
    "brute_force": r"FAIL.*login",
    "port_scan":   r"SYN.*multiple",
}

result = tag_logs(logs, patterns)

assert result[0] == (logs[0], ["brute_force"])
assert result[1] == (logs[1], ["port_scan"])
assert result[2] == (logs[2], [])
# Line 4 matches both
assert "brute_force" in result[3][1]
assert "port_scan"   in result[3][1]

# Empty patterns -> all untagged
assert tag_logs(logs, {}) == [(l, []) for l in logs]

# Empty lines -> []
assert tag_logs([], patterns) == []

print(tag_logs(logs, patterns))
