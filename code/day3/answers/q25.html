# Q25 - Call Stack Depth Limiter (Fibonacci with Fallback)
# Compute Fibonacci(n) with a maximum recursion depth of 10.
# If n requires deeper recursion than max_depth, switch to iterative.
# Report which path was taken.
#
# Security context:
#   Recursive bombs (deeply nested inputs) can exhaust the call stack.
#   Always guard recursion depth in security-sensitive code (parsers, policy eval, etc.).
#   Python's default recursion limit is ~1000 (sys.getrecursionlimit()).
#
# Approach:
#   - Naive recursive:   O(2^n) — classic; blows up fast.
#   - Recursive memoised: O(n)  — with lru_cache; still hits stack limit for large n.
#   - Depth-limited:     O(n)   — detect at call time; fall back to iterative O(n).
#
# Fibonacci definition: fib(0)=0, fib(1)=1, fib(n)=fib(n-1)+fib(n-2).
#
# Edge cases:
#   - n=0       -> 0
#   - n=1       -> 1
#   - n < 0     -> raises ValueError
#   - n=max_depth+1 -> iterative path triggered
#   - None input -> TypeError

import sys
from functools import lru_cache


# === APPROACH 1: NAIVE RECURSIVE — O(2^n) ===
# "Classic; exponential. Mention it immediately and say you'd never use it for large n."

def fib_naive(n):
    if n is None:
        raise TypeError("n cannot be None")
    if not isinstance(n, int) or isinstance(n, bool):
        raise TypeError(f"Expected int, got {type(n).__name__}")
    if n < 0:
        raise ValueError(f"n must be >= 0, got {n}")
    if n <= 1:
        return n
    return fib_naive(n - 1) + fib_naive(n - 2)


# === APPROACH 2: MEMOISED RECURSIVE — O(n) ===
# "lru_cache eliminates redundant calls. Still limited by call stack."

@lru_cache(maxsize=None)
def fib_memo(n):
    if n < 0:
        raise ValueError(f"n must be >= 0, got {n}")
    if n <= 1:
        return n
    return fib_memo(n - 1) + fib_memo(n - 2)


# === APPROACH 3: DEPTH-LIMITED WITH ITERATIVE FALLBACK — O(n) ===
# "Check depth threshold at entry. If n > max_depth, use iterative.
#  This is the answer that shows production awareness."

def fib_iterative(n):
    if n <= 1:
        return n
    a, b = 0, 1
    for _ in range(2, n + 1):
        a, b = b, a + b
    return b

def _fib_recursive_guarded(n, depth, max_depth):
    if depth > max_depth:
        raise RecursionError("Depth exceeded")
    if n <= 1:
        return n
    return (_fib_recursive_guarded(n - 1, depth + 1, max_depth) +
            _fib_recursive_guarded(n - 2, depth + 1, max_depth))

def fib(n, max_depth=10):
    """
    Returns (result, path) where path is 'recursive' or 'iterative'.
    Switches to iterative if n > max_depth to avoid stack overflow.
    """
    if n is None:
        raise TypeError("n cannot be None")
    if not isinstance(n, int) or isinstance(n, bool):
        raise TypeError(f"Expected int, got {type(n).__name__}")
    if not isinstance(max_depth, int) or max_depth < 1:
        raise ValueError("max_depth must be a positive integer")
    if n < 0:
        raise ValueError(f"n must be >= 0, got {n}")

    if n <= max_depth:
        try:
            result = _fib_recursive_guarded(n, 0, max_depth)
            return (result, "recursive")
        except RecursionError:
            pass  # fallback

    return (fib_iterative(n), "iterative")


# --- tests ---
# Known values
known = {0: 0, 1: 1, 2: 1, 3: 2, 4: 3, 5: 5, 6: 8, 7: 13, 8: 21, 9: 34, 10: 55}

for n, expected in known.items():
    assert fib_naive(n) == expected
    assert fib_memo(n)  == expected

# Depth-limited version
for n in range(11):
    result, path = fib(n, max_depth=10)
    assert result == known[n], f"fib({n}) = {result}, expected {known[n]}"
    assert path == "recursive"

# Above threshold -> iterative
result, path = fib(15, max_depth=10)
assert result == 610   # fib(15)
assert path   == "iterative"

result, path = fib(50, max_depth=10)
assert result == 12586269025
assert path   == "iterative"

# Edge
assert fib(0)[0] == 0
assert fib(1)[0] == 1

print(fib(10, max_depth=10))   # (55, 'recursive')
print(fib(15, max_depth=10))   # (610, 'iterative')
print(fib(50, max_depth=10))   # (12586269025, 'iterative')
