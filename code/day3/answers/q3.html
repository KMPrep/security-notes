# Q3 - Evaluate Security Policy Expression (Reverse Polish Notation)
# Evaluate a postfix boolean expression with AND, OR, NOT and literals T/F.
#
# Approach:
#   - Naive:          O(n) — iterate tokens; use stack; handle T/F/AND/OR/NOT.
#   - No built-ins:   O(n) — same logic; manual boolean ops using integer 0/1.
#   - With built-ins: O(n) — clean stack with Python bool; operand dict for ops.
#
# RPN rules:
#   Push T or F as boolean.
#   AND: pop two, push (a AND b).
#   OR:  pop two, push (a OR b).
#   NOT: pop one, push (NOT a).
#
# Edge cases:
#   - Single T/F            -> that value
#   - Malformed expression  -> raises ValueError
#   - Unknown token         -> raises ValueError
#   - Empty list            -> raises ValueError
#   - None input            -> TypeError


# === APPROACH 1: NAIVE — O(n) ===
# "Walk the tokens, push booleans, apply ops. Good first answer."

def eval_policy_naive(tokens):
    if tokens is None:
        raise TypeError("Input cannot be None")
    if not isinstance(tokens, list):
        raise TypeError(f"Expected list, got {type(tokens).__name__}")
    if not tokens:
        raise ValueError("Empty expression")

    stack = []
    for tok in tokens:
        if tok == "T":
            stack.append(True)
        elif tok == "F":
            stack.append(False)
        elif tok == "NOT":
            if len(stack) < 1:
                raise ValueError("NOT requires 1 operand on stack")
            a = stack.pop()
            stack.append(not a)
        elif tok == "AND":
            if len(stack) < 2:
                raise ValueError("AND requires 2 operands on stack")
            b = stack.pop()
            a = stack.pop()
            stack.append(a and b)
        elif tok == "OR":
            if len(stack) < 2:
                raise ValueError("OR requires 2 operands on stack")
            b = stack.pop()
            a = stack.pop()
            stack.append(a or b)
        else:
            raise ValueError(f"Unknown token: {tok!r}")

    if len(stack) != 1:
        raise ValueError(f"Malformed expression; stack has {len(stack)} values")
    return stack[0]


# === APPROACH 2: NO BUILT-INS — INTEGER BOOLEANS — O(n) ===
# "Represent True/False as 1/0. Shows understanding of boolean algebra without Python keywords."

def eval_policy_no_builtins(tokens):
    if tokens is None:
        raise TypeError("Input cannot be None")
    if not isinstance(tokens, list):
        raise TypeError(f"Expected list, got {type(tokens).__name__}")
    if not tokens:
        raise ValueError("Empty expression")

    stack = []
    for tok in tokens:
        if tok == "T":
            stack.append(1)
        elif tok == "F":
            stack.append(0)
        elif tok == "NOT":
            if len(stack) < 1:
                raise ValueError("NOT requires 1 operand on stack")
            a = stack[len(stack) - 1]
            stack = stack[:len(stack) - 1]
            stack.append(1 - a)          # NOT: flip 0/1
        elif tok == "AND":
            if len(stack) < 2:
                raise ValueError("AND requires 2 operands on stack")
            b = stack[len(stack) - 1]
            a = stack[len(stack) - 2]
            stack = stack[:len(stack) - 2]
            stack.append(1 if (a == 1 and b == 1) else 0)
        elif tok == "OR":
            if len(stack) < 2:
                raise ValueError("OR requires 2 operands on stack")
            b = stack[len(stack) - 1]
            a = stack[len(stack) - 2]
            stack = stack[:len(stack) - 2]
            stack.append(1 if (a == 1 or b == 1) else 0)
        else:
            raise ValueError(f"Unknown token: {tok!r}")

    if len(stack) != 1:
        raise ValueError("Malformed expression")
    return stack[0] == 1


# === APPROACH 3: WITH BUILT-INS — OPERATION DICT — O(n) ===
# "Dict of lambdas for the operators. Clean, extensible."

def eval_policy(tokens):
    if tokens is None:
        raise TypeError("Input cannot be None")
    if not isinstance(tokens, list):
        raise TypeError(f"Expected list, got {type(tokens).__name__}")
    if not tokens:
        raise ValueError("Empty expression")

    binary_ops = {
        "AND": lambda a, b: a and b,
        "OR":  lambda a, b: a or b,
    }
    stack = []

    for tok in tokens:
        if tok == "T":
            stack.append(True)
        elif tok == "F":
            stack.append(False)
        elif tok == "NOT":
            if not stack:
                raise ValueError("NOT requires 1 operand on stack")
            stack.append(not stack.pop())
        elif tok in binary_ops:
            if len(stack) < 2:
                raise ValueError(f"{tok} requires 2 operands on stack")
            b, a = stack.pop(), stack.pop()
            stack.append(binary_ops[tok](a, b))
        else:
            raise ValueError(f"Unknown token: {tok!r}")

    if len(stack) != 1:
        raise ValueError(f"Malformed expression; {len(stack)} values remain")
    return stack[0]


# --- tests ---
assert eval_policy(["T", "F", "AND"]) is False
assert eval_policy(["T", "F", "OR", "T", "AND"]) is True
assert eval_policy(["T", "NOT"]) is False
assert eval_policy(["F", "NOT"]) is True
assert eval_policy(["T"]) is True
assert eval_policy(["T", "T", "OR"]) is True
assert eval_policy(["F", "F", "AND"]) is False

assert eval_policy_naive(["T", "F", "AND"]) is False
assert eval_policy_no_builtins(["T", "F", "OR", "T", "AND"]) is True

print(eval_policy(["T", "F", "AND"]))               # False
print(eval_policy(["T", "F", "OR", "T", "AND"]))    # True
print(eval_policy(["T", "NOT"]))                    # False
