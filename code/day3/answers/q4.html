# Q4 - Decode Compressed Log Token
# Given a string where k[pattern] means repeat pattern k times, decode it.
# Supports arbitrary nesting: 2[a3[b]] -> "abbbabbb".
#
# Approach:
#   - Naive:          O(n * max_repeat) — iterative, single-level only (no nesting).
#   - No built-ins:   O(output_length) — stack of (pending_string, repeat_count); manual concat.
#   - With built-ins: O(output_length) — same stack logic with string * k.
#
# Stack state:
#   When we see '[': push (current_string, current_k) onto stack; reset both.
#   When we see ']': pop (prev_string, k); set current = prev + current * k.
#   Digits accumulate into current_k (handles multi-digit like 12[x]).
#   Letters accumulate into current_string.
#
# Edge cases:
#   - No brackets        -> return as-is
#   - k=0                -> empty string for that segment
#   - Nested brackets    -> resolved inside-out
#   - Empty string       -> ""
#   - None input         -> TypeError


# === APPROACH 1: NAIVE — SINGLE-LEVEL ONLY — O(n * max_repeat) ===
# "Handle non-nested case first. Mention it breaks for nesting — that motivates the stack."

def decode_token_naive(s):
    if s is None:
        raise TypeError("Input cannot be None")
    if not isinstance(s, str):
        raise TypeError(f"Expected str, got {type(s).__name__}")

    result = ""
    i = 0
    while i < len(s):
        if s[i].isdigit():
            k = 0
            while i < len(s) and s[i].isdigit():
                k = k * 10 + int(s[i])
                i += 1
            if i < len(s) and s[i] == "[":
                i += 1  # skip '['
                inner = ""
                # Only handles one level — stops at first ']'
                while i < len(s) and s[i] != "]":
                    inner += s[i]
                    i += 1
                i += 1  # skip ']'
                result += inner * k
        else:
            result += s[i]
            i += 1
    return result


# === APPROACH 2: STACK — NO BUILT-INS — O(output_length) ===
# "Push state on '[', restore and concat on ']'. No string multiplication built-in."

def decode_token_no_builtins(s):
    if s is None:
        raise TypeError("Input cannot be None")
    if not isinstance(s, str):
        raise TypeError(f"Expected str, got {type(s).__name__}")

    stack      = []
    current    = ""
    current_k  = 0

    for ch in s:
        if ch.isdigit():
            current_k = current_k * 10 + int(ch)
        elif ch == "[":
            stack.append((current, current_k))
            current   = ""
            current_k = 0
        elif ch == "]":
            prev_str, k = stack[len(stack) - 1]
            stack = stack[:len(stack) - 1]
            repeated = ""
            for _ in range(k):
                repeated += current
            current = prev_str + repeated
        else:
            current += ch

    return current


# === APPROACH 3: WITH BUILT-INS — O(output_length) ===
# "Same algorithm; use string * k. Clean and concise."

def decode_token(s):
    if s is None:
        raise TypeError("Input cannot be None")
    if not isinstance(s, str):
        raise TypeError(f"Expected str, got {type(s).__name__}")

    stack     = []
    current   = ""
    current_k = 0

    for ch in s:
        if ch.isdigit():
            current_k = current_k * 10 + int(ch)
        elif ch == "[":
            stack.append((current, current_k))
            current   = ""
            current_k = 0
        elif ch == "]":
            prev_str, k = stack.pop()
            current = prev_str + current * k
        else:
            current += ch

    return current


# --- tests ---
assert decode_token("3[ab]2[cd]")  == "ab" * 3 + "cd" * 2   # "ababab" + "cdcd"
assert decode_token("2[a3[b]]")    == "abbbabbb"              # nested
assert decode_token("abc")         == "abc"
assert decode_token("")            == ""
assert decode_token("1[x]")        == "x"
assert decode_token("10[a]")       == "a" * 10

assert decode_token_no_builtins("3[ab]2[cd]") == "ab" * 3 + "cd" * 2
assert decode_token_no_builtins("2[a3[b]]")   == "abbbabbb"

# Naive only handles single-level:
assert decode_token_naive("3[ab]2[cd]") == "ab" * 3 + "cd" * 2

print(decode_token("3[ab]2[cd]"))  # ababab + cdcd = ababababcdcd
print(decode_token("2[a3[b]]"))    # abbbabbb
print(decode_token("10[a]"))       # aaaaaaaaaa
