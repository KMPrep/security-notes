# Q6 - Largest Alert Window (Largest Rectangle in Histogram)
# Given alert counts per hour as bar heights, find the largest rectangle
# that fits entirely within the histogram (max width x min height in range).
#
# Approach:
#   - Naive:          O(n^2) — for each pair (i,j) compute min height * width.
#   - No built-ins:   O(n)   — monotonic increasing stack; manual min-height tracking.
#   - With built-ins: O(n)   — same stack; list.pop() + max().
#
# Key insight (stack approach):
#   Maintain a stack of indices with increasing heights.
#   When we find a bar shorter than the top, we know the top bar's
#   "right boundary". Pop and compute its rectangle width using the new top as left boundary.
#   At end, flush remaining stack with right boundary = n.
#
# Edge cases:
#   - Empty list          -> 0
#   - Single bar          -> that bar's height
#   - All equal heights   -> height * n
#   - All decreasing      -> first bar (tallest) stands alone, width 1
#   - None input          -> TypeError


# === APPROACH 1: NAIVE — O(n^2) ===
# "For each starting bar, sweep right tracking min height."

def largest_rect_naive(heights):
    if heights is None:
        raise TypeError("Input cannot be None")
    if not isinstance(heights, list):
        raise TypeError(f"Expected list, got {type(heights).__name__}")

    n    = len(heights)
    best = 0

    for i in range(n):
        min_h = heights[i]
        for j in range(i, n):
            if heights[j] < min_h:
                min_h = heights[j]
            area = min_h * (j - i + 1)
            if area > best:
                best = area

    return best


# === APPROACH 2: MONOTONIC STACK — NO BUILT-INS — O(n) ===
# "Track indices in increasing order. Pop when a shorter bar is encountered."

def largest_rect_no_builtins(heights):
    if heights is None:
        raise TypeError("Input cannot be None")
    if not isinstance(heights, list):
        raise TypeError(f"Expected list, got {type(heights).__name__}")

    n     = len(heights)
    stack = []  # indices with increasing heights
    best  = 0

    def process(i):
        nonlocal best
        while len(stack) > 0 and heights[stack[len(stack) - 1]] >= heights[i]:
            idx  = stack[len(stack) - 1]
            stack.pop()  # can't do stack = stack[:...] inside nested scope easily
            h    = heights[idx]
            left = stack[len(stack) - 1] + 1 if len(stack) > 0 else 0
            w    = i - left
            area = h * w
            if area > best:
                best = area

    # We need a mutable list so nonlocal works properly; using list directly.
    stack2 = []
    best2  = 0
    for i in range(n):
        while len(stack2) > 0 and heights[stack2[-1]] >= heights[i]:
            idx   = stack2.pop()
            h     = heights[idx]
            left  = stack2[-1] + 1 if stack2 else 0
            area  = h * (i - left)
            if area > best2:
                best2 = area
        stack2.append(i)

    # flush remaining
    for i in range(len(stack2) - 1, -1, -1):
        idx  = stack2[i]
        h    = heights[idx]
        left = stack2[i - 1] + 1 if i > 0 else 0
        area = h * (n - left)
        if area > best2:
            best2 = area

    return best2


# === APPROACH 3: WITH BUILT-INS — O(n) ===
# "Append sentinel 0 to flush stack cleanly at end."

def largest_rect(heights):
    if heights is None:
        raise TypeError("Input cannot be None")
    if not isinstance(heights, list):
        raise TypeError(f"Expected list, got {type(heights).__name__}")

    bars  = heights + [0]   # sentinel triggers full flush at end
    stack = []              # indices with strictly increasing heights
    best  = 0

    for i, h in enumerate(bars):
        while stack and bars[stack[-1]] >= h:
            height = bars[stack.pop()]
            width  = i - stack[-1] - 1 if stack else i
            best   = max(best, height * width)
        stack.append(i)

    return best


# --- tests ---
sample = [2, 1, 5, 6, 2, 3]

assert largest_rect_naive(sample)       == 10
assert largest_rect_no_builtins(sample) == 10
assert largest_rect(sample)             == 10

assert largest_rect([2, 4]) == 4
assert largest_rect([6, 2, 5, 4, 5, 1, 6]) == 12
assert largest_rect([]) == 0
assert largest_rect([5]) == 5
assert largest_rect([3, 3, 3]) == 9

print(largest_rect(sample))              # 10
print(largest_rect([6, 2, 5, 4, 5, 1, 6]))  # 12
print(largest_rect([3, 3, 3]))           # 9
