# Q7 - Zombie Process Chain
# Simulate a process call stack given push (P<name>) and kill (K<name>) operations.
# Return the names of processes in the order they were killed.
#
# Approach:
#   - Naive:          O(n) — list as stack; P pushes, K pops and records.
#   - No built-ins:   O(n) — same but manual pop via slicing.
#   - With built-ins: O(n) — clean with list.append() / list.pop().
#
# Format: operations list where each item is either "P<name>" or "K<name>".
#   P means push that process onto the stack.
#   K means kill (pop) that process; add it to the killed-order list.
#   The name in K must match the current top (validates LIFO order).
#
# Edge cases:
#   - Kill on empty stack         -> raises ValueError
#   - Kill with mismatched name   -> raises ValueError (not LIFO)
#   - Stack not empty at end      -> remaining processes are still running (not killed)
#   - None input                  -> TypeError
#   - Empty operation list        -> []


# === APPROACH 1: NAIVE — O(n) ===
# "Simple stack simulation. Good starting answer."

def zombie_chain_naive(calls):
    if calls is None:
        raise TypeError("Input cannot be None")
    if not isinstance(calls, list):
        raise TypeError(f"Expected list, got {type(calls).__name__}")

    stack  = []
    killed = []

    for op in calls:
        if not isinstance(op, str) or len(op) < 2:
            raise ValueError(f"Invalid operation: {op!r}")

        action = op[0]
        name   = op[1:]

        if action == "P":
            stack.append(name)
        elif action == "K":
            if not stack:
                raise ValueError(f"Kill '{name}' on empty stack")
            top = stack[-1]
            if top != name:
                raise ValueError(f"Kill '{name}' but top of stack is '{top}' (not LIFO)")
            stack.pop()
            killed.append(name)
        else:
            raise ValueError(f"Unknown action '{action}' in op '{op}'")

    return killed


# === APPROACH 2: NO BUILT-INS — O(n) ===
# "Manual pop via slicing. Shows awareness of what pop() does under the hood."

def zombie_chain_no_builtins(calls):
    if calls is None:
        raise TypeError("Input cannot be None")
    if not isinstance(calls, list):
        raise TypeError(f"Expected list, got {type(calls).__name__}")

    stack  = []
    killed = []

    for i in range(len(calls)):
        op = calls[i]
        if not isinstance(op, str) or len(op) < 2:
            raise ValueError(f"Invalid operation: {op!r}")

        action = op[0]
        name   = op[1:]

        if action == "P":
            stack = stack + [name]
        elif action == "K":
            if len(stack) == 0:
                raise ValueError(f"Kill '{name}' on empty stack")
            top = stack[len(stack) - 1]
            if top != name:
                raise ValueError(f"Kill '{name}' but top is '{top}'")
            stack  = stack[:len(stack) - 1]
            killed = killed + [name]
        else:
            raise ValueError(f"Unknown action '{action}'")

    return killed


# === APPROACH 3: WITH BUILT-INS — O(n) ===
# "Same as naive but cleaner syntax. What you'd write in production."

def zombie_chain(calls):
    if calls is None:
        raise TypeError("Input cannot be None")
    if not isinstance(calls, list):
        raise TypeError(f"Expected list, got {type(calls).__name__}")

    stack  = []
    killed = []

    for op in calls:
        if not isinstance(op, str) or len(op) < 2:
            raise ValueError(f"Invalid operation: {op!r}")
        action, name = op[0], op[1:]
        if action == "P":
            stack.append(name)
        elif action == "K":
            if not stack:
                raise ValueError(f"Kill '{name}' on empty stack")
            if stack[-1] != name:
                raise ValueError(f"Kill '{name}' but top is '{stack[-1]}'")
            stack.pop()
            killed.append(name)
        else:
            raise ValueError(f"Unknown action '{action}'")

    return killed


# --- tests ---
ops = ["PA","PB","PC","KC","PD","KD","KB","KA"]

assert zombie_chain_naive(ops)       == ["C","D","B","A"]
assert zombie_chain_no_builtins(ops) == ["C","D","B","A"]
assert zombie_chain(ops)             == ["C","D","B","A"]

assert zombie_chain(["PA","KA"]) == ["A"]
assert zombie_chain([]) == []

# Kill mismatch should raise
try:
    zombie_chain(["PA","PB","KA"])   # top is B, not A
    assert False, "Should raise ValueError"
except ValueError:
    pass

print(zombie_chain(ops))             # ['C', 'D', 'B', 'A']
print(zombie_chain(["PA","KA"]))     # ['A']
