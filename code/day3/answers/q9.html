# Q9 - Count Nodes in a Policy Tree
# Given a dict {node: [children]} and a root, recursively count all nodes.
#
# Approach:
#   - Naive/Recursive:  O(n) — DFS recursion, count 1 + sum of children counts.
#   - Iterative DFS:    O(n) — explicit stack; avoid recursion limit.
#   - Iterative BFS:    O(n) — queue-based level-by-level count.
#
# Edge cases:
#   - Root not in tree   -> root still counts as 1 (it's a leaf)
#   - Empty tree dict    -> 1 (just the root node)
#   - Single node        -> 1
#   - None input         -> TypeError
#   - Cycle in tree      -> handle via visited set (trees shouldn't cycle, but guard anyway)


# === APPROACH 1: RECURSIVE DFS — O(n) ===
# "Standard recursive DFS. Good first answer."

def count_nodes_recursive(tree, root):
    if tree is None:
        raise TypeError("tree cannot be None")
    if not isinstance(tree, dict):
        raise TypeError(f"Expected dict, got {type(tree).__name__}")
    if root is None:
        raise TypeError("root cannot be None")

    children = tree.get(root, [])
    total = 1
    for child in children:
        total += count_nodes_recursive(tree, child)
    return total


# === APPROACH 2: ITERATIVE DFS — NO RECURSION — O(n) ===
# "Explicit stack. Safe for large or deep policy trees."

def count_nodes_dfs(tree, root):
    if tree is None:
        raise TypeError("tree cannot be None")
    if not isinstance(tree, dict):
        raise TypeError(f"Expected dict, got {type(tree).__name__}")
    if root is None:
        raise TypeError("root cannot be None")

    stack   = [root]
    visited = set()
    count   = 0

    while stack:
        node = stack.pop()
        if node in visited:
            continue   # cycle guard
        visited.add(node)
        count += 1
        for child in tree.get(node, []):
            if child not in visited:
                stack.append(child)

    return count


# === APPROACH 3: BFS WITH COLLECTIONS.DEQUE — O(n) ===
# "Level-by-level; useful when you also need depth information."

from collections import deque

def count_nodes(tree, root):
    if tree is None:
        raise TypeError("tree cannot be None")
    if not isinstance(tree, dict):
        raise TypeError(f"Expected dict, got {type(tree).__name__}")
    if root is None:
        raise TypeError("root cannot be None")

    queue   = deque([root])
    visited = set([root])
    count   = 0

    while queue:
        node = queue.popleft()
        count += 1
        for child in tree.get(node, []):
            if child not in visited:
                visited.add(child)
                queue.append(child)

    return count


# --- tests ---
tree = {"root": ["a", "b"], "a": ["c"], "b": [], "c": []}

assert count_nodes_recursive(tree, "root") == 4
assert count_nodes_dfs(tree, "root")       == 4
assert count_nodes(tree, "root")           == 4

# Single node
assert count_nodes({}, "x")           == 1

# Root not in dict
assert count_nodes({"a": ["b"]}, "z") == 1

# Deeper tree
deep = {"r": ["a","b"], "a": ["c","d"], "b": ["e"], "c": [], "d": [], "e": []}
assert count_nodes(deep, "r") == 6

print(count_nodes(tree, "root"))     # 4
print(count_nodes(deep, "r"))        # 6
print(count_nodes({}, "x"))          # 1
