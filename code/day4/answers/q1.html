# Q1 - Classic Binary Search
# Given a sorted array, return the index of target or -1 if not found.
#
# ─── SOLUTION WALKTHROUGH ───────────────────────────────────────────────────
#
# Follow-up: "Implement binary search from scratch."
#
# STEP 1 — Clarify assumptions first (Google expects this)
#   "Is the array guaranteed sorted? Can it have duplicates? What type — ints only?"
#   "Can the array be None, or do I treat that as empty?"
#
# STEP 2 — Start naive to establish correctness baseline
#   "My first instinct is a linear scan — iterate and compare every element.
#    That's O(n). It works but ignores the sorted property."
#
# STEP 3 — Recognize the optimization
#   "Since it's sorted, every comparison can eliminate half the search space.
#    Binary search gives O(log n). The key insight: if arr[mid] < target, the
#    answer must be in the right half; otherwise the left half."
#
# STEP 4 — Edge cases BEFORE writing code
#   - arr is None or not a list                  → raise TypeError
#   - Empty array                                → return -1
#   - Single element (match)                     → return 0
#   - Single element (miss)                      → return -1
#   - Target smaller than arr[0]                 → lo > hi immediately, return -1
#   - Target larger than arr[-1]                 → same
#   - Target at index 0 or len-1                 → must still work
#
# STEP 5 — Overflow note (mention even in Python to show C++/Java awareness)
#   "In C++/Java, (lo + hi) can overflow int. Safe form: lo + (hi - lo) // 2.
#    Python has arbitrary-precision ints so it's not strictly needed, but
#    I'll mention it because these problems often have a follow-up."
#
# Complexity:
#   Time  — O(log n): each iteration halves the search space
#   Space — O(1) iterative; O(log n) recursive due to call stack
# ─────────────────────────────────────────────────────────────────────────────


# ── APPROACH 1: Naive linear scan (state this first, then improve) ────────────

def binary_search_naive(arr: list, target: int) -> int:
    """
    Linear scan — O(n) time, O(1) space.
    Mentioned as a starting point; not the answer we'd submit.
    """
    if not isinstance(arr, list):
        raise TypeError(f"Expected list, got {type(arr).__name__}")
    for i in range(len(arr)):
        if arr[i] == target:
            return i
    return -1


# ── APPROACH 2: Binary search — iterative (preferred at Google) ──────────────

def binary_search(arr: list, target: int) -> int:
    """
    Iterative binary search.

    Time:  O(log n) — search space halved each iteration
    Space: O(1)     — no recursion, no extra data structures
    """
    if not isinstance(arr, list):
        raise TypeError(f"Expected list, got {type(arr).__name__}")
    if not arr:
        return -1

    lo, hi = 0, len(arr) - 1

    while lo <= hi:
        mid = lo + (hi - lo) // 2          # overflow-safe form
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            lo = mid + 1
        else:
            hi = mid - 1

    return -1


# ── APPROACH 3: Binary search — recursive ────────────────────────────────────
# "If the reviewer asks for a recursive version, here it is.
#  Downside: call stack O(log n) — Python's default recursion limit is ~1000,
#  so for very deep recursion this could fail. Iterative is safer in production."

def binary_search_recursive(arr: list, target: int,
                            lo: int = None, hi: int = None) -> int:
    """
    Recursive binary search.

    Time:  O(log n)
    Space: O(log n) — call stack depth
    """
    if not isinstance(arr, list):
        raise TypeError(f"Expected list, got {type(arr).__name__}")
    if not arr:
        return -1

    if lo is None:
        lo = 0
    if hi is None:
        hi = len(arr) - 1

    if lo > hi:
        return -1

    mid = lo + (hi - lo) // 2
    if arr[mid] == target:
        return mid
    elif arr[mid] < target:
        return binary_search_recursive(arr, target, mid + 1, hi)
    else:
        return binary_search_recursive(arr, target, lo, mid - 1)


# ── MANUAL — no bisect, no sorted(), no stdlib helpers ───────────────────────
# "If you asked me to avoid all built-ins, here's the same logic written
#  with only basic Python operators and len()."

def binary_search_manual(arr: list, target: int) -> int:
    """
    Binary search without any stdlib imports.

    Time:  O(log n)
    Space: O(1)
    """
    if arr is None:
        raise ValueError("Input array cannot be None")
    n = len(arr)
    if n == 0:
        return -1

    lo = 0
    hi = n - 1

    while lo <= hi:
        mid = lo + (hi - lo) // 2
        val = arr[mid]
        if val == target:
            return mid
        if val < target:
            lo = mid + 1
        else:
            hi = mid - 1

    return -1


# ── WITH BUILT-IN: using bisect module ───────────────────────────────────────
# "In production Python I'd use bisect which has the same O(log n) guarantee
#  and is implemented in C."

import bisect

def binary_search_builtin(arr: list, target: int) -> int:
    """
    Uses bisect.bisect_left + index check.

    Time:  O(log n)
    Space: O(1)
    """
    if not isinstance(arr, list):
        raise TypeError(f"Expected list, got {type(arr).__name__}")
    if not arr:
        return -1

    idx = bisect.bisect_left(arr, target)
    if idx < len(arr) and arr[idx] == target:
        return idx
    return -1


# ── Tests ─────────────────────────────────────────────────────────────────────

if __name__ == "__main__":
    arr = [1, 3, 5, 7, 9, 11]

    # Basic cases
    assert binary_search(arr, 7)   == 3,  "Should find 7 at index 3"
    assert binary_search(arr, 4)   == -1, "4 not in array"
    assert binary_search(arr, 1)   == 0,  "First element"
    assert binary_search(arr, 11)  == 5,  "Last element"

    # Edge cases
    assert binary_search([], 5)    == -1, "Empty array"
    assert binary_search([5], 5)   == 0,  "Single element match"
    assert binary_search([5], 3)   == -1, "Single element miss"

    # Recursive
    assert binary_search_recursive(arr, 7)  == 3
    assert binary_search_recursive(arr, 4)  == -1
    assert binary_search_recursive([], 5)   == -1

    # Manual
    assert binary_search_manual(arr, 7)   == 3
    assert binary_search_manual(arr, 4)   == -1
    assert binary_search_manual([5], 5)   == 0

    # Built-in
    assert binary_search_builtin(arr, 7)  == 3
    assert binary_search_builtin(arr, 4)  == -1

    # TypeError guards
    try:
        binary_search("not a list", 3)
        assert False, "Should have raised TypeError"
    except TypeError:
        pass

    print("Q1 — All assertions passed.")
    print(f"binary_search([1,3,5,7,9,11], 7)  = {binary_search(arr, 7)}")   # 3
    print(f"binary_search([1,3,5,7,9,11], 4)  = {binary_search(arr, 4)}")   # -1
