# Q10 - Custom Sort: Severity-First Triage
# Given a list of (severity, timestamp, message) tuples, sort by:
#   1. Severity descending: CRITICAL > HIGH > MEDIUM > LOW > INFO
#   2. Timestamp ascending within same severity
#
# ─── SOLUTION WALKTHROUGH ───────────────────────────────────────────────────
#
# STEP 1 — Clarify
#   "The severity field is a string. The problem says use a custom key function
#    — meaning don't write a cascade of if/elif comparisons for every pair.
#    Is the severity always one of the five values? What happens on unknown?"
#
# STEP 2 — Naive approach
#   "I could write a comparator function with nested ifs — compare severity
#    strings directly. That's brittle and O(n²) if using compare-based sort."
#
# STEP 3 — Proper approach: key function with an order dict
#   "Map each severity string to a numeric rank. Smaller rank = higher priority.
#    CRITICAL → 0, HIGH → 1, MEDIUM → 2, LOW → 3, INFO → 4.
#    Unknown → large sentinel (99) so they sort to the bottom.
#    Key tuple: (rank, timestamp). Python sort is stable and handles tuple
#    comparison lexicographically — rank first, timestamp second."
#
# STEP 4 — Edge cases
#   - Empty list                         → return []
#   - Unknown severity string            → sort to bottom (use sentinel)
#   - All same severity                  → sorted by timestamp
#   - Negative timestamps                → still sortable (epoch before 1970)
#   - Ties on both severity AND timestamp → stable sort preserves insertion order
#
# Complexity:
#   Time  — O(n log n)
#   Space — O(n) for sorted() output; O(1) for in-place sort()
# ─────────────────────────────────────────────────────────────────────────────


# Severity ordering map — lower number = higher priority
SEVERITY_ORDER = {
    "CRITICAL": 0,
    "HIGH":     1,
    "MEDIUM":   2,
    "LOW":      3,
    "INFO":     4,
}

SEVERITY_SENTINEL = 99   # unknown severities sort to the bottom


# ── APPROACH 1: Naive — compare-based (fragile, O(n²) sort if not careful) ───

def sort_events_naive(events: list) -> list:
    """
    Manual bubble sort with explicit severity comparison.
    O(n²) and fragile — shows the problem with ad-hoc comparisons.
    """
    if not isinstance(events, list):
        raise TypeError(f"Expected list, got {type(events).__name__}")

    def sev_rank(s):
        return SEVERITY_ORDER.get(s, SEVERITY_SENTINEL)

    arr = list(events)
    n = len(arr)
    for i in range(n):
        for j in range(n - i - 1):
            sev_j   = sev_rank(arr[j][0])
            sev_j1  = sev_rank(arr[j + 1][0])
            ts_j    = arr[j][1]
            ts_j1   = arr[j + 1][1]
            # Compare: lower sev_rank = higher priority = should come FIRST
            if (sev_j, ts_j) > (sev_j1, ts_j1):
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
    return arr


# ── APPROACH 2: Built-in sort with custom key (correct O(n log n)) ───────────

def sort_events(events: list) -> list:
    """
    Sort using a key tuple: (severity_rank, timestamp).

    Key design:
      - severity_rank: lower = higher priority (CRITICAL=0, INFO=4)
      - timestamp: lower = earlier (ascending within same severity)

    Time:  O(n log n)
    Space: O(n)
    """
    if not isinstance(events, list):
        raise TypeError(f"Expected list, got {type(events).__name__}")
    if not events:
        return []

    for i, ev in enumerate(events):
        if not (isinstance(ev, tuple) and len(ev) == 3):
            raise ValueError(f"Event at index {i} must be (severity, ts, msg), got {ev!r}")

    def sort_key(event):
        severity, timestamp, _ = event
        rank = SEVERITY_ORDER.get(severity, SEVERITY_SENTINEL)
        return (rank, timestamp)

    return sorted(events, key=sort_key)


# ── APPROACH 3: Using functools.cmp_to_key for explicit comparator ────────────

from functools import cmp_to_key

def sort_events_cmp(events: list) -> list:
    """
    Explicit comparator via cmp_to_key.
    Less efficient than key-based sort (comparator called O(n log n) times
    vs key computed O(n) times then compared O(n log n) times).
    Useful when multi-field comparison logic is hard to express as a key.

    Time:  O(n log n)
    Space: O(n)
    """
    if not isinstance(events, list):
        raise TypeError(f"Expected list, got {type(events).__name__}")
    if not events:
        return []

    def cmp(a, b):
        rank_a = SEVERITY_ORDER.get(a[0], SEVERITY_SENTINEL)
        rank_b = SEVERITY_ORDER.get(b[0], SEVERITY_SENTINEL)
        if rank_a != rank_b:
            return rank_a - rank_b      # lower rank = comes first
        if a[1] != b[1]:
            return a[1] - b[1]         # earlier timestamp first
        return 0

    return sorted(events, key=cmp_to_key(cmp))


# ── MANUAL — no built-in sort, no sorted() ────────────────────────────────────

def sort_events_manual(events: list) -> list:
    """
    Stable merge sort with custom key comparison.
    No built-in sort functions.

    Time:  O(n log n)
    Space: O(n)
    """
    if events is None:
        raise ValueError("events cannot be None")

    def key_of(ev):
        return (SEVERITY_ORDER.get(ev[0], SEVERITY_SENTINEL), ev[1])

    def merge(left, right):
        result = []
        i = j = 0
        while i < len(left) and j < len(right):
            if key_of(left[i]) <= key_of(right[j]):
                result.append(left[i])
                i += 1
            else:
                result.append(right[j])
                j += 1
        while i < len(left):
            result.append(left[i])
            i += 1
        while j < len(right):
            result.append(right[j])
            j += 1
        return result

    def mergesort(arr):
        if len(arr) <= 1:
            return list(arr)
        mid = len(arr) // 2
        return merge(mergesort(arr[:mid]), mergesort(arr[mid:]))

    return mergesort(events)


# ── Tests ─────────────────────────────────────────────────────────────────────

if __name__ == "__main__":
    events = [
        ("LOW",      1620000010, "Low severity event"),
        ("CRITICAL", 1620000002, "Critical: data exfil"),
        ("HIGH",     1620000004, "High: lateral move"),
        ("CRITICAL", 1620000001, "Critical: root shell"),   # earlier than above
        ("MEDIUM",   1620000006, "Medium: suspicious scan"),
        ("INFO",     1620000008, "Info: user login"),
        ("UNKNOWN",  1620000003, "Unknown severity event"),  # sentinel test
    ]

    expected_severities = ["CRITICAL", "CRITICAL", "HIGH", "MEDIUM", "LOW", "INFO", "UNKNOWN"]
    expected_critical_ts = [1620000001, 1620000002]   # must be ascending within CRITICAL

    result = sort_events(events)
    assert [e[0] for e in result] == expected_severities, \
        f"Severity order wrong: {[e[0] for e in result]}"

    critical_events = [e for e in result if e[0] == "CRITICAL"]
    assert [e[1] for e in critical_events] == expected_critical_ts, \
        "CRITICAL events not sorted by timestamp"

    # All methods agree
    result_cmp    = sort_events_cmp(events)
    result_manual = sort_events_manual(events)
    result_naive  = sort_events_naive(events)

    assert [e[0] for e in result_cmp]    == expected_severities
    assert [e[0] for e in result_manual] == expected_severities
    assert [e[0] for e in result_naive]  == expected_severities

    # Edge cases
    assert sort_events([]) == []
    assert sort_events([("HIGH", 1, "only one")]) == [("HIGH", 1, "only one")]

    print("Q10 — All assertions passed.")
    print("\nSorted events:")
    for ev in result:
        print(f"  {ev[0]:8s}  ts={ev[1]}  {ev[2]}")
