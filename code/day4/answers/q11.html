# Q11 - Sort Strings by Security Relevance (Custom Comparator)
# Given a list of process names, sort so that any name containing "root" or
# "admin" comes first (alphabetically among themselves), followed by all others
# alphabetically. Use functools.cmp_to_key.
#
# ─── SOLUTION WALKTHROUGH ───────────────────────────────────────────────────
#
# STEP 1 — Clarify
#   "Case-sensitive or insensitive for 'root'/'admin'?
#    If two names are both privileged (or both non-privileged), sort
#    alphabetically between them?"
#
# STEP 2 — Naive approach
#   "Split into two lists: privileged and others. Sort each alphabetically.
#    Concatenate. That's O(n log n) and perfectly clean."
#
# STEP 3 — With cmp_to_key (as the problem requires)
#   "Define a comparator that:
#    - Returns negative if a should come before b
#    - Returns positive if b should come before a
#    - Returns 0 if equal
#    Logic:
#      a_priv = 'root' in a or 'admin' in a
#      b_priv = 'root' in b or 'admin' in b
#      If both privileged: compare alphabetically
#      If only a privileged: a comes first (return -1)
#      If only b privileged: b comes first (return +1)
#      If neither privileged: compare alphabetically"
#
# STEP 4 — Edge cases
#   - Empty list                             → []
#   - All privileged names                   → sorted alphabetically
#   - No privileged names                    → sorted alphabetically
#   - Name that is exactly "root" or "admin" → matches
#   - Name containing "rootkit" or "admin_panel" → also matches (substring)
#   - Case sensitivity: "Root" vs "root"     → ask the reviewer; assume
#     case-insensitive for security relevance
#
# Complexity:
#   Time  — O(n log n)
#   Space — O(n)
# ─────────────────────────────────────────────────────────────────────────────

from functools import cmp_to_key


def _is_privileged(name: str) -> bool:
    """Check if process name is security-relevant (case-insensitive)."""
    lower = name.lower()
    return "root" in lower or "admin" in lower


# ── APPROACH 1: Naive — split into two lists, sort each, concatenate ─────────

def sort_by_relevance_naive(processes: list) -> list:
    """
    Split → sort each group → concatenate.
    Simple and clear. O(n log n).
    Mention this first; the cmp_to_key version is the required answer.
    """
    if not isinstance(processes, list):
        raise TypeError(f"Expected list, got {type(processes).__name__}")
    if not processes:
        return []

    privileged = sorted([p for p in processes if _is_privileged(p)])
    others     = sorted([p for p in processes if not _is_privileged(p)])
    return privileged + others


# ── APPROACH 2: cmp_to_key comparator (as problem requires) ──────────────────

def sort_by_relevance(processes: list) -> list:
    """
    Custom sort using functools.cmp_to_key.

    Comparator rules:
      1. Privileged names come before non-privileged.
      2. Within same group, sort alphabetically.

    Time:  O(n log n)
    Space: O(n)
    """
    if not isinstance(processes, list):
        raise TypeError(f"Expected list, got {type(processes).__name__}")
    if not processes:
        return []

    for i, p in enumerate(processes):
        if not isinstance(p, str):
            raise TypeError(f"Process at index {i} must be a string, got {type(p).__name__}")

    def cmp(a: str, b: str) -> int:
        a_priv = _is_privileged(a)
        b_priv = _is_privileged(b)

        if a_priv and not b_priv:
            return -1                       # a (privileged) comes first
        if not a_priv and b_priv:
            return 1                        # b (privileged) comes first
        # Both in same group — alphabetical
        if a < b:
            return -1
        if a > b:
            return 1
        return 0

    return sorted(processes, key=cmp_to_key(cmp))


# ── APPROACH 3: Key-based sort (simpler, often preferred over cmp_to_key) ─────

def sort_by_relevance_key(processes: list) -> list:
    """
    Key tuple: (group_rank, name)
      group_rank 0 = privileged (comes first)
      group_rank 1 = others

    "I'd normally prefer this over cmp_to_key — cleaner and faster because
    the key is computed O(n) times rather than the comparator being called
    O(n log n) times."

    Time:  O(n log n)
    Space: O(n)
    """
    if not isinstance(processes, list):
        raise TypeError(f"Expected list, got {type(processes).__name__}")
    if not processes:
        return []

    return sorted(processes, key=lambda p: (0 if _is_privileged(p) else 1, p.lower()))


# ── MANUAL — no sorted(), no cmp_to_key, manual merge sort ───────────────────

def sort_by_relevance_manual(processes: list) -> list:
    """
    Manual stable merge sort with the same two-tier comparator.

    Time:  O(n log n)
    Space: O(n)
    """
    if processes is None:
        raise ValueError("processes cannot be None")

    def comes_before(a: str, b: str) -> bool:
        """True if a should come before b."""
        a_priv = _is_privileged(a)
        b_priv = _is_privileged(b)
        if a_priv and not b_priv:
            return True
        if not a_priv and b_priv:
            return False
        return a.lower() < b.lower()

    def merge(left, right):
        result = []
        i = j = 0
        while i < len(left) and j < len(right):
            if comes_before(left[i], right[j]) or left[i].lower() == right[j].lower():
                result.append(left[i])
                i += 1
            else:
                result.append(right[j])
                j += 1
        while i < len(left):
            result.append(left[i])
            i += 1
        while j < len(right):
            result.append(right[j])
            j += 1
        return result

    def mergesort(arr):
        if len(arr) <= 1:
            return list(arr)
        mid = len(arr) // 2
        return merge(mergesort(arr[:mid]), mergesort(arr[mid:]))

    return mergesort(processes)


# ── Tests ─────────────────────────────────────────────────────────────────────

if __name__ == "__main__":
    processes = [
        "nginx",
        "root_shell",
        "python3",
        "admin_panel",
        "sshd",
        "rootkit_scanner",
        "cron",
        "administrator",
    ]

    result = sort_by_relevance(processes)

    # Privileged names must come first
    priv_names = [p for p in result if _is_privileged(p)]
    other_names = [p for p in result if not _is_privileged(p)]

    assert result.index(priv_names[-1]) < result.index(other_names[0]), \
        "All privileged names must precede all others"

    # Privileged group alphabetically sorted
    assert priv_names == sorted(priv_names, key=str.lower)

    # Others group alphabetically sorted
    assert other_names == sorted(other_names, key=str.lower)

    # Naive and key-based agree with cmp_to_key
    assert sort_by_relevance_naive(processes) == result
    assert sort_by_relevance_key(processes)   == result

    # Edge cases
    assert sort_by_relevance([]) == []
    assert sort_by_relevance(["nginx"]) == ["nginx"]
    assert sort_by_relevance(["root"])  == ["root"]

    # All privileged
    all_priv = ["root_server", "admin_tool", "rootkit"]
    r = sort_by_relevance(all_priv)
    assert r == sorted(all_priv, key=str.lower)

    print("Q11 — All assertions passed.")
    print("\nSorted processes:")
    for p in result:
        tag = " ← privileged" if _is_privileged(p) else ""
        print(f"  {p}{tag}")
