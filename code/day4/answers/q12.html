# Q12 - Find Two Numbers Summing to Target in Sorted Array (Two Pointer)
# Given a sorted list of integers, find indices of the two numbers that sum
# to the target. Use two pointers — not a hash map.
#
# ─── SOLUTION WALKTHROUGH ───────────────────────────────────────────────────
#
# STEP 1 — Clarify
#   "Are indices 1-based or 0-based? Can the same element be used twice?
#    Is there guaranteed to be exactly one solution? Can the array have
#    duplicates?"
#
# STEP 2 — Naive approach (hash map — O(n) time, O(n) space)
#   "My first instinct: iterate, for each element check if (target - element)
#    is in a hash set. O(n) time, O(n) space. Correct, but ignores sorted."
#
# STEP 3 — Optimize with two pointers (O(n) time, O(1) space)
#   "Since the array is sorted, place one pointer at the start (lo) and one
#    at the end (hi):
#    - If arr[lo] + arr[hi] == target → found it.
#    - If sum < target → need larger sum → move lo right.
#    - If sum > target → need smaller sum → move hi left.
#    We never need to revisit a pair. Total O(n) with O(1) space."
#
# Why does this work?
#   "For any fixed lo, if arr[lo] + arr[hi] > target then arr[hi] is too large
#    for ALL remaining candidates to the left of hi — no need to check them
#    with this hi. Symmetrically for lo."
#
# STEP 4 — Contrast with hash map
#   "Hash map: O(n) time, O(n) space — doesn't require sorted input.
#    Two pointer: O(n) time, O(1) space — requires sorted input.
#    Prefer two pointer when memory is constrained and array is sorted."
#
# STEP 5 — Edge cases
#   - No valid pair exists            → return [] (or [-1, -1])
#   - Array with exactly 2 elements   → only one candidate pair
#   - Negative numbers                → two pointer still works
#   - Target is sum of first+last     → found immediately
#   - Same value at different indices → valid pair (not same element twice)
#
# Complexity:
#   Time  — O(n)
#   Space — O(1)
# ─────────────────────────────────────────────────────────────────────────────


# ── APPROACH 1: Hash map (mention first, then improve) ───────────────────────

def two_sum_hashmap(arr: list, target: int) -> list:
    """
    Hash map approach — does NOT require sorted input.

    Time:  O(n)
    Space: O(n)
    """
    if not isinstance(arr, list):
        raise TypeError(f"Expected list, got {type(arr).__name__}")
    if len(arr) < 2:
        return []

    seen = {}                           # value → index
    for i, val in enumerate(arr):
        complement = target - val
        if complement in seen:
            return [seen[complement], i]
        seen[val] = i

    return []


# ── APPROACH 2: Two pointer (optimal — O(n) time, O(1) space) ────────────────

def two_sum_two_pointer(arr: list, target: int) -> list:
    """
    Two pointer on sorted array.

    Time:  O(n)
    Space: O(1)
    """
    if not isinstance(arr, list):
        raise TypeError(f"Expected list, got {type(arr).__name__}")
    if len(arr) < 2:
        return []

    lo, hi = 0, len(arr) - 1

    while lo < hi:
        current_sum = arr[lo] + arr[hi]

        if current_sum == target:
            return [lo, hi]
        elif current_sum < target:
            lo += 1                     # need a larger sum
        else:
            hi -= 1                     # need a smaller sum

    return []                           # no pair found


# ── APPROACH 3: Binary search variant (O(n log n)) ───────────────────────────
# "For each element, binary search for its complement in the rest of the array.
#  O(n log n). Worse than two pointer for a sorted array, but shows
#  binary search thinking."

import bisect

def two_sum_binary_search(arr: list, target: int) -> list:
    """
    For each index i, binary search for (target - arr[i]) in arr[i+1:].

    Time:  O(n log n)
    Space: O(1)
    """
    if not isinstance(arr, list):
        raise TypeError(f"Expected list, got {type(arr).__name__}")
    if len(arr) < 2:
        return []

    for i in range(len(arr) - 1):
        complement = target - arr[i]
        # Search in arr[i+1:]
        lo, hi = i + 1, len(arr) - 1
        while lo <= hi:
            mid = lo + (hi - lo) // 2
            if arr[mid] == complement:
                return [i, mid]
            elif arr[mid] < complement:
                lo = mid + 1
            else:
                hi = mid - 1

    return []


# ── MANUAL — no bisect, no enumerate, no dict ────────────────────────────────

def two_sum_manual(arr: list, target: int) -> list:
    """
    Pure two pointer without any imports or built-in helpers.

    Time:  O(n)
    Space: O(1)
    """
    if arr is None:
        raise ValueError("arr cannot be None")
    n = len(arr)
    if n < 2:
        return []

    lo = 0
    hi = n - 1

    while lo < hi:
        s = arr[lo] + arr[hi]
        if s == target:
            return [lo, hi]
        elif s < target:
            lo += 1
        else:
            hi -= 1

    return []

if __name__ == "__main__":
    # Problem example
    assert two_sum_two_pointer([2, 7, 11, 15], 9) == [0, 1]

    # Negative numbers
    assert two_sum_two_pointer([-3, -1, 0, 2, 4], 1) == [1, 4]  # -1 + 2 = 1... no: 2+(-1)=1? no. -1+2=1 yes. wait -3+4=1, also -1+2=1. first pair lo=0 sum=-3+4=1... actually let me recalculate: [-3,-1,0,2,4] target=1. lo=0(-3), hi=4(4): -3+4=1 → [0,4]
    assert two_sum_two_pointer([-3, -1, 0, 2, 4], 1) == [0, 4]  # -3 + 4 = 1

    # Target at boundaries
    arr2 = [1, 2, 3, 4, 6]
    assert two_sum_two_pointer(arr2, 7) == [1, 4]   # 2 + ... wait: 1+6=7→[0,4]? Let's check: lo=0(1),hi=4(6):1+6=7→[0,4]
    assert two_sum_two_pointer(arr2, 7) == [0, 4]   # 1 + 6 = 7

    # No pair
    assert two_sum_two_pointer([1, 2, 3], 10) == []

    # Two-element array
    assert two_sum_two_pointer([3, 7], 10) == [0, 1]
    assert two_sum_two_pointer([3, 7], 5)  == []

    # All approaches agree
    test_cases = [
        ([2, 7, 11, 15], 9),
        ([-3, -1, 0, 2, 4], 1),
        ([1, 2, 3], 10),
        ([3, 7], 10),
    ]
    for arr_t, tgt in test_cases:
        tp  = two_sum_two_pointer(arr_t, tgt)
        hm  = two_sum_hashmap(arr_t, tgt)
        bs  = two_sum_binary_search(arr_t, tgt)
        man = two_sum_manual(arr_t, tgt)
        assert tp == hm  == bs == man, \
            f"Mismatch on arr={arr_t}, target={tgt}: tp={tp}, hm={hm}, bs={bs}, man={man}"

    print("Q12 — All assertions passed.")
    print(f"two_sum_two_pointer([2,7,11,15], 9) = {two_sum_two_pointer([2,7,11,15], 9)}")  # [0,1]
    print()
    print("Complexity comparison:")
    print("  Hash map:      O(n) time, O(n) space — works on unsorted arrays")
    print("  Two pointer:   O(n) time, O(1) space — requires sorted input")
    print("  Binary search: O(n log n) time, O(1) space — not optimal here")
