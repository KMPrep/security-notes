# Q13 - Complexity Analysis: Log Deduplication
# Analyze the time and space complexity of the given dedup_logs function.
# Could you make it O(1) space? What's the tradeoff?
#
# ─── SOLUTION WALKTHROUGH ───────────────────────────────────────────────────
#
# Given code to analyze:
#
#   def dedup_logs(logs):
#       seen = set()
#       result = []
#       for log in logs:
#           if log not in seen:
#               seen.add(log)
#               result.append(log)
#       return result
#
# ─────────────────────────────────────────────────────────────────────────────
#
# TIME COMPLEXITY — T(n):
#   - The for loop iterates once over each log: O(n) iterations.
#   - `log not in seen`: set membership test is O(1) average
#     (hash the log string, look up bucket).
#     Worst case O(k) where k = string length (for hashing), but we treat
#     each log entry as a unit → O(1) amortized.
#   - `seen.add(log)` and `result.append(log)`: both O(1) amortized.
#   Total: O(n)
#
# SPACE COMPLEXITY — S(n):
#   - `seen` set: in the worst case (all logs unique) stores n entries → O(n)
#   - `result` list: in the worst case stores n entries → O(n)
#   - Both are O(n). We cannot avoid O(n) space with this approach since we
#     MUST remember which logs we've seen.
#   Total: O(n)
#
# CAN WE MAKE IT O(1) SPACE?
#   "Yes, but only if we're allowed to mutate / sort the input:"
#
#   Option A — Sort in-place, then deduplicate with one pass:
#     - Sort: O(n log n) time, O(1) extra space (in-place sort)
#     - Dedup pass: O(n) time
#     - BUT: this changes the ORDER of output (no longer original order).
#       Also: the sort itself uses O(log n) stack space in Python's Timsort.
#
#   Option B — Sort + dedup in-place (modify original list):
#     - O(n log n) time, O(log n) stack
#     - Loses original insertion order.
#
#   TRADEOFF:
#     O(n) space solution → preserves original order, O(n) time.
#     "O(1)" space solution → destroys order, O(n log n) time, mutates input.
#     "For a log pipeline where arrival order matters (it always does for
#      correlation), sacrificing order is unacceptable. Use the O(n) space
#      version."
#
# SECURITY NOTE:
#   Hash sets are vulnerable to hash flooding attacks if the hash function
#   is predictable. An adversary could craft log entries that all hash to
#   the same bucket, degrading set operations from O(1) to O(n).
#   Python 3.3+ uses hash randomization (PYTHONHASHSEED) by default to
#   mitigate this.
#
# ─────────────────────────────────────────────────────────────────────────────


# ── Original function (as given) ─────────────────────────────────────────────

def dedup_logs(logs):
    """
    Deduplicate log entries while preserving first-seen order.

    Time:  O(n)   — one pass, O(1) amortized set/list ops
    Space: O(n)   — seen set + result list, worst case n unique entries
    """
    seen = set()
    result = []
    for log in logs:
        if log not in seen:
            seen.add(log)
            result.append(log)
    return result


# ── VARIANT A: O(1) extra space — sort then dedup (loses order) ──────────────

def dedup_logs_sorted(logs: list) -> list:
    """
    Sorted dedup — O(1) extra space but LOSES original insertion order.

    1. Sort the list in-place: O(n log n)
    2. Walk and skip consecutive duplicates: O(n)
    Total Time:  O(n log n)
    Total Space: O(log n) — Timsort's stack; O(1) data structures
    """
    if not logs:
        return []
    logs.sort()                         # mutates input — warn the reviewer
    result = [logs[0]]
    for i in range(1, len(logs)):
        if logs[i] != logs[i - 1]:
            result.append(logs[i])
    return result


# ── VARIANT B: Manual dedup without set() or any stdlib ──────────────────────

def dedup_logs_manual(logs: list) -> list:
    """
    Manual dedup: same O(n) algorithm but using a manual hash table
    (open-addressing) instead of Python's built-in set.
    Illustrates what happens under the hood.

    Time:  O(n)   amortized
    Space: O(n)
    """
    if logs is None:
        raise ValueError("logs cannot be None")
    if not logs:
        return []

    # Simple open-addressing hash table backed by a fixed-size list
    # We resize when load factor > 0.7 to keep O(1) amortized ops
    TABLE_INIT_SIZE = 16

    def _make_table(size):
        return [None] * size

    def _hash(s, size):
        # djb2-style hash
        h = 5381
        for ch in s:
            h = (h * 33 + ord(ch)) & 0xFFFFFFFF
        return h % size

    def _insert(table, key):
        """Returns True if key was newly inserted, False if already present."""
        size = len(table)
        idx  = _hash(key, size)
        while table[idx] is not None:
            if table[idx] == key:
                return False            # already in table
            idx = (idx + 1) % size     # linear probing
        table[idx] = key
        return True

    table    = _make_table(TABLE_INIT_SIZE)
    count    = 0
    capacity = TABLE_INIT_SIZE
    result   = []

    for log in logs:
        # Resize if load factor > 0.7
        if count >= capacity * 0.7:
            new_cap   = capacity * 2
            new_table = _make_table(new_cap)
            for entry in table:
                if entry is not None:
                    _insert(new_table, entry)
            table    = new_table
            capacity = new_cap

        if _insert(table, log):
            count += 1
            result.append(log)

    return result

if __name__ == "__main__":
    logs = [
        "user=alice action=login",
        "user=bob   action=login",
        "user=alice action=login",   # duplicate
        "user=carol action=scan",
        "user=bob   action=login",   # duplicate
        "user=alice action=logout",
    ]

    expected = [
        "user=alice action=login",
        "user=bob   action=login",
        "user=carol action=scan",
        "user=alice action=logout",
    ]

    # Original function
    assert dedup_logs(logs) == expected

    # Manual dedup (preserves order)
    assert dedup_logs_manual(logs) == expected

    # Edge cases
    assert dedup_logs([])       == []
    assert dedup_logs(["a"])    == ["a"]
    assert dedup_logs(["a","a","a"]) == ["a"]
    assert dedup_logs_manual(["a","b","a"]) == ["a","b"]

    # Sorted variant (order changes — just check dedup correctness)
    logs_copy = list(logs)
    deduped_sorted = dedup_logs_sorted(logs_copy)
    assert len(deduped_sorted) == len(expected)
    assert set(deduped_sorted) == set(expected)

    print("Q13 — All assertions passed.")
    print()
    print("Complexity summary:")
    print("  dedup_logs (original):  T(n) = O(n),        S(n) = O(n)")
    print("  dedup_logs_sorted:      T(n) = O(n log n),  S(n) = O(log n)  [loses order]")
    print("  dedup_logs_manual:      T(n) = O(n) amort., S(n) = O(n)")
