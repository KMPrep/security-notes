# Q14 - Complexity Analysis: Nested Loop Alert Correlation
# Analyze the time complexity of correlate(alerts, rules).
# How would you improve it? Propose an O(n + m) alternative.
#
# ─── SOLUTION WALKTHROUGH ───────────────────────────────────────────────────
#
# Given code to analyze:
#
#   def correlate(alerts, rules):
#       matches = []
#       for alert in alerts:           # n iterations
#           for rule in rules:         # m iterations
#               if rule in alert:      # substring search: O(|alert|) worst
#                   matches.append((alert, rule))
#       return matches
#
# ─────────────────────────────────────────────────────────────────────────────
#
# TIME COMPLEXITY — T(n, m):
#
#   - Outer loop: n iterations (n = len(alerts))
#   - Inner loop: m iterations (m = len(rules))
#   - `rule in alert`: substring search using Python's default algorithm.
#     For a rule of length r and alert of length a: O(a * r) worst case
#     (or O(a + r) with KMP/Boyer-Moore, but Python uses a simple scan).
#
#   If we treat each string comparison as O(1) (unit-cost model):
#     T(n, m) = O(n * m)
#
#   If we model string lengths explicitly (let A = avg alert length, R = avg rule length):
#     T(n, m) = O(n * m * A)   [using naive substring match]
#
#   "For a system with 100K alerts and 10K rules, that's 10^9 ops — too slow."
#
# SPACE COMPLEXITY:
#   - matches list: worst case O(n * m) if every alert matches every rule.
#   - O(1) extra working space.
#
# HOW TO IMPROVE:
#
#   Approach 1 — Preprocess rules into a set (exact match, not substring):
#     If "rule in alert" means "alert == rule" (exact match):
#     - Build rules_set = set(rules): O(m)
#     - For each alert: O(1) set lookup → total O(n + m)
#
#   Approach 2 — Trie for prefix/substring matching:
#     Build a trie of all rules: O(m * R)
#     For each alert, scan alert tokens and check trie: O(A) per alert
#     Total: O(m*R + n*A) — much better for large n and m
#
#   Approach 3 — Aho-Corasick automaton (multi-pattern substring search):
#     Build automaton from all rules: O(total rule chars)
#     Run each alert through automaton: O(alert length + number of matches)
#     Total: O(sum(rule lengths) + sum(alert lengths) + total matches)
#     This is the industry-standard approach for IDS/SIEM rule matching.
#
# ─────────────────────────────────────────────────────────────────────────────


# ── Original function (as given) — O(n * m) ──────────────────────────────────

def correlate(alerts, rules):
    """
    Naive nested loop.

    Time:  O(n * m) treating string ops as O(1)
           O(n * m * A) with string length factored in
    Space: O(k) where k = number of matches (output)
    """
    matches = []
    for alert in alerts:
        for rule in rules:
            if rule in alert:       # substring check
                matches.append((alert, rule))
    return matches


# ── IMPROVED: O(n + m) — exact keyword match using set ───────────────────────

def correlate_exact_set(alerts: list, rules: list) -> list:
    """
    When rules are exact tokens to match anywhere in the alert string,
    split alert into tokens and check each against a rules set.

    Build step: O(m) to create rules_set.
    Query step:
      - For each alert: tokenize O(A), check each token O(1) per token.
      - Total: O(n * A + m)
    In the unit-cost model (treating each alert as a unit): O(n + m)

    Time:  O(n + m)  unit-cost model
    Space: O(m)      for the rules set
    """
    if not isinstance(alerts, list) or not isinstance(rules, list):
        raise TypeError("alerts and rules must be lists")

    if not alerts or not rules:
        return []

    rules_set = set(rules)                  # O(m) build
    matches   = []

    for alert in alerts:
        # Tokenize the alert (split on whitespace; adjust for your log format)
        tokens = alert.split()
        for token in tokens:
            if token in rules_set:          # O(1) lookup
                matches.append((alert, token))
                # Note: could break here if we only want first match per alert

    return matches


# ── IMPROVED: O(n + m) — keyword set lookup on raw substring ─────────────────

def correlate_set(alerts: list, rules: list) -> list:
    """
    Variant: check if the FULL RULE STRING is a substring, but use a set to
    avoid redundant rule checks when rules contain duplicates.
    Still O(n * m) for substring matching, but O(n + m) for exact membership.

    For true O(n + m) substring multi-pattern matching, use Aho-Corasick.
    This version shows the set optimization for exact match:

    Time:  O(n + m)  (exact match in unit-cost model)
    Space: O(m)
    """
    if not isinstance(alerts, list) or not isinstance(rules, list):
        raise TypeError("alerts and rules must be lists")
    if not alerts or not rules:
        return []

    rules_set = set(rules)
    matches = []
    for alert in alerts:
        for rule in rules_set:              # deduplicated rules
            if rule in alert:
                matches.append((alert, rule))
    return matches


# ── MANUAL — no set(), explicit hash table ────────────────────────────────────

def correlate_manual(alerts: list, rules: list) -> list:
    """
    Manual dict-based lookup to replace inner loop.
    Demonstrates the O(n + m) principle without built-in set.

    Time:  O(n + m)  unit-cost exact match
    Space: O(m)
    """
    if alerts is None or rules is None:
        raise ValueError("alerts and rules cannot be None")
    if not alerts or not rules:
        return []

    # Build manual hash dict: rule → True
    rules_dict = {}
    for rule in rules:
        rules_dict[rule] = True

    matches = []
    for alert in alerts:
        tokens = alert.split()
        for token in tokens:
            if token in rules_dict:
                matches.append((alert, token))

    return matches


# ── Tests ─────────────────────────────────────────────────────────────────────

if __name__ == "__main__":
    alerts = [
        "user=alice event=LOGIN src=1.1.1.1",
        "user=bob   event=SCAN  src=2.2.2.2",
        "user=carol event=LOGIN src=3.3.3.3",
        "user=alice event=EXEC  src=1.1.1.1 cmd=wget",
    ]
    rules = ["LOGIN", "SCAN", "wget"]

    # Original correlate (substring match)
    result_orig = correlate(alerts, rules)
    assert len(result_orig) == 4   # alice LOGIN, bob SCAN, carol LOGIN, alice wget

    # Set-based exact token match
    result_set = correlate_exact_set(alerts, rules)
    # "LOGIN" token should match, "SCAN" should match, "wget" should match
    assert any("LOGIN" == tok for _, tok in result_set)
    assert any("SCAN"  == tok for _, tok in result_set)
    assert any("wget"  == tok for _, tok in result_set)

    # Manual agrees with set-based
    result_manual = correlate_manual(alerts, rules)
    assert len(result_manual) == len(result_set)

    # Edge cases
    assert correlate([], rules)    == []
    assert correlate(alerts, [])   == []

    print("Q14 — All assertions passed.")
    print()
    print("Original correlate matches:")
    for alert, rule in result_orig:
        print(f"  rule={rule!r:10s}  alert={alert[:40]!r}")
    print()
    print("Complexity summary:")
    print("  Original:           T(n,m) = O(n*m),     S = O(k)")
    print("  Set exact match:    T(n,m) = O(n+m),     S = O(m)")
    print("  Aho-Corasick ideal: T(n,m) = O(n+m+k),  S = O(m*R)")
