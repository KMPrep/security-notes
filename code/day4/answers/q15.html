# Q15 - Space Complexity: Recursive vs Iterative DFS
# Compare space complexity of recursive DFS vs iterative DFS on a policy tree
# of depth d with n total nodes.
#
# ─── SOLUTION WALKTHROUGH ───────────────────────────────────────────────────
#
# STEP 1 — Set up the problem
#   "A policy tree has n nodes and depth d. We want to visit every node once
#    (DFS). Two implementations: recursive (call stack) vs iterative
#    (explicit stack)."
#
# SPACE COMPLEXITY:
#
#   Recursive DFS:
#     - Each recursive call adds one frame to the call stack.
#     - Maximum depth of recursion = depth of tree = d.
#     - Space: O(d)
#     - Python's default recursion limit is ~1000 frames. For a tree deeper
#       than ~900 nodes, this WILL raise RecursionError.
#     - sys.setrecursionlimit() can raise it, but risks segfault on deep stacks.
#
#   Iterative DFS (explicit stack):
#     - We maintain our own stack of nodes to visit.
#     - At any point, the stack holds at most all nodes on one root-to-leaf path.
#     - Stack depth = O(d) in the worst case (a path-like chain tree).
#     - BUT: unlike the call stack, the heap-allocated explicit stack can grow
#       to millions of entries without Python's recursion limit.
#     - Space: O(d)
#
#   Both are O(d) in theory. Practical difference:
#     - Recursive: limited by Python's call stack (~1000 default).
#     - Iterative: limited only by available heap memory.
#     - For wide, shallow trees (d << n): both use very little space.
#     - For degenerate chains (d = n): recursive may crash; iterative is fine.
#
# STEP 2 — When does recursive DFS risk stack overflow?
#   "When d > ~900 in Python (default sys.getrecursionlimit() = 1000).
#    Example: a linked-list-shaped tree with 10,000 nodes → d = 10,000 →
#    RecursionError."
#
# STEP 3 — Security context
#   "In a policy engine traversing an RBAC hierarchy, an adversary might craft
#    a deeply nested role chain to trigger a stack overflow (DoS). Always use
#    iterative DFS or set a depth limit when the tree structure is user-supplied."
#
# ─────────────────────────────────────────────────────────────────────────────

import sys


# ── Tree node structure ───────────────────────────────────────────────────────

class PolicyNode:
    """Simple n-ary tree node representing a policy/role."""
    __slots__ = ("name", "children")

    def __init__(self, name: str):
        if not isinstance(name, str) or not name:
            raise ValueError("Node name must be a non-empty string")
        self.name     = name
        self.children = []

    def add_child(self, child: "PolicyNode") -> "PolicyNode":
        """Fluent child adder. Returns self for chaining."""
        if not isinstance(child, PolicyNode):
            raise TypeError(f"Expected PolicyNode, got {type(child).__name__}")
        self.children.append(child)
        return self


# ── APPROACH 1: Recursive DFS ────────────────────────────────────────────────

def dfs_recursive(root: PolicyNode, max_depth: int = 900) -> list:
    """
    Recursive DFS with an explicit depth guard to prevent stack overflow.

    Space: O(d)  — call stack frames
    Time:  O(n)  — visits each node once

    max_depth: safety limit; raise ValueError if exceeded.
               Prevents adversarial deep-tree DoS.
    """
    if root is None:
        return []

    visited = []

    def _dfs(node: PolicyNode, depth: int):
        if depth > max_depth:
            raise RecursionError(
                f"Tree depth {depth} exceeds safety limit {max_depth}. "
                "Consider using iterative DFS."
            )
        visited.append(node.name)
        for child in node.children:
            _dfs(child, depth + 1)

    _dfs(root, 0)
    return visited


# ── APPROACH 2: Iterative DFS (explicit stack) ───────────────────────────────

def dfs_iterative(root: PolicyNode) -> list:
    """
    Iterative DFS using an explicit list as a stack.

    Space: O(d)  — stack holds at most d nodes (depth of deepest path)
    Time:  O(n)  — visits each node once

    No recursion limit risk. Preferred for production code and
    user-supplied / potentially adversarial tree structures.
    """
    if root is None:
        return []

    stack   = [root]
    visited = []

    while stack:
        node = stack.pop()              # LIFO → DFS order
        visited.append(node.name)
        # Push children in REVERSE order so the first child is processed first
        for child in reversed(node.children):
            stack.append(child)

    return visited


# ── APPROACH 3: Iterative DFS with depth tracking ────────────────────────────

def dfs_iterative_with_depth(root: PolicyNode, max_depth: int = None) -> list:
    """
    Iterative DFS that also tracks depth at each node.
    Allows optional depth limiting (useful for DAG/RBAC cycle detection).

    Space: O(d)
    Time:  O(n)
    """
    if root is None:
        return []

    stack   = [(root, 0)]               # (node, current_depth)
    visited = []

    while stack:
        node, depth = stack.pop()

        if max_depth is not None and depth > max_depth:
            continue                    # skip nodes beyond depth limit

        visited.append(node.name)

        for child in reversed(node.children):
            stack.append((child, depth + 1))

    return visited


# ── MANUAL — iterative DFS without any list methods except append/pop ─────────

def dfs_manual(root: PolicyNode) -> list:
    """
    Iterative DFS using a manually managed array as a stack.
    No append/pop on children stack — we use index-based management.

    Space: O(n) worst case (we pre-allocate a flat traversal buffer)
    Time:  O(n)
    """
    if root is None:
        return []

    # Manual stack: store (node, child_index) pairs
    stack_nodes = [None] * 10000        # bounded pre-allocation
    stack_top   = 0
    stack_nodes[stack_top] = root
    stack_top += 1

    visited = []

    while stack_top > 0:
        stack_top -= 1
        node = stack_nodes[stack_top]
        visited.append(node.name)
        # Push children in reverse order
        for i in range(len(node.children) - 1, -1, -1):
            if stack_top >= len(stack_nodes):
                # Grow the stack array
                stack_nodes = stack_nodes + [None] * len(stack_nodes)
            stack_nodes[stack_top] = node.children[i]
            stack_top += 1

    return visited


# ── Tests ─────────────────────────────────────────────────────────────────────

if __name__ == "__main__":
    # Build a sample policy tree:
    #       root
    #      / | \
    #    A   B   C
    #   / \      \
    #  D   E      F

    root = PolicyNode("root")
    a    = PolicyNode("A")
    b    = PolicyNode("B")
    c    = PolicyNode("C")
    d    = PolicyNode("D")
    e    = PolicyNode("E")
    f    = PolicyNode("F")

    root.add_child(a).add_child(b).add_child(c)
    a.add_child(d).add_child(e)
    c.add_child(f)

    expected_dfs = ["root", "A", "D", "E", "B", "C", "F"]

    assert dfs_recursive(root)               == expected_dfs
    assert dfs_iterative(root)               == expected_dfs
    assert dfs_iterative_with_depth(root)    == expected_dfs
    assert dfs_manual(root)                  == expected_dfs

    # Depth-limited DFS
    depth_limited = dfs_iterative_with_depth(root, max_depth=1)
    assert "D" not in depth_limited         # depth 2, pruned
    assert "root" in depth_limited

    # Deep chain — would crash recursive at depth > 900
    chain_root = PolicyNode("n0")
    cur = chain_root
    for i in range(1, 200):
        child = PolicyNode(f"n{i}")
        cur.add_child(child)
        cur = child

    # Iterative handles it fine
    iterative_result = dfs_iterative(chain_root)
    assert len(iterative_result) == 200

    # Recursive with safety limit
    try:
        dfs_recursive(chain_root, max_depth=50)
        assert False, "Should have raised"
    except RecursionError:
        pass

    print("Q15 — All assertions passed.")
    print(f"DFS order: {dfs_iterative(root)}")
    print()
    print("Space complexity summary:")
    print("  Recursive DFS:  O(d) call stack — crashes at d > ~900 (Python default)")
    print("  Iterative DFS:  O(d) heap stack — safe for any d, heap-limited only")
    print("  Practical rule: always use iterative for user-supplied trees (security)")
