# Q16 - Amortized Analysis: Dynamic Array Append
# Python lists use dynamic arrays. Explain the amortized O(1) append cost
# using the doubling strategy. Apply this to log ingestion pipelines.
#
# ─── SOLUTION WALKTHROUGH ───────────────────────────────────────────────────
#
# THE QUESTION: Why is appending n items one by one O(n) total?
#
# NAIVE ANALYSIS (wrong):
#   "Every now and then, append triggers a resize which copies the entire
#    array. That sounds like it could be O(n) per append in the worst case,
#    making n appends O(n²)."
#
# CORRECT ANALYSIS — Doubling Strategy:
#   When the array is full and we append:
#     1. Allocate a new array of 2x the current capacity.
#     2. Copy ALL existing elements to the new array: O(n) work.
#     3. Add the new element: O(1).
#
#   But resizes happen RARELY. With doubling:
#     - Resize at sizes 1, 2, 4, 8, 16, ..., n
#     - Total copy work = 1 + 2 + 4 + ... + n = 2n - 1 = O(n)
#     - Spread over n appends: O(n) / n = O(1) amortized per append.
#
#   Proof via potential function (banker's argument):
#     - Assign each element a "credit" of 2 at the time it's appended.
#     - 1 credit pays for the append itself.
#     - 1 credit is saved for the future copy when the next resize happens.
#     - When resize triggers: each element in the second half of the array
#       has at least 1 saved credit to pay for its copy.
#     - Total credits charged: 2n → amortized O(1) per append.
#
# LOG INGESTION PIPELINE APPLICATION:
#   "In a log ingestion service, we receive events in a stream and buffer them
#    in a list before flushing to storage.
#    - Using a list and appending each event is O(1) amortized — ideal for
#      high-throughput ingestion.
#    - Pre-allocating if the batch size is known: avoids ALL resize copies.
#    - Worst case for resize: a single large copy while other threads are
#      waiting — avoid by pre-allocating or using deque for O(1) appends
#      from both ends (no resizes)."
#
# STEP — Edge cases / follow-ups
#   - What if we append and then pop? Hysteresis: only shrink when size < cap/4
#     to avoid O(n) thrashing on alternating push/pop.
#   - Python's list uses ~1.125x growth factor (not exactly 2x) for memory
#     efficiency. The amortized argument holds for any growth factor > 1.
#
# ─────────────────────────────────────────────────────────────────────────────


# ── Demonstration: track resize events ───────────────────────────────────────

import sys

def demonstrate_list_growth(n: int) -> None:
    """
    Append n items one by one and record when the underlying buffer grows.
    Uses sys.getsizeof to detect capacity changes.

    Time:  O(n) total
    Space: O(n) for the list
    """
    if not isinstance(n, int) or n <= 0:
        raise ValueError(f"n must be a positive integer, got {n!r}")

    arr       = []
    prev_size = sys.getsizeof(arr)
    resizes   = 0

    print(f"{'Index':>6}  {'List Size (bytes)':>18}  {'Resize?':>8}")
    print("-" * 40)

    for i in range(n):
        arr.append(i)
        cur_size = sys.getsizeof(arr)
        resized  = "YES <--" if cur_size > prev_size else ""
        if resized or i < 5 or i == n - 1:
            print(f"{i:>6}  {cur_size:>18}  {resized}")
        if cur_size > prev_size:
            resizes += 1
        prev_size = cur_size

    print(f"\nTotal resizes for {n} appends: {resizes}")
    print(f"log2({n}) ≈ {n.bit_length() - 1}  (resizes are O(log n))")


# ── MANUAL dynamic array with explicit doubling ───────────────────────────────

class DynamicArray:
    """
    Manual dynamic array implementation using a Python list as a backing store.
    Demonstrates the doubling strategy explicitly.

    All amortized O(1) operations:
      - append: O(1) amortized
      - get:    O(1)
      - len:    O(1)
    """

    def __init__(self, initial_capacity: int = 1):
        if initial_capacity < 1:
            raise ValueError("initial_capacity must be >= 1")
        self._data     = [None] * initial_capacity
        self._size     = 0
        self._capacity = initial_capacity
        self._resizes  = 0

    def append(self, value) -> None:
        """Amortized O(1) append using doubling strategy."""
        if self._size == self._capacity:
            self._resize(self._capacity * 2)
        self._data[self._size] = value
        self._size += 1

    def _resize(self, new_capacity: int) -> None:
        """Double the backing array. O(n) work, called O(log n) total times."""
        new_data = [None] * new_capacity
        for i in range(self._size):
            new_data[i] = self._data[i]
        self._data     = new_data
        self._capacity = new_capacity
        self._resizes += 1

    def get(self, index: int):
        """O(1) random access."""
        if not (0 <= index < self._size):
            raise IndexError(f"Index {index} out of range [0, {self._size})")
        return self._data[index]

    def __len__(self) -> int:
        return self._size

    def __repr__(self) -> str:
        return f"DynamicArray({[self._data[i] for i in range(self._size)]})"


# ── Log ingestion buffer using dynamic array ─────────────────────────────────

class LogIngestionBuffer:
    """
    Buffer for streaming log events.
    Uses amortized O(1) append for high-throughput ingestion.
    Flushes in bulk when buffer reaches flush_size.
    """

    def __init__(self, flush_size: int = 1000):
        if flush_size < 1:
            raise ValueError("flush_size must be >= 1")
        self._buffer     = []
        self._flush_size = flush_size
        self._flushed    = 0

    def ingest(self, event: str) -> bool:
        """
        Ingest a single log event. Returns True if flush was triggered.
        Time: O(1) amortized
        """
        if not isinstance(event, str):
            raise TypeError(f"Event must be a string, got {type(event).__name__}")
        self._buffer.append(event)
        if len(self._buffer) >= self._flush_size:
            self.flush()
            return True
        return False

    def flush(self) -> list:
        """Flush buffered events. Returns the flushed batch."""
        batch        = self._buffer[:]
        self._buffer = []
        self._flushed += len(batch)
        return batch

    @property
    def pending(self) -> int:
        return len(self._buffer)

    @property
    def total_flushed(self) -> int:
        return self._flushed
