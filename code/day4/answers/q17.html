# Q17 - Explain: Why is dict lookup O(1)?
# Give a clear verbal explanation of hash tables, collision handling,
# worst-case O(n), and hash flooding attacks.
#
# ─── SOLUTION WALKTHROUGH ───────────────────────────────────────────────────
#
# HOW DICT LOOKUP WORKS:
#
# 1. Hash function: key → integer
#    "When you look up d[key], Python calls hash(key) which maps the key to a
#     large integer. This is O(k) where k = size of key (e.g., length of string).
#     We treat this as O(1) in the unit-cost model."
#
# 2. Bucket index: hash(key) % table_size
#    "The hash integer is reduced mod the table size to get an array index.
#     The table is a fixed-size array of buckets."
#
# 3. Direct access: O(1)
#    "We index directly into the array at that position — O(1)."
#
# 4. Collision handling:
#    Two keys can map to the same bucket. Python uses OPEN ADDRESSING
#    (specifically a variation of quadratic probing / perturbation).
#    Other schemes: separate chaining (linked lists per bucket).
#    "On collision, probe adjacent slots until an empty one is found.
#     Average probe length is O(1) when the load factor is < 0.7."
#
# 5. Load factor and resizing:
#    "When len(d) / table_size > 2/3 (Python's threshold), the table is
#     resized (doubled). Insertion triggers a resize and a rehash of all keys.
#     Amortized O(1) per insertion."
#
# WORST CASE O(n):
#    "If all n keys hash to the same bucket, every lookup degenerates to a
#     linear scan through the collision chain. O(n) per operation."
#
# HASH FLOODING ATTACK (security context):
#    "An adversary who knows the hash function can craft n strings that all
#     hash to the same bucket, turning every dict operation into O(n).
#     This is a denial-of-service attack on web services that store user input
#     in dicts (e.g., HTTP request parameters parsed into a dict)."
#
# PYTHON's DEFENSE:
#    "Python 3.3+ uses hash randomization: PYTHONHASHSEED environment variable
#     seeds the hash function with a random secret at interpreter startup.
#     This means an attacker can't precompute colliding keys across runs.
#     Caveat: doesn't help if seed is leaked (e.g., predictable seeds in tests)."
#
# PYTHON 3.6+ DICTS:
#    "Python 3.6+ uses a compact hash table that preserves insertion order
#     by storing (key, value) pairs in a separate dense array and using the
#     hash table only for indexing. Memory-efficient and ordered."
#
# ─────────────────────────────────────────────────────────────────────────────


# ── Manual hash table: separate chaining ─────────────────────────────────────

class SimpleHashTable:
    """
    Hash table with separate chaining (list of (key, value) per bucket).
    Illustrates the O(1) average / O(n) worst-case lookup principle.
    """

    def __init__(self, initial_capacity: int = 16):
        if initial_capacity < 1:
            raise ValueError("initial_capacity must be >= 1")
        self._capacity  = initial_capacity
        self._buckets   = [[] for _ in range(self._capacity)]
        self._size      = 0
        self._load_max  = 0.7

    def _bucket_index(self, key) -> int:
        """Compute bucket index from key hash."""
        return hash(key) % self._capacity

    def set(self, key, value) -> None:
        """Insert or update key. Amortized O(1)."""
        idx    = self._bucket_index(key)
        bucket = self._buckets[idx]

        for i, (k, _) in enumerate(bucket):
            if k == key:
                bucket[i] = (key, value)    # update existing
                return

        bucket.append((key, value))         # new entry
        self._size += 1

        if self._size / self._capacity > self._load_max:
            self._resize()

    def get(self, key, default=None):
        """
        Look up key. Average O(1), worst case O(n) with all keys in one bucket.
        """
        idx    = self._bucket_index(key)
        bucket = self._buckets[idx]

        for k, v in bucket:
            if k == key:
                return v

        return default

    def __contains__(self, key) -> bool:
        return self.get(key, _SENTINEL) is not _SENTINEL

    def _resize(self) -> None:
        """Double capacity and rehash all entries. O(n) total."""
        new_capacity = self._capacity * 2
        new_buckets  = [[] for _ in range(new_capacity)]

        for bucket in self._buckets:
            for k, v in bucket:
                new_idx = hash(k) % new_capacity
                new_buckets[new_idx].append((k, v))

        self._capacity = new_capacity
        self._buckets  = new_buckets

    def __len__(self) -> int:
        return self._size

    def bucket_lengths(self) -> list:
        """Return the length of each bucket (for collision visualization)."""
        return [len(b) for b in self._buckets]

    def max_chain_length(self) -> int:
        """Worst-case lookup depth = longest chain."""
        return max(self.bucket_lengths())


_SENTINEL = object()


# ── Demonstration: hash flooding simulation ───────────────────────────────────

def demonstrate_collision_attack() -> None:
    """
    Simulate hash flooding with a fixed-seed hash function.

    NOTE: Python's PYTHONHASHSEED randomizes hashes at startup, so this
    demo uses a custom (insecure) hash function to show the collision effect.
    """
    def bad_hash(s: str, capacity: int) -> int:
        """Intentionally weak hash: sum of char codes mod capacity."""
        return sum(ord(c) for c in s) % capacity

    CAPACITY = 8
    buckets  = [[] for _ in range(CAPACITY)]

    # Craft keys that all hash to bucket 0 with this weak hash
    # (keys whose char sum ≡ 0 mod 8)
    crafted_keys = ["hh", "ip", "jo", "kn", "lm", "qb", "ra", "sz"]
    for key in crafted_keys:
        idx = bad_hash(key, CAPACITY)
        buckets[idx].append(key)

    max_chain = max(len(b) for b in buckets)
    print(f"  Crafted {len(crafted_keys)} keys into {CAPACITY} buckets.")
    print(f"  Longest collision chain: {max_chain}")
    print(f"  Each lookup in that chain costs O({max_chain}) instead of O(1).")


# ── Tests ─────────────────────────────────────────────────────────────────────

if __name__ == "__main__":
    ht = SimpleHashTable()

    # Basic set/get
    ht.set("alice", 42)
    ht.set("bob",   17)
    ht.set("carol", 99)

    assert ht.get("alice") == 42
    assert ht.get("bob")   == 17
    assert ht.get("carol") == 99
    assert ht.get("dan")   is None       # missing key returns default

    # Update existing key
    ht.set("alice", 100)
    assert ht.get("alice") == 100

    # Contains
    assert "alice" in ht
    assert "dan"   not in ht

    # Size
    assert len(ht) == 3

    # Force resize by adding many keys
    for i in range(50):
        ht.set(f"key_{i}", i)
    assert len(ht) == 53

    # After resize, all keys still accessible
    for i in range(50):
        assert ht.get(f"key_{i}") == i

    print("Q17 — All assertions passed.")
    print()
    print("=== Why dict lookup is O(1) ===")
    print("  1. hash(key)        → integer                     O(k) ≈ O(1)")
    print("  2. hash % capacity  → bucket index                O(1)")
    print("  3. arr[idx]         → direct array access         O(1)")
    print("  4. chain scan       → average O(1), worst O(n)    depends on load")
    print()
    print("=== Hash flooding attack demo ===")
    demonstrate_collision_attack()
    print()
    print("=== Python 3.6+ dict features ===")
    print("  - Compact hash table: (key,val) stored in dense array; hash table holds indices")
    print("  - Insertion-ordered (language spec since 3.7)")
    print("  - PYTHONHASHSEED randomization since 3.3 (mitigates flooding)")
    print()
    print(f"Max chain length in our test table: {ht.max_chain_length()}")
