# Q18 - Binary Search on Answer: Minimum Scan Threshold
# Find the minimum alert threshold T such that the number of alerts with
# score >= T is exactly <= K. Given sorted scores, find T using binary search
# on the answer (value space), not on the index space.
#
# ─── SOLUTION WALKTHROUGH ───────────────────────────────────────────────────
#
# STEP 1 — Clarify
#   "We want the MINIMUM T such that count(score >= T) <= K.
#    Since lowering T includes more alerts (count goes up), and raising T
#    includes fewer, we want the smallest T where count is still within K.
#    Scores are sorted ascending. Can scores be negative? Can K be 0?"
#
# STEP 2 — Naive approach
#   "Try every unique score value as a candidate T. For each, count how many
#    scores are >= T. O(n²) if we do a linear count for each of n candidates."
#
# STEP 3 — Binary search on the INDEX space
#   "Since scores are sorted, for a given T I can binary search (bisect_left)
#    to find the first index where score >= T.
#    Count = len(scores) - that_index.
#    But I still need to try each value of T — that's the outer search."
#
# STEP 4 — Binary search on the VALUE space (the key insight)
#   "The answer T lives somewhere in the range [scores[0], scores[-1]].
#    If I try T = mid_value:
#      - count(score >= mid) using bisect_left = n - bisect_left(scores, mid)
#      - If count <= K: T = mid might work, but maybe smaller T also works
#        → try lower (hi = mid)
#      - If count > K: T = mid is too low (too many alerts) → go higher (lo = mid+1)
#    This is binary search on the VALUE space.
#    Time: O(log(max_score - min_score) * log n)"
#
# STEP 5 — Simpler interpretation: binary search on INDEX
#   "Actually, the cleaner approach: we want index p (0-based from left) such
#    that scores[p] is the K-th largest score from the right.
#    That index is n - K (0-based). T = scores[n - K].
#    This is O(1) after O(log n) verification."
#
# STEP 6 — Edge cases
#   - K >= n: T = scores[0] (all alerts qualify, threshold is minimum)
#   - K == 0: T = scores[-1] + 1 (no alert qualifies)
#   - K == 1: T = scores[-1] (only max qualifies)
#   - All scores equal: T = that score
#   - Duplicates: count(score >= T) may jump
#
# Complexity:
#   Index-based: O(1) lookup after sorted input
#   Value binary search: O(log(V) * log n) where V = value range
# ─────────────────────────────────────────────────────────────────────────────

import bisect


# ── APPROACH 1: Naive — try every value as T ─────────────────────────────────

def min_threshold_naive(scores: list, k: int) -> int:
    """
    Try each unique score as a candidate threshold.
    For each T, count alerts >= T linearly. O(n²) time.
    """
    if not isinstance(scores, list) or not isinstance(k, int):
        raise TypeError("scores must be list, k must be int")
    if k < 0:
        raise ValueError(f"k must be >= 0, got {k}")
    if not scores:
        raise ValueError("scores must not be empty")
    if k >= len(scores):
        return scores[0]        # all alerts qualify

    n = len(scores)
    # Try each unique score as T, sorted in ascending order
    candidates = sorted(set(scores))
    best_T = scores[-1] + 1     # sentinel: no alerts qualify

    for T in candidates:
        count = sum(1 for s in scores if s >= T)
        if count <= k:
            best_T = T
            break               # first (smallest) T that satisfies count <= k

    return best_T


# ── APPROACH 2: Index-based O(1) answer ──────────────────────────────────────

def min_threshold_index(scores: list, k: int) -> int:
    """
    Direct index calculation.

    Sorted ascending: scores[n-k] is the (n-k)-th smallest = k-th largest.
    All scores at index >= n-k are >= scores[n-k].
    Count of scores >= scores[n-k] is exactly k (or more if duplicates).

    Time:  O(1) given sorted input
    Space: O(1)
    """
    if not isinstance(scores, list) or not isinstance(k, int):
        raise TypeError("scores must be list, k must be int")
    if k < 0:
        raise ValueError(f"k must be >= 0, got {k}")
    if not scores:
        raise ValueError("scores must not be empty")

    n = len(scores)
    if k == 0:
        return scores[-1] + 1       # threshold above max → 0 alerts qualify
    if k >= n:
        return scores[0]            # all alerts qualify

    return scores[n - k]            # k-th largest score from the right


# ── APPROACH 3: Binary search on value space ─────────────────────────────────

def min_threshold_bsearch(scores: list, k: int) -> int:
    """
    Binary search on the value range [min_score, max_score+1].
    Counts are monotone: higher T → fewer alerts above T.

    Find smallest T such that count(score >= T) <= k.

    Time:  O(log(V) * log n)  where V = max - min value range
    Space: O(1)
    """
    if not isinstance(scores, list) or not isinstance(k, int):
        raise TypeError("scores must be list, k must be int")
    if k < 0:
        raise ValueError(f"k must be >= 0, got {k}")
    if not scores:
        raise ValueError("scores must not be empty")

    n = len(scores)
    if k >= n:
        return scores[0]

    def count_gte(T: int) -> int:
        """Number of scores >= T using binary search."""
        idx = bisect.bisect_left(scores, T)
        return n - idx

    lo = scores[0]
    hi = scores[-1] + 1     # exclusive upper bound (ensures count = 0 at hi)
    result = hi

    while lo <= hi:
        mid = lo + (hi - lo) // 2
        if count_gte(mid) <= k:
            result = mid        # valid T; try smaller
            hi = mid - 1
        else:
            lo = mid + 1        # too many alerts; raise T

    return result


# ── MANUAL — no bisect ────────────────────────────────────────────────────────

def min_threshold_manual(scores: list, k: int) -> int:
    """
    Binary search on value space without bisect.
    Manual count_gte using binary search.

    Time:  O(log(V) * log n)
    Space: O(1)
    """
    if scores is None:
        raise ValueError("scores cannot be None")
    n = len(scores)
    if n == 0:
        raise ValueError("scores must not be empty")
    if k < 0:
        raise ValueError(f"k must be >= 0, got {k}")
    if k >= n:
        return scores[0]

    def count_gte(T: int) -> int:
        """Binary search for first index >= T."""
        lo_i, hi_i = 0, n
        while lo_i < hi_i:
            mid_i = lo_i + (hi_i - lo_i) // 2
            if scores[mid_i] < T:
                lo_i = mid_i + 1
            else:
                hi_i = mid_i
        return n - lo_i

    lo = scores[0]
    hi = scores[-1] + 1
    result = hi

    while lo <= hi:
        mid = lo + (hi - lo) // 2
        if count_gte(mid) <= k:
            result = mid
            hi = mid - 1
        else:
            lo = mid + 1

    return result


# ── Tests ─────────────────────────────────────────────────────────────────────

if __name__ == "__main__":
    scores = [1, 3, 5, 7, 9, 11]

    # Problem example: K=3 → T=7  (7,9,11 qualify — exactly 3 alerts)
    assert min_threshold_index(scores, 3)   == 7
    assert min_threshold_bsearch(scores, 3) == 7
    assert min_threshold_manual(scores, 3)  == 7
    assert min_threshold_naive(scores, 3)   == 7

    # K=1 → only max qualifies, T=11
    assert min_threshold_index(scores, 1) == 11

    # K=6 → all qualify, T = scores[0] = 1
    assert min_threshold_index(scores, 6) == 1
    assert min_threshold_index(scores, 10) == 1  # k >= n

    # K=0 → no alerts qualify, T above max
    assert min_threshold_index(scores, 0) == 12  # scores[-1]+1

    # Duplicates
    dup = [1, 3, 3, 5, 7, 7, 9]
    # K=3: top 3 are 7,7,9. T=7. count(>=7) = 3. ✓
    assert min_threshold_index(dup, 3) == 7
    assert min_threshold_bsearch(dup, 3) == 7

    # All approaches agree across multiple K values
    for k_val in range(1, 7):
        idx = min_threshold_index(scores, k_val)
        bs  = min_threshold_bsearch(scores, k_val)
        man = min_threshold_manual(scores, k_val)
        naive = min_threshold_naive(scores, k_val)
        assert idx == bs == man == naive, \
            f"Mismatch at k={k_val}: idx={idx}, bs={bs}, man={man}, naive={naive}"

    print("Q18 — All assertions passed.")
    print(f"min_threshold([1,3,5,7,9,11], k=3) = {min_threshold_index(scores, 3)}")  # 7
    print()
    print("Explanation:")
    print("  scores = [1,3,5,7,9,11], k=3")
    print("  n-k = 6-3 = 3  → scores[3] = 7")
    print("  count(score >= 7) = 3  (7,9,11) ✓")
