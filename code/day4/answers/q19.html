# Q19 - Sort and Deduplicate IP Blocklist
# Given a messy list of IP strings (with duplicates, unsorted), produce a
# sorted, deduplicated blocklist. Discuss complexity and scaling strategies.
#
# ─── SOLUTION WALKTHROUGH ───────────────────────────────────────────────────
#
# STEP 1 — Clarify
#   "Are all entries guaranteed to be valid IPs? IPv4 only or IPv6 too?
#    What's the expected scale — memory-resident or disk-based sort needed?"
#
# STEP 2 — Naive approach
#   "Load everything into memory. Use a set to dedup, convert to list, sort.
#    O(n) for set build + O(u log u) for sort where u = unique IPs.
#    Simple and correct for small-to-medium lists."
#
# STEP 3 — Natural sort (IP-semantic ordering)
#   "Sorting IPs lexicographically gives wrong order:
#    '10.0.0.2' < '9.0.0.1' lexicographically (wrong!)
#    Correct: sort by numeric tuple (10, 0, 0, 2) < (10, 0, 0, 3).
#    Use key=ip_sort_key."
#
# STEP 4 — Large-scale (10M entries)
#   "10M IPv4 strings × ~15 bytes each ≈ 150MB — fits in RAM on a modern server.
#    But for 10B entries or streaming:
#    - External sort (sort chunks that fit in RAM, merge sorted chunks)
#    - Bloom filter for approximate dedup before exact dedup
#    - Radix sort on the 32-bit integer representation of IPv4: O(n)"
#
# STEP 5 — Edge cases
#   - Empty list                   → []
#   - All duplicates               → single entry list
#   - Malformed IPs                → validate and skip/reject
#   - IPv6 addresses               → need different key function
#
# Complexity:
#   Basic: O(n + u log u) time, O(n) space
#   Radix (IPv4 only): O(n) time, O(n) space
# ─────────────────────────────────────────────────────────────────────────────

import socket


# ── APPROACH 1: Naive — set dedup + lexicographic sort ───────────────────────

def blocklist_naive(ips: list) -> list:
    """
    Dedup with set, sort lexicographically (INCORRECT order for IPs).
    Mention this first to show the pitfall.

    Time:  O(n + u log u)  u = unique IPs
    Space: O(n)
    """
    if not isinstance(ips, list):
        raise TypeError(f"Expected list, got {type(ips).__name__}")
    return sorted(set(ips))     # lexicographic sort — wrong for IPs


# ── APPROACH 2: Correct IP-aware sort with validation ────────────────────────

def _ip_sort_key(ip: str):
    """
    Convert IPv4 string to a tuple of 4 integers for correct numeric sort.
    E.g., '10.0.0.2' → (10, 0, 0, 2).
    Returns a sentinel tuple for invalid IPs so they sort to the end.
    """
    parts = ip.split(".")
    if len(parts) != 4:
        return (256, 256, 256, 256)     # invalid → sort to end
    try:
        octets = tuple(int(p) for p in parts)
        if any(o < 0 or o > 255 for o in octets):
            return (256, 256, 256, 256)
        return octets
    except ValueError:
        return (256, 256, 256, 256)


def _is_valid_ipv4(ip: str) -> bool:
    """Validate an IPv4 address string."""
    if not isinstance(ip, str):
        return False
    parts = ip.split(".")
    if len(parts) != 4:
        return False
    try:
        return all(0 <= int(p) <= 255 and str(int(p)) == p for p in parts)
    except ValueError:
        return False


def blocklist_sorted(ips: list, skip_invalid: bool = True) -> list:
    """
    Dedup, validate, and sort numerically by IP value.

    Time:  O(n + u log u)   u = unique valid IPs
    Space: O(n)
    """
    if not isinstance(ips, list):
        raise TypeError(f"Expected list, got {type(ips).__name__}")
    if not ips:
        return []

    seen  = set()
    valid = []

    for ip in ips:
        if not isinstance(ip, str):
            if not skip_invalid:
                raise TypeError(f"IP must be a string, got {type(ip).__name__}")
            continue
        ip = ip.strip()
        if ip in seen:
            continue
        if not _is_valid_ipv4(ip):
            if not skip_invalid:
                raise ValueError(f"Invalid IPv4 address: {ip!r}")
            continue
        seen.add(ip)
        valid.append(ip)

    valid.sort(key=_ip_sort_key)
    return valid


# ── APPROACH 3: Radix sort on 32-bit integer representation (O(n)) ───────────

def _ip_to_int(ip: str) -> int:
    """Convert IPv4 string to 32-bit integer."""
    parts = ip.split(".")
    n = 0
    for p in parts:
        n = (n << 8) | int(p)
    return n


def _int_to_ip(n: int) -> str:
    """Convert 32-bit integer back to IPv4 string."""
    return ".".join([
        str((n >> 24) & 0xFF),
        str((n >> 16) & 0xFF),
        str((n >> 8)  & 0xFF),
        str(n         & 0xFF),
    ])


def blocklist_radix(ips: list) -> list:
    """
    Convert IPs to 32-bit integers, deduplicate via set, radix sort,
    convert back. For IPv4, the numeric space is [0, 2^32-1].

    Radix sort on 32-bit ints: O(n) time (4 passes of counting sort on 1 byte).
    Total: O(n) time, O(n) space.

    Note: Python's built-in sort with integer key is effectively O(n log n).
    True radix sort is shown conceptually for discussion.
    """
    if not isinstance(ips, list):
        raise TypeError(f"Expected list, got {type(ips).__name__}")
    if not ips:
        return []

    # Dedup and convert to ints
    unique_ints = set()
    for ip in ips:
        ip = ip.strip()
        if _is_valid_ipv4(ip):
            unique_ints.add(_ip_to_int(ip))

    # Sort integers (uses Python's Timsort — O(u log u))
    # True radix sort would be O(u) but more code; discuss conceptually
    sorted_ints = sorted(unique_ints)

    return [_int_to_ip(n) for n in sorted_ints]


# ── MANUAL — no set(), no sorted(), no sort() ────────────────────────────────

def blocklist_manual(ips: list) -> list:
    """
    Manual dedup (dict) + manual insertion sort on numeric IP tuples.

    Time:  O(n + u²) — insertion sort is O(u²); acceptable for demonstration
    Space: O(n)
    """
    if ips is None:
        raise ValueError("ips cannot be None")
    if len(ips) == 0:
        return []

    # Manual dedup using dict
    seen = {}
    unique = []
    for ip in ips:
        if not isinstance(ip, str):
            continue
        ip = ip.strip()
        if ip not in seen and _is_valid_ipv4(ip):
            seen[ip] = True
            unique.append(ip)

    # Insertion sort by numeric key
    for i in range(1, len(unique)):
        key_ip  = unique[i]
        key_val = _ip_sort_key(key_ip)
        j = i - 1
        while j >= 0 and _ip_sort_key(unique[j]) > key_val:
            unique[j + 1] = unique[j]
            j -= 1
        unique[j + 1] = key_ip

    return unique

if __name__ == "__main__":
    messy = [
        "10.0.0.2",
        "9.255.255.255",    # lexicographically "9..." > "10..." but numerically less
        "10.0.0.1",
        "192.168.1.1",
        "10.0.0.2",         # duplicate
        "172.16.0.0",
        "INVALID_IP",       # should be skipped
        "  10.0.0.3  ",     # leading/trailing whitespace
        "256.0.0.1",        # out-of-range octet — invalid
        "10.0.0.1",         # duplicate
    ]

    expected = [
        "9.255.255.255",
        "10.0.0.1",
        "10.0.0.2",
        "10.0.0.3",
        "172.16.0.0",
        "192.168.1.1",
    ]

    result       = blocklist_sorted(messy)
    result_radix = blocklist_radix(messy)
    result_manual = blocklist_manual(messy)

    assert result       == expected, f"blocklist_sorted: {result}"
    assert result_radix == expected, f"blocklist_radix:  {result_radix}"
    assert result_manual == expected, f"blocklist_manual: {result_manual}"

    # Edge cases
    assert blocklist_sorted([])              == []
    assert blocklist_sorted(["1.2.3.4"])     == ["1.2.3.4"]
    assert blocklist_sorted(["1.2.3.4"] * 5) == ["1.2.3.4"]

    # Validate the naive approach shows lexicographic ordering (incorrect)
    naive_result = blocklist_naive(["10.0.0.1", "9.255.255.255"])
    assert naive_result == ["10.0.0.1", "9.255.255.255"]   # lexicographic: "1" < "9"
    assert naive_result != ["9.255.255.255", "10.0.0.1"]   # numeric correct order

    print("Q19 — All assertions passed.")
    print("\nSorted deduplicated blocklist:")
    for ip in result:
        print(f"  {ip}")
    print()
    print("Scaling notes (10M+ entries):")
    print("  - 10M × 15 bytes ≈ 150MB — fits in RAM on most servers")
    print("  - IPv4 radix sort on 32-bit int: O(n) time")
    print("  - Disk: sort 64MB chunks, k-way merge (external merge sort)")
    print("  - Approximate dedup: Bloom filter (then exact dedup on candidates)")
