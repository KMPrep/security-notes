# Q2 - Search Insert Position
# Given a sorted list of unique integers, return the index where target would
# be inserted to keep it sorted. If target exists, return its index.
#
# ─── SOLUTION WALKTHROUGH ───────────────────────────────────────────────────
#
# STEP 1 — Clarify
#   "Is the array guaranteed sorted and unique? Can it be empty? Can target
#    be outside the range of all elements?"
#
# STEP 2 — Naive approach (linear scan)
#   "Walk left-to-right. The first position where arr[i] >= target is where
#    target belongs. If no such position exists, insert at the end."
#   Time: O(n)
#
# STEP 3 — Optimize
#   "Since the array is sorted, I can binary search. The critical observation:
#    when the while loop exits (lo > hi), lo is exactly the insertion point.
#    This is precisely the semantics of bisect_left."
#   Why does lo work?
#   - If target < arr[mid], hi shrinks → lo stays, pointing at the first
#     element >= target.
#   - If target > arr[mid], lo advances past mid.
#   - At exit, lo == hi + 1, and lo points to the correct slot.
#   Time: O(log n)
#
# STEP 4 — Edge cases
#   - Empty array             → insert at 0
#   - Target < arr[0]         → lo stays 0
#   - Target > arr[-1]        → lo becomes len(arr)
#   - Target already present  → returns its index (same logic)
#
# Security context: "Where does this new severity score rank in our baseline?"
#   → binary search the sorted baseline; insertion point tells us rank.
#
# Complexity:
#   Time  — O(log n)
#   Space — O(1)
# ─────────────────────────────────────────────────────────────────────────────


# ── APPROACH 1: Naive linear scan ────────────────────────────────────────────

def search_insert_naive(arr: list, target: int) -> int:
    """
    Linear scan — O(n) time, O(1) space.
    First pass to establish baseline; mention this before optimizing.
    """
    if not isinstance(arr, list):
        raise TypeError(f"Expected list, got {type(arr).__name__}")
    if not arr:
        return 0
    for i in range(len(arr)):
        if arr[i] >= target:
            return i
    return len(arr)


# ── APPROACH 2: Binary search — manual ───────────────────────────────────────

def search_insert(arr: list, target: int) -> int:
    """
    Binary search for insertion position (bisect_left semantics).

    Key insight: when the loop exits, lo is the leftmost index where
    arr[lo] >= target — i.e., the correct insertion slot.

    Time:  O(log n)
    Space: O(1)
    """
    if not isinstance(arr, list):
        raise TypeError(f"Expected list, got {type(arr).__name__}")
    if not arr:
        return 0

    lo, hi = 0, len(arr) - 1

    while lo <= hi:
        mid = lo + (hi - lo) // 2
        if arr[mid] == target:
            return mid                  # exact match — return immediately
        elif arr[mid] < target:
            lo = mid + 1
        else:
            hi = mid - 1

    # lo > hi now; lo is the insertion point
    return lo


# ── APPROACH 3: Using bisect module (production Python) ──────────────────────

import bisect

def search_insert_builtin(arr: list, target: int) -> int:
    """
    bisect_left returns the leftmost index to insert target while keeping
    the list sorted. Handles the "already present" case identically.

    Time:  O(log n)  — bisect is implemented in C
    Space: O(1)
    """
    if not isinstance(arr, list):
        raise TypeError(f"Expected list, got {type(arr).__name__}")
    if not arr:
        return 0
    return bisect.bisect_left(arr, target)


# ── MANUAL — no bisect, no enumerate, no stdlib ──────────────────────────────

def search_insert_manual(arr: list, target: int) -> int:
    """
    Explicit binary search; overflow-safe mid calculation.

    Time:  O(log n)
    Space: O(1)
    """
    if arr is None:
        raise ValueError("arr cannot be None")
    n = len(arr)
    if n == 0:
        return 0

    lo = 0
    hi = n - 1

    while lo <= hi:
        mid = lo + (hi - lo) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            lo = mid + 1
        else:
            hi = mid - 1

    return lo

if __name__ == "__main__":
    # From the problem statement
    assert search_insert([1, 3, 5, 6], 5) == 2   # exact match
    assert search_insert([1, 3, 5, 6], 2) == 1   # between 1 and 3

    # Edge cases
    assert search_insert([1, 3, 5, 6], 7) == 4   # after last
    assert search_insert([1, 3, 5, 6], 0) == 0   # before first
    assert search_insert([], 5)           == 0   # empty array
    assert search_insert([1], 0)          == 0   # before single
    assert search_insert([1], 1)          == 0   # match single
    assert search_insert([1], 2)          == 1   # after single

    # Built-in
    assert search_insert_builtin([1, 3, 5, 6], 5) == 2
    assert search_insert_builtin([1, 3, 5, 6], 2) == 1
    assert search_insert_builtin([], 5)            == 0

    # Manual
    assert search_insert_manual([1, 3, 5, 6], 5) == 2
    assert search_insert_manual([1, 3, 5, 6], 2) == 1

    # Naive agrees with optimized
    for t in [0, 1, 2, 4, 5, 6, 7]:
        assert search_insert_naive([1, 3, 5, 6], t) == search_insert([1, 3, 5, 6], t), \
