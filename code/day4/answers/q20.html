# Q20 - Kth Largest Anomaly Score
# Given an unsorted list of floats (anomaly scores), find the Kth largest.
# Compare sort approach vs min-heap. Which is better for n=10M, k=5?
#
# ─── SOLUTION WALKTHROUGH ───────────────────────────────────────────────────
#
# STEP 1 — Clarify
#   "Kth largest (1-indexed): k=1 means the maximum. Are there duplicates?
#    What if k > n?"
#
# STEP 2 — Naive approach: sort descending, return index k-1
#   "Sort the entire list descending. scores[k-1] is the answer.
#    Time: O(n log n). Space: O(n) for a copy (or O(log n) for in-place sort)."
#
# STEP 3 — Better: min-heap of size k
#   "Maintain a min-heap of exactly k elements.
#    - For each score: if heap size < k, push it.
#    - Else if score > heap[0] (the current k-th largest): pop and push.
#    The heap always holds the k largest elements seen so far.
#    The minimum of the heap (heap[0]) is the k-th largest.
#    Time: O(n log k). Space: O(k)."
#
# STEP 4 — When does the heap approach win?
#   "n = 10M, k = 5:
#    - Sort:     O(10M × log(10M)) ≈ 10M × 23 = 230M ops, O(10M) space
#    - Heap:     O(10M × log(5))   ≈ 10M × 2.3 = 23M ops, O(5) space
#    Min-heap is ~10× faster and uses negligible extra memory.
#    General rule: use heap when k << n."
#
# STEP 5 — Quickselect (optimal average case)
#   "Quickselect: average O(n), worst case O(n²). For a one-time query on
#    mutable data, Quickselect can be faster than heapq.
#    heapq.nlargest uses Quickselect internally when k > n//2; otherwise heap."
#
# STEP 6 — Edge cases
#   - k > n             → raise ValueError
#   - k == 1            → return max
#   - k == n            → return min
#   - Negative scores   → heap still works
#   - Duplicate scores  → still works (2nd largest of [5,5,5] is 5)
#
# Complexity:
#   Sort:        O(n log n) time, O(n) space (copy) or O(log n) in-place
#   Min-heap:    O(n log k) time, O(k) space
#   Quickselect: O(n) average, O(n²) worst, O(1) extra space
# ─────────────────────────────────────────────────────────────────────────────

import heapq
import random


# ── APPROACH 1: Sort descending, index k-1 ───────────────────────────────────

def kth_largest_sort(scores: list, k: int) -> float:
    """
    Sort all scores descending. O(n log n) time, O(n) space.

    Time:  O(n log n)
    Space: O(n) for sorted copy
    """
    if not isinstance(scores, list):
        raise TypeError(f"Expected list, got {type(scores).__name__}")
    if not isinstance(k, int) or k <= 0:
        raise ValueError(f"k must be a positive integer, got {k!r}")
    if not scores:
        raise ValueError("scores must not be empty")
    if k > len(scores):
        raise ValueError(f"k={k} exceeds number of scores ({len(scores)})")

    sorted_scores = sorted(scores, reverse=True)
    return sorted_scores[k - 1]


# ── APPROACH 2: Min-heap of size k (optimal for k << n) ──────────────────────

def kth_largest_heap(scores: list, k: int) -> float:
    """
    Min-heap of size k.

    The heap maintains the k largest elements seen so far.
    heap[0] is always the k-th largest (minimum of the top-k set).

    Time:  O(n log k)
    Space: O(k)
    """
    if not isinstance(scores, list):
        raise TypeError(f"Expected list, got {type(scores).__name__}")
    if not isinstance(k, int) or k <= 0:
        raise ValueError(f"k must be a positive integer, got {k!r}")
    if not scores:
        raise ValueError("scores must not be empty")
    if k > len(scores):
        raise ValueError(f"k={k} exceeds number of scores ({len(scores)})")

    # heapq is a MIN-heap; we want to track the K largest
    min_heap = []

    for score in scores:
        heapq.heappush(min_heap, score)
        if len(min_heap) > k:
            heapq.heappop(min_heap)         # evict smallest if heap exceeds k

    return min_heap[0]                      # k-th largest


# ── APPROACH 3: heapq.nlargest (production Python) ────────────────────────────

def kth_largest_builtin(scores: list, k: int) -> float:
    """
    heapq.nlargest(k, scores)[k-1] — uses heap internally.

    Time:  O(n log k)
    Space: O(k)
    """
    if not isinstance(scores, list):
        raise TypeError(f"Expected list, got {type(scores).__name__}")
    if not isinstance(k, int) or k <= 0:
        raise ValueError(f"k must be a positive integer, got {k!r}")
    if not scores:
        raise ValueError("scores must not be empty")
    if k > len(scores):
        raise ValueError(f"k={k} exceeds number of scores ({len(scores)})")

    return heapq.nlargest(k, scores)[-1]


# ── APPROACH 4: Quickselect — O(n) average ───────────────────────────────────

def kth_largest_quickselect(scores: list, k: int) -> float:
    """
    Quickselect on a copy of scores.
    Partition around a pivot until the pivot lands at index n-k (0-based).

    Time:  O(n) average, O(n²) worst case (bad pivot choice)
    Space: O(1) extra (in-place on copy)
    """
    if not isinstance(scores, list):
        raise TypeError(f"Expected list, got {type(scores).__name__}")
    if not isinstance(k, int) or k <= 0:
        raise ValueError(f"k must be a positive integer, got {k!r}")
    if not scores:
        raise ValueError("scores must not be empty")
    if k > len(scores):
        raise ValueError(f"k={k} exceeds number of scores ({len(scores)})")

    arr = list(scores)          # work on a copy
    target_idx = len(arr) - k   # 0-based index of kth largest in sorted order

    def partition(lo: int, hi: int) -> int:
        # Randomized pivot to reduce worst-case probability
        pivot_idx = random.randint(lo, hi)
        arr[pivot_idx], arr[hi] = arr[hi], arr[pivot_idx]
        pivot = arr[hi]
        i = lo - 1
        for j in range(lo, hi):
            if arr[j] <= pivot:
                i += 1
                arr[i], arr[j] = arr[j], arr[i]
        arr[i + 1], arr[hi] = arr[hi], arr[i + 1]
        return i + 1

    lo, hi = 0, len(arr) - 1
    while lo <= hi:
        pivot_pos = partition(lo, hi)
        if pivot_pos == target_idx:
            return arr[pivot_pos]
        elif pivot_pos < target_idx:
            lo = pivot_pos + 1
        else:
            hi = pivot_pos - 1

    return arr[target_idx]


# ── MANUAL — no heapq, manual min-heap ───────────────────────────────────────

class MinHeap:
    """Manual min-heap of fixed max size k."""

    def __init__(self, max_size: int):
        self._data = []
        self._max  = max_size

    def push(self, val: float) -> None:
        self._data.append(val)
        self._sift_up(len(self._data) - 1)
        if len(self._data) > self._max:
            self._swap(0, len(self._data) - 1)
            self._data.pop()
            self._sift_down(0)

    def peek(self) -> float:
        if not self._data:
            raise IndexError("Heap is empty")
        return self._data[0]

    def _swap(self, i: int, j: int) -> None:
        self._data[i], self._data[j] = self._data[j], self._data[i]

    def _sift_up(self, i: int) -> None:
        while i > 0:
            parent = (i - 1) // 2
            if self._data[parent] > self._data[i]:
                self._swap(parent, i)
                i = parent
            else:
                break

    def _sift_down(self, i: int) -> None:
        n = len(self._data)
        while True:
            smallest = i
            left     = 2 * i + 1
            right    = 2 * i + 2
            if left  < n and self._data[left]  < self._data[smallest]:
                smallest = left
            if right < n and self._data[right] < self._data[smallest]:
                smallest = right
            if smallest == i:
                break
            self._swap(i, smallest)
            i = smallest


def kth_largest_manual(scores: list, k: int) -> float:
    """
    Manual min-heap of size k without heapq.

    Time:  O(n log k)
    Space: O(k)
    """
    if scores is None:
        raise ValueError("scores cannot be None")
    if not scores:
        raise ValueError("scores must not be empty")
    if k <= 0 or k > len(scores):
        raise ValueError(f"k must be in [1, {len(scores)}], got {k}")

    heap = MinHeap(max_size=k)
    for score in scores:
        heap.push(score)

    return heap.peek()

if __name__ == "__main__":
    scores = [3.2, 1.5, 7.8, 2.1, 9.0, 4.6, 0.3, 8.8]

    # k=1 → maximum = 9.0
    assert kth_largest_sort(scores, 1)          == 9.0
    assert kth_largest_heap(scores, 1)          == 9.0
    assert kth_largest_builtin(scores, 1)       == 9.0
    assert kth_largest_quickselect(scores, 1)   == 9.0
    assert kth_largest_manual(scores, 1)        == 9.0

    # k=3 → [9.0, 8.8, 7.8] → 7.8
    assert kth_largest_sort(scores, 3)          == 7.8
    assert kth_largest_heap(scores, 3)          == 7.8
    assert kth_largest_builtin(scores, 3)       == 7.8
    assert kth_largest_manual(scores, 3)        == 7.8

    # k=n → minimum
    assert kth_largest_sort(scores, len(scores)) == min(scores)

    # Negatives
    neg = [-5.0, -1.0, -3.0, -2.0]
    assert kth_largest_sort(neg, 2)    == -2.0  # 2nd largest
    assert kth_largest_heap(neg, 2)    == -2.0
    assert kth_largest_manual(neg, 2)  == -2.0

    # Duplicates
    dup = [5.0, 5.0, 5.0, 3.0]
    assert kth_largest_sort(dup, 2)   == 5.0

    # Error guards
    try:
        kth_largest_sort(scores, 0)
        assert False
    except ValueError:
        pass

        assert False
