# Q21 - Count Inversions in Threat Timeline
# An "inversion" is a pair (i, j) where i < j but scores[i] > scores[j].
# Count total inversions. Brute force O(n²). Describe the O(n log n) approach.
#
# ─── SOLUTION WALKTHROUGH ───────────────────────────────────────────────────
#
# STEP 1 — Clarify
#   "An inversion means a score DECREASED from time i to time j (i < j),
#    which in a threat timeline signals 'expected escalation didn't happen'
#    or data was received out of order. Count ALL such pairs."
#
# STEP 2 — Naive O(n²)
#   "Check every pair (i, j) where i < j: if scores[i] > scores[j], count it.
#    Two nested loops. O(n²) time, O(1) space.
#    For n=100K that's 10^10 ops — too slow."
#
# STEP 3 — Merge-sort based O(n log n)
#   "The key insight: during MERGE of two sorted halves, if we pick an element
#    from the RIGHT half before we're done with the LEFT half, every remaining
#    element in the left half forms an inversion with the picked right element.
#
#    Why? Left half is sorted, right half is sorted. If right[j] < left[i],
#    then right[j] < left[i], left[i+1], ..., left[-1] — all of them form
#    inversions with right[j]. Count them as (len(left) - i).
#
#    Total inversions = inversions in left half + inversions in right half
#                     + inversions across the split (counted during merge).
#    T(n) = 2T(n/2) + O(n) = O(n log n) by master theorem."
#
# STEP 4 — Edge cases
#   - Empty or single element    → 0 inversions
#   - Already sorted             → 0 inversions
#   - Reverse sorted             → n*(n-1)/2 inversions (maximum)
#   - Duplicates (equal scores)  → equal elements are NOT inversions
#
# Complexity:
#   Brute force: O(n²) time, O(1) space
#   Merge sort:  O(n log n) time, O(n) space
# ─────────────────────────────────────────────────────────────────────────────


# ── APPROACH 1: Brute force O(n²) ────────────────────────────────────────────

def count_inversions_naive(scores: list) -> int:
    """
    Check all pairs — O(n²) time, O(1) space.
    Mention first to establish the baseline.
    """
    if not isinstance(scores, list):
        raise TypeError(f"Expected list, got {type(scores).__name__}")
    n = len(scores)
    count = 0
    for i in range(n):
        for j in range(i + 1, n):
            if scores[i] > scores[j]:
                count += 1
    return count


# ── APPROACH 2: Merge sort — O(n log n) ──────────────────────────────────────

def _merge_count(left: list, right: list) -> tuple:
    """
    Merge two sorted lists and count split inversions.

    A split inversion occurs when right[j] < left[i]:
    all remaining elements in left (indices i..end) are > right[j].
    Count = len(left) - i.

    Returns: (merged_sorted_list, inversion_count)
    """
    merged  = []
    count   = 0
    i = j   = 0

    while i < len(left) and j < len(right):
        if left[i] <= right[j]:
            merged.append(left[i])
            i += 1
        else:
            # right[j] < left[i]: inversion with ALL remaining left elements
            count += len(left) - i
            merged.append(right[j])
            j += 1

    while i < len(left):
        merged.append(left[i])
        i += 1
    while j < len(right):
        merged.append(right[j])
        j += 1

    return merged, count


def _merge_sort_count(arr: list) -> tuple:
    """
    Merge sort that accumulates inversion count at each merge step.

    Returns: (sorted_arr, total_inversions)
    """
    if len(arr) <= 1:
        return list(arr), 0

    mid   = len(arr) // 2
    left,  left_inv  = _merge_sort_count(arr[:mid])
    right, right_inv = _merge_sort_count(arr[mid:])
    merged, split_inv = _merge_count(left, right)

    return merged, left_inv + right_inv + split_inv


def count_inversions(scores: list) -> int:
    """
    Count inversions using modified merge sort.

    Time:  O(n log n)
    Space: O(n) for merge buffer
    """
    if not isinstance(scores, list):
        raise TypeError(f"Expected list, got {type(scores).__name__}")
    if len(scores) <= 1:
        return 0

    _, total = _merge_sort_count(scores)
    return total


# ── APPROACH 3: Using sorted() + BIT (Binary Indexed Tree / Fenwick Tree) ─────
# "For completeness: a BIT/Fenwick tree can also count inversions in O(n log n).
#  Process elements right-to-left; for each element, query BIT for count of
#  elements < current seen so far (those form inversions with current).
#  More complex to implement but O(n log n) and O(n) space."
#  (We implement merge sort above as it's more easy to explain.)


# ── MANUAL — no stdlib for sort/merge ────────────────────────────────────────

def count_inversions_manual(scores: list) -> int:
    """
    Same merge-sort logic written without any imports or list slicing with copy.

    Time:  O(n log n)
    Space: O(n)
    """
    if scores is None:
        raise ValueError("scores cannot be None")
    n = len(scores)
    if n <= 1:
        return 0

    def merge_count_manual(arr):
        if len(arr) <= 1:
            return arr, 0
        mid = len(arr) // 2
        # Manual slice without list()
        left_arr  = [arr[i] for i in range(mid)]
        right_arr = [arr[i] for i in range(mid, len(arr))]
        left_sorted, lc  = merge_count_manual(left_arr)
        right_sorted, rc = merge_count_manual(right_arr)

        merged = []
        inv_c  = 0
        i = j  = 0
        while i < len(left_sorted) and j < len(right_sorted):
            if left_sorted[i] <= right_sorted[j]:
                merged.append(left_sorted[i])
                i += 1
            else:
                inv_c += len(left_sorted) - i
                merged.append(right_sorted[j])
                j += 1
        while i < len(left_sorted):
            merged.append(left_sorted[i])
            i += 1
        while j < len(right_sorted):
            merged.append(right_sorted[j])
            j += 1

        return merged, lc + rc + inv_c

    _, total = merge_count_manual(list(scores))
    return total


# ── Tests ─────────────────────────────────────────────────────────────────────

if __name__ == "__main__":
    # Basic
    assert count_inversions([2, 4, 1, 3, 5]) == 3   # (2,1),(4,1),(4,3)
    assert count_inversions([1, 2, 3, 4, 5]) == 0   # sorted → no inversions
    assert count_inversions([5, 4, 3, 2, 1]) == 10  # reverse → n*(n-1)/2 = 10

    # Empty and single
    assert count_inversions([])  == 0
    assert count_inversions([7]) == 0

    # Duplicates — equal elements NOT inversions
    assert count_inversions([2, 2, 2]) == 0

    # Manual
    assert count_inversions_manual([2, 4, 1, 3, 5]) == 3
    assert count_inversions_manual([5, 4, 3, 2, 1]) == 10

    # Naive agrees with merge sort
    test_cases = [
        [2, 4, 1, 3, 5],
        [1, 2, 3, 4, 5],
        [5, 4, 3, 2, 1],
        [3, 1, 2],
        [1],
        [],
    ]
    for tc in test_cases:
        naive  = count_inversions_naive(tc)
        merged = count_inversions(tc)
        manual = count_inversions_manual(tc)
        assert naive == merged == manual, \
            f"Mismatch on {tc}: naive={naive}, merged={merged}, manual={manual}"

    print("Q21 — All assertions passed.")
    print(f"count_inversions([2,4,1,3,5]) = {count_inversions([2,4,1,3,5])}")  # 3
    print(f"count_inversions([5,4,3,2,1]) = {count_inversions([5,4,3,2,1])}")  # 10
    print()
    print("Algorithm: modified merge sort")
    print("  At each merge: if right[j] < left[i], count += len(left) - i")
    print("  T(n) = 2T(n/2) + O(n) → O(n log n) by master theorem")
