# Q22 - Stable Partition by Severity
# Given a mixed list of events, partition so all CRITICAL events come before
# non-CRITICAL, preserving relative order within each group. O(n), O(1) extra?
#
# ─── SOLUTION WALKTHROUGH ───────────────────────────────────────────────────
#
# STEP 1 — Clarify
#   "We want all CRITICAL events to appear before all non-CRITICAL events.
#    Within each group, the original order must be preserved (stable).
#    Events are (severity, ...) tuples."
#
# STEP 2 — Naive O(n) time, O(n) space
#   "Two passes: collect all CRITICAL events preserving order, then all
#    non-CRITICAL events preserving order, concatenate.
#    This is O(n) time and O(n) space."
#
# STEP 3 — Can we do O(1) extra space?
#   "For an UNSTABLE partition (like Lomuto / Hoare in quicksort), yes — O(1).
#    But for a STABLE partition in O(1) extra space, it's theoretically
#    possible but requires a complex algorithm (block swaps, buffer techniques).
#    In practice, the O(n) space two-pass is the accepted answer.
#
#    Python's built-in filter() / list comprehension is exactly this O(n) stable
#    partition. Python's sort() with a key is also stable."
#
# STEP 4 — Python built-in that achieves this
#   "sorted() with key that puts CRITICAL first is stable O(n log n).
#    But for binary partitioning (only 2 groups), two-pass is O(n)."
#
# STEP 5 — Edge cases
#   - All CRITICAL               → all events in one group, order preserved
#   - No CRITICAL                → no change to order
#   - Empty list                 → []
#   - Single event               → unchanged
#
# Complexity:
#   Two-pass (O(n) space): O(n) time, O(n) space
#   In-place unstable:     O(n) time, O(1) space (loses relative order)
#   sorted() with key:     O(n log n) time, O(n) space
# ─────────────────────────────────────────────────────────────────────────────


# ── APPROACH 1: Two-pass stable partition (O(n) space) — start here ───────────

def stable_partition_critical(events: list) -> list:
    """
    Two passes:
      1. Collect all CRITICAL events in order.
      2. Collect all non-CRITICAL events in order.
      3. Concatenate.

    Stable: relative order within each group is preserved.

    Time:  O(n)
    Space: O(n) — two output lists
    """
    if not isinstance(events, list):
        raise TypeError(f"Expected list, got {type(events).__name__}")
    if not events:
        return []

    critical     = []
    non_critical = []

    for event in events:
        if not isinstance(event, (tuple, list)) or len(event) < 1:
            raise ValueError(f"Event must be a non-empty tuple/list, got {event!r}")
        if event[0] == "CRITICAL":
            critical.append(event)
        else:
            non_critical.append(event)

    return critical + non_critical


# ── APPROACH 2: In-place unstable partition (O(1) extra space) ───────────────

def unstable_partition_critical(events: list) -> list:
    """
    Lomuto-style in-place partition. NOT stable.
    Shown as contrast — demonstrates the stability trade-off.

    Time:  O(n)
    Space: O(1)
    """
    if not isinstance(events, list):
        raise TypeError(f"Expected list, got {type(events).__name__}")
    if not events:
        return []

    arr = list(events)      # copy to avoid mutating input
    boundary = 0            # all elements [0..boundary) are CRITICAL

    for i in range(len(arr)):
        if arr[i][0] == "CRITICAL":
            arr[i], arr[boundary] = arr[boundary], arr[i]
            boundary += 1

    return arr


# ── APPROACH 3: Using sorted() with key (stable, but O(n log n)) ─────────────

def sorted_partition_critical(events: list) -> list:
    """
    sorted() with key=(0 if CRITICAL else 1). Stable O(n log n).

    Overkill for binary partition, but useful when you want multi-level
    grouping (e.g., CRITICAL < HIGH < MEDIUM < LOW < INFO).

    Time:  O(n log n)
    Space: O(n)
    """
    if not isinstance(events, list):
        raise TypeError(f"Expected list, got {type(events).__name__}")
    if not events:
        return []

    return sorted(events, key=lambda e: (0 if e[0] == "CRITICAL" else 1))


# ── APPROACH 4: Python filter (most Pythonic for stable partition) ─────────────

def filter_partition_critical(events: list) -> list:
    """
    list(filter()) — functionally equivalent to two-pass.
    Most readable Python idiom.

    Time:  O(n)
    Space: O(n)
    """
    if not isinstance(events, list):
        raise TypeError(f"Expected list, got {type(events).__name__}")
    if not events:
        return []

    critical     = list(filter(lambda e: e[0] == "CRITICAL", events))
    non_critical = list(filter(lambda e: e[0] != "CRITICAL", events))
    return critical + non_critical


# ── MANUAL — no filter, no sorted, no list comprehension sugar ───────────────

def stable_partition_manual(events: list) -> list:
    """
    Manual two-pass without any built-in helpers.

    Time:  O(n)
    Space: O(n)
    """
    if events is None:
        raise ValueError("events cannot be None")
    n = len(events)
    if n == 0:
        return []

    critical_list     = []
    non_critical_list = []

    i = 0
    while i < n:
        ev = events[i]
        if ev[0] == "CRITICAL":
            critical_list.append(ev)
        else:
            non_critical_list.append(ev)
        i += 1

    result = []
    for ev in critical_list:
        result.append(ev)
    for ev in non_critical_list:
        result.append(ev)

    return result


# ── Tests ─────────────────────────────────────────────────────────────────────

if __name__ == "__main__":
    events = [
        ("HIGH",     1000, "High alert 1"),
        ("CRITICAL", 1001, "Critical A"),
        ("MEDIUM",   1002, "Medium alert 1"),
        ("CRITICAL", 1003, "Critical B"),
        ("LOW",      1004, "Low alert 1"),
        ("CRITICAL", 1005, "Critical C"),
        ("INFO",     1006, "Info alert"),
    ]

    # Expected: all 3 CRITICAL events first, in original order, then rest in order
    result = stable_partition_critical(events)

    critical_events     = [e for e in result if e[0] == "CRITICAL"]
    non_critical_events = [e for e in result if e[0] != "CRITICAL"]

    # All CRITICAL before non-CRITICAL
    last_crit_idx  = max(i for i, e in enumerate(result) if e[0] == "CRITICAL")
    first_non_idx  = min(i for i, e in enumerate(result) if e[0] != "CRITICAL")
    assert last_crit_idx < first_non_idx, "All CRITICAL events must precede non-CRITICAL"

    # Stable: original relative order preserved
    assert [e[2] for e in critical_events] == ["Critical A", "Critical B", "Critical C"]
    non_expected = ["High alert 1", "Medium alert 1", "Low alert 1", "Info alert"]
    assert [e[2] for e in non_critical_events] == non_expected

    # All stable methods agree
    r2 = sorted_partition_critical(events)
    r3 = filter_partition_critical(events)
    r4 = stable_partition_manual(events)
    assert result == r2 == r3 == r4

    # Unstable partition: correct grouping but possibly different order within groups
    unstable = unstable_partition_critical(events)
    u_critical = [e for e in unstable if e[0] == "CRITICAL"]
    u_non      = [e for e in unstable if e[0] != "CRITICAL"]
    assert len(u_critical) == 3
    assert len(u_non)      == 4

    # Edge cases
    assert stable_partition_critical([]) == []
    assert stable_partition_critical([("HIGH", 1, "msg")]) == [("HIGH", 1, "msg")]
    all_crit = [("CRITICAL", i, f"c{i}") for i in range(3)]
    assert stable_partition_critical(all_crit) == all_crit  # unchanged

    print("Q22 — All assertions passed.")
    print("\nStable partition result:")
    for e in result:
        print(f"  {e[0]:9s}  ts={e[1]}  {e[2]}")
    print()
    print("Space complexity note:")
    print("  Two-pass stable: O(n) time, O(n) space")
    print("  In-place unstable: O(n) time, O(1) space — but loses relative order")
    print("  True in-place stable partition: O(n log n) time, O(1) space (complex)")
