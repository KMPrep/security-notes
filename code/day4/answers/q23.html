# Q23 - Bisect: Insert Alert into Sorted Feed
# Use bisect module to maintain a sorted list of timestamps as new alerts arrive.
# Insert 1000 alerts one by one. What is total complexity vs sorting at the end?
#
# ─── SOLUTION WALKTHROUGH ───────────────────────────────────────────────────
#
# STEP 1 — Clarify
#   "We want to keep a sorted list at all times as alerts arrive one by one.
#    Should we use bisect_insort? And compare against batching all then sorting?"
#
# STEP 2 — bisect_insort approach
#   "bisect.insort(arr, x) finds the correct position in O(log n) via binary
#    search, then inserts in O(n) because Python lists shift elements right.
#    Per insert: O(log n) search + O(n) shift = O(n) dominated.
#    For n inserts: O(n²) total.
#
#    Wait — the binary search is O(log n), but the insertion into the list is
#    O(n) due to array shift. So each insort is O(n) worst case."
#
# STEP 3 — Sort at the end
#   "Append n items: O(n) amortized.
#    Sort once at the end: O(n log n).
#    Total: O(n log n). Significantly better than O(n²)."
#
# STEP 4 — When does bisect_insort win?
#   "When you need the list to be sorted AT EVERY POINT IN TIME — e.g., you're
#    answering range queries between inserts.
#    If you only need the final sorted result, batch sort is always better."
#
# STEP 5 — Heap alternative
#   "If you only need to poll the minimum repeatedly (priority queue semantics),
#    use heapq: push O(log n), pop O(log n). n pushes = O(n log n) total.
#    But heapq does NOT give you a sorted random-access list."
#
# STEP 6 — Edge cases
#   - Duplicate timestamps         → bisect_insort preserves duplicates (right)
#   - Inserting in sorted order    → O(log n) bisect, O(1) shift (append)
#   - Inserting in reverse order   → O(log n) bisect, O(n) shift always
#
# Complexity summary:
#   bisect_insort × n: O(n²) total (dominated by list shifts)
#   append + sort:     O(n log n) total
#   heap × n:          O(n log n) total, O(log n) per operation
# ─────────────────────────────────────────────────────────────────────────────

import bisect
import heapq
import time
import random


# ── APPROACH 1: bisect_insort — sorted after every insert ────────────────────

def build_sorted_feed_insort(timestamps: list) -> list:
    """
    Maintain a sorted list using bisect.insort_right.

    Per insert: O(log n) search + O(n) shift = O(n)
    Total for n inserts: O(n²)

    Use when you NEED sorted access between inserts.
    """
    if not isinstance(timestamps, list):
        raise TypeError(f"Expected list, got {type(timestamps).__name__}")

    feed = []
    for ts in timestamps:
        bisect.insort_right(feed, ts)   # O(log n) search + O(n) shift
    return feed


# ── APPROACH 2: Append all, sort at end (better for final result only) ────────

def build_sorted_feed_batch(timestamps: list) -> list:
    """
    Collect all timestamps, sort once at the end.

    Time:  O(n) append + O(n log n) sort = O(n log n)
    Space: O(n)

    Preferred when you only need the final sorted list.
    """
    if not isinstance(timestamps, list):
        raise TypeError(f"Expected list, got {type(timestamps).__name__}")

    feed = list(timestamps)     # O(n) copy
    feed.sort()                 # O(n log n) Timsort
    return feed


# ── APPROACH 3: Heap — O(n log n) with O(log n) per insert/pop ───────────────

def build_sorted_feed_heap(timestamps: list) -> list:
    """
    Push all into heap, then pop in order to get sorted output.

    Time:  O(n log n)  — n pushes, each O(log n)
    Space: O(n)

    Useful when processing in priority order (not random access).
    """
    if not isinstance(timestamps, list):
        raise TypeError(f"Expected list, got {type(timestamps).__name__}")

    heap = list(timestamps)
    heapq.heapify(heap)             # O(n) to build heap from unsorted list

    result = []
    while heap:
        result.append(heapq.heappop(heap))  # O(log n) each

    return result


# ── MANUAL bisect + insort without bisect module ─────────────────────────────

def _bisect_right_manual(arr: list, val) -> int:
    """Find rightmost insertion index for val in sorted arr."""
    lo, hi = 0, len(arr)
    while lo < hi:
        mid = lo + (hi - lo) // 2
        if arr[mid] <= val:
            lo = mid + 1
        else:
            hi = mid
    return lo


def build_sorted_feed_manual(timestamps: list) -> list:
    """
    Manual insort using binary search + list insert.

    Per insert: O(log n) search + O(n) shift = O(n)
    Total: O(n²)
    """
    if timestamps is None:
        raise ValueError("timestamps cannot be None")

    feed = []
    for ts in timestamps:
        idx = _bisect_right_manual(feed, ts)
        feed.insert(idx, ts)    # list.insert is O(n) due to shift
    return feed


# ── Performance comparison ────────────────────────────────────────────────────

def benchmark_comparison(n: int = 1000) -> None:
    """
    Time insort vs batch sort for n random timestamps.
    """
    timestamps = [random.randint(0, 10_000_000) for _ in range(n)]

    # bisect_insort
    t0 = time.perf_counter()
    result_insort = build_sorted_feed_insort(timestamps)
    t1 = time.perf_counter()
    insort_ms = (t1 - t0) * 1000

    # batch sort
    t2 = time.perf_counter()
    result_batch = build_sorted_feed_batch(timestamps)
    t3 = time.perf_counter()
    batch_ms = (t3 - t2) * 1000

    print(f"n={n} timestamps:")
    print(f"  bisect_insort:  {insort_ms:.2f} ms  (O(n²))")
    print(f"  batch sort:     {batch_ms:.2f} ms  (O(n log n))")
    assert result_insort == result_batch, "Results must be identical"


# ── Tests ─────────────────────────────────────────────────────────────────────

if __name__ == "__main__":
    timestamps = [1620000005, 1620000001, 1620000003, 1620000002, 1620000004]
    expected   = sorted(timestamps)

    # All approaches produce the same sorted result
    assert build_sorted_feed_insort(timestamps) == expected
    assert build_sorted_feed_batch(timestamps)  == expected
    assert build_sorted_feed_heap(timestamps)   == expected
    assert build_sorted_feed_manual(timestamps) == expected

    # Duplicates
    dup = [5, 3, 5, 1, 3]
    assert build_sorted_feed_insort(dup) == [1, 3, 3, 5, 5]
    assert build_sorted_feed_batch(dup)  == [1, 3, 3, 5, 5]

    # Edge cases
    assert build_sorted_feed_insort([]) == []
    assert build_sorted_feed_batch([])  == []
    assert build_sorted_feed_insort([42]) == [42]

    print("Q23 — All assertions passed.")
    print()
    benchmark_comparison(n=2000)
    print()
    print("Complexity comparison:")
    print("  bisect_insort × n:  O(n²)      — correct if sorted access needed at all times")
    print("  append + sort:      O(n log n) — optimal if only final order needed")
    print("  heap:               O(n log n) — optimal for priority-queue workloads")
    print()
    print("Rule: if you only need final sorted result → ALWAYS batch sort.")
    print("      if you need sorted list between inserts → insort (but O(n²) cost).")
    print("      if you need min/max access only → heap.")
