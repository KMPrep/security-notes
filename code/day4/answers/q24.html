# Q24 - Find Earliest Conflicting Firewall Rule
# Given a sorted list of rules each with port range [lo, hi], find the first
# rule that overlaps with ANY previous rule. Return its index.
#
# ─── SOLUTION WALKTHROUGH ───────────────────────────────────────────────────
#
# STEP 1 — Clarify
#   "Rules are sorted — by what? Start port? We need to define overlap:
#    rule A [a_lo, a_hi] overlaps rule B [b_lo, b_hi] if a_lo <= b_hi AND b_lo <= a_hi.
#    'Conflicts with any previous rule' means overlaps with any rule at a lower index.
#    Return the index of the FIRST such rule."
#
# STEP 2 — Naive approach
#   "For each rule i, compare it against ALL previous rules 0..i-1.
#    If any overlap, return i. O(n²) worst case."
#
# STEP 3 — Observation to optimize
#   "Since rules are sorted by start port, we only need to check if the new
#    rule's start (b_lo) falls within any previously seen rule.
#    We can track the MAXIMUM hi seen so far from all previous rules.
#    If the new rule's b_lo <= max_hi_so_far, it overlaps with the rule
#    that had that max_hi. Return current index.
#    This is O(n) — one linear pass."
#
# STEP 4 — Can binary search help?
#   "If rules are sorted by start, and we want to check if b_lo <= max_hi_so_far:
#    We could maintain a sorted structure of (hi values) and binary search,
#    but since we just need the maximum hi, a single running max is sufficient.
#    Binary search doesn't improve over O(n) here for this exact problem."
#
# STEP 5 — Edge cases
#   - Empty list or single rule    → -1 (no conflict possible)
#   - Rules touching (a_hi == b_lo) → overlapping if rule is [lo, hi] inclusive
#   - Rule fully contained         → overlap
#   - No conflicts                 → return -1
#
# Complexity:
#   Naive: O(n²)
#   Linear (max hi tracking): O(n) time, O(1) space
# ─────────────────────────────────────────────────────────────────────────────


# ── APPROACH 1: Naive O(n²) ──────────────────────────────────────────────────

def first_conflict_naive(rules: list) -> int:
    """
    Check each rule against all previous rules.

    Time:  O(n²)
    Space: O(1)
    """
    if not isinstance(rules, list):
        raise TypeError(f"Expected list, got {type(rules).__name__}")

    def overlaps(r1, r2) -> bool:
        return r1[0] <= r2[1] and r2[0] <= r1[1]

    for i in range(1, len(rules)):
        for j in range(i):
            if overlaps(rules[j], rules[i]):
                return i
    return -1


# ── APPROACH 2: Linear scan with max-hi tracking ─────────────────────────────

def first_conflict_linear(rules: list) -> int:
    """
    Track the maximum hi port seen so far. If new rule's start <= max_hi,
    it overlaps with some previous rule.

    Assumes rules are sorted by start port (lo).

    Time:  O(n)
    Space: O(1)
    """
    if not isinstance(rules, list):
        raise TypeError(f"Expected list, got {type(rules).__name__}")
    if not rules:
        return -1

    # Validate input
    for i, r in enumerate(rules):
        if not (isinstance(r, (list, tuple)) and len(r) == 2):
            raise ValueError(f"Rule at index {i} must be [lo, hi], got {r!r}")
        if r[0] > r[1]:
            raise ValueError(f"Rule lo={r[0]} must be <= hi={r[1]}")

    if len(rules) == 1:
        return -1

    max_hi = rules[0][1]    # maximum hi port seen among all previous rules

    for i in range(1, len(rules)):
        lo, hi = rules[i]
        if lo <= max_hi:
            # new rule's start port is within a previously covered range → conflict
            return i
        max_hi = max(max_hi, hi)

    return -1   # no conflict found


# ── APPROACH 3: Binary search on the sorted hi-endpoints list ────────────────
# "This is the reviewer's hint: can we use binary search?
#  We maintain a sorted list of hi-values seen so far.
#  For the new rule [lo, hi]: if bisect_right(hi_list, lo-1) < len(hi_list),
#  then there exists a previous hi >= lo → conflict.
#  Alternatively: max of hi_list >= lo.
#  Since we need the MAX, a sorted list adds O(log n) insert for no benefit
#  over the O(1) running max. So binary search doesn't help here.
#  Worth articulating this reasoning to the reviewer."

import bisect

def first_conflict_with_binary_search(rules: list) -> int:
    """
    Uses a sorted list of hi-values + binary search to check for conflict.
    Same O(n log n) as sorting — demonstrates the approach even though
    the O(n) linear method is strictly better.

    Time:  O(n log n)  due to insort
    Space: O(n)
    """
    if not isinstance(rules, list):
        raise TypeError(f"Expected list, got {type(rules).__name__}")
    if len(rules) <= 1:
        return -1

    seen_hi = [rules[0][1]]   # sorted list of hi values from previous rules

    for i in range(1, len(rules)):
        lo, hi = rules[i]
        # Is there any previous hi >= lo?
        # bisect_left(seen_hi, lo) < len(seen_hi) means some hi >= lo
        idx = bisect.bisect_left(seen_hi, lo)
        if idx < len(seen_hi):
            return i
        bisect.insort(seen_hi, hi)

    return -1


# ── MANUAL — no bisect, no max() ─────────────────────────────────────────────

def first_conflict_manual(rules: list) -> int:
    """
    Linear scan with manual max tracking. No imports.

    Time:  O(n)
    Space: O(1)
    """
    if rules is None:
        raise ValueError("rules cannot be None")
    n = len(rules)
    if n <= 1:
        return -1

    max_hi = rules[0][1]

    i = 1
    while i < n:
        lo = rules[i][0]
        hi = rules[i][1]
        if lo <= max_hi:
            return i
        if hi > max_hi:
            max_hi = hi
        i += 1

    return -1


# ── Tests ─────────────────────────────────────────────────────────────────────

if __name__ == "__main__":
    # Basic: [[1,5],[3,8],...] — rule at index 1 conflicts with rule at index 0
    rules1 = [[1, 5], [3, 8], [10, 15], [20, 25]]
    assert first_conflict_naive(rules1)               == 1
    assert first_conflict_linear(rules1)              == 1
    assert first_conflict_with_binary_search(rules1)  == 1
    assert first_conflict_manual(rules1)              == 1

    # No conflict
    rules2 = [[1, 5], [6, 10], [11, 15]]
    assert first_conflict_linear(rules2) == -1
    assert first_conflict_naive(rules2)  == -1

    # Conflict at index 2 (not index 1)
    rules3 = [[1, 5], [7, 9], [4, 12]]  # [4,12] overlaps with [1,5]
    # Note: rules3 is NOT sorted by start, so results may differ.
    # Sort first for linear/binary search approaches:
    rules3_sorted = sorted(rules3, key=lambda r: r[0])
    # [[1,5],[4,12],[7,9]] — rule [4,12] at index 1 conflicts with [1,5]
    assert first_conflict_linear(rules3_sorted) == 1

    # Touching (hi == lo-1): NOT a conflict (adjacent, non-overlapping)
    rules4 = [[1, 5], [6, 10]]          # 5 and 6 are adjacent, not overlapping
    assert first_conflict_linear(rules4) == -1

    # Touching at same port (lo == hi of previous): overlap
    rules5 = [[1, 5], [5, 10]]          # share port 5 → overlap
    assert first_conflict_linear(rules5) == 1

    # Single and empty
    assert first_conflict_linear([]) == -1
    assert first_conflict_linear([[1, 5]]) == -1

    # All approaches agree on sorted inputs
    test_cases = [rules1, rules2, [[1,3],[2,4],[5,7]]]
    for tc in test_cases:
        tc_sorted = sorted(tc, key=lambda r: r[0])
        naive  = first_conflict_naive(tc_sorted)
        linear = first_conflict_linear(tc_sorted)
        bs     = first_conflict_with_binary_search(tc_sorted)
        manual = first_conflict_manual(tc_sorted)
        assert naive == linear == bs == manual, \
            f"Mismatch on {tc_sorted}: {naive}, {linear}, {bs}, {manual}"

    print("Q24 — All assertions passed.")
    print(f"first_conflict_linear([[1,5],[3,8],[10,15],[20,25]]) = "
          f"{first_conflict_linear([[1,5],[3,8],[10,15],[20,25]])}")  # 1
    print()
    print("Algorithm: O(n) linear scan with running max-hi tracker")
    print("  For each rule i: if rule[i].lo <= max_hi_so_far → conflict at i")
    print("  Binary search: same asymptotic bound; running max is simpler and faster")
