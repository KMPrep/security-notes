# Q25 - Design: What Algorithm Would You Use?
# Five real-world security engineering scenarios. For each, describe the
# best algorithm, complexity, and trade-offs — as in a system design.
#
# ─── SOLUTION WALKTHROUGH ───────────────────────────────────────────────────
#
# SCENARIO 1 — Search sorted blocklist of 1M IPs for a given IP
#   Answer:  Binary search  O(log n) = ~20 comparisons for 1M entries
#   Why:     List is sorted and static (or rarely updated) → binary search ideal
#   Alt:     Hash set for O(1) average lookup — but uses more memory and doesn't
#            preserve sort order. If you need range queries (CIDR blocks),
#            binary search on sorted prefix tree or interval tree.
#   Don't:   Linear scan O(n) — 1M comparisons per lookup is unacceptable.
#
# SCENARIO 2 — Top 10 most active scanners from 50M log entries
#   Answer:  Frequency dict + min-heap of size k=10
#   Step 1:  O(n) pass to build frequency dict: {ip: count}
#   Step 2:  Iterate unique IPs (~u entries), maintain min-heap of size 10
#            → O(u log 10) ≈ O(u) since log(10) is constant
#   Total:   O(n + u) ≈ O(n)
#   Alt:     Counter.most_common(10) — same O(n + u log 10) under the hood
#   Don't:   Sort all unique IPs — O(u log u) — worse, unnecessary.
#
# SCENARIO 3 — Detect exact duplicate events in a stream
#   Answer:  Hash set (seen = set()) — O(1) average per insert and lookup
#   Hash:    Use a canonical hash of (timestamp, source_ip, event_type, payload)
#   Concern: Hash collisions — extremely rare with SHA-256; for security,
#            use cryptographic hash to avoid adversarial collision attacks.
#   Memory:  O(n) — for very large streams, use Bloom filter (probabilistic,
#            O(1) space per element, small false-positive rate, zero false negatives)
#   Don't:   Sort and compare adjacent — O(n log n) and requires buffering.
#
# SCENARIO 4 — Merge 100 sorted streams of 10K events each (1M total)
#   Answer:  k-way merge with min-heap (priority queue)
#   Init:    Push first element of each stream into heap: O(k log k) = O(100 * 7) ≈ O(1)
#   Process: Each pop + push = O(log k); N total events → O(N log k)
#            For N=1M, k=100: O(1M * log 100) = O(1M * 7) ≈ 7M operations
#   Alt:     Sort all 1M events together: O(N log N) = O(1M * 20) ≈ 20M operations
#            k-way merge wins because log k << log N when streams are pre-sorted.
#   Don't:   Merge pairs sequentially (merge sort of streams) — same O(N log k)
#            but more complex to implement.
#
# SCENARIO 5 — Find first rule where priority drops (first inversion)
#   Answer:  Linear scan O(n) — return index of first pair where priority[i] > priority[i+1]
#   Why:     You need to find the FIRST occurrence — must check from the start.
#            No shortcut via binary search because violations aren't monotone.
#   Edge:    If priorities must be strictly increasing and you only need to CHECK
#            if the list is sorted (not find the violation), you still need O(n).
#   Binary search would work for: "Is there any violation in the right half?"
#            But to find the FIRST, you must scan left-to-right regardless.
#
# ─────────────────────────────────────────────────────────────────────────────

import bisect
import heapq
import hashlib
from collections import Counter


# ══ SCENARIO 1: Search sorted blocklist ══════════════════════════════════════

def search_blocklist_linear(blocklist: list, ip: str) -> bool:
    """Naive O(n) — do NOT use in production for large lists."""
    return ip in blocklist


def search_blocklist_binary(blocklist: list, ip: str) -> bool:
    """
    Binary search on sorted blocklist.

    Time:  O(log n)
    Space: O(1)

    Assumes blocklist is sorted (sort semantics: lexicographic for IPs is wrong;
    for correct IP order, store as integers. Here we demonstrate binary search logic.)
    """
    lo, hi = 0, len(blocklist) - 1
    while lo <= hi:
        mid = lo + (hi - lo) // 2
        if blocklist[mid] == ip:
            return True
        elif blocklist[mid] < ip:
            lo = mid + 1
        else:
            hi = mid - 1
    return False


def search_blocklist_set(blocklist: list, ip: str) -> bool:
    """
    Hash set lookup.

    Time:  O(1) average
    Space: O(n) for the set

    Use when you have many lookups and memory is not a constraint.
    Pre-build the set once: O(n). Each lookup: O(1).
    """
    blocklist_set = set(blocklist)
    return ip in blocklist_set


# ══ SCENARIO 2: Top-10 most active scanners ═══════════════════════════════════

def top_k_scanners(log_entries: list, k: int = 10) -> list:
    """
    Build frequency map + maintain min-heap of size k.

    Time:  O(n + u log k)  where u = unique IP count, n = total logs
    Space: O(u)
    """
    if k <= 0:
        return []

    freq = {}
    for ip in log_entries:
        freq[ip] = freq.get(ip, 0) + 1

    # Min-heap of (count, ip) — heap preserves smallest count at root
    heap = []
    for ip, count in freq.items():
        if len(heap) < k:
            heapq.heappush(heap, (count, ip))
        elif count > heap[0][0]:
            heapq.heapreplace(heap, (count, ip))

    # Return sorted descending by count
    return sorted(heap, key=lambda x: -x[0])


def top_k_scanners_builtin(log_entries: list, k: int = 10) -> list:
    """
    Counter.most_common(k) — idiomatic Python, same complexity.

    Time:  O(n + u log k)  via internal heapq.nlargest
    """
    if k <= 0:
        return []
    return Counter(log_entries).most_common(k)


# ══ SCENARIO 3: Detect exact duplicate events ════════════════════════════════

def _event_fingerprint(event: dict) -> str:
    """
    Canonical SHA-256 fingerprint for an event.
    Use cryptographic hash to resist adversarial collisions.
    """
    key = f"{event.get('ts')}:{event.get('src')}:{event.get('type')}:{event.get('payload')}"
    return hashlib.sha256(key.encode()).hexdigest()


def detect_duplicates(events: list) -> list:
    """
    Use a hash set to find duplicate events in O(n) time.

    Time:  O(n)
    Space: O(n)

    Returns list of duplicate events (second and subsequent occurrences).
    """
    seen = set()
    duplicates = []
    for event in events:
        fp = _event_fingerprint(event)
        if fp in seen:
            duplicates.append(event)
        else:
            seen.add(fp)
    return duplicates


# ══ SCENARIO 4: k-way merge of sorted streams ════════════════════════════════

def kway_merge(streams: list) -> list:
    """
    Merge k sorted streams into one sorted output using a min-heap.

    Time:  O(N log k)  where N = total events, k = number of streams
    Space: O(k)  for the heap + O(N) for output

    heap entries: (value, stream_index, element_index)
    """
    if not streams:
        return []

    heap   = []
    result = []

    # Seed the heap with the first element of each non-empty stream
    for si, stream in enumerate(streams):
        if stream:
            heapq.heappush(heap, (stream[0], si, 0))

    while heap:
        val, si, ei = heapq.heappop(heap)
        result.append(val)
        next_ei = ei + 1
        if next_ei < len(streams[si]):
            heapq.heappush(heap, (streams[si][next_ei], si, next_ei))

    return result


# ══ SCENARIO 5: Find first priority inversion ════════════════════════════════

def first_priority_drop(priorities: list) -> int:
    """
    Find index i where priorities[i] > priorities[i+1] (first drop).
    Linear scan — must check from start to find FIRST occurrence.

    Time:  O(n)
    Space: O(1)

    Returns index i (0-based) of the first inversion, or -1 if none.
    """
    if not isinstance(priorities, list):
        raise TypeError(f"Expected list, got {type(priorities).__name__}")

    for i in range(len(priorities) - 1):
        if priorities[i] > priorities[i + 1]:
            return i
    return -1


def first_priority_drop_manual(priorities: list) -> int:
    """Same logic, explicit while loop."""
    n = len(priorities)
    i = 0
    while i < n - 1:
        if priorities[i] > priorities[i + 1]:
            return i
        i += 1
    return -1

if __name__ == "__main__":
    # ── Scenario 1: blocklist search
    blocklist = sorted(["10.0.0.1", "192.168.1.100", "172.16.0.5", "8.8.8.8"])

    assert search_blocklist_binary(blocklist, "1.2.3.4")       is False
    assert search_blocklist_set(blocklist,    "8.8.8.8")        is True
    assert search_blocklist_set(blocklist,    "9.9.9.9")        is False

    # ── Scenario 2: top-k scanners

           ["10.0.0.4"] * 15 + ["10.0.0.5"] * 5
    top3 = top_k_scanners(logs, k=3)
