# Q3 - First and Last Occurrence
# Given a sorted array with duplicates, find the first and last position
# of a given target. Return [-1, -1] if not found.
#
# ─── SOLUTION WALKTHROUGH ───────────────────────────────────────────────────
#
# STEP 1 — Clarify
#   "Array is sorted ascending, can contain duplicates. Target may not exist.
#    Return [-1, -1] when not found, right?"
#
# STEP 2 — Naive approach
#   "Walk left to right to find first occurrence, right to left for last.
#    Or a single pass tracking first/last seen index."
#   Time: O(n) — correct but doesn't use sorted property.
#
# STEP 3 — Optimize with two binary searches
#   "I'll run binary search twice:
#    - First search: when arr[mid] == target, DON'T stop — keep going LEFT
#      (hi = mid - 1) to find the leftmost occurrence.
#    - Second search: when arr[mid] == target, keep going RIGHT
#      (lo = mid + 1) to find the rightmost occurrence.
#    Both are O(log n), total O(log n)."
#
# STEP 4 — Edge cases
#   - Empty array                  → [-1, -1]
#   - Target not present           → [-1, -1]
#   - All elements are target      → [0, n-1]
#   - Single occurrence            → [i, i]
#   - Target at start/end          → must still work
#
# Complexity:
#   Time  — O(log n): two independent binary searches
#   Space — O(1)
# ─────────────────────────────────────────────────────────────────────────────


# ── APPROACH 1: Naive linear scan ────────────────────────────────────────────

def find_first_last_naive(arr: list, target: int) -> list:
    """
    Single-pass linear scan — O(n) time, O(1) space.
    Simple but ignores sorted property.
    """
    if not isinstance(arr, list):
        raise TypeError(f"Expected list, got {type(arr).__name__}")

    first, last = -1, -1
    for i in range(len(arr)):
        if arr[i] == target:
            if first == -1:
                first = i
            last = i
    return [first, last]


# ── APPROACH 2: Two binary searches — manual ─────────────────────────────────

def _find_leftmost(arr: list, target: int) -> int:
    """
    Binary search variant: find leftmost (first) occurrence.
    When arr[mid] == target, record result but continue searching LEFT.
    """
    lo, hi = 0, len(arr) - 1
    result = -1
    while lo <= hi:
        mid = lo + (hi - lo) // 2
        if arr[mid] == target:
            result = mid        # record this candidate
            hi = mid - 1       # keep searching left for an earlier occurrence
        elif arr[mid] < target:
            lo = mid + 1
        else:
            hi = mid - 1
    return result


def _find_rightmost(arr: list, target: int) -> int:
    """
    Binary search variant: find rightmost (last) occurrence.
    When arr[mid] == target, record result but continue searching RIGHT.
    """
    lo, hi = 0, len(arr) - 1
    result = -1
    while lo <= hi:
        mid = lo + (hi - lo) // 2
        if arr[mid] == target:
            result = mid        # record this candidate
            lo = mid + 1       # keep searching right for a later occurrence
        elif arr[mid] < target:
            lo = mid + 1
        else:
            hi = mid - 1
    return result


def find_first_last(arr: list, target: int) -> list:
    """
    Two binary searches for first and last position.

    Time:  O(log n)  — two passes, each O(log n)
    Space: O(1)
    """
    if not isinstance(arr, list):
        raise TypeError(f"Expected list, got {type(arr).__name__}")
    if not arr:
        return [-1, -1]

    first = _find_leftmost(arr, target)
    if first == -1:
        return [-1, -1]     # not found at all — skip the second search

    last = _find_rightmost(arr, target)
    return [first, last]


# ── APPROACH 3: Using bisect module (built-in) ────────────────────────────────

import bisect

def find_first_last_builtin(arr: list, target: int) -> list:
    """
    bisect_left  → index of first occurrence (or insertion point if not found)
    bisect_right → index just AFTER last occurrence

    Time:  O(log n)
    Space: O(1)
    """
    if not isinstance(arr, list):
        raise TypeError(f"Expected list, got {type(arr).__name__}")
    if not arr:
        return [-1, -1]

    left  = bisect.bisect_left(arr, target)
    right = bisect.bisect_right(arr, target) - 1

    # Verify target actually exists at the computed left index
    if left > right or left >= len(arr) or arr[left] != target:
        return [-1, -1]

    return [left, right]


# ── MANUAL — no bisect, no stdlib helpers ────────────────────────────────────

def find_first_last_manual(arr: list, target: int) -> list:
    """
    Same two-search approach written without any imports.

    Time:  O(log n)
    Space: O(1)
    """
    if arr is None:
        raise ValueError("arr cannot be None")
    n = len(arr)
    if n == 0:
        return [-1, -1]

    # --- find first ---
    lo, hi = 0, n - 1
    first = -1
    while lo <= hi:
        mid = lo + (hi - lo) // 2
        if arr[mid] == target:
            first = mid
            hi = mid - 1
        elif arr[mid] < target:
            lo = mid + 1
        else:
            hi = mid - 1

    if first == -1:
        return [-1, -1]

    # --- find last ---
    lo, hi = 0, n - 1
    last = -1
    while lo <= hi:
        mid = lo + (hi - lo) // 2
        if arr[mid] == target:
            last = mid
            lo = mid + 1
        elif arr[mid] < target:
            lo = mid + 1
        else:
            hi = mid - 1

    return [first, last]

if __name__ == "__main__":
    arr = [5, 7, 7, 8, 8, 10]

    assert find_first_last(arr, 8)  == [3, 4]
    assert find_first_last(arr, 6)  == [-1, -1]
    assert find_first_last(arr, 7)  == [1, 2]
    assert find_first_last(arr, 5)  == [0, 0]   # single at start
    assert find_first_last(arr, 10) == [5, 5]   # single at end

    # All same element
    assert find_first_last([2, 2, 2, 2], 2) == [0, 3]

    # Empty
    assert find_first_last([], 8) == [-1, -1]

    # Single element
    assert find_first_last([8], 8)  == [0, 0]
    assert find_first_last([8], 7)  == [-1, -1]

    # Built-in
    assert find_first_last_builtin(arr, 8)  == [3, 4]
    assert find_first_last_builtin(arr, 6)  == [-1, -1]

    # Manual
    assert find_first_last_manual(arr, 8)  == [3, 4]
    assert find_first_last_manual(arr, 6)  == [-1, -1]

    # Naive vs optimized agreement
    for t in [5, 6, 7, 8, 10]:
        assert find_first_last_naive(arr, t) == find_first_last(arr, t), \
            f"Mismatch at target={t}"

    print("Q3 — All assertions passed.")
