# Q4 - Find Peak in Alert Spike Data
# Given a unimodal list (values first increase then decrease), find the
# index of the peak value WITHOUT a linear scan.
#
# ─── SOLUTION WALKTHROUGH ───────────────────────────────────────────────────
#
# STEP 1 — Clarify
#   "Is the array guaranteed unimodal (strictly increases then decreases)?
#    Can the peak be at index 0 or the last index?
#    Are there duplicates?"
#
# STEP 2 — Naive approach
#   "First thought: scan every element, track the index of the max.
#    That's O(n). Correct, but the reviewer said 'without scanning linearly'."
#
# STEP 3 — Binary search insight
#   "At any mid point:
#    - If arr[mid] < arr[mid + 1]: slope is still going UP → peak is to the RIGHT
#    - If arr[mid] > arr[mid + 1]: slope is going DOWN → peak is to the LEFT
#                                  (mid itself could be the peak)
#    This halves the search space each step — O(log n)."
#
# STEP 4 — Edge cases
#   - Single element              → return index 0 (it IS the peak)
#   - Peak at index 0             → strictly decreasing array: index 0
#   - Peak at last index          → strictly increasing: index n-1
#   - Two elements                → max of arr[0], arr[1]
#
# Complexity:
#   Time  — O(log n)
#   Space — O(1)
# ─────────────────────────────────────────────────────────────────────────────


# ── APPROACH 1: Naive linear scan ────────────────────────────────────────────

def find_peak_naive(arr: list) -> int:
    """
    Scan for max — O(n) time, O(1) space.
    Correct but linear; mentioned first before optimizing.
    """
    if not isinstance(arr, list):
        raise TypeError(f"Expected list, got {type(arr).__name__}")
    if not arr:
        raise ValueError("Array must not be empty")

    peak_idx = 0
    for i in range(1, len(arr)):
        if arr[i] > arr[peak_idx]:
            peak_idx = i
    return peak_idx


# ── APPROACH 2: Binary search — manual ───────────────────────────────────────

def find_peak(arr: list) -> int:
    """
    Binary search on unimodal array.

    At each mid, compare arr[mid] with arr[mid + 1]:
      - arr[mid] < arr[mid+1]  → still climbing → peak is in right half
      - arr[mid] > arr[mid+1]  → descending    → peak is in left half (incl. mid)

    We keep lo..hi as the window that definitely contains the peak.
    When lo == hi, that's the peak.

    Time:  O(log n)
    Space: O(1)
    """
    if not isinstance(arr, list):
        raise TypeError(f"Expected list, got {type(arr).__name__}")
    if not arr:
        raise ValueError("Array must not be empty")

    n = len(arr)
    if n == 1:
        return 0

    lo, hi = 0, n - 1

    while lo < hi:
        mid = lo + (hi - lo) // 2
        if arr[mid] < arr[mid + 1]:
            # Slope going up → peak must be to the right of mid
            lo = mid + 1
        else:
            # arr[mid] >= arr[mid+1]: descending or flat → peak at mid or left
            hi = mid

    # lo == hi: this is the peak index
    return lo


# ── APPROACH 3: Using max() built-in ─────────────────────────────────────────
# "Simplest production code — but O(n), which violates the problem constraint."

def find_peak_builtin(arr: list) -> int:
    """
    max() with index — O(n) time, O(1) space.
    Only valid if O(n) is acceptable; violates the O(log n) requirement here.
    """
    if not isinstance(arr, list):
        raise TypeError(f"Expected list, got {type(arr).__name__}")
    if not arr:
        raise ValueError("Array must not be empty")
    return arr.index(max(arr))


# ── MANUAL — no max(), no index(), no enumerate ──────────────────────────────

def find_peak_manual(arr: list) -> int:
    """
    Binary search without any stdlib helpers.
    Same O(log n) logic, fully manual.

    Time:  O(log n)
    Space: O(1)
    """
    if arr is None:
        raise ValueError("arr cannot be None")
    n = len(arr)
    if n == 0:
        raise ValueError("Array must not be empty")
    if n == 1:
        return 0

    lo = 0
    hi = n - 1

    while lo < hi:
        mid = lo + (hi - lo) // 2
        if arr[mid] < arr[mid + 1]:
            lo = mid + 1
        else:
            hi = mid

    return lo


# ── Tests ─────────────────────────────────────────────────────────────────────

if __name__ == "__main__":
    # Problem example
    arr = [1, 3, 7, 12, 9, 5, 2]
    assert find_peak(arr) == 3       # value 12 at index 3

    # Peak at start
    assert find_peak([5, 4, 3, 2, 1]) == 0

    # Peak at end
    assert find_peak([1, 2, 3, 4, 5]) == 4

    # Single element
    assert find_peak([42]) == 0

    # Two elements — ascending
    assert find_peak([1, 3]) == 1

    # Two elements — descending
    assert find_peak([3, 1]) == 0

    # Longer plateau-ish (last highest)
    assert find_peak([1, 5, 10, 8, 3]) == 2

    # Manual
    assert find_peak_manual(arr) == 3
    assert find_peak_manual([5, 4, 3]) == 0

    # Naive agrees with optimized
    for test in [[1, 3, 7, 12, 9, 5, 2], [5, 4, 3], [1, 2, 3], [42]]:
        assert find_peak_naive(test) == find_peak(test), f"Mismatch on {test}"

    # Error guard
    try:
        find_peak([])
        assert False, "Should raise"
    except ValueError:
        pass

    print("Q4 — All assertions passed.")
    print(f"find_peak([1,3,7,12,9,5,2]) = {find_peak(arr)}")       # 3 (value 12)
    print(f"Peak value                  = {arr[find_peak(arr)]}")  # 12
