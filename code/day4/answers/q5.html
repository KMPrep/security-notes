# Q5 - Search in Rotated Sorted Array
# A sorted array of threat scores was rotated at an unknown pivot.
# Find the target without sorting again. Return index or -1.
#
# ─── SOLUTION WALKTHROUGH ───────────────────────────────────────────────────
#
# STEP 1 — Clarify
#   "Is the array guaranteed to have unique values? Can it be empty?
#    Rotation: [4,5,6,7,0,1,2] was originally [0,1,2,4,5,6,7], rotated at 4."
#
# STEP 2 — Naive approach
#   "Find the pivot (min element) with a linear scan, then decide which half
#    to standard-binary-search in. That's O(n) for pivot + O(log n) for search.
#    Dominated by O(n)."
#
# STEP 3 — Slightly better: find pivot in O(log n) first
#   "Binary search for the minimum element. Then do a second binary search
#    in the correct half."
#
# STEP 4 — Best: single-pass binary search
#   "At any mid, one of the two halves [lo..mid] or [mid..hi] MUST be
#    fully sorted (because at most one rotation happened).
#    Identify the sorted half, check if target falls in it, narrow accordingly.
#
#    Two cases:
#    Case A: left half [lo..mid] is sorted  → arr[lo] <= arr[mid]
#      - If arr[lo] <= target < arr[mid]: target is in left half
#      - Else: target is in right half
#    Case B: right half [mid..hi] is sorted → arr[mid] <= arr[hi]
#      - If arr[mid] < target <= arr[hi]: target is in right half
#      - Else: target is in left half"
#
# STEP 5 — Edge cases
#   - Empty array, single element
#   - Target not in array
#   - No rotation (pivot = 0, normal sorted array)
#   - Rotation puts pivot at index 1 or n-1
#
# Complexity:
#   Time  — O(log n): one pass, halves space each iteration
#   Space — O(1)
# ─────────────────────────────────────────────────────────────────────────────


# ── APPROACH 1: Naive linear scan ────────────────────────────────────────────

def search_rotated_naive(arr: list, target: int) -> int:
    """
    Linear scan — O(n) time, O(1) space.
    Correct but ignores structure of the rotated array.
    """
    if not isinstance(arr, list):
        raise TypeError(f"Expected list, got {type(arr).__name__}")
    for i in range(len(arr)):
        if arr[i] == target:
            return i
    return -1


# ── APPROACH 2: Two-step — find pivot via binary search, then search ──────────

def _find_pivot(arr: list) -> int:
    """
    Find index of minimum element (rotation point).
    Returns 0 if array is not rotated.
    Time: O(log n)
    """
    lo, hi = 0, len(arr) - 1
    while lo < hi:
        mid = lo + (hi - lo) // 2
        if arr[mid] > arr[hi]:
            lo = mid + 1        # minimum is in right half
        else:
            hi = mid            # minimum is at mid or in left half
    return lo


def search_rotated_two_step(arr: list, target: int) -> int:
    """
    Find pivot → determine which half → binary search that half.

    Time:  O(log n)
    Space: O(1)
    """
    if not isinstance(arr, list):
        raise TypeError(f"Expected list, got {type(arr).__name__}")
    if not arr:
        return -1

    n = len(arr)
    pivot = _find_pivot(arr)

    # Decide which half to search
    if pivot == 0 or target < arr[0]:
        lo, hi = pivot, n - 1
    else:
        lo, hi = 0, pivot - 1

    while lo <= hi:
        mid = lo + (hi - lo) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            lo = mid + 1
        else:
            hi = mid - 1

    return -1


# ── APPROACH 3: Single-pass binary search (best) ─────────────────────────────

def search_rotated(arr: list, target: int) -> int:
    """
    Single-pass binary search on rotated sorted array.

    At every step, one of the two halves is always fully sorted.
    We identify the sorted half and check if target belongs there.

    Time:  O(log n)
    Space: O(1)
    """
    if not isinstance(arr, list):
        raise TypeError(f"Expected list, got {type(arr).__name__}")
    if not arr:
        return -1

    lo, hi = 0, len(arr) - 1

    while lo <= hi:
        mid = lo + (hi - lo) // 2

        if arr[mid] == target:
            return mid

        # Determine which half is sorted
        if arr[lo] <= arr[mid]:
            # Left half [lo..mid] is sorted
            if arr[lo] <= target < arr[mid]:
                # Target falls within the sorted left half
                hi = mid - 1
            else:
                lo = mid + 1
        else:
            # Right half [mid..hi] is sorted
            if arr[mid] < target <= arr[hi]:
                # Target falls within the sorted right half
                lo = mid + 1
            else:
                hi = mid - 1

    return -1


# ── MANUAL — no stdlib, explicit logic ───────────────────────────────────────

def search_rotated_manual(arr: list, target: int) -> int:
    """
    Same single-pass logic; no imports, fully manual.

    Time:  O(log n)
    Space: O(1)
    """
    if arr is None:
        raise ValueError("arr cannot be None")
    n = len(arr)
    if n == 0:
        return -1

    lo = 0
    hi = n - 1

    while lo <= hi:
        mid = lo + (hi - lo) // 2
        if arr[mid] == target:
            return mid

        if arr[lo] <= arr[mid]:                 # left half sorted
            if arr[lo] <= target < arr[mid]:
                hi = mid - 1
            else:
                lo = mid + 1
        else:                                   # right half sorted
            if arr[mid] < target <= arr[hi]:
                lo = mid + 1
            else:
                hi = mid - 1

    return -1


# ── Tests ─────────────────────────────────────────────────────────────────────

if __name__ == "__main__":
    # Problem examples
    arr1 = [4, 5, 6, 7, 0, 1, 2]
    assert search_rotated(arr1, 0)  == 4
    assert search_rotated(arr1, 3)  == -1

    # No rotation
    arr2 = [1, 2, 3, 4, 5]
    assert search_rotated(arr2, 3)  == 2
    assert search_rotated(arr2, 6)  == -1

    # Rotation at index 1
    arr3 = [2, 0, 1]
    assert search_rotated(arr3, 0)  == 1

    # Rotation at last
    arr4 = [1, 2, 3, 4, 0]
    assert search_rotated(arr4, 0)  == 4
    assert search_rotated(arr4, 1)  == 0

    # Empty and single
    assert search_rotated([], 1)    == -1
    assert search_rotated([1], 1)   == 0
    assert search_rotated([1], 0)   == -1

    # All approaches agree
    for arr, t in [(arr1, 0), (arr1, 3), (arr2, 3), (arr3, 0)]:
        assert search_rotated_naive(arr, t)    == search_rotated(arr, t)
        assert search_rotated_two_step(arr, t) == search_rotated(arr, t)
        assert search_rotated_manual(arr, t)   == search_rotated(arr, t)

    print("Q5 — All assertions passed.")
    print(f"search_rotated([4,5,6,7,0,1,2], 0) = {search_rotated(arr1, 0)}")  # 4
    print(f"search_rotated([4,5,6,7,0,1,2], 3) = {search_rotated(arr1, 3)}")  # -1
