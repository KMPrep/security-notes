# Q6 - Count Elements in Range
# Given a sorted list of timestamps (integers), count how many fall within
# [lo, hi] inclusive using binary search. O(log n) required.
#
# ─── SOLUTION WALKTHROUGH ───────────────────────────────────────────────────
#
# STEP 1 — Clarify
#   "Is the list sorted ascending? Can lo > hi? Can timestamps repeat?
#    Is the range inclusive on both ends?"
#
# STEP 2 — Naive approach
#   "Linear scan: iterate and count every element between lo and hi.
#    That's O(n). Correct but ignores sorted property."
#
# STEP 3 — Optimize with binary search
#   "Two binary searches:
#    - bisect_left(arr, lo)  → first index where arr[i] >= lo  (call it L)
#    - bisect_right(arr, hi) → first index where arr[i] >  hi  (call it R)
#    Count = R - L
#    This is O(log n) per search, total O(log n)."
#
# Why bisect_left for lo and bisect_right for hi?
#   - bisect_left(arr, lo):  leftmost index to insert lo → first element >= lo
#   - bisect_right(arr, hi): rightmost insert position for hi → first el > hi
#   Elements in range are exactly indices [L, R)
#
# STEP 4 — Edge cases
#   - Empty array                       → 0
#   - lo > hi                           → 0 (invalid range)
#   - No elements in range              → 0
#   - All elements in range             → n
#   - lo and hi equal                   → count of that exact value
#
# Complexity:
#   Time  — O(log n)
#   Space — O(1)
# ─────────────────────────────────────────────────────────────────────────────


# ── APPROACH 1: Naive linear scan ────────────────────────────────────────────

def count_in_range_naive(arr: list, lo: int, hi: int) -> int:
    """
    Linear scan — O(n) time, O(1) space.
    Start here; then optimize.
    """
    if not isinstance(arr, list):
        raise TypeError(f"Expected list, got {type(arr).__name__}")
    if lo > hi:
        return 0
    count = 0
    for val in arr:
        if lo <= val <= hi:
            count += 1
    return count


# ── APPROACH 2: Two manual binary searches ───────────────────────────────────

def _bisect_left_manual(arr: list, target: int) -> int:
    """
    Leftmost index where arr[i] >= target (insertion point for target).
    Mirrors bisect.bisect_left.
    """
    lo_idx, hi_idx = 0, len(arr)
    while lo_idx < hi_idx:
        mid = lo_idx + (hi_idx - lo_idx) // 2
        if arr[mid] < target:
            lo_idx = mid + 1
        else:
            hi_idx = mid
    return lo_idx


def _bisect_right_manual(arr: list, target: int) -> int:
    """
    Leftmost index where arr[i] > target (insertion point just after target).
    Mirrors bisect.bisect_right.
    """
    lo_idx, hi_idx = 0, len(arr)
    while lo_idx < hi_idx:
        mid = lo_idx + (hi_idx - lo_idx) // 2
        if arr[mid] <= target:
            lo_idx = mid + 1
        else:
            hi_idx = mid
    return lo_idx


def count_in_range(arr: list, lo: int, hi: int) -> int:
    """
    Count elements in [lo, hi] using two binary searches.

    Time:  O(log n)
    Space: O(1)
    """
    if not isinstance(arr, list):
        raise TypeError(f"Expected list, got {type(arr).__name__}")
    if not arr or lo > hi:
        return 0

    left  = _bisect_left_manual(arr, lo)   # first index >= lo
    right = _bisect_right_manual(arr, hi)  # first index > hi
    return right - left


# ── APPROACH 3: Using bisect module (built-in) ────────────────────────────────

import bisect

def count_in_range_builtin(arr: list, lo: int, hi: int) -> int:
    """
    Production version using bisect module.

    Time:  O(log n)
    Space: O(1)
    """
    if not isinstance(arr, list):
        raise TypeError(f"Expected list, got {type(arr).__name__}")
    if not arr or lo > hi:
        return 0

    left  = bisect.bisect_left(arr, lo)
    right = bisect.bisect_right(arr, hi)
    return right - left


# ── MANUAL — no imports ───────────────────────────────────────────────────────

def count_in_range_manual(arr: list, lo: int, hi: int) -> int:
    """
    Fully manual two-pass binary search; no imports.

    Time:  O(log n)
    Space: O(1)
    """
    if arr is None:
        raise ValueError("arr cannot be None")
    n = len(arr)
    if n == 0 or lo > hi:
        return 0

    # Find first index >= lo
    l, r = 0, n
    while l < r:
        mid = l + (r - l) // 2
        if arr[mid] < lo:
            l = mid + 1
        else:
            r = mid
    left_idx = l

    # Find first index > hi (i.e., >= hi + 1)
    l, r = 0, n
    while l < r:
        mid = l + (r - l) // 2
        if arr[mid] <= hi:
            l = mid + 1
        else:
            r = mid
    right_idx = l

    return right_idx - left_idx


# ── Tests ─────────────────────────────────────────────────────────────────────

if __name__ == "__main__":
    arr = [100, 200, 300, 400, 500]

    # Problem example
    assert count_in_range(arr, 150, 450) == 3  # 200, 300, 400

    # Exact boundary hits
    assert count_in_range(arr, 100, 500) == 5  # all elements
    assert count_in_range(arr, 100, 100) == 1  # single element lo==hi
    assert count_in_range(arr, 200, 200) == 1

    # No elements in range
    assert count_in_range(arr, 101, 199) == 0  # gap between 100 and 200
    assert count_in_range(arr, 501, 600) == 0  # beyond end
    assert count_in_range(arr,   0,  99) == 0  # before start

    # Invalid range
    assert count_in_range(arr, 400, 200) == 0  # lo > hi

    # Empty
    assert count_in_range([], 0, 100) == 0

    # Duplicates
    dup = [100, 200, 200, 300, 300, 300, 400]
    assert count_in_range(dup, 200, 300) == 5  # 200,200,300,300,300

    # All approaches agree
    for lo_t, hi_t in [(150, 450), (100, 100), (0, 99), (501, 600)]:
        assert count_in_range_naive(arr, lo_t, hi_t) == count_in_range(arr, lo_t, hi_t)
        assert count_in_range_builtin(arr, lo_t, hi_t) == count_in_range(arr, lo_t, hi_t)
        assert count_in_range_manual(arr, lo_t, hi_t) == count_in_range(arr, lo_t, hi_t)

    print("Q6 — All assertions passed.")
    print(f"count_in_range([100..500], 150, 450) = {count_in_range(arr, 150, 450)}")  # 3
