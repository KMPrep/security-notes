# Q7 - Sort Log Events by Timestamp (Stable)
# Given a list of (timestamp, event_type, message) tuples, sort by timestamp
# ascending. When timestamps tie, preserve original order (stable sort).
#
# ─── SOLUTION WALKTHROUGH ───────────────────────────────────────────────────
#
# STEP 1 — Clarify
#   "Tuples have (timestamp, event_type, message) in that order?
#    Timestamps are integers (epoch seconds)? Can they repeat?
#    'Stable' means ties preserve insertion order, correct?"
#
# STEP 2 — Naive approach
#   "Write a simple bubble sort or selection sort. O(n²). Works for small logs."
#
# STEP 3 — Optimize
#   "Python's list.sort() uses Timsort — O(n log n) worst case, O(n) on nearly
#    sorted data, and it IS stable by default. That's exactly what we need.
#    We key on the first element (timestamp) only; stability handles ties."
#
# Why is Timsort stable?
#   "When merging runs, equal elements from the left run always come before
#    equal elements from the right run — preserving original index order."
#
# STEP 4 — Manual stable sort
#   "Merge sort is naturally stable if implemented carefully (left element wins
#    ties). I'll implement it manually to show I understand the mechanism."
#
# STEP 5 — Edge cases
#   - Empty list                 → return []
#   - Single event               → unchanged
#   - All same timestamp         → original order preserved
#   - Negative timestamps (epoch offsets before 1970) → still sortable
#
# Complexity:
#   Time  — O(n log n) sort; O(n) for nearly-sorted input (Timsort)
#   Space — O(n) for merge sort auxiliary space; O(1) for in-place sort
# ─────────────────────────────────────────────────────────────────────────────


# ── APPROACH 1: Naive bubble sort (O(n²)) ────────────────────────────────────

def sort_logs_bubble(events: list) -> list:
    """
    Bubble sort — stable, O(n²) time, O(1) extra space.
    Mention for baseline; not production-ready.
    """
    if not isinstance(events, list):
        raise TypeError(f"Expected list, got {type(events).__name__}")
    arr = list(events)          # copy so we don't mutate input
    n = len(arr)
    for i in range(n):
        for j in range(0, n - i - 1):
            if arr[j][0] > arr[j + 1][0]:
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
    return arr


# ── APPROACH 2: Built-in sort (Timsort — production choice) ──────────────────

def sort_logs(events: list) -> list:
    """
    list.sort() / sorted() with key=lambda e: e[0].

    Python's sort is:
      - Timsort: O(n log n) worst, O(n) on sorted/nearly-sorted data
      - Stable: equal timestamps preserve original ordering
      - In-place: O(1) extra space (sort()) or O(n) for sorted()

    Time:  O(n log n)
    Space: O(1) using .sort() in-place; O(n) using sorted()
    """
    if not isinstance(events, list):
        raise TypeError(f"Expected list, got {type(events).__name__}")
    if not events:
        return []

    # Validate each entry
    for i, ev in enumerate(events):
        if not (isinstance(ev, tuple) and len(ev) == 3):
            raise ValueError(f"Event at index {i} must be a 3-tuple, got {ev!r}")

    result = list(events)               # preserve original list
    result.sort(key=lambda e: e[0])     # sort by timestamp only; stable by default
    return result


# ── APPROACH 3: Manual stable merge sort (no built-ins for sort) ──────────────

def _merge(left: list, right: list) -> list:
    """
    Stable merge: when timestamps are equal, left element comes first
    (preserves the relative order from before the split).
    """
    merged = []
    i = j = 0
    while i < len(left) and j < len(right):
        # '<=' ensures stability (left wins on ties)
        if left[i][0] <= right[j][0]:
            merged.append(left[i])
            i += 1
        else:
            merged.append(right[j])
            j += 1
    # Append any remaining elements
    while i < len(left):
        merged.append(left[i])
        i += 1
    while j < len(right):
        merged.append(right[j])
        j += 1
    return merged


def sort_logs_manual(events: list) -> list:
    """
    Stable merge sort without any built-in sort functions.

    Time:  O(n log n)
    Space: O(n) for merge buffer
    """
    if events is None:
        raise ValueError("events cannot be None")
    if len(events) <= 1:
        return list(events)

    mid = len(events) // 2
    left  = sort_logs_manual(events[:mid])
    right = sort_logs_manual(events[mid:])
    return _merge(left, right)


# ── Tests ─────────────────────────────────────────────────────────────────────

if __name__ == "__main__":
    events = [
        (1620000003, "AUTH_FAIL",    "Failed login for user alice"),
        (1620000001, "SCAN",         "Port scan from 10.0.0.1"),
        (1620000002, "AUTH_SUCCESS", "Login succeeded for user bob"),
        (1620000001, "SCAN",         "Port scan from 10.0.0.2"),   # same ts as prev
        (1620000004, "MALWARE",      "Malware detected on host X"),
    ]

    expected_order_ts = [
        1620000001,
        1620000001,   # tie — original order: "10.0.0.1" before "10.0.0.2"
        1620000002,
        1620000003,
        1620000004,
    ]

    # --- sort() built-in ---
    result = sort_logs(events)
    assert [e[0] for e in result] == expected_order_ts
    # Stability check: the two ts=1620000001 events must stay in original order
    ties = [e for e in result if e[0] == 1620000001]
    assert ties[0][2] == "Port scan from 10.0.0.1"
    assert ties[1][2] == "Port scan from 10.0.0.2"

    # --- merge sort manual ---
    result_manual = sort_logs_manual(events)
    assert [e[0] for e in result_manual] == expected_order_ts
    ties_manual = [e for e in result_manual if e[0] == 1620000001]
    assert ties_manual[0][2] == "Port scan from 10.0.0.1"

    # --- bubble sort ---
    result_bubble = sort_logs_bubble(events)
    assert [e[0] for e in result_bubble] == expected_order_ts

    # Edge cases
    assert sort_logs([]) == []
    assert sort_logs([(1, "A", "msg")]) == [(1, "A", "msg")]

    # Error guard
    try:
        sort_logs([(1, "only_two_elements")])
        assert False, "Should raise ValueError"
    except ValueError:
        pass

    print("Q7 — All assertions passed.")
    print("Sorted events (by timestamp, stable):")
    for ev in result:
        print(f"  ts={ev[0]}  type={ev[1]:15s}  msg={ev[2]}")
