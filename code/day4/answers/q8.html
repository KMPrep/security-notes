# Q8 - Top-K Most Active IPs
# Given a list of IP strings from a log, find the K most frequently occurring IPs.
#
# ─── SOLUTION WALKTHROUGH ───────────────────────────────────────────────────
#
# STEP 1 — Clarify
#   "Is K guaranteed <= number of unique IPs? Should I return IPs in frequency
#    order (descending)? What if multiple IPs have the same count — any ordering
#    among ties acceptable?"
#
# STEP 2 — Naive approach
#   "Count frequencies with a dict, sort all unique IPs by count descending,
#    take the first K. That's O(n) to count + O(u log u) to sort all unique IPs
#    (u = unique IPs). If u ≈ n this is O(n log n)."
#
# STEP 3 — Optimize with a min-heap of size K
#   "We don't need to sort ALL IPs — we only want the top K.
#    Maintain a min-heap of size K. For each (ip, count) pair:
#    - Push to heap if size < K
#    - If size == K and new count > heap[0].count, pop min and push new.
#    Final heap contains top-K.
#    Cost: O(n) build freq + O(u log K) heap ops = O(n + u log K)
#    Better than full sort when K << u (K << number of unique IPs)."
#
# STEP 4 — When to use each approach?
#   "If K is close to u: full sort is fine (simpler code).
#    If K is very small (top-5 from 50M IPs): min-heap is significantly better.
#    Python's heapq.nlargest already implements this optimally."
#
# STEP 5 — Edge cases
#   - Empty list                  → return []
#   - K == 0                      → return []
#   - K >= number of unique IPs   → return all IPs sorted by count desc
#   - All IPs same                → return that one IP
#
# Complexity:
#   Full sort: O(n + u log u)  where u = unique IPs
#   Min-heap:  O(n + u log K)
# ─────────────────────────────────────────────────────────────────────────────


# ── APPROACH 1: Naive — dict + full sort ─────────────────────────────────────

def top_k_ips_sort(ips: list, k: int) -> list:
    """
    Count frequencies → sort all unique IPs by count desc → slice [:k].

    Time:  O(n + u log u)  — n to build freq, u log u to sort
    Space: O(u)            — frequency dict
    """
    if not isinstance(ips, list):
        raise TypeError(f"Expected list, got {type(ips).__name__}")
    if not isinstance(k, int) or k < 0:
        raise ValueError(f"k must be a non-negative integer, got {k!r}")
    if not ips or k == 0:
        return []

    freq = {}
    for ip in ips:
        if not isinstance(ip, str):
            raise TypeError(f"Each IP must be a string, got {type(ip).__name__}")
        freq[ip] = freq.get(ip, 0) + 1

    # Sort all unique IPs by count descending
    sorted_ips = sorted(freq.keys(), key=lambda ip: freq[ip], reverse=True)
    return sorted_ips[:k]


# ── APPROACH 2: Min-heap of size K (better when K << u) ─────────────────────

import heapq

def top_k_ips_heap(ips: list, k: int) -> list:
    """
    Maintain a min-heap of size K to track top-K IPs.

    Time:  O(n + u log K)
    Space: O(n)  — freq dict O(u) + heap O(K)
    """
    if not isinstance(ips, list):
        raise TypeError(f"Expected list, got {type(ips).__name__}")
    if not isinstance(k, int) or k < 0:
        raise ValueError(f"k must be a non-negative integer, got {k!r}")
    if not ips or k == 0:
        return []

    # Build frequency map
    freq = {}
    for ip in ips:
        if not isinstance(ip, str):
            raise TypeError(f"Each IP must be a string, got {type(ip).__name__}")
        freq[ip] = freq.get(ip, 0) + 1

    # heapq.nlargest already implements optimal heap strategy
    return [ip for ip, _ in heapq.nlargest(k, freq.items(), key=lambda x: x[1])]


# ── APPROACH 3: Using Counter.most_common() (most Pythonic) ──────────────────

from collections import Counter

def top_k_ips_builtin(ips: list, k: int) -> list:
    """
    Counter.most_common(k) uses heapq internally → O(n + u log K).

    Time:  O(n + u log K)
    Space: O(u)
    """
    if not isinstance(ips, list):
        raise TypeError(f"Expected list, got {type(ips).__name__}")
    if not isinstance(k, int) or k < 0:
        raise ValueError(f"k must be a non-negative integer, got {k!r}")
    if not ips or k == 0:
        return []

    counter = Counter(ips)
    return [ip for ip, _ in counter.most_common(k)]


# ── MANUAL — no heapq, no Counter, manual dict + manual sort ─────────────────

def top_k_ips_manual(ips: list, k: int) -> list:
    """
    Manual frequency dict and manual selection sort for top-K.
    (Selection sort is O(u*K) which is acceptable for demonstrationnstration.)

    Time:  O(n + u*K)  — n for freq, u*K for selection of top-K
    Space: O(u)
    """
    if ips is None:
        raise ValueError("ips cannot be None")
    if k < 0:
        raise ValueError(f"k must be >= 0, got {k}")
    if len(ips) == 0 or k == 0:
        return []

    # Build frequency dict manually
    freq = {}
    for ip in ips:
        if ip in freq:
            freq[ip] += 1
        else:
            freq[ip] = 1

    # Build a list of (ip, count) pairs
    items = []
    for ip in freq:
        items.append((ip, freq[ip]))

    # Select top-K by repeatedly finding and removing the maximum
    result = []
    k = min(k, len(items))
    for _ in range(k):
        max_idx = 0
        for j in range(1, len(items)):
            if items[j][1] > items[max_idx][1]:
                max_idx = j
        result.append(items[max_idx][0])
        items[max_idx], items[-1] = items[-1], items[max_idx]
        items.pop()

    return result
