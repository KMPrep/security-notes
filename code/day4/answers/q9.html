# Q9 - Interval Merge for Firewall Rule Consolidation
# Given a list of port ranges [start, end], merge all overlapping ranges.
#
# ─── SOLUTION WALKTHROUGH ───────────────────────────────────────────────────
#
# STEP 1 — Clarify
#   "Each range is [start, end] inclusive? Can end < start? Can ranges be
#    adjacent (e.g., [1,3] and [4,6] — should those merge? Usually no —
#    they share no port. But [1,3] and [3,6] overlap at port 3, so yes.)
#    Can the input be unsorted?"
#
# STEP 2 — Naive approach
#   "For every interval, compare it with every other interval and merge if
#    they overlap. O(n²). Hard to implement cleanly too."
#
# STEP 3 — Optimized two-step approach
#   "Sort by start port. Then a single linear pass:
#    - Maintain a 'current merged' interval.
#    - For each next interval: if next.start <= current.end → they overlap,
#      extend current.end = max(current.end, next.end).
#    - Else: no overlap, push current to result, start fresh.
#
#    Why does sorting first help?
#    After sorting, any overlapping interval MUST be adjacent in the array
#    — we only need to look at the next interval, not all remaining ones."
#
# STEP 4 — Edge cases
#   - Empty list                           → []
#   - Single interval                      → same interval
#   - All intervals overlap into one       → one giant range
#   - No intervals overlap                 → same list (sorted)
#   - end == start (point interval)        → valid
#   - Unsorted input                       → sort handles it
#   - Duplicate intervals                  → merge to one
#
# Complexity:
#   Time  — O(n log n) for sort + O(n) for merge pass = O(n log n)
#   Space — O(n) for output list (can't do better since output may be n intervals)
# ─────────────────────────────────────────────────────────────────────────────


# ── APPROACH 1: Naive O(n²) ──────────────────────────────────────────────────

def merge_intervals_naive(intervals: list) -> list:
    """
    Brute-force: repeatedly scan for any pair of overlapping intervals and
    merge them until none remain. O(n²) time.
    Mention this to show thought process; not the answer.
    """
    if not isinstance(intervals, list):
        raise TypeError(f"Expected list, got {type(intervals).__name__}")
    if not intervals:
        return []

    arr = [list(iv) for iv in intervals]    # work with copies

    changed = True
    while changed:
        changed = False
        new_arr = []
        used = [False] * len(arr)
        for i in range(len(arr)):
            if used[i]:
                continue
            cur = arr[i]
            for j in range(i + 1, len(arr)):
                if used[j]:
                    continue
                # Overlap: one starts before the other ends
                if cur[0] <= arr[j][1] and arr[j][0] <= cur[1]:
                    cur = [min(cur[0], arr[j][0]), max(cur[1], arr[j][1])]
                    used[j] = True
                    changed = True
            new_arr.append(cur)
        arr = new_arr

    return sorted(arr, key=lambda x: x[0])


# ── APPROACH 2: Sort + linear merge (optimal) ────────────────────────────────

def merge_intervals(intervals: list) -> list:
    """
    Step 1: Sort by start.
    Step 2: Single linear pass — extend current interval or start a new one.

    Time:  O(n log n) sort + O(n) merge = O(n log n)
    Space: O(n) for result
    """
    if not isinstance(intervals, list):
        raise TypeError(f"Expected list, got {type(intervals).__name__}")
    if not intervals:
        return []

    # Validate and copy
    validated = []
    for i, iv in enumerate(intervals):
        if not (isinstance(iv, (list, tuple)) and len(iv) == 2):
            raise ValueError(f"Interval at index {i} must be [start, end], got {iv!r}")
        start, end = int(iv[0]), int(iv[1])
        if start > end:
            raise ValueError(f"Interval start {start} must be <= end {end}")
        validated.append([start, end])

    # Sort by start port
    validated.sort(key=lambda x: x[0])

    merged = [validated[0]]

    for start, end in validated[1:]:
        current = merged[-1]
        if start <= current[1]:
            # Overlapping — extend
            current[1] = max(current[1], end)
        else:
            # No overlap — start new interval
            merged.append([start, end])

    return merged


# ── MANUAL — no sorted(), no sort(), no lambda (fully manual sort) ────────────

def _insertion_sort_by_start(arr: list) -> list:
    """Insertion sort on intervals by start value — O(n²) but avoids .sort()."""
    result = [iv[:] for iv in arr]
    for i in range(1, len(result)):
        key = result[i]
        j = i - 1
        while j >= 0 and result[j][0] > key[0]:
            result[j + 1] = result[j]
            j -= 1
        result[j + 1] = key
    return result


def merge_intervals_manual(intervals: list) -> list:
    """
    Merge intervals without built-in sort.
    Uses manual insertion sort (O(n²) for sort, fine for demonstration).

    Time:  O(n²) due to insertion sort; O(n log n) if swap in built-in sort
    Space: O(n)
    """
    if intervals is None:
        raise ValueError("intervals cannot be None")
    if len(intervals) == 0:
        return []

    arr = [[iv[0], iv[1]] for iv in intervals]
    arr = _insertion_sort_by_start(arr)

    merged = [arr[0][:]]
    i = 1
    while i < len(arr):
        start = arr[i][0]
        end   = arr[i][1]
        current = merged[-1]
        if start <= current[1]:
            # extend
            if end > current[1]:
                current[1] = end
        else:
            merged.append([start, end])
        i += 1

    return merged


# ── Tests ─────────────────────────────────────────────────────────────────────

if __name__ == "__main__":
    # Problem examples
    assert merge_intervals([[1, 3], [2, 6], [8, 10], [15, 18]]) \
        == [[1, 6], [8, 10], [15, 18]]

    assert merge_intervals([[1, 4], [4, 5]]) \
        == [[1, 5]]          # touching intervals merge

    # Edge cases
    assert merge_intervals([]) == []
    assert merge_intervals([[5, 7]]) == [[5, 7]]    # single interval

    # All merge into one
    assert merge_intervals([[1, 10], [2, 6], [8, 15], [12, 20]]) \
        == [[1, 20]]

    # No overlaps (already merged, sorted)
    assert merge_intervals([[1, 2], [4, 5], [7, 8]]) \
        == [[1, 2], [4, 5], [7, 8]]

    # Unsorted input
    assert merge_intervals([[8, 10], [1, 3], [15, 18], [2, 6]]) \
        == [[1, 6], [8, 10], [15, 18]]

    # Manual
    assert merge_intervals_manual([[1, 3], [2, 6], [8, 10], [15, 18]]) \
        == [[1, 6], [8, 10], [15, 18]]

    # Naive agrees (on small inputs)
    test_cases = [
        [[1, 3], [2, 6], [8, 10], [15, 18]],
        [[1, 4], [4, 5]],
        [[1, 2], [4, 5]],
    ]
    for tc in test_cases:
        assert merge_intervals_naive(tc) == merge_intervals(tc), \
            f"Mismatch on {tc}"

    print("Q9 — All assertions passed.")
    print(f"merge_intervals([[1,3],[2,6],[8,10],[15,18]]) = "
          f"{merge_intervals([[1,3],[2,6],[8,10],[15,18]])}")
