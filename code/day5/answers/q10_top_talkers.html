"""
Q10. Top Talkers Report
========================
Given a list of (src_ip, dst_ip, bytes) flow records:
- Top 5 source IPs by total bytes sent.
- Top 5 destination IPs by total bytes received.
- All (src, dst) pairs where bytes > threshold (e.g., 1 GB).

=============================================================================
SOLUTION PROGRESSION
=============================================================================

PROMPT: "Find the heaviest network talkers from a list of flow records."

--- CANDIDATE FIRST RESPONSE (sort the raw list repeatedly) ---
"I'd sort the whole list by bytes for each question."

Follow-up: "That doesn't aggregate — the same IP appears multiple times."
→ Need to sum bytes per IP first, then sort.

--- CANDIDATE SECOND RESPONSE (dict aggregation, then sort) ---
"I'll build a dict: src_ip → total_bytes, then sort it."

Follow-up: "Good. What's more efficient than sorting the entire dict for top-5?"
→ heapq.nlargest — O(N log k) vs O(N log N). For k=5, much faster on large N.

--- FINAL GOOGLE-LEVEL RESPONSE ---
Dict aggregation, top-K with manual heap (without builtins) and heapq (with),
threshold filtering, clean report.

=============================================================================
"""

import sys
import heapq
from collections import defaultdict

# ─────────────────────────────────────────────────────────────────────────────
# SAMPLE DATA  (bytes, realistic scale)
# ─────────────────────────────────────────────────────────────────────────────

GB = 1_073_741_824
MB = 1_048_576

SAMPLE_FLOWS = [
    ("10.0.0.1",  "8.8.8.8",          5 * MB),
    ("10.0.0.2",  "1.1.1.1",         50 * MB),
    ("10.0.0.1",  "8.8.8.8",        200 * MB),
    ("10.0.0.3",  "203.0.113.1",       2 * GB),   # suspicious exfil
    ("10.0.0.2",  "1.1.1.1",        800 * MB),
    ("10.0.0.4",  "8.8.4.4",          3 * MB),
    ("10.0.0.5",  "203.0.113.2",       1.5 * GB), # suspicious exfil
    ("10.0.0.1",  "192.168.1.100",   900 * MB),
    ("10.0.0.6",  "10.0.0.99",       10 * MB),
    ("10.0.0.7",  "8.8.8.8",         20 * MB),
    ("10.0.0.3",  "203.0.113.1",     500 * MB),   # same pair, more bytes
    ("10.0.0.8",  "172.217.0.0",      80 * MB),
    ("10.0.0.9",  "172.217.0.0",     900 * MB),
    ("10.0.0.10", "1.1.1.1",          15 * MB),
    ("10.0.0.2",  "10.0.0.99",       400 * MB),
]

EXFIL_THRESHOLD_BYTES = GB  # Flag (src, dst) pairs exceeding 1 GB total


# ─────────────────────────────────────────────────────────────────────────────
# APPROACH 1 — WITHOUT BUILT-IN HIGH-LEVEL HELPERS
# (manual dict, manual top-K via partial sort, no heapq / sorted key)
# ─────────────────────────────────────────────────────────────────────────────

def _manual_top_k_items(count_dict, k):
    """
    Extract top-k (key, value) pairs by value descending.
    Uses selection (find max k times) — O(N*k) which is fine for small k.
    """
    remaining = list(count_dict.items())
    result = []
    for _ in range(min(k, len(remaining))):
        # Find index of max value
        max_idx = 0
        for i in range(1, len(remaining)):
            if remaining[i][1] > remaining[max_idx][1]:
                max_idx = i
        result.append(remaining[max_idx])
        # Remove it
        remaining[max_idx] = remaining[-1]
        remaining.pop()
    return result


def _bytes_to_human(b: float) -> str:
    """Convert bytes to human-readable string without f-string formatting tricks."""
    gb = b / (1024 ** 3)
    if gb >= 1:
        val = int(gb * 100) / 100  # 2 decimal places manually
        return str(val) + " GB"
    mb = b / (1024 ** 2)
    val = int(mb * 100) / 100
    return str(val) + " MB"


def analyze_without_builtins(flows, threshold_bytes):
    """
    Aggregate flows and compute top talkers.
    flows: list of (src_ip, dst_ip, bytes)
    """
    src_totals  = {}   # src_ip   → total bytes sent
    dst_totals  = {}   # dst_ip   → total bytes received
    pair_totals = {}   # (src, dst) → total bytes

    for item in flows:
        if not (isinstance(item, (list, tuple)) and len(item) == 3):
            print(f"[WARN] Skipping malformed flow record: {item}", file=sys.stderr)
            continue
        src, dst, byte_count = item

        # Validate
        if not isinstance(byte_count, (int, float)) or byte_count < 0:
            print(f"[WARN] Invalid byte count for ({src},{dst}): {byte_count}",
                  file=sys.stderr)
            continue

        src_totals[src]       = src_totals.get(src, 0) + byte_count
        dst_totals[dst]       = dst_totals.get(dst, 0) + byte_count
        pair                  = (src, dst)
        pair_totals[pair]     = pair_totals.get(pair, 0) + byte_count

    top5_src  = _manual_top_k_items(src_totals,  5)
    top5_dst  = _manual_top_k_items(dst_totals,  5)

    # Heavy pairs: all (src, dst) with total > threshold
    heavy_pairs = []
    for pair, total in pair_totals.items():
        if total > threshold_bytes:
            heavy_pairs.append((pair[0], pair[1], total))

    # Sort heavy_pairs descending by bytes (manual insertion sort)
    for i in range(1, len(heavy_pairs)):
        cur = heavy_pairs[i]
        j = i - 1
        while j >= 0 and heavy_pairs[j][2] < cur[2]:
            heavy_pairs[j + 1] = heavy_pairs[j]
            j -= 1
        heavy_pairs[j + 1] = cur

    return top5_src, top5_dst, heavy_pairs


# ─────────────────────────────────────────────────────────────────────────────
# APPROACH 2 — WITH BUILT-IN HELPERS (defaultdict, heapq.nlargest, sorted)
# ─────────────────────────────────────────────────────────────────────────────

def analyze_with_builtins(flows, threshold_bytes):
    src_totals  = defaultdict(int)
    dst_totals  = defaultdict(int)
    pair_totals = defaultdict(int)

    for item in flows:
        if not (isinstance(item, (list, tuple)) and len(item) == 3):
            print(f"[WARN] Skipping malformed flow record: {item}", file=sys.stderr)
            continue
        src, dst, byte_count = item

        if not isinstance(byte_count, (int, float)) or byte_count < 0:
            print(f"[WARN] Invalid byte count for ({src},{dst}): {byte_count}",
                  file=sys.stderr)
            continue

        src_totals[src]     += byte_count
        dst_totals[dst]     += byte_count
        pair_totals[(src, dst)] += byte_count

    # heapq.nlargest is O(N log k) — better than sorting the entire dict
    top5_src = heapq.nlargest(5, src_totals.items(),  key=lambda x: x[1])
    top5_dst = heapq.nlargest(5, dst_totals.items(),  key=lambda x: x[1])

    heavy_pairs = sorted(
        [(src, dst, total) for (src, dst), total in pair_totals.items()
         if total > threshold_bytes],
        key=lambda x: x[2],
        reverse=True
    )

    return top5_src, top5_dst, heavy_pairs


# ─────────────────────────────────────────────────────────────────────────────
# REPORT
# ─────────────────────────────────────────────────────────────────────────────

def _fmt_bytes(b: float) -> str:
    gb = b / (1024 ** 3)
    if gb >= 1:
        return f"{gb:.2f} GB"
    return f"{b / (1024**2):.2f} MB"


def print_report(top5_src, top5_dst, heavy_pairs, threshold_bytes, label=""):
    print(f"\n{'='*65}")
    print(f"  TOP TALKERS REPORT  {label}")
    print(f"{'='*65}")

    print("\nTop 5 Source IPs by Bytes Sent:")
    for rank, (ip, total) in enumerate(top5_src, 1):
        print(f"  {rank}. {ip:<20} {_fmt_bytes(total)}")

    print("\nTop 5 Destination IPs by Bytes Received:")
    for rank, (ip, total) in enumerate(top5_dst, 1):
        print(f"  {rank}. {ip:<20} {_fmt_bytes(total)}")

    threshold_str = _fmt_bytes(threshold_bytes)
    print(f"\n(src, dst) Pairs Exceeding {threshold_str}:")
    if heavy_pairs:
        for src, dst, total in heavy_pairs:
            print(f"  {src:<20} → {dst:<20} : {_fmt_bytes(total)}")
    else:
        print("  None.")
    print()


# ─────────────────────────────────────────────────────────────────────────────
# MAIN
# ─────────────────────────────────────────────────────────────────────────────

def main():
    print("[INFO] Using built-in sample data.\n")

    top5_src_manual,  top5_dst_manual,  heavy_manual  = analyze_without_builtins(
        SAMPLE_FLOWS, EXFIL_THRESHOLD_BYTES)
    top5_src_builtin, top5_dst_builtin, heavy_builtin = analyze_with_builtins(
        SAMPLE_FLOWS, EXFIL_THRESHOLD_BYTES)

    print_report(top5_src_manual,  top5_dst_manual,  heavy_manual,
                 EXFIL_THRESHOLD_BYTES, label="(Manual / No Builtins)")
    print_report(top5_src_builtin, top5_dst_builtin, heavy_builtin,
                 EXFIL_THRESHOLD_BYTES, label="(With Builtins)")

    # Sanity: top-5 sets should be equal (order may differ between manual and heapq)
    assert set(ip for ip, _ in top5_src_manual) == set(ip for ip, _ in top5_src_builtin), \
        "Top-5 src mismatch!"
    assert len(heavy_manual) == len(heavy_builtin), \
        f"Heavy pair count mismatch: {len(heavy_manual)} vs {len(heavy_builtin)}"
    print("[OK] Both approaches agree.\n")

    # Edge case tests
    print("--- Edge Case Tests ---")
    print("Empty flows:", analyze_with_builtins([], GB))
    print("Single flow:", analyze_with_builtins([("1.1.1.1", "2.2.2.2", 5 * MB)], GB))
    print()


if __name__ == "__main__":
    main()

# Top-K       : O(N log k) with heapq.nlargest; O(N log N) with full sort

# Heavy pairs : O(P)  — P = unique (src,dst) pairs

# GOOGLE FOLLOW-UP QUESTIONS:
# - "The flow log has 500 million records. How do you handle it?"
#     → Stream aggregate: read line by line, maintain running totals in dicts.
#       Use only O(U) space where U = unique IPs/pairs. No need to load all records.
# - "What if you need top-K in real-time as flows arrive?"
#     → Use a min-heap of size k. For each new flow: update aggregate,
#       if the IP's new total exceeds the heap's minimum, replace it.
# - "How would you handle private vs public IP distinction?"
#     → Use ipaddress.ip_address(ip).is_private to classify. Exfil typically
#       means private src → public dst with high byte count.
# - "Can the same (src, dst) pair appear across multiple days of logs?"
#     → Yes. When loading multiple files, continue aggregating into the same dicts.
