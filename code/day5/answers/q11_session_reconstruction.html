"""
Q11. Session Reconstruction
============================
Given (user, timestamp, action) events sorted by timestamp:
- Group consecutive events per user into sessions (gap > 30 min = new session).
- For each session report: user, session start, session end, actions taken.
- Flag sessions with any "DELETE" or "EXPORT" action.

=============================================================================
SOLUTION PROGRESSION
=============================================================================

PROMPT: "Reconstruct user sessions from an event log."

--- CANDIDATE FIRST RESPONSE (group by user, ignore gaps) ---
"I'd group all events by user and that's one session per user."

Follow-up: "What about the 30-minute gap rule?"
→ Need to track the PREVIOUS event's timestamp per user and start a new
  session when the gap exceeds 30 minutes.

Follow-up: "Are events guaranteed to be globally sorted by time?"
→ Not necessarily — need to sort per-user by timestamp before reconstructing.

Follow-up: "What if the timestamp format is inconsistent?"
→ Parse to a numeric (epoch) value first; compare numerically.

--- CANDIDATE SECOND RESPONSE (sort per user, gap-based session split) ---
Better. Follow-up: "How do you handle two users whose sessions interleave
in the original log?"
→ Already handled by sorting per-user independently — global order doesn't matter.

--- FINAL GOOGLE-LEVEL RESPONSE ---
Parse timestamps to epoch, sort per user, session reconstruction with gap
rule, flag suspicious sessions, clean structured output.

=============================================================================
"""

import re
import sys
from datetime import datetime, timezone
from collections import defaultdict

# ─────────────────────────────────────────────────────────────────────────────
# SAMPLE DATA
# ─────────────────────────────────────────────────────────────────────────────

# (user, ISO-timestamp, action)
SAMPLE_EVENTS = [
    ("alice", "2024-01-15T09:00:00Z", "LOGIN"),
    ("bob",   "2024-01-15T09:01:00Z", "LOGIN"),
    ("alice", "2024-01-15T09:05:00Z", "VIEW"),
    ("alice", "2024-01-15T09:10:00Z", "DOWNLOAD"),
    ("bob",   "2024-01-15T09:15:00Z", "VIEW"),
    ("alice", "2024-01-15T09:20:00Z", "EXPORT"),   # suspicious
    ("bob",   "2024-01-15T09:45:00Z", "DELETE"),   # suspicious
    ("alice", "2024-01-15T10:00:00Z", "LOGOUT"),
    # Alice logs in again 40 min later — new session
    ("alice", "2024-01-15T10:40:00Z", "LOGIN"),
    ("alice", "2024-01-15T10:45:00Z", "VIEW"),
    ("alice", "2024-01-15T10:50:00Z", "DELETE"),   # suspicious, second session
    ("alice", "2024-01-15T10:55:00Z", "LOGOUT"),
    ("bob",   "2024-01-15T09:50:00Z", "LOGOUT"),
    # Charlie - single session
    ("charlie", "2024-01-15T11:00:00Z", "LOGIN"),
    ("charlie", "2024-01-15T11:05:00Z", "VIEW"),
    ("charlie", "2024-01-15T11:10:00Z", "LOGOUT"),
]

SESSION_GAP_SECONDS = 30 * 60   # 30 minutes
SUSPICIOUS_ACTIONS  = {"DELETE", "EXPORT"}

# ─────────────────────────────────────────────────────────────────────────────
# TIMESTAMP PARSER
# ─────────────────────────────────────────────────────────────────────────────

def _parse_ts(ts_str: str) -> float:
    """Parse ISO-8601 timestamp to Unix epoch float. Returns -1.0 on failure."""
    try:
        ts_str = ts_str.strip().replace("Z", "+00:00")
        dt = datetime.fromisoformat(ts_str)
        if dt.tzinfo is None:
            dt = dt.replace(tzinfo=timezone.utc)
        return dt.timestamp()
    except (ValueError, AttributeError):
        return -1.0


def _epoch_to_iso(epoch: float) -> str:
    """Convert epoch float back to ISO-8601 UTC string."""
    return datetime.fromtimestamp(epoch, tz=timezone.utc).strftime(
        "%Y-%m-%dT%H:%M:%SZ"
    )


# ─────────────────────────────────────────────────────────────────────────────
# APPROACH 1 — WITHOUT BUILT-IN HIGH-LEVEL HELPERS
# (manual grouping, manual sort, manual session split, no defaultdict)
# ─────────────────────────────────────────────────────────────────────────────

def _manual_sort_events(events):
    """Insertion sort by timestamp (float) ascending."""
    for i in range(1, len(events)):
        cur = events[i]
        j = i - 1
        while j >= 0 and events[j]["ts_epoch"] > cur["ts_epoch"]:
            events[j + 1] = events[j]
            j -= 1
        events[j + 1] = cur


def reconstruct_without_builtins(raw_events):
    """
    raw_events: list of (user, timestamp_str, action)
    Returns list of session dicts.
    """
    malformed = 0
    # Step 1: Parse and group by user (manual dict)
    user_events = {}
    for item in raw_events:
        if not (isinstance(item, (list, tuple)) and len(item) == 3):
            print(f"[WARN] Malformed event record: {item}", file=sys.stderr)
            malformed += 1
            continue
        user, ts_str, action = item
        ts = _parse_ts(ts_str)
        if ts < 0:
            print(f"[WARN] Bad timestamp '{ts_str}' for user {user}, skipping.",
                  file=sys.stderr)
            malformed += 1
            continue

        ev = {"user": user, "ts_epoch": ts, "ts_str": ts_str, "action": action}
        if user not in user_events:
            user_events[user] = []
        user_events[user].append(ev)

    # Step 2: Sort each user's events by timestamp
    for user in user_events:
        _manual_sort_events(user_events[user])

    # Step 3: Reconstruct sessions
    sessions = []
    session_id = 0

    for user, events in user_events.items():
        if not events:
            continue

        # Start the first session
        session_actions = [events[0]["action"]]
        session_start   = events[0]["ts_epoch"]
        session_end     = events[0]["ts_epoch"]

        for i in range(1, len(events)):
            ev   = events[i]
            gap  = ev["ts_epoch"] - events[i - 1]["ts_epoch"]

            if gap > SESSION_GAP_SECONDS:
                # Close current session
                session_id += 1
                is_suspicious = False
                for act in session_actions:
                    if act in SUSPICIOUS_ACTIONS:
                        is_suspicious = True
                        break

                sessions.append({
                    "id":           session_id,
                    "user":         user,
                    "start":        _epoch_to_iso(session_start),
                    "end":          _epoch_to_iso(session_end),
                    "actions":      session_actions,
                    "suspicious":   is_suspicious,
                    "duration_min": round((session_end - session_start) / 60, 1),
                })

                # Start new session
                session_actions = [ev["action"]]
                session_start   = ev["ts_epoch"]
                session_end     = ev["ts_epoch"]
            else:
                session_actions.append(ev["action"])
                session_end = ev["ts_epoch"]

        # Close last session
        session_id += 1
        is_suspicious = False
        for act in session_actions:
            if act in SUSPICIOUS_ACTIONS:
                is_suspicious = True
                break
        sessions.append({
            "id":           session_id,
            "user":         user,
            "start":        _epoch_to_iso(session_start),
            "end":          _epoch_to_iso(session_end),
            "actions":      session_actions,
            "suspicious":   is_suspicious,
            "duration_min": round((session_end - session_start) / 60, 1),
        })

    return sessions, malformed


# ─────────────────────────────────────────────────────────────────────────────
# APPROACH 2 — WITH BUILT-IN HELPERS (defaultdict, sorted, any())
# ─────────────────────────────────────────────────────────────────────────────

def reconstruct_with_builtins(raw_events):
    malformed  = 0
    user_events = defaultdict(list)

    for item in raw_events:
        if not (isinstance(item, (list, tuple)) and len(item) == 3):
            print(f"[WARN] Malformed event: {item}", file=sys.stderr)
            malformed += 1
            continue
        user, ts_str, action = item
        ts = _parse_ts(ts_str)
        if ts < 0:
            print(f"[WARN] Bad timestamp '{ts_str}' for {user}.", file=sys.stderr)
            malformed += 1
            continue
        user_events[user].append({"ts": ts, "ts_str": ts_str, "action": action})

    sessions   = []
    session_id = 0

    for user, events in user_events.items():
        events = sorted(events, key=lambda e: e["ts"])
        if not events:
            continue

        start   = events[0]["ts"]
        end     = events[0]["ts"]
        actions = [events[0]["action"]]

        for i in range(1, len(events)):
            gap = events[i]["ts"] - events[i - 1]["ts"]
            if gap > SESSION_GAP_SECONDS:
                session_id += 1
                sessions.append({
                    "id":           session_id,
                    "user":         user,
                    "start":        _epoch_to_iso(start),
                    "end":          _epoch_to_iso(end),
                    "actions":      actions,
                    "suspicious":   any(a in SUSPICIOUS_ACTIONS for a in actions),
                    "duration_min": round((end - start) / 60, 1),
                })
                start   = events[i]["ts"]
                end     = events[i]["ts"]
                actions = [events[i]["action"]]
            else:
                actions.append(events[i]["action"])
                end = events[i]["ts"]

        session_id += 1
        sessions.append({
            "id":           session_id,
            "user":         user,
            "start":        _epoch_to_iso(start),
            "end":          _epoch_to_iso(end),
            "actions":      actions,
            "suspicious":   any(a in SUSPICIOUS_ACTIONS for a in actions),
            "duration_min": round((end - start) / 60, 1),
        })

    return sessions, malformed


# ─────────────────────────────────────────────────────────────────────────────
# REPORT
# ─────────────────────────────────────────────────────────────────────────────

def print_report(sessions, malformed, label=""):
    print(f"\n{'='*70}")
    print(f"  SESSION RECONSTRUCTION REPORT  {label}")
    print(f"{'='*70}")
    print(f"Malformed events skipped: {malformed}")
    print(f"Session gap threshold   : {SESSION_GAP_SECONDS // 60} minutes")
    print(f"Total sessions found    : {len(sessions)}\n")

    flagged = [s for s in sessions if s["suspicious"]]
    normal  = [s for s in sessions if not s["suspicious"]]

    print(f"All Sessions ({len(sessions)}):")
    for s in sorted(sessions, key=lambda x: (x["user"], x["start"])):
        flag_str = " *** SUSPICIOUS ***" if s["suspicious"] else ""
        print(f"  [{s['id']:>2}] {s['user']:<12} {s['start']} → {s['end']}  "
              f"({s['duration_min']} min)  actions={s['actions']}{flag_str}")

    print(f"\nSuspicious Sessions ({len(flagged)}):")
    if flagged:
        for s in flagged:
            suspect_acts = [a for a in s["actions"] if a in SUSPICIOUS_ACTIONS]
            print(f"  User: {s['user']}  |  {s['start']} → {s['end']}"
                  f"  |  Suspicious actions: {suspect_acts}")
    else:
        print("  None.")
    print()


# ─────────────────────────────────────────────────────────────────────────────
# MAIN
# ─────────────────────────────────────────────────────────────────────────────

def main():
    print("[INFO] Using built-in sample data.\n")

    sessions_manual,  mal_manual  = reconstruct_without_builtins(SAMPLE_EVENTS)
    sessions_builtin, mal_builtin = reconstruct_with_builtins(SAMPLE_EVENTS)

    print_report(sessions_manual,  mal_manual,  label="(Manual / No Builtins)")
    print_report(sessions_builtin, mal_builtin, label="(With Builtins)")

    # Sanity check
    assert len(sessions_manual) == len(sessions_builtin), \
        f"Session count mismatch: {len(sessions_manual)} vs {len(sessions_builtin)}"
    manual_suspicious  = sorted(s["user"] for s in sessions_manual  if s["suspicious"])
    builtin_suspicious = sorted(s["user"] for s in sessions_builtin if s["suspicious"])
    assert manual_suspicious == builtin_suspicious, \
        f"Suspicious user mismatch: {manual_suspicious} vs {builtin_suspicious}"
    print("[OK] Both approaches agree.\n")


if __name__ == "__main__":
    main()
