"""
Q13. Rule-Based Log Tagger
===========================
Given (tag, regex) detection rules and a log file:
- For each line, find all matching tags.
- Output (line_number, line, [tags]) for any line with at least one match.

rules = [
    ("brute_force", r"Failed password.*(\d+) times"),
    ("privesc",     r"sudo:.*COMMAND=/bin/bash"),
    ("exfil",       r"bytes_sent=[0-9]{7,}"),
]

=============================================================================
SOLUTION PROGRESSION
=============================================================================

PROMPT: "Build a detection engine: apply a ruleset to each log line."

--- CANDIDATE FIRST RESPONSE (compile regex per line, per rule) ---
"For each line, for each rule, do re.search(pattern, line)."

Follow-up: "What's wrong with compiling the regex inside the loop?"
→ Python re.compile caches patterns internally, but explicitly compiling
  ONCE before the loop is clearer and guarantees no recompilation cost.

Follow-up: "What if two rules match the same part of a line? Should both fire?"
→ Yes — a line can have multiple tags. We're not doing exclusive matching.

Follow-up: "What if a rule's regex is malformed (e.g., unclosed parenthesis)?"
→ re.compile() raises re.error. We should validate rules at startup and
  skip/reject invalid ones rather than crashing mid-run.

Follow-up: "What about performance with 10,000 rules on 10M log lines?"
→ Compile all patterns once. Consider combining patterns into one alternation
  regex (Aho-Corasick for substring matching, or combined regex).
  For Google-scale: use a YARA-like engine or distribute across workers.

--- FINAL GOOGLE-LEVEL RESPONSE ---
Pre-compile all rules, validate each pattern, tag multiple matches per line,
streaming input, structured output.

=============================================================================
"""

import re
import sys

# ─────────────────────────────────────────────────────────────────────────────
# DETECTION RULES
# ─────────────────────────────────────────────────────────────────────────────

RULES = [
    ("brute_force",       r"Failed password.*(\d+) times"),
    ("failed_login",      r"Failed password"),
    ("privesc",           r"sudo:.*COMMAND=/bin/bash"),
    ("invalid_user",      r"[Ii]nvalid user"),
    ("exfil",             r"bytes_sent=[0-9]{7,}"),
    ("large_upload",      r"bytes_out=[0-9]{8,}"),
    ("lateral_movement",  r"ssh.*\b(?:10|172|192)\.\d+\.\d+\.\d+"),
    ("INVALID_RULE",      r"(unclosed"),   # intentionally bad — should be rejected gracefully
]

# ─────────────────────────────────────────────────────────────────────────────
# SAMPLE LOG
# ─────────────────────────────────────────────────────────────────────────────

SAMPLE_LOG = """\
Jan 15 10:00:01 host sshd[100]: Failed password for root from 10.0.0.1 port 22 ssh2
Jan 15 10:00:02 host sshd[100]: Failed password for root from 10.0.0.1 5 times
Jan 15 10:00:03 host sudo[200]:   alice : TTY=pts/0 ; COMMAND=/bin/bash
Jan 15 10:00:04 host nginx[300]: 10.0.0.5 GET /api/download bytes_sent=10500000
Jan 15 10:00:05 host sshd[101]: Invalid user admin from 203.0.113.5
Jan 15 10:00:06 host app[400]: upload bytes_out=250000000 dst=198.51.100.1
Jan 15 10:00:07 host sshd[102]: Accepted publickey for alice from 10.0.0.2
Jan 15 10:00:08 host kernel: normal log line, nothing suspicious here
Jan 15 10:00:09 host sshd[103]: ssh connection from 192.168.1.200 established
Jan 15 10:00:10 host sudo[201]:   bob : TTY=pts/1 ; COMMAND=/usr/bin/vi
"""

# ─────────────────────────────────────────────────────────────────────────────
# RULE COMPILER
# ─────────────────────────────────────────────────────────────────────────────

def compile_rules(rules):
    """
    Pre-compile all rules. Skip and warn on invalid patterns.
    Returns list of (tag, compiled_pattern).
    """
    compiled = []
    for tag, pattern in rules:
        try:
            compiled.append((tag, re.compile(pattern)))
        except re.error as exc:
            print(f"[WARN] Rule '{tag}' has invalid regex '{pattern}': {exc}. "
                  f"Skipping.", file=sys.stderr)
    return compiled


# ─────────────────────────────────────────────────────────────────────────────
# APPROACH 1 — WITHOUT BUILT-IN HIGH-LEVEL HELPERS
# (manual tag deduplication, no list comprehension, no any())
# ─────────────────────────────────────────────────────────────────────────────

def tag_lines_without_builtins(lines, compiled_rules):
    """
    Returns list of (line_number, line_text, [tags]).
    Only includes lines that match at least one rule.
    """
    results = []

    for line_num, raw in enumerate(lines, start=1):
        line = raw.rstrip("\n")
        if not line.strip():
            continue

        matched_tags = []
        for tag, pattern in compiled_rules:
            if pattern.search(line):
                # Manual dedup
                already = False
                for t in matched_tags:
                    if t == tag:
                        already = True
                        break
                if not already:
                    matched_tags.append(tag)

        if matched_tags:
            results.append((line_num, line, matched_tags))

    return results


# ─────────────────────────────────────────────────────────────────────────────
# APPROACH 2 — WITH BUILT-IN HELPERS (list comprehension, dict.fromkeys dedup)
# ─────────────────────────────────────────────────────────────────────────────

def tag_lines_with_builtins(lines, compiled_rules):
    results = []

    for line_num, raw in enumerate(lines, start=1):
        line = raw.rstrip("\n")
        if not line.strip():
            continue

        # dict.fromkeys preserves insertion order and deduplicates
        matched_tags = list(dict.fromkeys(
            tag for tag, pattern in compiled_rules
            if pattern.search(line)
        ))

        if matched_tags:
            results.append((line_num, line, matched_tags))

    return results


# ─────────────────────────────────────────────────────────────────────────────
# REPORT
# ─────────────────────────────────────────────────────────────────────────────

def print_report(results, total_lines, label=""):
    print(f"\n{'='*70}")
    print(f"  RULE-BASED LOG TAGGER  {label}")
    print(f"{'='*70}")
    print(f"Total lines processed  : {total_lines}")
    print(f"Lines with detections  : {len(results)}\n")

    if not results:
        print("  No detections.\n")
        return

    # Tag frequency summary (manual count for clarity)
    tag_freq = {}
    for _, _, tags in results:
        for tag in tags:
            tag_freq[tag] = tag_freq.get(tag, 0) + 1

    print("Tag Frequency:")
    for tag, count in sorted(tag_freq.items(), key=lambda x: -x[1]):
        print(f"  {tag:<25} {count} line(s)")

    print(f"\nDetected Lines:")
    for line_num, line, tags in results:
        tags_str = ", ".join(tags)
        # Truncate long lines for display
        display = line if len(line) <= 80 else line[:77] + "..."
        print(f"  Line {line_num:>4} [{tags_str}]")
        print(f"         {display}")
    print()


# ─────────────────────────────────────────────────────────────────────────────
# STREAMING FILE READER
# ─────────────────────────────────────────────────────────────────────────────

def stream_lines(filepath: str):
    try:
        with open(filepath, "r", encoding="utf-8", errors="replace") as fh:
            for line in fh:
                yield line
    except FileNotFoundError:
        print(f"[ERROR] File not found: {filepath}", file=sys.stderr)
    except PermissionError:
        print(f"[ERROR] Permission denied: {filepath}", file=sys.stderr)
    except OSError as exc:
        print(f"[ERROR] {exc}", file=sys.stderr)


# ─────────────────────────────────────────────────────────────────────────────
# MAIN
# ─────────────────────────────────────────────────────────────────────────────

def main():
    # Compile rules once — invalid rules are skipped with a warning
    compiled_rules = compile_rules(RULES)
    print(f"[INFO] Loaded {len(compiled_rules)} valid rule(s) "
          f"(skipped {len(RULES) - len(compiled_rules)} invalid).\n")

    if len(sys.argv) > 1:
        lines = list(stream_lines(sys.argv[1]))
        if not lines:
            print("[WARN] No lines read.", file=sys.stderr)
            sys.exit(1)
    else:
        print("[INFO] No file provided — using built-in sample data.\n")
        lines = SAMPLE_LOG.strip().splitlines(keepends=True)

    results_manual  = tag_lines_without_builtins(lines, compiled_rules)
    results_builtin = tag_lines_with_builtins(lines, compiled_rules)

    total_lines = sum(1 for raw in lines if raw.strip())

    print_report(results_manual,  total_lines, label="(Manual / No Builtins)")
    print_report(results_builtin, total_lines, label="(With Builtins)")

    # Sanity check
    assert len(results_manual) == len(results_builtin), \
        f"Result count mismatch: {len(results_manual)} vs {len(results_builtin)}"
    for m, b in zip(results_manual, results_builtin):
        assert m[0] == b[0],        f"Line number mismatch: {m[0]} vs {b[0]}"
        assert sorted(m[2]) == sorted(b[2]), \
            f"Tags mismatch at line {m[0]}: {m[2]} vs {b[2]}"
    print("[OK] Both approaches agree.\n")


if __name__ == "__main__":
    main()


# =============================================================================
# COMPLEXITY
# =============================================================================
# Time  : O(L * R) where L = lines, R = number of rules
#         Each rule is a compiled regex: O(|line|) per rule per line.
# Space : O(L_flagged) for storing flagged results
#
# SCALING APPROACHES:
# 1. Combine all regexes into one alternation with named groups:
#    pattern = re.compile("|".join(f"(?P<{tag}>{pat})" for tag, pat in rules))
#    Then scan once per line → O(L * |combined pattern|)
#    Drawback: Python's re engine recompiles combined patterns slowly for large R.
#
# 2. For substring rules (no anchors): use Aho-Corasick (pyahocorasick).
#    O(L * |line| + R * |pattern|) preprocessing, O(|line|) per line matching.
#
# 3. For Google-scale: use a compiled YARA ruleset. YARA is optimized for
#    exactly this use case and can process millions of lines/sec.
#
# GOOGLE FOLLOW-UP QUESTIONS:
# - "A rule has a regex with catastrophic backtracking. How do you detect it?"
#     → Use a timeout wrapper around re.search (signal.alarm on Unix, or
#       run in a separate thread with a timeout). Consider using re2 (Google's
#       library) which guarantees O(n) matching with no backtracking.
# - "Rules come from user input (analyst-written). What are the security risks?"
#     → ReDoS (catastrophic backtracking). Sanitize with re2 or a timeout.
#       Also: ensure rules are sandboxed — a rule should only match/not match,
#       not execute code (use re, not eval).
# - "You need to add/remove rules without restarting the process."
#     → Hot-reload: watch the rules file for changes, recompile in a background
#       thread, swap the compiled ruleset atomically (replace reference).
# =============================================================================
