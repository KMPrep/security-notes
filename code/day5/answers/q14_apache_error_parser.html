"""
Q14. Apache Error Log Parser
==============================
Parse Apache error log lines, extract fields, group by error message pattern,
count frequency.

Apache error log format:
  [Mon Jan 15 10:00:00.123456 2024] [error] [pid 1234] [client 192.168.1.1:54321] File does not exist: /var/www/html/admin

=============================================================================
SOLUTION PROGRESSION
=============================================================================

PROMPT: "Parse Apache error logs and group by error pattern."

--- CANDIDATE FIRST RESPONSE (naive split on ']') ---
"I'd split on '] [' to get the bracket-enclosed fields."

Follow-up: "What about the timestamp which has spaces inside the brackets?"
→ The format is `[Mon Jan 15 10:00:00.123456 2024]` — splitting on '] [' can
  work, but you'd need to strip the leading/trailing brackets manually.
  A regex is cleaner and more robust.

Follow-up: "How do you 'group by error message pattern'? The path varies."
→ Strip the variable part (file path, IP, port numbers, etc.) from the
  message and use the remainder as the 'pattern key'.
  E.g., "File does not exist: /var/www/html/admin" and
        "File does not exist: /var/www/html/login" both map to
        "File does not exist: <path>"

--- CANDIDATE SECOND RESPONSE (regex parse + message normalization) ---
Better. Follow-up: "What normalization would you apply?"
→ Replace: file paths (/...), IP addresses, port numbers, PIDs, hex values
  with placeholders. This clusters similar errors together.

--- FINAL GOOGLE-LEVEL RESPONSE ---
Full regex parse, message normalization pipeline, defaultdict grouping,
frequency-sorted report.

=============================================================================
"""

import re
import sys
from collections import defaultdict

# ─────────────────────────────────────────────────────────────────────────────
# SAMPLE LOG
# ─────────────────────────────────────────────────────────────────────────────

# ─────────────────────────────────────────────────────────────────────────────
# PARSER
# ─────────────────────────────────────────────────────────────────────────────

# Apache error log format (httpd 2.4+)
APACHE_ERROR_RE = re.compile(
    r'^\[(?P<timestamp>[^\]]+)\]'          # [timestamp]
    r'\s+\[(?P<level>\w+)\]'              # [level]
    r'(?:\s+\[pid\s+(?P<pid>\d+)\])?'    # [pid N] — optional
    r'(?:\s+\[client\s+(?P<client>[^\]]+)\])?' # [client ip:port] — optional
    r'\s+(?P<message>.+)$'               # message (rest of line)
)

# For extracting client IP and port separately
CLIENT_RE = re.compile(r'^(?P<ip>.+?):(?P<port>\d+)$')


def parse_apache_error_line(raw: str):
    """
    Returns dict with: timestamp, level, pid, client_ip, client_port, message.
    Returns None if malformed.
    """
    raw = raw.strip()
    if not raw:
        return None
    m = APACHE_ERROR_RE.match(raw)
    if not m:
        return None

    client_ip   = None
    client_port = None
    client_str  = m.group("client")
    if client_str:
        cm = CLIENT_RE.match(client_str.strip())
        if cm:
            client_ip   = cm.group("ip")
            client_port = cm.group("port")
        else:
            client_ip = client_str.strip()

    return {
        "timestamp":   m.group("timestamp"),
        "level":       m.group("level").lower(),
        "pid":         m.group("pid"),
        "client_ip":   client_ip,
        "client_port": client_port,
        "message":     m.group("message").strip(),
    }


# ─────────────────────────────────────────────────────────────────────────────
# MESSAGE NORMALIZER
# Strip variable parts so similar errors cluster under one pattern.
# ─────────────────────────────────────────────────────────────────────────────

# Order matters: more specific patterns first
_NORMALIZERS = [
    # Absolute file paths → <path>
    (re.compile(r'/[a-zA-Z0-9_/.\-]+'),          "<path>"),
    # Port numbers  :1234  → :<port>
    (re.compile(r':\d{2,5}\b'),                   ":<port>"),
    # Line numbers in PHP errors  :42  → :<line>  (already handled by port rule mostly)
    # Hex values
    (re.compile(r'\b0x[0-9a-fA-F]+\b'),           "<hex>"),
    # Standalone numbers (pids, error codes) — be conservative
    (re.compile(r'\b\d{4,}\b'),                   "<num>"),
    # IPv4 addresses
    (re.compile(r'\b(?:\d{1,3}\.){3}\d{1,3}\b'), "<ip>"),
    # Collapse multiple spaces
    (re.compile(r'\s{2,}'),                        " "),
]


def normalize_message(message: str) -> str:
    """Apply all normalizers to produce a pattern key."""
    for pattern, replacement in _NORMALIZERS:
        message = pattern.sub(replacement, message)
    return message.strip()


# ─────────────────────────────────────────────────────────────────────────────
# APPROACH 1 — WITHOUT BUILT-IN HIGH-LEVEL HELPERS
# (manual dict grouping, manual top-K sort, no defaultdict)
# ─────────────────────────────────────────────────────────────────────────────

def _manual_top_k(count_dict, k):
    """Top-k by value descending using selection."""
    remaining = list(count_dict.items())
    result = []
    for _ in range(min(k, len(remaining))):
        max_idx = 0
        for i in range(1, len(remaining)):
            if remaining[i][1] > remaining[max_idx][1]:
                max_idx = i
        result.append(remaining[max_idx])
        remaining[max_idx] = remaining[-1]
        remaining.pop()
    return result


def analyze_without_builtins(lines):
    pattern_groups = {}   # normalized_msg → { count, level_counts, clients }
    level_counts   = {}   # level → count
    malformed      = 0

    for raw in lines:
        entry = parse_apache_error_line(raw)
        if entry is None:
            malformed += 1
            continue

        level = entry["level"]
        level_counts[level] = level_counts.get(level, 0) + 1

        pattern = normalize_message(entry["message"])

        if pattern not in pattern_groups:
            pattern_groups[pattern] = {
                "count":     0,
                "levels":    {},
                "clients":   {},
                "sample":    entry["message"],  # keep one real example
            }

        grp = pattern_groups[pattern]
        grp["count"] += 1
        grp["levels"][level] = grp["levels"].get(level, 0) + 1
        if entry["client_ip"]:
            grp["clients"][entry["client_ip"]] = True

    top_patterns = _manual_top_k(
        {p: g["count"] for p, g in pattern_groups.items()}, k=10
    )

    return pattern_groups, top_patterns, level_counts, malformed


# ─────────────────────────────────────────────────────────────────────────────
# APPROACH 2 — WITH BUILT-IN HELPERS
# ─────────────────────────────────────────────────────────────────────────────

def analyze_with_builtins(lines):
    pattern_groups = defaultdict(lambda: {
        "count": 0, "levels": defaultdict(int),
        "clients": set(), "sample": ""
    })
    level_counts = defaultdict(int)
    malformed    = 0

    for raw in lines:
        entry = parse_apache_error_line(raw)
        if entry is None:
            malformed += 1
            continue

        level = entry["level"]
        level_counts[level] += 1

        pattern = normalize_message(entry["message"])
        grp = pattern_groups[pattern]
        grp["count"] += 1
        grp["levels"][level] += 1
        if entry["client_ip"]:
            grp["clients"].add(entry["client_ip"])
        if not grp["sample"]:
            grp["sample"] = entry["message"]

    top_patterns = sorted(
        ((p, g["count"]) for p, g in pattern_groups.items()),
        key=lambda x: x[1],
        reverse=True
    )[:10]

    return dict(pattern_groups), top_patterns, dict(level_counts), malformed


# ─────────────────────────────────────────────────────────────────────────────
# REPORT
# ─────────────────────────────────────────────────────────────────────────────

def print_report(pattern_groups, top_patterns, level_counts, malformed, label=""):
    print(f"\n{'='*70}")
    print(f"  APACHE ERROR LOG REPORT  {label}")
    print(f"{'='*70}")
    print(f"Malformed lines skipped: {malformed}")

    print("\nEvent Counts by Log Level:")
    for level, count in sorted(level_counts.items()):
        print(f"  {level:<10} {count}")

    print(f"\nTop Error Patterns (by frequency):")
    for pattern, count in top_patterns:
        grp = pattern_groups[pattern]
        clients_val = grp["clients"]
        client_count = len(clients_val)
        levels_str = ", ".join(
            f"{lv}={cnt}" for lv, cnt in sorted(grp["levels"].items())
        )
        print(f"\n  Count={count}  UniqueClients={client_count}  Levels=[{levels_str}]")
        print(f"  Pattern : {pattern}")
        sample = grp["sample"]
        if len(sample) > 80:
            sample = sample[:77] + "..."
        print(f"  Example : {sample}")
    print()


# ─────────────────────────────────────────────────────────────────────────────
# MAIN
# ─────────────────────────────────────────────────────────────────────────────

def main():
    if len(sys.argv) > 1:
        try:
            with open(sys.argv[1], "r", encoding="utf-8", errors="replace") as fh:
                lines = fh.readlines()
        except (FileNotFoundError, PermissionError, OSError) as exc:
            print(f"[ERROR] {exc}", file=sys.stderr)
            sys.exit(1)
    else:
        print("[INFO] No file provided — using built-in sample data.\n")
        lines = SAMPLE_LOG.strip().splitlines(keepends=True)

    pg_manual,  tp_manual,  lv_manual,  mal_manual  = analyze_without_builtins(lines)
    pg_builtin, tp_builtin, lv_builtin, mal_builtin = analyze_with_builtins(lines)

    print_report(pg_manual,  tp_manual,  lv_manual,  mal_manual,  label="(Manual)")
    print_report(pg_builtin, tp_builtin, lv_builtin, mal_builtin, label="(Builtins)")

    # Sanity check
    assert mal_manual == mal_builtin, f"Malformed mismatch: {mal_manual} vs {mal_builtin}"
    assert len(pg_manual) == len(pg_builtin), \
        f"Pattern group count mismatch: {len(pg_manual)} vs {len(pg_builtin)}"
    print("[OK] Both approaches agree on pattern count.\n")


if __name__ == "__main__":
    main()
