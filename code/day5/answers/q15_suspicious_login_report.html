"""
Q15. Suspicious Login Report Generator
========================================
Complete Python script:

1. GENERATE a sample auth.log (50+ lines) containing:
   - Users with 3+ consecutive failures
   - Users logging in from 2+ different IPs in the same hour
   - Logins outside business hours (before 8am or after 8pm)

2. ANALYZE the log and produce a formatted report:

   === Suspicious Login Report ===
   Brute Force Candidates:
     alice: 7 consecutive failures from 10.0.0.1

   Multi-IP Logins:
     bob: logged in from [10.0.0.2, 172.16.0.5] within 1 hour

   After-Hours Logins:
     charlie: login at 02:13am from 192.168.1.99

=============================================================================
SOLUTION PROGRESSION
=============================================================================

PROMPT: "This is an end-to-end problem. Write a complete working script."

--- CANDIDATE FIRST RESPONSE (verbal walk-through) ---

 1. A generator to create the sample log.
 2. A parser to read the log into structured records.
 3. Three detection functions, one per suspicious pattern.
 Then a report formatter."

Follow-up: "Good structure. What does each detection function look like?
Start with brute force."

--- CANDIDATE WORKS THROUGH BRUTE FORCE ---
"Track consecutive failures per user. When we see OK, reset. If streak >= 3,
flag the user. Store the src_ip of the failing attempts."

Follow-up: "Multi-IP logins within one hour — how do you implement that?"
"Group successful logins by (user, hour). Collect the set of source IPs per
(user, hour). If len(ips) >= 2, flag it."

Follow-up: "After-hours — what counts as 'business hours'?"

import re
import sys
import random
from datetime import datetime, timezone, timedelta
from collections import defaultdict

# ─────────────────────────────────────────────────────────────────────────────
# CONSTANTS
# ─────────────────────────────────────────────────────────────────────────────

BUSINESS_START_HOUR = 8   # 08:00
BUSINESS_END_HOUR   = 20  # 20:00
BRUTE_FORCE_THRESHOLD = 3  # consecutive failures to flag
MULTI_IP_THRESHOLD    = 2  # distinct IPs in one hour to flag

# ─────────────────────────────────────────────────────────────────────────────
# PART 1 — SAMPLE LOG GENERATOR
# ─────────────────────────────────────────────────────────────────────────────

def generate_sample_log():
    """
    Generate a realistic auth.log covering all three suspicious patterns.
    Returns list of log lines (strings).
    Lines are timestamped in chronological order.
    """
    random.seed(42)  # deterministic output

    base = datetime(2024, 1, 15, 0, 0, 0, tzinfo=timezone.utc)
    lines = []

    def ts(offset_seconds):
        return (base + timedelta(seconds=offset_seconds)).strftime(
            "%Y-%m-%dT%H:%M:%SZ"
        )

    def make_line(offset, user, result, src):
        return f"{ts(offset)} auth: user={user} result={result} src={src}"

    t = 0   # running offset in seconds

    # ── Normal users (background noise) ──────────────────────────────────────
    normal_events = [
        (3600,  "dave",  "OK",   "10.10.0.1"),    # 01:00
        (7200,  "eve",   "OK",   "10.10.0.2"),    # 02:00
        (10800, "frank", "OK",   "10.10.0.3"),    # 03:00
        (14400, "grace", "OK",   "10.10.0.4"),    # 04:00
        (32400, "dave",  "OK",   "10.10.0.1"),    # 09:00
        (36000, "eve",   "OK",   "10.10.0.2"),    # 10:00
        (39600, "frank", "OK",   "10.10.0.3"),    # 11:00
        (43200, "grace", "OK",   "10.10.0.4"),    # 12:00
    ]
    for offset, user, result, src in normal_events:
        lines.append((offset, make_line(offset, user, result, src)))

    # ── PATTERN 1: Brute force — alice: 7 consecutive failures ───────────────
    # Start at 02:30
    t = 9000
    for i in range(7):
        lines.append((t, make_line(t, "alice", "FAIL", "10.0.0.1")))
        t += 10
    lines.append((t, make_line(t, "alice", "OK", "10.0.0.1")))   # eventually succeeds
    t += 30

    # ── PATTERN 2a: bob — 3 consecutive failures ─────────────────────────────
    t = 18000  # 05:00
    for i in range(3):
        lines.append((t, make_line(t, "bob", "FAIL", "192.168.2.1")))
        t += 5
    lines.append((t, make_line(t, "bob", "OK", "192.168.2.1")))
    t += 30

    # ── PATTERN 3: Multi-IP — bob logs in from 2 IPs in same hour ────────────
    # Bob logs in at 10:00 from two different IPs
    t = 36000   # 10:00
    lines.append((t, make_line(t, "bob", "OK", "10.0.0.2")))
    t += 1200   # +20 minutes
    lines.append((t, make_line(t, "bob", "OK", "172.16.0.5")))

    # ── PATTERN 4: carol — multi-IP login (3 different IPs) ──────────────────
    t = 39600   # 11:00
    ips = ["10.1.0.1", "10.1.0.2", "10.1.0.3"]
    for ip in ips:
        lines.append((t, make_line(t, "carol", "OK", ip)))
        t += 300   # 5 minutes apart — all within same hour

    # ── PATTERN 5: After-hours logins ────────────────────────────────────────
    # charlie logs in at 02:13
    t = 7980    # 02:13:00
    lines.append((t, make_line(t, "charlie", "OK", "192.168.1.99")))

    # heidi logs in at 23:45
    t = 85500   # 23:45:00
    lines.append((t, make_line(t, "heidi", "OK", "172.16.99.1")))

    # ivan logs in at 06:30 (before 8am)
    t = 23400   # 06:30:00
    lines.append((t, make_line(t, "ivan", "OK", "10.9.9.9")))

    # ── More noise ────────────────────────────────────────────────────────────
    noise_users = ["judy", "kevin", "linda", "mike", "nancy"]
    t_noise = 32400  # start at 09:00
    for u in noise_users:
        ip = f"10.0.{random.randint(1,50)}.{random.randint(1,250)}"
        lines.append((t_noise, make_line(t_noise, u, "OK", ip)))
        t_noise += 300
        # one fail each
        lines.append((t_noise, make_line(t_noise, u, "FAIL", ip)))
        t_noise += 10
        lines.append((t_noise, make_line(t_noise, u, "OK", ip)))
        t_noise += 290

    # Sort by timestamp offset
    lines.sort(key=lambda x: x[0])
    return [line for _, line in lines]


# ─────────────────────────────────────────────────────────────────────────────
# PART 2 — PARSER
# ─────────────────────────────────────────────────────────────────────────────

LINE_RE = re.compile(
    r'(?P<ts>\S+)\s+auth:\s+user=(?P<user>\S+)'
    r'\s+result=(?P<result>FAIL|OK)\s+src=(?P<src>\S+)'
)


def parse_line(raw: str):
    """Returns dict or None."""
    m = LINE_RE.search(raw.strip())
    if not m:
        return None
    ts_str = m.group("ts").replace("Z", "+00:00")
    try:
        dt = datetime.fromisoformat(ts_str)
        if dt.tzinfo is None:
            dt = dt.replace(tzinfo=timezone.utc)
    except ValueError:
        return None
    return {
        "ts":     dt,
        "user":   m.group("user"),
        "result": m.group("result"),
        "src":    m.group("src"),
    }


def parse_all(lines):
    """Parse all lines; return list of event dicts and malformed count."""
    events   = []
    malformed = 0
    for raw in lines:
        ev = parse_line(raw)
        if ev is None:
            malformed += 1
        else:
            events.append(ev)
    return events, malformed


# ─────────────────────────────────────────────────────────────────────────────
# PART 3 — DETECTION FUNCTIONS
# ─────────────────────────────────────────────────────────────────────────────

def detect_brute_force(events):
    """
    Flag users who had >= BRUTE_FORCE_THRESHOLD consecutive failures.
    Returns list of dicts: { user, max_streak, src_ips }
    """
    current_streak = defaultdict(int)
    max_streak     = defaultdict(int)
    streak_ips     = defaultdict(set)
    all_fail_ips   = defaultdict(set)
    flagged        = set()

    for ev in events:
        user   = ev["user"]
        result = ev["result"]
        src    = ev["src"]

        if result == "FAIL":
            current_streak[user] += 1
            streak_ips[user].add(src)
            all_fail_ips[user].add(src)
            if current_streak[user] > max_streak[user]:
                max_streak[user] = current_streak[user]
            if current_streak[user] >= BRUTE_FORCE_THRESHOLD:
                flagged.add(user)
        else:
            current_streak[user] = 0
            streak_ips[user]     = set()

    return [
        {
            "user":       u,
            "max_streak": max_streak[u],
            "src_ips":    sorted(all_fail_ips[u]),
        }
        for u in sorted(flagged)
    ]


def detect_multi_ip(events):
    """
    Flag users who had >= MULTI_IP_THRESHOLD distinct source IPs for
    SUCCESSFUL logins within the same calendar hour.
    Returns list of dicts: { user, hour, ips }
    """
    # Group by (user, hour-bucket)
    hour_ips = defaultdict(set)   # (user, hour_key) → set of IPs

    for ev in events:
        if ev["result"] != "OK":
            continue
        # Round timestamp down to the hour
        dt     = ev["ts"]
        hour_key = dt.strftime("%Y-%m-%dT%H:00Z")
        hour_ips[(ev["user"], hour_key)].add(ev["src"])

    results = []
    for (user, hour_key), ips in sorted(hour_ips.items()):
        if len(ips) >= MULTI_IP_THRESHOLD:
            results.append({
                "user":     user,
                "hour":     hour_key,
                "ips":      sorted(ips),
            })

    return results


def detect_after_hours(events):
    """
    Flag SUCCESSFUL logins outside business hours (before 8am or >= 8pm).
    Returns list of dicts: { user, ts, src }
    """
    results = []
    for ev in events:
        if ev["result"] != "OK":
            continue
        hour = ev["ts"].hour
        if hour < BUSINESS_START_HOUR or hour >= BUSINESS_END_HOUR:
            results.append({
                "user":   ev["user"],
                "ts":     ev["ts"],
                "src":    ev["src"],
            })
    return sorted(results, key=lambda x: x["ts"])


# ─────────────────────────────────────────────────────────────────────────────
# PART 4 — REPORT FORMATTER
# ─────────────────────────────────────────────────────────────────────────────

def print_report(brute_force, multi_ip, after_hours):
    print("\n" + "=" * 50)
    print("   === Suspicious Login Report ===")
    print("=" * 50)

    # ── Brute Force ──────────────────────────────────────────────────────────
    print(f"\nBrute Force Candidates (>= {BRUTE_FORCE_THRESHOLD} consecutive failures):")
    if not brute_force:
        print("  (none detected)")
    else:
        for rec in brute_force:
            ips_str = ", ".join(rec["src_ips"])
            print(f"  {rec['user']}: {rec['max_streak']} consecutive failures"
                  f" from [{ips_str}]")

    # ── Multi-IP ─────────────────────────────────────────────────────────────
    print(f"\nMulti-IP Logins (>= {MULTI_IP_THRESHOLD} different IPs in same hour):")
    if not multi_ip:
        print("  (none detected)")
    else:
        for rec in multi_ip:
            ips_str = ", ".join(rec["ips"])
            print(f"  {rec['user']}: logged in from [{ips_str}]"
                  f" within hour starting {rec['hour']}")

    # ── After-Hours ───────────────────────────────────────────────────────────
    print(f"\nAfter-Hours Logins (before {BUSINESS_START_HOUR:02d}:00"
          f" or after {BUSINESS_END_HOUR:02d}:00 UTC):")
    if not after_hours:
        print("  (none detected)")
    else:
        for rec in after_hours:
            time_str = rec["ts"].strftime("%I:%M%p").lower()
            print(f"  {rec['user']}: login at {time_str} from {rec['src']}")

    print()


# ─────────────────────────────────────────────────────────────────────────────
# MAIN
# ─────────────────────────────────────────────────────────────────────────────

def main():
    if len(sys.argv) > 1:
        # Read from file
        filepath = sys.argv[1]
        try:
            with open(filepath, "r", encoding="utf-8", errors="replace") as fh:
                log_lines = fh.readlines()
            print(f"[INFO] Read {len(log_lines)} lines from {filepath}")
        except (FileNotFoundError, PermissionError, OSError) as exc:
            print(f"[ERROR] {exc}", file=sys.stderr)
            sys.exit(1)
    else:
        # Generate sample log
        print("[INFO] Generating sample auth.log ...\n")
        log_lines = generate_sample_log()

        # Show generated log (truncated for readability)
        print(f"Generated {len(log_lines)} log lines. First 10:")
        for line in log_lines[:10]:
            print(f"  {line}")
        print(f"  ... and {len(log_lines) - 10} more lines.\n")

        # Optionally write to file for inspection
        output_path = "generated_auth.log"
        try:
            with open(output_path, "w", encoding="utf-8") as fh:
                for line in log_lines:
                    fh.write(line + "\n")
            print(f"[INFO] Sample log written to: {output_path}\n")
        except OSError as exc:
            print(f"[WARN] Could not write sample log: {exc}", file=sys.stderr)

    # Parse
    events, malformed = parse_all(log_lines)
    print(f"[INFO] Parsed {len(events)} events ({malformed} malformed skipped)\n")

    if not events:
        print("[WARN] No events to analyze.", file=sys.stderr)
        sys.exit(1)

    # Detect
    brute_force  = detect_brute_force(events)
    multi_ip     = detect_multi_ip(events)
    after_hours  = detect_after_hours(events)

    # Report
    print_report(brute_force, multi_ip, after_hours)

    # Summary counts
    print(f"Summary:")
    print(f"  Brute force candidates : {len(brute_force)}")
    print(f"  Multi-IP login users   : {len(multi_ip)}")
    print(f"  After-hours logins     : {len(after_hours)}")
    print()


if __name__ == "__main__":
    main()
