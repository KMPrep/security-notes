"""
Q1. HTTP Access Log Parser
==========================
Parse HTTP access logs in Combined Log Format and produce a summary report.

Format:
  192.168.1.1 - frank [10/Oct/2023:13:55:36 -0700] "GET /admin HTTP/1.1" 404 512

=============================================================================
SOLUTION PROGRESSION
=============================================================================

PROMPT: "Parse an HTTP access log and report: total requests, status code
distribution, top 5 IPs, top 5 paths, and IPs hitting /admin or /passwd."

--- CANDIDATE FIRST RESPONSE (naive, no regex, no error handling) ---
"My first instinct is to split each line by spaces and grab the fields
positionally. I know the format is fixed-ish so let me try that first,
then we can tighten it up."

Issue: splitting by space breaks on the timestamp (it has a space inside []).
The quoted request field also has spaces. So naive split won't work cleanly.

--- CANDIDATE SECOND RESPONSE (regex, but still rough) ---
"OK, I'll use a regex. Let me capture each field group explicitly."

Still missing: error handling for malformed lines, edge cases like '-' for
missing bytes, file not found, encoding issues.

--- FINAL GOOGLE-LEVEL RESPONSE ---
Full regex, graceful skip of malformed lines, proper aggregation, clear report.

=============================================================================
"""

# APPROACH 1 — WITHOUT BUILT-IN HIGH-LEVEL HELPERS
# (manual dict aggregation, manual top-K, no Counter / no sorted key lambda)

# Combined Log Format regex — compiled once for performance
LOG_PATTERN = re.compile(
    r'(?P<ip>\S+)'           # client IP
    r'\s+\S+'                # ident (usually -)
    r'\s+(?P<user>\S+)'      # auth user
    r'\s+\[(?P<ts>[^\]]+)\]' # timestamp inside []
    r'\s+"(?P<method>\S+)'   # HTTP method
    r'\s+(?P<path>\S+)'      # request path
    r'\s+\S+"'               # HTTP version
    r'\s+(?P<status>\d{3})'  # status code
    r'\s+(?P<bytes>\S+)'     # bytes (can be -)
)


def parse_line(line: str):
    """
    Parse a single log line.
    Returns a dict on success, None if malformed.
    """
    line = line.strip()
    if not line:
        return None
    m = LOG_PATTERN.match(line)
    if not m:
        return None

    raw_bytes = m.group("bytes")
    byte_count = 0
    if raw_bytes != "-":
        try:
            byte_count = int(raw_bytes)
        except ValueError:
            byte_count = 0  # unexpected non-numeric — treat as 0

    return {
        "ip":     m.group("ip"),
        "user":   m.group("user"),
        "ts":     m.group("ts"),
        "method": m.group("method"),
        "path":   m.group("path"),
        "status": m.group("status"),
        "bytes":  byte_count,
    }


def _manual_top_k(count_dict, k):
    """
    Return top-k (key, count) pairs sorted by count descending.
    Implemented WITHOUT sorted() / max() builtins — uses insertion sort.
    """
    pairs = []
    for key, val in count_dict.items():
        pairs.append((key, val))

    # Insertion sort descending by count (fine for small N; log lines ~ hundreds)
    for i in range(1, len(pairs)):
        current = pairs[i]
        j = i - 1
        while j >= 0 and pairs[j][1] < current[1]:
            pairs[j + 1] = pairs[j]
            j -= 1
        pairs[j + 1] = current

    return pairs[:k]


def analyze_without_builtins(lines):
    """
    First-pass approach:
    - Manual dict counting
    - Manual top-K (insertion sort)
    - No collections.Counter, no sorted(key=...)
    """
    total = 0
    status_counts   = {}   # { "200": 5, ... }
    ip_counts        = {}   # { "1.2.3.4": 3, ... }
    path_counts      = {}   # { "/admin": 2, ... }
    sensitive_ips    = {}   # IPs that hit /admin or /passwd (use dict as set)
    malformed        = 0

    SENSITIVE_PATHS = ("/admin", "/passwd")

    for raw in lines:
        entry = parse_line(raw)
        if entry is None:
            malformed += 1
            continue

        total += 1

        # Status distribution
        status_counts[entry["status"]] = status_counts.get(entry["status"], 0) + 1

        # IP count
        ip_counts[entry["ip"]] = ip_counts.get(entry["ip"], 0) + 1

        # Path count
        path_counts[entry["path"]] = path_counts.get(entry["path"], 0) + 1

        # Sensitive path check — check if path contains /admin or /passwd
        for sensitive in SENSITIVE_PATHS:
            if sensitive in entry["path"]:
                sensitive_ips[entry["ip"]] = True
                break

    top5_ips   = _manual_top_k(ip_counts, 5)
    top5_paths = _manual_top_k(path_counts, 5)

    return {
        "total":         total,
        "malformed":     malformed,
        "status_counts": status_counts,
        "top5_ips":      top5_ips,
        "top5_paths":    top5_paths,
        "sensitive_ips": list(sensitive_ips.keys()),
    }


# ─────────────────────────────────────────────────────────────────────────────
# APPROACH 2 — WITH BUILT-IN HELPERS (cleaner, idiomatic Python)
# ─────────────────────────────────────────────────────────────────────────────

def analyze_with_builtins(lines):
    """
    Cleaner version using collections.defaultdict and sorted().
    Same logic, less code, easier to read in production.
    """
    total            = 0
    malformed        = 0
    status_counts    = defaultdict(int)
    ip_counts        = defaultdict(int)
    path_counts      = defaultdict(int)
    sensitive_ips    = set()

    SENSITIVE_PATHS = ("/admin", "/passwd")

    for raw in lines:
        entry = parse_line(raw)
        if entry is None:
            malformed += 1
            continue

        total += 1
        status_counts[entry["status"]] += 1
        ip_counts[entry["ip"]]         += 1
        path_counts[entry["path"]]     += 1

        for sensitive in SENSITIVE_PATHS:
            if sensitive in entry["path"]:
                sensitive_ips.add(entry["ip"])
                break

    top5_ips   = sorted(ip_counts.items(),   key=lambda x: x[1], reverse=True)[:5]
    top5_paths = sorted(path_counts.items(), key=lambda x: x[1], reverse=True)[:5]

    return {
        "total":         total,
        "malformed":     malformed,
        "status_counts": dict(status_counts),
        "top5_ips":      top5_ips,
        "top5_paths":    top5_paths,
        "sensitive_ips": sorted(sensitive_ips),
    }


# ─────────────────────────────────────────────────────────────────────────────
# REPORT PRINTER
# ─────────────────────────────────────────────────────────────────────────────

def print_report(result, label=""):
    print(f"\n{'='*60}")
    print(f"  HTTP ACCESS LOG REPORT  {label}")
    print(f"{'='*60}")
    print(f"Total requests   : {result['total']}")
    print(f"Malformed lines  : {result['malformed']}")

    print("\nStatus Code Distribution:")
    for code, count in sorted(result["status_counts"].items()):
        print(f"  {code} : {count}")

    print("\nTop 5 IPs by Request Count:")
    for ip, count in result["top5_ips"]:
        print(f"  {ip:<20} {count} requests")

    print("\nTop 5 Paths by Request Count:")
    for path, count in result["top5_paths"]:
        print(f"  {path:<30} {count} requests")

    print("\nIPs that hit /admin or /passwd:")
    for ip in result["sensitive_ips"]:
        print(f"  {ip}")
    print()

# FILE READER  (memory-efficient line-by-line, as required)
# ─────────────────────────────────────────────────────────────────────────────

def read_log_file(filepath: str):
    """
    Generator: yields lines one at a time — does NOT load full file into memory.
    Handles missing file, permission errors, and encoding issues.
    """
    try:
        with open(filepath, "r", encoding="utf-8", errors="replace") as fh:
            for line in fh:
                yield line
    except FileNotFoundError:
        print(f"[ERROR] Log file not found: {filepath}", file=sys.stderr)
        return
    except PermissionError:
        print(f"[ERROR] Permission denied: {filepath}", file=sys.stderr)
        return
    except OSError as exc:
        print(f"[ERROR] Could not read file {filepath}: {exc}", file=sys.stderr)
        return

def main():
    """
    Entry point.
    Usage:
      python q1_http_log_parser.py [logfile]
      python q1_http_log_parser.py          ← uses built-in sample data
    """
    if len(sys.argv) > 1:
        filepath = sys.argv[1]
        lines_source = read_log_file(filepath)
        # We need to iterate twice (once for each approach) — materialize once.
        # For huge files you'd pick ONE approach; here we demo both.
        lines = list(lines_source)
        if not lines:
            print("[WARN] No lines read. Exiting.", file=sys.stderr)
            sys.exit(1)
    else:
        print("[INFO] No log file provided — using built-in sample data.\n")
        lines = SAMPLE_LOG.strip().splitlines(keepends=True)

    result_manual   = analyze_without_builtins(lines)
    result_builtins = analyze_with_builtins(lines)

    print_report(result_manual,   label="(Manual / No Builtins)")
    print_report(result_builtins, label="(With Builtins)")

    # Sanity check: both approaches should agree
    assert result_manual["total"]    == result_builtins["total"],    "Mismatch in totals!"
    assert result_manual["malformed"] == result_builtins["malformed"], "Mismatch in malformed!"
    print("[OK] Both approaches produce identical totals.\n")


if __name__ == "__main__":
    main()
