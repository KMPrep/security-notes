"""
Q2. Failed Login Detector
==========================
Parse an auth log, count consecutive failures per user before a success
resets the counter, flag any user who hits >= 5 consecutive failures.

Log format:
  2024-01-15T10:30:00Z auth: user=alice result=FAIL src=10.0.0.1

=============================================================================
SOLUTION PROGRESSION
=============================================================================

PROMPT: "Detect brute-force attempts from an auth log.
Count consecutive failures per user; a success resets the counter.
Flag anyone who hits >= 5 consecutive failures."

--- CANDIDATE FIRST RESPONSE (naive — reads entire file into memory, no regex) ---
"I'd loop through lines, split on spaces, track a per-user counter in a dict.
When I see FAIL I increment; when I see OK I reset to 0."

Follow-ups:
  - "What if a user has multiple bursts — 4 fails, then success, then 6 fails?"
    → Need to track MAX consecutive failures seen, not just current.
  - "What source IPs did those failures come from?"
    → Need a per-user set of IPs seen during failure bursts.
  - "What if the log format is slightly inconsistent (extra spaces)?"
    → Use regex instead of naive split.

--- CANDIDATE SECOND RESPONSE (regex + proper state tracking) ---
Uses regex, tracks current_streak and max_streak per user.
Still no error handling.

--- FINAL GOOGLE-LEVEL RESPONSE ---
Regex parsing, full state machine per user, graceful error handling,
memory-efficient streaming, clear report.

=============================================================================
"""

import re
import sys
from collections import defaultdict

# ─────────────────────────────────────────────────────────────────────────────
# SAMPLE DATA
# ─────────────────────────────────────────────────────────────────────────────

SAMPLE_LOG = """\

THRESHOLD = 5  # flag if consecutive failures reach this

# ─────────────────────────────────────────────────────────────────────────────
# PARSE HELPER
# ─────────────────────────────────────────────────────────────────────────────

LINE_RE = re.compile(
    r'(?P<ts>\S+)'
    r'\s+auth:\s+'
    r'user=(?P<user>\S+)'
    r'\s+result=(?P<result>FAIL|OK)'
    r'\s+src=(?P<src>\S+)'
)


def parse_auth_line(raw: str):
    """Return dict or None for malformed lines."""
    raw = raw.strip()
    if not raw:
        return None
    m = LINE_RE.search(raw)
    if not m:
        return None
    return {
        "ts":     m.group("ts"),
        "user":   m.group("user"),
        "result": m.group("result"),
        "src":    m.group("src"),
    }


# ─────────────────────────────────────────────────────────────────────────────
# APPROACH 1 — WITHOUT BUILT-IN HIGH-LEVEL HELPERS
# (no defaultdict, no set — use plain dicts and lists, manual deduplication)
# ─────────────────────────────────────────────────────────────────────────────

def detect_without_builtins(lines):
    """
    State per user (all plain dicts/lists):
      current_streak  : int  — consecutive failures in current run
      max_streak      : int  — highest streak seen for this user
      streak_ips      : list — IPs seen during the CURRENT failure streak
      all_fail_ips    : list — all IPs ever seen during ANY failure streak
      flagged         : bool — did this user ever hit >= THRESHOLD?
    """
    # user_state[user] = { current_streak, max_streak, streak_ips, all_fail_ips, flagged }
    user_state = {}
    malformed = 0

    def _add_unique(lst, item):
        """Append item to list only if not already present (manual set)."""
        for existing in lst:
            if existing == item:
                return
        lst.append(item)

    for raw in lines:
        entry = parse_auth_line(raw)
        if entry is None:
            malformed += 1
            continue

        user   = entry["user"]
        result = entry["result"]
        src    = entry["src"]

        # Initialise state for new user
        if user not in user_state:
            user_state[user] = {
                "current_streak": 0,
                "max_streak":     0,
                "streak_ips":     [],
                "all_fail_ips":   [],
                "flagged":        False,
            }

        state = user_state[user]

        if result == "FAIL":
            state["current_streak"] += 1
            _add_unique(state["streak_ips"],   src)
            _add_unique(state["all_fail_ips"], src)

            if state["current_streak"] > state["max_streak"]:
                state["max_streak"] = state["current_streak"]

            if state["current_streak"] >= THRESHOLD:
                state["flagged"] = True

        else:  # result == "OK"
            # Reset current streak; preserve historical data
            state["current_streak"] = 0
            state["streak_ips"]     = []   # fresh streak IPs for next burst

    # Collect flagged users
    flagged_users = []
    for user, state in user_state.items():
        if state["flagged"]:
            flagged_users.append({
                "user":       user,
                "max_streak": state["max_streak"],
                "src_ips":    state["all_fail_ips"],
            })

    return flagged_users, malformed, user_state


# ─────────────────────────────────────────────────────────────────────────────
# APPROACH 2 — WITH BUILT-IN HELPERS (cleaner)
# ─────────────────────────────────────────────────────────────────────────────

def detect_with_builtins(lines):
    """
    Same logic using defaultdict and set for brevity.
    """
    current_streak = defaultdict(int)
    max_streak     = defaultdict(int)
    streak_ips     = defaultdict(set)   # IPs in the CURRENT burst
    all_fail_ips   = defaultdict(set)   # all IPs across all failure bursts
    flagged        = set()
    malformed      = 0

    for raw in lines:
        entry = parse_auth_line(raw)
        if entry is None:
            malformed += 1
            continue

        user   = entry["user"]
        result = entry["result"]
        src    = entry["src"]

        if result == "FAIL":
            current_streak[user] += 1
            streak_ips[user].add(src)
            all_fail_ips[user].add(src)

            if current_streak[user] > max_streak[user]:
                max_streak[user] = current_streak[user]

            if current_streak[user] >= THRESHOLD:
                flagged.add(user)

        else:  # OK — reset
            current_streak[user] = 0
            streak_ips[user]     = set()

    flagged_users = [
        {
            "user":       u,
            "max_streak": max_streak[u],
            "src_ips":    sorted(all_fail_ips[u]),
        }
        for u in sorted(flagged)
    ]
    return flagged_users, malformed


# ─────────────────────────────────────────────────────────────────────────────
# REPORT
# ─────────────────────────────────────────────────────────────────────────────

def print_report(flagged_users, malformed, label=""):
    print(f"\n{'='*60}")
    print(f"  FAILED LOGIN DETECTION REPORT  {label}")
    print(f"{'='*60}")
    print(f"Malformed lines skipped: {malformed}")
    print(f"Threshold for flagging : {THRESHOLD} consecutive failures\n")

    if not flagged_users:
        print("  No users flagged.\n")
        return

    print(f"Flagged users ({len(flagged_users)}):")
    for rec in flagged_users:
        ips_str = ", ".join(rec["src_ips"])
        print(f"  {rec['user']:<15} max_consecutive={rec['max_streak']}  "
              f"src_ips=[{ips_str}]")
    print()


# ─────────────────────────────────────────────────────────────────────────────
# FILE READER
# ─────────────────────────────────────────────────────────────────────────────

def stream_lines(filepath: str):
    """Generator — line-by-line, does not load entire file into memory."""
    try:
        with open(filepath, "r", encoding="utf-8", errors="replace") as fh:
            for line in fh:
                yield line
    except FileNotFoundError:
        print(f"[ERROR] File not found: {filepath}", file=sys.stderr)
    except PermissionError:
        print(f"[ERROR] Permission denied: {filepath}", file=sys.stderr)
    except OSError as exc:
        print(f"[ERROR] OS error reading {filepath}: {exc}", file=sys.stderr)


# ─────────────────────────────────────────────────────────────────────────────
# MAIN
# ─────────────────────────────────────────────────────────────────────────────

def main():
    if len(sys.argv) > 1:
        filepath = sys.argv[1]
        lines = list(stream_lines(filepath))
        if not lines:
            print("[WARN] No lines read. Exiting.", file=sys.stderr)
            sys.exit(1)
    else:
        print("[INFO] No log file provided — using built-in sample data.\n")
        lines = SAMPLE_LOG.strip().splitlines(keepends=True)

    flagged_manual, malformed_manual, _ = detect_without_builtins(lines)
    flagged_builtin, malformed_builtin  = detect_with_builtins(lines)

    print_report(flagged_manual,  malformed_manual,  label="(Manual / No Builtins)")
    print_report(flagged_builtin, malformed_builtin, label="(With Builtins)")

    # Sanity check
    manual_names  = sorted(r["user"] for r in flagged_manual)
    builtin_names = sorted(r["user"] for r in flagged_builtin)
    assert manual_names == builtin_names, \
        f"Mismatch! manual={manual_names}  builtin={builtin_names}"
    print("[OK] Both approaches agree on flagged users.\n")


if __name__ == "__main__":
    main()
