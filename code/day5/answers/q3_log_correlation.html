"""
Q3. Multi-Source Log Correlation
==================================
Given two log files (firewall log and auth log), correlate events by IP
within a 60-second window.

Firewall log : timestamp ip port action       (ALLOW / DENY)
Auth log     : timestamp ip username result   (OK / FAIL)

Flag: any IP with a firewall DENY followed by an auth FAIL within 60 seconds.
Output: (ip, fw_timestamp, auth_timestamp, username)

=============================================================================
SOLUTION PROGRESSION
=============================================================================

PROMPT: "You have two different log files. Correlate them by IP and time."

--- CANDIDATE FIRST RESPONSE (naive — two nested loops, O(F*A)) ---
"Load all firewall DENY events. Load all auth FAIL events.
For every DENY, scan every auth FAIL for the same IP and check the delta."

Follow-up: "What's the complexity?"
→ O(F * A) per IP. If firewall has 1M DENYs and auth has 500K FAILs, that's
  500 billion comparisons. Too slow.

Follow-up: "How do you fix it?"
→ Group events by IP first. For each IP, sort both lists by timestamp.
  Use a two-pointer or binary search to find auth events within [t, t+60].

--- CANDIDATE SECOND RESPONSE (group by IP, sort, sliding window) ---
Better. O((F+A) log(F+A)) overall due to sorting.

Follow-up: "What if files are too large to fit in memory?"
→ Sort both files externally (sort utility or merge sort on disk),
  then stream-merge them line-by-line by IP+timestamp.
  For this problem scope we keep the in-memory approach.

--- FINAL GOOGLE-LEVEL RESPONSE ---
Parse both files, group by IP, sort per-IP lists, two-pointer correlation,
proper datetime parsing with timezone awareness.

=============================================================================
"""

import re
import sys
from datetime import datetime, timezone
from collections import defaultdict

# ─────────────────────────────────────────────────────────────────────────────
# SAMPLE DATA
# ─────────────────────────────────────────────────────────────────────────────

SAMPLE_FW_LOG = """\
2024-01-15T10:00:00Z 10.0.0.5  443  ALLOW
2024-01-15T10:00:05Z 10.0.0.6  22   DENY
2024-01-15T10:00:10Z 10.0.0.7  80   ALLOW
2024-01-15T10:00:15Z 10.0.0.8  22   DENY
2024-01-15T10:00:20Z 10.0.0.6  22   DENY
2024-01-15T10:00:25Z 10.0.0.9  443  DENY
2024-01-15T10:01:00Z 10.0.0.8  22   DENY
MALFORMED FW LINE
"""

SAMPLE_AUTH_LOG = """\
2024-01-15T10:00:08Z 10.0.0.6  attacker FAIL
2024-01-15T10:00:18Z 10.0.0.8  hacker   FAIL
2024-01-15T10:00:22Z 10.0.0.6  attacker FAIL
2024-01-15T10:00:30Z 10.0.0.9  bot      OK
2024-01-15T10:00:50Z 10.0.0.5  alice    OK
2024-01-15T10:01:05Z 10.0.0.8  hacker   FAIL
MALFORMED AUTH LINE
"""

WINDOW_SECS = 60

# ─────────────────────────────────────────────────────────────────────────────
# PARSERS
# ─────────────────────────────────────────────────────────────────────────────

FW_RE = re.compile(
    r'(?P<ts>\S+)\s+(?P<ip>\S+)\s+(?P<port>\d+)\s+(?P<action>ALLOW|DENY)'
)
AUTH_RE = re.compile(
    r'(?P<ts>\S+)\s+(?P<ip>\S+)\s+(?P<user>\S+)\s+(?P<result>OK|FAIL)'
)


def _parse_ts(ts_str: str) -> float:
    """
    Parse ISO-8601 UTC timestamp to a Unix epoch float.
    Returns -1.0 on failure so callers can discard the record.
    """
    ts_str = ts_str.strip()
    # Handle both 'Z' suffix and '+00:00'
    ts_str = ts_str.replace("Z", "+00:00")
    try:
        dt = datetime.fromisoformat(ts_str)
        # If no tzinfo, assume UTC
        if dt.tzinfo is None:
            dt = dt.replace(tzinfo=timezone.utc)
        return dt.timestamp()
    except ValueError:
        return -1.0


def parse_fw_line(raw: str):
    """Returns dict or None."""
    m = FW_RE.search(raw.strip())
    if not m:
        return None
    ts = _parse_ts(m.group("ts"))
    if ts < 0:
        return None
    return {"ts": ts, "ts_raw": m.group("ts"),
            "ip": m.group("ip"), "port": m.group("port"),
            "action": m.group("action")}


def parse_auth_line(raw: str):
    """Returns dict or None."""
    m = AUTH_RE.search(raw.strip())
    if not m:
        return None
    ts = _parse_ts(m.group("ts"))
    if ts < 0:
        return None
    return {"ts": ts, "ts_raw": m.group("ts"),
            "ip": m.group("ip"), "user": m.group("user"),
            "result": m.group("result")}


# ─────────────────────────────────────────────────────────────────────────────
# APPROACH 1 — WITHOUT BUILT-IN HIGH-LEVEL HELPERS
# (manual grouping, manual binary search, no defaultdict/bisect)
# ─────────────────────────────────────────────────────────────────────────────

def _manual_binary_search_left(arr, target):
    """
    Return the leftmost index i such that arr[i] >= target.
    arr is a list of floats, sorted ascending.
    Implements bisect_left manually.
    """
    lo, hi = 0, len(arr)
    while lo < hi:
        mid = (lo + hi) // 2
        if arr[mid] < target:
            lo = mid + 1
        else:
            hi = mid
    return lo


def _manual_insertion_sort_by_ts(records):
    """Sort list of dicts by 'ts' (float) in-place, ascending."""
    for i in range(1, len(records)):
        key = records[i]
        j = i - 1
        while j >= 0 and records[j]["ts"] > key["ts"]:
            records[j + 1] = records[j]
            j -= 1
        records[j + 1] = key


def correlate_without_builtins(fw_lines, auth_lines):
    malformed = {"fw": 0, "auth": 0}

    # Group DENY events by IP
    fw_denies = {}   # ip -> list of fw event dicts
    for raw in fw_lines:
        ev = parse_fw_line(raw)
        if ev is None:
            malformed["fw"] += 1
            continue
        if ev["action"] != "DENY":
            continue
        if ev["ip"] not in fw_denies:
            fw_denies[ev["ip"]] = []
        fw_denies[ev["ip"]].append(ev)

    # Group FAIL events by IP
    auth_fails = {}  # ip -> list of auth event dicts
    for raw in auth_lines:
        ev = parse_auth_line(raw)
        if ev is None:
            malformed["auth"] += 1
            continue
        if ev["result"] != "FAIL":
            continue
        if ev["ip"] not in auth_fails:
            auth_fails[ev["ip"]] = []
        auth_fails[ev["ip"]].append(ev)

    # Sort per-IP lists by timestamp
    for ip in fw_denies:
        _manual_insertion_sort_by_ts(fw_denies[ip])
    for ip in auth_fails:
        _manual_insertion_sort_by_ts(auth_fails[ip])

    # Correlate: for each DENY, find auth FAILs within [deny_ts, deny_ts + 60]
    results = []
    for ip, denies in fw_denies.items():
        if ip not in auth_fails:
            continue
        auths = auth_fails[ip]
        auth_ts_list = [a["ts"] for a in auths]  # sorted list of timestamps

        for deny in denies:
            window_start = deny["ts"]
            window_end   = deny["ts"] + WINDOW_SECS

            # Find first auth event >= window_start
            idx = _manual_binary_search_left(auth_ts_list, window_start)

            # Scan forward while within window
            while idx < len(auths) and auths[idx]["ts"] <= window_end:
                results.append((
                    ip,
                    deny["ts_raw"],
                    auths[idx]["ts_raw"],
                    auths[idx]["user"],
                ))
                idx += 1

    return results, malformed


# ─────────────────────────────────────────────────────────────────────────────
# APPROACH 2 — WITH BUILT-IN HELPERS (bisect, defaultdict, sorted)
# ─────────────────────────────────────────────────────────────────────────────

import bisect

def correlate_with_builtins(fw_lines, auth_lines):
    malformed = {"fw": 0, "auth": 0}

    fw_denies  = defaultdict(list)
    auth_fails = defaultdict(list)

    for raw in fw_lines:
        ev = parse_fw_line(raw)
        if ev is None:
            malformed["fw"] += 1
            continue
        if ev["action"] == "DENY":
            fw_denies[ev["ip"]].append(ev)

    for raw in auth_lines:
        ev = parse_auth_line(raw)
        if ev is None:
            malformed["auth"] += 1
            continue
        if ev["result"] == "FAIL":
            auth_fails[ev["ip"]].append(ev)

    # Sort each IP's list
    for ip in fw_denies:
        fw_denies[ip].sort(key=lambda e: e["ts"])
    for ip in auth_fails:
        auth_fails[ip].sort(key=lambda e: e["ts"])

    results = []
    for ip, denies in fw_denies.items():
        if ip not in auth_fails:
            continue
        auths = auth_fails[ip]
        auth_ts_list = [a["ts"] for a in auths]

        for deny in denies:
            window_start = deny["ts"]
            window_end   = deny["ts"] + WINDOW_SECS

            idx = bisect.bisect_left(auth_ts_list, window_start)
            while idx < len(auths) and auths[idx]["ts"] <= window_end:
                results.append((
                    ip,
                    deny["ts_raw"],
                    auths[idx]["ts_raw"],
                    auths[idx]["user"],
                ))
                idx += 1

    return results, malformed


# ─────────────────────────────────────────────────────────────────────────────
# REPORT
# ─────────────────────────────────────────────────────────────────────────────

def print_report(results, malformed, label=""):
    print(f"\n{'='*70}")
    print(f"  MULTI-SOURCE CORRELATION REPORT  {label}")
    print(f"{'='*70}")
    print(f"Malformed FW   lines skipped: {malformed['fw']}")
    print(f"Malformed auth lines skipped: {malformed['auth']}")
    print(f"Window size: {WINDOW_SECS} seconds\n")

    if not results:
        print("  No correlated events found.\n")
        return

    print(f"Correlated (FW DENY → Auth FAIL) events ({len(results)} found):\n")
    print(f"  {'IP':<15}  {'FW Deny Time':<25}  {'Auth Fail Time':<25}  {'User'}")
    print(f"  {'-'*14}  {'-'*24}  {'-'*24}  {'-'*10}")
    for ip, fw_ts, auth_ts, user in results:
        print(f"  {ip:<15}  {fw_ts:<25}  {auth_ts:<25}  {user}")
    print()


# ─────────────────────────────────────────────────────────────────────────────
# FILE READER
# ─────────────────────────────────────────────────────────────────────────────

def read_file_lines(filepath: str):
    try:
        with open(filepath, "r", encoding="utf-8", errors="replace") as fh:
            return fh.readlines()
    except FileNotFoundError:
        print(f"[ERROR] File not found: {filepath}", file=sys.stderr)
        return []
    except PermissionError:
        print(f"[ERROR] Permission denied: {filepath}", file=sys.stderr)
        return []
    except OSError as exc:
        print(f"[ERROR] Could not read {filepath}: {exc}", file=sys.stderr)
        return []


# ─────────────────────────────────────────────────────────────────────────────
# MAIN
# ─────────────────────────────────────────────────────────────────────────────

def main():
    if len(sys.argv) == 3:
        fw_lines   = read_file_lines(sys.argv[1])
        auth_lines = read_file_lines(sys.argv[2])
        if not fw_lines or not auth_lines:
            print("[WARN] One or both files are empty/unreadable. Exiting.",
                  file=sys.stderr)
            sys.exit(1)
    else:
        print("[INFO] Usage: python q3_log_correlation.py fw.log auth.log")
        print("[INFO] Using built-in sample data.\n")
        fw_lines   = SAMPLE_FW_LOG.strip().splitlines(keepends=True)
        auth_lines = SAMPLE_AUTH_LOG.strip().splitlines(keepends=True)

    res_manual,  mal_manual  = correlate_without_builtins(fw_lines, auth_lines)
    res_builtin, mal_builtin = correlate_with_builtins(fw_lines, auth_lines)

    print_report(res_manual,  mal_manual,  label="(Manual / No Builtins)")
    print_report(res_builtin, mal_builtin, label="(With Builtins)")

    # Sanity check
    assert sorted(res_manual) == sorted(res_builtin), \
        "Mismatch between manual and builtin results!"
    print("[OK] Both approaches produce identical results.\n")


if __name__ == "__main__":
    main()

# - "Events can arrive out of order by up to 5 seconds. How do you handle it?"

# - "The window should be 'within N seconds of each other' not directional."
