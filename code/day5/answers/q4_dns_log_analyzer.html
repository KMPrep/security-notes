"""
Q4. DNS Query Log Analyzer
===========================
Given a DNS query log, count unique domains per client IP,
detect DGA (Domain Generation Algorithm) candidates by Shannon entropy,
and report top-10 queried domains.

Log format:
  2024-01-15T10:00:00Z client=10.0.0.5 query=evil.example.com type=A

DGA heuristic: Shannon entropy of the domain label > 3.5 bits/char.

=============================================================================
SOLUTION PROGRESSION
=============================================================================

PROMPT: "Analyze a DNS log. Detect which domains look algorithmically
generated based on character-frequency entropy."

--- CANDIDATE FIRST RESPONSE (no entropy, just counting) ---
"First, I'll parse lines, count domains per client, count total queries per
domain, and print top-10. I'll skip entropy for now."

Follow-up: "Good. Now explain how you'd detect DGA domains."
→ DGA domains look random: 'akjdhqkjwehqwe.com' vs 'google.com'.
  Shannon entropy measures randomness in a string.
  H = -Σ p(c) log2(p(c))

Follow-up: "Implement it without using math.log."
→ Use manual power-of-2 log: log2(x) = ln(x)/ln(2)
  But if we can't use math at all, approximate with a lookup table or
  bit-counting. More practical: use math module but implement the
  frequency counting manually.

Follow-up: "What's the threshold and why 3.5?"
→ Empirically, legitimate domains like 'google', 'facebook' score ~2.5-3.
  Truly random 12-char strings score ~3.8-4.0. 3.5 is a common heuristic
  threshold balancing false positives and false negatives.

--- FINAL GOOGLE-LEVEL RESPONSE ---
Full regex parsing, proper entropy calculation, top-10 with manual heap,
DGA detection with threshold, per-client unique domain tracking.

=============================================================================
"""

from collections import defaultdict

DGA_ENTROPY_THRESHOLD = 3.5  # bits per character

# ─────────────────────────────────────────────────────────────────────────────
# ENTROPY CALCULATOR
# ─────────────────────────────────────────────────────────────────────────────

def shannon_entropy_without_builtins(label: str) -> float:
    """
    Compute Shannon entropy of a string WITHOUT using collections.Counter.
    H = -Σ (count/n) * log2(count/n)

    log2 implemented manually as ln(x)/ln(2) using math.log (the base
    math.log is kept since implementing ln from scratch would be out of scope).
    If reviewer insists on no math module at all, note that you would use
    a precomputed ln(2) = 0.693147... constant.
    """
    if not label:
        return 0.0

    n = len(label)

    # Manual frequency count (no Counter)
    freq = {}
    for ch in label:
        if ch in freq:
            freq[ch] += 1
        else:
            freq[ch] = 1

    LN2 = 0.6931471805599453  # math.log(2) — a constant, not a "builtin call"
    entropy = 0.0
    for count in freq.values():
        p = count / n
        # log2(p) = ln(p) / ln(2)
        entropy -= p * (math.log(p) / LN2)

    return entropy


def shannon_entropy_with_builtins(label: str) -> float:
    """
    Same calculation using math.log2 directly — cleaner.
    """
    if not label:
        return 0.0
    from collections import Counter
    n = len(label)
    freq = Counter(label)
    return -sum((c / n) * math.log2(c / n) for c in freq.values())


# ─────────────────────────────────────────────────────────────────────────────
# PARSER
# ─────────────────────────────────────────────────────────────────────────────

LINE_RE = re.compile(
    r'(?P<ts>\S+)\s+'
    r'client=(?P<client>\S+)\s+'
    r'query=(?P<query>\S+)\s+'
    r'type=(?P<qtype>\S+)'
)


def parse_dns_line(raw: str):
    """Returns dict or None."""
    m = LINE_RE.search(raw.strip())
    if not m:
        return None
    return {
        "ts":     m.group("ts"),
        "client": m.group("client"),
        "query":  m.group("query").lower(),   # normalise to lowercase
        "qtype":  m.group("qtype"),
    }


def _extract_label(fqdn: str) -> str:
    """
    Return the leftmost label of an FQDN (the 'random-looking' part for DGA).
    e.g. 'xkjdhqwe12ab.com' → 'xkjdhqwe12ab'
         'maps.google.com'  → 'maps'  (subdomain, less interesting)
    For DGA detection we check the most-left label.
    """
    parts = fqdn.rstrip(".").split(".")
    return parts[0] if parts else fqdn


# ─────────────────────────────────────────────────────────────────────────────
# APPROACH 1 — WITHOUT BUILT-IN HIGH-LEVEL HELPERS
# (manual dict counting, manual top-K, manual entropy)
# ─────────────────────────────────────────────────────────────────────────────

def _manual_top_k(count_dict, k):
    """Insertion-sort top-k descending."""
    pairs = list(count_dict.items())
    for i in range(1, len(pairs)):
        cur = pairs[i]
        j = i - 1
        while j >= 0 and pairs[j][1] < cur[1]:
            pairs[j + 1] = pairs[j]
            j -= 1
        pairs[j + 1] = cur
    return pairs[:k]


def analyze_without_builtins(lines):
    malformed      = 0
    domain_counts  = {}   # domain -> total query count
    client_domains = {}   # client_ip -> {domain: True} (manual set via dict)
    dga_candidates = {}   # domain -> entropy  (domains flagged as DGA)

    for raw in lines:
        entry = parse_dns_line(raw)
        if entry is None:
            malformed += 1
            continue

        domain = entry["query"]
        client = entry["client"]

        # Global domain count
        domain_counts[domain] = domain_counts.get(domain, 0) + 1

        # Per-client unique domains
        if client not in client_domains:
            client_domains[client] = {}
        client_domains[client][domain] = True

        # DGA check — only compute once per domain
        if domain not in dga_candidates:
            label   = _extract_label(domain)
            entropy = shannon_entropy_without_builtins(label)
            if entropy > DGA_ENTROPY_THRESHOLD:
                dga_candidates[domain] = round(entropy, 3)

    top10 = _manual_top_k(domain_counts, 10)

    # Convert client_domains from {domain: True} to counts
    client_unique = {}
    for client, dom_dict in client_domains.items():
        client_unique[client] = len(dom_dict)

    return {
        "malformed":      malformed,
        "top10":          top10,
        "dga_candidates": dga_candidates,
        "client_unique":  client_unique,
    }


# ─────────────────────────────────────────────────────────────────────────────
# APPROACH 2 — WITH BUILT-IN HELPERS
# ─────────────────────────────────────────────────────────────────────────────

def analyze_with_builtins(lines):
    malformed      = 0
    domain_counts  = defaultdict(int)
    client_domains = defaultdict(set)
    dga_candidates = {}
    seen_entropy   = {}  # cache entropy per domain

    for raw in lines:
        entry = parse_dns_line(raw)
        if entry is None:
            malformed += 1
            continue

        domain = entry["query"]
        client = entry["client"]

        domain_counts[domain]        += 1
        client_domains[client].add(domain)

        if domain not in seen_entropy:
            label   = _extract_label(domain)
            entropy = shannon_entropy_with_builtins(label)
            seen_entropy[domain] = entropy
            if entropy > DGA_ENTROPY_THRESHOLD:
                dga_candidates[domain] = round(entropy, 3)

    top10 = sorted(domain_counts.items(), key=lambda x: x[1], reverse=True)[:10]
    client_unique = {c: len(domains) for c, domains in client_domains.items()}

    return {
        "malformed":      malformed,
        "top10":          top10,
        "dga_candidates": dga_candidates,
        "client_unique":  client_unique,
    }


# ─────────────────────────────────────────────────────────────────────────────
# REPORT
# ─────────────────────────────────────────────────────────────────────────────

def print_report(result, label=""):
    print(f"\n{'='*60}")
    print(f"  DNS QUERY LOG ANALYSIS  {label}")
    print(f"{'='*60}")
    print(f"Malformed lines skipped: {result['malformed']}")
    print(f"DGA entropy threshold  : > {DGA_ENTROPY_THRESHOLD} bits/char\n")

    print("Top 10 Most Queried Domains:")
    for i, (domain, count) in enumerate(result["top10"], 1):
        print(f"  {i:>2}. {domain:<35} {count} queries")

    print("\nPer-Client Unique Domain Count:")
    for client, count in sorted(result["client_unique"].items()):
        print(f"  {client:<20} {count} unique domains")

    print("\nDGA Candidates (entropy > {:.1f}):".format(DGA_ENTROPY_THRESHOLD))
    if result["dga_candidates"]:
        for domain, entropy in sorted(result["dga_candidates"].items(),
                                       key=lambda x: -x[1]):
            print(f"  {domain:<35} entropy={entropy}")
    else:
        print("  None detected.")
    print()


# ─────────────────────────────────────────────────────────────────────────────
# MAIN
# ─────────────────────────────────────────────────────────────────────────────

def main():
    if len(sys.argv) > 1:
        filepath = sys.argv[1]
        try:
            with open(filepath, "r", encoding="utf-8", errors="replace") as fh:
                lines = fh.readlines()
        except (FileNotFoundError, PermissionError, OSError) as exc:
            print(f"[ERROR] {exc}", file=sys.stderr)
            sys.exit(1)
    else:
        print("[INFO] No file provided — using built-in sample data.\n")
        lines = SAMPLE_LOG.strip().splitlines(keepends=True)

    result_manual  = analyze_without_builtins(lines)
    result_builtin = analyze_with_builtins(lines)

    print_report(result_manual,  label="(Manual / No Builtins)")
    print_report(result_builtin, label="(With Builtins)")

    # Sanity checks
    assert set(result_manual["dga_candidates"].keys()) == \
           set(result_builtin["dga_candidates"].keys()), \
           "DGA candidate mismatch!"
    print("[OK] Both approaches agree on DGA candidates.\n")


if __name__ == "__main__":
    main()
