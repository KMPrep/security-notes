"""
Q5. Syslog Severity Summarizer
================================
Parse syslog-format file, extract timestamp/hostname/process/message,
group by process, flag lines matching dangerous patterns.

Syslog format:
  Jan 15 10:00:01 hostname kernel: [12345.678] something happened
  Jan 15 10:00:02 hostname sshd[1234]: Failed password for root from 192.168.1.1

=============================================================================
SOLUTION PROGRESSION
=============================================================================

PROMPT: "Parse a syslog file. Group by process name. Flag dangerous lines."

--- CANDIDATE FIRST RESPONSE (naive split, no regex) ---
"Split each line by spaces. Index 0-2 is the timestamp, index 3 is hostname,
index 4 has the process name. I'll strip the colon from it."

Follow-up: "What about sshd[1234]: — the PID in brackets?"
→ Need to strip `[pid]` from the process name.
Follow-up: "What if a line doesn't match the format?"
→ Need graceful skip.
Follow-up: "How will you flag patterns like 'Failed password'?"
→ Precompile regexes for each detection rule.

--- CANDIDATE SECOND RESPONSE (regex for process extraction, flag list) ---
Better. Still no proper error handling or structured output.

--- FINAL GOOGLE-LEVEL RESPONSE ---
Full regex parse, compiled detection patterns, per-process grouping,
flagged events list, streaming line-by-line reading.

=============================================================================
"""

import re
import sys
from collections import defaultdict

# ─────────────────────────────────────────────────────────────────────────────
# SAMPLE DATA
# ─────────────────────────────────────────────────────────────────────────────

SAMPLE_LOG = """\
Jan 15 10:00:01 webserver kernel: [12345.678] EXT4-fs: mounted filesystem
Jan 15 10:00:02 webserver sshd[1234]: Failed password for root from 192.168.1.1
Jan 15 10:00:03 webserver sshd[1234]: Failed password for invalid user admin from 10.0.0.5
Jan 15 10:00:04 webserver nginx[5678]: 127.0.0.1 - GET /health HTTP/1.1 200
Jan 15 10:00:05 webserver sshd[1235]: Accepted publickey for alice from 10.0.0.2
Jan 15 10:00:06 webserver sudo[9999]:     bob : TTY=pts/1 ; PWD=/home/bob ; USER=root ; COMMAND=/bin/bash

THIS IS A MALFORMED LINE WITHOUT PROPER SYSLOG FORMAT
Jan 15 10:00:14 webserver sshd[1238]: Failed password for root from 192.168.1.1 port 54321 ssh2

# ─────────────────────────────────────────────────────────────────────────────
# DETECTION RULES
# ─────────────────────────────────────────────────────────────────────────────

DETECTION_RULES = [
    ("failed_password",   r"Failed password"),
    ("invalid_user",      r"[Ii]nvalid user"),
    ("connection_refused", r"[Cc]onnection refused"),
    ("sudo_shell",        r"sudo.*COMMAND=/bin/bash"),
    ("root_login_attempt", r"Failed password for root"),
]

# ─────────────────────────────────────────────────────────────────────────────
# PARSER
# ─────────────────────────────────────────────────────────────────────────────

# Standard syslog: month day time hostname process[pid]: message
# OR:             month day time hostname process: message
SYSLOG_RE = re.compile(
    r'^(?P<month>[A-Za-z]{3})\s+'
    r'(?P<day>\s*\d+)\s+'
    r'(?P<time>\d{2}:\d{2}:\d{2})\s+'
    r'(?P<host>\S+)\s+'
    r'(?P<proc>[^\[:\s]+)'   # process name (no [, :, or space)
    r'(?:\[\d+\])?'          # optional [pid]
    r':\s+'
    r'(?P<msg>.*)'
)


def parse_syslog_line(raw: str):
    """
    Returns dict with keys: timestamp_str, host, process, message.
    Returns None if the line doesn't match syslog format.
    """
    raw = raw.strip()
    if not raw:
        return None
    m = SYSLOG_RE.match(raw)
    if not m:
        return None

    ts_str = f"{m.group('month')} {m.group('day').strip()} {m.group('time')}"
    return {
        "timestamp": ts_str,
        "host":      m.group("host"),
        "process":   m.group("proc"),
        "message":   m.group("msg"),
        "raw":       raw,
    }


# ─────────────────────────────────────────────────────────────────────────────
# APPROACH 1 — WITHOUT BUILT-IN HIGH-LEVEL HELPERS
# (manual dict grouping, manual pattern matching with re, no defaultdict)
# ─────────────────────────────────────────────────────────────────────────────

def _compile_rules(rules):
    """Pre-compile all detection rule regexes."""
    return [(tag, re.compile(pattern)) for tag, pattern in rules]


def analyze_without_builtins(lines):
    """
    Returns:
      process_groups : dict { process_name -> [event_dict, ...] }
      flagged        : list of { line_number, process, timestamp, message, tags }
      malformed      : int
    """
    compiled_rules = _compile_rules(DETECTION_RULES)
    process_groups = {}
    flagged        = []
    malformed      = 0

    for line_num, raw in enumerate(lines, start=1):
        entry = parse_syslog_line(raw)
        if entry is None:
            malformed += 1
            continue

        proc = entry["process"]

        # Manual grouping (no defaultdict)
        if proc not in process_groups:
            process_groups[proc] = []
        process_groups[proc].append(entry)

        # Check each detection rule manually
        matched_tags = []
        for tag, pattern in compiled_rules:
            if pattern.search(entry["message"]):
                # Manual append-if-not-present
                already = False
                for t in matched_tags:
                    if t == tag:
                        already = True
                        break
                if not already:
                    matched_tags.append(tag)

        if matched_tags:
            flagged.append({
                "line_number": line_num,
                "process":     proc,
                "timestamp":   entry["timestamp"],
                "message":     entry["message"],
                "tags":        matched_tags,
            })

    return process_groups, flagged, malformed


# ─────────────────────────────────────────────────────────────────────────────
# APPROACH 2 — WITH BUILT-IN HELPERS (defaultdict, set, list comprehension)
# ─────────────────────────────────────────────────────────────────────────────

def analyze_with_builtins(lines):
    compiled_rules = _compile_rules(DETECTION_RULES)
    process_groups = defaultdict(list)
    flagged        = []
    malformed      = 0

    for line_num, raw in enumerate(lines, start=1):
        entry = parse_syslog_line(raw)
        if entry is None:
            malformed += 1
            continue

        process_groups[entry["process"]].append(entry)

        matched_tags = [tag for tag, pat in compiled_rules
                        if pat.search(entry["message"])]
        if matched_tags:
            flagged.append({
                "line_number": line_num,
                "process":     entry["process"],
                "timestamp":   entry["timestamp"],
                "message":     entry["message"],
                "tags":        matched_tags,
            })

    return dict(process_groups), flagged, malformed


# ─────────────────────────────────────────────────────────────────────────────
# REPORT
# ─────────────────────────────────────────────────────────────────────────────

def print_report(process_groups, flagged, malformed, label=""):
    print(f"\n{'='*70}")
    print(f"  SYSLOG SEVERITY SUMMARY  {label}")
    print(f"{'='*70}")
    print(f"Malformed lines skipped: {malformed}\n")

    print("Per-Process Event Count:")
    for proc in sorted(process_groups.keys()):
        count = len(process_groups[proc])
        print(f"  {proc:<20} {count} events")

    print(f"\nFlagged Events ({len(flagged)} total):")
    if not flagged:
        print("  None.")
    else:
        for ev in flagged:
            tags_str = ", ".join(ev["tags"])
            print(f"  Line {ev['line_number']:>4} | {ev['timestamp']} | "
                  f"{ev['process']:<12} | [{tags_str}]")
            # Truncate long messages for readability
            msg = ev["message"]
            if len(msg) > 80:
                msg = msg[:77] + "..."
            print(f"             {msg}")
    print()


# ─────────────────────────────────────────────────────────────────────────────
# STREAMING FILE READER
# ─────────────────────────────────────────────────────────────────────────────

def stream_lines(filepath: str):
    try:
        with open(filepath, "r", encoding="utf-8", errors="replace") as fh:
            for line in fh:
                yield line
    except FileNotFoundError:
        print(f"[ERROR] File not found: {filepath}", file=sys.stderr)
    except PermissionError:
        print(f"[ERROR] Permission denied: {filepath}", file=sys.stderr)
    except OSError as exc:
        print(f"[ERROR] OS error: {exc}", file=sys.stderr)


# ─────────────────────────────────────────────────────────────────────────────
# MAIN
# ─────────────────────────────────────────────────────────────────────────────

def main():
    if len(sys.argv) > 1:
        lines = list(stream_lines(sys.argv[1]))
        if not lines:
            print("[WARN] No lines read.", file=sys.stderr)
            sys.exit(1)
    else:
        print("[INFO] No file provided — using built-in sample data.\n")
        lines = SAMPLE_LOG.strip().splitlines(keepends=True)

    pg_manual, fl_manual, mal_manual = analyze_without_builtins(lines)
    pg_builtin, fl_builtin, mal_builtin = analyze_with_builtins(lines)

    print_report(pg_manual, fl_manual, mal_manual, label="(Manual / No Builtins)")
    print_report(pg_builtin, fl_builtin, mal_builtin, label="(With Builtins)")

    # Sanity check
    assert mal_manual == mal_builtin, "Malformed count mismatch!"
    assert len(fl_manual) == len(fl_builtin), "Flagged count mismatch!"
    print("[OK] Both approaches agree.\n")


if __name__ == "__main__":
    main()
