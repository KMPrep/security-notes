"""
Q6. JSON Log Normalizer
========================
Given a JSONL file with inconsistent schemas, normalize every record to a
canonical form: {"timestamp": ..., "src_ip": ..., "action": ...}

Known field aliases:
  timestamp : "ts", "timestamp", "time", "date"
  src_ip    : "src_ip", "ip", "source", "client_ip", "src"
  action    : "action", "event", "type", "verdict", "result"

Missing fields → None in output.

=============================================================================
SOLUTION PROGRESSION
=============================================================================

PROMPT: "You have logs from 5 different products, all different schemas.
Normalize them to one canonical schema."

--- CANDIDATE FIRST RESPONSE (hardcoded if-else per variant) ---
"I'd check what keys are in the dict and pick the right one with if/elif."

Follow-up: "What if a 6th product is added?"
→ Hardcoded approach breaks. Better: define an alias mapping as data.

--- CANDIDATE SECOND RESPONSE (alias map, dict.get cascade) ---
"Define a dict of canonical_name → [list of aliases].
For each incoming record, try each alias in order until one matches."

Follow-up: "What if the value itself needs normalization? e.g. 'BLOCK'
vs 'block' vs 'deny' vs 'DROP'?"
→ Add a value normalizer step after field extraction.

Follow-up: "What about lines that are not valid JSON at all?"
→ Wrap json.loads in try/except, skip with a warning.

--- FINAL GOOGLE-LEVEL RESPONSE ---
Alias mapping as data, value normalization, full error handling,
streaming JSONL reader, output to JSONL.

=============================================================================
"""

import sys
import json

# ─────────────────────────────────────────────────────────────────────────────
# SAMPLE DATA
# ─────────────────────────────────────────────────────────────────────────────

SAMPLE_JSONL = [
    '{"ts": "2024-01-15", "src_ip": "1.1.1.1", "action": "BLOCK"}',
    '{"timestamp": "2024-01-15T10:00:00", "ip": "2.2.2.2", "event": "deny"}',
    '{"time": "2024-01-15T11:00:00Z", "source": "3.3.3.3", "verdict": "allow"}',
    '{"date": "2024-01-15", "client_ip": "4.4.4.4", "type": "ALLOW"}',
    '{"ts": "2024-01-15T12:00:00Z", "src": "5.5.5.5", "result": "DROP"}',
    '{"timestamp": "2024-01-15T13:00:00Z", "ip": "6.6.6.6"}',  # missing action
    '{"ts": "2024-01-15T14:00:00Z"}',                           # missing ip and action
    'THIS IS NOT VALID JSON {{{',                                # malformed JSON
    '',                                                          # empty line
    '{"ts": "2024-01-15T15:00:00Z", "src_ip": "7.7.7.7", "action": "ACCEPT"}',
]

# ─────────────────────────────────────────────────────────────────────────────
# SCHEMA CONFIGURATION
# ─────────────────────────────────────────────────────────────────────────────

# Canonical field → list of known aliases (tried in order)
FIELD_ALIASES = {
    "timestamp": ["timestamp", "ts", "time", "date"],
    "src_ip":    ["src_ip", "ip", "source", "client_ip", "src"],
    "action":    ["action", "event", "verdict", "type", "result"],
}

# Value normalization for 'action' field
ACTION_MAP = {
    "block":  "BLOCK",
    "deny":   "BLOCK",
    "drop":   "BLOCK",
    "reject": "BLOCK",
    "allow":  "ALLOW",
    "accept": "ALLOW",
    "permit": "ALLOW",
    "pass":   "ALLOW",
}


# ─────────────────────────────────────────────────────────────────────────────
# APPROACH 1 — WITHOUT BUILT-IN HIGH-LEVEL HELPERS
# (manual key lookup, no dict comprehension, no json module — parse manually)
#
# Note: truly parsing JSON without json.loads would require writing a full
# JSON parser — that's out of scope. The "without builtins" here means:
# no dict comprehensions, no .get() chaining in one expression,
# no str.lower() in a generator — do everything step-by-step.
# ─────────────────────────────────────────────────────────────────────────────

def _find_field_manual(record, aliases):
    """
    Given a dict `record` and a list of `aliases`, return (found_value, True)
    for the first alias that exists in the record.
    Returns (None, False) if none found.
    """
    for alias in aliases:
        if alias in record:
            return record[alias], True
    return None, False


def _normalize_action_manual(value):
    """Normalize action values to ALLOW/BLOCK. Return as-is if unknown."""
    if value is None:
        return None
    lowered = ""
    for ch in str(value):
        lowered += ch.lower()   # manual lower without str.lower()
    if lowered in ACTION_MAP:
        return ACTION_MAP[lowered]
    return str(value).upper()  # best-effort uppercase


def normalize_record_without_builtins(record):
    """
    Normalize a single dict to canonical schema.
    Returns a new dict with keys: timestamp, src_ip, action.
    """
    canonical = {}

    ts_val, _ = _find_field_manual(record, FIELD_ALIASES["timestamp"])
    canonical["timestamp"] = ts_val

    ip_val, _ = _find_field_manual(record, FIELD_ALIASES["src_ip"])
    canonical["src_ip"] = ip_val

    action_val, _ = _find_field_manual(record, FIELD_ALIASES["action"])
    canonical["action"] = _normalize_action_manual(action_val)

    return canonical


def process_jsonl_without_builtins(lines):
    """
    Process JSONL lines one at a time (generator pattern).
    Yields normalized dicts.
    Reports errors to stderr.
    """
    stats = {"parsed": 0, "malformed": 0, "empty": 0}

    for line_num, raw in enumerate(lines, start=1):
        raw = raw.strip()
        if not raw:
            stats["empty"] += 1
            continue

        try:
            record = json.loads(raw)
        except json.JSONDecodeError as exc:
            print(f"[WARN] Line {line_num}: JSON parse error — {exc}",
                  file=sys.stderr)
            stats["malformed"] += 1
            continue

        if not isinstance(record, dict):
            print(f"[WARN] Line {line_num}: Expected JSON object, got {type(record).__name__}",
                  file=sys.stderr)
            stats["malformed"] += 1
            continue

        stats["parsed"] += 1
        normalized = normalize_record_without_builtins(record)
        yield normalized

    return stats


# ─────────────────────────────────────────────────────────────────────────────
# APPROACH 2 — WITH BUILT-IN HELPERS (dict.get, comprehensions, str.lower)
# ─────────────────────────────────────────────────────────────────────────────

def _find_field_builtin(record, aliases):
    """Return value of first matching alias, or None."""
    for alias in aliases:
        val = record.get(alias)
        if val is not None:
            return val
    return None


def _normalize_action_builtin(value):
    if value is None:
        return None
    return ACTION_MAP.get(str(value).lower(), str(value).upper())


def normalize_record_with_builtins(record):
    return {
        "timestamp": _find_field_builtin(record, FIELD_ALIASES["timestamp"]),
        "src_ip":    _find_field_builtin(record, FIELD_ALIASES["src_ip"]),
        "action":    _normalize_action_builtin(
                         _find_field_builtin(record, FIELD_ALIASES["action"])),
    }


def process_jsonl_with_builtins(lines):
    stats = {"parsed": 0, "malformed": 0, "empty": 0}

    for line_num, raw in enumerate(lines, start=1):
        raw = raw.strip()
        if not raw:
            stats["empty"] += 1
            continue

        try:
            record = json.loads(raw)
        except json.JSONDecodeError as exc:
            print(f"[WARN] Line {line_num}: JSON parse error — {exc}",
                  file=sys.stderr)
            stats["malformed"] += 1
            continue

        if not isinstance(record, dict):
            print(f"[WARN] Line {line_num}: Expected JSON object.",
                  file=sys.stderr)
            stats["malformed"] += 1
            continue

        stats["parsed"] += 1
        yield normalize_record_with_builtins(record)

    return stats  # NOTE: generators can't return values easily; use a wrapper


# ─────────────────────────────────────────────────────────────────────────────
# WRAPPER to capture stats from generators
# ─────────────────────────────────────────────────────────────────────────────

def run_normalizer(lines, normalizer_func):
    """Run a normalizer generator and collect results + stats."""
    stats = {"parsed": 0, "malformed": 0, "empty": 0}
    results = []

    for line_num, raw in enumerate(lines, start=1):
        raw = raw.strip()
        if not raw:
            stats["empty"] += 1
            continue

        try:
            record = json.loads(raw)
        except json.JSONDecodeError as exc:
            print(f"[WARN] Line {line_num}: {exc}", file=sys.stderr)
            stats["malformed"] += 1
            continue

        if not isinstance(record, dict):
            stats["malformed"] += 1
            continue

        stats["parsed"] += 1
        results.append(normalizer_func(record))

    return results, stats


# ─────────────────────────────────────────────────────────────────────────────
# REPORT + OUTPUT
# ─────────────────────────────────────────────────────────────────────────────

def print_report(results, stats, label=""):
    print(f"\n{'='*60}")
    print(f"  JSON LOG NORMALIZER  {label}")
    print(f"{'='*60}")
    print(f"Lines processed : {stats['parsed'] + stats['malformed'] + stats['empty']}")
    print(f"  Parsed OK     : {stats['parsed']}")
    print(f"  Malformed JSON: {stats['malformed']}")
    print(f"  Empty lines   : {stats['empty']}")
    print(f"\nNormalized records ({len(results)}):")
    for rec in results:
        print(f"  {json.dumps(rec)}")
    print()


def write_jsonl(results, filepath: str):
    """Write normalized records to a JSONL output file."""
    try:
        with open(filepath, "w", encoding="utf-8") as fh:
            for rec in results:
                fh.write(json.dumps(rec) + "\n")
        print(f"[INFO] Output written to: {filepath}")
    except (PermissionError, OSError) as exc:
        print(f"[ERROR] Could not write output: {exc}", file=sys.stderr)


# ─────────────────────────────────────────────────────────────────────────────
# FILE READER
# ─────────────────────────────────────────────────────────────────────────────

def read_jsonl_file(filepath: str):
    """Read JSONL file lines. Returns empty list on error."""
    try:
        with open(filepath, "r", encoding="utf-8", errors="replace") as fh:
            return fh.readlines()
    except FileNotFoundError:
        print(f"[ERROR] File not found: {filepath}", file=sys.stderr)
        return []
    except PermissionError:
        print(f"[ERROR] Permission denied: {filepath}", file=sys.stderr)
        return []
    except OSError as exc:
        print(f"[ERROR] {exc}", file=sys.stderr)
        return []


# ─────────────────────────────────────────────────────────────────────────────
# MAIN
# ─────────────────────────────────────────────────────────────────────────────

def main():
    if len(sys.argv) > 1:
        filepath = sys.argv[1]
        lines = read_jsonl_file(filepath)
        if not lines:
            sys.exit(1)
    else:
        print("[INFO] No file provided — using built-in sample data.\n")
        lines = SAMPLE_JSONL

    results_manual, stats_manual   = run_normalizer(lines, normalize_record_without_builtins)
    results_builtin, stats_builtin = run_normalizer(lines, normalize_record_with_builtins)

    print_report(results_manual,  stats_manual,  label="(Manual / No Builtins)")
    print_report(results_builtin, stats_builtin, label="(With Builtins)")

    # Optionally write output
    # write_jsonl(results_builtin, "normalized_output.jsonl")

    # Sanity check
    assert len(results_manual) == len(results_builtin), "Result count mismatch!"
    for r1, r2 in zip(results_manual, results_builtin):
        assert r1 == r2, f"Record mismatch: {r1} vs {r2}"
    print("[OK] Both approaches produce identical normalized records.\n")


if __name__ == "__main__":
    main()
