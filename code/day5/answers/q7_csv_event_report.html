"""

=============================================================================

REQUIRED_COLUMNS = {
    "event_id", "timestamp", "severity",
    "source_ip", "destination_ip", "event_type", "description"
}
SEVERITY_ORDER = {"CRITICAL": 0, "HIGH": 1, "MEDIUM": 2, "LOW": 3, "INFO": 4}


# ─────────────────────────────────────────────────────────────────────────────
# APPROACH 1 — WITHOUT BUILT-IN HIGH-LEVEL HELPERS
# (manual CSV parsing using csv.reader not DictReader, manual dicts, no sorted key)
# ─────────────────────────────────────────────────────────────────────────────

def _manual_insertion_sort_by_count(pairs):
    """Sort (key, count) pairs descending by count in-place."""
    for i in range(1, len(pairs)):
        cur = pairs[i]
        j = i - 1
        while j >= 0 and pairs[j][1] < cur[1]:
            pairs[j + 1] = pairs[j]
            j -= 1
        pairs[j + 1] = cur


def analyze_without_builtins(csv_text: str):
    """
    Uses csv.reader (lower-level than DictReader) + manual dict mapping.
    No defaultdict, no sorted with key=, no Counter.
    """
    reader = csv.reader(io.StringIO(csv_text))

    # Read header row
    try:
        header = next(reader)
    except StopIteration:
        print("[ERROR] CSV is empty.", file=sys.stderr)
        return {}, {}, [], 0

    # Strip whitespace from header columns
    header = [h.strip() for h in header]

    # Build column index map manually
    col_index = {}
    for i, col in enumerate(header):
        col_index[col] = i

    # Validate required columns
    for col in REQUIRED_COLUMNS:
        if col not in col_index:
            print(f"[ERROR] Missing required column: {col}", file=sys.stderr)
            return {}, {}, [], 0

    severity_groups = {}   # severity -> list of row dicts
    pair_counts     = {}   # (src_ip, dst_ip) -> count
    critical_events = []
    malformed       = 0

    for row_num, row in enumerate(reader, start=2):
        if len(row) < len(header):
            print(f"[WARN] Row {row_num}: fewer columns than header, skipping.",
                  file=sys.stderr)
            malformed += 1
            continue

        # Build a dict from the row manually
        record = {}
        for col, idx in col_index.items():
            record[col] = row[idx].strip() if idx < len(row) else ""

        severity = record.get("severity", "").upper()
        src_ip   = record.get("source_ip", "")
        dst_ip   = record.get("destination_ip", "")

        if not severity:
            print(f"[WARN] Row {row_num}: empty severity, skipping.",
                  file=sys.stderr)
            malformed += 1
            continue

        # Severity grouping (manual)
        if severity not in severity_groups:
            severity_groups[severity] = []
        severity_groups[severity].append(record)

        # (src, dst) pair counting (manual)
        pair = (src_ip, dst_ip)
        pair_counts[pair] = pair_counts.get(pair, 0) + 1

        if severity == "CRITICAL":
            critical_events.append(record)

    # Convert pair_counts to sorted list
    pair_list = list(pair_counts.items())
    _manual_insertion_sort_by_count(pair_list)

    return severity_groups, pair_list, critical_events, malformed


# ─────────────────────────────────────────────────────────────────────────────
# APPROACH 2 — WITH BUILT-IN HELPERS (csv.DictReader, defaultdict, sorted)
# ─────────────────────────────────────────────────────────────────────────────

def analyze_with_builtins(csv_text: str):
    reader = csv.DictReader(
        io.StringIO(csv_text),
        skipinitialspace=True
    )

    # Validate header
    if reader.fieldnames is None:
        print("[ERROR] CSV has no headers.", file=sys.stderr)
        return {}, [], [], 0

    missing = REQUIRED_COLUMNS - set(reader.fieldnames)
    if missing:
        print(f"[ERROR] Missing columns: {missing}", file=sys.stderr)
        return {}, [], [], 0

    severity_groups = defaultdict(list)
    pair_counts     = defaultdict(int)
    critical_events = []
    malformed       = 0

    for row_num, row in enumerate(reader, start=2):
        severity = (row.get("severity") or "").strip().upper()
        src_ip   = (row.get("source_ip") or "").strip()
        dst_ip   = (row.get("destination_ip") or "").strip()

        if not severity:
            print(f"[WARN] Row {row_num}: empty severity.", file=sys.stderr)
            malformed += 1
            continue

        severity_groups[severity].append(dict(row))
        pair_counts[(src_ip, dst_ip)] += 1

        if severity == "CRITICAL":
            critical_events.append(dict(row))

    pair_list = sorted(pair_counts.items(), key=lambda x: x[1], reverse=True)

    return dict(severity_groups), pair_list, critical_events, malformed


# ─────────────────────────────────────────────────────────────────────────────
# REPORT PRINTER
# ─────────────────────────────────────────────────────────────────────────────

def print_report(severity_groups, pair_list, critical_events, malformed, label=""):
    print(f"\n{'='*70}")
    print(f"  CSV SECURITY EVENT REPORT  {label}")
    print(f"{'='*70}")
    print(f"Malformed / skipped rows: {malformed}\n")

    print("Events by Severity:")
    for sev in ["CRITICAL", "HIGH", "MEDIUM", "LOW", "INFO"]:
        events = severity_groups.get(sev, [])
        if events:
            print(f"  {sev:<10} {len(events):>4} events")

    print(f"\nTop (src_ip, dst_ip) Pairs by Event Count:")
    for (src, dst), count in pair_list[:10]:   # top-10
        print(f"  {src:<20} → {dst:<20} : {count} events")

    print(f"\nCRITICAL Events ({len(critical_events)}):")
    for ev in critical_events:
        print(f"  [{ev.get('event_id','?')}] {ev.get('timestamp','')}  "
              f"{ev.get('event_type','')}  {ev.get('source_ip','')} → "
              f"{ev.get('destination_ip','')}  \"{ev.get('description','')}\"")
    print()


# ─────────────────────────────────────────────────────────────────────────────
# CSV WRITER for CRITICAL events
# ─────────────────────────────────────────────────────────────────────────────

def write_critical_csv(critical_events, filepath: str):
    """Write CRITICAL events to a new CSV file."""
    if not critical_events:
        print("[INFO] No CRITICAL events to write.")
        return
    try:
        fieldnames = list(REQUIRED_COLUMNS)  # consistent column order
        with open(filepath, "w", newline="", encoding="utf-8") as fh:
            writer = csv.DictWriter(fh, fieldnames=fieldnames,
                                    extrasaction="ignore")
            writer.writeheader()
            for ev in critical_events:
                writer.writerow(ev)
        print(f"[INFO] CRITICAL events written to: {filepath}")
    except (PermissionError, OSError) as exc:
        print(f"[ERROR] Could not write {filepath}: {exc}", file=sys.stderr)


# ─────────────────────────────────────────────────────────────────────────────
# FILE READER
# ─────────────────────────────────────────────────────────────────────────────

def read_csv_text(filepath: str) -> str:
    """Read entire CSV text. Returns empty string on error."""
    try:
        with open(filepath, "r", encoding="utf-8-sig",
                  errors="replace", newline="") as fh:
            return fh.read()
    except FileNotFoundError:
        print(f"[ERROR] File not found: {filepath}", file=sys.stderr)
        return ""
    except PermissionError:
        print(f"[ERROR] Permission denied: {filepath}", file=sys.stderr)
        return ""
    except OSError as exc:
        print(f"[ERROR] {exc}", file=sys.stderr)
        return ""


# ─────────────────────────────────────────────────────────────────────────────
# MAIN
# ─────────────────────────────────────────────────────────────────────────────

def main():
    if len(sys.argv) > 1:
        csv_text = read_csv_text(sys.argv[1])
        if not csv_text:
            sys.exit(1)
    else:
        print("[INFO] No file provided — using built-in sample data.\n")
        csv_text = SAMPLE_CSV

    sg_manual,  pl_manual,  crit_manual,  mal_manual  = analyze_without_builtins(csv_text)
    sg_builtin, pl_builtin, crit_builtin, mal_builtin = analyze_with_builtins(csv_text)

    print_report(sg_manual,  pl_manual,  crit_manual,  mal_manual,  label="(Manual)")
    print_report(sg_builtin, pl_builtin, crit_builtin, mal_builtin, label="(Builtins)")

    # Write critical events CSV (use builtin result — the cleaner one)
    # write_critical_csv(crit_builtin, "critical_events.csv")

    # Sanity checks
    assert len(crit_manual) == len(crit_builtin), "CRITICAL count mismatch!"
    print("[OK] Both approaches agree on CRITICAL event count.\n")


if __name__ == "__main__":
    main()
