"""
Q8. Log Line Enricher
======================
Given a log file of IPs and a threat intel dict mapping IP → category,
enrich each log line: append the threat category if the IP is known bad.
Flag enriched lines for review.
Output: enriched lines, count by category.

threat_intel = {"1.2.3.4": "C2", "5.6.7.8": "scanner", ...}

=============================================================================
SOLUTION PROGRESSION
=============================================================================

PROMPT: "You have a log file. For each line, check if the IP appears in
a threat intel feed and annotate the line."

--- CANDIDATE FIRST RESPONSE (linear scan through threat intel per line) ---
"For each log line, I'd extract the IP and then loop through the threat intel
dict to find a match."

Follow-up: "What's the complexity of that?"
→ O(L * T) where L = lines, T = threat intel entries. If T = 10M and L = 100M
  that's 10^15 operations. Way too slow.

Follow-up: "How do you fix it?"
→ The threat intel dict is already a hash map — O(1) lookup.
  Just do `threat_intel.get(ip)`. No need to loop through it.
  Candidate was confused because they thought of it as a list first.

--- CANDIDATE SECOND RESPONSE (dict lookup, but no regex for IP extraction) ---
"Use ip = line.split()[3]" — positional. Breaks if format varies.

Follow-up: "What if IPs appear anywhere in the line?"
→ Use regex to find all IPs in the line, check each against threat intel.

--- FINAL GOOGLE-LEVEL RESPONSE ---
Regex IP extraction, O(1) dict lookup, streaming line-by-line,
enriched output with category counts, original line preserved.

=============================================================================
"""

import re
import sys
from collections import defaultdict

# ─────────────────────────────────────────────────────────────────────────────
# THREAT INTEL (in production this would load from a file / database / API)
# ─────────────────────────────────────────────────────────────────────────────

THREAT_INTEL = {
    "1.2.3.4":      "C2",
    "5.6.7.8":      "scanner",
    "10.0.0.99":    "C2",
    "192.168.1.50": "exfil",
    "203.0.113.10": "phishing",
    "198.51.100.5": "botnet",
}

# ─────────────────────────────────────────────────────────────────────────────
# SAMPLE LOG
# ─────────────────────────────────────────────────────────────────────────────

SAMPLE_LOG = """\
2024-01-15T10:00:00Z src=1.2.3.4 dst=172.16.0.1 bytes=1024 action=ALLOW
2024-01-15T10:00:01Z src=10.0.0.1 dst=8.8.8.8 bytes=512 action=ALLOW
2024-01-15T10:00:02Z src=5.6.7.8 dst=172.16.0.2 bytes=4096 action=ALLOW
2024-01-15T10:00:03Z src=10.0.0.2 dst=10.0.0.99 bytes=256 action=ALLOW
2024-01-15T10:00:04Z src=192.168.1.50 dst=203.0.113.10 bytes=99999 action=ALLOW

2024-01-15T10:00:06Z src=198.51.100.5 dst=172.16.0.3 bytes=2048 action=DENY

# ─────────────────────────────────────────────────────────────────────────────
# IP EXTRACTOR
# ─────────────────────────────────────────────────────────────────────────────

# Strict IPv4 pattern (avoid matching version strings like 10.0.0.99x)
IP_RE = re.compile(r'\b(?:(?:25[0-5]|2[0-4]\d|[01]?\d\d?)\.){3}'
                   r'(?:25[0-5]|2[0-4]\d|[01]?\d\d?)\b')


def extract_ips(line: str):
    """Return list of all IPv4 addresses found in a line."""
    return IP_RE.findall(line)


# ─────────────────────────────────────────────────────────────────────────────
# APPROACH 1 — WITHOUT BUILT-IN HIGH-LEVEL HELPERS
# (no defaultdict, manual dict for category counts, no any()/all())
# ─────────────────────────────────────────────────────────────────────────────

def enrich_without_builtins(lines, threat_intel):
    """
    Returns:
      enriched_lines : list of strings (original line + optional annotations)
      flagged_lines  : list of (line_number, line, categories)
      category_counts: dict { category -> count of enriched lines }
    """
    enriched_lines  = []
    flagged_lines   = []
    category_counts = {}

    for line_num, raw in enumerate(lines, start=1):
        raw = raw.rstrip("\n")
        if not raw.strip():
            enriched_lines.append(raw)
            continue

        ips = extract_ips(raw)
        matched_categories = []

        # Check each IP against threat intel — O(1) per lookup
        for ip in ips:
            if ip in threat_intel:
                category = threat_intel[ip]
                # Manual dedup
                already = False
                for c in matched_categories:
                    if c == category:
                        already = True
                        break
                if not already:
                    matched_categories.append(category)

        if matched_categories:
            # Append annotation to line
            annotation = " [THREAT:" + ",".join(matched_categories) + "]"
            enriched_line = raw + annotation
            enriched_lines.append(enriched_line)
            flagged_lines.append((line_num, enriched_line, matched_categories))

            # Count by category (manual)
            for cat in matched_categories:
                category_counts[cat] = category_counts.get(cat, 0) + 1
        else:
            enriched_lines.append(raw)

    return enriched_lines, flagged_lines, category_counts


# ─────────────────────────────────────────────────────────────────────────────
# APPROACH 2 — WITH BUILT-IN HELPERS (defaultdict, set, f-strings)
# ─────────────────────────────────────────────────────────────────────────────

def enrich_with_builtins(lines, threat_intel):
    enriched_lines  = []
    flagged_lines   = []
    category_counts = defaultdict(int)

    for line_num, raw in enumerate(lines, start=1):
        raw = raw.rstrip("\n")
        if not raw.strip():
            enriched_lines.append(raw)
            continue

        ips = extract_ips(raw)
        # Use a set to deduplicate categories, preserve insertion order with dict
        matched_categories = list(dict.fromkeys(
            threat_intel[ip] for ip in ips if ip in threat_intel
        ))

        if matched_categories:
            annotation    = f" [THREAT:{','.join(matched_categories)}]"
            enriched_line = raw + annotation
            enriched_lines.append(enriched_line)
            flagged_lines.append((line_num, enriched_line, matched_categories))
            for cat in matched_categories:
                category_counts[cat] += 1
        else:
            enriched_lines.append(raw)

    return enriched_lines, flagged_lines, dict(category_counts)


# ─────────────────────────────────────────────────────────────────────────────
# REPORT
# ─────────────────────────────────────────────────────────────────────────────

def print_report(enriched_lines, flagged_lines, category_counts, label=""):
    print(f"\n{'='*70}")
    print(f"  LOG LINE ENRICHER REPORT  {label}")
    print(f"{'='*70}")
    print(f"Total lines      : {len(enriched_lines)}")
    print(f"Enriched (flagged): {len(flagged_lines)}\n")

    print("Category Counts:")
    if category_counts:
        for cat, count in sorted(category_counts.items()):
            print(f"  {cat:<15} {count} line(s)")
    else:
        print("  None.")

    print(f"\nFlagged Lines:")
    for line_num, line, cats in flagged_lines:
        print(f"  Line {line_num:>3}: {line}")
    print()


def write_enriched_output(enriched_lines, filepath: str):
    """Write enriched lines to a file."""
    try:
        with open(filepath, "w", encoding="utf-8") as fh:
            for line in enriched_lines:
                fh.write(line + "\n")
        print(f"[INFO] Enriched output written to: {filepath}")
    except (PermissionError, OSError) as exc:
        print(f"[ERROR] Could not write output: {exc}", file=sys.stderr)


# ─────────────────────────────────────────────────────────────────────────────
# THREAT INTEL LOADER (from a flat file: ip<tab>category)
# ─────────────────────────────────────────────────────────────────────────────

def load_threat_intel(filepath: str) -> dict:
    """
    Load threat intel from a file with lines: ip<whitespace>category
    Returns dict. Malformed lines are skipped with a warning.
    """
    intel = {}
    try:
        with open(filepath, "r", encoding="utf-8", errors="replace") as fh:
            for line_num, raw in enumerate(fh, start=1):
                raw = raw.strip()
                if not raw or raw.startswith("#"):
                    continue
                parts = raw.split()
                if len(parts) < 2:
                    print(f"[WARN] Threat intel line {line_num}: malformed, skipping.",
                          file=sys.stderr)
                    continue
                intel[parts[0]] = parts[1]
    except FileNotFoundError:
        print(f"[WARN] Threat intel file not found: {filepath}. "
              f"Using empty intel.", file=sys.stderr)
    except (PermissionError, OSError) as exc:
        print(f"[ERROR] Could not read threat intel: {exc}", file=sys.stderr)
    return intel


# ─────────────────────────────────────────────────────────────────────────────
# MAIN
# ─────────────────────────────────────────────────────────────────────────────

def main():
    """
    Usage:
      python q8_log_enricher.py [logfile] [threat_intel_file]
      python q8_log_enricher.py                 ← uses sample data
    """
    if len(sys.argv) == 3:
        logfile    = sys.argv[1]
        intel_file = sys.argv[2]
        threat_intel = load_threat_intel(intel_file)
        try:
            with open(logfile, "r", encoding="utf-8", errors="replace") as fh:
                lines = fh.readlines()
        except (FileNotFoundError, PermissionError, OSError) as exc:
            print(f"[ERROR] {exc}", file=sys.stderr)
            sys.exit(1)
    elif len(sys.argv) == 1:
        print("[INFO] Using built-in sample data and threat intel.\n")
        lines        = SAMPLE_LOG.strip().splitlines(keepends=True)
        threat_intel = THREAT_INTEL
    else:
        print("Usage: python q8_log_enricher.py [logfile] [threat_intel_file]",
              file=sys.stderr)
        sys.exit(1)

    if not threat_intel:
        print("[WARN] Threat intel is empty — no enrichment will occur.",
              file=sys.stderr)

    enriched_manual,  flagged_manual,  cats_manual  = enrich_without_builtins(lines, threat_intel)
    enriched_builtin, flagged_builtin, cats_builtin = enrich_with_builtins(lines, threat_intel)

    print_report(enriched_manual,  flagged_manual,  cats_manual,  label="(Manual)")
    print_report(enriched_builtin, flagged_builtin, cats_builtin, label="(Builtins)")

    # write_enriched_output(enriched_builtin, "enriched.log")

    # Sanity check
    assert len(flagged_manual) == len(flagged_builtin), "Flagged count mismatch!"
    assert cats_manual == cats_builtin, f"Category count mismatch: {cats_manual} vs {cats_builtin}"
    print("[OK] Both approaches agree.\n")


if __name__ == "__main__":
    main()
