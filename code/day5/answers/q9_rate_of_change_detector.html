"""
Q9. Rate-of-Change Anomaly Detector
=====================================
Given a time-series list of (minute, event_count) tuples, flag any minute
where the count increased by more than 3x compared to the previous minute.

Output: list of (minute, prev_count, curr_count, multiplier)

=============================================================================
SOLUTION PROGRESSION
=============================================================================

PROMPT: "Detect DDoS spikes: flag any minute where event count jumps
more than 3x over the previous minute."

--- CANDIDATE FIRST RESPONSE (simple loop, no edge case handling) ---
"Loop from index 1 to end. If data[i] > 3 * data[i-1], flag it."

Follow-up: "What if prev_count is 0?"
→ Division by zero / infinite multiplier. Need to handle zero baseline.
  Options: skip if prev == 0, or treat any non-zero following a zero as a flag.

Follow-up: "What if data is not sorted by minute?"
→ Sort by minute first.

Follow-up: "What if there are gaps in minutes (minute 5 missing, then 6)?"
→ Should the gap count as '0 events' for minute 5? Likely yes for DDoS detection.
  Fill gaps with 0 before computing rate-of-change.

--- FINAL GOOGLE-LEVEL RESPONSE ---
Sort by minute, fill gaps with 0, handle zero-baseline,
compute multiplier as float, clear output format.

=============================================================================
"""

import sys

# ─────────────────────────────────────────────────────────────────────────────
# SAMPLE DATA
# ─────────────────────────────────────────────────────────────────────────────

# (minute, event_count) tuples
SAMPLE_DATA = [
    (1,  10),
    (2,  12),
    (3,  11),
    (4,  13),
    (5,   0),   # quiet minute — gap filled
    (6,  45),   # spike after quiet: 0 → 45 (treat 0-baseline specially)
    (7,  50),
    (8, 180),   # 180 / 50 = 3.6x — should be FLAGGED
    (9,  55),
    (10, 200),  # 200 / 55 = 3.6x — should be FLAGGED
    (11, 210),
    (12, 205),
    (13,  60),
    # Gap: minute 14 is missing
    (15, 250),  # comes after a gap — gap filled with 0
    (16, 255),
]

SPIKE_THRESHOLD = 3.0  # flag if curr / prev > this multiplier


# ─────────────────────────────────────────────────────────────────────────────
# GAP FILLER
# ─────────────────────────────────────────────────────────────────────────────

def fill_gaps(time_series):
    """
    Given a list of (minute, count) sorted by minute,
    return a new list with missing minutes filled as (minute, 0).
    """
    if not time_series:
        return []

    filled = []
    for i, (minute, count) in enumerate(time_series):
        if i == 0:
            filled.append((minute, count))
        else:
            prev_minute = time_series[i - 1][0]
            # Fill any missing minutes in between
            for gap_minute in range(prev_minute + 1, minute):
                filled.append((gap_minute, 0))
            filled.append((minute, count))

    return filled


# ─────────────────────────────────────────────────────────────────────────────
# APPROACH 1 — WITHOUT BUILT-IN HIGH-LEVEL HELPERS
# (manual sort, manual gap fill, no enumerate, explicit indexing)
# ─────────────────────────────────────────────────────────────────────────────

def _manual_sort_by_minute(data):
    """Sort list of (minute, count) tuples by minute ascending."""
    # Insertion sort
    for i in range(1, len(data)):
        cur = data[i]
        j = i - 1
        while j >= 0 and data[j][0] > cur[0]:
            data[j + 1] = data[j]
            j -= 1
        data[j + 1] = cur
    return data


def detect_without_builtins(time_series):
    """
    Returns list of (minute, prev_count, curr_count, multiplier) for spikes.
    """
    if not time_series:
        return []

    # Validate input
    for item in time_series:
        if not (isinstance(item, (list, tuple)) and len(item) == 2):
            raise ValueError(f"Expected (minute, count) tuple, got: {item}")
        if item[1] < 0:
            raise ValueError(f"Event count cannot be negative: {item}")

    # Sort by minute
    data = _manual_sort_by_minute(list(time_series))

    # Fill gaps manually
    filled = []
    for i in range(len(data)):
        minute, count = data[i]
        if i == 0:
            filled.append((minute, count))
        else:
            prev_minute = data[i - 1][0]
            gap = minute - prev_minute - 1
            gap_idx = 0
            while gap_idx < gap:
                filled.append((prev_minute + 1 + gap_idx, 0))
                gap_idx += 1
            filled.append((minute, count))

    # Detect spikes
    flagged = []
    for i in range(1, len(filled)):
        curr_minute, curr_count = filled[i]
        _, prev_count           = filled[i - 1]

        if prev_count == 0:
            # Can't compute a meaningful multiplier.
            # Flag if curr_count > 0 as a spike from silence.
            if curr_count > 0:
                flagged.append((curr_minute, prev_count, curr_count, float("inf")))
            continue

        multiplier = curr_count / prev_count
        if multiplier > SPIKE_THRESHOLD:
            # Round to 2 decimal places manually
            rounded = int(multiplier * 100) / 100
            flagged.append((curr_minute, prev_count, curr_count, rounded))

    return flagged


# ─────────────────────────────────────────────────────────────────────────────
# APPROACH 2 — WITH BUILT-IN HELPERS (sorted, enumerate, round)
# ─────────────────────────────────────────────────────────────────────────────

def detect_with_builtins(time_series):
    if not time_series:
        return []

    for item in time_series:
        if not (isinstance(item, (list, tuple)) and len(item) == 2):
            raise ValueError(f"Expected (minute, count) tuple, got: {item}")
        if item[1] < 0:
            raise ValueError(f"Event count cannot be negative: {item}")

    data   = sorted(time_series, key=lambda x: x[0])
    filled = fill_gaps(data)

    flagged = []
    for i, (curr_minute, curr_count) in enumerate(filled):
        if i == 0:
            continue
        _, prev_count = filled[i - 1]

        if prev_count == 0:
            if curr_count > 0:
                flagged.append((curr_minute, prev_count, curr_count, float("inf")))
            continue

        multiplier = curr_count / prev_count
        if multiplier > SPIKE_THRESHOLD:
            flagged.append((curr_minute, prev_count, curr_count, round(multiplier, 2)))

    return flagged


# ─────────────────────────────────────────────────────────────────────────────
# REPORT
# ─────────────────────────────────────────────────────────────────────────────

def print_report(flagged, label=""):
    print(f"\n{'='*65}")
    print(f"  RATE-OF-CHANGE ANOMALY REPORT  {label}")
    print(f"{'='*65}")
    print(f"Spike threshold: > {SPIKE_THRESHOLD}x increase\n")

    if not flagged:
        print("  No anomalies detected.\n")
        return

    print(f"{'Minute':<8} {'Prev Count':<12} {'Curr Count':<12} {'Multiplier'}")
    print(f"{'-'*7}  {'-'*11}  {'-'*11}  {'-'*10}")
    for minute, prev, curr, mult in flagged:
        mult_str = "∞ (zero→nonzero)" if mult == float("inf") else f"{mult:.2f}x"
        print(f"{minute:<8} {prev:<12} {curr:<12} {mult_str}")
    print()


# ─────────────────────────────────────────────────────────────────────────────
# MAIN
# ─────────────────────────────────────────────────────────────────────────────

def main():
    """
    Demonstrates with built-in sample data.
    In production: load (minute, count) data from a metrics store or log aggregator.
    """
    print("[INFO] Using built-in sample data.\n")

    try:
        flagged_manual  = detect_without_builtins(SAMPLE_DATA)
        flagged_builtin = detect_with_builtins(SAMPLE_DATA)
    except ValueError as exc:
        print(f"[ERROR] Invalid input: {exc}", file=sys.stderr)
        sys.exit(1)

    print_report(flagged_manual,  label="(Manual / No Builtins)")
    print_report(flagged_builtin, label="(With Builtins)")

    # Sanity check (inf values can't be compared with ==, so compare carefully)
    assert len(flagged_manual) == len(flagged_builtin), \
        f"Count mismatch: {len(flagged_manual)} vs {len(flagged_builtin)}"
    for m, b in zip(flagged_manual, flagged_builtin):
        assert m[0] == b[0], f"Minute mismatch: {m[0]} vs {b[0]}"
    print("[OK] Both approaches agree.\n")

    # Edge case tests
    print("--- Edge Case Tests ---")
    print("Empty input:      ", detect_with_builtins([]))
    print("Single point:     ", detect_with_builtins([(1, 100)]))
    print("All zeros:        ", detect_with_builtins([(1,0),(2,0),(3,0)]))
    print("Zero then spike:  ", detect_with_builtins([(1,0),(2,50)]))
    print("No spike:         ", detect_with_builtins([(1,10),(2,11),(3,12)]))
    print()


if __name__ == "__main__":
    main()

# COMPLEXITY
# =============================================================================
# Sort    : O(N log N)
# Gap fill: O(M) where M = total minutes span (can be >> N if sparse)
# Scan    : O(M)
# Overall : O(N log N + M)
#
# NOTE: If M >> N (e.g., N=1000 events spanning 100000 minutes), gap filling
# is expensive. Use a sparse approach: only compare consecutive actual data
# points and compute the effective rate across the gap.
#
# GOOGLE FOLLOW-UP QUESTIONS:
# - "What if you need to detect spikes in a STREAMING feed, not a batch?"
#     → Keep only the last event count in state. No need for the full list.
#       Emit an alert immediately when threshold is crossed.
# - "What if the data has noise — one-off spikes that aren't real DDoS?"
#     → Use a rolling average instead of previous minute:
#       flag if curr > threshold * avg(last_N_minutes).
#       This makes it more robust to single-minute noise.
# - "How would you set the threshold dynamically?"
#     → Use statistical methods: flag if curr > mean + k * stddev of a
#       rolling window. Adjust k based on desired false-positive rate.
# - "Can you detect a sustained attack that ramps up gradually (no single spike)?"
#     → Rate-of-change alone won't catch it. Combine with an absolute
#       threshold and a trend detector (e.g., 10 consecutive increases).
