# Q1 (Easy) - First Repeated Privileged Command
# Given a list of shell commands run by a user, return the first command that
# appears more than once. Return None if all commands are unique.
#
# ─── SOLUTION WALKTHROUGH ───────────────────────────────────────────────────
#
# STEP 1 — Restate
#   "I have a list of shell commands in the order they were executed. I need to
#    return the first command that I've seen BEFORE. Not the most frequent —
#    the first one that is a repeat."
#
# STEP 2 — Assumptions
#   - Case-sensitive: "ls" ≠ "LS"
#   - Order matters: return the command at the FIRST point of recurrence
#   - Input may be empty or have a single command → return None
#   - Commands are non-empty strings
#   - No null/None entries in list (validate at boundary)
#
# STEP 3 — Pseudocode
#   seen = empty hash set
#   for each command in commands:
#       if command in seen → return command   ← first recurrence
#       add command to seen
#   return None
#
# STEP 4 — Why hash set?
#   Hash set lookup and insert are O(1) average. One pass → O(n) total.
#
# STEP 5 — Naive first
#   "The brute force is two nested loops: for each command, scan everything
#    before it. O(n²) time. Not acceptable for large inputs."
#
# STEP 6 — Follow-up: 100M entries
#   "With 100M entries the set itself could be large (memory concern).
#    Options:
#    a) Stream with external hash map (Redis) — O(1) per lookup, distributed.
#    b) Bloom filter: probabilistic, O(1) space per entry. Small false-positive
#       rate, zero false negatives. Good for dedup screening.
#    c) Partition the stream by hash(command) % P into P parallel workers.
#    For exact answer: we still need the full set in some form.
#    For approximate dedup screening: Bloom filter is ideal."
#
# STEP 7 — Security framing
#   "An adversary could send millions of unique-looking commands to force
#    memory growth (hash flooding). Python uses PYTHONHASHSEED randomisation
#    for string hashes, mitigating predictable collisions. In prod, set an
#    upper bound on the number of unique commands tracked."
#
# Complexity:
#   Naive:     O(n²) time, O(1) space
#   Hash set:  O(n) time,  O(n) space  ← target
# ─────────────────────────────────────────────────────────────────────────────


# ── APPROACH 1: Brute force O(n²) — state first ─────────────────

def first_repeated_naive(commands: list) -> str | None:
    """
    For each command, scan all previous commands.

    Time:  O(n²)
    Space: O(1)
    """
    if not isinstance(commands, list):
        raise TypeError(f"Expected list, got {type(commands).__name__}")

    for i in range(1, len(commands)):
        for j in range(i):
            if commands[j] == commands[i]:
                return commands[i]
    return None


# ── APPROACH 2: Hash set O(n) ─────────────────────────────────────────────────

def first_repeated(commands: list) -> str | None:
    """
    Single pass with a seen set.

    Time:  O(n) average
    Space: O(n) worst case (all unique)

    Returns the first command that is a repeat, or None.
    """
    if not isinstance(commands, list):
        raise TypeError(f"Expected list, got {type(commands).__name__}")

    seen = set()
    for cmd in commands:
        if not isinstance(cmd, str):
            raise ValueError(f"Command must be a string, got {type(cmd).__name__}: {cmd!r}")
        if cmd in seen:
            return cmd
        seen.add(cmd)

    return None


# ── APPROACH 3: Using dict to also track first seen index ─────────────────────

def first_repeated_with_index(commands: list) -> tuple | None:
    """
    Returns (command, first_seen_index, repeated_at_index) for the first repeat.
    Useful for audit logging — you know when the command was first issued.

    Time:  O(n)
    Space: O(n)
    """
    if not isinstance(commands, list):
        raise TypeError(f"Expected list, got {type(commands).__name__}")

    first_seen = {}
    for i, cmd in enumerate(commands):
        if cmd in first_seen:
            return (cmd, first_seen[cmd], i)
        first_seen[cmd] = i

    return None


# ── MANUAL — no set, no dict built-ins (manual open-addressing hash table) ───

class _SimpleSet:
    """Minimal open-addressing hash set (linear probing)."""

    EMPTY    = object()
    DELETED  = object()

    def __init__(self, initial_capacity: int = 16):
        self._cap   = initial_capacity
        self._table = [self.EMPTY] * self._cap
        self._size  = 0

    def _hash(self, key: str) -> int:
        h = 5381
        for ch in key:
            h = ((h << 5) + h) ^ ord(ch)
        return h % self._cap

    def _resize(self):
        old_table = self._table
        self._cap  *= 2
        self._table = [self.EMPTY] * self._cap
        self._size  = 0
        for item in old_table:
            if item is not self.EMPTY and item is not self.DELETED:
                self.add(item)

    def add(self, key: str) -> None:
        if self._size / self._cap > 0.7:
            self._resize()
        idx = self._hash(key)
        while self._table[idx] is not self.EMPTY and self._table[idx] is not self.DELETED:
            if self._table[idx] == key:
                return  # already present
            idx = (idx + 1) % self._cap
        self._table[idx] = key
        self._size += 1

    def __contains__(self, key: str) -> bool:
        idx = self._hash(key)
        while self._table[idx] is not self.EMPTY:
            if self._table[idx] == key:
                return True
            idx = (idx + 1) % self._cap
        return False


def first_repeated_manual(commands: list) -> str | None:
    """
    Same hash-set logic using a hand-rolled open-addressing table.

    Time:  O(n) amortized
    Space: O(n)
    """
    if commands is None:
        raise ValueError("commands cannot be None")

    seen = _SimpleSet()
    for cmd in commands:
        if cmd in seen:
            return cmd
        seen.add(cmd)
    return None


# ── Tests ─────────────────────────────────────────────────────────────────────

if __name__ == "__main__":
    # Basic examples from question
    cmds1 = ["ls", "sudo su", "whoami", "sudo su", "id"]
    assert first_repeated(cmds1) == "sudo su"

    cmds2 = ["ls", "cat", "pwd"]
    assert first_repeated(cmds2) is None

    # First repeat is NOT the most frequent — order matters
    cmds3 = ["ls", "ls", "whoami", "whoami"]
    assert first_repeated(cmds3) == "ls"    # "ls" repeats first

    # Case-sensitive
    cmds4 = ["Sudo su", "sudo su"]
    assert first_repeated(cmds4) is None    # different strings

    # All same
    cmds5 = ["id", "id", "id"]
    assert first_repeated(cmds5) == "id"

    # Empty and single
    assert first_repeated([])    is None
    assert first_repeated(["ls"]) is None

    # All approaches agree
    for tc in [cmds1, cmds2, cmds3, cmds4, cmds5, [], ["x"]]:
        r1 = first_repeated_naive(tc)
        r2 = first_repeated(tc)
        r3 = first_repeated_manual(tc)
        assert r1 == r2 == r3, f"Mismatch on {tc}: {r1}, {r2}, {r3}"

    # Index variant
    result = first_repeated_with_index(cmds1)
    assert result == ("sudo su", 1, 3)   # "sudo su" first at index 1, repeated at 3

    print("Q1 — All assertions passed.")
    print(f"first_repeated({cmds1}) = {first_repeated(cmds1)!r}")
    print()
    print("Complexity: O(n) time, O(n) space")
    print("Follow-up (100M entries): Bloom filter for approximate dedup screening,")
    print("  or partition stream by hash(cmd) across parallel workers for exact answer.")
