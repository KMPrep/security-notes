# Q3 (Medium) - ACL Rule Conflict Detector
# Given firewall ACL rules evaluated in priority order (lowest number = highest
# priority), find all pairs that conflict (same src/dst/port, different action).
# For each conflict, report which rule wins.
#
# ─── SOLUTION WALKTHROUGH ───────────────────────────────────────────────────
#
# STEP 1 — Restate
#   "I have a list of firewall rules. Two rules conflict when they match the
#    same traffic (same src, dst, port) but have different actions (ALLOW vs DENY).
#    I need to find ALL such conflicting pairs and report which rule wins.
#    The winning rule is the one with the LOWER priority number (evaluated first)."
#
# STEP 2 — Assumptions
#   - Priority numbers are unique (if tied, define a tiebreak — here assume unique)
#   - 'src', 'dst', 'port' are the match keys. 'action' is either "ALLOW" or "DENY"
#   - A rule can conflict with more than one other rule
#   - We want ALL conflicting pairs, not just the first
#   - Input may be empty → return []
#
# STEP 3 — Pseudocode (O(n) grouping approach)
#   group = dict keyed by (src, dst, port)
#   for each rule:
#       append to group[(rule.src, rule.dst, rule.port)]
#   conflicts = []
#   for key, group_rules in group.items():
#       if any two rules in group have different actions:
#           for each pair (r1, r2) with r1.action != r2.action:
#               winner = min(r1, r2, key=priority)
#               conflicts.append((r1, r2, winner))
#   return conflicts
#
# STEP 4 — Why O(n) not O(n²)
#   "The dict groups rules by key in one pass O(n). Within each group,
#    comparing all pairs is O(g²) where g = group size. In practice,
#    g is small (few rules share same src/dst/port). Total = O(n + sum(g²)).
#    For small groups: effectively O(n)."
#
# STEP 5 — Naive approach
#   "Compare every pair of rules: O(n²). Fine for small rulesets."
#
# STEP 6 — Security framing
#   "ACL conflicts are a real security risk — a DENY rule might be shadowed by
#    an ALLOW rule at higher priority. Tools like Batfish and Tufin detect these.
#    In practice we want to surface: (a) shadow rules (a rule can NEVER match
#    because a higher-priority rule always matches first), and (b) redundant rules."
#
# Complexity:
#   Naive:    O(n²) time, O(1) extra space
#   Grouped:  O(n) time,  O(n) space  ← target
# ─────────────────────────────────────────────────────────────────────────────

from itertools import combinations


# ── APPROACH 1: Brute force O(n²) ────────────────────────────────────────────

def find_conflicts_naive(rules: list) -> list:
    """
    Compare every pair of rules.

    Time:  O(n²)
    Space: O(c) where c = number of conflicts
    """
    if not isinstance(rules, list):
        raise TypeError(f"Expected list, got {type(rules).__name__}")
    if not rules:
        return []

    def _validate_rule(r, idx):
        required = {"priority", "src", "dst", "port", "action"}
        missing = required - set(r.keys())
        if missing:
            raise ValueError(f"Rule at index {idx} missing fields: {missing}")
        if r["action"] not in ("ALLOW", "DENY"):
            raise ValueError(f"Invalid action {r['action']!r} at index {idx}")

    for i, r in enumerate(rules):
        _validate_rule(r, i)

    conflicts = []
    n = len(rules)
    for i in range(n):
        for j in range(i + 1, n):
            r1, r2 = rules[i], rules[j]
            same_match = (r1["src"] == r2["src"] and
                          r1["dst"] == r2["dst"] and
                          r1["port"] == r2["port"])
            if same_match and r1["action"] != r2["action"]:
                winner = r1 if r1["priority"] < r2["priority"] else r2
                loser  = r2 if winner is r1 else r1
                conflicts.append({
                    "rule_a":  r1,
                    "rule_b":  r2,
                    "winner":  winner,
                    "loser":   loser,
                    "key":     (r1["src"], r1["dst"], r1["port"]),
                })
    return conflicts


# ── APPROACH 2: Dict grouping O(n) ───────────────────────────────────────────

def find_conflicts(rules: list) -> list:
    """
    Group rules by (src, dst, port). Within each group, find pairs with
    different actions.

    Time:  O(n + sum of g² for each group g) ≈ O(n) for small groups
    Space: O(n)
    """
    if not isinstance(rules, list):
        raise TypeError(f"Expected list, got {type(rules).__name__}")
    if not rules:
        return []

    # Group by match key
    groups: dict = {}
    for i, rule in enumerate(rules):
        if not isinstance(rule, dict):
            raise ValueError(f"Rule at index {i} must be a dict, got {type(rule).__name__}")
        for field in ("priority", "src", "dst", "port", "action"):
            if field not in rule:
                raise ValueError(f"Rule at index {i} missing required field: {field!r}")
        if rule["action"] not in ("ALLOW", "DENY"):
            raise ValueError(f"Invalid action {rule['action']!r} at index {i}")

        key = (rule["src"], rule["dst"], rule["port"])
        if key not in groups:
            groups[key] = []
        groups[key].append(rule)

    # Find conflicting pairs within each group
    conflicts = []
    for key, group_rules in groups.items():
        # Use combinations to get all pairs
        for r1, r2 in combinations(group_rules, 2):
            if r1["action"] != r2["action"]:
                winner = r1 if r1["priority"] < r2["priority"] else r2
                loser  = r2 if winner is r1 else r1
                conflicts.append({
                    "rule_a":  r1,
                    "rule_b":  r2,
                    "winner":  winner,
                    "loser":   loser,
                    "key":     key,
                })

    # Sort by winning rule's priority for deterministic output
    conflicts.sort(key=lambda c: c["winner"]["priority"])
    return conflicts


def format_conflict(c: dict) -> str:
    """Human-readable conflict description."""
    w = c["winner"]
    l = c["loser"]
    key = c["key"]
    return (
        f"Conflict on (src={key[0]}, dst={key[1]}, port={key[2]}): "
        f"priority={w['priority']} {w['action']} wins over "
        f"priority={l['priority']} {l['action']}"
    )


# ── MANUAL — no itertools, no dict.items() sugar ─────────────────────────────

def find_conflicts_manual(rules: list) -> list:
    """
    Same grouping logic without itertools or built-in helpers.

    Time:  O(n + sum g²)
    Space: O(n)
    """
    if rules is None:
        raise ValueError("rules cannot be None")
    n = len(rules)
    if n == 0:
        return []

    # Manual grouping
    keys    = []
    buckets = []

    for rule in rules:
        key = (rule["src"], rule["dst"], rule["port"])
        found = False
        for idx in range(len(keys)):
            if keys[idx] == key:
                buckets[idx].append(rule)
                found = True
                break
        if not found:
            keys.append(key)
            buckets.append([rule])

    conflicts = []
    for idx in range(len(keys)):
        key   = keys[idx]
        group = buckets[idx]
        gn    = len(group)
        for i in range(gn):
            for j in range(i + 1, gn):
                r1, r2 = group[i], group[j]
                if r1["action"] != r2["action"]:
                    if r1["priority"] < r2["priority"]:
                        winner, loser = r1, r2
                    else:
                        winner, loser = r2, r1
                    conflicts.append({
                        "rule_a": r1,
                        "rule_b": r2,
                        "winner": winner,
                        "loser":  loser,
                        "key":    key,
                    })

    return conflicts


# ── Tests ─────────────────────────────────────────────────────────────────────

if __name__ == "__main__":
    rules = [
        {"priority": 10, "src": "10.0.0.0/8",     "dst": "any", "port": 443, "action": "ALLOW"},
        {"priority": 20, "src": "10.0.0.0/8",     "dst": "any", "port": 443, "action": "DENY"},
        {"priority": 5,  "src": "192.168.1.0/24", "dst": "any", "port": 80,  "action": "DENY"},
        {"priority": 30, "src": "192.168.1.0/24", "dst": "any", "port": 80,  "action": "ALLOW"},
    ]

    conflicts = find_conflicts(rules)
    assert len(conflicts) == 2, f"Expected 2 conflicts, got {len(conflicts)}"

    # Conflict 1: port 80  → priority 5 DENY wins over priority 30 ALLOW
    c80 = next(c for c in conflicts if c["key"][2] == 80)
    assert c80["winner"]["priority"] == 5
    assert c80["winner"]["action"]   == "DENY"

    # Conflict 2: port 443 → priority 10 ALLOW wins over priority 20 DENY
    c443 = next(c for c in conflicts if c["key"][2] == 443)
    assert c443["winner"]["priority"] == 10
    assert c443["winner"]["action"]   == "ALLOW"

    # Print formatted output
    print("Conflicts found:")
    for c in conflicts:
        print(f"  {format_conflict(c)}")
    print()

    # No conflicts
    no_conflict_rules = [
        {"priority": 1, "src": "10.0.0.1", "dst": "any", "port": 80,  "action": "ALLOW"},
        {"priority": 2, "src": "10.0.0.1", "dst": "any", "port": 443, "action": "ALLOW"},
    ]
    assert find_conflicts(no_conflict_rules) == []

    # Same action — not a conflict
    same_action = [
        {"priority": 1, "src": "10.0.0.1", "dst": "any", "port": 80, "action": "DENY"},
        {"priority": 2, "src": "10.0.0.1", "dst": "any", "port": 80, "action": "DENY"},
    ]
    assert find_conflicts(same_action) == []

    # Empty
    assert find_conflicts([]) == []

    # All approaches agree
    for tc in [rules, no_conflict_rules, same_action, []]:
        n = find_conflicts_naive(tc)
        g = find_conflicts(tc)
        m = find_conflicts_manual(tc)
        assert len(n) == len(g) == len(m), \
            f"Count mismatch on {tc}: {len(n)}, {len(g)}, {len(m)}"

    print("Q3 — All assertions passed.")
    print()
    print("Complexity:")
    print("  Naive:    O(n²) time, O(c) space")
    print("  Grouped:  O(n) time, O(n) space  (O(n + g²) per group, small g in practice)")
    print()
    print("Security note: ACL conflicts risk ALLOW overriding intended DENY.")
    print("  Always audit: which is the effective rule for a given flow?")
    print("  Shadow rule: rule that can NEVER match (always blocked by higher-priority rule)")
