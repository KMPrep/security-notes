# Q4 (Hard) - Threat Event Correlation Engine
# Two sorted event lists from different sensors. A correlated threat occurs when
# the same source_ip appears in both sensors within time window T. Return all
# correlated pairs with their time delta.
#
# ─── SOLUTION WALKTHROUGH ───────────────────────────────────────────────────
#
# STEP 1 — Restate
#   "I have two sorted lists of security events (timestamp, event_type, source_ip).
#    For every event in sensor_a, I want to find all events in sensor_b from the
#    same IP where |ts_a - ts_b| <= T. Return all correlated pairs sorted by
#    sensor_a timestamp ascending."
#
# STEP 2 — Assumptions
#   - Both lists are sorted ascending by timestamp
#   - The same IP can appear multiple times in both lists
#   - If sensor_a event correlates with MULTIPLE sensor_b events, report ALL pairs
#   - T >= 0 (T=0 means exact timestamp match only)
#   - Tuples are (timestamp: int, event_type: str, source_ip: str)
#
# STEP 3 — Naive approach
#   "For each event in sensor_a, scan ALL events in sensor_b with the same IP.
#    O(n * m) where n=|sensor_a|, m=|sensor_b|. For 1M events each: 10^12 ops."
#
# STEP 4 — Better approach: group by IP, then two-pointer per IP
#   "Step 1: Group sensor_b events by source_ip → dict: {ip: [events sorted by ts]}
#    Step 2: For each sensor_a event (ts_a, type_a, ip):
#              look up sensor_b events for that IP
#              use binary search to find the window [ts_a-T, ts_a+T]
#              collect all sensor_b events in that window
#    Step 3: Append to results, sort by sensor_a timestamp.
#
#    Time:  O(n log n + m log m) for grouping and binary search per lookup
#           = O((n+m) log(n+m)) overall
#    Space: O(m) for the grouped sensor_b dict"
#
# STEP 5 — Why not just two-pointer over merged stream?
#   "Two-pointer over the merged stream works ONLY if events are interleaved and
#    you process one IP at a time. Grouping by IP first is cleaner and handles
#    the 'same IP' requirement explicitly."
#
# STEP 6 — Follow-up: 1 billion events per day
#   "1B events/day ≈ 11,500 events/second.
#    Approach:
#    a) Partition by source_ip using consistent hashing across workers.
#       Each worker handles a subset of IPs independently — fully parallel.
#    b) Use Apache Kafka: partition logs by source_ip key → events for the same
#       IP always go to the same partition → natural parallelism.
#    c) Within each partition, maintain a sliding window per IP using a deque.
#       Push events; evict events older than T; on each push, check cross-sensor.
#    d) State store (Redis/RocksDB) for IP → recent events, with TTL = T.
#    e) Apache Flink or Spark Streaming for stateful windowed joins.
#    Key: never hold ALL events in memory; process in windows."
#
# STEP 7 — Security framing
#   "This is a multi-stage attack detection pattern: port_scan on sensor_a
#    followed by exec_shell on sensor_b from the same IP within T seconds = kill chain.
#    False positives: CDN IPs, NAT — correlate on (ip, port, user) not just ip.
#    False negatives: IP rotation/spoofing — use session tokens or user-agent."
#
# Complexity:
#   Naive:          O(n*m)
#   Group + bisect: O((n+m) log m)  ← target
# ─────────────────────────────────────────────────────────────────────────────

import bisect
from collections import defaultdict


# ── APPROACH 1: Brute force O(n*m) ───────────────────────────────────────────

def correlate_naive(sensor_a: list, sensor_b: list, T: int) -> list:
    """
    For each event in sensor_a, scan all sensor_b events for same IP within T.

    Time:  O(n*m)
    Space: O(c) where c = number of correlated pairs
    """
    if not isinstance(sensor_a, list) or not isinstance(sensor_b, list):
        raise TypeError("sensor_a and sensor_b must be lists")
    if T < 0:
        raise ValueError(f"T must be >= 0, got {T}")
    if not sensor_a or not sensor_b:
        return []

    results = []
    for ts_a, type_a, ip_a in sensor_a:
        for ts_b, type_b, ip_b in sensor_b:
            if ip_a == ip_b and abs(ts_a - ts_b) <= T:
                results.append({
                    "sensor_a_event": (ts_a, type_a, ip_a),
                    "sensor_b_event": (ts_b, type_b, ip_b),
                    "ip":             ip_a,
                    "delta_seconds":  abs(ts_a - ts_b),
                })
    results.sort(key=lambda r: r["sensor_a_event"][0])
    return results


# ── APPROACH 2: Group by IP + binary search O((n+m) log m) ───────────────────

def correlate(sensor_a: list, sensor_b: list, T: int) -> list:
    """
    1. Group sensor_b events by source_ip: {ip: [(ts, type), ...]}  O(m)
    2. For each sensor_a event, binary search sensor_b[ip] for window [ts-T, ts+T]
       and collect all events in that window.  O(n log m) per event
    3. Sort output by sensor_a timestamp.  O(c log c)

    Total: O(m + n log m + c log c)

    Args:
        sensor_a: list of (timestamp, event_type, source_ip) — sorted ascending
        sensor_b: list of (timestamp, event_type, source_ip) — sorted ascending
        T:        correlation window in seconds

    Returns:
        list of correlated pair dicts, sorted by sensor_a timestamp
    """
    if not isinstance(sensor_a, list) or not isinstance(sensor_b, list):
        raise TypeError("sensor_a and sensor_b must be lists")
    if T < 0:
        raise ValueError(f"T must be >= 0, got {T}")
    if not sensor_a or not sensor_b:
        return []

    # Step 1: Group sensor_b by IP — store (ts, event_type) per IP, sorted by ts
    b_by_ip: dict = defaultdict(list)
    for ts_b, type_b, ip_b in sensor_b:
        b_by_ip[ip_b].append((ts_b, type_b))
    # Ensure each group is sorted (sensor_b is globally sorted, but per-IP may not be)
    for ip in b_by_ip:
        b_by_ip[ip].sort()

    results = []

    # Step 2: For each sensor_a event, binary search in sensor_b[ip]
    for ts_a, type_a, ip_a in sensor_a:
        if ip_a not in b_by_ip:
            continue

        b_events = b_by_ip[ip_a]       # sorted list of (ts_b, type_b)
        b_ts     = [e[0] for e in b_events]   # timestamps only, for bisect

        # Find range [ts_a - T, ts_a + T] in b_ts
        lo_idx = bisect.bisect_left(b_ts,  ts_a - T)
        hi_idx = bisect.bisect_right(b_ts, ts_a + T)

        for idx in range(lo_idx, hi_idx):
            ts_b, type_b = b_events[idx]
            results.append({
                "sensor_a_event": (ts_a, type_a, ip_a),
                "sensor_b_event": (ts_b, type_b, ip_a),
                "ip":             ip_a,
                "delta_seconds":  abs(ts_a - ts_b),
            })

    # Step 3: Sort output by sensor_a timestamp
    results.sort(key=lambda r: r["sensor_a_event"][0])
    return results


# ── APPROACH 3: Two-pointer per IP (avoids extracting ts list) ────────────────

def correlate_two_pointer(sensor_a: list, sensor_b: list, T: int) -> list:
    """
    For each IP, use two pointers over the per-IP events from each sensor.
    The outer pointer moves through sensor_a events for that IP;
    the inner window tracks the valid sensor_b range.

    Time:  O(m + n + c) amortized for each IP; O(n + m + c) total
    Space: O(n + m) for grouping
    """
    if not isinstance(sensor_a, list) or not isinstance(sensor_b, list):
        raise TypeError("sensor_a and sensor_b must be lists")
    if T < 0:
        raise ValueError(f"T must be >= 0, got {T}")

    # Group both sensors by IP
    a_by_ip: dict = defaultdict(list)
    b_by_ip: dict = defaultdict(list)
    for ts, etype, ip in sensor_a:
        a_by_ip[ip].append((ts, etype))
    for ts, etype, ip in sensor_b:
        b_by_ip[ip].append((ts, etype))

    results = []

    for ip in a_by_ip:
        if ip not in b_by_ip:
            continue

        a_events = sorted(a_by_ip[ip])
        b_events = sorted(b_by_ip[ip])
        mb = len(b_events)

        left = 0    # earliest b_event that could still match

        for ts_a, type_a in a_events:
            # Advance left past events that are too old (ts_b < ts_a - T)
            while left < mb and b_events[left][0] < ts_a - T:
                left += 1
            # Collect all b_events in [ts_a-T, ts_a+T]
            idx = left
            while idx < mb and b_events[idx][0] <= ts_a + T:
                ts_b, type_b = b_events[idx]
                results.append({
                    "sensor_a_event": (ts_a, type_a, ip),
                    "sensor_b_event": (ts_b, type_b, ip),
                    "ip":             ip,
                    "delta_seconds":  abs(ts_a - ts_b),
                })
                idx += 1

    results.sort(key=lambda r: r["sensor_a_event"][0])
    return results


# ── MANUAL — no defaultdict, no bisect ───────────────────────────────────────

def _manual_bisect_left(arr, val):
    lo, hi = 0, len(arr)
    while lo < hi:
        mid = lo + (hi - lo) // 2
        if arr[mid] < val:
            lo = mid + 1
        else:
            hi = mid
    return lo

def _manual_bisect_right(arr, val):
    lo, hi = 0, len(arr)
    while lo < hi:
        mid = lo + (hi - lo) // 2
        if arr[mid] <= val:
            lo = mid + 1
        else:
            hi = mid
    return lo

def correlate_manual(sensor_a: list, sensor_b: list, T: int) -> list:
    """
    Group + binary search without defaultdict or bisect module.

    Time:  O(m + n log m + c log c)
    Space: O(n + m)
    """
    if not sensor_a or not sensor_b:
        return []

    # Manual grouping dict
    b_groups = {}
    for ts_b, type_b, ip_b in sensor_b:
        if ip_b not in b_groups:
            b_groups[ip_b] = []
        b_groups[ip_b].append((ts_b, type_b))

    # Sort each group
    for ip in b_groups:
        b_groups[ip].sort()

    results = []
    for ts_a, type_a, ip_a in sensor_a:
        if ip_a not in b_groups:
            continue
        b_events = b_groups[ip_a]
        b_ts = [e[0] for e in b_events]
        lo = _manual_bisect_left(b_ts, ts_a - T)
        hi = _manual_bisect_right(b_ts, ts_a + T)
        for i in range(lo, hi):
            ts_b, type_b = b_events[i]
            results.append({
                "sensor_a_event": (ts_a, type_a, ip_a),
                "sensor_b_event": (ts_b, type_b, ip_a),
                "ip":             ip_a,
                "delta_seconds":  abs(ts_a - ts_b),
            })

    results.sort(key=lambda r: r["sensor_a_event"][0])
    return results

        (210, "exec_shell",  "2.2.2.2"),
        (320, "http_req",    "3.3.3.3"),
        (395, "dns_query",   "1.1.1.1"),
        (500, "exec_shell",  "1.1.1.1"),

    T = 30

    results = correlate(sensor_a, sensor_b, T)

    # Manually verify:
    # (100, 1.1.1.1) ↔ (110, 1.1.1.1): delta=10 ✓ (T=30)
    # (200, 2.2.2.2) ↔ (210, 2.2.2.2): delta=10 ✓
    # (300, 3.3.3.3) ↔ (320, 3.3.3.3): delta=20 ✓
    # (400, 1.1.1.1) ↔ (395, 1.1.1.1): delta=5  ✓
    # (400, 1.1.1.1) ↔ (500, 1.1.1.1): delta=100 ✗ (T=30)
    assert len(results) == 4, f"Expected 4 correlations, got {len(results)}"

    # Sorted by sensor_a timestamp
    a_ts = [r["sensor_a_event"][0] for r in results]
    assert a_ts == sorted(a_ts), "Results must be sorted by sensor_a timestamp"

    # Verify IPs
    for r in results:
        assert r["sensor_a_event"][2] == r["sensor_b_event"][2], "IPs must match"
        assert r["delta_seconds"] <= T

    # All approaches agree
    r1 = correlate_naive(sensor_a, sensor_b, T)
    r2 = correlate(sensor_a, sensor_b, T)
    r3 = correlate_two_pointer(sensor_a, sensor_b, T)
    r4 = correlate_manual(sensor_a, sensor_b, T)
    assert len(r1) == len(r2) == len(r3) == len(r4), \
        f"Mismatch: {len(r1)}, {len(r2)}, {len(r3)}, {len(r4)}"

    assert correlate([], sensor_b, T) == []
    assert correlate(sensor_a, [], T) == []
    assert correlate([], [], T)       == []
