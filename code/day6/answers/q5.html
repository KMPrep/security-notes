# Q5 (Hard) - Minimal Covering Policy (Set Cover)
# Given required permissions and a set of roles (each granting a subset of
# permissions), find the minimum number of roles that together cover ALL
# required permissions.
#
# ─── SOLUTION WALKTHROUGH ───────────────────────────────────────────────────
#
# STEP 1 — Restate
#   "I have a set of permissions a user needs. I have a catalog of roles, each
#    granting a subset of permissions. I want to find the MINIMUM NUMBER of roles
#    that together cover every required permission. If coverage is impossible,
#    say so. This is the classic Set Cover problem."
#
# STEP 2 — Assumptions
#   - Permissions and role names are strings
#   - A role's permissions may exceed required (extra perms are fine)
#   - 'required' is a set — no duplicates
#   - Roles dict values are sets of permissions
#   - If no subset of roles can cover required → return None / raise
#   - If required is empty → return [] (no roles needed)
#
# STEP 3 — Critical key insight: NP-hard
#   "SET COVER IS NP-HARD. The exact optimal solution requires checking ALL 2^n
#    subsets of roles — exponential. Mention this explicitly to the reviewer:
#
#    'This is Set Cover, which is NP-hard. For small M (number of roles),
#     exact brute-force is feasible. For large M, the greedy algorithm gives a
#     (1 + ln(|required|))-approximation guarantee — it's within a log factor
#     of optimal and runs in polynomial time.'"
#
# STEP 4 — Greedy approach
#   "At each step, pick the role that covers the MOST uncovered permissions.
#    Repeat until all required permissions are covered.
#
#    Pseudocode:
#      uncovered = set(required)
#      chosen = []
#      while uncovered:
#          best_role = role with max |role_perms ∩ uncovered|
#          if max coverage == 0: return None  ← impossible
#          chosen.append(best_role)
#          uncovered -= role_perms[best_role]
#      return chosen"
#
# STEP 5 — Is greedy always optimal?
#   "NO. Greedy gives (1 + ln k) approximation where k = |required|.
#    Example where greedy fails: required = {a,b,c,d}, roles:
#      R1 = {a,b},  R2 = {c,d},  R3 = {a,b,c,d}
#    Greedy might pick R1 first (covers 2), then R2 (covers 2). Answer: 2 roles.
#    Optimal is also 2 (or 1 if R3 is available). In adversarial cases, greedy
#    takes O(log n) times the optimal number of sets."
#
# STEP 6 — Exact brute force (for small M)
#   "Check all 2^M subsets. Pick smallest covering subset.
#    Time: O(2^M * M * P) where M = #roles, P = #permissions.
#    Only feasible for M <= 20 or so."
#
# STEP 7 — Security framing
#   "Least-privilege principle: we want the MINIMUM roles covering required
#    permissions. Over-permissioned users = attack surface expansion.
#    Privilege escalation: an attacker grants themselves extra roles → monitor
#    delta between required and actual permissions.
#    Role explosion: too many fine-grained roles → governance problem."
#
# Complexity:
#   Brute force: O(2^M * M * P)
#   Greedy:      O(M * P) per step, O(M² * P) total  ← target
# ─────────────────────────────────────────────────────────────────────────────

from itertools import combinations


# ── APPROACH 1: Brute force exact (enumerate all subsets) ────────────────────

def min_roles_exact(required: set, roles: dict) -> list | None:
    """
    Check all subsets of roles from smallest to largest.
    Returns the first (smallest) subset that covers all required permissions.

    Time:  O(2^M * M * P) — only feasible for small M (< 20 roles)
    Space: O(M)
    """
    if not isinstance(required, set):
        required = set(required)
    if not isinstance(roles, dict):
        raise TypeError(f"roles must be a dict, got {type(roles).__name__}")
    if not required:
        return []

    role_names = list(roles.keys())
    M = len(role_names)

    # Try subsets of increasing size
    for size in range(1, M + 1):
        for subset in combinations(role_names, size):
            covered = set()
            for role in subset:
                covered |= roles[role]
            if required.issubset(covered):
                return list(subset)

    return None   # impossible to cover


# ── APPROACH 2: Greedy (polynomial, approximately optimal) ────────────────────

def min_roles_greedy(required: set, roles: dict) -> list | None:
    """
    At each step, pick the role covering the most UNCOVERED permissions.
    Guaranteed to cover if a solution exists (complete coverage test at end).

    Approximation ratio: O(1 + ln(|required|))
    NOT always optimal — but polynomial time and practical.

    Time:  O(M² * P)  — M greedy steps, each scans M roles and P permissions
    Space: O(M + P)
    """
    if not isinstance(required, set):
        required = set(required)
    if not isinstance(roles, dict):
        raise TypeError(f"roles must be a dict, got {type(roles).__name__}")
    if not required:
        return []

    # Validate all role permissions are sets
    for name, perms in roles.items():
        if not isinstance(perms, (set, frozenset)):
            roles = {k: set(v) for k, v in roles.items()}
            break

    uncovered = set(required)
    chosen    = []
    remaining = dict(roles)    # roles not yet chosen

    while uncovered:
        if not remaining:
            # No more roles but still uncovered permissions
            return None

        # Pick role with max intersection with uncovered
        best_role  = None
        best_cover = set()

        for role_name, role_perms in remaining.items():
            covered_by_this = role_perms & uncovered
            if len(covered_by_this) > len(best_cover):
                best_role  = role_name
                best_cover = covered_by_this

        if not best_cover:
            # No role covers any remaining uncovered permission
            return None

        chosen.append(best_role)
        uncovered -= best_cover
        del remaining[best_role]

    return chosen


# ── APPROACH 3: Greedy with coverage explanation ──────────────────────────────

def min_roles_greedy_explained(required: set, roles: dict) -> dict:
    """
    Same greedy, but returns detailed step-by-step coverage breakdown.
    Useful for audit reports: "why was this role assigned?"

    Time:  O(M² * P)
    """
    if not isinstance(required, set):
        required = set(required)
    if not required:
        return {"chosen": [], "coverage_steps": [], "covered": set()}

    uncovered = set(required)
    chosen    = []
    steps     = []
    remaining = {k: set(v) for k, v in roles.items()}

    while uncovered:
        if not remaining:
            return {"chosen": None, "error": "Cannot cover all required permissions",
                    "uncovered": uncovered}

        best_role  = None
        best_cover = set()
        for name, perms in remaining.items():
            c = perms & uncovered
            if len(c) > len(best_cover):
                best_role  = name
                best_cover = c

        if not best_cover:
            return {"chosen": None, "error": "Cannot cover all required permissions",
                    "uncovered": uncovered}

        chosen.append(best_role)
        steps.append({
            "role":           best_role,
            "newly_covered":  best_cover,
            "remaining_after": uncovered - best_cover,
        })
        uncovered -= best_cover
        del remaining[best_role]

    return {"chosen": chosen, "coverage_steps": steps, "covered": set(required)}


# ── MANUAL — no sets/frozensets for operations (manual intersection) ──────────

def _manual_intersection(set_a: set, set_b: set) -> set:
    """Return elements in both set_a and set_b without using & operator."""
    result = set()
    for item in set_a:
        found = False
        for b in set_b:
            if item == b:
                found = True
                break
        if found:
            result.add(item)
    return result


def min_roles_manual(required: set, roles: dict) -> list | None:
    """
    Greedy set cover without set operators (manual intersection).

    Time:  O(M² * P²) — manual intersection is O(P²)
    Space: O(M + P)
    """
    if not required:
        return []

    uncovered = set(required)
    chosen    = []
    remaining = {k: set(v) for k, v in roles.items()}

    while uncovered:
        if not remaining:
            return None

        best_role  = None
        best_count = 0
        best_cover = set()

        for name, perms in remaining.items():
            covered_by_this = _manual_intersection(perms, uncovered)
            if len(covered_by_this) > best_count:
                best_role  = name
                best_count = len(covered_by_this)
                best_cover = covered_by_this

        if not best_cover:
            return None

        chosen.append(best_role)
        for perm in best_cover:
            uncovered.discard(perm)
        del remaining[best_role]

    return chosen


# ── Tests ─────────────────────────────────────────────────────────────────────

if __name__ == "__main__":
    # Example 1: single role covers all
    required1 = {"read", "write", "exec", "admin"}
    roles1 = {
        "viewer":    {"read"},
        "editor":    {"read", "write"},
        "developer": {"read", "write", "exec"},
        "admin":     {"read", "write", "exec", "admin"},
    }
    result1_exact  = min_roles_exact(required1, roles1)
    result1_greedy = min_roles_greedy(required1, roles1)
    assert len(result1_exact)  == 1, f"Expected 1 role, got {result1_exact}"
    assert len(result1_greedy) == 1, f"Expected 1 role, got {result1_greedy}"
    assert result1_exact[0]  == "admin"
    assert result1_greedy[0] == "admin"
    print(f"Example 1: exact={result1_exact}, greedy={result1_greedy}")

    # Example 2: harder input — no single role covers all
    required2 = {"read", "write", "exec", "audit", "deploy"}
    roles2 = {
        "viewer":    {"read"},
        "editor":    {"read", "write"},
        "developer": {"read", "write", "exec"},
        "auditor":   {"audit", "read"},
        "deployer":  {"deploy", "exec"},
    }
    result2_exact  = min_roles_exact(required2, roles2)
    result2_greedy = min_roles_greedy(required2, roles2)
    # Verify coverage
    for result in [result2_exact, result2_greedy]:
        covered = set()
        for role in result:
            covered |= roles2[role]
        assert required2.issubset(covered), \
            f"Result {result} does not cover required={required2}"
    print(f"Example 2: exact={result2_exact}, greedy={result2_greedy}")

    # Example 3: impossible — required has permission not in any role
    required3 = {"read", "root_shell"}
    result3 = min_roles_greedy(required3, roles1)
    assert result3 is None, f"Expected None (impossible), got {result3}"

    # Example 4: empty required → no roles needed
    assert min_roles_greedy(set(), roles1) == []
    assert min_roles_exact(set(),  roles1) == []

    # Example 5: single required permission
    required5 = {"read"}
    result5 = min_roles_greedy(required5, roles1)
    assert len(result5) == 1
    assert "read" in roles1[result5[0]]

    # Manual agrees with greedy on examples 1 and 2
    m1 = min_roles_manual(required1, roles1)
    assert len(m1) == 1

    m2 = min_roles_manual(required2, roles2)
    covered_m2 = set()
    for role in m2:
        covered_m2 |= roles2[role]
    assert required2.issubset(covered_m2)

    # Explained output
    explained = min_roles_greedy_explained(required2, roles2)
    assert "chosen" in explained
    print(f"\nGreedy steps for Example 2:")
    for step in explained["coverage_steps"]:
        print(f"  → assign '{step['role']}', covers: {sorted(step['newly_covered'])}")

    print("\nQ5 — All assertions passed.")
    print()

    print("  Set Cover is NP-hard. Exact solution: O(2^M). Only feasible M < 20.")
    print("  Greedy: O(M²P), gives (1 + ln k) approximation (k = |required|).")
    print("  For M=100 roles, P=50 perms: greedy is practical; exact is not.")
    print()
    print("Security principle: minimum roles = least privilege.")
    print("  Audit delta(required, actual) to detect over-provisioned accounts.")
    print("  Role explosion risk: 1000s of fine-grained roles → use ABAC instead.")
