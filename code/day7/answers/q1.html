# Q1 (Consolidation from Day 2 Q25) - Minimum Window Containing All Severity Levels
# Find the smallest contiguous subarray containing at least one each of:
# "INFO", "WARN", "ERROR". Return start index, end index, and the subarray.
#
# ─── SOLUTION WALKTHROUGH ───────────────────────────────────────────────────
#
# STEP 1 — Restate
#   "Given a list of log events, I need the shortest subarray that contains at
#    least one INFO, one WARN, and one ERROR. Return start index, end index,
#    and the actual slice."
#
# STEP 2 — Assumptions
#   - Events are strings; only "INFO", "WARN", "ERROR" matter — others are ignored
#   - "Contiguous subarray" means consecutive elements (indices are preserved)
#   - If no valid window exists (e.g., missing severity), return None
#   - Empty input → return None
#   - Case-sensitive: "info" ≠ "INFO"
#
# STEP 3 — Naive O(n²)
#   "Check every subarray [i..j]. For each, check if it contains all three.
#    O(n²) subarrays × O(n) check = O(n³). Reduce check to O(1) with a set:
#    O(n²) time."
#
# STEP 4 — Sliding window O(n)
#   "Use two pointers left/right.
#    Expand right until window contains all 3 severities.
#    Then shrink left as far as possible while still valid.
#    Record best window. Move right again.
#
#    Track counts: need = {'INFO':0,'WARN':0,'ERROR':0}
#    have = how many of the 3 we've satisfied (count > 0)
#    When have == 3: record window, try shrinking."
#
# STEP 5 — Edge cases
#   - One severity missing entirely → return None
#   - Minimum window at the start → indices [0, k]
#   - Multiple valid windows of same length → return the first (leftmost)
#
# Complexity:
#   Naive: O(n²) time, O(1) space
#   Sliding window: O(n) time, O(1) space (fixed 3-key dict)
# ─────────────────────────────────────────────────────────────────────────────

REQUIRED = {"INFO", "WARN", "ERROR"}


# ── APPROACH 1: Brute force O(n²) ────────────────────────────────────────────

def min_window_naive(events: list) -> dict | None:
    """
    Check every subarray. O(n²) time, O(1) space.
    """
    if not isinstance(events, list):
        raise TypeError(f"Expected list, got {type(events).__name__}")
    n = len(events)
    if n == 0:
        return None

    best = None

    for i in range(n):
        found = set()
        for j in range(i, n):
            if events[j] in REQUIRED:
                found.add(events[j])
            if found == REQUIRED:
                length = j - i + 1
                if best is None or length < best["length"]:
                    best = {"start": i, "end": j, "length": length,
                            "window": events[i:j + 1]}
                break  # shrink inner loop — can't do better from this i

    return best


# ── APPROACH 2: Sliding window O(n) ──────────────────────────────────────────

def min_window(events: list) -> dict | None:
    """
    Two-pointer sliding window with count tracking.

    - Expand right until all 3 severities are in window.
    - Shrink left while window remains valid.
    - Record best window (smallest length) seen.

    Time:  O(n) — each pointer moves at most n times
    Space: O(1) — 3-key count dict
    """
    if not isinstance(events, list):
        raise TypeError(f"Expected list, got {type(events).__name__}")
    if not events:
        return None

    need   = {"INFO": 0, "WARN": 0, "ERROR": 0}   # current window counts
    have   = 0      # number of severity types with count >= 1
    left   = 0
    best   = None   # (start, end)

    for right in range(len(events)):
        ev = events[right]
        if ev in need:
            if need[ev] == 0:
                have += 1       # this severity just became satisfied
            need[ev] += 1

        # Shrink from left while window is still valid (have == 3)
        while have == 3:
            length = right - left + 1
            if best is None or length < best[1] - best[0] + 1:
                best = (left, right)

            # Remove events[left] from window
            lev = events[left]
            if lev in need:
                need[lev] -= 1
                if need[lev] == 0:
                    have -= 1   # lost this severity — window no longer valid
            left += 1

    if best is None:
        return None

    start, end = best
    return {
        "start":  start,
        "end":    end,
        "length": end - start + 1,
        "window": events[start:end + 1],
    }


# ── MANUAL — no dict comprehension / set operators ────────────────────────────

def min_window_manual(events: list) -> dict | None:
    """
    Same sliding window, manually tracking counts without dict.get() or comprehensions.

    Time:  O(n)
    Space: O(1)
    """
    if events is None:
        raise ValueError("events cannot be None")
    n = len(events)
    if n == 0:
        return None

    cnt_info = cnt_warn = cnt_err = 0
    have = 0
    left = 0
    best_start = best_end = -1
    best_len   = n + 1

    for right in range(n):
        ev = events[right]
        if ev == "INFO":
            if cnt_info == 0:
                have += 1
            cnt_info += 1
        elif ev == "WARN":
            if cnt_warn == 0:
                have += 1
            cnt_warn += 1
        elif ev == "ERROR":
            if cnt_err == 0:
                have += 1
            cnt_err += 1

        while have == 3:
            length = right - left + 1
            if length < best_len:
                best_len   = length
                best_start = left
                best_end   = right

            lev = events[left]
            if lev == "INFO":
                cnt_info -= 1
                if cnt_info == 0:
                    have -= 1
            elif lev == "WARN":
                cnt_warn -= 1
                if cnt_warn == 0:
                    have -= 1
            elif lev == "ERROR":
                cnt_err -= 1
                if cnt_err == 0:
                    have -= 1
            left += 1

    if best_start == -1:
        return None

    return {
        "start":  best_start,
        "end":    best_end,
        "length": best_len,
        "window": events[best_start:best_end + 1],
    }


# ── Tests ─────────────────────────────────────────────────────────────────────

if __name__ == "__main__":
    # Basic from question
    events1 = ["INFO", "XSS", "WARN", "INFO", "ERROR", "WARN"]
    r1 = min_window(events1)
    assert r1 is not None
    assert set(r1["window"]) & REQUIRED == REQUIRED, "Window must contain all 3"
    # Minimum window is [2..4] (WARN,INFO,ERROR) length 3, not [0..4] length 5
    assert r1["start"] == 2 and r1["end"] == 4, f"Expected [2,4], got {r1}"
    print(f"events1 → start={r1['start']}, end={r1['end']}, window={r1['window']}")

    # Minimum window is at the end
    events2 = ["XSS", "XSS", "INFO", "WARN", "ERROR"]
    r2 = min_window(events2)
    assert r2["length"] == 3
    assert r2["start"] == 2 and r2["end"] == 4

    # Earlier window wins when two are equal length
    events3 = ["INFO", "WARN", "ERROR", "XSS", "INFO", "WARN", "ERROR"]
    r3 = min_window(events3)
    assert r3["start"] == 0 and r3["end"] == 2    # first window

    # No WARN → None
    events4 = ["INFO", "ERROR", "INFO", "ERROR"]
    assert min_window(events4) is None

    # Empty
    assert min_window([]) is None

    # All three
    events5 = ["ERROR", "INFO", "WARN"]
    r5 = min_window(events5)
    assert r5["length"] == 3

    # All approaches agree
    for tc in [events1, events2, events3, events4, events5, []]:
        rn = min_window_naive(tc)
        rw = min_window(tc)
        rm = min_window_manual(tc)
        if rn is None:
            assert rw is None and rm is None
        else:
            assert rn["length"] == rw["length"] == rm["length"], \
                f"Mismatch on {tc}: {rn['length']}, {rw['length']}, {rm['length']}"

    print("\nQ1 — All assertions passed.")
    print("Complexity: O(n) time, O(1) space (sliding window with 3-key count dict)")
    print("Key: expand right until valid, shrink left while still valid, track best.")
