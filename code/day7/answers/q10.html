# Q10 (Gap Fixer) - Binary Search: Find Minimum in Rotated Sorted Array
# Given a rotated sorted array of unique integers, find the minimum element.
# Variant: also identify whether the array is rotated at all.
#
# ─── SOLUTION WALKTHROUGH ───────────────────────────────────────────────────
#
# STEP 1 — Restate
#   "A sorted array was rotated at some unknown pivot. Find the minimum element
#    in O(log n) without scanning every element. Also report the pivot index
#    where the rotation occurred."
#
# STEP 2 — Insight
#   "In a rotated array [3,4,5,1,2], the minimum element is the one whose
#    left neighbor is greater (arr[pivot-1] > arr[pivot]). Binary search can
#    narrow to this location by comparing mid to the rightmost element:
#    - If arr[mid] > arr[hi]: minimum is in the RIGHT half (lo = mid+1)
#    - Else:                  minimum is in the LEFT half or AT mid (hi = mid)
#    Terminate when lo == hi."
#
# STEP 3 — Not-rotated case
#   "If arr[0] < arr[-1], the array was never rotated (or rotated by n).
#    Minimum is arr[0], pivot is 0 (no rotation — we define pivot as the
#    index of the minimum, which for a sorted array is 0)."
#
# STEP 4 — Why compare to arr[hi], not arr[lo]?
#   "Comparing arr[mid] to arr[hi] tells us whether the right half is sorted
#    (arr[mid] <= arr[hi] → right half sorted → min in left half or at mid).
#    Comparing to arr[lo] would fail when lo == mid in 2-element arrays."
#
# STEP 5 — Edge cases
#   - Single element: min = arr[0], pivot = 0
#   - Not rotated (fully sorted): min = arr[0], pivot = 0
#   - Rotated by 1: arr[-1] is the old first element moved to end... wait:
#     rotated by 1 means arr[0] is now arr[n-1]'s successor — min is arr[1]
#     Actually: if rotated by k, arr[k] is the original first element = min
#   - All same values: not handled (assumed distinct — state this assumption)
#
# Complexity:
#   Naive:         O(n) linear scan
#   Binary search: O(log n)
# ─────────────────────────────────────────────────────────────────────────────


# ── APPROACH 1: Naive — O(n) linear scan ─────────────────────────────────────

def find_min_naive(arr: list) -> dict:
    """
    Linear scan for minimum. O(n).

    Returns {'min_value': int, 'min_index': int, 'pivot_index': int,
             'is_rotated': bool}
    """
    if not isinstance(arr, list):
        raise TypeError(f"Expected list, got {type(arr).__name__}")
    if not arr:
        raise ValueError("Array must be non-empty")

    min_val = arr[0]
    min_idx = 0
    for i in range(1, len(arr)):
        if arr[i] < min_val:
            min_val = arr[i]
            min_idx = i

    is_rotated = min_idx != 0
    return {
        "min_value":   min_val,
        "min_index":   min_idx,
        "pivot_index": min_idx,    # pivot = index of minimum
        "is_rotated":  is_rotated,
    }


# ── APPROACH 2: Binary search — O(log n) ─────────────────────────────────────

def find_min(arr: list) -> dict:
    """
    Binary search for minimum in rotated sorted array.

    Key invariant:
      Compare arr[mid] with arr[hi]:
        arr[mid] > arr[hi] → minimum is in (mid, hi]   → lo = mid + 1
        arr[mid] < arr[hi] → minimum is in [lo, mid]   → hi = mid
      When lo == hi: arr[lo] is the minimum (= pivot index).

    Time:  O(log n)
    Space: O(1)

    Assumes: distinct integers, array is non-empty.
    """
    if not isinstance(arr, list):
        raise TypeError(f"Expected list, got {type(arr).__name__}")
    if not arr:
        raise ValueError("Array must be non-empty")

    lo, hi = 0, len(arr) - 1

    while lo < hi:
        mid = lo + (hi - lo) // 2

        if arr[mid] > arr[hi]:
            # Right half is broken; min is somewhere in (mid, hi]
            lo = mid + 1
        else:
            # arr[mid] <= arr[hi]: left half or mid holds the min
            hi = mid

    # lo == hi == pivot (index of minimum element)
    pivot = lo
    return {
        "min_value":   arr[pivot],
        "min_index":   pivot,
        "pivot_index": pivot,
        "is_rotated":  pivot != 0,
    }


# ── APPROACH 3: Python built-in (for reference) ───────────────────────────────

def find_min_builtin(arr: list) -> dict:
    """
    Use min() for simplicity comparison. O(n).
    DO NOT use in practice as the primary answer.
    """
    if not arr:
        raise ValueError("Array must be non-empty")
    min_val = min(arr)
    min_idx = arr.index(min_val)
    return {
        "min_value":   min_val,
        "min_index":   min_idx,
        "pivot_index": min_idx,
        "is_rotated":  min_idx != 0,
    }


# ── APPROACH 4: Manual — no builtins, no imports ──────────────────────────────

def find_min_manual(arr: list) -> dict:
    """
    Binary search with explicit integer arithmetic only.
    Same algorithm as find_min but expressed without any imports or builtins
    other than len() and basic operators.

    Time:  O(log n)
    Space: O(1)
    """
    if not arr:
        raise ValueError("Array must be non-empty")

    lo = 0
    hi = len(arr) - 1

    while lo < hi:
        mid = lo + (hi - lo) // 2
        if arr[mid] > arr[hi]:
            lo = mid + 1
        else:
            hi = mid

    pivot = lo
    is_rotated = True if pivot != 0 else False
    return {
        "min_value":   arr[pivot],
        "min_index":   pivot,
        "pivot_index": pivot,
        "is_rotated":  is_rotated,
    }
