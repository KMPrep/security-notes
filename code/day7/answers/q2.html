# Q2 (Consolidation from Day 3 Q4) - Decode Compressed Log Token — Nested
# Decode k[pattern] encoding with nesting up to 4 levels deep.
# Also validate input before decoding — return None if invalid.
#
# ─── SOLUTION WALKTHROUGH ───────────────────────────────────────────────────
#
# STEP 1 — Restate
#   "The encoding is: a number followed by a bracket-enclosed string means repeat
#    that string that many times. Nesting means the inner pattern is decoded first,
#    then the outer count applies. I also need to validate the input — bad input
#    should return None instead of crashing."
#
# STEP 2 — Assumptions
#   - Numbers are positive integers (k >= 1)
#   - Only lowercase letters outside brackets
#   - Brackets are always paired with a preceding number
#   - Nesting depth up to 4 levels
#   - Invalid: unmatched brackets, number without '[', non-letters in wrong place,
#              empty brackets '[]'
#   - Valid characters: digits [0-9], lowercase letters [a-z], '[', ']'
#
# STEP 3 — Validation approach
#   "Use a simple regex to match the full structure, or a manual scan:
#    - Count brackets: must be balanced
#    - Every '[' must be preceded by a digit
#    - Every digit sequence must be followed by '['
#    - No trailing digits without '['
#    - No empty '[]'"
#
# STEP 4 — Decoding approach (stack)
#   "Use a stack. When we see:
#    - digit: build current number
#    - '[': push (current_string, current_number) onto stack; reset both
#    - letter: append to current_string
#    - ']': pop (prev_string, repeat_count); current_string = prev_string + repeat_count * current_string"
#
# STEP 5 — Edge cases
#   - Plain string with no encoding: "abc" → "abc"
#   - k=1: "1[abc]" → "abc"
#   - Deep nesting: "2[a3[b2[c]]]"
#   - Invalid: "2[a]3" — trailing digit without bracket
#   - Invalid: "abc[3]" — bracket not preceded by digit
#   - Invalid: "[2[a]]" — outer bracket not preceded by digit
#   - Invalid: "2[]" — empty bracket content
#
# Complexity:
#   Decode: O(n * k_max) time where n=len(input), k_max=max repeat count
#   Validate: O(n) time
# ─────────────────────────────────────────────────────────────────────────────

import re


# ── VALIDATION ────────────────────────────────────────────────────────────────

def validate_encoded(s: str) -> tuple[bool, str]:
    """
    Validate the encoded string. Returns (True, "") if valid, or (False, reason).

    Rules:
    1. Only chars: digits, lowercase letters, '[', ']'
    2. Every '[' must be immediately preceded by a digit
    3. Every ']' must close a previously opened '['
    4. Brackets must be balanced
    5. No standalone digit not followed eventually by '['
    6. No empty '[]'
    """
    if not isinstance(s, str):
        return False, f"Expected string, got {type(s).__name__}"
    if not s:
        return True, ""   # empty string is technically valid (decodes to "")

    # Rule 1: allowed characters only
    if not re.fullmatch(r'[0-9a-z\[\]]+', s):
        return False, "Invalid characters detected (only digits, lowercase letters, brackets allowed)"

    # Use a stack to validate structure
    stack = []
    i = 0
    n = len(s)
    pending_number = False   # True after we finish parsing a number

    while i < n:
        ch = s[i]
        if ch.isdigit():
            # Read full number
            while i < n and s[i].isdigit():
                i += 1
            # After a number, next non-trivial char must be '['
            if i >= n or s[i] != '[':
                return False, f"Digit at position {i} not followed by '['"
            pending_number = True
            continue
        elif ch == '[':
            if not pending_number:
                return False, f"'[' at position {i} not preceded by a number"
            stack.append('[')
            pending_number = False
            # Check for empty bracket immediately
            if i + 1 < n and s[i + 1] == ']':
                return False, f"Empty bracket at position {i}"
        elif ch == ']':
            if not stack:
                return False, f"Unmatched ']' at position {i}"
            stack.pop()
        # lowercase letters: always valid between brackets or at start
        i += 1

    if stack:
        return False, f"Unmatched '[': {len(stack)} bracket(s) not closed"
    if pending_number:
        return False, "Trailing digit(s) not followed by '['"

    return True, ""


# ── APPROACH 1: Stack-based decoder ──────────────────────────────────────────

def decode_string(s: str) -> str | None:
    """
    Decode k[pattern] encoded string using a stack.

    Algorithm:
      stack holds (accumulated_string_so_far, repeat_count) when entering a bracket.
      On ']': pop and concatenate.

    Time:  O(n * k_max) — output can be exponentially large
    Space: O(n) stack depth

    Returns None if input is invalid.
    """
    if not isinstance(s, str):
        return None

    valid, reason = validate_encoded(s)
    if not valid:
        return None

    stack        = []    # stack of (string_before_bracket, repeat_count)
    current_str  = ""
    current_num  = 0

    for ch in s:
        if ch.isdigit():
            current_num = current_num * 10 + int(ch)
        elif ch == '[':
            # Save state and start fresh
            stack.append((current_str, current_num))
            current_str = ""
            current_num = 0
        elif ch == ']':
            prev_str, repeat = stack.pop()
            current_str = prev_str + repeat * current_str
        else:
            current_str += ch

    return current_str


# ── APPROACH 2: Recursive (for comparison — harder to validate) ───────────────

def _decode_recursive(s: str, i: int) -> tuple[str, int]:
    """
    Recursive helper. Returns (decoded_string, new_index).
    Called when we're inside a bracket group.
    """
    result = ""
    while i < len(s) and s[i] != ']':
        if s[i].isdigit():
            # Parse number
            num = 0
            while i < len(s) and s[i].isdigit():
                num = num * 10 + int(s[i])
                i += 1
            # Expect '['
            i += 1   # skip '['
            inner, i = _decode_recursive(s, i)
            i += 1   # skip ']'
            result += num * inner
        else:
            result += s[i]
            i += 1
    return result, i


def decode_string_recursive(s: str) -> str | None:
    """
    Recursive decoder. Validates first.

    Time:  O(n * k_max)
    Space: O(depth) stack frames
    """
    if not isinstance(s, str):
        return None
    valid, _ = validate_encoded(s)
    if not valid:
        return None
    result, _ = _decode_recursive(s, 0)
    return result


# ── MANUAL — no f-strings for building, no string multiplication ───────────────

def decode_string_manual(s: str) -> str | None:
    """
    Stack decoder without string multiplication operator.
    Manually repeats by appending in a loop.

    Time:  O(n * k_max)
    Space: O(n)
    """
    if s is None:
        return None
    valid, _ = validate_encoded(s)
    if not valid:
        return None

    stack       = []
    current_str = []   # list of chars (efficient append)
    current_num = 0

    for ch in s:
        if ch.isdigit():
            current_num = current_num * 10 + int(ch)
        elif ch == '[':
            stack.append((list(current_str), current_num))
            current_str = []
            current_num = 0
        elif ch == ']':
            prev_str, repeat = stack.pop()
            repeated = []
            for _ in range(repeat):
                for c in current_str:
                    repeated.append(c)
            current_str = prev_str + repeated
        else:
            current_str.append(ch)

    return "".join(current_str)


# ── Tests ─────────────────────────────────────────────────────────────────────

if __name__ == "__main__":
    # From the question
    assert decode_string("2[a3[b2[c]]]") == "abccbccbccabccbccbcc"

    # Simple cases
    assert decode_string("3[a]")       == "aaa"
    assert decode_string("2[abc]")     == "abcabc"
    assert decode_string("2[ab]3[c]")  == "ababccc"
    assert decode_string("abc")        == "abc"          # no encoding
    assert decode_string("")           == ""

    # More nesting
    assert decode_string("3[a2[b]]")   == "abbabbabb"
    assert decode_string("1[a1[b]]")   == "ab"

    # All three approaches agree
    test_cases = [
        "2[a3[b2[c]]]",
        "3[a]",
        "2[ab]3[c]",
        "3[a2[b]]",
        "abc",
        "",
    ]
    for tc in test_cases:
        s = decode_string(tc)
        r = decode_string_recursive(tc)
        m = decode_string_manual(tc)
        assert s == r == m, f"Mismatch on {tc!r}: {s!r}, {r!r}, {m!r}"

    # Validation — invalid inputs
    assert decode_string("2[a]3")     is None, "Trailing digit should be invalid"
    assert decode_string("[abc]")     is None, "'[' without preceding number"
    assert decode_string("2[]")       is None, "Empty brackets"
    assert decode_string("2[a")       is None, "Unmatched '['"
    assert decode_string("2a]")       is None, "Unmatched ']'"
    assert decode_string("2[A]")      is None, "Uppercase letter not allowed"
    assert decode_string(None)        is None, "None input"

    # Validate function directly
    ok, _  = validate_encoded("2[a3[b]]")
    bad, r = validate_encoded("2[a]3")
    assert ok  is True
    assert bad is False

    print("Q2 — All assertions passed.")
    print(f"decode_string('2[a3[b2[c]]]') = {decode_string('2[a3[b2[c]]]')!r}")
