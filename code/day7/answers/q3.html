# Q3 (Consolidation from Day 4 Q5) - Search in Rotated Sorted Array
# Re-implement from scratch. Also return the pivot index (where rotation happened).
#
# ─── SOLUTION WALKTHROUGH ───────────────────────────────────────────────────
#
# STEP 1 — Restate
#   "The array was originally sorted but rotated at some pivot. I need to:
#    1. Find the index of the target (or -1 if not found)
#    2. Find the pivot index — the position where the rotation happened
#       (i.e., the index of the minimum element)
#    Both in O(log n)."
#
# STEP 2 — Assumptions
#   - All elements are distinct (no duplicates)
#   - Array may not be rotated at all (pivot = 0)
#   - Rotation means: arr[pivot-1] > arr[pivot] where pivot is minimum element index
#   - 0 <= target search index <= n-1
#   - Empty array → return (-1, -1)
#
# STEP 3 — Pivot finding (binary search for minimum)
#   "After rotation, arr looks like: [big..big, min, small..big_left_of_rotation]
#    The minimum element is the only element smaller than both its neighbours.
#
#    Binary search: compare arr[mid] with arr[hi]:
#    - If arr[mid] > arr[hi]: minimum is to the RIGHT of mid → lo = mid + 1
#    - Else: minimum is at mid or to the LEFT → hi = mid
#    When lo == hi: that's the minimum (pivot)."
#
# STEP 4 — Target search (single-pass)
#   "With pivot known, apply standard binary search on whichever sorted half
#    the target falls in."
#
# STEP 5 — Alternative: single-pass without finding pivot
#   "At every mid point, one of the two halves [lo..mid] or [mid+1..hi] is
#    definitely sorted. Check which one, then check if target is in that half."
#
# STEP 6 — Edge cases
#   - Not rotated: pivot = 0 (minimum is at index 0)
#   - Target not present → -1
#   - Single element → return (0 if target matches else -1, 0)
#   - Target at pivot → return (pivot, pivot)
#
# Complexity:
#   Both find_pivot and search_target: O(log n)
#   Total: O(log n)
# ─────────────────────────────────────────────────────────────────────────────


# ── STEP A: Find pivot (index of minimum element) ────────────────────────────

def find_pivot(arr: list) -> int:
    """
    Binary search for the pivot (minimum element index).

    Invariant: arr[lo..hi] contains the minimum.
    Compare arr[mid] with arr[hi]:
    - arr[mid] > arr[hi]: min is strictly to the right  → lo = mid + 1
    - arr[mid] < arr[hi]: min is at mid or to the left → hi = mid

    Returns: index of minimum element (0 if not rotated)
    Time: O(log n)
    """
    if not arr:
        return -1

    lo, hi = 0, len(arr) - 1

    while lo < hi:
        mid = lo + (hi - lo) // 2
        if arr[mid] > arr[hi]:
            lo = mid + 1    # minimum is in right half
        else:
            hi = mid        # minimum is at mid or in left half

    return lo   # lo == hi == pivot index


# ── APPROACH 1: Find pivot first, then binary search ─────────────────────────

def search_rotated(arr: list, target: int) -> dict:
    """
    Two-step O(log n) approach:
    1. Find pivot (minimum index)
    2. Binary search in the correct sorted half

    Returns dict with 'target_index' and 'pivot_index'.
    """
    if not isinstance(arr, list):
        raise TypeError(f"Expected list, got {type(arr).__name__}")
    if not arr:
        return {"target_index": -1, "pivot_index": -1}

    n     = len(arr)
    pivot = find_pivot(arr)

    def _binary_search(lo: int, hi: int) -> int:
        while lo <= hi:
            mid = lo + (hi - lo) // 2
            if arr[mid] == target:
                return mid
            elif arr[mid] < target:
                lo = mid + 1
            else:
                hi = mid - 1
        return -1

    # Determine which half the target lives in
    if arr[pivot] == target:
        idx = pivot
    elif pivot == 0:
        # Array is not rotated — binary search the entire array
        idx = _binary_search(0, n - 1)
    elif target >= arr[0]:
        # Target is in the left sorted segment [0..pivot-1]
        idx = _binary_search(0, pivot - 1)
    else:
        # Target is in the right sorted segment [pivot..n-1]
        idx = _binary_search(pivot, n - 1)

    return {"target_index": idx, "pivot_index": pivot}


# ── APPROACH 2: Single-pass (no pivot pre-computation) ───────────────────────

def search_rotated_single_pass(arr: list, target: int) -> dict:
    """
    Single binary search — identify sorted half at each step.

    At each mid:
    - Left half [lo..mid] is sorted if arr[lo] <= arr[mid]
      → check if target in [arr[lo], arr[mid]]; if yes go left, else right
    - Otherwise right half [mid+1..hi] is sorted
      → check if target in [arr[mid+1], arr[hi]]; if yes go right, else left

    Returns dict with 'target_index' and 'pivot_index' (found separately).
    Time: O(log n)
    """
    if not isinstance(arr, list):
        raise TypeError(f"Expected list, got {type(arr).__name__}")
    if not arr:
        return {"target_index": -1, "pivot_index": -1}

    lo, hi = 0, len(arr) - 1

    while lo <= hi:
        mid = lo + (hi - lo) // 2
        if arr[mid] == target:
            return {"target_index": mid, "pivot_index": find_pivot(arr)}

        if arr[lo] <= arr[mid]:
            # Left half [lo..mid] is sorted
            if arr[lo] <= target < arr[mid]:
                hi = mid - 1
            else:
                lo = mid + 1
        else:
            # Right half [mid+1..hi] is sorted
            if arr[mid] < target <= arr[hi]:
                lo = mid + 1
            else:
                hi = mid - 1

    return {"target_index": -1, "pivot_index": find_pivot(arr)}


# ── APPROACH 3: Naive linear scan (for baseline) ─────────────────────────────

def search_rotated_naive(arr: list, target: int) -> dict:
    """
    Linear scan. O(n). State first to establish baseline.
    """
    if not arr:
        return {"target_index": -1, "pivot_index": -1}

    idx   = -1
    pivot = 0
    for i in range(len(arr)):
        if arr[i] == target:
            idx = i
        if i > 0 and arr[i] < arr[i - 1]:
            pivot = i

    return {"target_index": idx, "pivot_index": pivot}


# ── MANUAL — no helper functions, no dict ────────────────────────────────────

def search_rotated_manual(arr: list, target: int) -> tuple:
    """
    Returns (target_index, pivot_index). No helper functions, no imports.

    Time: O(log n)
    """
    if not arr:
        return (-1, -1)

    n = len(arr)

    # Find pivot
    lo, hi = 0, n - 1
    while lo < hi:
        mid = lo + (hi - lo) // 2
        if arr[mid] > arr[hi]:
            lo = mid + 1
        else:
            hi = mid
    pivot = lo

    # Binary search in correct half
    if arr[pivot] == target:
        return (pivot, pivot)

    if pivot == 0:
        # Array is not rotated — search entire array
        lo2, hi2 = 0, n - 1
    elif target >= arr[0]:
        lo2, hi2 = 0, pivot - 1
    else:
        lo2, hi2 = pivot, n - 1

    while lo2 <= hi2:
        mid = lo2 + (hi2 - lo2) // 2
        if arr[mid] == target:
            return (mid, pivot)
        elif arr[mid] < target:
            lo2 = mid + 1
        else:
            hi2 = mid - 1

    return (-1, pivot)

if __name__ == "__main__":
    # From the question
    arr1 = [6, 7, 0, 1, 2, 3, 4, 5]
    r1   = search_rotated(arr1, 0)
    assert r1["target_index"] == 2,  f"Expected index 2, got {r1['target_index']}"
    assert r1["pivot_index"]  == 2,  f"Expected pivot 2, got {r1['pivot_index']}"
    print(f"[6,7,0,1,2,3,4,5] target=0 → index={r1['target_index']}, pivot={r1['pivot_index']}")

    # Not rotated
    arr2 = [1, 2, 3, 4, 5]
    r2   = search_rotated(arr2, 3)
    assert r2["target_index"] == 2
    assert r2["pivot_index"]  == 0    # no rotation → minimum at index 0

    # Target not found
    assert search_rotated(arr1, 9)["target_index"] == -1

    # Target at pivot
    r3 = search_rotated([4, 5, 6, 7, 0, 1, 2], 0)
    assert r3["target_index"] == 4
    assert r3["pivot_index"]  == 4

    # Single element
    assert search_rotated([1], 1)["target_index"]  == 0
    assert search_rotated([1], 2)["target_index"]  == -1

    # Empty
    assert search_rotated([], 1)["target_index"]   == -1

    # All approaches agree
