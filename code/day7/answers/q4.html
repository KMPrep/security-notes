# Q4 (Consolidation from Day 5 Q15) - Suspicious Login Report Generator
# Rebuild the full script from scratch. Generate a synthetic auth.log (30+ lines),
# detect brute force, multi-IP logins, and after-hours logins, output a report.
#
# ─── SOLUTION WALKTHROUGH ───────────────────────────────────────────────────
#
# STEP 1 — Restate
#   "Parse an SSH auth.log and produce a security report identifying:
#    1. Brute force: a user with >= 5 failed password attempts
#    2. Multi-IP: a user who successfully logged in from >= 2 distinct IPs
#    3. After-hours: any successful login between 20:00 and 06:00
#    Output must show user, evidence, and severity."
#
# STEP 2 — Assumptions
#   - Log format: syslog/OpenSSH format
#     'Mon DD HH:MM:SS host sshd[PID]: {message}'
#   - "Failed password for {user} from {ip}" for failures
#   - "Accepted password for {user} from {ip}" for successes
#   - Case-sensitive usernames
#   - After-hours: hour >= 20 OR hour < 6 (local server time)
#   - No timezone handling needed
#   - Brute force threshold: 5 failures (not per-IP — global per user)
#
# STEP 3 — Approach
#   Single pass over log lines:
#   - Regex extract: timestamp, user, ip, event type
#   - Maintain per-user: fail_count, success_ips, after_hours_logins
#   - After parsing: apply detection rules
#
# STEP 4 — Security framing
#   "In production:
#    - Parse incrementally (tail -f or logrotate-aware)
#    - Account lockout after N failures (pam_faillock / fail2ban)
#    - Multi-IP alert: could be VPN roaming (reduce false positives with geo-distance check)
#    - After-hours: baseline what's normal per user before alerting
#    - PII: log IPs but anonymize in reports if required by policy"
#
# Complexity:
#   Parsing: O(L * R) where L = lines, R = regex match time ≈ O(L)
#   Detection: O(U) where U = unique users
# ─────────────────────────────────────────────────────────────────────────────

# ── REGEX PATTERNS ────────────────────────────────────────────────────────────

_FAILED_PAT  = re.compile(
    r'(\w{3}\s+\d+\s+\d{2}:\d{2}:\d{2}).*Failed password for (\S+) from ([\d.]+)'
)
_ACCEPTED_PAT = re.compile(
    r'(\w{3}\s+\d+\s+\d{2}:\d{2}:\d{2}).*Accepted password for (\S+) from ([\d.]+)'
)
_TS_PAT = re.compile(r'(\d{2}):(\d{2}):(\d{2})')


def _is_after_hours(timestamp_str: str) -> bool:
    """Return True if hour >= 20 or hour < 6."""
    m = _TS_PAT.search(timestamp_str)
    if not m:
        return False
    hour = int(m.group(1))
    return hour >= 20 or hour < 6


# ── PARSING ───────────────────────────────────────────────────────────────────

def parse_auth_log(log_text: str) -> dict:
    """
    Parse auth.log text. Returns per-user stats:
    {
      user: {
        'fail_count':          int,
        'fail_ips':            set,
        'success_ips':         set,
        'after_hours_logins':  list of (timestamp, ip),
        'success_events':      list of (timestamp, ip),
      }
    }
    """
    stats = defaultdict(lambda: {
        "fail_count":         0,
        "fail_ips":           set(),
        "success_ips":        set(),
        "after_hours_logins": [],
        "success_events":     [],
    })

    for line in log_text.splitlines():
        line = line.strip()
        if not line:
            continue

        m = _FAILED_PAT.search(line)
        if m:
            ts, user, ip = m.group(1), m.group(2), m.group(3)
            stats[user]["fail_count"] += 1
            stats[user]["fail_ips"].add(ip)
            continue

        m = _ACCEPTED_PAT.search(line)
        if m:
            ts, user, ip = m.group(1), m.group(2), m.group(3)
            stats[user]["success_ips"].add(ip)
            stats[user]["success_events"].append((ts, ip))
            if _is_after_hours(ts):
                stats[user]["after_hours_logins"].append((ts, ip))

    return stats


# ── DETECTION RULES ───────────────────────────────────────────────────────────

BRUTE_FORCE_THRESHOLD = 5
MULTI_IP_THRESHOLD    = 2

def detect_suspicious(stats: dict) -> list:
    """
    Apply three detection rules to per-user stats.
    Returns list of finding dicts sorted by severity then username.
    """
    SEVERITY_ORDER = {"CRITICAL": 0, "HIGH": 1, "MEDIUM": 2, "LOW": 3}
    findings = []

    for user, s in stats.items():
        # Rule 1: Brute force (>= threshold failed attempts)
        if s["fail_count"] >= BRUTE_FORCE_THRESHOLD:
            findings.append({
                "severity":  "HIGH",
                "rule":      "BRUTE_FORCE",
                "user":      user,
                "detail":    f"{s['fail_count']} failed attempts from IPs: {sorted(s['fail_ips'])}",
            })

        # Rule 2: Multi-IP successful logins (>= 2 distinct source IPs)
        if len(s["success_ips"]) >= MULTI_IP_THRESHOLD:
            findings.append({
                "severity":  "MEDIUM",
                "rule":      "MULTI_IP_LOGIN",
                "user":      user,
                "detail":    f"Successful logins from {len(s['success_ips'])} distinct IPs: "
                             f"{sorted(s['success_ips'])}",
            })

        # Rule 3: After-hours successful login
        for ts, ip in s["after_hours_logins"]:
            findings.append({
                "severity":  "LOW",
                "rule":      "AFTER_HOURS_LOGIN",
                "user":      user,
                "detail":    f"Login at {ts} from {ip}",
            })

    findings.sort(key=lambda f: (SEVERITY_ORDER.get(f["severity"], 99), f["user"]))
    return findings


# ── REPORT FORMATTING ─────────────────────────────────────────────────────────

def format_report(findings: list) -> str:
    """Produce human-readable security report."""
    if not findings:
        return "No suspicious activity detected.\n"

    lines = [
        "=" * 60,
        "  SUSPICIOUS LOGIN REPORT",
        "=" * 60,
        f"  Total findings: {len(findings)}",
        "=" * 60,
        "",
    ]
    current_sev = None
    for f in findings:
        if f["severity"] != current_sev:
            current_sev = f["severity"]
            lines.append(f"[{current_sev}]")
        lines.append(f"  Rule:   {f['rule']}")
        lines.append(f"  User:   {f['user']}")
        lines.append(f"  Detail: {f['detail']}")
        lines.append("")

    lines.append("=" * 60)
    return "\n".join(lines)
