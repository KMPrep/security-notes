# Q5 (Consolidation from Day 6 Q4) - Threat Event Correlation Engine (Twist)
# Same as Day 6 Q4, but: (1) sensor events may arrive OUT OF ORDER — sort first.
# (2) if a sensor_b event matches multiple sensor_a events, report only the
# CLOSEST match (smallest time delta).
#
# ─── SOLUTION WALKTHROUGH ───────────────────────────────────────────────────
#
# STEP 1 — Restate
#   "Two sensors emit (timestamp, event_type, source_ip) events. They may arrive
#    out of order. A correlated threat is the same IP appearing in both sensors
#    within T seconds. If one sensor_b event correlates with multiple sensor_a
#    events, I only want the ONE with the smallest time delta."
#
# STEP 2 — Assumptions
#   - Sort both sensors by timestamp before processing
#   - 'Closest match' means smallest |ts_a - ts_b|
#   - If two sensor_a events tie for closest delta, take the one with smaller ts_a
#   - Still return all correlated pairs after deduplication (one per sensor_b event)
#   - Output sorted by sensor_a timestamp ascending
#
# STEP 3 — Algorithm
#   Step 1: Sort sensor_a and sensor_b by timestamp O((n+m) log(n+m))
#   Step 2: Group sensor_b by IP; sort each IP's events O(m)
#   Step 3: For each sensor_a event, binary search sensor_b[ip] for window
#           [ts_a-T, ts_a+T]; collect candidates
#   Step 4: Track for each sensor_b event key (ts_b, ip): best (min delta) sensor_a
#   Step 5: Emit one pair per sensor_b event (best match)
#   Step 6: Sort output by sensor_a timestamp
#
# STEP 4 — Data structure for "best sensor_a per sensor_b event"
#   "Use a dict keyed by (ts_b, type_b, ip) → best (delta, ts_a, type_a, ip)
#    After processing all sensor_a events, emit the best match for each sensor_b key."
#
# STEP 5 — Edge cases
#   - Out-of-order input: must sort first
#   - T=0: only exact timestamp matches
#   - Same IP in sensor_b, multiple events in window: one per sensor_b event
#   - No correlation for any IP → return []
#
# Complexity:
#   Sorting: O((n+m) log(n+m))
#   Processing: O(n log m + c)
#   Total: O((n+m) log(n+m))
# ─────────────────────────────────────────────────────────────────────────────

import bisect
from collections import defaultdict


# ── APPROACH 1: Naive — sort + O(n*m) correlation ────────────────────────────

def correlate_closest_naive(sensor_a: list, sensor_b: list, T: int) -> list:
    """
    Sort both sensors, then brute-force all pairs. For each sensor_b event,
    keep only the closest sensor_a match.

    Time:  O((n+m) log(n+m) + n*m)
    Space: O(c)
    """
    if not isinstance(sensor_a, list) or not isinstance(sensor_b, list):
        raise TypeError("sensor_a and sensor_b must be lists")
    if T < 0:
        raise ValueError(f"T must be >= 0, got {T}")
    if not sensor_a or not sensor_b:
        return []

    # Step 1: Sort both by timestamp
    a = sorted(sensor_a, key=lambda e: e[0])
    b = sorted(sensor_b, key=lambda e: e[0])

    # Step 2: For each sensor_b event, find best (min delta) sensor_a match
    # Key: (ts_b, type_b, ip_b) → best_pair dict
    best: dict = {}

    for ts_a, type_a, ip_a in a:
        for ts_b, type_b, ip_b in b:
            if ip_a != ip_b:
                continue
            delta = abs(ts_a - ts_b)
            if delta > T:
                continue
            key = (ts_b, type_b, ip_b)
            if key not in best or delta < best[key]["delta_seconds"] or \
               (delta == best[key]["delta_seconds"] and ts_a < best[key]["sensor_a_event"][0]):
                best[key] = {
                    "sensor_a_event": (ts_a, type_a, ip_a),
                    "sensor_b_event": (ts_b, type_b, ip_b),
                    "ip":             ip_a,
                    "delta_seconds":  delta,
                }

    result = list(best.values())
    result.sort(key=lambda r: r["sensor_a_event"][0])
    return result


# ── APPROACH 2: Group by IP + binary search + best-match tracking ─────────────

def correlate_closest(sensor_a: list, sensor_b: list, T: int) -> list:
    """
    Optimized approach:

    1. Sort both sensors by timestamp.
    2. Group sensor_b events by IP.
    3. For each sensor_a event, binary search in sensor_b[ip] for window.
    4. Track best (minimum delta) sensor_a for each sensor_b event key.
    5. Emit one pair per sensor_b event.

    Time:  O((n+m) log(n+m) + n log m + c)
    Space: O(m)
    """
    if not isinstance(sensor_a, list) or not isinstance(sensor_b, list):
        raise TypeError("sensor_a and sensor_b must be lists")
    if T < 0:
        raise ValueError(f"T must be >= 0, got {T}")
    if not sensor_a or not sensor_b:
        return []

    # Step 1: Sort both sensors by timestamp
    a_sorted = sorted(sensor_a, key=lambda e: e[0])
    b_sorted = sorted(sensor_b, key=lambda e: e[0])

    # Step 2: Group sensor_b by IP → {ip: [(ts_b, type_b), ...]} sorted by ts
    b_by_ip: dict = defaultdict(list)
    for ts_b, type_b, ip_b in b_sorted:
        b_by_ip[ip_b].append((ts_b, type_b))
    # Already sorted since b_sorted is sorted globally; but per-IP may need re-sort
    for ip in b_by_ip:
        b_by_ip[ip].sort()

    # Step 3+4: Process each sensor_a event
    # best_match: sensor_b event key → best match dict
    best_match: dict = {}

    for ts_a, type_a, ip_a in a_sorted:
        if ip_a not in b_by_ip:
            continue

        b_events = b_by_ip[ip_a]
        b_ts     = [e[0] for e in b_events]

        # Find all sensor_b events within [ts_a-T, ts_a+T]
        lo = bisect.bisect_left(b_ts,  ts_a - T)
        hi = bisect.bisect_right(b_ts, ts_a + T)

        for idx in range(lo, hi):
            ts_b, type_b = b_events[idx]
            delta = abs(ts_a - ts_b)
            b_key = (ts_b, type_b, ip_a)    # unique key for this sensor_b event

            if b_key not in best_match:
                best_match[b_key] = {
                    "sensor_a_event": (ts_a, type_a, ip_a),
                    "sensor_b_event": (ts_b, type_b, ip_a),
                    "ip":             ip_a,
                    "delta_seconds":  delta,
                }
            else:
                # Update if this sensor_a event is closer
                prev = best_match[b_key]
                if delta < prev["delta_seconds"] or \
                   (delta == prev["delta_seconds"] and ts_a < prev["sensor_a_event"][0]):
                    best_match[b_key]["sensor_a_event"] = (ts_a, type_a, ip_a)
                    best_match[b_key]["delta_seconds"]  = delta

    # Step 5: Sort output by sensor_a timestamp
    result = list(best_match.values())
    result.sort(key=lambda r: r["sensor_a_event"][0])
    return result


# ── MANUAL — no defaultdict, no bisect ────────────────────────────────────────

def _sort_events(events: list) -> list:
    """Manual insertion sort by timestamp (for small inputs)."""
    arr = list(events)
    for i in range(1, len(arr)):
        key = arr[i]
        j = i - 1
        while j >= 0 and arr[j][0] > key[0]:
            arr[j + 1] = arr[j]
            j -= 1
        arr[j + 1] = key
    return arr

def _manual_bisect_left(arr, val):
    lo, hi = 0, len(arr)
    while lo < hi:
        mid = lo + (hi - lo) // 2
        if arr[mid] < val:
            lo = mid + 1
        else:
            hi = mid
    return lo

def _manual_bisect_right(arr, val):
    lo, hi = 0, len(arr)
    while lo < hi:
        mid = lo + (hi - lo) // 2
        if arr[mid] <= val:
            lo = mid + 1
        else:
            hi = mid
    return lo

def correlate_closest_manual(sensor_a: list, sensor_b: list, T: int) -> list:
    """
    Same algorithm without defaultdict or bisect module.

    Time:  O((n+m)² for manual sort) — use proper sort for large input
    """
    if not sensor_a or not sensor_b:
        return []

    a_sorted = _sort_events(sensor_a)
    b_sorted = _sort_events(sensor_b)

    # Manual grouping
    b_groups = {}
    for ts_b, type_b, ip_b in b_sorted:
        if ip_b not in b_groups:
            b_groups[ip_b] = []
        b_groups[ip_b].append((ts_b, type_b))

    best_match = {}

    for ts_a, type_a, ip_a in a_sorted:
        if ip_a not in b_groups:
            continue
        b_events = b_groups[ip_a]
        b_ts = [e[0] for e in b_events]
        lo = _manual_bisect_left(b_ts, ts_a - T)
        hi = _manual_bisect_right(b_ts, ts_a + T)
        for idx in range(lo, hi):
            ts_b, type_b = b_events[idx]
            delta = abs(ts_a - ts_b) if ts_a >= ts_b else ts_b - ts_a
            b_key = (ts_b, type_b, ip_a)
            if b_key not in best_match:
                best_match[b_key] = {
                    "sensor_a_event": (ts_a, type_a, ip_a),
                    "sensor_b_event": (ts_b, type_b, ip_a),
                    "ip":             ip_a,
                    "delta_seconds":  delta,
                }
            else:
                prev_delta = best_match[b_key]["delta_seconds"]
                prev_ts_a  = best_match[b_key]["sensor_a_event"][0]
                if delta < prev_delta or (delta == prev_delta and ts_a < prev_ts_a):
                    best_match[b_key]["sensor_a_event"] = (ts_a, type_a, ip_a)
                    best_match[b_key]["delta_seconds"]  = delta

    result = list(best_match.values())
    result.sort(key=lambda r: r["sensor_a_event"][0])
    return result

if __name__ == "__main__":
    # Out-of-order inputs (the key twist)
    sensor_a = [
        (200, "login_fail", "2.2.2.2"),    # out of order vs (100,...)
        (100, "port_scan",  "1.1.1.1"),
        (400, "login_fail", "1.1.1.1"),
        (300, "port_scan",  "3.3.3.3"),
    ]
    sensor_b = [
        (395, "dns_query",  "1.1.1.1"),    # out of order
        (110, "dns_query",  "1.1.1.1"),
        (210, "exec_shell", "2.2.2.2"),
        (320, "http_req",   "3.3.3.3"),
        (500, "exec_shell", "1.1.1.1"),    # outside T=30 from both 100 and 400
    ]
    T = 30

    results = correlate_closest(sensor_a, sensor_b, T)

    print(f"Correlated pairs (T={T}s, out-of-order input, closest-match only):")
    for r in results:
        a, b = r["sensor_a_event"], r["sensor_b_event"]
        print(f"  IP={r['ip']}  A:[ts={a[0]},{a[1]}] ↔ B:[ts={b[0]},{b[1]}]  Δ={r['delta_seconds']}s")
    print()

    # Verify: sensor_b event (110, dns_query, 1.1.1.1) is closer to sensor_a ts=100 (delta=10)
    # than to sensor_a ts=400 (delta=290). So it should pair with ts=100.
    b_dns_110 = next((r for r in results
                      if r["sensor_b_event"] == (110, "dns_query", "1.1.1.1")), None)
    assert b_dns_110 is not None, "Correlation for sensor_b ts=110 not found"
    assert b_dns_110["sensor_a_event"][0] == 100, \
        f"sensor_b ts=110 should pair with sensor_a ts=100 (closest), got {b_dns_110['sensor_a_event'][0]}"
    assert b_dns_110["delta_seconds"] == 10

    # Verify sensor_b ts=395 pairs with sensor_a ts=400 (delta=5, closest)
    b_dns_395 = next((r for r in results
                      if r["sensor_b_event"] == (395, "dns_query", "1.1.1.1")), None)
    assert b_dns_395 is not None
    assert b_dns_395["sensor_a_event"][0] == 400
    assert b_dns_395["delta_seconds"] == 5

    # Output must be sorted by sensor_a timestamp
    a_ts = [r["sensor_a_event"][0] for r in results]
    assert a_ts == sorted(a_ts), "Output must be sorted by sensor_a timestamp"

    # All correlations respect T
    for r in results:
        assert r["delta_seconds"] <= T

    # Edge cases
    assert correlate_closest([], sensor_b, T) == []
    assert correlate_closest(sensor_a, [], T) == []
    # T=0: only exact timestamp match
    exact_a = [(100, "scan", "1.1.1.1")]
    exact_b = [(100, "dns", "1.1.1.1"), (101, "http", "1.1.1.1")]
    exact_res = correlate_closest(exact_a, exact_b, T=0)
    assert len(exact_res) == 1
    assert exact_res[0]["delta_seconds"] == 0

    # Manual agrees
    m_results = correlate_closest_manual(sensor_a, sensor_b, T)
    assert len(m_results) == len(results)

    # Naive agrees on count
    n_results = correlate_closest_naive(sensor_a, sensor_b, T)
    assert len(n_results) == len(results)

    print("Q5 — All assertions passed.")
    print()
    print("Twist 1: Out-of-order input → sort both sensors first O((n+m)log(n+m))")
    print("Twist 2: Closest match only → track best sensor_a per sensor_b event key")
    print(f"Total complexity: O((n+m) log(n+m) + n log m)")
