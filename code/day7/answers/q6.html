# Q6 (Gap Fixer) - Two-Pointer: All Pair Sums in Threat Scores
# Given a sorted list of threat scores and a target, find ALL pairs that sum
# to the target. Deduplicate. No hash map. O(n) two-pointer.
#
# ─── SOLUTION WALKTHROUGH ───────────────────────────────────────────────────
#
# STEP 1 — Restate
#   "I have a sorted list of threat scores (may contain duplicates). I need
#    every UNIQUE pair (a, b) where a <= b and a + b == target. Return all such
#    pairs. The list may contain duplicates — deduplicate the pair output."
#
# STEP 2 — Assumptions
#   - Input list is sorted ascending
#   - Pairs are value-based, not index-based: (1,7) is the same pair regardless
#     of which index the 1 and 7 come from
#   - a <= b in the returned pair (canonical order)
#   - Negative scores are allowed
#   - If the same value appears multiple times in the input, we still report
#     the pair at most once (e.g., [3,3,3] target=6 → [(3,3)])
#
# STEP 3 — Naive O(n²)
#   "Two nested loops: for each i, scan j = i+1..n-1 for arr[j] == target - arr[i].
#    Use a set to deduplicate pairs. O(n²) time, O(p) space (p=pairs found)."
#
# STEP 4 — Two-pointer O(n)
#   "lo=0, hi=n-1.
#    if arr[lo]+arr[hi] == target → found pair; skip duplicates at lo and hi
#    if sum < target → lo++
#    if sum > target → hi--
#    Terminate when lo >= hi."
#
# STEP 5 — Why deduplication?
#   "If arr=[1,1,7], target=8: both lo=0 and lo=1 pair with hi=2. We skip the
#    duplicate by incrementing lo past all equal values after recording the pair."
#
# STEP 6 — Edge cases
#   - No pairs found → []
#   - All elements the same: [3,3,3], target=6 → [(3,3)]
#   - Negative numbers: [-3,-1,2,5,6], target=3 → [(-3,6),(-1,4) if present]
#   - Single element → []
#   - Empty → []
#
# Complexity:
#   Naive:       O(n²) time, O(p) space
#   Two-pointer: O(n) time, O(p) space  ← target
# ─────────────────────────────────────────────────────────────────────────────


# ── APPROACH 1: Brute force O(n²) ────────────────────────────────────────────

def all_pairs_naive(scores: list, target: int) -> list:
    """
    Two nested loops + set for deduplication.

    Time:  O(n²)
    Space: O(p) where p = number of unique pairs
    """
    if not isinstance(scores, list):
        raise TypeError(f"Expected list, got {type(scores).__name__}")
    if not scores:
        return []

    seen = set()
    pairs = []
    n = len(scores)
    for i in range(n):
        for j in range(i + 1, n):
            if scores[i] + scores[j] == target:
                pair = (min(scores[i], scores[j]), max(scores[i], scores[j]))
                if pair not in seen:
                    seen.add(pair)
                    pairs.append(pair)
    return sorted(pairs)


# ── APPROACH 2: Two-pointer O(n) ─────────────────────────────────────────────

def all_pairs_two_pointer(scores: list, target: int) -> list:
    """
    Two-pointer on sorted array. Skip duplicates after recording each pair.

    Time:  O(n)
    Space: O(p)

    Assumes input is sorted ascending.
    """
    if not isinstance(scores, list):
        raise TypeError(f"Expected list, got {type(scores).__name__}")
    if not scores:
        return []
    if not all(scores[i] <= scores[i + 1] for i in range(len(scores) - 1)):
        raise ValueError("Input must be sorted ascending")

    pairs = []
    lo, hi = 0, len(scores) - 1

    while lo < hi:
        total = scores[lo] + scores[hi]

        if total == target:
            pairs.append((scores[lo], scores[hi]))

            # Skip duplicates at lo
            while lo < hi and scores[lo] == scores[lo + 1]:
                lo += 1
            # Skip duplicates at hi
            while lo < hi and scores[hi] == scores[hi - 1]:
                hi -= 1

            lo += 1
            hi -= 1

        elif total < target:
            lo += 1
        else:
            hi -= 1

    return pairs


# ── APPROACH 3: Using dict (one-pass hash map) — for comparison ───────────────

def all_pairs_hashmap(scores: list, target: int) -> list:
    """
    Hash map approach. O(n) time, O(n) space.
    Use this if input is NOT sorted. Shown for contrast.
    """
    if not isinstance(scores, list):
        raise TypeError(f"Expected list, got {type(scores).__name__}")

    seen_pairs = set()
    visited    = {}    # value → count seen so far
    pairs      = []

    for score in scores:
        complement = target - score
        if complement in visited:
            pair = (min(score, complement), max(score, complement))
            if pair not in seen_pairs:
                seen_pairs.add(pair)
                pairs.append(pair)
        visited[score] = visited.get(score, 0) + 1

    return sorted(pairs)


# ── MANUAL — no set, no sort (for manual deduplication) ──────────────────────

def all_pairs_manual(scores: list, target: int) -> list:
    """
    Two-pointer without using set or sort. Manual duplicate skip.
    Relies on input already being sorted.

    Time:  O(n)
    Space: O(p)
    """
    if not scores:
        return []

    pairs = []
    lo, hi = 0, len(scores) - 1

    while lo < hi:
        total = scores[lo] + scores[hi]

        if total == target:
            # Manual check: don't add if last pair has same values
            new_pair = (scores[lo], scores[hi])
            if not pairs or pairs[-1] != new_pair:
                pairs.append(new_pair)

            # Manual skip of duplicates
            lo_val, hi_val = scores[lo], scores[hi]
            while lo < hi and scores[lo] == lo_val:
                lo += 1
            while lo < hi and scores[hi] == hi_val:
                hi -= 1

        elif total < target:
            lo += 1
        else:
            hi -= 1

    return pairs


# ── Tests ─────────────────────────────────────────────────────────────────────

if __name__ == "__main__":
    # From the question
    scores1 = [1, 2, 3, 4, 5, 6, 7]
    target1 = 8
    expected1 = [(1, 7), (2, 6), (3, 5)]

    assert all_pairs_two_pointer(scores1, target1) == expected1
    assert all_pairs_naive(scores1, target1)       == expected1
    assert all_pairs_manual(scores1, target1)      == expected1
    assert all_pairs_hashmap(scores1, target1)     == expected1
    print(f"all_pairs({scores1}, target={target1}) = {all_pairs_two_pointer(scores1, target1)}")

    # Duplicates in input — dedup pairs
    scores2 = [1, 1, 7, 7]
    result2 = all_pairs_two_pointer(scores2, 8)
    assert result2 == [(1, 7)], f"Expected [(1,7)], got {result2}"

    # All same elements
    scores3 = [3, 3, 3]
    assert all_pairs_two_pointer(scores3, 6) == [(3, 3)]

    # No pairs
    assert all_pairs_two_pointer([1, 2, 3], 10) == []

    # Negative numbers
    scores4 = [-3, -1, 2, 4, 7]
    result4  = all_pairs_two_pointer(scores4, 4)
    assert (-3, 7) in result4
    assert (-1, 5) not in result4   # 5 not in array

    # Edge cases
    assert all_pairs_two_pointer([], 5)  == []
    assert all_pairs_two_pointer([5], 5) == []    # single element, lo >= hi

    # All approaches agree
    test_cases = [
        ([1, 2, 3, 4, 5, 6, 7], 8),
        ([1, 1, 7, 7],           8),
        ([3, 3, 3],              6),
        ([-3, -1, 2, 4, 7],     4),
        ([1, 2, 3],             10),
    ]
    for tc_scores, tc_target in test_cases:
        r_naive  = all_pairs_naive(tc_scores, tc_target)
        r_tp     = all_pairs_two_pointer(tc_scores, tc_target)
        r_manual = all_pairs_manual(tc_scores, tc_target)
        assert set(r_naive) == set(r_tp) == set(r_manual), \
            f"Mismatch on {tc_scores} t={tc_target}: {r_naive},{r_tp},{r_manual}"

    print("Q6 — All assertions passed.")
    print()
    print("Two-pointer deduplication:")
    print("  After recording pair (lo, hi):")
    print("  → skip right while scores[lo] == scores[lo+1], then lo++")
    print("  → skip left  while scores[hi] == scores[hi-1], then hi--")
    print()
    print("Complexity: O(n) time, O(p) space — no hash map needed on sorted input")
