# Q7 (Gap Fixer) - Stack: Evaluate Nested DENY/ALLOW Firewall Policies
# Parse firewall policy rules from a text format.
# Rules are grouped in PUSH/POP context blocks (nesting allowed).
# On POP, finalize the context — report rules and whether any traffic is allowed.
#
# ─── SOLUTION WALKTHROUGH ───────────────────────────────────────────────────
#
# STEP 1 — Restate
#   "Parse a sequence of firewall instructions using a stack:
#    - PUSH context=<name> → push new context onto stack
#    - ALLOW proto=<P> port=<N> → add ALLOW rule to current context
#    - DENY proto=<P> port=<N>  → add DENY rule to current context
#    - POP → finalize current context (evaluate if any traffic passes)
#   Output: list of finalized contexts with their rules and evaluation."
#
# STEP 2 — Assumptions
#   - Rules are evaluated first-match-wins (order matters)
#   - ALLOW takes precedence if it appears before DENY for same proto/port
#   - A context evaluates to 'ALLOW_SOME' if at least one ALLOW rule exists
#     before any matching DENY (conservative estimate without specific traffic)
#   - Nested contexts inherit no rules; each context is independent
#   - POP without a matching PUSH → error
#   - PUSH without a matching POP → warn, finalize remaining on done
#
# STEP 3 — Algorithm
#   Stack of context dicts: {'name': str, 'rules': [(action,proto,port), ...], 'depth': int}
#   - PUSH: push new context onto stack
#   - ALLOW/DENY: append rule to stack[-1]
#   - POP: pop context, finalize, append to results list
#   - At end: finalize any unclosed contexts (stack not empty → warn)
#
# STEP 4 — Evaluation logic
#   'evaluate_context(rules)':
#   - If no rules: IMPLICIT_DENY (default deny)
#   - Scan rules in order; first ALLOW → at least some traffic passes → ALLOW_SOME
#   - If only DENY rules → DENY_ALL
#   - If ALLOW before DENY → ALLOW_SOME
#   - If DENY before ALLOW → DENY_FIRST_ALLOW_REST
#
# STEP 5 — Stack invariant
#   "The stack guarantees correct nesting. I never need to scan for a matching
#    open — I just pop the top. A mismatch in name is a warning, not a crash."
#
# Complexity:
#   O(L) where L = number of lines in the policy text
# ─────────────────────────────────────────────────────────────────────────────


# ── PARSING HELPERS ───────────────────────────────────────────────────────────

def _parse_rule_line(line: str) -> dict | None:
    """
    Parse a single policy line. Returns a dict or None if unrecognised.

    Supported formats:
      PUSH context=<name>
      POP
      ALLOW proto=<proto> port=<port>
      DENY  proto=<proto> port=<port>
    """
    line = line.strip()
    if not line or line.startswith("#"):
        return None

    parts = line.split()
    if not parts:
        return None

    keyword = parts[0].upper()

    if keyword == "PUSH":
        # PUSH context=<name>
        for part in parts[1:]:
            if part.startswith("context="):
                return {"op": "PUSH", "context": part.split("=", 1)[1]}
        return {"op": "PUSH", "context": "unnamed"}

    if keyword == "POP":
        return {"op": "POP"}

    if keyword in ("ALLOW", "DENY"):
        kv = {}
        for part in parts[1:]:
            if "=" in part:
                k, v = part.split("=", 1)
                kv[k.lower()] = v
        return {
            "op":    keyword,
            "proto": kv.get("proto", "ANY"),
            "port":  kv.get("port", "ANY"),
        }

    return None


# ── CONTEXT EVALUATION ────────────────────────────────────────────────────────

def _evaluate_context(rules: list) -> str:
    """
    Evaluate ordered rules list → human-readable verdict.

    Rules are first-match-wins.
    Returns one of: IMPLICIT_DENY | DENY_ALL | ALLOW_SOME | MIXED
    """
    if not rules:
        return "IMPLICIT_DENY"

    has_allow = any(r["op"] == "ALLOW" for r in rules)
    has_deny  = any(r["op"] == "DENY"  for r in rules)

    if not has_allow:
        return "DENY_ALL"

    if not has_deny:
        return "ALLOW_ALL_LISTED"

    # Both present — check first occurrence
    for r in rules:
        if r["op"] == "ALLOW":
            return "ALLOW_SOME"    # first match is ALLOW
        if r["op"] == "DENY":
            return "DENY_FIRST_ALLOW_REST"   # first match is DENY

    return "MIXED"


# ── MAIN PARSER — single stack pass ──────────────────────────────────────────

def parse_policy(policy_text: str) -> list:
    """
    Parse the full policy text using a stack. Returns finalized contexts.

    Each finalized context:
    {
      'name':     str,
      'depth':    int,         # nesting depth (0 = top level)
      'rules':    list of {'op','proto','port'},
      'verdict':  str,         # result of evaluate_context
      'warnings': list of str,
    }

    Time:  O(L)  — single pass over L lines
    Space: O(D + R) — D=max depth, R=total rules
    """
    if not isinstance(policy_text, str):
        raise TypeError("policy_text must be a str")

    stack:   list = []   # active contexts (top = stack[-1])
    results: list = []
    warnings: list = []

    for lineno, raw_line in enumerate(policy_text.splitlines(), start=1):
        instr = _parse_rule_line(raw_line)
        if instr is None:
            continue

        op = instr["op"]

        if op == "PUSH":
            stack.append({
                "name":     instr["context"],
                "depth":    len(stack),
                "rules":    [],
                "warnings": [],
            })

        elif op == "POP":
            if not stack:
                warnings.append(f"Line {lineno}: POP with empty stack — ignored")
                continue
            ctx = stack.pop()
            ctx["verdict"] = _evaluate_context(ctx["rules"])
            results.append(ctx)

        elif op in ("ALLOW", "DENY"):
            if not stack:
                warnings.append(f"Line {lineno}: {op} outside any context — ignored")
                continue
            stack[-1]["rules"].append({
                "op":    op,
                "proto": instr["proto"],
                "port":  instr["port"],
            })

    # Unclosed contexts: finalize with warning
    while stack:
        ctx = stack.pop()
        ctx["warnings"].append("Context was never closed with POP")
        ctx["verdict"] = _evaluate_context(ctx["rules"])
        results.append(ctx)
        warnings.append(f"Unclosed context: '{ctx['name']}'")

    return results, warnings


# ── REPORT FORMATTER ──────────────────────────────────────────────────────────

def format_policy_report(results: list, warnings: list) -> str:
    """Format parsed policy into human-readable report."""
    lines = [
        "=" * 60,
        "  FIREWALL POLICY EVALUATION REPORT",
        "=" * 60,
        f"  Contexts evaluated: {len(results)}",
    ]
    if warnings:
        lines.append(f"  Warnings: {len(warnings)}")
    lines.append("")

    for ctx in results:
        indent = "  " * ctx["depth"]
        lines.append(f"{indent}[Context: {ctx['name']}] (depth={ctx['depth']})")
        lines.append(f"{indent}  Verdict: {ctx['verdict']}")
        if not ctx["rules"]:
            lines.append(f"{indent}  (no rules)")
        for r in ctx["rules"]:
            action_sym = "+" if r["op"] == "ALLOW" else "-"
            lines.append(f"{indent}  [{action_sym}] {r['op']:5} proto={r['proto']} port={r['port']}")
        if ctx.get("warnings"):
            for w in ctx["warnings"]:
                lines.append(f"{indent}  WARN: {w}")
        lines.append("")

    if warnings:
        lines.append("Global warnings:")
        for w in warnings:
            lines.append(f"  ! {w}")
        lines.append("")

    lines.append("=" * 60)
    return "\n".join(lines)


# ── Tests ─────────────────────────────────────────────────────────────────────

SAMPLE_POLICY = """
# Firewall policy for DMZ
PUSH context=DMZ
ALLOW proto=TCP port=443
ALLOW proto=TCP port=80
DENY  proto=TCP port=22
DENY  proto=UDP port=53

  PUSH context=DB-tier
  DENY  proto=TCP port=443
  DENY  proto=TCP port=80
  ALLOW proto=TCP port=5432
  POP

POP

PUSH context=Admin
ALLOW proto=TCP port=22
ALLOW proto=TCP port=443
POP

PUSH context=Monitoring
POP
"""

if __name__ == "__main__":
    results, warnings = parse_policy(SAMPLE_POLICY)
    report = format_policy_report(results, warnings)
    print(report)

    # Verify structure
    assert len(results) >= 3, f"Expected >= 3 contexts, got {len(results)}"

    # Find DMZ context
    dmz = next((r for r in results if r["name"] == "DMZ"), None)
    assert dmz is not None, "DMZ context not found"
    assert dmz["verdict"] == "ALLOW_SOME", f"DMZ: expected ALLOW_SOME, got {dmz['verdict']}"
    assert len(dmz["rules"]) == 4, f"DMZ should have 4 rules, got {len(dmz['rules'])}"
    assert dmz["rules"][0] == {"op": "ALLOW", "proto": "TCP", "port": "443"}

    # Find DB-tier context
    db = next((r for r in results if r["name"] == "DB-tier"), None)
    assert db is not None
    assert db["verdict"] == "DENY_FIRST_ALLOW_REST", f"DB-tier verdict: {db['verdict']}"
    assert db["depth"] == 1, "DB-tier should be at depth 1 (nested)"

    # Find Admin context
    admin = next((r for r in results if r["name"] == "Admin"), None)
    assert admin is not None
    assert admin["verdict"] == "ALLOW_ALL_LISTED"

    # Find Monitoring context (empty rules)
    mon = next((r for r in results if r["name"] == "Monitoring"), None)
    assert mon is not None
    assert mon["verdict"] == "IMPLICIT_DENY"

    # No global warnings in valid policy
    assert len(warnings) == 0, f"Unexpected warnings: {warnings}"

    # Test POP without PUSH
    bad_policy = "POP\nALLOW proto=TCP port=80\n"
    _, bad_warn = parse_policy(bad_policy)
    assert any("POP with empty stack" in w for w in bad_warn)
    assert any("outside any context" in w for w in bad_warn)

    # Test unclosed context
    unclosed = "PUSH context=X\nALLOW proto=TCP port=443\n"
    uc_results, uc_warn = parse_policy(unclosed)
    assert any("Unclosed context" in w for w in uc_warn)
    assert uc_results[0]["verdict"] == "ALLOW_ALL_LISTED"

    print("Q7 — All assertions passed.")
    print()
    print("Stack invariant: PUSH adds to top, POP finalizes top → O(L) per pass")
    print("evaluate_context: first-match-wins scan of ordered rules → O(R)")
    print("Overall complexity: O(L) where L = number of policy lines")
