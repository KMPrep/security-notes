# Q9 (Gap Fixer) - LRU Cache from Scratch
# Implement an LRU cache supporting get(key) and put(key, value).
# Evict the Least Recently Used entry when capacity is exceeded.
# Show: naive dict+list (O(n) moves), then O(1) with OrderedDict.
#
# ─── SOLUTION WALKTHROUGH ───────────────────────────────────────────────────
#
# STEP 1 — Restate
#   "Implement an LRU (Least Recently Used) cache with:
#    - get(key): return value if exists, else -1. Accessing a key makes it MRU.
#    - put(key, value): insert or update. If at capacity, evict LRU first.
#    Both operations must be O(1) in the optimal solution."
#
# STEP 2 — Data structures
#   Naive: dict + list  (dict for O(1) lookup, list for ordering)
#     - get: O(n) remove + append
#     - put: O(n) capacity check + possible eviction
#
#   Optimal: OrderedDict (Python 3.7+)
#     - move_to_end(key): O(1) — moves to MRU position
#     - popitem(last=False): O(1) — evicts LRU (front of ordered dict)
#     - get + put: O(1) each
#
#   Manual O(1): doubly-linked list + hash map
#     - hash map: key → node
#     - doubly-linked list: maintains insertion/access order
#     - Head sentinel: LRU end; Tail sentinel: MRU end
#
# STEP 3 — Gotchas
#   "In Python, dict is ordered since 3.7, but LRU requires explicit order
#    tracking. OrderedDict is the right tool. The move_to_end(key) call is
#    what makes it LRU rather than just an ordered dict."
#
# STEP 4 — Security relevance
#   "LRU caches appear in:
#    - DNS resolvers (cache most-recently-queried records)
#    - Auth token validation (cache recent JWTs to avoid DB round-trip)
#    - Web application WAFs (cache recent IP reputation lookups)
#    If capacity is mis-tuned, hot entries get evicted → performance cliff"
#
# Complexity:
#   Naive:        get O(n), put O(n)
#   OrderedDict:  get O(1), put O(1)
#   Manual DLL:   get O(1), put O(1)
# ─────────────────────────────────────────────────────────────────────────────

from collections import OrderedDict


# ── APPROACH 1: Naive — dict + list (O(n) moves) ─────────────────────────────

class LRUCacheNaive:
    """
    LRU cache using a dict (for O(1) lookups) and a list (for ordering).
    The list stores keys in LRU→MRU order (index 0 = LRU, end = MRU).

    get: O(n)  — remove key from list + append
    put: O(n)  — evict from front if needed
    """

    def __init__(self, capacity: int):
        if capacity <= 0:
            raise ValueError(f"Capacity must be > 0, got {capacity}")
        self.capacity = capacity
        self.cache: dict = {}    # key → value
        self.order: list = []    # LRU at index 0, MRU at end

    def get(self, key) -> int:
        if key not in self.cache:
            return -1
        self.order.remove(key)    # O(n)
        self.order.append(key)    # MRU
        return self.cache[key]

    def put(self, key, value) -> None:
        if key in self.cache:
            self.order.remove(key)    # O(n)
        elif len(self.cache) >= self.capacity:
            lru_key = self.order.pop(0)    # O(n)
            del self.cache[lru_key]
        self.cache[key] = value
        self.order.append(key)

    def __repr__(self):
        return f"LRUCacheNaive(cap={self.capacity}, order={self.order}, cache={self.cache})"


# ── APPROACH 2: OrderedDict — O(1) ────────────────────────────────────────────

class LRUCache:
    """
    LRU cache using collections.OrderedDict.

    OrderedDict remembers insertion order.
    move_to_end(key)    → O(1) — promote to MRU (end)
    popitem(last=False) → O(1) — evict LRU (front)

    get: O(1)
    put: O(1)
    """

    def __init__(self, capacity: int):
        if capacity <= 0:
            raise ValueError(f"Capacity must be > 0, got {capacity}")
        self.capacity = capacity
        self._od: OrderedDict = OrderedDict()

    def get(self, key) -> int:
        if key not in self._od:
            return -1
        self._od.move_to_end(key)    # promote to MRU
        return self._od[key]

    def put(self, key, value) -> None:
        if key in self._od:
            self._od.move_to_end(key)   # update existing: promote to MRU
        self._od[key] = value
        if len(self._od) > self.capacity:
            self._od.popitem(last=False)  # evict LRU (front)

    def __repr__(self):
        items = list(self._od.items())
        return f"LRUCache(cap={self.capacity}, LRU→MRU={items})"


# ── APPROACH 3: Manual doubly-linked list + hash map (O(1)) ──────────────────

class _Node:
    __slots__ = ("key", "val", "prev", "next")
    def __init__(self, key=None, val=None):
        self.key  = key
        self.val  = val
        self.prev = None
        self.next = None


class LRUCacheManual:
    """
    O(1) LRU cache using a doubly-linked list + hash map.

    List layout: head(sentinel) ↔ LRU ↔ ... ↔ MRU ↔ tail(sentinel)
    - Insert at tail.prev → MRU position
    - Remove from head.next → LRU eviction
    - Hash map: key → node (for O(1) access)

    get: O(1)
    put: O(1)
    """

    def __init__(self, capacity: int):
        if capacity <= 0:
            raise ValueError(f"Capacity must be > 0, got {capacity}")
        self.capacity = capacity
        self.map:  dict  = {}

        # Sentinel nodes — never contain real data
        self.head = _Node()    # LRU side
        self.tail = _Node()    # MRU side
        self.head.next = self.tail
        self.tail.prev = self.head

    def _remove(self, node: _Node) -> None:
        """Unlink node from doubly-linked list."""
        node.prev.next = node.next
        node.next.prev = node.prev

    def _insert_before_tail(self, node: _Node) -> None:
        """Insert node just before tail (MRU position)."""
        node.prev       = self.tail.prev
        node.next       = self.tail
        self.tail.prev.next = node
        self.tail.prev  = node

    def get(self, key) -> int:
        if key not in self.map:
            return -1
        node = self.map[key]
        self._remove(node)
        self._insert_before_tail(node)   # promote to MRU
        return node.val

    def put(self, key, value) -> None:
        if key in self.map:
            node = self.map[key]
            node.val = value
            self._remove(node)
            self._insert_before_tail(node)
        else:
            if len(self.map) >= self.capacity:
                lru = self.head.next    # LRU is first real node
                self._remove(lru)
                del self.map[lru.key]
            new_node = _Node(key, value)
            self._insert_before_tail(new_node)
            self.map[key] = new_node

    def __repr__(self):
        items = []
        cur = self.head.next
        while cur is not self.tail:
            items.append((cur.key, cur.val))
            cur = cur.next
        return f"LRUCacheManual(cap={self.capacity}, LRU→MRU={items})"


# ── Tests ─────────────────────────────────────────────────────────────────────

def _run_test_sequence(cache):
    """
    Standard test sequence for capacity=3:
      put(1,1), put(2,2), get(1), put(3,3), get(2), put(4,4),
      get(1), get(3), get(4)

    After put(1,1):     cache={1:1}
    After put(2,2):     cache={1:1, 2:2}
    After get(1):       cache={2:2, 1:1}   → 1 promoted to MRU
    After put(3,3):     cache={2:2, 1:1, 3:3}
    After get(2):       cache={1:1, 3:3, 2:2}  → 2 promoted
    After put(4,4):     evict LRU=1; cache={3:3, 2:2, 4:4}
    get(1) → -1  (evicted)
    get(3) → 3
    get(4) → 4
    """
    cache.put(1, 1)
    cache.put(2, 2)
    assert cache.get(1) == 1            # hit; 1 promoted to MRU

    cache.put(3, 3)
    assert cache.get(2) == 2            # hit; 2 promoted to MRU; now LRU=3

    cache.put(4, 4)                     # evict LRU=3
    assert cache.get(1) == -1           # 1 was evicted before put(4)
    assert cache.get(3) == 3            # 3 is still present (2 was promoted before put(4), so 1 is LRU)
    assert cache.get(4) == 4


if __name__ == "__main__":
    # Run with all three implementations
    for CacheClass in (LRUCacheNaive, LRUCache, LRUCacheManual):
        c = CacheClass(capacity=3)
        _run_test_sequence(c)
        print(f"{CacheClass.__name__}: test sequence passed")

    # Additional tests
    c2 = LRUCache(capacity=2)
    c2.put(1, 1)
    c2.put(2, 2)
    assert c2.get(1) == 1    # 1 → MRU
    c2.put(3, 3)             # evict 2 (LRU)
    assert c2.get(2) == -1   # evicted
    assert c2.get(3) == 3
    assert c2.get(1) == 1

    # Update existing key does not grow beyond capacity
    c3 = LRUCache(capacity=2)
    c3.put(1, 1)
    c3.put(2, 2)
    c3.put(1, 10)            # update; should not evict anything
    assert c3.get(1) == 10
    assert c3.get(2) == 2

    # Capacity 1
    c4 = LRUCacheManual(capacity=1)
    c4.put(1, 1)
    assert c4.get(1) == 1
    c4.put(2, 2)             # evict 1
    assert c4.get(1) == -1
    assert c4.get(2) == 2

    print("\nQ9 — All assertions passed.")
