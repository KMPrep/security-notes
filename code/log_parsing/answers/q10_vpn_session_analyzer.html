"""
Q10. VPN Session Log Analyzer — Impossible Travel + Exfil
===========================================================
Detect impossible travel, after-hours access, data exfiltration,
and Tor exit node usage from VPN session logs.

=============================================================================
SOLUTION PROGRESSION
=============================================================================

NAIVE: Flag sessions outside business hours.
  Problem: Misses the most interesting patterns — impossible travel (same
  user in two countries within hours) and large data exfil.

BETTER: Sort sessions per user by timestamp; compare consecutive countries.
  Problem: Concurrent sessions for the same user from different regions are
  legitimate in some enterprises (split-tunnelling, multi-device). Need to
  check the OVERLAP between sessions, not just order. Also needs actual
  travel-time estimation — two hours for US→Russia is physically impossible.

GOOGLE-LEVEL (this file):
  - Impossible travel: overlapping or temporally close sessions from distinct
    country codes. Use a min-travel-time table (simplified) for realism.
  - After-hours: configurable time window check on session start.
  - Exfil: bytes_out > threshold in a single session.
  - Tor: check src_ip against a provided list (simulated here inline).
  - Per-user risk timeline output for analyst triage.

Complexity: O(N log N) sort per user; O(U × S) where U = users, S = sessions.
=============================================================================
"""

import re
import sys
from collections import defaultdict
from datetime import datetime, timezone

# ─────────────────────────────────────────────────────────────────────────────
# SAMPLE DATA
# ─────────────────────────────────────────────────────────────────────────────

SAMPLE_LOG = """\

# Simulated Tor exit node list (in practice, loaded from a threat feed)
TOR_EXIT_IPS = {
    "185.220.101.1",
    "185.220.101.5",
    "185.220.101.34",
    "198.96.155.3",
    "46.165.230.5",
}

# Simplified minimum travel time in hours between countries (symmetric)
# In production, use a geolocation DB and great-circle distance / max speed of travel
MIN_TRAVEL_HOURS: dict[frozenset, float] = {
    frozenset({"US", "RU"}): 10.0,
    frozenset({"US", "CN"}): 12.0,
    frozenset({"US", "AU"}): 20.0,
    frozenset({"US", "DE"}): 8.0,
    frozenset({"US", "JP"}): 13.0,
    frozenset({"DE", "JP"}): 11.0,
    frozenset({"RU", "JP"}): 9.0,
}
DEFAULT_MIN_TRAVEL_HOURS = 1.0  # same-country or unknown pair

EXFIL_BYTES_THRESHOLD   = 100_000_000  # 100 MB
AFTER_HOURS_START_UTC   = 22   # 22:00 UTC
AFTER_HOURS_END_UTC     = 6    # 06:00 UTC


# ─────────────────────────────────────────────────────────────────────────────
# PARSER
# ─────────────────────────────────────────────────────────────────────────────

LOG_RE = re.compile(
    r'(?P<ts>\S+)\s+'
    r'user=(?P<user>\S+)\s+'
    r'src_ip=(?P<src_ip>\S+)\s+'
    r'country=(?P<country>\S+)\s+'
    r'bytes_in=(?P<bytes_in>\d+)\s+'
    r'bytes_out=(?P<bytes_out>\d+)\s+'
    r'event=(?P<event>\S+)\s+'
    r'session_id=(?P<session_id>\S+)'
)


def parse_ts(ts_str: str) -> float | None:
    try:
        return datetime.strptime(ts_str, "%Y-%m-%dT%H:%M:%SZ").replace(tzinfo=timezone.utc).timestamp()
    except ValueError:
        return None


def parse_line(line: str) -> dict | None:
    line = line.strip()
    if not line:
        return None
    m = LOG_RE.match(line)
    if not m:
        return None
    ts = parse_ts(m.group("ts"))
    if ts is None:
        return None
    return {
        "ts":         ts,
        "user":       m.group("user"),
        "src_ip":     m.group("src_ip"),
        "country":    m.group("country"),
        "bytes_in":   int(m.group("bytes_in")),
        "bytes_out":  int(m.group("bytes_out")),
        "event":      m.group("event"),
        "session_id": m.group("session_id"),
    }


# ─────────────────────────────────────────────────────────────────────────────
# DETECTORS
# ─────────────────────────────────────────────────────────────────────────────

def min_travel_hours(c1: str, c2: str) -> float:
    key = frozenset({c1, c2})
    return MIN_TRAVEL_HOURS.get(key, DEFAULT_MIN_TRAVEL_HOURS) if c1 != c2 else float("inf")


def detect_impossible_travel(user_sessions: list[dict]) -> list[dict]:
    """
    For each pair of sessions from distinct countries, check if the time gap
    is less than the minimum feasible travel time.
    """
    alerts = []
    sessions = sorted(user_sessions, key=lambda x: x["ts"])
    for i in range(len(sessions)):
        for j in range(i + 1, len(sessions)):
            s1, s2 = sessions[i], sessions[j]
            if s1["country"] == s2["country"]:
                continue
            gap_hours = (s2["ts"] - s1["ts"]) / 3600.0
            req_hours = min_travel_hours(s1["country"], s2["country"])
            if gap_hours < req_hours:
                alerts.append({
                    "type":       "IMPOSSIBLE_TRAVEL",
                    "risk":       "CRITICAL",
                    "session1":   s1["session_id"],
                    "session2":   s2["session_id"],
                    "country1":   s1["country"],
                    "country2":   s2["country"],
                    "gap_hours":  round(gap_hours, 2),
                    "min_travel": req_hours,
                    "ip1":        s1["src_ip"],
                    "ip2":        s2["src_ip"],
                })
    return alerts


def detect_after_hours(sessions: list[dict]) -> list[dict]:
    alerts = []
    for s in sessions:
        hour = datetime.utcfromtimestamp(s["ts"]).hour
        if hour >= AFTER_HOURS_START_UTC or hour < AFTER_HOURS_END_UTC:
            alerts.append({
                "type":       "AFTER_HOURS_ACCESS",
                "risk":       "MEDIUM",
                "session_id": s["session_id"],
                "user":       s["user"],
                "src_ip":     s["src_ip"],
                "country":    s["country"],
                "hour_utc":   hour,
                "ts":         s["ts"],
            })
    return alerts


def detect_exfil(sessions: list[dict]) -> list[dict]:
    return [
        {
            "type":       "DATA_EXFIL",
            "risk":       "HIGH",
            "session_id": s["session_id"],
            "user":       s["user"],
            "src_ip":     s["src_ip"],
            "bytes_out":  s["bytes_out"],
            "ts":         s["ts"],
        }
        for s in sessions
        if s["bytes_out"] > EXFIL_BYTES_THRESHOLD
    ]


def detect_tor(sessions: list[dict]) -> list[dict]:
    return [
        {
            "type":       "TOR_EXIT_NODE",
            "risk":       "HIGH",
            "session_id": s["session_id"],
            "user":       s["user"],
            "src_ip":     s["src_ip"],
            "ts":         s["ts"],
        }
        for s in sessions
        if s["src_ip"] in TOR_EXIT_IPS
    ]


# ─────────────────────────────────────────────────────────────────────────────
# AGGREGATOR
# ─────────────────────────────────────────────────────────────────────────────

def analyze_vpn_logs(records: list[dict]) -> dict[str, list[dict]]:
    user_sessions: dict[str, list[dict]] = defaultdict(list)
    for r in records:
        user_sessions[r["user"]].append(r)

    user_findings: dict[str, list[dict]] = defaultdict(list)
    for user, sessions in user_sessions.items():
        user_findings[user].extend(detect_impossible_travel(sessions))
        user_findings[user].extend(detect_after_hours(sessions))
        user_findings[user].extend(detect_exfil(sessions))
        user_findings[user].extend(detect_tor(sessions))

    return dict(user_findings)


# ─────────────────────────────────────────────────────────────────────────────
# REPORT
# ─────────────────────────────────────────────────────────────────────────────

RISK_ORDER = {"CRITICAL": 0, "HIGH": 1, "MEDIUM": 2, "LOW": 3}


def ts_fmt(ts: float) -> str:
    return datetime.utcfromtimestamp(ts).strftime("%Y-%m-%dT%H:%M:%SZ")


def print_report(user_findings: dict) -> None:
    print("=" * 65)
    print("VPN SESSION ANALYSIS REPORT")
    print("=" * 65)

    for user in sorted(user_findings):
        findings = sorted(user_findings[user], key=lambda x: RISK_ORDER.get(x["risk"], 9))
        if not findings:
            continue
        max_risk = findings[0]["risk"]
        print(f"\n  User: {user}  max_risk={max_risk}  findings={len(findings)}")
        for f in findings:
            print(f"    [{f['risk']}] {f['type']}")
            for k, v in f.items():
                if k in ("type", "risk", "user"):
                    continue
                if isinstance(v, float) and v > 1e9:
                    print(f"      {k}={ts_fmt(v)}")
                elif k == "bytes_out":
                    print(f"      {k}={v/1_000_000:.1f}MB")
                else:
                    print(f"      {k}={v}")
    print("=" * 65)


# ─────────────────────────────────────────────────────────────────────────────
# MAIN
# ─────────────────────────────────────────────────────────────────────────────

if __name__ == "__main__":
    if len(sys.argv) > 1:
        with open(sys.argv[1], encoding="utf-8", errors="replace") as fh:
            raw_lines = fh.readlines()
    else:
        print("[INFO] Using embedded sample data.\n")
        raw_lines = SAMPLE_LOG.splitlines(keepends=True)

    records  = [r for r in (parse_line(l) for l in raw_lines) if r is not None]
    findings = analyze_vpn_logs(records)
    total    = sum(len(v) for v in findings.values())
    print(f"Parsed {len(records)} VPN session events.  "
          f"Found {total} finding(s) across {len(findings)} user(s).\n")
    print_report(findings)
