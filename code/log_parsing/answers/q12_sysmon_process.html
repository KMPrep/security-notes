"""
Q12. Sysmon Process Creation Log Parser (Event ID 1)
=====================================================
Detect LOLBin abuse, unusual parent-child chains, and obfuscated
command lines from Sysmon process creation events.

=============================================================================
SOLUTION PROGRESSION
=============================================================================

NAIVE: Flag powershell.exe in process creation logs.
  Problem: powershell.exe is used legitimately everywhere. Without parent
  context, flagging it produces massive noise. Also misses LOLBins like
  certutil, regsvr32, mshta which are rarely discussed.

BETTER: Check parent process against a block-list of office apps.
  Problem: Need to also detect LOLBin misuse based on arguments (not just
  presence), long command lines (obfuscation), and 3-level chains
  (grandparent → parent → child) for full attack chain reconstruction.

GOOGLE-LEVEL (this file):
  - LOLBin detection: binary name + suspicious argument pattern.
  - Parent-child: office/browser → shell/scripting engine.
  - Grandparent chain: track parent-child relationships by PID/PPID.
  - Long command line: > 500 chars often indicates obfuscated one-liners.
  - Output shows full chain with risk tags.

Complexity: O(N) build + O(N) walk chain per flagged process.
=============================================================================
"""

# ─────────────────────────────────────────────────────────────────────────────
# CONFIG
# ─────────────────────────────────────────────────────────────────────────────

LONG_CMDLINE_THRESHOLD = 500

# LOLBins and their suspicious argument patterns
LOLBIN_PATTERNS: list[tuple[str, re.Pattern, str]] = [
    ("certutil.exe",  re.compile(r'-decode|-encode|-urlcache|-f\s+http', re.IGNORECASE), "certutil decode/download"),
    ("regsvr32.exe",  re.compile(r'/i:http|/i:ftp|scrobj\.dll|/s\s+/n\s+/u', re.IGNORECASE), "regsvr32 COM scriptlet"),
    ("mshta.exe",     re.compile(r'http|javascript:|vbscript:', re.IGNORECASE), "mshta remote/script execution"),
    ("wscript.exe",   re.compile(r'\.js|\.vbs|\.hta|http', re.IGNORECASE), "wscript script execution"),
    ("cscript.exe",   re.compile(r'\.js|\.vbs|\.hta|http', re.IGNORECASE), "cscript script execution"),
    ("rundll32.exe",  re.compile(r'javascript:|shell32|shell\.dll|powershell', re.IGNORECASE), "rundll32 abuse"),
    ("msiexec.exe",   re.compile(r'/i\s+http|/q\s+http|\.msi.*http', re.IGNORECASE), "msiexec remote install"),
    ("powershell.exe",re.compile(r'-enc\s+|-encodedcommand\s+|-nop\s+.*-exec\s+bypass|-w\s+hidden', re.IGNORECASE), "powershell encoded/hidden"),
]

# Office/browser processes that should NOT spawn shells
SPAWNER_BLOCKLIST = {
    "winword.exe", "excel.exe", "powerpnt.exe", "outlook.exe",
    "onenote.exe", "chrome.exe", "firefox.exe", "iexplore.exe", "msedge.exe",
}

# Shell/scripting engines that should not appear as children of the above
SHELL_CHILDREN = {
    "cmd.exe", "powershell.exe", "wscript.exe", "cscript.exe",
    "mshta.exe", "wmic.exe", "regsvr32.exe", "rundll32.exe",
}


# ─────────────────────────────────────────────────────────────────────────────
# PARSER
# ─────────────────────────────────────────────────────────────────────────────

def parse_line(line: str, header: list[str]) -> dict | None:
    line = line.strip()
    if not line:
        return None
    parts = line.split(",", len(header) - 1)
    if len(parts) != len(header):
        return None
    row = dict(zip(header, parts))
    try:
        ts = datetime.strptime(row["TimeCreated"], "%Y-%m-%dT%H:%M:%SZ").timestamp()
        pid  = int(row["ProcessId"])
        ppid = int(row["ParentProcessId"])
    except (ValueError, KeyError):
        return None
    image_name  = row["Image"].replace("\\", "/").split("/")[-1].lower()
    parent_name = row["ParentImage"].replace("\\", "/").split("/")[-1].lower()
    return {
        "ts":           ts,
        "computer":     row["ComputerName"],
        "user":         row["UserName"],
        "pid":          pid,
        "ppid":         ppid,
        "image":        row["Image"],
        "image_name":   image_name,
        "parent_image": row["ParentImage"],
        "parent_name":  parent_name,
        "cmdline":      row["CommandLine"],
    }


# ─────────────────────────────────────────────────────────────────────────────
# DETECTORS
# ─────────────────────────────────────────────────────────────────────────────

def detect_lolbin(records: list[dict]) -> list[dict]:
    alerts = []
    for r in records:
        for binary, pattern, description in LOLBIN_PATTERNS:
            if r["image_name"] == binary and pattern.search(r["cmdline"]):
                alerts.append({
                    "type":        "LOLBIN_ABUSE",
                    "binary":      binary,
                    "description": description,
                    "user":        r["user"],
                    "computer":    r["computer"],
                    "pid":         r["pid"],
                    "cmdline":     r["cmdline"],
                    "ts":          r["ts"],
                })
    return alerts


def detect_malicious_parent_child(records: list[dict]) -> list[dict]:
    alerts = []
    for r in records:
        if r["parent_name"] in SPAWNER_BLOCKLIST and r["image_name"] in SHELL_CHILDREN:
            alerts.append({
                "type":         "MALICIOUS_PARENT_CHILD",
                "parent":       r["parent_name"],
                "child":        r["image_name"],
                "user":         r["user"],
                "computer":     r["computer"],
                "ppid":         r["ppid"],
                "pid":          r["pid"],
                "cmdline":      r["cmdline"],
                "ts":           r["ts"],
            })
    return alerts


def detect_long_cmdline(records: list[dict]) -> list[dict]:
    return [
        {
            "type":     "LONG_CMDLINE",
            "length":   len(r["cmdline"]),
            "image":    r["image_name"],
            "user":     r["user"],
            "computer": r["computer"],
            "pid":      r["pid"],
            "cmdline":  r["cmdline"][:200] + "...",
            "ts":       r["ts"],
        }
        for r in records
        if len(r["cmdline"]) > LONG_CMDLINE_THRESHOLD
    ]


def build_chain(pid: int, pid_map: dict[int, dict], depth: int = 3) -> list[str]:
    """Walk the process tree upward to reconstruct the chain."""
    chain = []
    current_pid = pid
    for _ in range(depth):
        proc = pid_map.get(current_pid)
        if not proc:
            break
        chain.append(proc["image_name"])
        current_pid = proc["ppid"]
    return list(reversed(chain))


# ─────────────────────────────────────────────────────────────────────────────
# REPORT
# ─────────────────────────────────────────────────────────────────────────────

def ts_fmt(ts: float) -> str:
    return datetime.utcfromtimestamp(ts).strftime("%Y-%m-%dT%H:%M:%SZ")


def print_report(lolbin, parent_child, long_cmd, pid_map) -> None:
    print("=" * 65)
    print("SYSMON PROCESS CREATION — THREAT DETECTION REPORT")
    print("=" * 65)

    print(f"\n[LOLBIN_ABUSE] {len(lolbin)} alert(s)")
    for a in lolbin:
        chain = build_chain(a["pid"], pid_map)
        print(f"  {a['binary']} ({a['description']})")
        print(f"  user={a['user']}  host={a['computer']}  ts={ts_fmt(a['ts'])}")
        print(f"  chain: {' → '.join(chain)}")
        print(f"  cmd: {a['cmdline'][:100]}")

    print(f"\n[MALICIOUS_PARENT_CHILD] {len(parent_child)} alert(s)")
    for a in parent_child:
        chain = build_chain(a["pid"], pid_map)
        print(f"  {a['parent']} → {a['child']}")
        print(f"  user={a['user']}  host={a['computer']}  ts={ts_fmt(a['ts'])}")
        print(f"  chain: {' → '.join(chain)}")
        print(f"  cmd: {a['cmdline'][:100]}")

    print(f"\n[LONG_CMDLINE] {len(long_cmd)} alert(s)")
    for a in long_cmd:
        print(f"  {a['image']}  len={a['length']}  user={a['user']}  ts={ts_fmt(a['ts'])}")
        print(f"  cmd: {a['cmdline'][:120]}")
    print("=" * 65)

if __name__ == "__main__":
    if len(sys.argv) > 1:
        with open(sys.argv[1], encoding="utf-8", errors="replace") as fh:
            raw_lines = fh.readlines()
    else:
        print("[INFO] Using embedded sample data.\n")
        raw_lines = SAMPLE_LOG.splitlines(keepends=True)

    # First line is the header
    lines = [l.strip() for l in raw_lines if l.strip()]
    if not lines:
        print("Empty input.")
        sys.exit(1)

    header_line = lines[0]
    header = header_line.split(",", 8)  # 9 columns max

    records = [r for r in (parse_line(l, header) for l in lines[1:]) if r is not None]
    pid_map = {r["pid"]: r for r in records}

    print(f"Parsed {len(records)} Sysmon process creation events.\n")
    print_report(
        detect_lolbin(records),
        detect_malicious_parent_child(records),
        detect_long_cmdline(records),
        pid_map,
    )
