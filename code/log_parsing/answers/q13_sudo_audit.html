"""
Q13. sudo / auditd Command Audit Parser
=========================================
Detect dangerous commands, privilege escalation chains, off-hours activity,
and failed sudo followed by success from audit logs.

=============================================================================
SOLUTION PROGRESSION
=============================================================================

NAIVE: Grep for "sudo" and list commands.
  Problem: Every sudo invocation would be listed. No detection logic,
  no dangerous command patterns, no context.

BETTER: Flag commands containing "curl" or "wget".
  Problem: `curl https://updates.example.com/check` is legitimate.
  Need to detect the dangerous pattern: curl/wget piped to shell execution
  or combined with base64 decode.

GOOGLE-LEVEL (this file):
  - Regex-based dangerous command patterns with justification for each.
  - Privilege escalation chain: sudo → su → sudo within a TTY session.
  - Off-hours detection with configurable window.
  - Failed sudo → success within 60s: possible password enumeration.
  - TTY-aware: track sessions per TTY to reconstruct chains.

Complexity: O(N) parse, O(T × C) where T = TTYs, C = commands per TTY.
=============================================================================
"""

import re
import sys
from collections import defaultdict
from datetime import datetime, time as dtime

# ─────────────────────────────────────────────────────────────────────────────
# SAMPLE DATA
# ─────────────────────────────────────────────────────────────────────────────

SAMPLE_LOG = """\

# ─────────────────────────────────────────────────────────────────────────────
# CONFIG
# ─────────────────────────────────────────────────────────────────────────────

BUSINESS_HOURS_START = dtime(9, 0)
BUSINESS_HOURS_END   = dtime(18, 0)
FAILED_SUCCESS_WINDOW_SECS = 60
MONTH_MAP = {"Jan":1,"Feb":2,"Mar":3,"Apr":4,"May":5,"Jun":6,
             "Jul":7,"Aug":8,"Sep":9,"Oct":10,"Nov":11,"Dec":12}
REFERENCE_YEAR = 2024

# Dangerous command patterns with descriptions
DANGEROUS_PATTERNS: list[tuple[str, re.Pattern, str]] = [
    ("CHMOD_WORLD_WRITE",
     re.compile(r'chmod\s+(777|o\+w)', re.IGNORECASE),
     "world-writable permission grant"),
    ("SETUID_BIT",
     re.compile(r'chmod\s+[+]s|chmod\s+[0-9]*[246][0-9]{3}', re.IGNORECASE),
     "setuid/setgid bit set — privilege escalation vector"),
    ("DD_DISK_DUMP",
     re.compile(r'\bdd\b.*if=/dev/', re.IGNORECASE),
     "raw disk dump — potential data exfil"),
    ("DOWNLOAD_EXEC",
     re.compile(r'(wget|curl).*http.*[|&]\s*(bash|sh|python|perl)', re.IGNORECASE),
     "download and execute — common dropper pattern"),
    ("BASE64_DECODE_EXEC",
     re.compile(r'base64\s+-d.*[|&]\s*(bash|sh)', re.IGNORECASE),
     "base64 decode piped to shell — obfuscated payload"),
    ("IPTABLES_FLUSH",
     re.compile(r'iptables\s+-F', re.IGNORECASE),
     "firewall rule flush — defence evasion"),
    ("SU_COMMAND",
     re.compile(r'(?:^|/)su\b\s+\w+', re.IGNORECASE),
     "user substitution — possible privilege chain"),
    ("SHELL_SPAWN",
     re.compile(r'(?:^|/)(bash|sh|zsh|fish|dash)\s*$', re.IGNORECASE),
     "interactive shell spawned as root"),
]

# sudo success line
SUDO_SUCCESS_RE = re.compile(
    r'(?P<month>\w{3})\s+(?P<day>\d+)\s+(?P<time>\d{2}:\d{2}:\d{2})\s+'
    r'\S+\s+sudo:\s+'
    r'(?P<user>\S+)\s+:\s+TTY=(?P<tty>\S+)\s+;\s+PWD=(?P<pwd>\S+)\s+;\s+'
    r'USER=(?P<runas>\S+)\s+;\s+COMMAND=(?P<cmd>.+)'
)

# sudo failure line
SUDO_FAIL_RE = re.compile(
    r'(?P<month>\w{3})\s+(?P<day>\d+)\s+(?P<time>\d{2}:\d{2}:\d{2})\s+'
    r'\S+\s+sudo.*authentication failure.*tty=(?P<tty>\S+).*user=(?P<user>\S+)',
    re.IGNORECASE,
)


def parse_ts(month: str, day: str, time_str: str) -> float:
    return datetime(REFERENCE_YEAR, MONTH_MAP[month], int(day),
                    *map(int, time_str.split(":"))).timestamp()


def parse_line(line: str) -> dict | None:
    line = line.strip()
    if not line:
        return None

    # Try success
    m = SUDO_SUCCESS_RE.match(line)
    if m:
        return {
            "event":  "success",
            "ts":     parse_ts(m.group("month"), m.group("day"), m.group("time")),
            "user":   m.group("user"),
            "tty":    m.group("tty"),
            "runas":  m.group("runas"),
            "cmd":    m.group("cmd").strip(),
        }

    # Try failure
    m = SUDO_FAIL_RE.match(line)
    if m:
        return {
            "event": "failure",
            "ts":    parse_ts(m.group("month"), m.group("day"), m.group("time")),
            "user":  m.group("user"),
            "tty":   m.group("tty"),
            "runas": "root",
            "cmd":   "",
        }

    return None


# ─────────────────────────────────────────────────────────────────────────────
# DETECTORS
# ─────────────────────────────────────────────────────────────────────────────

def detect_dangerous_commands(records: list[dict]) -> list[dict]:
    alerts = []
    for r in records:
        if r["event"] != "success":
            continue
        for name, pattern, description in DANGEROUS_PATTERNS:
            if pattern.search(r["cmd"]):
                alerts.append({
                    "type":        name,
                    "description": description,
                    "user":        r["user"],
                    "runas":       r["runas"],
                    "tty":         r["tty"],
                    "cmd":         r["cmd"],
                    "ts":          r["ts"],
                })
    return alerts


def detect_escalation_chain(records: list[dict]) -> list[dict]:
    """Detect TTY session where sudo → su → another sudo occurs within 5 min."""
    tty_events: dict[str, list[dict]] = defaultdict(list)
    for r in records:
        if r["event"] == "success":
            tty_events[r["tty"]].append(r)

    alerts = []
    for tty, events in tty_events.items():
        events_sorted = sorted(events, key=lambda x: x["ts"])
        # Look for su followed by another command, all within 5 minutes of first
        has_su   = any(re.search(r'(?:^|/)su\b', e["cmd"]) for e in events_sorted)
        has_sudo = len(events_sorted) >= 3
        if has_su and has_sudo:
            duration = events_sorted[-1]["ts"] - events_sorted[0]["ts"]
            if duration <= 300:
                chain = [f"{e['user']}→{e['runas']}: {e['cmd'][:40]}" for e in events_sorted]
                alerts.append({
                    "type":     "ESCALATION_CHAIN",
                    "tty":      tty,
                    "duration": round(duration),
                    "chain":    chain,
                    "ts_start": events_sorted[0]["ts"],
                })
    return alerts


def detect_off_hours(records: list[dict]) -> list[dict]:
    alerts = []
    for r in records:
        if r["event"] != "success":
            continue
        t = datetime.fromtimestamp(r["ts"]).time()
        if not (BUSINESS_HOURS_START <= t <= BUSINESS_HOURS_END):
            alerts.append({
                "type": "OFF_HOURS_SUDO",
                "user": r["user"],
                "cmd":  r["cmd"][:80],
                "tty":  r["tty"],
                "ts":   r["ts"],
            })
    return alerts


def detect_fail_then_success(records: list[dict]) -> list[dict]:
    """Failed sudo immediately followed by successful sudo within window."""
    user_failures: dict[str, list[float]] = defaultdict(list)
    alerts = []

    for r in sorted(records, key=lambda x: x["ts"]):
        user = r["user"]
        if r["event"] == "failure":
            user_failures[user].append(r["ts"])
        elif r["event"] == "success":
            recent = [t for t in user_failures[user]
                      if r["ts"] - FAILED_SUCCESS_WINDOW_SECS <= t < r["ts"]]
            if recent:
                alerts.append({
                    "type":          "FAIL_THEN_SUCCESS",
                    "user":          user,
                    "cmd":           r["cmd"],
                    "success_ts":    r["ts"],
                    "failure_count": len(recent),
                    "gap_secs":      round(r["ts"] - min(recent), 1),
                })
    return alerts


# ─────────────────────────────────────────────────────────────────────────────
# REPORT
# ─────────────────────────────────────────────────────────────────────────────

def ts_fmt(ts: float) -> str:
    return datetime.fromtimestamp(ts).strftime("%Y-%m-%dT%H:%M:%S")


def print_report(dangerous, escalation, off_hours, fail_success) -> None:
    print("=" * 65)
    print("SUDO / AUDITD COMMAND AUDIT REPORT")
    print("=" * 65)

    print(f"\n[DANGEROUS_COMMANDS] {len(dangerous)} alert(s)")
    for a in dangerous:
        print(f"  [{a['type']}] {a['description']}")
        print(f"    user={a['user']} runas={a['runas']} tty={a['tty']} ts={ts_fmt(a['ts'])}")
        print(f"    cmd: {a['cmd'][:100]}")

    print(f"\n[ESCALATION_CHAIN] {len(escalation)} alert(s)")
    for a in escalation:
        print(f"  tty={a['tty']}  duration={a['duration']}s  ts={ts_fmt(a['ts_start'])}")
        for step in a["chain"]:
            print(f"    {step}")

    print(f"\n[OFF_HOURS_SUDO] {len(off_hours)} alert(s)")
    for a in off_hours:
        print(f"  user={a['user']}  ts={ts_fmt(a['ts'])}  cmd={a['cmd'][:80]}")

    print(f"\n[FAIL_THEN_SUCCESS] {len(fail_success)} alert(s)")
    for a in fail_success:
        print(f"  user={a['user']}  failures={a['failure_count']}  "
              f"gap={a['gap_secs']}s  success_at={ts_fmt(a['success_ts'])}")
        print(f"    cmd: {a['cmd'][:80]}")
    print("=" * 65)


# ─────────────────────────────────────────────────────────────────────────────
# MAIN
# ─────────────────────────────────────────────────────────────────────────────

if __name__ == "__main__":
    if len(sys.argv) > 1:
        with open(sys.argv[1], encoding="utf-8", errors="replace") as fh:
            raw_lines = fh.readlines()
    else:
        print("[INFO] Using embedded sample data.\n")
        raw_lines = SAMPLE_LOG.splitlines(keepends=True)

    records = [r for r in (parse_line(l) for l in raw_lines) if r is not None]
    print(f"Parsed {len(records)} sudo audit events.\n")
    print_report(
        detect_dangerous_commands(records),
        detect_escalation_chain(records),
        detect_off_hours(records),
        detect_fail_then_success(records),
    )
