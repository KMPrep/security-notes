"""
Q14. Container / Docker Runtime Escape Detector
=================================================
Parse Docker/containerd event logs to detect privileged containers,
dangerous volume mounts, capability additions, disabled security
profiles, and namespace escape attempts.

=============================================================================
SOLUTION PROGRESSION
=============================================================================

NAIVE: Grep logs for "--privileged".
  Problem: Detection fires on flags in docker run but containerd uses
  different field names. Also misses volume mounts, cap additions,
  seccomp=unconfined, and nsenter/unshare runtime commands.

BETTER: Parse docker inspect output and check fields.
  Problem: Reactive. Log-based, real-time detection requires parsing
  stream events (OCI lifecycle events, daemon audit). Also doesn't catch
  nsenter/unshare executed inside a container at runtime.

GOOGLE-LEVEL (this file):
  - Covers CREATE events (privileged, volumes, caps) and EXEC events
    (nsenter, unshare inside container).
  - Risk scoring to separate "informational" from "critical".
  - Each dangerous indicator is independent: allows policy tuning per org.
  - Properly handles both Docker JSON events and simplified audit lines.

Complexity: O(N) over all log events.
=============================================================================
"""

import re
import sys
from datetime import datetime

# ─────────────────────────────────────────────────────────────────────────────
# SAMPLE DATA  (simplified docker/containerd audit line format)
# ─────────────────────────────────────────────────────────────────────────────

SAMPLE_LOG = """\

2024-03-12T08:01:00Z ACTION=create CONTAINER=deadbeef IMAGE=attackimage:latest CMD="bash" PRIVILEGED=true VOLUMES=[] CAPS=[] SECCOMP=runtime/default APPARMOR=docker-default
2024-03-12T08:02:00Z ACTION=create CONTAINER=aabbcc IMAGE=sidecar:v1 CMD="/app/run" PRIVILEGED=false VOLUMES=[/etc:/host-etc,/var/run/docker.sock:/var/run/docker.sock] CAPS=[CAP_SYS_ADMIN,CAP_NET_ADMIN] SECCOMP=unconfined APPARMOR=unconfined
2024-03-12T08:03:00Z ACTION=create CONTAINER=11223344 IMAGE=app:stable CMD="/app" PRIVILEGED=false VOLUMES=[/tmp:/data] CAPS=[] SECCOMP=runtime/default APPARMOR=docker-default

"""

# ─────────────────────────────────────────────────────────────────────────────
# CONFIG
# ─────────────────────────────────────────────────────────────────────────────

# Volume mounts that grant host access
DANGEROUS_VOLUMES = {
    "/etc", "/proc", "/sys", "/root",
    "/var/run/docker.sock", "/var/run/containerd/containerd.sock",
    "/run/docker.sock",
}

# Capabilities that allow escaping the container
DANGEROUS_CAPS = {
    "CAP_SYS_ADMIN",    # remount, pivot_root, full-kernel access
    "CAP_NET_ADMIN",    # iptables, packet injection
    "CAP_SYS_PTRACE",   # ptrace host processes
    "CAP_SYS_MODULE",   # insert kernel modules
    "CAP_DAC_OVERRIDE", # bypass file permissions
    "CAP_NET_RAW",      # raw sockets
    "CAP_SETUID",       # change UID
}

# Namespace escape commands
NSENTER_RE = re.compile(r'\bnsenter\b|\bunshare\b', re.IGNORECASE)

# ─────────────────────────────────────────────────────────────────────────────
# PARSER
# ─────────────────────────────────────────────────────────────────────────────

# Capture key=value pairs, handling quoted values and bracket-delimited lists
KV_RE = re.compile(
    r'(?P<key>[A-Z_]+)='
    r'(?:(?P<quoted>"[^"]*")|(?P<bracket>\[[^\]]*\])|(?P<bare>\S+))'
)


def extract_kv(text: str) -> dict[str, str]:
    result: dict[str, str] = {}
    for m in KV_RE.finditer(text):
        key = m.group("key")
        val = m.group("quoted") or m.group("bracket") or m.group("bare")
        result[key] = val.strip('"')
    return result


def parse_list_field(raw: str) -> list[str]:
    """Parse [item1,item2] or empty [] from a string field."""
    raw = raw.strip("[]")
    if not raw:
        return []
    return [x.strip() for x in raw.split(",") if x.strip()]


def parse_line(line: str) -> dict | None:
    line = line.strip()
    if not line:
        return None
    # Extract timestamp (first token)
    parts = line.split(" ", 1)
    if len(parts) < 2:
        return None
    try:
        ts = datetime.strptime(parts[0], "%Y-%m-%dT%H:%M:%SZ").timestamp()
    except ValueError:
        return None

    kv = extract_kv(parts[1])
    action = kv.get("ACTION", "").lower()
    if action not in ("create", "exec"):
        return None

    event: dict = {
        "ts":        ts,
        "action":    action,
        "container": kv.get("CONTAINER", "?"),
        "image":     kv.get("IMAGE", "?"),
        "cmd":       kv.get("CMD", ""),
    }

    if action == "create":
        event["privileged"] = kv.get("PRIVILEGED", "false").lower() == "true"
        event["volumes"]    = parse_list_field(kv.get("VOLUMES", "[]"))
        event["caps"]       = parse_list_field(kv.get("CAPS", "[]"))
        event["seccomp"]    = kv.get("SECCOMP", "")
        event["apparmor"]   = kv.get("APPARMOR", "")
    elif action == "exec":
        event["user"] = kv.get("USER", "?")

    return event


# ─────────────────────────────────────────────────────────────────────────────
# DETECTORS
# ─────────────────────────────────────────────────────────────────────────────

def detect_create_events(records: list[dict]) -> list[dict]:
    alerts = []
    for r in records:
        if r["action"] != "create":
            continue

        issues = []
        score  = 0

        if r["privileged"]:
            issues.append("PRIVILEGED_FLAG")
            score += 5

        dangerous_vols = [v.split(":")[0] for v in r["volumes"]
                          if v.split(":")[0] in DANGEROUS_VOLUMES]
        if dangerous_vols:
            issues.append(f"DANGEROUS_VOLUMES:{','.join(dangerous_vols)}")
            score += 3 * len(dangerous_vols)

        # docker.sock alone is a container escape vector
        if any("docker.sock" in v or "containerd.sock" in v for v in r["volumes"]):
            issues.append("DOCKER_SOCKET_MOUNT")
            score += 4

        dangerous_caps = [c for c in r["caps"] if c.upper() in DANGEROUS_CAPS]
        if dangerous_caps:
            issues.append(f"DANGEROUS_CAPS:{','.join(dangerous_caps)}")
            score += 2 * len(dangerous_caps)

        if r["seccomp"] in ("unconfined", ""):
            issues.append("SECCOMP_DISABLED")
            score += 2

        if r["apparmor"] in ("unconfined", ""):
            issues.append("APPARMOR_DISABLED")
            score += 2

        if issues:
            alerts.append({
                "type":      "CONTAINER_CREATE_RISK",
                "container": r["container"],
                "image":     r["image"],
                "issues":    issues,
                "score":     score,
                "ts":        r["ts"],
            })
    return alerts


def detect_exec_events(records: list[dict]) -> list[dict]:
    alerts = []
    for r in records:
        if r["action"] != "exec":
            continue
        if NSENTER_RE.search(r["cmd"]):
            alerts.append({
                "type":      "NAMESPACE_ESCAPE_ATTEMPT",
                "container": r["container"],
                "cmd":       r["cmd"],
                "user":      r.get("user", "?"),
                "ts":        r["ts"],
            })
    return alerts


# ─────────────────────────────────────────────────────────────────────────────
# REPORT
# ─────────────────────────────────────────────────────────────────────────────

RISK_LABELS = {0: "INFO", 3: "LOW", 5: "MEDIUM", 8: "HIGH", 12: "CRITICAL"}


def risk_label(score: int) -> str:
    for threshold in sorted(RISK_LABELS.keys(), reverse=True):
        if score >= threshold:
            return RISK_LABELS[threshold]
    return "INFO"


def ts_fmt(ts: float) -> str:
    return datetime.utcfromtimestamp(ts).strftime("%Y-%m-%dT%H:%M:%SZ")


def print_report(create_alerts: list[dict], exec_alerts: list[dict]) -> None:
    print("=" * 65)
    print("CONTAINER / DOCKER ESCAPE DETECTION REPORT")
    print("=" * 65)

    sorted_creates = sorted(create_alerts, key=lambda x: x["score"], reverse=True)

    print(f"\n[CONTAINER_CREATE_RISK] {len(sorted_creates)} container(s) with issues")
    for a in sorted_creates:
        print(f"  [{risk_label(a['score'])}] score={a['score']}  "
              f"container={a['container']}  image={a['image']}  ts={ts_fmt(a['ts'])}")
        for issue in a["issues"]:
            print(f"    - {issue}")

    print(f"\n[NAMESPACE_ESCAPE_ATTEMPT] {len(exec_alerts)} event(s)")
    for a in exec_alerts:
        print(f"  [CRITICAL] container={a['container']}  user={a['user']}  ts={ts_fmt(a['ts'])}")
        print(f"    cmd: {a['cmd']}")
    print("=" * 65)


# ─────────────────────────────────────────────────────────────────────────────
# MAIN
# ─────────────────────────────────────────────────────────────────────────────

if __name__ == "__main__":
    if len(sys.argv) > 1:
        with open(sys.argv[1], encoding="utf-8", errors="replace") as fh:
            raw_lines = fh.readlines()
    else:
        print("[INFO] Using embedded sample data.\n")
        raw_lines = SAMPLE_LOG.splitlines(keepends=True)

    records = [r for r in (parse_line(l) for l in raw_lines) if r is not None]
    print(f"Parsed {len(records)} container events.\n")
    print_report(
        detect_create_events(records),
        detect_exec_events(records),
    )
