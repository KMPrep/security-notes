"""
Q15. Process Snapshot Diff (ps aux) — Fileless & Masquerade Detector
======================================================================
Parse periodic ps aux snapshots and diff them to detect transient
processes (fileless payloads), process masquerading, high-memory
surprise processes, and deleted-exe references.

=============================================================================
SOLUTION PROGRESSION
=============================================================================

NAIVE: Compare two snapshots, print new processes.
  Problem: Every new process appears here — sshd child, cron job,
  a developer's shell. High false-positive rate, no filtering or
  categorization.

BETTER: Alert only on processes that appeared AND disappeared
  between snapshots.
  Problem: Still flags short-lived legitimate processes (sleep,
  grep in a pipeline). Needs masquerade detection (systemd- prefix
  from /tmp) and deleted-exe markers to be actionable.

GOOGLE-LEVEL (this file):
  - Transient process: present in snapshot N but gone in snapshot N+1.
  - Masquerade: system-sounding name from a non-system path.
  - High-memory: RSS > threshold, not a known hungry process.
  - Deleted exe: /proc/PID/exe is marked "(deleted)" — common for
    in-memory payload execution.
  - Handles multiple snapshot files passed as CLI args.

Complexity: O(S × P) where S = number of snapshots, P = processes.
=============================================================================
"""

import re
import sys
from collections import defaultdict
from datetime import datetime

# ─────────────────────────────────────────────────────────────────────────────
# SAMPLE DATA  (two ps aux snapshots separated by --- SNAPSHOT_BREAK ---)
# ─────────────────────────────────────────────────────────────────────────────

# ─────────────────────────────────────────────────────────────────────────────
# CONFIG
# ─────────────────────────────────────────────────────────────────────────────

HIGH_RSS_THRESHOLD_KB = 500_000  # 500 MB

# Processes that are legitimately large (exclude from HIGH_RSS alerts)
KNOWN_HUNGRY = {
    "java", "node", "python", "python3", "chrome", "firefox",
    "mysqld", "postgres", "mongod", "redis-server",
}

# Name prefixes that look like system processes but shouldn't run from /tmp, /dev, /run/user
SYSTEM_LOOKING_NAMES = re.compile(
    r'^(systemd|kworker|kthread|ksoftirqd|sshd|init|udev|dbus|cron|rsyslogd)',
    re.IGNORECASE,
)

NON_SYSTEM_PATHS = re.compile(
    r'^(/tmp|/dev/shm|/var/tmp|/run/user|/home|/root)',
    re.IGNORECASE,
)

DELETED_EXE_RE = re.compile(r'\(deleted\)|/memfd:', re.IGNORECASE)

SNAPSHOT_MARKER_RE = re.compile(r'^---\s+SNAPSHOT\s+(\S+)\s+---')

# ─────────────────────────────────────────────────────────────────────────────
# PARSER
# ─────────────────────────────────────────────────────────────────────────────

# USER PID %CPU %MEM VSZ RSS TTY STAT START TIME COMMAND
PS_RE = re.compile(
    r'^(?P<user>\S+)\s+'
    r'(?P<pid>\d+)\s+'
    r'(?P<cpu>\S+)\s+'
    r'(?P<mem>\S+)\s+'
    r'(?P<vsz>\d+)\s+'
    r'(?P<rss>\d+)\s+'
    r'(?P<tty>\S+)\s+'
    r'(?P<stat>\S+)\s+'
    r'(?P<start>\S+)\s+'
    r'(?P<time>\S+)\s+'
    r'(?P<cmd>.+)$'
)


def parse_snapshot(lines: list[str]) -> dict[int, dict]:
    """Return dict of PID → process record from a block of ps aux lines."""
    procs: dict[int, dict] = {}
    for line in lines:
        line = line.strip()
        m = PS_RE.match(line)
        if not m:
            continue
        pid = int(m.group("pid"))
        cmd = m.group("cmd").strip()
        # Extract binary name (first token, strip path)
        cmd_name = cmd.split()[0].replace("\\", "/").split("/")[-1].lstrip("[").rstrip("]")
        try:
            rss = int(m.group("rss"))
        except ValueError:
            rss = 0
        procs[pid] = {
            "user":     m.group("user"),
            "pid":      pid,
            "rss":      rss,
            "stat":     m.group("stat"),
            "cmd":      cmd,
            "cmd_name": cmd_name,
        }
    return procs


def split_snapshots(lines: list[str]) -> list[tuple[str, dict[int, dict]]]:
    """Split raw lines into list of (timestamp_str, process_dict) tuples."""
    snapshots: list[tuple[str, dict[int, dict]]] = []
    current_ts = "unknown"
    current_lines: list[str] = []

    for line in lines:
        m = SNAPSHOT_MARKER_RE.match(line.strip())
        if m:
            if current_lines:
                snapshots.append((current_ts, parse_snapshot(current_lines)))
            current_ts = m.group(1)
            current_lines = []
        else:
            current_lines.append(line)

    if current_lines:
        snapshots.append((current_ts, parse_snapshot(current_lines)))

    return snapshots


# ─────────────────────────────────────────────────────────────────────────────
# DETECTORS
# ─────────────────────────────────────────────────────────────────────────────

def detect_transient(snapshots: list[tuple[str, dict[int, dict]]]) -> list[dict]:
    """Processes that appear in snapshot N but are gone in snapshot N+1."""
    alerts = []
    for i in range(len(snapshots) - 1):
        ts_a, procs_a = snapshots[i]
        ts_b, procs_b = snapshots[i + 1]
        for pid, proc in procs_a.items():
            if pid not in procs_b:
                alerts.append({
                    "type":        "TRANSIENT_PROCESS",
                    "pid":         pid,
                    "user":        proc["user"],
                    "cmd":         proc["cmd"],
                    "ts_appeared": ts_a,
                    "ts_gone":     ts_b,
                })
    return alerts


def detect_masquerade(snapshots: list[tuple[str, dict[int, dict]]]) -> list[dict]:
    """System-looking name running from non-system path."""
    seen: set[int] = set()  # avoid duplicate alerts across snapshots
    alerts = []
    for ts, procs in snapshots:
        for pid, proc in procs.items():
            if pid in seen:
                continue
            if SYSTEM_LOOKING_NAMES.match(proc["cmd_name"]):
                # Extract the path (first token of cmd)
                path = proc["cmd"].split()[0] if proc["cmd"] else ""
                if NON_SYSTEM_PATHS.match(path):
                    alerts.append({
                        "type":     "MASQUERADE",
                        "pid":      pid,
                        "user":     proc["user"],
                        "cmd":      proc["cmd"],
                        "ts":       ts,
                    })
                    seen.add(pid)
    return alerts


def detect_high_rss(snapshots: list[tuple[str, dict[int, dict]]]) -> list[dict]:
    seen: set[int] = set()
    alerts = []
    for ts, procs in snapshots:
        for pid, proc in procs.items():
            if pid in seen:
                continue
            if proc["rss"] > HIGH_RSS_THRESHOLD_KB and proc["cmd_name"] not in KNOWN_HUNGRY:
                alerts.append({
                    "type":    "HIGH_RSS_SURPRISE",
                    "pid":     pid,
                    "user":    proc["user"],
                    "rss_mb":  round(proc["rss"] / 1024, 1),
                    "cmd":     proc["cmd"],
                    "ts":      ts,
                })
                seen.add(pid)
    return alerts


def detect_deleted_exe(snapshots: list[tuple[str, dict[int, dict]]]) -> list[dict]:
    seen: set[int] = set()
    alerts = []
    for ts, procs in snapshots:
        for pid, proc in procs.items():
            if pid in seen:
                continue
            if DELETED_EXE_RE.search(proc["cmd"]):
                alerts.append({
                    "type": "DELETED_EXE",
                    "pid":  pid,
                    "user": proc["user"],
                    "cmd":  proc["cmd"],
                    "ts":   ts,
                })
                seen.add(pid)
    return alerts


# ─────────────────────────────────────────────────────────────────────────────
# REPORT
# ─────────────────────────────────────────────────────────────────────────────

def print_report(transient, masquerade, high_rss, deleted_exe) -> None:
    print("=" * 65)
    print("PROCESS SNAPSHOT DIFF — ANOMALY DETECTION REPORT")
    print("=" * 65)

    print(f"\n[TRANSIENT_PROCESS] {len(transient)} event(s)  "
          f"(appeared and disappeared between snapshots)")
    for a in transient:
        print(f"  pid={a['pid']}  user={a['user']}")
        print(f"    appeared={a['ts_appeared']}  gone={a['ts_gone']}")
        print(f"    cmd: {a['cmd'][:100]}")

    print(f"\n[MASQUERADE] {len(masquerade)} event(s)  "
          f"(system-looking name from non-system path)")
    for a in masquerade:
        print(f"  pid={a['pid']}  user={a['user']}  ts={a['ts']}")
        print(f"    cmd: {a['cmd'][:100]}")

    print(f"\n[HIGH_RSS_SURPRISE] {len(high_rss)} event(s)  "
          f"(RSS > {HIGH_RSS_THRESHOLD_KB // 1024} MB, not in allowlist)")
    for a in high_rss:
        print(f"  pid={a['pid']}  user={a['user']}  rss={a['rss_mb']} MB  ts={a['ts']}")
        print(f"    cmd: {a['cmd'][:100]}")

    print(f"\n[DELETED_EXE] {len(deleted_exe)} event(s)  "
          f"(exe deleted/memfd — in-memory payload execution)")
    for a in deleted_exe:
        print(f"  pid={a['pid']}  user={a['user']}  ts={a['ts']}")
        print(f"    cmd: {a['cmd'][:100]}")
    print("=" * 65)

if __name__ == "__main__":
    if len(sys.argv) > 1:
        with open(sys.argv[1], encoding="utf-8", errors="replace") as fh:
            raw_lines = fh.readlines()
    else:
        print("[INFO] Using embedded sample data.\n")
        raw_lines = SAMPLE_LOG.splitlines(keepends=True)

    snapshots = split_snapshots(raw_lines)
    print(f"Loaded {len(snapshots)} snapshot(s).\n")

    print_report(
        detect_transient(snapshots),
        detect_masquerade(snapshots),
        detect_high_rss(snapshots),
        detect_deleted_exe(snapshots),
    )
