"""
Q16. AWS CloudTrail Anomaly Detector
======================================
Detect root account usage, IAM privilege escalation, credential
harvesting, suspicious cross-region activity, and S3 data exfiltration
from CloudTrail JSON event logs.

=============================================================================
SOLUTION PROGRESSION
=============================================================================

NAIVE: Alert on every root login.
  Problem: Root actions show up in break-glass scenarios. Without context
  (what they did, how often, which service), alerts are noise. Also misses
  IAM self-escalation, which is more common in real compromises.

BETTER: Alert on root + CreatePolicy/AttachPolicy/PutUserPolicy.
  Problem: Need to also detect credential harvesting (ListAccessKeys on
  OTHER users), cross-region launches from a user who never used that region,
  and bulk S3 GetObject (exfiltration pattern).

GOOGLE-LEVEL (this file):
  - Root usage detector: any non-read API call by root.
  - IAM escalation: CreatePolicy/PutUserPolicy/AttachRolePolicy targeting
    another principal, or AddUserToGroup (group with existing FullAccess).
  - Credential harvesting: ListAccessKeys, GetAccessKeyLastUsed, GetLoginProfile
    for users other than self.
  - Cross-region: call to a region the user has never used before (baseline).
  - S3 exfil: bulk GetObject (> threshold) or GetObject on sensitive prefixes.

Complexity: O(N) over all events, O(U) memory for per-user baselines.
=============================================================================
"""

import sys
import re
from collections import defaultdict
from datetime import datetime

# ─────────────────────────────────────────────────────────────────────────────
# CONFIG
# ─────────────────────────────────────────────────────────────────────────────

# APIs used to harvest other users' credentials/profile info
HARVEST_APIS = {
    "ListAccessKeys", "GetAccessKeyLastUsed", "GetLoginProfile",
    "GetUser", "ListMFADevices", "ListUserPolicies",
}

# IAM privilege escalation actions
IAM_ESCALATION_APIS = {
    "CreatePolicy", "PutUserPolicy", "PutRolePolicy",
    "AttachUserPolicy", "AttachRolePolicy", "AttachGroupPolicy",
    "AddUserToGroup", "CreatePolicyVersion", "SetDefaultPolicyVersion",
}

SENSITIVE_S3_PREFIXES = re.compile(
    r'^(finance|hr|payroll|ssn|pci|credentials|secrets|keys)',
    re.IGNORECASE,
)

S3_BULK_THRESHOLD = 5  # GetObject calls from same user to same bucket within session

KNOWN_REGIONS: dict[str, set] = defaultdict(set)  # user → set of previously seen regions


def parse_event(line: str) -> dict | None:
    line = line.strip()
    if not line:
        return None
    try:
        return json.loads(line)
    except json.JSONDecodeError:
        return None


def get_user(event: dict) -> str:
    uid = event.get("userIdentity", {})
    return uid.get("userName") or uid.get("arn", "unknown")


def get_user_type(event: dict) -> str:
    return event.get("userIdentity", {}).get("type", "")


def get_ts(event: dict) -> float:
    try:
        return datetime.strptime(event["eventTime"], "%Y-%m-%dT%H:%M:%SZ").timestamp()
    except (KeyError, ValueError):
        return 0.0


# ─────────────────────────────────────────────────────────────────────────────
# DETECTORS
# ─────────────────────────────────────────────────────────────────────────────

def detect_root_usage(events: list[dict]) -> list[dict]:
    alerts = []
    for e in events:
        if get_user_type(e) == "Root":
            alerts.append({
                "type":   "ROOT_USAGE",
                "api":    e.get("eventName"),
                "region": e.get("awsRegion"),
                "ts":     get_ts(e),
            })
    return alerts


def detect_iam_escalation(events: list[dict]) -> list[dict]:
    alerts = []
    for e in events:
        if e.get("eventName") not in IAM_ESCALATION_APIS:
            continue
        user  = get_user(e)
        params = e.get("requestParameters") or {}
        target = params.get("userName") or params.get("roleName") or params.get("groupName", "?")
        alerts.append({
            "type":    "IAM_ESCALATION",
            "actor":   user,
            "api":     e.get("eventName"),
            "target":  target,
            "region":  e.get("awsRegion"),
            "details": str(params)[:120],
            "ts":      get_ts(e),
        })
    return alerts


def detect_credential_harvesting(events: list[dict]) -> list[dict]:
    """Flag when a user calls harvest APIs against a DIFFERENT user."""
    alerts = []
    for e in events:
        if e.get("eventName") not in HARVEST_APIS:
            continue
        actor  = get_user(e)
        params = e.get("requestParameters") or {}
        target = params.get("userName", "")
        if target and target != actor:
            alerts.append({
                "type":   "CREDENTIAL_HARVESTING",
                "actor":  actor,
                "api":    e.get("eventName"),
                "target": target,
                "region": e.get("awsRegion"),
                "ts":     get_ts(e),
            })
    return alerts


def detect_cross_region(events: list[dict]) -> list[dict]:
    """Alert when a user makes a mutating API call to a region never seen before."""
    READ_ONLY_PREFIXES = {"Describe", "List", "Get"}
    user_regions: dict[str, set] = defaultdict(set)
    alerts = []

    for e in sorted(events, key=get_ts):
        user   = get_user(e)
        region = e.get("awsRegion", "")
        api    = e.get("eventName", "")
        is_read = any(api.startswith(p) for p in READ_ONLY_PREFIXES)

        if region in user_regions[user] or is_read:
            user_regions[user].add(region)
            continue

        if user_regions[user]:  # user has history → this is a new region
            alerts.append({
                "type":       "NEW_REGION",
                "user":       user,
                "new_region": region,
                "api":        api,
                "ts":         get_ts(e),
            })
        user_regions[user].add(region)

    return alerts


def detect_s3_exfil(events: list[dict]) -> list[dict]:
    """Bulk GetObject or sensitive-prefix access."""
    bulk_tracker: dict[tuple, list] = defaultdict(list)
    sensitive_alerts = []

    for e in events:
        if e.get("eventName") != "GetObject":
            continue
        user   = get_user(e)
        params = e.get("requestParameters") or {}
        bucket = params.get("bucketName", "?")
        key    = params.get("key", "")

        if SENSITIVE_S3_PREFIXES.match(key):
            sensitive_alerts.append({
                "type":   "S3_SENSITIVE_PREFIX",
                "user":   user,
                "bucket": bucket,
                "key":    key,
                "ts":     get_ts(e),
            })
        bulk_tracker[(user, bucket)].append(get_ts(e))

    bulk_alerts = [
        {
            "type":   "S3_BULK_DOWNLOAD",
            "user":   user,
            "bucket": bucket,
            "count":  len(tss),
            "ts":     min(tss),
        }
        for (user, bucket), tss in bulk_tracker.items()
        if len(tss) >= S3_BULK_THRESHOLD
    ]

    return sensitive_alerts + bulk_alerts

# REPORT
# ─────────────────────────────────────────────────────────────────────────────

def ts_fmt(ts: float) -> str:
    return datetime.utcfromtimestamp(ts).strftime("%Y-%m-%dT%H:%M:%SZ")


def print_report(root, escalation, harvesting, cross_region, s3_exfil) -> None:
    print("=" * 65)
    print("AWS CLOUDTRAIL ANOMALY DETECTION REPORT")
    print("=" * 65)

    print(f"\n[ROOT_USAGE] {len(root)} event(s)")
    for a in root:
        print(f"  api={a['api']}  region={a['region']}  ts={ts_fmt(a['ts'])}")

    print(f"\n[IAM_ESCALATION] {len(escalation)} event(s)")
    for a in escalation:
        print(f"  actor={a['actor']}  api={a['api']}  target={a['target']}  ts={ts_fmt(a['ts'])}")
        print(f"    details: {a['details'][:100]}")

    print(f"\n[CREDENTIAL_HARVESTING] {len(harvesting)} event(s)")
    for a in harvesting:
        print(f"  actor={a['actor']}  api={a['api']}  target={a['target']}  ts={ts_fmt(a['ts'])}")

    print(f"\n[NEW_REGION] {len(cross_region)} event(s)")
    for a in cross_region:
        print(f"  user={a['user']}  new_region={a['new_region']}  api={a['api']}  ts={ts_fmt(a['ts'])}")

    print(f"\n[S3_EXFIL] {len(s3_exfil)} event(s)")
    for a in s3_exfil:
        if a["type"] == "S3_BULK_DOWNLOAD":
            print(f"  [BULK] user={a['user']}  bucket={a['bucket']}  count={a['count']}  ts={ts_fmt(a['ts'])}")
        else:
            print(f"  [SENSITIVE] user={a['user']}  bucket={a['bucket']}  key={a['key'][:60]}  ts={ts_fmt(a['ts'])}")
    print("=" * 65)

if __name__ == "__main__":
    if len(sys.argv) > 1:
        with open(sys.argv[1], encoding="utf-8", errors="replace") as fh:
            raw_lines = fh.readlines()
    else:
        print("[INFO] Using embedded sample data.\n")
        raw_lines = SAMPLE_LOG.splitlines(keepends=True)

    events = [e for e in (parse_event(l) for l in raw_lines) if e is not None]
    print(f"Parsed {len(events)} CloudTrail events.\n")
    print_report(
        detect_root_usage(events),
        detect_iam_escalation(events),
        detect_credential_harvesting(events),
        detect_cross_region(events),
        detect_s3_exfil(events),
    )
