"""
Q17. S3 Server Access Log Auditor
===================================
Detect bulk downloads, sensitive prefix access, cross-account requests,
anonymous access, and 403 storms from S3 server access logs.

=============================================================================
SOLUTION PROGRESSION
=============================================================================

NAIVE: Parse the log, print rows with status != 200.
  Problem: 403s are meaningful in aggregate (scanning) but individually
  they're normal. A bulk download of 1000 objects, all returning 200,
  is far more dangerous and would be missed entirely.

BETTER: Alert on any anonymous requester.
  Problem: Misses cross-account access (authenticated, but by a foreign
  account), bulk download, and sensitive-prefix patterns.

GOOGLE-LEVEL (this file):
  - Bulk download: > threshold GetObject calls per requester per bucket window.
  - Sensitive prefix: regex on key path (ssn, pci, credentials, etc.).
  - Cross-account: requester account != bucket-owner account.
  - Anonymous: requester == "-" or AuthType contains anonymous.
  - 403 storm: sustained stream of 403s from one IP → permission-probing.

Complexity: O(N) parse, O(U × B) for aggregation tables.
=============================================================================
"""

import re
import sys
from collections import defaultdict
from datetime import datetime

# ─────────────────────────────────────────────────────────────────────────────
# SAMPLE DATA  (S3 server access log format)
# Columns: BucketOwner Bucket Time RemoteIP Requester RequestID Operation Key
#          RequestURI Status ErrorCode BytesSent ObjectSize TotalTime
#          TurnAroundTime Referrer UserAgent VersionId HostId SignatureVersion
#          CipherSuite AuthType HostHeader TLSVersion
# Simplified subset used here.
# ─────────────────────────────────────────────────────────────────────────────

SAMPLE_LOG = """\

OWNER_ACCT corp-sensitive-data [12/Mar/2024:08:03:03 +0000] 13.14.15.16 arn:aws:iam::222222222222:user/bob REST.GET.OBJECT db/backup.sql 403 AccessDenied 0 0 10 0 - aws-cli - - TLSv1.2

# ─────────────────────────────────────────────────────────────────────────────
# CONFIG
# ─────────────────────────────────────────────────────────────────────────────

BULK_DOWNLOAD_THRESHOLD = 5   # GetObject calls per (requester, bucket) in one session
FORBIDDEN_STORM_THRESHOLD = 3 # 403s from one IP within window

SENSITIVE_KEY_RE = re.compile(
    r'^(finance|hr|payroll|ssn|pci|credentials|secrets|keys|tokens|db|backup|config)',
    re.IGNORECASE,
)

OWNER_ACCOUNT = "OWNER_ACCT"  # bucket-owner account token in logs

# ─────────────────────────────────────────────────────────────────────────────
# PARSER
# ─────────────────────────────────────────────────────────────────────────────

# S3 log is space-delimited; time field contains spaces but is bracketed [...]
# Fields we care about: BucketOwner Bucket [Time] RemoteIP Requester
#                       _ Operation Key Status ...
LOG_RE = re.compile(
    r'^(?P<owner>\S+)\s+'          # 1 BucketOwner
    r'(?P<bucket>\S+)\s+'          # 2 Bucket
    r'\[(?P<time>[^\]]+)\]\s+'     # 3 Time
    r'(?P<ip>\S+)\s+'              # 4 RemoteIP
    r'(?P<requester>\S+)\s+'       # 5 Requester (ARN or -)
    r'\S+\s+'                      # 6 RequestID
    r'(?P<operation>\S+)\s+'       # 7 Operation
    r'(?P<key>\S+)\s+'             # 8 Key
    r'".+?"\s+'                    # 9 RequestURI (quoted)
    r'(?P<status>\d+)'             # 10 HTTP Status
)

TIME_FMT = "%d/%b/%Y:%H:%M:%S +0000"


def parse_line(line: str) -> dict | None:
    line = line.strip()
    if not line:
        return None
    m = LOG_RE.match(line)
    if not m:
        return None
    try:
        ts = datetime.strptime(m.group("time"), TIME_FMT).timestamp()
    except ValueError:
        return None
    return {
        "ts":         ts,
        "owner":      m.group("owner"),
        "bucket":     m.group("bucket"),
        "ip":         m.group("ip"),
        "requester":  m.group("requester"),
        "operation":  m.group("operation"),
        "key":        m.group("key"),
        "status":     int(m.group("status")),
    }


def requester_account(requester: str) -> str:
    """Extract AWS account ID from IAM ARN, or return '-' for anonymous."""
    if requester == "-":
        return "-"
    m = re.search(r'::(\d{12}):', requester)
    return m.group(1) if m else requester


# ─────────────────────────────────────────────────────────────────────────────
# DETECTORS
# ─────────────────────────────────────────────────────────────────────────────

def detect_bulk_download(records: list[dict]) -> list[dict]:
    counts: dict[tuple, list] = defaultdict(list)
    for r in records:
        if r["operation"] == "REST.GET.OBJECT" and r["status"] == 200:
            counts[(r["requester"], r["bucket"])].append(r["ts"])
    return [
        {
            "type":      "BULK_DOWNLOAD",
            "requester": req,
            "bucket":    bucket,
            "count":     len(tss),
            "ts":        min(tss),
        }
        for (req, bucket), tss in counts.items()
        if len(tss) >= BULK_DOWNLOAD_THRESHOLD
    ]


def detect_sensitive_prefix(records: list[dict]) -> list[dict]:
    return [
        {
            "type":      "SENSITIVE_KEY_ACCESS",
            "requester": r["requester"],
            "bucket":    r["bucket"],
            "key":       r["key"],
            "status":    r["status"],
            "ts":        r["ts"],
        }
        for r in records
        if SENSITIVE_KEY_RE.match(r["key"])
    ]


def detect_cross_account(records: list[dict]) -> list[dict]:
    alerts = []
    for r in records:
        acct = requester_account(r["requester"])
        # Non-anonymous and account differs from bucket owner token
        if acct != "-" and r["owner"] != acct:
            alerts.append({
                "type":      "CROSS_ACCOUNT_ACCESS",
                "requester": r["requester"],
                "bucket":    r["bucket"],
                "key":       r["key"],
                "status":    r["status"],
                "ts":        r["ts"],
            })
    return alerts


def detect_anonymous(records: list[dict]) -> list[dict]:
    return [
        {
            "type":   "ANONYMOUS_ACCESS",
            "ip":     r["ip"],
            "bucket": r["bucket"],
            "key":    r["key"],
            "status": r["status"],
            "ts":     r["ts"],
        }
        for r in records
        if r["requester"] == "-"
    ]


def detect_403_storm(records: list[dict]) -> list[dict]:
    forbidden: dict[str, list] = defaultdict(list)
    for r in records:
        if r["status"] == 403:
            forbidden[r["ip"]].append(r["ts"])
    return [
        {
            "type":  "403_STORM",
            "ip":    ip,
            "count": len(tss),
            "ts":    min(tss),
        }
        for ip, tss in forbidden.items()
        if len(tss) >= FORBIDDEN_STORM_THRESHOLD
    ]


# ─────────────────────────────────────────────────────────────────────────────
# REPORT
# ─────────────────────────────────────────────────────────────────────────────

def ts_fmt(ts: float) -> str:
    return datetime.utcfromtimestamp(ts).strftime("%Y-%m-%dT%H:%M:%SZ")


def print_report(bulk, sensitive, cross_acct, anon, storm) -> None:
    print("=" * 65)
    print("S3 SERVER ACCESS LOG AUDIT REPORT")
    print("=" * 65)

    print(f"\n[BULK_DOWNLOAD] {len(bulk)} alert(s)")
    for a in bulk:
        print(f"  requester={a['requester'][:60]}  bucket={a['bucket']}  "
              f"objects={a['count']}  ts={ts_fmt(a['ts'])}")

    print(f"\n[SENSITIVE_KEY_ACCESS] {len(sensitive)} event(s)")
    for a in sensitive:
        print(f"  requester={a['requester'][:60]}  key={a['key']}  "
              f"status={a['status']}  ts={ts_fmt(a['ts'])}")

    print(f"\n[CROSS_ACCOUNT_ACCESS] {len(cross_acct)} event(s)")
    for a in cross_acct:
        print(f"  requester={a['requester'][:60]}  bucket={a['bucket']}  "
              f"key={a['key']}  status={a['status']}  ts={ts_fmt(a['ts'])}")

    print(f"\n[ANONYMOUS_ACCESS] {len(anon)} event(s)")
    for a in anon:
        print(f"  ip={a['ip']}  bucket={a['bucket']}  key={a['key']}  ts={ts_fmt(a['ts'])}")

    print(f"\n[403_STORM] {len(storm)} alert(s)")
    for a in storm:
        print(f"  ip={a['ip']}  count={a['count']}  ts={ts_fmt(a['ts'])}")
    print("=" * 65)


# ─────────────────────────────────────────────────────────────────────────────
# MAIN
# ─────────────────────────────────────────────────────────────────────────────

if __name__ == "__main__":
    if len(sys.argv) > 1:
        with open(sys.argv[1], encoding="utf-8", errors="replace") as fh:
            raw_lines = fh.readlines()
    else:
        print("[INFO] Using embedded sample data.\n")
        raw_lines = SAMPLE_LOG.splitlines(keepends=True)

    records = [r for r in (parse_line(l) for l in raw_lines) if r is not None]
    print(f"Parsed {len(records)} S3 access log entries.\n")
    print_report(
        detect_bulk_download(records),
        detect_sensitive_prefix(records),
        detect_cross_account(records),
        detect_anonymous(records),
        detect_403_storm(records),
    )
