"""
Q18. Kubernetes Audit Log Analyzer
=====================================
Detect container exec, secret reads, RBAC escalation, hostNetwork/PID
pod specs, and anonymous API server access from K8s audit logs.

=============================================================================
SOLUTION PROGRESSION
=============================================================================

NAIVE: Alert on exec into pods.
  Problem: kubectl exec is used by legitimate devs. Without filtering
  for production namespaces, time of day, or user context, the alert rate
  overwhelms. Misses the silent ways to get root: create a hostPID pod.

BETTER: Alert on exec + secret read in production namespace.
  Problem: Misses RBAC escalation (creating ClusterRoleBindings),
  hostNetwork/PID pod creation (node escape), and anonymous API access.

GOOGLE-LEVEL (this file):
  - exec/attach: flag by user + namespace (prod vs dev) + time.
  - Secret access: read/list secrets, cross-namespace.
  - RBAC escalation: create/update ClusterRoleBinding or RoleBinding
    that grants cluster-admin or wildcard verbs.
  - hostNetwork/hostPID/hostIPC pods: node-level escape vectors.
  - Anonymous access: user == "system:anonymous".

Complexity: O(N) over all audit events.
=============================================================================
"""

import json
import sys
import re
from collections import defaultdict
from datetime import datetime

# ─────────────────────────────────────────────────────────────────────────────
# SAMPLE DATA  (K8s audit log, one JSON event per line)
# ─────────────────────────────────────────────────────────────────────────────

SAMPLE_LOG = r"""
{"timestamp":"2024-03-12T08:00:00Z","user":{"username":"alice","groups":["system:masters"]},"verb":"create","objectRef":{"resource":"pods","namespace":"production"},"requestObject":{"spec":{"hostNetwork":true,"hostPID":true,"containers":[{"name":"debug","image":"ubuntu"}]}},"responseStatus":{"code":201}}
{"timestamp":"2024-03-12T08:01:00Z","user":{"username":"alice"},"verb":"create","objectRef":{"resource":"clusterrolebindings","namespace":""},"requestObject":{"subjects":[{"kind":"User","name":"eve"}],"roleRef":{"kind":"ClusterRole","name":"cluster-admin"}},"responseStatus":{"code":201}}
{"timestamp":"2024-03-12T08:02:00Z","user":{"username":"bob"},"verb":"get","objectRef":{"resource":"secrets","namespace":"production","name":"db-credentials"},"responseStatus":{"code":200}}
{"timestamp":"2024-03-12T08:02:30Z","user":{"username":"bob"},"verb":"list","objectRef":{"resource":"secrets","namespace":"kube-system"},"responseStatus":{"code":200}}
{"timestamp":"2024-03-12T08:03:00Z","user":{"username":"carol"},"verb":"create","objectRef":{"resource":"pods/exec","namespace":"production","name":"web-app-xyz"},"requestObject":{"command":["/bin/bash"]},"responseStatus":{"code":101}}
{"timestamp":"2024-03-12T08:04:00Z","user":{"username":"system:anonymous"},"verb":"list","objectRef":{"resource":"pods","namespace":"default"},"responseStatus":{"code":200}}

# ─────────────────────────────────────────────────────────────────────────────
# CONFIG
# ─────────────────────────────────────────────────────────────────────────────

PRIVILEGED_NAMESPACES = {"production", "kube-system", "kube-public", "default"}
HIGH_PRIVILEGE_ROLES  = {"cluster-admin", "admin", "edit"}


# ─────────────────────────────────────────────────────────────────────────────
# PARSER
# ─────────────────────────────────────────────────────────────────────────────

def parse_event(line: str) -> dict | None:
    line = line.strip()
    if not line:
        return None
    try:
        return json.loads(line)
    except json.JSONDecodeError:
        return None


def get_username(event: dict) -> str:
    return event.get("user", {}).get("username", "unknown")


def get_ts(event: dict) -> float:
    try:
        return datetime.strptime(event["timestamp"], "%Y-%m-%dT%H:%M:%SZ").timestamp()
    except (KeyError, ValueError):
        return 0.0


def get_obj(event: dict) -> dict:
    return event.get("objectRef", {})


def get_status(event: dict) -> int:
    return event.get("responseStatus", {}).get("code", 0)


# ─────────────────────────────────────────────────────────────────────────────
# DETECTORS
# ─────────────────────────────────────────────────────────────────────────────

def detect_exec_attach(events: list[dict]) -> list[dict]:
    alerts = []
    for e in events:
        obj  = get_obj(e)
        resource = obj.get("resource", "")
        if resource not in ("pods/exec", "pods/attach"):
            continue
        ns   = obj.get("namespace", "")
        user = get_username(e)
        verb = e.get("verb", "")
        alerts.append({
            "type":      "POD_EXEC_ATTACH",
            "user":      user,
            "namespace": ns,
            "pod":       obj.get("name", "?"),
            "verb":      verb,
            "sensitive": ns in PRIVILEGED_NAMESPACES,
            "ts":        get_ts(e),
        })
    return alerts


def detect_secret_access(events: list[dict]) -> list[dict]:
    alerts = []
    for e in events:
        obj = get_obj(e)
        if obj.get("resource") != "secrets":
            continue
        ns   = obj.get("namespace", "")
        user = get_username(e)
        verb = e.get("verb", "")
        if verb in ("get", "list", "watch"):
            alerts.append({
                "type":      "SECRET_READ",
                "user":      user,
                "namespace": ns,
                "secret":    obj.get("name", "(all)"),
                "verb":      verb,
                "status":    get_status(e),
                "ts":        get_ts(e),
            })
    return alerts


def detect_rbac_escalation(events: list[dict]) -> list[dict]:
    alerts = []
    for e in events:
        obj      = get_obj(e)
        resource = obj.get("resource", "")
        verb     = e.get("verb", "")
        if resource not in ("clusterrolebindings", "rolebindings") or verb not in ("create", "update", "patch"):
            continue

        req_obj  = e.get("requestObject", {})
        role_ref = req_obj.get("roleRef", {})
        role_name = role_ref.get("name", "")
        subjects  = req_obj.get("subjects", [])

        if role_name in HIGH_PRIVILEGE_ROLES or role_name == "*":
            alerts.append({
                "type":     "RBAC_ESCALATION",
                "actor":    get_username(e),
                "resource": resource,
                "role":     role_name,
                "subjects": [s.get("name") for s in subjects],
                "ts":       get_ts(e),
            })
    return alerts


def detect_host_namespace_pods(events: list[dict]) -> list[dict]:
    alerts = []
    for e in events:
        obj  = get_obj(e)
        verb = e.get("verb", "")
        if obj.get("resource") != "pods" or verb not in ("create", "update"):
            continue

        spec = e.get("requestObject", {}).get("spec", {})
        flags = []
        if spec.get("hostNetwork"):
            flags.append("hostNetwork")
        if spec.get("hostPID"):
            flags.append("hostPID")
        if spec.get("hostIPC"):
            flags.append("hostIPC")

        # Check privileged containers
        for container in spec.get("containers", []):
            sc = container.get("securityContext", {})
            if sc.get("privileged"):
                flags.append(f"privileged_container:{container.get('name','?')}")

        if flags:
            alerts.append({
                "type":      "HOST_NAMESPACE_POD",
                "user":      get_username(e),
                "namespace": obj.get("namespace", ""),
                "flags":     flags,
                "ts":        get_ts(e),
            })
    return alerts


def detect_anonymous_access(events: list[dict]) -> list[dict]:
    return [
        {
            "type":      "ANONYMOUS_API_ACCESS",
            "verb":      e.get("verb"),
            "resource":  get_obj(e).get("resource"),
            "namespace": get_obj(e).get("namespace", ""),
            "status":    get_status(e),
            "ts":        get_ts(e),
        }
        for e in events
        if get_username(e) == "system:anonymous"
    ]


# ─────────────────────────────────────────────────────────────────────────────
# REPORT
# ─────────────────────────────────────────────────────────────────────────────

def ts_fmt(ts: float) -> str:
    return datetime.utcfromtimestamp(ts).strftime("%Y-%m-%dT%H:%M:%SZ")


def print_report(execs, secrets, rbac, host_ns, anon) -> None:
    print("=" * 65)
    print("KUBERNETES AUDIT LOG — THREAT DETECTION REPORT")
    print("=" * 65)

    print(f"\n[POD_EXEC_ATTACH] {len(execs)} event(s)")
    for a in execs:
        tag = " *** PRIVILEGED NAMESPACE ***" if a["sensitive"] else ""
        print(f"  user={a['user']}  pod={a['pod']}  ns={a['namespace']}{tag}  ts={ts_fmt(a['ts'])}")

    print(f"\n[SECRET_READ] {len(secrets)} event(s)")
    for a in secrets:
        print(f"  user={a['user']}  ns={a['namespace']}  secret={a['secret']}  "
              f"verb={a['verb']}  status={a['status']}  ts={ts_fmt(a['ts'])}")

    print(f"\n[RBAC_ESCALATION] {len(rbac)} event(s)")
    for a in rbac:
        print(f"  [CRITICAL] actor={a['actor']}  resource={a['resource']}  "
              f"role={a['role']}  subjects={a['subjects']}  ts={ts_fmt(a['ts'])}")

    print(f"\n[HOST_NAMESPACE_POD] {len(host_ns)} event(s)")
    for a in host_ns:
        print(f"  [CRITICAL] user={a['user']}  ns={a['namespace']}  "
              f"flags={a['flags']}  ts={ts_fmt(a['ts'])}")

    print(f"\n[ANONYMOUS_API_ACCESS] {len(anon)} event(s)")
    for a in anon:
        print(f"  verb={a['verb']}  resource={a['resource']}  ns={a['namespace']}  "
              f"status={a['status']}  ts={ts_fmt(a['ts'])}")
    print("=" * 65)


# ─────────────────────────────────────────────────────────────────────────────
# MAIN
# ─────────────────────────────────────────────────────────────────────────────

if __name__ == "__main__":
    if len(sys.argv) > 1:
        with open(sys.argv[1], encoding="utf-8", errors="replace") as fh:
            raw_lines = fh.readlines()
    else:
        print("[INFO] Using embedded sample data.\n")
        raw_lines = SAMPLE_LOG.splitlines(keepends=True)

    events = [e for e in (parse_event(l) for l in raw_lines) if e is not None]
    print(f"Parsed {len(events)} Kubernetes audit events.\n")
    print_report(
        detect_exec_attach(events),
        detect_secret_access(events),
        detect_rbac_escalation(events),
        detect_host_namespace_pods(events),
        detect_anonymous_access(events),
    )
