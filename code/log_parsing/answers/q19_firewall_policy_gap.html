"""
Q19. Firewall Policy Gap Analyzer
===================================
Detect shadowed rules, policy gaps, east-west lateral movement,
and exfiltration indicators from firewall traffic logs.

=============================================================================
SOLUTION PROGRESSION
=============================================================================

NAIVE: Show all DENY traffic — blocked is blocked.
  Problem: Attacker traffic may be allowed (the whole point of policy gaps).
  Showing only denies misses: allowed traffic to unexpected destinations,
  east-west between segments that should be isolated, and outbound on
  non-standard ports.

BETTER: Flag ALLOW traffic on unusual ports.
  Problem: What's "unusual" is environment-specific. A port allowed by a
  permissive rule (e.g., 0/0 → 0/0) but never explicitly intended is a
  "shadowed-rule" problem. Need rule-level analysis, not just port analysis.

GOOGLE-LEVEL (this file):
  - Shadowed rules: a specific DENY rule is shadowed by an earlier ALLOW *
    rule — the deny can never fire.
  - Policy gaps: ALLOW traffic from untrusted zone → sensitive zone that has
    no explicit rule (matched the catch-all).
  - East-west movement: internal subnet→internal subnet on sensitive ports
    (SMB, RDP, SSH, WinRM) where the subnets are in different security zones.
  - Exfil indicators: large outbound bytes to external IPs on non-standard ports,
    or high distinct-destination counts (beaconing/scanning).

Complexity: O(R²) for rule shadowing, O(N) for traffic analysis.
=============================================================================
"""

import sys
from collections import defaultdict
from datetime import datetime
from ipaddress import ip_network, ip_address, IPv4Network, IPv4Address

# ─────────────────────────────────────────────────────────────────────────────
# CONFIG
# ─────────────────────────────────────────────────────────────────────────────

# Security zones and their subnets
ZONES: dict[str, list[IPv4Network]] = {
    "sensitive": [ip_network("10.20.0.0/24")],
    "internal":  [ip_network("10.0.0.0/8")],
    "dmz":       [ip_network("10.10.0.0/24")],
    "untrusted": [ip_network("192.168.0.0/16")],
    "external":  [ip_network("0.0.0.0/0")],  # catch-all
}

EAST_WEST_PORTS = {445: "SMB", 3389: "RDP", 22: "SSH", 5985: "WinRM", 5986: "WinRM-SSL"}
EXFIL_BYTES_THRESHOLD = 500_000       # 500 KB per flow to external
EXFIL_DISTINCT_DSTS_THRESHOLD = 3     # distinct external IPs on same non-standard port


def zone_of(ip_str: str) -> str:
    try:
        addr = ip_address(ip_str)
    except ValueError:
        return "unknown"
    for zone, nets in ZONES.items():
        if zone == "external":
            continue
        for net in nets:
            if addr in net:
                return zone
    return "external"


def is_internal(ip_str: str) -> bool:
    return zone_of(ip_str) not in ("external", "unknown")


# ─────────────────────────────────────────────────────────────────────────────
# RULE PARSER
# ─────────────────────────────────────────────────────────────────────────────

def parse_rules(lines: list[str]) -> list[dict]:
    rules = []
    for line in lines:
        line = line.strip()
        if not line or line.startswith("#"):
            continue
        parts = line.split()
        if len(parts) < 9:
            continue
        try:
            rules.append({
                "id":       int(parts[0]),
                "priority": int(parts[1]),
                "src_zone": parts[2],
                "dst_zone": parts[3],
                "src_net":  ip_network(parts[4], strict=False),
                "dst_net":  ip_network(parts[5], strict=False),
                "proto":    parts[6],
                "dport":    parts[7],  # may be "any" or numeric string
                "action":   parts[8].upper(),
            })
        except (ValueError, IndexError):
            continue
    return sorted(rules, key=lambda r: r["priority"])


def rules_conflict(r_a: dict, r_b: dict) -> bool:
    """Return True if r_b (higher priority number) is shadowed by r_a (lower priority/catch-all)."""
    def net_overlap(n1: IPv4Network, n2: IPv4Network) -> bool:
        return n1.overlaps(n2)

    if r_a["priority"] >= r_b["priority"]:
        return False  # r_b fires first — no shadow
    if r_a["action"] != "ALLOW" or r_b["action"] != "DENY":
        return False  # shadow only matters if a DENY is made unreachable by an ALLOW

    # Check if r_a covers r_b's src, dst, and ports
    if not net_overlap(r_a["src_net"], r_b["src_net"]):
        return False
    if not net_overlap(r_a["dst_net"], r_b["dst_net"]):
        return False
    if r_a["proto"] not in ("any", r_b["proto"]) and r_b["proto"] != "any":
        return False
    if r_a["dport"] not in ("any", r_b["dport"]) and r_b["dport"] != "any":
        return False
    return True


def detect_shadowed_rules(rules: list[dict]) -> list[dict]:
    alerts = []
    for i, r_a in enumerate(rules):
        for r_b in rules[i + 1:]:
            if rules_conflict(r_a, r_b):
                alerts.append({
                    "type":         "SHADOWED_RULE",
                    "deny_rule_id": r_b["id"],
                    "shadowed_by":  r_a["id"],
                    "deny_action":  f"{r_b['src_net']} → {r_b['dst_net']} {r_b['proto']}/{r_b['dport']}",
                })
    return alerts


# ─────────────────────────────────────────────────────────────────────────────
# TRAFFIC PARSER
# ─────────────────────────────────────────────────────────────────────────────

TRAFFIC_RE = re.compile(
    r'(?P<ts>\S+)\s+(?P<action>ALLOW|DENY)\s+'
    r'src=(?P<src>\S+)\s+dst=(?P<dst>\S+)\s+'
    r'proto=(?P<proto>\S+)\s+dport=(?P<dport>\d+)\s+'
    r'bytes=(?P<bytes>\d+)\s+rule=(?P<rule>\d+)',
    re.IGNORECASE,
)


def parse_traffic(lines: list[str]) -> list[dict]:
    records = []
    for line in lines:
        line = line.strip()
        m = TRAFFIC_RE.match(line)
        if not m:
            continue
        try:
            ts = datetime.strptime(m.group("ts"), "%Y-%m-%dT%H:%M:%SZ").timestamp()
        except ValueError:
            continue
        records.append({
            "ts":     ts,
            "action": m.group("action").upper(),
            "src":    m.group("src"),
            "dst":    m.group("dst"),
            "proto":  m.group("proto"),
            "dport":  int(m.group("dport")),
            "bytes":  int(m.group("bytes")),
            "rule":   int(m.group("rule")),
        })
    return records


# ─────────────────────────────────────────────────────────────────────────────
# TRAFFIC DETECTORS
# ─────────────────────────────────────────────────────────────────────────────

def detect_east_west(records: list[dict]) -> list[dict]:
    alerts = []
    for r in records:
        if r["action"] != "ALLOW":
            continue
        dport = r["dport"]
        if dport not in EAST_WEST_PORTS:
            continue
        src_zone = zone_of(r["src"])
        dst_zone = zone_of(r["dst"])
        if src_zone != "external" and dst_zone != "external" and src_zone != dst_zone:
            alerts.append({
                "type":     "EAST_WEST_LATERAL",
                "src":      r["src"],
                "dst":      r["dst"],
                "port":     f"{dport}/{EAST_WEST_PORTS[dport]}",
                "src_zone": src_zone,
                "dst_zone": dst_zone,
                "bytes":    r["bytes"],
                "ts":       r["ts"],
            })
    return alerts


def detect_exfil(records: list[dict]) -> list[dict]:
    # Large single-flow exfil
    large_flows = [
        {
            "type":  "EXFIL_LARGE_FLOW",
            "src":   r["src"],
            "dst":   r["dst"],
            "dport": r["dport"],
            "bytes": r["bytes"],
            "ts":    r["ts"],
        }
        for r in records
        if r["action"] == "ALLOW" and r["bytes"] > EXFIL_BYTES_THRESHOLD
        and not is_internal(r["dst"])
    ]

    # Many distinct external dsts on same non-standard port from same src
    src_port_dsts: dict[tuple, set] = defaultdict(set)
    src_port_ts: dict[tuple, float] = {}
    for r in records:
        if r["action"] == "ALLOW" and not is_internal(r["dst"]) and r["dport"] > 1024:
            key = (r["src"], r["dport"])
            src_port_dsts[key].add(r["dst"])
            src_port_ts.setdefault(key, r["ts"])

    scatter = [
        {
            "type":  "EXFIL_MULTI_DST",
            "src":   src,
            "dport": dport,
            "count": len(dsts),
            "ts":    src_port_ts[(src, dport)],
        }
        for (src, dport), dsts in src_port_dsts.items()
        if len(dsts) >= EXFIL_DISTINCT_DSTS_THRESHOLD
    ]

    return large_flows + scatter


# ─────────────────────────────────────────────────────────────────────────────
# REPORT
# ─────────────────────────────────────────────────────────────────────────────

def ts_fmt(ts: float) -> str:
    return datetime.utcfromtimestamp(ts).strftime("%Y-%m-%dT%H:%M:%SZ")


def print_report(shadowed, east_west, exfil) -> None:
    print("=" * 65)
    print("FIREWALL POLICY GAP & TRAFFIC ANOMALY REPORT")
    print("=" * 65)

    print(f"\n[SHADOWED_RULES] {len(shadowed)} issue(s)")
    for a in shadowed:
        print(f"  DENY rule #{a['deny_rule_id']} ({a['deny_action']})")
        print(f"    shadowed by ALLOW rule #{a['shadowed_by']} — the DENY can never fire")

    print(f"\n[EAST_WEST_LATERAL] {len(east_west)} event(s)")
    for a in east_west:
        print(f"  {a['src']} ({a['src_zone']}) → {a['dst']} ({a['dst_zone']}) "
              f"port={a['port']}  bytes={a['bytes']}  ts={ts_fmt(a['ts'])}")

    print(f"\n[EXFIL_INDICATORS] {len(exfil)} event(s)")
    for a in exfil:
        if a["type"] == "EXFIL_LARGE_FLOW":
            print(f"  [LARGE_FLOW] {a['src']} → {a['dst']}:{a['dport']}  "
                  f"bytes={a['bytes']}  ts={ts_fmt(a['ts'])}")
        else:
            print(f"  [MULTI_DST] {a['src']}:{a['dport']} → {a['count']} distinct external IPs  "
                  f"ts={ts_fmt(a['ts'])}")
    print("=" * 65)

# MAIN
# ─────────────────────────────────────────────────────────────────────────────

if __name__ == "__main__":
    if len(sys.argv) > 1:
        with open(sys.argv[1], encoding="utf-8", errors="replace") as fh:
            all_lines = fh.readlines()
    else:
        print("[INFO] Using embedded sample data.\n")
        all_lines = (SAMPLE_RULES + "\n" + SAMPLE_LOG).splitlines(keepends=True)

    # Split into rule section and traffic section by blank line
    rule_lines: list[str] = []
    traffic_lines: list[str] = []
    in_traffic = False
    for line in all_lines:
        if not in_traffic and (line.strip().startswith("2024") or
                                re.match(r'\d{4}-\d{2}-\d{2}', line.strip())):
            in_traffic = True
        if in_traffic:
            traffic_lines.append(line)
        else:
            rule_lines.append(line)

    rules   = parse_rules(rule_lines)
    records = parse_traffic(traffic_lines)

    print(f"Loaded {len(rules)} firewall rules, {len(records)} traffic events.\n")
    print_report(
        detect_shadowed_rules(rules),
        detect_east_west(records),
        detect_exfil(records),
    )
