"""
Q1. Apache/Nginx Combined Log Deep Parser
==========================================
Parse HTTP access logs, detect scanners, path traversal, error spikes,
and large-response exfiltration candidates.

=============================================================================
SOLUTION PROGRESSION
=============================================================================

NAIVE: Split by space, grab positional fields.
  Problem: timestamp contains a space inside []; quoted request contains spaces.
  Breaks immediately on any real log line.

BETTER: Single regex for the full line; basic field extraction.
  Problem: still no URL-decoding for traversal detection, no aggregation,
  no reporting structure.

GOOGLE-LEVEL (this file):
  - Compiled regex for Combined Log Format.
  - URL-decode paths before traversal matching (catches %2e%2e bypasses).
  - Streaming parse — O(1) memory per line; only aggregation dicts grow.
  - Scanner fingerprint DB as a compiled set of patterns (not repeated re.search calls).
  - Error-spike detection via per-IP error counter with threshold.
  - Clear, structured report at the end.

Complexity: O(N) time, O(U) space where U = unique IPs/paths (U << N in practice).
=============================================================================
"""

# ─────────────────────────────────────────────────────────────────────────────
# CONFIGURATION
# ─────────────────────────────────────────────────────────────────────────────

ERROR_SPIKE_THRESHOLD = 10       # flag IPs with > this many 4xx/5xx responses
LARGE_RESPONSE_BYTES  = 1_048_576  # 1 MB

# Scanner fingerprints — compiled once
SCANNER_PATTERNS = re.compile(
    r'sqlmap|nikto|masscan|nmap|dirbuster|gobuster|dirb|wfuzz|'
    r'nuclei|zap|burpsuite|acunetix|w3af|skipfish|hydra|medusa',
    re.IGNORECASE,
)

# Path traversal indicators (checked AFTER URL-decoding)
TRAVERSAL_PATTERN = re.compile(r'\.\.[\\/]|%252e|%2e%2e', re.IGNORECASE)

# Combined Log Format — single compiled regex
LOG_RE = re.compile(
    r'(?P<ip>\S+)'                   # client IP
    r'\s+\S+'                        # ident (-)
    r'\s+(?P<user>\S+)'              # auth user
    r'\s+\[(?P<ts>[^\]]+)\]'         # [timestamp]
    r'\s+"(?P<method>\S+)'           # "METHOD
    r'\s+(?P<path>\S+)'              #  /path
    r'\s+[^"]+"\s*'                  #  HTTP/1.1"
    r'(?P<status>\d{3})'             # status code
    r'\s+(?P<bytes>\S+)'             # bytes
    r'(?:\s+"(?P<referrer>[^"]*)")?'  # optional referrer
    r'(?:\s+"(?P<ua>[^"]*)")?'        # optional User-Agent
)


# ─────────────────────────────────────────────────────────────────────────────
# PARSE HELPER
# ─────────────────────────────────────────────────────────────────────────────

def parse_line(line: str) -> dict | None:
    """Return parsed dict or None for malformed lines."""
    line = line.strip()
    if not line:
        return None
    m = LOG_RE.match(line)
    if not m:
        return None
    raw_bytes = m.group("bytes")
    try:
        byte_count = int(raw_bytes) if raw_bytes != "-" else 0
    except ValueError:
        byte_count = 0
    return {
        "ip":       m.group("ip"),
        "user":     m.group("user"),
        "ts":       m.group("ts"),
        "method":   m.group("method"),
        "path":     m.group("path"),
        "status":   int(m.group("status")),
        "bytes":    byte_count,
        "referrer": m.group("referrer") or "-",
        "ua":       m.group("ua") or "-",
    }


# ─────────────────────────────────────────────────────────────────────────────
# APPROACH 1 — Manual aggregation (no Counter, no sorted key lambda)
# ─────────────────────────────────────────────────────────────────────────────

def analyze_logs_manual(lines: list[str]) -> dict:
    ip_requests:    dict[str, int]       = defaultdict(int)
    ip_errors:      dict[str, int]       = defaultdict(int)
    status_dist:    dict[int, int]       = defaultdict(int)
    path_requests:  dict[str, int]       = defaultdict(int)
    scanner_hits:   list[dict]           = []
    traversal_hits: list[dict]           = []
    large_response: list[dict]           = []
    malformed:      int                  = 0

    for line in lines:
        record = parse_line(line)
        if record is None:
            malformed += 1
            continue

        ip     = record["ip"]
        status = record["status"]
        path   = record["path"]
        ua     = record["ua"]
        byt    = record["bytes"]

        ip_requests[ip]   += 1
        status_dist[status] += 1
        path_requests[path] += 1

        if status >= 400:
            ip_errors[ip] += 1

        # Scanner detection
        if SCANNER_PATTERNS.search(ua):
            scanner_hits.append({"ip": ip, "ua": ua, "path": path, "ts": record["ts"]})

        # Path traversal — URL-decode first to catch encoded variants
        decoded_path = unquote_plus(path)
        if TRAVERSAL_PATTERN.search(decoded_path) or TRAVERSAL_PATTERN.search(path):
            traversal_hits.append({"ip": ip, "path": path, "decoded": decoded_path, "status": status, "ts": record["ts"]})

        # Large response
        if byt > LARGE_RESPONSE_BYTES:
            large_response.append({"ip": ip, "path": path, "bytes": byt, "status": status, "ts": record["ts"]})

    # Top-5 IPs manually
    sorted_ips   = sorted(ip_requests.items(), key=lambda x: x[1], reverse=True)[:5]
    sorted_paths = sorted(path_requests.items(), key=lambda x: x[1], reverse=True)[:5]

    error_spikes = {ip: cnt for ip, cnt in ip_errors.items() if cnt > ERROR_SPIKE_THRESHOLD}

    return {
        "total_requests": sum(ip_requests.values()),
        "malformed_lines": malformed,
        "status_distribution": dict(sorted(status_dist.items())),
        "top5_ips": sorted_ips,
        "top5_paths": sorted_paths,
        "scanner_hits": scanner_hits,
        "traversal_hits": traversal_hits,
        "large_response_hits": large_response,
        "error_spike_ips": error_spikes,
    }

from collections import Counter

def analyze_logs_idiomatic(lines: list[str]) -> dict:
    ip_counter     = Counter()
    status_counter = Counter()
    path_counter   = Counter()
    ip_errors      = Counter()
    scanner_hits   = []
    traversal_hits = []
    large_response = []
    malformed      = 0

    for line in lines:
        r = parse_line(line)
        if r is None:
            malformed += 1
            continue

        ip_counter[r["ip"]]       += 1
        status_counter[r["status"]] += 1
        path_counter[r["path"]]   += 1

        if r["status"] >= 400:
            ip_errors[r["ip"]] += 1

        if SCANNER_PATTERNS.search(r["ua"]):
            scanner_hits.append(r)

        decoded = unquote_plus(r["path"])
        if TRAVERSAL_PATTERN.search(decoded) or TRAVERSAL_PATTERN.search(r["path"]):
            traversal_hits.append({**r, "decoded_path": decoded})

        if r["bytes"] > LARGE_RESPONSE_BYTES:
            large_response.append(r)

    return {
        "total_requests":    sum(ip_counter.values()),
        "malformed_lines":   malformed,
        "status_distribution": dict(sorted(status_counter.items())),
        "top5_ips":          ip_counter.most_common(5),
        "top5_paths":        path_counter.most_common(5),
        "scanner_hits":      scanner_hits,
        "traversal_hits":    traversal_hits,
        "large_response_hits": large_response,
        "error_spike_ips":   {ip: cnt for ip, cnt in ip_errors.items() if cnt > ERROR_SPIKE_THRESHOLD},
    }

# REPORT PRINTER

def print_report(result: dict) -> None:
    print("=" * 60)
    print("HTTP ACCESS LOG ANALYSIS REPORT")
    print("=" * 60)
    print(f"Total requests  : {result['total_requests']}")
    print(f"Malformed lines : {result['malformed_lines']}")

    print("\n── Status Code Distribution ──")
    for code, cnt in result["status_distribution"].items():
        bar = "█" * min(cnt, 40)
        print(f"  {code}  {cnt:>4}  {bar}")

    print("\n── Top 5 IPs ──")
    for ip, cnt in result["top5_ips"]:
        print(f"  {ip:<18} {cnt} requests")

    print("\n── Top 5 Paths ──")
    for path, cnt in result["top5_paths"]:
        print(f"  {path:<40} {cnt} requests")

    print("\n── [ALERT] Scanner Fingerprints Detected ──")
    if result["scanner_hits"]:
        for h in result["scanner_hits"]:
            print(f"  IP={h['ip']}  UA={h['ua'][:60]}  path={h['path']}")
    else:
        print("  None")

    print("\n── [ALERT] Path Traversal Attempts ──")
    if result["traversal_hits"]:
        for h in result["traversal_hits"]:
            print(f"  IP={h['ip']}  raw={h['path']}  decoded={h.get('decoded_path', h['path'])}  status={h['status']}")
    else:
        print("  None")

    print("\n── [ALERT] Large Response / Possible Exfiltration ──")
    if result["large_response_hits"]:
        for h in result["large_response_hits"]:
            print(f"  IP={h['ip']}  path={h['path']}  bytes={h['bytes']:,}  status={h['status']}")
    else:
        print("  None")

    print("\n── [ALERT] Error Spike IPs (>{} errors) ──".format(ERROR_SPIKE_THRESHOLD))
    if result["error_spike_ips"]:
        for ip, cnt in result["error_spike_ips"].items():
            print(f"  IP={ip}  errors={cnt}")
    else:
        print("  None")
    print("=" * 60)
