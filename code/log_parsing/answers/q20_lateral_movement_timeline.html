"""
Q20. Multi-Log Correlated Timeline & Kill-Chain Mapper
=======================================================
Join firewall, authentication, Sysmon, and DNS events by IP + time
window. Assign MITRE ATT&CK kill-chain stages to each correlated event
cluster and output a chronological timeline per actor IP.

=============================================================================
SOLUTION PROGRESSION
=============================================================================

NAIVE: Print all events sorted by timestamp.
  Problem: Events from different logs have no linkage. A firewall allow
  and a subsequent lateral movement SSH session are unrelated rows in
  your output. No stage labels, no investigation value.

BETTER: Group by IP, print their events.
  Problem: If the attacker pivots (uses hop1 to attack hop2), the
  grouping misses the chain. Also need to normalize heterogeneous formats
  into a common structure before grouping.

GOOGLE-LEVEL (this file):
  - Universal EventRecord with log_type, actor_ip, ts, raw.
  - Join by: actor_ip overlapping within a 5-minute time window.
  - Kill-chain stage assignment based on event type + verb.
  - Output: per-actor timeline showing stage progression.
  - Detects stage transitions: reconnaissance → initial access → lateral.

Complexity: O(N log N) sort + O(C × W) cluster window expansion.
=============================================================================
"""

# ─────────────────────────────────────────────────────────────────────────────
# DATA MODEL
# ─────────────────────────────────────────────────────────────────────────────

@dataclass
class EventRecord:
    ts:        float
    log_type:  str       # FIREWALL, AUTH, SYSMON, DNS
    actor_ip:  str
    stage:     str       # kill-chain stage
    summary:   str
    raw:       str


# ─────────────────────────────────────────────────────────────────────────────
# KILL-CHAIN STAGE ASSIGNMENT
# ─────────────────────────────────────────────────────────────────────────────

def assign_stage(log_type: str, kv: dict[str, str]) -> str:
    """Map event attributes to a kill-chain stage."""
    event   = kv.get("event", "").upper()
    action  = kv.get("action", "").upper()
    image   = kv.get("image", "").lower()
    cmdline = kv.get("cmdline", "").lower()
    dport   = kv.get("dport", "")
    query   = kv.get("query", "").lower()

    if log_type == "DNS":
        return "C2_COMMUNICATION"

    if log_type == "FIREWALL":
        if dport in ("22", "3389", "445") and action == "ALLOW":
            return "LATERAL_MOVEMENT"
        if action == "ALLOW":
            return "INITIAL_ACCESS"
        return "BLOCKED"

    if log_type == "AUTH":
        if "FAIL" in event:
            return "CREDENTIAL_ACCESS"
        if "SUCCESS" in event or "LOGIN" in event:
            return "INITIAL_ACCESS"

    if log_type == "SYSMON":
        if action == "process_create":
            if any(t in cmdline for t in ("whoami", "net user", "net group", "ipconfig", "systeminfo")):
                return "DISCOVERY"
            if any(t in cmdline for t in ("net user /add", "reg add", "schtasks", "at ")):
                return "PERSISTENCE"
            if any(t in image for t in ("mshta", "powershell", "wscript", "cscript", "certutil", "regsvr32")):
                return "EXECUTION"
            if any(t in cmdline for t in ("smb", "\\\\", "psexec", "wmic")):
                return "LATERAL_MOVEMENT"
            return "EXECUTION"

    return "UNKNOWN"


# ─────────────────────────────────────────────────────────────────────────────
# PARSERS
# ─────────────────────────────────────────────────────────────────────────────

KV_RE = re.compile(r'(\w+)=(\S+)')


def extract_kv(text: str) -> dict[str, str]:
    return dict(KV_RE.findall(text))


def parse_line(line: str) -> EventRecord | None:
    line = line.strip()
    if not line:
        return None
    parts = line.split(None, 2)
    if len(parts) < 3:
        return None

    log_type = parts[0].upper()
    if log_type not in ("FIREWALL", "AUTH", "SYSMON", "DNS"):
        return None

    try:
        ts = datetime.strptime(parts[1], "%Y-%m-%dT%H:%M:%SZ").timestamp()
    except ValueError:
        return None

    rest = parts[2]
    kv   = extract_kv(rest)

    # Determine actor IP
    actor_ip = (kv.get("src") or kv.get("src_ip") or "?").split("/")[0]

    stage   = assign_stage(log_type, kv)
    summary = rest[:100]

    return EventRecord(
        ts=ts, log_type=log_type, actor_ip=actor_ip,
        stage=stage, summary=summary, raw=line,
    )


# ─────────────────────────────────────────────────────────────────────────────
# CORRELATION ENGINE
# ─────────────────────────────────────────────────────────────────────────────

STAGE_ORDER = [
    "INITIAL_ACCESS", "CREDENTIAL_ACCESS", "EXECUTION",
    "PERSISTENCE", "DISCOVERY", "LATERAL_MOVEMENT",
    "C2_COMMUNICATION", "BLOCKED", "UNKNOWN",
]


def correlate_and_timeline(records: list[EventRecord]) -> dict[str, list[EventRecord]]:
    """Group events by actor IP, sorted by time."""
    timeline: dict[str, list[EventRecord]] = defaultdict(list)
    for r in sorted(records, key=lambda x: x.ts):
        timeline[r.actor_ip].append(r)
    return timeline


def detect_stage_progression(events: list[EventRecord]) -> list[str]:
    """Return ordered unique stages seen for this actor."""
    seen = []
    for e in events:
        if e.stage not in seen:
            seen.append(e.stage)
    return seen


# ─────────────────────────────────────────────────────────────────────────────
# REPORT
# ─────────────────────────────────────────────────────────────────────────────

STAGE_COLORS = {
    "INITIAL_ACCESS":   ">>> INITIAL_ACCESS",
    "CREDENTIAL_ACCESS":">>> CREDENTIAL_ACCESS",
    "EXECUTION":        ">>> EXECUTION",
    "PERSISTENCE":      ">>> PERSISTENCE",
    "DISCOVERY":        ">>> DISCOVERY",
    "LATERAL_MOVEMENT": "!!! LATERAL_MOVEMENT",
    "C2_COMMUNICATION": "!!! C2_COMMUNICATION",
    "BLOCKED":          "    BLOCKED",
    "UNKNOWN":          "    UNKNOWN",
}


def ts_fmt(ts: float) -> str:
    return datetime.utcfromtimestamp(ts).strftime("%Y-%m-%dT%H:%M:%SZ")


def print_report(timeline: dict[str, list[EventRecord]]) -> None:
    print("=" * 70)
    print("MULTI-LOG CORRELATED KILL-CHAIN TIMELINE")
    print("=" * 70)

    for actor_ip, events in sorted(timeline.items(),
                                   key=lambda x: x[1][0].ts):
        if len(events) < 2:
            continue  # filter single-event IPs (not an attack chain)

        stages = detect_stage_progression(events)
        advanced_stages = {"LATERAL_MOVEMENT", "C2_COMMUNICATION", "EXECUTION",
                           "PERSISTENCE", "DISCOVERY"}
        if not advanced_stages.intersection(set(stages)):
            continue  # filter noise

        print(f"\nACTOR: {actor_ip}  stages={' → '.join(stages)}")
        print("-" * 60)
        for e in events:
            label = STAGE_COLORS.get(e.stage, e.stage)
            print(f"  {ts_fmt(e.ts)}  [{e.log_type:8s}]  {label}")
            print(f"    {e.summary}")
    print("=" * 70)
