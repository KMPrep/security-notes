"""
Q21. API Gateway Credential Stuffing Detector (JWT-based)
===========================================================
Detect credential stuffing attacks, low-and-slow distributed campaigns,
post-burst success, and latency anomalies from API gateway JWT logs.

=============================================================================
SOLUTION PROGRESSION
=============================================================================

NAIVE: Rate-limit per IP.
  Problem: Distributed stuffing uses thousands of IPs — each IP sends
  only 1-3 requests. Rate limiting per IP misses the entire campaign.

BETTER: Track failure rate per username, alert above threshold.
  Problem: "Low and slow distributed" spreads across many usernames too.
  Need to look at account-level AND global failure rate AND the post-burst
  success (attacker found valid creds after burst from a single IP).

GOOGLE-LEVEL (this file):
  - Per-IP burst: many failures from one IP in a sliding 60s window.
  - Distributed: total global unique-username failures across all IPs in
    5-minute window > threshold — detects campaign without any single hot IP.
  - Post-burst success: an IP that had ≥ N failures suddenly succeeds.
  - Latency anomaly: successful login whose latency is significantly lower
    than the account's historical average (pre-computed password check
    might explain this — or credential stuffing tool bypassing 2FA).

Complexity: O(N) with per-IP and per-user sliding window accounting.
=============================================================================
"""

# ─────────────────────────────────────────────────────────────────────────────
# CONFIG
# ─────────────────────────────────────────────────────────────────────────────

BURST_WINDOW_SECS     = 60
BURST_FAIL_THRESHOLD  = 5    # per-IP failures in window

DISTRIBUTED_WINDOW    = 300  # 5-minute window
DISTRIBUTED_THRESHOLD = 8    # distinct usernames targeted globally

POST_BURST_FAIL_MIN   = 3    # IP needs this many failures before success counts

LATENCY_STDDEV_FACTOR = 2.0  # flag if success latency < mean - 2*stddev

# ─────────────────────────────────────────────────────────────────────────────
# PARSER
# ─────────────────────────────────────────────────────────────────────────────

LOG_RE = re.compile(
    r'(?P<ts>\S+)\s+\S+\s+\S+\s+(?P<status>\d+)\s+'
    r'src=(?P<src>\S+)\s+user=(?P<user>\S+)\s+lat_ms=(?P<lat>\d+)'
    r'(?:\s+jwt=(?P<jwt>\S+))?'
)


def parse_line(line: str) -> dict | None:
    line = line.strip()
    if not line:
        return None
    m = LOG_RE.match(line)
    if not m:
        return None
    try:
        ts = datetime.strptime(m.group("ts"), "%Y-%m-%dT%H:%M:%SZ").timestamp()
    except ValueError:
        return None
    return {
        "ts":     ts,
        "status": int(m.group("status")),
        "src":    m.group("src"),
        "user":   m.group("user"),
        "lat_ms": int(m.group("lat")),
        "jwt":    m.group("jwt"),
    }


# ─────────────────────────────────────────────────────────────────────────────
# DETECTORS
# ─────────────────────────────────────────────────────────────────────────────

def detect_per_ip_burst(records: list[dict]) -> list[dict]:
    """Sliding window count of 401s per source IP."""
    ip_window: dict[str, deque] = defaultdict(deque)
    alerts = []
    alerted: set[str] = set()

    for r in sorted(records, key=lambda x: x["ts"]):
        if r["status"] != 401:
            continue
        ip  = r["src"]
        win = ip_window[ip]
        win.append(r["ts"])
        cutoff = r["ts"] - BURST_WINDOW_SECS
        while win and win[0] < cutoff:
            win.popleft()
        if len(win) >= BURST_FAIL_THRESHOLD and ip not in alerted:
            alerts.append({
                "type":  "PER_IP_BURST",
                "src":   ip,
                "count": len(win),
                "ts":    r["ts"],
            })
            alerted.add(ip)
    return alerts


def detect_distributed_stuffing(records: list[dict]) -> list[dict]:
    """Global count of distinct usernames targeted in sliding 5-minute window."""
    sorted_r = sorted(records, key=lambda x: x["ts"])
    idx      = 0
    window: deque = deque()  # (ts, user)
    alerts = []
    alerted_until = 0.0

    for r in sorted_r:
        if r["status"] != 401:
            continue
        window.append((r["ts"], r["user"]))
        cutoff = r["ts"] - DISTRIBUTED_WINDOW
        while window and window[0][0] < cutoff:
            window.popleft()
        distinct_users = len({u for _, u in window})
        if distinct_users >= DISTRIBUTED_THRESHOLD and r["ts"] > alerted_until:
            alerts.append({
                "type":           "DISTRIBUTED_STUFFING",
                "distinct_users": distinct_users,
                "window_secs":    DISTRIBUTED_WINDOW,
                "ts":             r["ts"],
            })
            alerted_until = r["ts"] + DISTRIBUTED_WINDOW  # suppress for next window
    return alerts


def detect_post_burst_success(records: list[dict]) -> list[dict]:
    """IP that accumulated failures then suddenly logged in successfully."""
    ip_fails: dict[str, int] = defaultdict(int)
    alerts = []

    for r in sorted(records, key=lambda x: x["ts"]):
        ip = r["src"]
        if r["status"] == 401:
            ip_fails[ip] += 1
        elif r["status"] == 200 and ip_fails[ip] >= POST_BURST_FAIL_MIN:
            alerts.append({
                "type":       "POST_BURST_SUCCESS",
                "src":        ip,
                "user":       r["user"],
                "fail_count": ip_fails[ip],
                "ts":         r["ts"],
            })
    return alerts


def detect_latency_anomaly(records: list[dict]) -> list[dict]:
    """
    Success latency significantly below historical mean — could indicate
    a bypassed authentication check (stuffing tool hitting a cached path).
    """
    # Build baseline from successful logins per user (first pass)
    user_latencies: dict[str, list[int]] = defaultdict(list)
    for r in records:
        if r["status"] == 200:
            user_latencies[r["user"]].append(r["lat_ms"])

    def stats(lats: list[int]) -> tuple[float, float]:
        n = len(lats)
        if n < 2:
            return 0.0, 0.0
        mean = sum(lats) / n
        var  = sum((x - mean) ** 2 for x in lats) / n
        return mean, math.sqrt(var)

    alerts = []
    for r in records:
        if r["status"] != 200:
            continue
        lats = user_latencies[r["user"]]
        if len(lats) < 2:
            continue
        mean, stddev = stats(lats)
        threshold = mean - LATENCY_STDDEV_FACTOR * stddev
        if stddev > 0 and r["lat_ms"] < threshold:
            alerts.append({
                "type":      "LATENCY_ANOMALY",
                "user":      r["user"],
                "src":       r["src"],
                "lat_ms":    r["lat_ms"],
                "mean_ms":   round(mean, 1),
                "stddev_ms": round(stddev, 1),
                "ts":        r["ts"],
            })
    return alerts

def ts_fmt(ts: float) -> str:
    return datetime.utcfromtimestamp(ts).strftime("%Y-%m-%dT%H:%M:%SZ")


def print_report(burst, distributed, post_burst, latency) -> None:
    print("=" * 65)
    print("API GATEWAY CREDENTIAL STUFFING DETECTION REPORT")
    print("=" * 65)

    print(f"\n[PER_IP_BURST] {len(burst)} alert(s)  (≥{BURST_FAIL_THRESHOLD} fails/{BURST_WINDOW_SECS}s per IP)")
    for a in burst:
        print(f"  src={a['src']}  failures={a['count']}  ts={ts_fmt(a['ts'])}")

    print(f"\n[DISTRIBUTED_STUFFING] {len(distributed)} alert(s)  "
          f"(≥{DISTRIBUTED_THRESHOLD} distinct users targeted in {DISTRIBUTED_WINDOW}s)")
    for a in distributed:
        print(f"  distinct_users={a['distinct_users']}  ts={ts_fmt(a['ts'])}")

    print(f"\n[POST_BURST_SUCCESS] {len(post_burst)} alert(s)")
    for a in post_burst:
        print(f"  [CRITICAL] src={a['src']}  user={a['user']}  "
              f"prior_failures={a['fail_count']}  ts={ts_fmt(a['ts'])}")

    print(f"\n[LATENCY_ANOMALY] {len(latency)} alert(s)")
    for a in latency:
        print(f"  user={a['user']}  src={a['src']}  lat={a['lat_ms']}ms  "
              f"mean={a['mean_ms']}ms  stddev={a['stddev_ms']}ms  ts={ts_fmt(a['ts'])}")
    print("=" * 65)
