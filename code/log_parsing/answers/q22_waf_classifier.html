"""
Q22. WAF Log Classifier
=========================
Detect SQL injection, XSS, path traversal, command injection, and
common WAF bypass techniques from web application firewall logs.

=============================================================================
SOLUTION PROGRESSION
=============================================================================

NAIVE: Grep for "UNION SELECT" and "script>".
  Problem: Attackers URL-encode, hex-encode, use case variation, insert
  SQL comments, or split payloads across parameters. A static string
  grep will have both false negatives (bypasses) and false positives.

BETTER: Use a multi-pattern regex per attack type.
  Problem: Doesn't handle URL/double-URL encoding, nor does it detect
  bypass techniques like `UNI/**/ON` or `<scr%00ipt>`.

GOOGLE-LEVEL (this file):
  - Normalize: URL-decode (up to 3 levels), strip null bytes, collapse
    repeated whitespace/comments before matching.
  - Category patterns: SQLi, XSS, path traversal, command injection.
  - Bypass patterns: SQL comment injection, double encoding, null byte,
    case-scrambling indicators (mixed-case SELECT).
  - Score per event: additive weights; report sorted by severity.

Complexity: O(N × P) where P = number of patterns.
=============================================================================
"""

import re
import sys
from urllib.parse import unquote_plus
from collections import defaultdict
from datetime import datetime

# ─────────────────────────────────────────────────────────────────────────────
# SAMPLE DATA
# ─────────────────────────────────────────────────────────────────────────────

# ─────────────────────────────────────────────────────────────────────────────
# DETECTION PATTERNS  (name, pattern, weight, description)
# ─────────────────────────────────────────────────────────────────────────────

# These are applied AFTER normalization of the input
ATTACK_PATTERNS: list[tuple[str, re.Pattern, int, str]] = [
    # SQL Injection
    ("SQLI_UNION",
     re.compile(r'\bunion\s+(?:all\s+)?select\b', re.IGNORECASE), 5,
     "UNION-based SQL injection"),
    ("SQLI_BOOLEAN",
     re.compile(r"'\s*(?:or|and)\s+['\"0-9]+=", re.IGNORECASE), 4,
     "Boolean-based blind SQLi"),
    ("SQLI_COMMENT_BYPASS",
     re.compile(r'/\*[^*]*\*/\s*(union|select|or|and|from|where)', re.IGNORECASE), 4,
     "SQL comment injection bypass"),
    ("SQLI_TAUTOLOGY",
     re.compile(r"'\s*or\s+(?:'|\")?\s*\d+\s*=\s*\d+\s*(?:--|#|/\*)", re.IGNORECASE), 5,
     "Tautology SQLi"),
    ("SQLI_STACKED",
     re.compile(r";\s*(drop|insert|update|delete|create|exec)\s", re.IGNORECASE), 5,
     "Stacked query SQLi"),

    # XSS
    ("XSS_SCRIPT_TAG",
     re.compile(r'<script[\s>]', re.IGNORECASE), 5,
     "Script tag XSS"),
    ("XSS_EVENT_HANDLER",
     re.compile(r'\bon(?:load|error|click|mouseover|focus|blur)\s*=', re.IGNORECASE), 4,
     "DOM event handler XSS"),
    ("XSS_JAVASCRIPT_URI",
     re.compile(r'javascript\s*:', re.IGNORECASE), 4,
     "JavaScript URI XSS"),
    ("XSS_EXPRESSION",
     re.compile(r'expression\s*\(', re.IGNORECASE), 3,
     "CSS expression XSS (IE)"),

    # Path Traversal
    ("PATH_TRAVERSAL",
     re.compile(r'\.\.(?:/|\\)', re.IGNORECASE), 4,
     "Directory traversal"),
    ("PATH_TRAVERSAL_ENCODED",
     re.compile(r'%2e%2e(?:%2f|%5c)', re.IGNORECASE), 4,
     "Encoded directory traversal"),
    ("SENSITIVE_FILE",
     re.compile(r'(?:etc/passwd|etc/shadow|win\.ini|boot\.ini|web\.config)', re.IGNORECASE), 5,
     "Sensitive file access attempt"),

    # Command Injection
    ("CMD_PIPE",
     re.compile(r'[;|`]\s*(?:cat|ls|id|whoami|wget|curl|nc|bash|sh)\b', re.IGNORECASE), 5,
     "Command injection via pipe/semicolon"),
    ("CMD_SUBSHELL",
     re.compile(r'\$\(.*?\)|\`[^`]+\`', re.IGNORECASE), 4,
     "Command substitution injection"),
]

# Bypass patterns (applied separately, additive score)
BYPASS_PATTERNS: list[tuple[str, re.Pattern, int, str]] = [
    ("BYPASS_NULL_BYTE",
     re.compile(r'\x00|%00'), 2,
     "Null byte WAF bypass attempt"),
    ("BYPASS_DOUBLE_ENCODE",
     re.compile(r'%25[0-9a-f]{2}', re.IGNORECASE), 2,
     "Double URL encoding bypass"),
    ("BYPASS_CASE_SCRAMBLE",
     re.compile(r'(?:sElEcT|uNiOn|ScRiPt|aLeRt)', re.IGNORECASE), 2,
     "Case-scrambling bypass (mixed case)"),
]

# ─────────────────────────────────────────────────────────────────────────────
# NORMALIZER
# ─────────────────────────────────────────────────────────────────────────────

NULL_BYTE_RE     = re.compile(r'\x00|%00', re.IGNORECASE)
SQL_COMMENT_PASS = re.compile(r'/\*!\d*\s*', re.IGNORECASE)  # MySQL inline comments


def normalize(text: str) -> str:
    """Decode URL encoding up to 3 levels, strip null bytes, collapse whitespace."""
    for _ in range(3):
        decoded = unquote_plus(text)
        if decoded == text:
            break
        text = decoded
    text = NULL_BYTE_RE.sub("", text)
    text = re.sub(r'\s+', ' ', text)
    return text


# ─────────────────────────────────────────────────────────────────────────────
# PARSER
# ─────────────────────────────────────────────────────────────────────────────

LOG_RE = re.compile(
    r'(?P<ts>\S+)\s+src=(?P<src>\S+)\s+method=(?P<method>\S+)\s+'
    r'path=(?P<path>\S+)\s+status=(?P<status>\d+)\s+waf=(?P<waf>\S+)'
)


def parse_line(line: str) -> dict | None:
    line = line.strip()
    if not line:
        return None
    m = LOG_RE.match(line)
    if not m:
        return None
    try:
        ts = datetime.strptime(m.group("ts"), "%Y-%m-%dT%H:%M:%SZ").timestamp()
    except ValueError:
        return None
    return {
        "ts":     ts,
        "src":    m.group("src"),
        "method": m.group("method"),
        "path":   m.group("path"),
        "status": int(m.group("status")),
        "waf":    m.group("waf"),
    }


# ─────────────────────────────────────────────────────────────────────────────
# CLASSIFIER
# ─────────────────────────────────────────────────────────────────────────────

def classify(record: dict) -> dict:
    raw  = record["path"]
    norm = normalize(raw)

    matched_attacks: list[dict] = []
    matched_bypasses: list[dict] = []
    score = 0

    for name, pattern, weight, description in ATTACK_PATTERNS:
        if pattern.search(norm):
            matched_attacks.append({"name": name, "description": description, "weight": weight})
            score += weight

    for name, pattern, weight, description in BYPASS_PATTERNS:
        # Check bypass on raw (before decoding) to catch encoding tricks
        if pattern.search(raw):
            matched_bypasses.append({"name": name, "description": description, "weight": weight})
            score += weight

    return {
        **record,
        "attacks":  matched_attacks,
        "bypasses": matched_bypasses,
        "score":    score,
        "norm":     norm[:200],
    }


def risk_label(score: int, waf: str) -> str:
    label = "LOW"
    if score >= 8:
        label = "CRITICAL"
    elif score >= 5:
        label = "HIGH"
    elif score >= 3:
        label = "MEDIUM"
    if waf == "PASS" and score > 0:
        label += "+BYPASS"
    return label


# ─────────────────────────────────────────────────────────────────────────────
# REPORT
# ─────────────────────────────────────────────────────────────────────────────

def ts_fmt(ts: float) -> str:
    return datetime.utcfromtimestamp(ts).strftime("%Y-%m-%dT%H:%M:%SZ")


def print_report(results: list[dict]) -> None:
    print("=" * 65)
    print("WAF LOG CLASSIFICATION REPORT")
    print("=" * 65)

    flagged = [r for r in results if r["score"] > 0]
    flagged.sort(key=lambda x: x["score"], reverse=True)

    bypassed = [r for r in flagged if r["waf"] == "PASS"]
    blocked  = [r for r in flagged if r["waf"] == "BLOCK"]

    print(f"\n[BYPASSED WAF] {len(bypassed)} event(s)  *** Needs immediate attention ***")
    for r in bypassed:
        label = risk_label(r["score"], r["waf"])
        print(f"\n  [{label}] src={r['src']}  ts={ts_fmt(r['ts'])}")
        print(f"  path: {r['path'][:80]}")
        print(f"  norm: {r['norm'][:80]}")
        for a in r["attacks"] + r["bypasses"]:
            print(f"    [{a['name']}] {a['description']}  weight={a['weight']}")

    print(f"\n[BLOCKED by WAF] {len(blocked)} event(s)")
    for r in blocked:
        label = risk_label(r["score"], r["waf"])
        print(f"  [{label}] src={r['src']}  score={r['score']}  "
              f"attacks={[a['name'] for a in r['attacks']]}  ts={ts_fmt(r['ts'])}")
    print("=" * 65)

if __name__ == "__main__":
    if len(sys.argv) > 1:
        with open(sys.argv[1], encoding="utf-8", errors="replace") as fh:
            raw_lines = fh.readlines()
    else:
        print("[INFO] Using embedded sample data.\n")
        raw_lines = SAMPLE_LOG.splitlines(keepends=True)

    records = [r for r in (parse_line(l) for l in raw_lines) if r is not None]
    print(f"Parsed {len(records)} WAF log entries.\n")
    results = [classify(r) for r in records]
    print_report(results)
