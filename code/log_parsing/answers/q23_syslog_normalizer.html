"""
Q23. Universal Syslog Normalizer
==================================
Parse five different log formats (RFC 5424, RFC 3164, JSON, Windows
key=value, Apache CLF) and output a unified JSON record per event.

=============================================================================
SOLUTION PROGRESSION
=============================================================================

NAIVE: Parse only Apache logs with a regex.
  Problem: Real SOC environments receive logs from Linux syslog, Windows
  Event Log, structured JSON from cloud services, and web servers — all
  in the same pipeline. A single-format parser is useless.

BETTER: Write a parser per format, select by filename.
  Problem: In a SIEM stream you don't know the format in advance.
  Need to auto-detect the format based on the first character / schema.

GOOGLE-LEVEL (this file):
  - Auto-detect: each parser returns None if the line doesn't match its format.
  - Ordered probe: cheapest/most-specific first, catch-all last.
  - Output: unified schema with ts (epoch float), host, source_type,
    severity, message, and extra dict for format-specific fields.
  - Handles malformed lines without crashing.

Complexity: O(N × F) worst-case where F = 5 formats.
=============================================================================
"""

import re
import sys
import json
from datetime import datetime, timezone
from typing import Any

# ─────────────────────────────────────────────────────────────────────────────
# SAMPLE DATA  (mixed formats in one stream)
# ─────────────────────────────────────────────────────────────────────────────

# ─────────────────────────────────────────────────────────────────────────────
# UNIFIED SCHEMA
# ─────────────────────────────────────────────────────────────────────────────

def make_record(
    ts: float,
    source_type: str,
    host: str,
    severity: str,
    message: str,
    extra: dict[str, Any] | None = None,
) -> dict:
    return {
        "ts":          ts,
        "ts_iso":      datetime.fromtimestamp(ts, tz=timezone.utc).strftime("%Y-%m-%dT%H:%M:%SZ"),
        "source_type": source_type,
        "host":        host,
        "severity":    severity,
        "message":     message,
        "extra":       extra or {},
    }


# ─────────────────────────────────────────────────────────────────────────────
# PARSERS
# ─────────────────────────────────────────────────────────────────────────────

# ── RFC 5424 ─────────────────────────────────────────────────────────────────
# <PRI>VERSION TIMESTAMP HOST APP PID MSGID STRUCTURED-DATA MSG
RFC5424_RE = re.compile(
    r'^<(?P<pri>\d+)>(?P<ver>\d+)\s+'
    r'(?P<ts>\S+)\s+(?P<host>\S+)\s+(?P<app>\S+)\s+'
    r'(?P<pid>\S+)\s+(?P<msgid>\S+)\s+(?P<sd>\S+)\s*(?P<msg>.*)'
)

SEVERITY_NAMES = ["EMERGENCY","ALERT","CRITICAL","ERROR","WARNING","NOTICE","INFO","DEBUG"]


def parse_rfc5424(line: str) -> dict | None:
    m = RFC5424_RE.match(line)
    if not m:
        return None
    try:
        ts_str = m.group("ts").rstrip("Z")
        # Handle optional fractional seconds
        for fmt in ("%Y-%m-%dT%H:%M:%S.%f", "%Y-%m-%dT%H:%M:%S"):
            try:
                ts = datetime.strptime(ts_str, fmt).replace(tzinfo=timezone.utc).timestamp()
                break
            except ValueError:
                continue
        else:
            return None
    except Exception:
        return None

    pri      = int(m.group("pri"))
    sev_idx  = pri % 8
    severity = SEVERITY_NAMES[sev_idx]
    return make_record(
        ts=ts,
        source_type="RFC5424",
        host=m.group("host"),
        severity=severity,
        message=m.group("msg").strip(),
        extra={"app": m.group("app"), "pid": m.group("pid"), "msgid": m.group("msgid")},
    )


# ── RFC 3164 ─────────────────────────────────────────────────────────────────
# <PRI>Mon DD HH:MM:SS host process[pid]: message
RFC3164_RE = re.compile(
    r'^<(?P<pri>\d+)>(?P<ts>\w{3}\s+\d+\s+\d{2}:\d{2}:\d{2})\s+'
    r'(?P<host>\S+)\s+(?P<proc>[^:\[]+)(?:\[(?P<pid>\d+)\])?:\s*(?P<msg>.*)'
)
# Also handles the bare form without PRI
RFC3164_BARE_RE = re.compile(
    r'^(?P<ts>\w{3}\s+\d+\s+\d{2}:\d{2}:\d{2})\s+'
    r'(?P<host>\S+)\s+(?P<proc>[^:\[]+)(?:\[(?P<pid>\d+)\])?:\s*(?P<msg>.*)'
)

MONTH_MAP = {"Jan":1,"Feb":2,"Mar":3,"Apr":4,"May":5,"Jun":6,
             "Jul":7,"Aug":8,"Sep":9,"Oct":10,"Nov":11,"Dec":12}
RFC3164_YEAR = 2024


def _parse_rfc3164_ts(ts_str: str) -> float | None:
    parts = ts_str.split()
    if len(parts) != 3:
        return None
    try:
        month = MONTH_MAP.get(parts[0], 0)
        day   = int(parts[1])
        h, mi, s = map(int, parts[2].split(":"))
        return datetime(RFC3164_YEAR, month, day, h, mi, s,
                        tzinfo=timezone.utc).timestamp()
    except (ValueError, KeyError):
        return None


def parse_rfc3164(line: str) -> dict | None:
    m = RFC3164_RE.match(line) or RFC3164_BARE_RE.match(line)
    if not m:
        return None
    ts = _parse_rfc3164_ts(m.group("ts"))
    if ts is None:
        return None
    pri      = int(m.group("pri")) if "pri" in m.groupdict() and m.group("pri") else 14
    sev_idx  = pri % 8
    severity = SEVERITY_NAMES[sev_idx]
    return make_record(
        ts=ts,
        source_type="RFC3164",
        host=m.group("host"),
        severity=severity,
        message=m.group("msg").strip(),
        extra={"process": m.group("proc").strip(), "pid": m.group("pid")},
    )


# ── JSON ──────────────────────────────────────────────────────────────────────
def parse_json(line: str) -> dict | None:
    if not line.startswith("{"):
        return None
    try:
        obj = json.loads(line)
    except json.JSONDecodeError:
        return None

    ts_str = obj.get("timestamp") or obj.get("time") or obj.get("@timestamp")
    if not ts_str:
        return None
    try:
        ts = datetime.strptime(ts_str, "%Y-%m-%dT%H:%M:%SZ").replace(tzinfo=timezone.utc).timestamp()
    except ValueError:
        return None

    level = (obj.get("level") or obj.get("severity") or "INFO").upper()
    msg   = obj.get("message") or obj.get("msg") or ""
    host  = obj.get("host") or obj.get("hostname") or "?"
    extra = {k: v for k, v in obj.items() if k not in ("timestamp","time","@timestamp","level","severity","message","msg","host","hostname")}
    return make_record(ts=ts, source_type="JSON", host=host, severity=level, message=msg, extra=extra)


# ── Windows key=value ─────────────────────────────────────────────────────────
WINDOWS_KV_RE = re.compile(r'EventID=\d+.*TimeCreated=', re.IGNORECASE)
KV_PAIR_RE    = re.compile(r'(\w+)=(\S+)')


def parse_windows_kv(line: str) -> dict | None:
    if not WINDOWS_KV_RE.search(line):
        return None
    kv = dict(KV_PAIR_RE.findall(line))
    ts_str = kv.get("TimeCreated", "")
    try:
        ts = datetime.strptime(ts_str, "%Y-%m-%dT%H:%M:%SZ").replace(tzinfo=timezone.utc).timestamp()
    except ValueError:
        return None

    # Map EventID to rough severity
    eid      = int(kv.get("EventID", "0"))
    severity = "WARNING" if eid in (4624, 4625, 4648, 4768, 4769) else "INFO"
    msg      = f"EventID={eid} User={kv.get('UserName','?')} Status={kv.get('Status','?')}"
    return make_record(
        ts=ts, source_type="WINDOWS_KV",
        host=kv.get("ComputerName", "?"),
        severity=severity,
        message=msg,
        extra={k: v for k, v in kv.items() if k not in ("TimeCreated",)},
    )


# ── Apache CLF ────────────────────────────────────────────────────────────────
APACHE_CLF_RE = re.compile(
    r'^(?P<ip>\S+)\s+\S+\s+(?P<user>\S+)\s+'
    r'\[(?P<ts>[^\]]+)\]\s+"(?P<req>[^"]+)"\s+(?P<status>\d+)\s+(?P<bytes>\S+)'
)
CLF_TIME_FMT = "%d/%b/%Y:%H:%M:%S %z"


def parse_apache_clf(line: str) -> dict | None:
    m = APACHE_CLF_RE.match(line)
    if not m:
        return None
    try:
        ts = datetime.strptime(m.group("ts"), CLF_TIME_FMT).timestamp()
    except ValueError:
        return None
    status   = int(m.group("status"))
    severity = "WARNING" if status >= 400 else "INFO"
    return make_record(
        ts=ts, source_type="APACHE_CLF",
        host=m.group("ip"),
        severity=severity,
        message=m.group("req"),
        extra={"user": m.group("user"), "status": status, "bytes": m.group("bytes")},
    )


# ─────────────────────────────────────────────────────────────────────────────
# PIPELINE
# ─────────────────────────────────────────────────────────────────────────────

PARSERS = [parse_json, parse_windows_kv, parse_rfc5424, parse_rfc3164, parse_apache_clf]


def parse_any(line: str) -> dict | None:
    line = line.strip()
    if not line:
        return None
    for parser in PARSERS:
        result = parser(line)
        if result is not None:
            return result
    return None


# ─────────────────────────────────────────────────────────────────────────────
# REPORT
# ─────────────────────────────────────────────────────────────────────────────

def print_report(records: list[dict], raw_count: int) -> None:
    print("=" * 65)
    print("UNIVERSAL SYSLOG NORMALIZATION REPORT")
    print("=" * 65)
    print(f"\nInput lines: {raw_count}   Parsed: {len(records)}   "
          f"Failed: {raw_count - len(records)}")

    by_type: dict[str, int] = {}
    for r in records:
        by_type[r["source_type"]] = by_type.get(r["source_type"], 0) + 1
    print("\nFormat breakdown:")
    for fmt, count in sorted(by_type.items()):
        print(f"  {fmt:15s} {count}")

    print("\nSample normalized records (sorted by timestamp):")
    for r in sorted(records, key=lambda x: x["ts"])[:20]:
        extra_preview = ", ".join(f"{k}={v}" for k, v in list(r["extra"].items())[:3])
        print(f"  [{r['source_type']:12s}] [{r['severity']:10s}]  "
              f"ts={r['ts_iso']}  host={r['host']}")
        print(f"    msg: {r['message'][:80]}")
        if extra_preview:
            print(f"    extra: {extra_preview}")
    print("=" * 65)

if __name__ == "__main__":
    if len(sys.argv) > 1:
        with open(sys.argv[1], encoding="utf-8", errors="replace") as fh:
            raw_lines = fh.readlines()
    else:
        print("[INFO] Using embedded sample data.\n")
        raw_lines = SAMPLE_LOG.splitlines(keepends=True)

    stripped = [l for l in raw_lines if l.strip()]
    records  = [r for r in (parse_any(l) for l in stripped) if r is not None]
    print_report(records, len(stripped))
