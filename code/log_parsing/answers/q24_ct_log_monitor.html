"""
Q24. Certificate Transparency (CT) Log Monitor
================================================
Detect unexpected CAs, wildcard certs, short-lived certificates,
SAN mismatches, and lookalike domains from CT log entries.

=============================================================================
SOLUTION PROGRESSION
=============================================================================

NAIVE: Alert if a certificate is issued by an unknown CA.
  Problem: Hundreds of CAs are trusted. Maintaining the list is hard.
  More importantly, this misses the common attack: an *authorized* CA
  issuing a cert for a lookalike domain (g00gle.com, paypa1.com).

BETTER: Flag short-lived certs and wildcard certs.
  Problem: Short-lived certs are increasingly the norm (Let's Encrypt).
  Need to combine signals: short-lived + lookalike + unexpected CA.
  Also need SAN mismatch detection (CN and SANs disagree — may indicate
  misconfiguration or cert transparency abuse).

GOOGLE-LEVEL (this file):
  - Lookalike detection: Levenshtein distance ≤ 2 against known-good domains.
  - Wildcard: *.example.com → over-permissive; track by issuer.
  - Unexpected CA: CA not in approved list.
  - Short-lived: validity < 7 days.
  - SAN vs CN mismatch: CN not in SANs.
  - Risk scoring: additive, sorted output.

Complexity: O(N × D) for Levenshtein where D = approved domain count.
=============================================================================
"""

import json
import sys
import re
from datetime import datetime, timezone
from collections import defaultdict

# ─────────────────────────────────────────────────────────────────────────────
# SAMPLE DATA  (simplified CT log, one JSON per line)
# ─────────────────────────────────────────────────────────────────────────────

SAMPLE_LOG = r"""

# ─────────────────────────────────────────────────────────────────────────────
# CONFIG
# ─────────────────────────────────────────────────────────────────────────────

APPROVED_CAS = {
    "Let's Encrypt",
    "DigiCert",
    "Sectigo",
    "GlobalSign",
    "Entrust",
    "Amazon",
}

# Your protected/monitored domains
PROTECTED_DOMAINS = {
    "google.com", "paypal.com", "mybank.com", "example.com",
    "microsoft.com", "amazon.com", "apple.com",
}

SHORT_LIVED_DAYS = 7
LOOKALIKE_DISTANCE = 2  # Levenshtein distance threshold


# ─────────────────────────────────────────────────────────────────────────────
# LEVENSHTEIN DISTANCE
# ─────────────────────────────────────────────────────────────────────────────

def levenshtein(s1: str, s2: str) -> int:
    if len(s1) < len(s2):
        return levenshtein(s2, s1)
    if len(s2) == 0:
        return len(s1)
    prev = list(range(len(s2) + 1))
    for i, c1 in enumerate(s1):
        curr = [i + 1]
        for j, c2 in enumerate(s2):
            curr.append(min(prev[j + 1] + 1, curr[j] + 1, prev[j] + (c1 != c2)))
        prev = curr
    return prev[-1]


def extract_sld(domain: str) -> str:
    """Return second-level domain (e.g., 'google' from 'www.google.com')."""
    parts = domain.strip("*. ").split(".")
    if len(parts) >= 2:
        return parts[-2] + "." + parts[-1]
    return domain


# ─────────────────────────────────────────────────────────────────────────────
# PARSER
# ─────────────────────────────────────────────────────────────────────────────

def parse_event(line: str) -> dict | None:
    line = line.strip()
    if not line:
        return None
    try:
        obj = json.loads(line)
    except json.JSONDecodeError:
        return None
    required = ("not_before", "not_after", "issuer", "cn", "sans")
    if not all(k in obj for k in required):
        return None
    try:
        nb = datetime.strptime(obj["not_before"], "%Y-%m-%dT%H:%M:%SZ").replace(tzinfo=timezone.utc)
        na = datetime.strptime(obj["not_after"],  "%Y-%m-%dT%H:%M:%SZ").replace(tzinfo=timezone.utc)
    except ValueError:
        return None
    return {
        "idx":       obj.get("log_index", 0),
        "not_before": nb,
        "not_after":  na,
        "validity_days": (na - nb).days,
        "issuer":    obj["issuer"],
        "cn":        obj["cn"].lower(),
        "sans":      [s.lower() for s in obj["sans"]],
    }


# ─────────────────────────────────────────────────────────────────────────────
# DETECTORS
# ─────────────────────────────────────────────────────────────────────────────

def detect_unexpected_ca(records: list[dict]) -> list[dict]:
    return [
        {
            "type":   "UNEXPECTED_CA",
            "issuer": r["issuer"],
            "cn":     r["cn"],
            "idx":    r["idx"],
            "score":  3,
        }
        for r in records
        if r["issuer"] not in APPROVED_CAS
    ]


def detect_wildcard(records: list[dict]) -> list[dict]:
    return [
        {
            "type":   "WILDCARD_CERT",
            "issuer": r["issuer"],
            "cn":     r["cn"],
            "idx":    r["idx"],
            "score":  2,
        }
        for r in records
        if r["cn"].startswith("*") or any(s.startswith("*") for s in r["sans"])
    ]


def detect_short_lived(records: list[dict]) -> list[dict]:
    return [
        {
            "type":    "SHORT_LIVED",
            "cn":      r["cn"],
            "issuer":  r["issuer"],
            "days":    r["validity_days"],
            "idx":     r["idx"],
            "score":   2,
        }
        for r in records
        if 0 < r["validity_days"] < SHORT_LIVED_DAYS
    ]


def detect_san_mismatch(records: list[dict]) -> list[dict]:
    alerts = []
    for r in records:
        cn_clean = r["cn"].lstrip("*.")
        if cn_clean and not any(cn_clean in s or s.endswith("." + cn_clean) for s in r["sans"]):
            alerts.append({
                "type":  "SAN_CN_MISMATCH",
                "cn":    r["cn"],
                "sans":  r["sans"],
                "idx":   r["idx"],
                "score": 2,
            })
    return alerts


def detect_lookalike(records: list[dict]) -> list[dict]:
    alerts = []
    for r in records:
        cn_sld = extract_sld(r["cn"])
        for protected in PROTECTED_DOMAINS:
            protected_sld = extract_sld(protected)
            if cn_sld == protected_sld:
                continue  # exact match is fine
            dist = levenshtein(cn_sld.split(".")[0], protected_sld.split(".")[0])
            if dist <= LOOKALIKE_DISTANCE:
                alerts.append({
                    "type":     "LOOKALIKE_DOMAIN",
                    "cn":       r["cn"],
                    "target":   protected,
                    "distance": dist,
                    "issuer":   r["issuer"],
                    "idx":      r["idx"],
                    "score":    5,
                })
                break  # one alert per cert per detection type
    return alerts


# ─────────────────────────────────────────────────────────────────────────────
# REPORT
# ─────────────────────────────────────────────────────────────────────────────

def print_report(unexpected_ca, wildcard, short_lived, san_mismatch, lookalike) -> None:
    print("=" * 65)
    print("CT LOG MONITOR — SUSPICIOUS CERTIFICATE REPORT")
    print("=" * 65)

    all_alerts = unexpected_ca + wildcard + short_lived + san_mismatch + lookalike
    all_alerts.sort(key=lambda x: x["score"], reverse=True)

    counts = {
        "UNEXPECTED_CA": len(unexpected_ca),
        "WILDCARD_CERT": len(wildcard),
        "SHORT_LIVED":   len(short_lived),
        "SAN_CN_MISMATCH": len(san_mismatch),
        "LOOKALIKE_DOMAIN": len(lookalike),
    }
    for k, v in counts.items():
        print(f"  {k:20s} {v}")

    print(f"\n{'─' * 65}")
    for a in all_alerts:
        t = a["type"]
        print(f"\n  [{t}]  idx={a['idx']}  score={a['score']}")
        if t == "LOOKALIKE_DOMAIN":
            print(f"    cn={a['cn']}  target={a['target']}  distance={a['distance']}  issuer={a['issuer']}")
        elif t == "SHORT_LIVED":
            print(f"    cn={a['cn']}  days={a['days']}  issuer={a['issuer']}")
        elif t == "SAN_CN_MISMATCH":
            print(f"    cn={a['cn']}  sans={a['sans']}")
        else:
            print(f"    cn={a.get('cn','?')}  issuer={a.get('issuer','?')}")
    print("=" * 65)


# ─────────────────────────────────────────────────────────────────────────────
# MAIN
# ─────────────────────────────────────────────────────────────────────────────

if __name__ == "__main__":
    if len(sys.argv) > 1:
        with open(sys.argv[1], encoding="utf-8", errors="replace") as fh:
            raw_lines = fh.readlines()
    else:
        print("[INFO] Using embedded sample data.\n")
        raw_lines = SAMPLE_LOG.splitlines(keepends=True)

    records = [r for r in (parse_event(l) for l in raw_lines) if r is not None]
    print(f"Parsed {len(records)} CT log entries.\n")
    print_report(
        detect_unexpected_ca(records),
        detect_wildcard(records),
        detect_short_lived(records),
        detect_san_mismatch(records),
        detect_lookalike(records),
    )
