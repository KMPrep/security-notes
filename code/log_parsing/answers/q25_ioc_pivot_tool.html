"""
Q25. IOC Pivot & Cross-Reference Tool
=======================================
Scan a log corpus against an IOC list (IPs, domains, hashes, URLs).
Extract context (3 lines before/after each hit), deduplicate, and
produce a pivot report showing each IOC and all matching log lines.

=============================================================================
SOLUTION PROGRESSION
=============================================================================

NAIVE: grep -r evil.com .
  Problem: You'd need to run grep for each IOC. With 10,000 IOCs and
  50 log files, that's 500,000 subprocess invocations. Also, grep output
  doesn't give you context across multiple log types or deduplicated hits.

BETTER: Precompile all IOC patterns into one alternation regex.
  Problem: A single alternation regex with 10,000 patterns is slow and
  the backtracking makes it exponential in pathological cases.
  Better: Aho-Corasick for exact string matching (O(N + M + Z) where
  Z = match count). Python doesn't have it in stdlib, so we simulate
  the key property with a compiled set-lookup after line splitting.

GOOGLE-LEVEL (this file):
  - Normalize IOCs into typed buckets (IP, domain, hash, URL).
  - Per-line tokenization: extract IPs, domains, hashes from each log line
    and do O(1) set-lookup against the IOC buckets.
  - Context window: maintain a circular buffer of last N lines per file;
    emit N lines before + N lines after each hit.
  - Deduplication: (ioc, file, hit_line_no) tuple prevents reporting the
    same match from context of an adjacent hit.
  - Pivot report: IOC → list of {file, line, context}.

Complexity: O(N × T) where T = tokens per line (≈ 10), lookups O(1).
=============================================================================
"""

import sys
from collections import defaultdict, deque
from datetime import datetime
from pathlib import Path

# ─────────────────────────────────────────────────────────────────────────────
# CONFIG
# ─────────────────────────────────────────────────────────────────────────────

CONTEXT_LINES = 3  # lines before and after each match

# ─────────────────────────────────────────────────────────────────────────────
# IOC PARSER
# ─────────────────────────────────────────────────────────────────────────────

def load_iocs(text: str) -> dict[str, set[str]]:
    """
    Parse IOC list. Return dict with keys: ip, domain, hash, url.
    Auto-detects type if no prefix given.
    """
    buckets: dict[str, set[str]] = {
        "ip":     set(),
        "domain": set(),
        "hash":   set(),
        "url":    set(),
    }

    IP_RE   = re.compile(r'^\d{1,3}(?:\.\d{1,3}){3}$')
    HASH_RE = re.compile(r'^[0-9a-fA-F]{32,64}$')
    URL_RE  = re.compile(r'^https?://')

    for line in text.splitlines():
        line = line.strip()
        if not line or line.startswith("#"):
            continue

        if ":" in line and line.split(":", 1)[0].lower() in buckets:
            ioc_type, value = line.split(":", 1)
            buckets[ioc_type.lower()].add(value.strip().lower())
        else:
            value = line.lower()
            if IP_RE.match(value):
                buckets["ip"].add(value)
            elif URL_RE.match(value):
                buckets["url"].add(value)
            elif HASH_RE.match(value):
                buckets["hash"].add(value)
            else:
                buckets["domain"].add(value)

    return buckets


# ─────────────────────────────────────────────────────────────────────────────
# TOKENIZERS  (extract typed values from a log line)
# ─────────────────────────────────────────────────────────────────────────────

IP_TOKEN_RE    = re.compile(r'\b\d{1,3}(?:\.\d{1,3}){3}\b')
DOMAIN_TOKEN_RE = re.compile(r'\b(?:[a-z0-9](?:[a-z0-9-]{0,61}[a-z0-9])?\.)+[a-z]{2,}\b',
                              re.IGNORECASE)
HASH_TOKEN_RE  = re.compile(r'\b[0-9a-fA-F]{32,64}\b')
URL_TOKEN_RE   = re.compile(r'https?://\S+')


def extract_tokens(line: str) -> dict[str, set[str]]:
    lower = line.lower()
    return {
        "ip":     set(IP_TOKEN_RE.findall(lower)),
        "domain": set(DOMAIN_TOKEN_RE.findall(lower)),
        "hash":   set(HASH_TOKEN_RE.findall(lower)),
        "url":    set(URL_TOKEN_RE.findall(lower)),
    }


# ─────────────────────────────────────────────────────────────────────────────
# SCANNER
# ─────────────────────────────────────────────────────────────────────────────

def scan_lines(
    lines: list[str],
    ioc_buckets: dict[str, set[str]],
    source_name: str = "stream",
) -> list[dict]:
    """
    Scan lines against IOC buckets.
    Returns list of {ioc, ioc_type, source, line_no, line, context_before, context_after}.
    """
    results = []
    n = len(lines)
    already_matched: set[tuple] = set()  # (ioc, line_no) dedup

    for idx, line in enumerate(lines):
        tokens = extract_tokens(line)
        for ioc_type, ioc_set in ioc_buckets.items():
            hits = tokens[ioc_type].intersection(ioc_set)
            for ioc in hits:
                key = (ioc, idx)
                if key in already_matched:
                    continue
                already_matched.add(key)

                before = lines[max(0, idx - CONTEXT_LINES): idx]
                after  = lines[idx + 1: min(n, idx + CONTEXT_LINES + 1)]

                results.append({
                    "ioc":            ioc,
                    "ioc_type":       ioc_type,
                    "source":         source_name,
                    "line_no":        idx + 1,
                    "line":           line.strip(),
                    "context_before": [l.strip() for l in before],
                    "context_after":  [l.strip() for l in after],
                })

    return results


# ─────────────────────────────────────────────────────────────────────────────
# PIVOT REPORT BUILDER
# ─────────────────────────────────────────────────────────────────────────────

def build_pivot(results: list[dict]) -> dict[str, list[dict]]:
    """Group results by IOC value for the pivot report."""
    pivot: dict[str, list[dict]] = defaultdict(list)
    for r in results:
        pivot[r["ioc"]].append(r)
    return pivot


# ─────────────────────────────────────────────────────────────────────────────
# REPORT
# ─────────────────────────────────────────────────────────────────────────────

def print_report(pivot: dict[str, list[dict]], ioc_buckets: dict[str, set[str]]) -> None:
    total_iocs    = sum(len(v) for v in ioc_buckets.values())
    matched_iocs  = len(pivot)
    total_matches = sum(len(v) for v in pivot.values())

    print("=" * 65)
    print("IOC PIVOT & CROSS-REFERENCE REPORT")
    print("=" * 65)
    print(f"\nIOC list: {total_iocs} indicators  |  "
          f"Matched: {matched_iocs}  |  Total hits: {total_matches}")

    print("\nIOC Breakdown:")
    for ioc_type, ioc_set in ioc_buckets.items():
        matched = [ioc for ioc in ioc_set if ioc in pivot]
        print(f"  {ioc_type:8s} total={len(ioc_set)}  matched={len(matched)}")

    print(f"\n{'─' * 65}")
    for ioc in sorted(pivot.keys()):
        hits = pivot[ioc]
        ioc_type = hits[0]["ioc_type"]
        print(f"\n  IOC [{ioc_type.upper()}]: {ioc}  ({len(hits)} hit(s))")
        for h in hits:
            print(f"    [{h['source']}] line {h['line_no']}: {h['line'][:90]}")
            for ctx in h["context_before"]:
                print(f"      << {ctx[:80]}")
            for ctx in h["context_after"]:
                print(f"      >> {ctx[:80]}")
    print("=" * 65)

if __name__ == "__main__":
    # Usage: q25_ioc_pivot_tool.py [ioc_file] [log_file ...]
    # Without args: uses embedded sample data.

    if len(sys.argv) > 1:
        ioc_path  = Path(sys.argv[1])
        log_paths = [Path(p) for p in sys.argv[2:]] if len(sys.argv) > 2 else []

        ioc_text = ioc_path.read_text(encoding="utf-8")
        all_results: list[dict] = []

        if log_paths:
            for lp in log_paths:
                lines = lp.read_text(encoding="utf-8", errors="replace").splitlines()
                ioc_buckets = load_iocs(ioc_text)
                all_results.extend(scan_lines(lines, ioc_buckets, source_name=lp.name))
        else:
            print("[WARN] No log files provided. Pass log file paths after the IOC file.")
            sys.exit(1)
    else:
        print("[INFO] Using embedded sample data.\n")
        ioc_buckets = load_iocs(SAMPLE_IOCS)
        log_lines   = SAMPLE_LOG.splitlines()
        all_results = scan_lines(log_lines, ioc_buckets, source_name="embedded_sample")

    pivot = build_pivot(all_results)
    print_report(pivot, ioc_buckets if len(sys.argv) == 1 else load_iocs(
        Path(sys.argv[1]).read_text(encoding="utf-8")))
