"""
Q2. Nginx Rate-Limit Evasion Detector
=======================================
Detect IPs rotating user-agents, slow-drip attacks, and distributed
path-hammering from Nginx access logs.

=============================================================================
SOLUTION PROGRESSION
=============================================================================

NAIVE: Count requests per IP; flag anything over a hard threshold.
  Problem: misses UA rotation (same IP splits across many UAs to stay under
  per-UA limits), slow-drip attacks (stay under rate-limit by spacing requests),
  and distributed attacks (no single IP triggers threshold).

BETTER: Track (IP, UA) pairs. Still misses temporal slow-drip and distributed.

GOOGLE-LEVEL (this file):
  - UA rotation: group requests by IP; if one IP has > 5 distinct UAs AND
    > 50 total requests, flag it.
  - Slow-drip: bucket requests per IP per 10s slot; if every bucket has
    exactly 1–2 requests for > 20 consecutive slots, it's pacing.
  - Distributed: bucket all requests by path per 60s; if > 20 distinct IPs
    hit the same path in a 60s window, flag the path.
  - All pattern detectors are independent passes to keep logic clean.

Complexity: O(N) time per pass, O(U) space where U = unique IPs.
=============================================================================
"""

# ─────────────────────────────────────────────────────────────────────────────
# CONSTANTS
# ─────────────────────────────────────────────────────────────────────────────

UA_ROTATION_UA_THRESHOLD  = 5    # distinct UAs per IP
UA_ROTATION_REQ_THRESHOLD = 50   # minimum total requests to flag
SLOW_DRIP_SLOTS           = 20   # minimum consecutive 5-second slots with 1–2 requests
SLOW_DRIP_SLOT_SECONDS    = 5
DISTRIBUTED_WINDOW_SECS   = 60   # window size for distributed attack
DISTRIBUTED_IP_THRESHOLD  = 20   # distinct IPs per path per window

LOG_RE = re.compile(
    r'(?P<ip>\S+)\s+\S+\s+\S+\s+'
    r'\[(?P<ts>[^\]]+)\]\s+'
    r'"(?P<method>\S+)\s+(?P<path>\S+)[^"]*"\s+'
    r'(?P<status>\d{3})\s+\S+'
    r'(?:\s+"[^"]*")?'
    r'(?:\s+"(?P<ua>[^"]*)")?'
)

TS_FORMAT = "%d/%b/%Y:%H:%M:%S %z"


def parse_ts(ts_str: str) -> float | None:
    try:
        return datetime.strptime(ts_str, TS_FORMAT).timestamp()
    except ValueError:
        return None


def parse_line(line: str) -> dict | None:
    line = line.strip()
    if not line:
        return None
    m = LOG_RE.match(line)
    if not m:
        return None
    ts = parse_ts(m.group("ts"))
    if ts is None:
        return None
    return {
        "ip":     m.group("ip"),
        "ts":     ts,
        "method": m.group("method"),
        "path":   m.group("path"),
        "status": int(m.group("status")),
        "ua":     m.group("ua") or "-",
    }


# ─────────────────────────────────────────────────────────────────────────────
# DETECTOR 1: UA Rotation
# ─────────────────────────────────────────────────────────────────────────────

def detect_ua_rotation(records: list[dict]) -> list[dict]:
    """Flag IPs using many distinct UAs to evade per-UA rate limits."""
    ip_uas:  dict[str, set]  = defaultdict(set)
    ip_reqs: dict[str, int]  = defaultdict(int)

    for r in records:
        ip_uas[r["ip"]].add(r["ua"])
        ip_reqs[r["ip"]] += 1

    alerts = []
    for ip, uas in ip_uas.items():
        if len(uas) > UA_ROTATION_UA_THRESHOLD and ip_reqs[ip] > UA_ROTATION_REQ_THRESHOLD:
            alerts.append({
                "type":    "UA_ROTATION",
                "ip":      ip,
                "ua_count": len(uas),
                "req_count": ip_reqs[ip],
                "uas":     sorted(uas),
            })
    return alerts


# ─────────────────────────────────────────────────────────────────────────────
# DETECTOR 2: Slow Drip
# ─────────────────────────────────────────────────────────────────────────────

def detect_slow_drip(records: list[dict]) -> list[dict]:
    """Detect IPs pacing requests at 1–2 per N seconds to avoid thresholds."""
    # Group timestamps per IP
    ip_ts: dict[str, list[float]] = defaultdict(list)
    for r in records:
        ip_ts[r["ip"]].append(r["ts"])

    alerts = []
    for ip, timestamps in ip_ts.items():
        if len(timestamps) < SLOW_DRIP_SLOTS:
            continue
        timestamps.sort()
        t_min = int(timestamps[0])
        t_max = int(timestamps[-1])

        # Build slot counts
        slot_counts: dict[int, int] = defaultdict(int)
        for t in timestamps:
            slot = int(t) // SLOW_DRIP_SLOT_SECONDS
            slot_counts[slot] += 1

        # Count consecutive slots with 1–2 requests
        all_slots = sorted(slot_counts.keys())
        consecutive = 0
        max_consecutive = 0
        prev = None
        for slot in all_slots:
            cnt = slot_counts[slot]
            if 1 <= cnt <= 2 and (prev is None or slot == prev + 1):
                consecutive += 1
            else:
                max_consecutive = max(max_consecutive, consecutive)
                consecutive = 1 if 1 <= cnt <= 2 else 0
            prev = slot
        max_consecutive = max(max_consecutive, consecutive)

        if max_consecutive >= SLOW_DRIP_SLOTS:
            alerts.append({
                "type":                "SLOW_DRIP",
                "ip":                  ip,
                "total_requests":      len(timestamps),
                "max_consecutive_drip_slots": max_consecutive,
                "duration_seconds":    t_max - t_min,
            })
    return alerts


# ─────────────────────────────────────────────────────────────────────────────
# DETECTOR 3: Distributed Attack on Same Path
# ─────────────────────────────────────────────────────────────────────────────

def detect_distributed_attack(records: list[dict]) -> list[dict]:
    """Detect many IPs hitting the same path within a 60-second window."""
    # path -> list of (ts, ip)
    path_events: dict[str, list[tuple[float, str]]] = defaultdict(list)
    for r in records:
        path_events[r["path"]].append((r["ts"], r["ip"]))

    alerts = []
    for path, events in path_events.items():
        events.sort(key=lambda x: x[0])
        # Sliding window
        left = 0
        for right in range(len(events)):
            while events[right][0] - events[left][0] > DISTRIBUTED_WINDOW_SECS:
                left += 1
            window_ips = {events[i][1] for i in range(left, right + 1)}
            if len(window_ips) > DISTRIBUTED_IP_THRESHOLD:
                window_start = events[left][0]
                window_end   = events[right][0]
                alerts.append({
                    "type":         "DISTRIBUTED_ATTACK",
                    "path":         path,
                    "distinct_ips": len(window_ips),
                    "window_start": window_start,
                    "window_end":   window_end,
                    "sample_ips":   sorted(window_ips)[:5],
                })
                # Advance left past this window to avoid duplicate alerts
                left = right + 1
                break
    return alerts


# ─────────────────────────────────────────────────────────────────────────────
# REPORT
# ─────────────────────────────────────────────────────────────────────────────

def print_report(ua_alerts, drip_alerts, dist_alerts) -> None:
    print("=" * 60)
    print("RATE-LIMIT EVASION DETECTION REPORT")
    print("=" * 60)

    print(f"\n[UA_ROTATION] Flagged IPs: {len(ua_alerts)}")
    for a in ua_alerts:
        print(f"  IP={a['ip']}  UAs={a['ua_count']}  requests={a['req_count']}")
        for ua in a["uas"][:5]:
            print(f"      {ua}")

    print(f"\n[SLOW_DRIP] Flagged IPs: {len(drip_alerts)}")
    for a in drip_alerts:
        print(f"  IP={a['ip']}  requests={a['total_requests']}  "
              f"drip_slots={a['max_consecutive_drip_slots']}  "
              f"duration={a['duration_seconds']}s")

    print(f"\n[DISTRIBUTED_ATTACK] Flagged paths: {len(dist_alerts)}")
    for a in dist_alerts:
        print(f"  path={a['path']}  distinct_ips={a['distinct_ips']}")
        print(f"  sample_ips={a['sample_ips']}")
    print("=" * 60)
