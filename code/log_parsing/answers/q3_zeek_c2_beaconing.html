"""
Q3. Zeek HTTP Log C2 Beaconing Detector
=========================================
Detect periodic outbound C2 communications by computing jitter coefficient
on inter-arrival times per (src_ip, dst_ip, dst_port) connection tuple.

=============================================================================
SOLUTION PROGRESSION
=============================================================================

NAIVE: Flag any IP that connects to the same external host more than N times.
  Problem: Legitimate services (telemetry, health checks) also do this.
  No temporal analysis — completely useless as a detector.

BETTER: Compute mean inter-arrival time. Flag low variance.
  Problem: Using raw variance without normalising by mean causes large-interval
  beacons (hourly) to look more "stable" than short ones. Need the coefficient
  of variation (CV = stddev / mean) to normalize.

GOOGLE-LEVEL (this file):
  - Parse Zeek TSV with proper header-driven field mapping.
  - Compute CV (jitter coefficient) per (src, dst, port) tuple.
  - Require minimum connection count before flagging (statistical significance).
  - Ignore internal-to-internal traffic (RFC 1918 → RFC 1918).
  - Sort candidates by ascending CV (most periodic = most suspicious).

Complexity: O(N) parse + O(C × K log K) sort where C = connection tuples,
            K = connections per tuple.  Space: O(N).
=============================================================================
"""

# ─────────────────────────────────────────────────────────────────────────────
# CONFIG
# ─────────────────────────────────────────────────────────────────────────────

MIN_CONNECTIONS   = 10     # need at least this many connections to compute statistics
JITTER_THRESHOLD  = 0.20   # CV below this → periodic (suspicious)

# Private address ranges to filter internal-to-internal noise
PRIVATE_RANGES = [
    ipaddress.ip_network("10.0.0.0/8"),
    ipaddress.ip_network("172.16.0.0/12"),
    ipaddress.ip_network("192.168.0.0/16"),
    ipaddress.ip_network("127.0.0.0/8"),
]


def is_private(ip_str: str) -> bool:
    try:
        addr = ipaddress.ip_address(ip_str)
        return any(addr in net for net in PRIVATE_RANGES)
    except ValueError:
        return False


# ─────────────────────────────────────────────────────────────────────────────
# PARSER
# ─────────────────────────────────────────────────────────────────────────────

def parse_zeek_log(lines: list[str]) -> list[dict]:
    """
    Parse Zeek TSV log.
    Respects #fields header for column mapping; skips all other # directives.
    """
    fields = None
    records = []

    for line in lines:
        line = line.rstrip("\n")
        if line.startswith("#fields"):
            # Tab-separated after the directive tag
            fields = line.split("\t")[1:]
            continue
        if line.startswith("#"):
            continue
        if not line.strip():
            continue
        if fields is None:
            continue  # No header seen yet — skip

        parts = line.split("\t")
        if len(parts) != len(fields):
            continue  # Mismatched column count — skip gracefully

        row = dict(zip(fields, parts))
        try:
            records.append({
                "ts":       float(row["ts"]),
                "src_ip":   row["id.orig_h"],
                "src_port": int(row["id.orig_p"]),
                "dst_ip":   row["id.resp_h"],
                "dst_port": int(row["id.resp_p"]),
                "method":   row.get("method", "-"),
                "host":     row.get("host", "-"),
                "uri":      row.get("uri", "-"),
            })
        except (KeyError, ValueError):
            continue

    return records


# ─────────────────────────────────────────────────────────────────────────────
# STATISTICS HELPERS
# ─────────────────────────────────────────────────────────────────────────────

def mean(values: list[float]) -> float:
    return sum(values) / len(values) if values else 0.0


def stddev(values: list[float]) -> float:
    if len(values) < 2:
        return 0.0
    mu = mean(values)
    variance = sum((x - mu) ** 2 for x in values) / (len(values) - 1)
    return math.sqrt(variance)


def coefficient_of_variation(values: list[float]) -> float:
    """CV = stddev / mean.  Lower = more periodic = more suspicious."""
    mu = mean(values)
    if mu == 0:
        return float("inf")
    return stddev(values) / mu


# ─────────────────────────────────────────────────────────────────────────────
# BEACON DETECTOR
# ─────────────────────────────────────────────────────────────────────────────

def detect_beaconing(records: list[dict]) -> list[dict]:
    """
    Group connections by (src_ip, dst_ip, dst_port).
    Compute inter-arrival CV; flag low-jitter, high-frequency tuples.
    Skip internal→internal traffic.
    """
    # Aggregate timestamps per connection tuple
    tuple_ts: dict[tuple, list[float]] = defaultdict(list)

    for r in records:
        if is_private(r["src_ip"]) and is_private(r["dst_ip"]):
            continue  # Ignore internal-to-internal
        key = (r["src_ip"], r["dst_ip"], r["dst_port"])
        tuple_ts[key].append(r["ts"])

    candidates = []
    for (src, dst, port), timestamps in tuple_ts.items():
        if len(timestamps) < MIN_CONNECTIONS:
            continue
        timestamps.sort()

        # Inter-arrival times
        intervals = [timestamps[i] - timestamps[i - 1] for i in range(1, len(timestamps))]
        if not intervals:
            continue

        cv      = coefficient_of_variation(intervals)
        avg_int = mean(intervals)

        if cv < JITTER_THRESHOLD:
            candidates.append({
                "src_ip":      src,
                "dst_ip":      dst,
                "dst_port":    port,
                "connections": len(timestamps),
                "mean_interval_s": round(avg_int, 2),
                "jitter_cv":   round(cv, 4),
                "first_seen":  timestamps[0],
                "last_seen":   timestamps[-1],
            })

    # Most periodic first
    candidates.sort(key=lambda x: x["jitter_cv"])
    return candidates


# ─────────────────────────────────────────────────────────────────────────────
# REPORT
# ─────────────────────────────────────────────────────────────────────────────

def print_report(candidates: list[dict]) -> None:
    print("=" * 65)
    print("ZEEK HTTP LOG — C2 BEACONING DETECTION REPORT")
    print("=" * 65)
    if not candidates:
        print("  No beaconing candidates detected.")
    for c in candidates:
        print(f"\n  [BEACON CANDIDATE]")
        print(f"    src={c['src_ip']}  dst={c['dst_ip']}:{c['dst_port']}")
        print(f"    connections={c['connections']}")
        print(f"    mean_interval={c['mean_interval_s']}s")
        print(f"    jitter_cv={c['jitter_cv']}  (threshold < {JITTER_THRESHOLD})")
        from datetime import datetime
        print(f"    first_seen={datetime.utcfromtimestamp(c['first_seen']).isoformat()}Z")
        print(f"    last_seen ={datetime.utcfromtimestamp(c['last_seen']).isoformat()}Z")
    print("=" * 65)
