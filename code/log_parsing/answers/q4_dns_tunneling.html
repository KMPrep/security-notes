"""
Q4. DNS Tunneling Detector
============================
Detect DNS-based data exfiltration and covert channels by analysing
query entropy, FQDN length, query volume, and suspicious record types.

=============================================================================
SOLUTION PROGRESSION
=============================================================================

NAIVE: Flag domains longer than some fixed length.
  Problem: CDNs and tracking pixels have long subdomains legitimately.
  No entropy analysis — misses base32/base64 encoded payloads.

BETTER: Entropy check on the subdomain label.
  Problem: Still needs volume analysis (tunneling tools hammer the same
  base domain). Missing TXT/NULL record type check. Shannon entropy
  on the entire FQDN is wrong — need entropy on the leftmost label only.

GOOGLE-LEVEL (this file):
  - Extract the leftmost subdomain label (not the TLD or SLD) for entropy.
  - Volume-per-base-domain check per client within a sliding window.
  - Suspicious record type detection (TXT, NULL, PRIVATE).
  - Composite risk score per client so analysts can triage.
  - Streaming parse — handles large log files without loading all into memory.

Complexity: O(N) time, O(C × D) space where C = clients, D = domains.
=============================================================================
"""

# ─────────────────────────────────────────────────────────────────────────────
# CONFIG
# ─────────────────────────────────────────────────────────────────────────────

ENTROPY_THRESHOLD     = 3.8   # Shannon entropy bits/char on subdomain label
LONG_FQDN_THRESHOLD   = 52    # chars before the registered domain (SLD + TLD)
VOLUME_THRESHOLD      = 100   # queries to same SLD from one client in 5 min
VOLUME_WINDOW_SECS    = 300
SUSPICIOUS_QTYPES     = {"TXT", "NULL", "PRIVATE", "ANY"}

LOG_RE = re.compile(
    r'(?P<ts>\S+)\s+client=(?P<client>\S+)\s+query=(?P<query>\S+)\s+type=(?P<qtype>\S+)'
)


# ─────────────────────────────────────────────────────────────────────────────
# HELPERS
# ─────────────────────────────────────────────────────────────────────────────

def shannon_entropy(s: str) -> float:
    """Shannon entropy in bits per character."""
    if not s:
        return 0.0
    freq: dict[str, int] = {}
    for ch in s:
        freq[ch] = freq.get(ch, 0) + 1
    n = len(s)
    return -sum((c / n) * math.log2(c / n) for c in freq.values() if c > 0)


def extract_sld_tld(fqdn: str) -> tuple[str, str]:
    """
    Very simple SLD+TLD extractor (not a full PSL — good enough here).
    Returns (subdomain_portion, sld_tld).
    E.g. 'abc.xyz.evil.com' → ('abc.xyz', 'evil.com')
    """
    parts = fqdn.rstrip(".").split(".")
    if len(parts) <= 2:
        return ("", fqdn)
    sld_tld    = ".".join(parts[-2:])
    subdomain  = ".".join(parts[:-2])
    return (subdomain, sld_tld)


def leftmost_label(fqdn: str) -> str:
    return fqdn.split(".")[0]


def parse_ts(ts_str: str) -> float | None:
    for fmt in ("%Y-%m-%dT%H:%M:%SZ", "%Y-%m-%dT%H:%M:%S"):
        try:
            return datetime.strptime(ts_str, fmt).timestamp()
        except ValueError:
            continue
    return None


def parse_line(line: str) -> dict | None:
    line = line.strip()
    if not line:
        return None
    m = LOG_RE.match(line)
    if not m:
        return None
    ts = parse_ts(m.group("ts"))
    if ts is None:
        return None
    return {
        "ts":     ts,
        "client": m.group("client"),
        "query":  m.group("query").lower(),
        "qtype":  m.group("qtype").upper(),
    }


# ─────────────────────────────────────────────────────────────────────────────
# DETECTOR
# ─────────────────────────────────────────────────────────────────────────────

def detect_dns_tunneling(records: list[dict]) -> dict:
    """
    Returns per-client findings: high-entropy labels, long FQDNs,
    suspicious record types, and high-volume base domains.
    """
    # Per-client: list of records grouped by (client, sld_tld) for volume check
    client_domain_ts: dict[tuple, list[float]] = defaultdict(list)
    client_findings:  dict[str, list[dict]]    = defaultdict(list)

    for r in records:
        query  = r["query"]
        client = r["client"]
        qtype  = r["qtype"]
        ts     = r["ts"]

        subdomain, sld_tld = extract_sld_tld(query)
        label   = leftmost_label(query)
        entropy = shannon_entropy(label)

        # Track for volume analysis
        client_domain_ts[(client, sld_tld)].append(ts)

        # High-entropy subdomain label
        if entropy > ENTROPY_THRESHOLD and len(label) >= 8:
            client_findings[client].append({
                "type":    "HIGH_ENTROPY_LABEL",
                "query":   query,
                "label":   label,
                "entropy": round(entropy, 3),
                "ts":      ts,
            })

        # Long FQDN subdomain portion
        if len(subdomain) > LONG_FQDN_THRESHOLD:
            client_findings[client].append({
                "type":   "LONG_FQDN",
                "query":  query,
                "sub_len": len(subdomain),
                "ts":     ts,
            })

        # Suspicious record type
        if qtype in SUSPICIOUS_QTYPES:
            client_findings[client].append({
                "type":  "SUSPICIOUS_QTYPE",
                "query": query,
                "qtype": qtype,
                "ts":    ts,
            })

    # Volume check: per-client sliding window over 5 minutes
    for (client, sld_tld), timestamps in client_domain_ts.items():
        timestamps.sort()
        left = 0
        for right in range(len(timestamps)):
            while timestamps[right] - timestamps[left] > VOLUME_WINDOW_SECS:
                left += 1
            if right - left + 1 > VOLUME_THRESHOLD:
                client_findings[client].append({
                    "type":   "HIGH_VOLUME",
                    "sld":    sld_tld,
                    "count":  right - left + 1,
                    "window": VOLUME_WINDOW_SECS,
                    "ts":     timestamps[left],
                })
                break  # One alert per (client, domain) is enough

    return dict(client_findings)


def risk_score(findings: list[dict]) -> int:
    """Simple additive risk score for triage sorting."""
    weights = {"HIGH_ENTROPY_LABEL": 3, "LONG_FQDN": 2, "SUSPICIOUS_QTYPE": 2, "HIGH_VOLUME": 4}
    return sum(weights.get(f["type"], 1) for f in findings)


# ─────────────────────────────────────────────────────────────────────────────
# REPORT
# ─────────────────────────────────────────────────────────────────────────────

def print_report(findings_by_client: dict) -> None:
    print("=" * 65)
    print("DNS TUNNELING DETECTION REPORT")
    print("=" * 65)
    if not findings_by_client:
        print("  No suspicious activity detected.")
        return

    sorted_clients = sorted(
        findings_by_client.items(),
        key=lambda x: risk_score(x[1]),
        reverse=True,
    )

    for client, findings in sorted_clients:
        score = risk_score(findings)
        print(f"\n  Client: {client}  risk_score={score}")
        for f in findings:
            ftype = f["type"]
            ts    = datetime.utcfromtimestamp(f["ts"]).isoformat() + "Z"
            if ftype == "HIGH_ENTROPY_LABEL":
                print(f"    [{ftype}] label='{f['label']}'  entropy={f['entropy']}  query={f['query']}")
            elif ftype == "LONG_FQDN":
                print(f"    [{ftype}] sub_len={f['sub_len']}  query={f['query']}")
            elif ftype == "SUSPICIOUS_QTYPE":
                print(f"    [{ftype}] qtype={f['qtype']}  query={f['query']}")
            elif ftype == "HIGH_VOLUME":
                print(f"    [{ftype}] {f['count']} queries to '{f['sld']}' within {f['window']}s")
    print("=" * 65)
