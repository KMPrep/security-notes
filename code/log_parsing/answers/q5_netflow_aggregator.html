"""
Q5. NetFlow/IPFIX Log Aggregator
==================================
Parse NetFlow records to detect port scans, data exfiltration,
and internal SMB/lateral movement reconnaissance.

=============================================================================
SOLUTION PROGRESSION
=============================================================================

NAIVE: Sum all bytes per src IP, flag if total > threshold.
  Problem: A single large FTP backup download also triggers this.
  No port-scan detection, no internal movement analysis.

BETTER: Group by (src, dst), check port diversity.
  Problem: Port scan detection needs a time window — otherwise a slow scan
  over many hours looks identical to a fast one. No distinction between
  internal and external destinations.

GOOGLE-LEVEL (this file):
  - Typed NetFlow record parsing with protocol mapping.
  - Port scan: sliding 60-second time window per (src, dst_ip) pair;
    count distinct dst_ports in window.
  - Exfil: single flow with bytes > threshold to external IP on non-std port.
  - Internal recon: src touching > N distinct internal dst_ips on port 445
    within a time window (SMB scanning).
  - RFC 1918 helper to distinguish internal vs external.
  - All three detectors are independent so each can be unit-tested.

Complexity: O(N log N) due to time-window sort; O(N) space.
=============================================================================
"""

# ─────────────────────────────────────────────────────────────────────────────
# CONFIG
# ─────────────────────────────────────────────────────────────────────────────

SCAN_WINDOW_SECS        = 60
SCAN_PORT_THRESHOLD     = 50
EXFIL_BYTES_THRESHOLD   = 50_000_000  # 50 MB
SMB_RECON_THRESHOLD     = 30          # distinct internal IPs on port 445
SMB_RECON_WINDOW_SECS   = 60

PRIVATE_RANGES = [
    ipaddress.ip_network("10.0.0.0/8"),
    ipaddress.ip_network("172.16.0.0/12"),
    ipaddress.ip_network("192.168.0.0/16"),
]


def is_private(ip_str: str) -> bool:
    try:
        addr = ipaddress.ip_address(ip_str)
        return any(addr in net for net in PRIVATE_RANGES)
    except ValueError:
        return False


# ─────────────────────────────────────────────────────────────────────────────
# PARSER
# ─────────────────────────────────────────────────────────────────────────────

LOG_RE = re.compile(
    r'(?P<ts>\S+)\s+'
    r'(?P<src_ip>\S+)\s+(?P<src_port>\d+)\s+'
    r'(?P<dst_ip>\S+)\s+(?P<dst_port>\d+)\s+'
    r'(?P<proto>\S+)\s+(?P<bytes>\d+)\s+(?P<packets>\d+)\s+(?P<flags>\S+)'
)


def parse_ts(ts_str: str) -> float | None:
    try:
        return datetime.strptime(ts_str, "%Y-%m-%dT%H:%M:%SZ").timestamp()
    except ValueError:
        return None


def parse_line(line: str) -> dict | None:
    line = line.strip()
    if not line:
        return None
    m = LOG_RE.match(line)
    if not m:
        return None
    ts = parse_ts(m.group("ts"))
    if ts is None:
        return None
    return {
        "ts":       ts,
        "src_ip":   m.group("src_ip"),
        "src_port": int(m.group("src_port")),
        "dst_ip":   m.group("dst_ip"),
        "dst_port": int(m.group("dst_port")),
        "proto":    m.group("proto"),
        "bytes":    int(m.group("bytes")),
        "packets":  int(m.group("packets")),
        "flags":    m.group("flags"),
    }


# ─────────────────────────────────────────────────────────────────────────────
# DETECTORS
# ─────────────────────────────────────────────────────────────────────────────

def detect_port_scan(records: list[dict]) -> list[dict]:
    """Single src contacting > SCAN_PORT_THRESHOLD distinct dst_ports on same dst_ip within window."""
    # (src_ip, dst_ip) -> sorted list of (ts, dst_port)
    pair_events: dict[tuple, list[tuple]] = defaultdict(list)
    for r in records:
        pair_events[(r["src_ip"], r["dst_ip"])].append((r["ts"], r["dst_port"]))

    alerts = []
    for (src, dst), events in pair_events.items():
        events.sort()
        left = 0
        for right in range(len(events)):
            while events[right][0] - events[left][0] > SCAN_WINDOW_SECS:
                left += 1
            ports_in_window = {events[i][1] for i in range(left, right + 1)}
            if len(ports_in_window) > SCAN_PORT_THRESHOLD:
                alerts.append({
                    "type":         "PORT_SCAN",
                    "src_ip":       src,
                    "dst_ip":       dst,
                    "distinct_ports": len(ports_in_window),
                    "window_start": events[left][0],
                    "window_end":   events[right][0],
                    "sample_ports": sorted(ports_in_window)[:10],
                })
                break
    return alerts


def detect_exfiltration(records: list[dict]) -> list[dict]:
    """Single flow exceeding EXFIL_BYTES_THRESHOLD to an external IP on a non-standard port."""
    alerts = []
    for r in records:
        if r["bytes"] > EXFIL_BYTES_THRESHOLD and not is_private(r["dst_ip"]) and r["dst_port"] > 1024:
            alerts.append({
                "type":     "EXFILTRATION",
                "src_ip":   r["src_ip"],
                "dst_ip":   r["dst_ip"],
                "dst_port": r["dst_port"],
                "bytes":    r["bytes"],
                "proto":    r["proto"],
                "ts":       r["ts"],
            })
    return alerts


def detect_smb_recon(records: list[dict]) -> list[dict]:
    """Internal src touching > threshold distinct internal IPs on port 445 in a time window."""
    smb_records = [r for r in records if r["dst_port"] == 445 and is_private(r["src_ip"]) and is_private(r["dst_ip"])]

    src_events: dict[str, list[tuple]] = defaultdict(list)
    for r in smb_records:
        src_events[r["src_ip"]].append((r["ts"], r["dst_ip"]))

    alerts = []
    for src, events in src_events.items():
        events.sort()
        left = 0
        for right in range(len(events)):
            while events[right][0] - events[left][0] > SMB_RECON_WINDOW_SECS:
                left += 1
            ips_in_window = {events[i][1] for i in range(left, right + 1)}
            if len(ips_in_window) > SMB_RECON_THRESHOLD:
                alerts.append({
                    "type":          "SMB_RECON",
                    "src_ip":        src,
                    "distinct_hosts": len(ips_in_window),
                    "window_start":  events[left][0],
                    "window_end":    events[right][0],
                    "sample_hosts":  sorted(ips_in_window)[:5],
                })
                break
    return alerts


# ─────────────────────────────────────────────────────────────────────────────
# REPORT
# ─────────────────────────────────────────────────────────────────────────────

def ts_fmt(ts: float) -> str:
    return datetime.utcfromtimestamp(ts).isoformat() + "Z"


def print_report(scan_alerts, exfil_alerts, smb_alerts) -> None:
    print("=" * 65)
    print("NETFLOW AGGREGATOR — THREAT DETECTION REPORT")
    print("=" * 65)

    print(f"\n[PORT_SCAN] {len(scan_alerts)} alert(s)")
    for a in scan_alerts:
        print(f"  src={a['src_ip']} → dst={a['dst_ip']}  ports={a['distinct_ports']}")
        print(f"  window={ts_fmt(a['window_start'])} → {ts_fmt(a['window_end'])}")
        print(f"  sample_ports={a['sample_ports']}")

    print(f"\n[EXFILTRATION] {len(exfil_alerts)} alert(s)")
    for a in exfil_alerts:
        mb = a["bytes"] / 1_000_000
        print(f"  src={a['src_ip']} → dst={a['dst_ip']}:{a['dst_port']}  "
              f"bytes={mb:.1f}MB  proto={a['proto']}  ts={ts_fmt(a['ts'])}")

    print(f"\n[SMB_RECON] {len(smb_alerts)} alert(s)")
    for a in smb_alerts:
        print(f"  src={a['src_ip']}  distinct_hosts={a['distinct_hosts']}")
        print(f"  window={ts_fmt(a['window_start'])} → {ts_fmt(a['window_end'])}")
        print(f"  sample_hosts={a['sample_hosts']}")
    print("=" * 65)

    if len(sys.argv) > 1:
        with open(sys.argv[1], encoding="utf-8", errors="replace") as fh:
            raw_lines = fh.readlines()
    else:
        print("[INFO] Using embedded sample data.\n")
        raw_lines = SAMPLE_LOG.splitlines(keepends=True)

    records       = [r for r in (parse_line(l) for l in raw_lines) if r is not None]
    scan_alerts   = detect_port_scan(records)
    exfil_alerts  = detect_exfiltration(records)
    smb_alerts    = detect_smb_recon(records)

    print(f"Parsed {len(records)} NetFlow records.\n")
    print_report(scan_alerts, exfil_alerts, smb_alerts)
