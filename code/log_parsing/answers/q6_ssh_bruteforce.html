"""
Q6. SSH/Auth Brute-Force Detector with IP Clustering
======================================================
Parse /var/log/auth.log to detect brute-force, credential stuffing,
distributed attacks, and successful logins following failure bursts.

=============================================================================
SOLUTION PROGRESSION
=============================================================================

NAIVE: Count total failed logins per IP; flag if > N.
  Problem: Misses credential stuffing (many usernames from one IP), misses
  distributed attacks (5 IPs each trying 2 passwords to avoid thresholds),
  and misses post-burst success (the actual compromise event).

BETTER: Per-IP counter with time window.
  Problem: Still missing the distributed and post-burst success patterns.
  Requires four separate independent detector passes.

GOOGLE-LEVEL (this file):
  - Four independent detectors, each with clear thresholds.
  - Sliding window for rate-based detectors (not just total counts).
  - Post-burst success: look backward from each success event to find
    any failure burst for that user within the preceding N seconds.
  - Streaming parse with regex so any auth.log format works.

Complexity: O(N log N) sort; O(U) space for user/IP state.
=============================================================================
"""

# ─────────────────────────────────────────────────────────────────────────────
# CONFIG
# ─────────────────────────────────────────────────────────────────────────────

BRUTE_FORCE_THRESHOLD    = 10   # failed logins from same IP within window
BRUTE_FORCE_WINDOW_SECS  = 60
STUFFING_USERNAME_THRESH = 20   # distinct usernames tried from one IP
DISTRIBUTED_IP_THRESHOLD = 5    # IPs all targeting same username within window
DISTRIBUTED_WINDOW_SECS  = 300  # 5 minutes
POST_BURST_WINDOW_SECS   = 300  # success within 5 min of a failure burst

# Parse auth.log RFC 3164-style
LOG_RE = re.compile(
    r'(?P<month>\w{3})\s+(?P<day>\d+)\s+(?P<time>\d{2}:\d{2}:\d{2})\s+'
    r'\S+\s+sshd\[\d+\]:\s+'
    r'(?P<result>Failed|Accepted)\s+\S+\s+for\s+(?P<user>\S+)\s+'
    r'from\s+(?P<ip>\S+)\s+port\s+(?P<port>\d+)'
)

MONTH_MAP = {"Jan":1,"Feb":2,"Mar":3,"Apr":4,"May":5,"Jun":6,
             "Jul":7,"Aug":8,"Sep":9,"Oct":10,"Nov":11,"Dec":12}

REFERENCE_YEAR = 2024  # no year in syslog RFC 3164


def parse_ts(month: str, day: str, time_str: str) -> float:
    dt = datetime(REFERENCE_YEAR, MONTH_MAP[month], int(day),
                  *map(int, time_str.split(":")))
    return dt.timestamp()


def parse_line(line: str) -> dict | None:
    line = line.strip()
    if not line:
        return None
    m = LOG_RE.search(line)
    if not m:
        return None
    return {
        "ts":     parse_ts(m.group("month"), m.group("day"), m.group("time")),
        "result": m.group("result"),
        "user":   m.group("user"),
        "ip":     m.group("ip"),
        "port":   int(m.group("port")),
    }


# ─────────────────────────────────────────────────────────────────────────────
# DETECTOR 1: Brute-Force (same IP, many failures in window)
# ─────────────────────────────────────────────────────────────────────────────

def detect_brute_force(records: list[dict]) -> list[dict]:
    ip_fail_events: dict[str, list[float]] = defaultdict(list)
    for r in records:
        if r["result"] == "Failed":
            ip_fail_events[r["ip"]].append(r["ts"])

    alerts = []
    for ip, timestamps in ip_fail_events.items():
        timestamps.sort()
        left = 0
        for right in range(len(timestamps)):
            while timestamps[right] - timestamps[left] > BRUTE_FORCE_WINDOW_SECS:
                left += 1
            count = right - left + 1
            if count >= BRUTE_FORCE_THRESHOLD:
                alerts.append({
                    "type":   "BRUTE_FORCE",
                    "ip":     ip,
                    "count":  count,
                    "window": (timestamps[left], timestamps[right]),
                })
                break
    return alerts


# ─────────────────────────────────────────────────────────────────────────────
# DETECTOR 2: Credential Stuffing (one IP, many distinct usernames)
# ─────────────────────────────────────────────────────────────────────────────

def detect_credential_stuffing(records: list[dict]) -> list[dict]:
    ip_users: dict[str, set] = defaultdict(set)
    for r in records:
        if r["result"] == "Failed":
            ip_users[r["ip"]].add(r["user"])

    return [
        {"type": "CREDENTIAL_STUFFING", "ip": ip, "distinct_users": len(users), "users": sorted(users)}
        for ip, users in ip_users.items()
        if len(users) > STUFFING_USERNAME_THRESH
    ]


# ─────────────────────────────────────────────────────────────────────────────
# DETECTOR 3: Distributed Attack (many IPs, same username, same window)
# ─────────────────────────────────────────────────────────────────────────────

def detect_distributed(records: list[dict]) -> list[dict]:
    user_events: dict[str, list[tuple]] = defaultdict(list)  # user -> [(ts, ip)]
    for r in records:
        if r["result"] == "Failed":
            user_events[r["user"]].append((r["ts"], r["ip"]))

    alerts = []
    for user, events in user_events.items():
        events.sort()
        left = 0
        for right in range(len(events)):
            while events[right][0] - events[left][0] > DISTRIBUTED_WINDOW_SECS:
                left += 1
            ips = {events[i][1] for i in range(left, right + 1)}
            if len(ips) > DISTRIBUTED_IP_THRESHOLD:
                alerts.append({
                    "type":        "DISTRIBUTED_ATTACK",
                    "target_user": user,
                    "distinct_ips": len(ips),
                    "window":      (events[left][0], events[right][0]),
                    "sample_ips":  sorted(ips)[:5],
                })
                break
    return alerts


# ─────────────────────────────────────────────────────────────────────────────
# DETECTOR 4: Post-Burst Success
# ─────────────────────────────────────────────────────────────────────────────

def detect_post_burst_success(records: list[dict]) -> list[dict]:
    """Successful login for a user preceded by a failure burst within POST_BURST_WINDOW_SECS."""
    user_failures: dict[str, list[float]] = defaultdict(list)
    successes:     list[dict]             = []

    for r in records:
        if r["result"] == "Failed":
            user_failures[r["user"]].append(r["ts"])
        elif r["result"] == "Accepted":
            successes.append(r)

    alerts = []
    for s in successes:
        user      = s["user"]
        success_ts = s["ts"]
        fails = user_failures.get(user, [])
        # Count failures within window before success
        recent_fails = [t for t in fails if success_ts - POST_BURST_WINDOW_SECS <= t < success_ts]
        if len(recent_fails) >= 2:
            alerts.append({
                "type":          "POST_BURST_SUCCESS",
                "user":          user,
                "login_ip":      s["ip"],
                "success_ts":    success_ts,
                "preceding_fails": len(recent_fails),
                "fail_window_s": round(success_ts - min(recent_fails), 1),
            })
    return alerts

# ─────────────────────────────────────────────────────────────────────────────

def ts_fmt(ts: float) -> str:
    return datetime.utcfromtimestamp(ts).strftime("%Y-%m-%dT%H:%M:%SZ")


def print_report(bf, cs, dist, pbs) -> None:
    print("=" * 65)
    print("SSH AUTH LOG — BRUTE-FORCE DETECTION REPORT")
    print("=" * 65)

    print(f"\n[BRUTE_FORCE] {len(bf)} alert(s)")
    for a in bf:
        print(f"  ip={a['ip']}  failures={a['count']}  "
              f"window={ts_fmt(a['window'][0])} → {ts_fmt(a['window'][1])}")

    print(f"\n[CREDENTIAL_STUFFING] {len(cs)} alert(s)")
    for a in cs:
        print(f"  ip={a['ip']}  distinct_users={a['distinct_users']}")
        print(f"  first 5 users: {a['users'][:5]}")

    print(f"\n[DISTRIBUTED_ATTACK] {len(dist)} alert(s)")
    for a in dist:
        print(f"  target={a['target_user']}  ips={a['distinct_ips']}  "
              f"window={ts_fmt(a['window'][0])} → {ts_fmt(a['window'][1])}")

    print(f"\n[POST_BURST_SUCCESS] {len(pbs)} alert(s)")
    for a in pbs:
        print(f"  user={a['user']}  login_ip={a['login_ip']}  "
              f"success_at={ts_fmt(a['success_ts'])}")
        print(f"  preceded by {a['preceding_fails']} failures in {a['fail_window_s']}s")
    print("=" * 65)
