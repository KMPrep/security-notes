"""
Q7. Windows Security Event Log Parser (4624/4625/4648)
=======================================================
Parse Windows Security events to detect Pass-the-Hash, lateral movement,
Over-Pass-the-Hash, and after-hours Kerberos explicit credential use.

=============================================================================
SOLUTION PROGRESSION
=============================================================================

NAIVE: Count 4625 events per IP; flag high-failure sources.
  Problem: Completely misses the sophistication of PtH — which succeeds via
  NTLM Type 3, so there are NO 4625 events. The attacker's activity is
  entirely in 4624 events.

BETTER: Parse all three event IDs, check AuthenticationPackage field.
  Problem: Still misses the correlated chain. PtH requires: Type 3 logon
  + NTLM package + workstation ≠ target server. Lateral movement requires
  counting Type 3 events from one source to distinct target hosts.

GOOGLE-LEVEL (this file):
  - Pass-the-Hash: Logon Type 3 + NTLM + WorkstationName ≠ TargetServerName.
  - Lateral movement: single SubjectIP generating Type 3 to > 5 hosts / 10 min.
  - Over-Pass-the-Hash: Type 9 from a host followed by Type 3 from same host
    within 30 seconds (NewCredentials → Kerberos → network logon).
  - After-hours: Event 4648 outside 09:00–17:00 UTC.
  - All detectors produce event chains, not just flag booleans.

Complexity: O(N) parse, O(E) space where E = distinct event tuples.
=============================================================================
"""

# ─────────────────────────────────────────────────────────────────────────────
# CONFIG
# ─────────────────────────────────────────────────────────────────────────────

LATERAL_MOVE_HOST_THRESHOLD = 5    # distinct target hosts from one source IP in window
LATERAL_MOVE_WINDOW_SECS    = 600  # 10 minutes
OPTH_WINDOW_SECS            = 30   # Type 9 → Type 3 within this window = OverPtH
BUSINESS_HOURS_START        = dtime(9, 0)
BUSINESS_HOURS_END          = dtime(17, 0)

# ─────────────────────────────────────────────────────────────────────────────
# PARSER
# ─────────────────────────────────────────────────────────────────────────────

def parse_kv(line: str) -> dict:
    """Parse key=value pairs from a log line."""
    result = {}
    for m in re.finditer(r'(\w+)=(\S+)', line):
        result[m.group(1)] = m.group(2)
    return result


def parse_line(line: str) -> dict | None:
    line = line.strip()
    if not line:
        return None
    kv = parse_kv(line)
    if "EventID" not in kv or "TimeCreated" not in kv:
        return None
    try:
        ts = datetime.strptime(kv["TimeCreated"], "%Y-%m-%dT%H:%M:%SZ").replace(tzinfo=timezone.utc)
    except ValueError:
        return None
    return {
        "ts":          ts,
        "event_id":    int(kv["EventID"]),
        "user":        kv.get("TargetUserName", kv.get("SubjectUserName", "-")),
        "subject":     kv.get("SubjectUserName", "-"),
        "logon_type":  int(kv.get("LogonType", "0")),
        "auth_pkg":    kv.get("AuthenticationPackageName", "-"),
        "workstation": kv.get("WorkstationName", "-"),
        "target_srv":  kv.get("TargetServerName", "-"),
        "src_ip":      kv.get("IpAddress", "-"),
        "domain":      kv.get("SubjectDomainName", "-"),
        "raw":         kv,
    }


# ─────────────────────────────────────────────────────────────────────────────
# DETECTOR 1: Pass-the-Hash
# ─────────────────────────────────────────────────────────────────────────────

def detect_pth(records: list[dict]) -> list[dict]:
    """
    PtH signature: Logon Type 3 + NTLM package + WorkstationName != TargetServerName.
    The workstation field reflects where the hash was injected; target is the victim server.
    """
    alerts = []
    for r in records:
        if (r["event_id"] == 4624
                and r["logon_type"] == 3
                and "NTLM" in r["auth_pkg"].upper()
                and r["workstation"] != r["target_srv"]
                and r["workstation"] not in ("-", "")
                and r["target_srv"] not in ("-", "")):
            alerts.append({
                "type":        "PASS_THE_HASH",
                "user":        r["user"],
                "src_ip":      r["src_ip"],
                "workstation": r["workstation"],
                "target_srv":  r["target_srv"],
                "ts":          r["ts"].isoformat(),
            })
    return alerts


# ─────────────────────────────────────────────────────────────────────────────
# DETECTOR 2: Lateral Movement (Type 3 to many hosts from one source)
# ─────────────────────────────────────────────────────────────────────────────

def detect_lateral_movement(records: list[dict]) -> list[dict]:
    """Single src_ip generates Logon Type 3 to > N distinct targets in 10 minutes."""
    # src_ip -> sorted list of (ts_epoch, target_srv)
    src_events: dict[str, list[tuple]] = defaultdict(list)
    for r in records:
        if r["event_id"] == 4624 and r["logon_type"] == 3 and r["src_ip"] not in ("-", ""):
            src_events[r["src_ip"]].append((r["ts"].timestamp(), r["target_srv"]))

    alerts = []
    for src, events in src_events.items():
        events.sort()
        left = 0
        for right in range(len(events)):
            while events[right][0] - events[left][0] > LATERAL_MOVE_WINDOW_SECS:
                left += 1
            hosts = {events[i][1] for i in range(left, right + 1)}
            if len(hosts) > LATERAL_MOVE_HOST_THRESHOLD:
                alerts.append({
                    "type":           "LATERAL_MOVEMENT",
                    "src_ip":         src,
                    "distinct_hosts": len(hosts),
                    "hosts":          sorted(hosts),
                    "window_start":   events[left][0],
                    "window_end":     events[right][0],
                })
                break
    return alerts


# ─────────────────────────────────────────────────────────────────────────────
# DETECTOR 3: Over-Pass-the-Hash (Type 9 → Type 3 within 30s, same workstation)
# ─────────────────────────────────────────────────────────────────────────────

def detect_opth(records: list[dict]) -> list[dict]:
    """
    OverPtH: mimikatz sekurlsa::pth injects a hash and creates a Type 9 logon session,
    then immediately uses it for a Type 3 network logon to another server.
    """
    # workstation -> list of (ts, logon_type, user)
    ws_events: dict[str, list[dict]] = defaultdict(list)
    for r in records:
        if r["event_id"] == 4624 and r["logon_type"] in (3, 9):
            ws_events[r["workstation"]].append({
                "ts":         r["ts"].timestamp(),
                "logon_type": r["logon_type"],
                "user":       r["user"],
                "target_srv": r["target_srv"],
            })

    alerts = []
    for ws, events in ws_events.items():
        type9_events = [e for e in events if e["logon_type"] == 9]
        type3_events = [e for e in events if e["logon_type"] == 3]
        for t9 in type9_events:
            for t3 in type3_events:
                if 0 <= t3["ts"] - t9["ts"] <= OPTH_WINDOW_SECS:
                    alerts.append({
                        "type":        "OVERPASS_THE_HASH",
                        "workstation": ws,
                        "user":        t9["user"],
                        "type9_ts":    t9["ts"],
                        "type3_ts":    t3["ts"],
                        "target_srv":  t3["target_srv"],
                        "gap_secs":    round(t3["ts"] - t9["ts"], 1),
                    })
    return alerts


# ─────────────────────────────────────────────────────────────────────────────
# DETECTOR 4: After-Hours 4648 (Explicit Credential Logon)
# ─────────────────────────────────────────────────────────────────────────────

def detect_after_hours_4648(records: list[dict]) -> list[dict]:
    alerts = []
    for r in records:
        if r["event_id"] == 4648:
            t = r["ts"].timetz().replace(tzinfo=None)
            if not (BUSINESS_HOURS_START <= t <= BUSINESS_HOURS_END):
                alerts.append({
                    "type":       "AFTER_HOURS_EXPLICIT_CRED",
                    "user":       r["subject"],
                    "target":     r["target_srv"],
                    "src_ip":     r["src_ip"],
                    "ts":         r["ts"].isoformat(),
                    "local_time": r["ts"].strftime("%H:%M:%S UTC"),
                })
    return alerts

def ts_fmt(epoch: float) -> str:
    return datetime.utcfromtimestamp(epoch).strftime("%Y-%m-%dT%H:%M:%SZ")


def print_report(pth, lat, opth, afh) -> None:
    print("=" * 65)
    print("WINDOWS SECURITY EVENT LOG — THREAT DETECTION REPORT")
    print("=" * 65)

    print(f"\n[PASS_THE_HASH] {len(pth)} alert(s)")
    for a in pth:
        print(f"  user={a['user']}  src={a['src_ip']}  "
              f"{a['workstation']} → {a['target_srv']}  ts={a['ts']}")

    print(f"\n[LATERAL_MOVEMENT] {len(lat)} alert(s)")
    for a in lat:
        print(f"  src={a['src_ip']}  hosts={a['distinct_hosts']}  "
              f"window={ts_fmt(a['window_start'])} → {ts_fmt(a['window_end'])}")
        print(f"  targets={a['hosts']}")

    print(f"\n[OVERPASS_THE_HASH] {len(opth)} alert(s)")
    for a in opth:
        print(f"  ws={a['workstation']}  user={a['user']}  "
              f"gap={a['gap_secs']}s → target={a['target_srv']}")

    print(f"\n[AFTER_HOURS_EXPLICIT_CRED] {len(afh)} alert(s)")
    for a in afh:
        print(f"  user={a['user']}  target={a['target']}  ts={a['ts']}")
    print("=" * 65)
