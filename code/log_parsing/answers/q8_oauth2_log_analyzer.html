"""
Q8. OAuth2 Authorization Server Log Analyzer
==============================================
Detect token theft, scope escalation, token grinding, and expired token replay
from OAuth2 authorization server logs.

=============================================================================
SOLUTION PROGRESSION
=============================================================================

NAIVE: Flag users with many refresh events.
  Problem: Service accounts legitimately refresh tokens frequently.
  No multi-IP detection (stolen token), no scope comparison, no replay detection.

BETTER: Group by token_id; detect IP changes per token.
  Problem: Need to reconstruct the original grant scope to compare with
  refresh scope (scope escalation). Replay detection requires tracking token
  state across EXPIRED → OK transitions.

GOOGLE-LEVEL (this file):
  - Token state machine: track first_ip, scope per token_id.
  - Multi-IP: token used from > 1 distinct IP in 10-minute window = STOLEN.
  - Scope escalation: refresh results in broader scope than original grant.
  - Token grinding: > 20 refreshes in 5 minutes for same token.
  - Expired replay: token first seen as EXPIRED but later succeeds (clock skew
    exploits or server-side bypass).
  - Risk level assigned: CRITICAL/HIGH/MEDIUM.

Complexity: O(N) parse, O(T) space where T = unique tokens.
=============================================================================
"""

# ─────────────────────────────────────────────────────────────────────────────
# CONFIG
# ─────────────────────────────────────────────────────────────────────────────

MULTI_IP_WINDOW_SECS   = 600   # 10 minutes
GRINDING_THRESHOLD     = 20    # refresh attempts in grinding window
GRINDING_WINDOW_SECS   = 300   # 5 minutes

LOG_RE = re.compile(
    r'(?P<ts>\S+)\s+'
    r'client_id=(?P<client_id>\S+)\s+'
    r'user=(?P<user>\S+)\s+'
    r'grant=(?P<grant>\S+)\s+'
    r'token=(?P<token>\S+)\s+'
    r'scope=(?P<scope>\S+)\s+'
    r'ip=(?P<ip>\S+)\s+'
    r'result=(?P<result>\S+)'
)


def parse_ts(ts_str: str) -> float | None:
    try:
        return datetime.strptime(ts_str, "%Y-%m-%dT%H:%M:%SZ").timestamp()
    except ValueError:
        return None


def parse_line(line: str) -> dict | None:
    line = line.strip()
    if not line:
        return None
    m = LOG_RE.match(line)
    if not m:
        return None
    ts = parse_ts(m.group("ts"))
    if ts is None:
        return None
    scope_set = frozenset(m.group("scope").split(","))
    return {
        "ts":        ts,
        "client_id": m.group("client_id"),
        "user":      m.group("user"),
        "grant":     m.group("grant"),
        "token":     m.group("token"),
        "scope":     scope_set,
        "scope_raw": m.group("scope"),
        "ip":        m.group("ip"),
        "result":    m.group("result"),
    }


def scope_is_broader(original: frozenset, refreshed: frozenset) -> bool:
    """Return True if refreshed scope contains permissions not in original grant."""
    return not refreshed.issubset(original)


# ─────────────────────────────────────────────────────────────────────────────
# ANALYZERS
# ─────────────────────────────────────────────────────────────────────────────

def analyze_tokens(records: list[dict]) -> list[dict]:
    """
    Build a state machine per token_id; emit findings across all four patterns.
    Single pass over records (sorted by ts) for efficiency.
    """
    records_sorted = sorted(records, key=lambda x: x["ts"])

    # Per-token state
    token_state: dict[str, dict] = {}

    findings = []

    for r in records_sorted:
        tok = r["token"]

        if tok not in token_state:
            token_state[tok] = {
                "original_scope": r["scope"],
                "ips":            [],           # list of (ts, ip)
                "refresh_ts":     [],           # timestamps of refresh events
                "expired_seen":   False,
                "user":           r["user"],
                "client_id":      r["client_id"],
            }

        state = token_state[tok]

        # Track IPs used with this token
        state["ips"].append((r["ts"], r["ip"]))

        # Token grinding: track refresh timestamps
        if r["grant"] == "refresh_token":
            state["refresh_ts"].append(r["ts"])

        # Expired token replay
        if r["result"] == "EXPIRED":
            state["expired_seen"] = True
        elif r["result"] == "OK" and state["expired_seen"] and r["grant"] == "refresh_token":
            findings.append({
                "type":      "EXPIRED_TOKEN_REPLAY",
                "risk":      "CRITICAL",
                "token":     tok,
                "user":      state["user"],
                "client_id": state["client_id"],
                "ip":        r["ip"],
                "ts":        r["ts"],
                "detail":    "Token previously returned EXPIRED but was accepted as OK",
            })

        # Scope escalation
        if r["grant"] == "refresh_token" and r["result"] == "OK":
            if scope_is_broader(state["original_scope"], r["scope"]):
                findings.append({
                    "type":           "SCOPE_ESCALATION",
                    "risk":           "HIGH",
                    "token":          tok,
                    "user":           state["user"],
                    "client_id":      state["client_id"],
                    "original_scope": ",".join(sorted(state["original_scope"])),
                    "escalated_scope": r["scope_raw"],
                    "ip":             r["ip"],
                    "ts":             r["ts"],
                })

    # Second pass: multi-IP and grinding (needs all events per token)
    for tok, state in token_state.items():
        user      = state["user"]
        client_id = state["client_id"]

        # Multi-IP (stolen token): same token from distinct IPs within window
        ip_events = sorted(state["ips"], key=lambda x: x[0])
        for i in range(len(ip_events)):
            window_ips = set()
            for j in range(i, len(ip_events)):
                if ip_events[j][0] - ip_events[i][0] > MULTI_IP_WINDOW_SECS:
                    break
                window_ips.add(ip_events[j][1])
            if len(window_ips) > 1:
                findings.append({
                    "type":        "MULTI_IP_TOKEN_USE",
                    "risk":        "CRITICAL",
                    "token":       tok,
                    "user":        user,
                    "client_id":   client_id,
                    "distinct_ips": len(window_ips),
                    "ips":         sorted(window_ips),
                    "window_start": ip_events[i][0],
                })
                break  # One alert per token

        # Token grinding
        refresh_ts = sorted(state["refresh_ts"])
        for i in range(len(refresh_ts)):
            count = sum(1 for t in refresh_ts if refresh_ts[i] <= t <= refresh_ts[i] + GRINDING_WINDOW_SECS)
            if count > GRINDING_THRESHOLD:
                findings.append({
                    "type":      "TOKEN_GRINDING",
                    "risk":      "HIGH",
                    "token":     tok,
                    "user":      user,
                    "client_id": client_id,
                    "refreshes": count,
                    "window_s":  GRINDING_WINDOW_SECS,
                    "ts":        refresh_ts[i],
                })
                break

    return findings

def ts_fmt(epoch: float) -> str:
    return datetime.utcfromtimestamp(epoch).strftime("%Y-%m-%dT%H:%M:%SZ")


def print_report(findings: list[dict]) -> None:
    print("=" * 65)
    print("OAUTH2 LOG ANALYSIS REPORT")
    print("=" * 65)
    if not findings:
        print("  No anomalies detected.")
    for f in sorted(findings, key=lambda x: {"CRITICAL": 0, "HIGH": 1, "MEDIUM": 2}.get(x["risk"], 3)):
        print(f"\n  [{f['risk']}] {f['type']}")
        print(f"    token={f['token']}  user={f['user']}  client={f['client_id']}")
        for k, v in f.items():
            if k not in ("type", "risk", "token", "user", "client_id"):
                if isinstance(v, float) and v > 1e9:
                    print(f"    {k}={ts_fmt(v)}")
                else:
                    print(f"    {k}={v}")
    print("=" * 65)
