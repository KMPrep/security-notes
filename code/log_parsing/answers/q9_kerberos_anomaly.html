"""
Q9. Kerberos TGT/TGS Anomaly Detector
========================================
Detect Golden Ticket, Silver Ticket, Kerberoasting, and encryption downgrade
from KDC event logs.

=============================================================================
SOLUTION PROGRESSION
=============================================================================

NAIVE: Look for too many TGS requests.
  Problem: Service accounts may legitimately request many tickets.
  No lifetime analysis (Golden Ticket), no RC4 downgrade check.

BETTER: Flag TGS requests > threshold per user.
  Problem: Kerberoasting targets specific SPN accounts (roastable = RC4 +
  no constrained delegation). Golden Ticket requires inspecting ticket
  lifetime fields, not just count. Silver Ticket has NO corresponding TGT
  in the logs — that's the key indicator.

GOOGLE-LEVEL (this file):
  - Golden Ticket: TGT lifetime > 10h or renewal window > 7 days.
  - Silver Ticket: TGS for a service with no preceding AS-REQ (TGT request)
    from that client host in the log window.
  - Kerberoasting: same account requests TGS for > 5 SPNs within 60s.
  - RC4 downgrade: etype=23 (RC4-HMAC) for an account that normally uses AES.
  - Overpass-the-Hash: AS-REQ using RC4 for an account whose prior logins used AES.

Complexity: O(N) parse, O(U) space where U = unique users/hosts.
=============================================================================
"""

import re
import sys
from collections import defaultdict
from datetime import datetime

# ─────────────────────────────────────────────────────────────────────────────
# SAMPLE DATA  (simplified KDC Security Event log, key=value format)
# Fields: ts, event_type (AS-REQ=TGT request, TGS-REQ=service ticket),
#         client, client_host, service_spn, etype, ticket_lifetime_h,
#         renew_window_h
# ─────────────────────────────────────────────────────────────────────────────

SAMPLE_LOG = """\
ts=2024-03-12T08:00:00Z event=AS-REQ client=alice@CORP client_host=WKSTN01 etype=18 ticket_lifetime_h=10 renew_window_h=168
ts=2024-03-12T08:00:05Z event=TGS-REQ client=alice@CORP client_host=WKSTN01 service_spn=cifs/FILE01 etype=18
ts=2024-03-12T08:01:00Z event=AS-REQ client=ATTACKER@CORP client_host=EVIL01 etype=23 ticket_lifetime_h=99 renew_window_h=9999
ts=2024-03-12T08:02:00Z event=TGS-REQ client=bob@CORP client_host=WKSTN02 service_spn=MSSQLSvc/SQL01 etype=23
ts=2024-03-12T08:02:05Z event=TGS-REQ client=bob@CORP client_host=WKSTN02 service_spn=MSSQLSvc/SQL02 etype=23
ts=2024-03-12T08:02:10Z event=TGS-REQ client=bob@CORP client_host=WKSTN02 service_spn=MSSQLSvc/SQL03 etype=23

# ─────────────────────────────────────────────────────────────────────────────
# CONFIG
# ─────────────────────────────────────────────────────────────────────────────

GOLDEN_TICKET_MAX_LIFETIME_H  = 10    # hours
GOLDEN_TICKET_MAX_RENEW_H     = 168   # 7 days
KERBEROASTING_SPN_THRESHOLD   = 5     # distinct SPNs within window
KERBEROASTING_WINDOW_SECS     = 60
AES_ETYPE                     = {17, 18}   # AES128, AES256
RC4_ETYPE                     = {23}       # RC4-HMAC (DES is deprecated; etype 3 = DES-CBC-MD5)

LOG_RE = re.compile(
    r'ts=(?P<ts>\S+)\s+event=(?P<event>\S+)\s+client=(?P<client>\S+)\s+'
    r'client_host=(?P<host>\S+)'
    r'(?:\s+service_spn=(?P<spn>\S+))?'
    r'(?:\s+etype=(?P<etype>\d+))?'
    r'(?:\s+ticket_lifetime_h=(?P<lifetime>\S+))?'
    r'(?:\s+renew_window_h=(?P<renew>\S+))?'
)


def parse_ts(ts_str: str) -> float | None:
    try:
        return datetime.strptime(ts_str, "%Y-%m-%dT%H:%M:%SZ").timestamp()
    except ValueError:
        return None


def parse_line(line: str) -> dict | None:
    line = line.strip()
    if not line:
        return None
    m = LOG_RE.match(line)
    if not m:
        return None
    ts = parse_ts(m.group("ts"))
    if ts is None:
        return None
    return {
        "ts":       ts,
        "event":    m.group("event"),
        "client":   m.group("client"),
        "host":     m.group("host"),
        "spn":      m.group("spn"),
        "etype":    int(m.group("etype")) if m.group("etype") else None,
        "lifetime": float(m.group("lifetime")) if m.group("lifetime") else None,
        "renew":    float(m.group("renew")) if m.group("renew") else None,
    }


# ─────────────────────────────────────────────────────────────────────────────
# DETECTOR 1: Golden Ticket
# ─────────────────────────────────────────────────────────────────────────────

def detect_golden_ticket(records: list[dict]) -> list[dict]:
    """TGT with ticket lifetime > 10h or renewal window > 7 days."""
    alerts = []
    for r in records:
        if r["event"] != "AS-REQ":
            continue
        golden = False
        reasons = []
        if r["lifetime"] is not None and r["lifetime"] > GOLDEN_TICKET_MAX_LIFETIME_H:
            golden = True
            reasons.append(f"lifetime={r['lifetime']}h (max {GOLDEN_TICKET_MAX_LIFETIME_H}h)")
        if r["renew"] is not None and r["renew"] > GOLDEN_TICKET_MAX_RENEW_H:
            golden = True
            reasons.append(f"renew_window={r['renew']}h (max {GOLDEN_TICKET_MAX_RENEW_H}h)")
        if golden:
            alerts.append({
                "type":    "GOLDEN_TICKET",
                "client":  r["client"],
                "host":    r["host"],
                "ts":      r["ts"],
                "reasons": reasons,
            })
    return alerts


# ─────────────────────────────────────────────────────────────────────────────
# DETECTOR 2: Silver Ticket
# ─────────────────────────────────────────────────────────────────────────────

def detect_silver_ticket(records: list[dict]) -> list[dict]:
    """
    TGS-REQ for a service with no preceding AS-REQ from the same host.
    Silver tickets are forged offline — no KDC contact for TGT.
    """
    # Track hosts that have made a valid AS-REQ
    hosts_with_tgt: set[str] = set()
    for r in records:
        if r["event"] == "AS-REQ":
            hosts_with_tgt.add(r["host"])

    alerts = []
    for r in records:
        if r["event"] == "TGS-REQ" and r["host"] not in hosts_with_tgt:
            alerts.append({
                "type":    "SILVER_TICKET",
                "client":  r["client"],
                "host":    r["host"],
                "spn":     r["spn"],
                "ts":      r["ts"],
                "detail":  "TGS-REQ with no prior AS-REQ from this host",
            })
    return alerts


# ─────────────────────────────────────────────────────────────────────────────
# DETECTOR 3: Kerberoasting
# ─────────────────────────────────────────────────────────────────────────────

def detect_kerberoasting(records: list[dict]) -> list[dict]:
    """Single client requesting TGS for > 5 distinct SPNs within 60 seconds."""
    client_tgs: dict[str, list[tuple]] = defaultdict(list)
    for r in records:
        if r["event"] == "TGS-REQ" and r["spn"]:
            client_tgs[r["client"]].append((r["ts"], r["spn"]))

    alerts = []
    for client, events in client_tgs.items():
        events.sort()
        left = 0
        for right in range(len(events)):
            while events[right][0] - events[left][0] > KERBEROASTING_WINDOW_SECS:
                left += 1
            spns = {events[i][1] for i in range(left, right + 1)}
            if len(spns) > KERBEROASTING_SPN_THRESHOLD:
                alerts.append({
                    "type":    "KERBEROASTING",
                    "client":  client,
                    "spns":    sorted(spns),
                    "count":   len(spns),
                    "window":  (events[left][0], events[right][0]),
                })
                break
    return alerts


# ─────────────────────────────────────────────────────────────────────────────
# DETECTOR 4: RC4 Downgrade
# ─────────────────────────────────────────────────────────────────────────────

def detect_rc4_downgrade(records: list[dict]) -> list[dict]:
    """
    Flag TGS-REQ or AS-REQ using RC4 (etype 23) for accounts that have
    previously used AES. Also flag any AS-REQ with RC4 outright (OPtH indicator).
    """
    # Track etype history per client
    client_etypes: dict[str, set] = defaultdict(set)
    for r in records:
        if r["etype"] is not None:
            client_etypes[r["client"]].add(r["etype"])

    # Clients that have used AES previously
    aes_clients = {c for c, etypes in client_etypes.items() if etypes & AES_ETYPE}

    alerts = []
    for r in records:
        if r["etype"] in RC4_ETYPE:
            is_downgrade = r["client"] in aes_clients
            alerts.append({
                "type":         "RC4_DOWNGRADE" if is_downgrade else "RC4_USAGE",
                "client":       r["client"],
                "event":        r["event"],
                "spn":          r["spn"],
                "ts":           r["ts"],
                "is_downgrade": is_downgrade,
                "detail":       "Client previously used AES; RC4 now requested" if is_downgrade else "RC4 used in environment",
            })
    return alerts


# ─────────────────────────────────────────────────────────────────────────────
# REPORT
# ─────────────────────────────────────────────────────────────────────────────

def ts_fmt(ts: float) -> str:
    return datetime.utcfromtimestamp(ts).strftime("%Y-%m-%dT%H:%M:%SZ")


def print_report(golden, silver, roasting, rc4) -> None:
    print("=" * 65)
    print("KERBEROS KDC LOG — ANOMALY DETECTION REPORT")
    print("=" * 65)

    print(f"\n[GOLDEN_TICKET] {len(golden)} alert(s)")
    for a in golden:
        print(f"  client={a['client']}  host={a['host']}  ts={ts_fmt(a['ts'])}")
        for r in a["reasons"]:
            print(f"    reason: {r}")

    print(f"\n[SILVER_TICKET] {len(silver)} alert(s)")
    for a in silver:
        print(f"  client={a['client']}  host={a['host']}  spn={a['spn']}  ts={ts_fmt(a['ts'])}")
        print(f"    {a['detail']}")

    print(f"\n[KERBEROASTING] {len(roasting)} alert(s)")
    for a in roasting:
        print(f"  client={a['client']}  spns_in_window={a['count']}")
        print(f"  window={ts_fmt(a['window'][0])} → {ts_fmt(a['window'][1])}")
        print(f"  spns={a['spns']}")

    print(f"\n[RC4_DOWNGRADE/USAGE] {len(rc4)} alert(s)")
    for a in rc4:
        severity = "DOWNGRADE" if a["is_downgrade"] else "INFO"
        print(f"  [{severity}] client={a['client']}  event={a['event']}  ts={ts_fmt(a['ts'])}")
        print(f"    {a['detail']}")
    print("=" * 65)


# ─────────────────────────────────────────────────────────────────────────────
# MAIN
# ─────────────────────────────────────────────────────────────────────────────

if __name__ == "__main__":
    if len(sys.argv) > 1:
        with open(sys.argv[1], encoding="utf-8", errors="replace") as fh:
            raw_lines = fh.readlines()
    else:
        print("[INFO] Using embedded sample data.\n")
        raw_lines = SAMPLE_LOG.splitlines(keepends=True)

    records = [r for r in (parse_line(l) for l in raw_lines) if r is not None]
    print(f"Parsed {len(records)} KDC events.\n")
    print_report(
        detect_golden_ticket(records),
        detect_silver_ticket(records),
        detect_kerberoasting(records),
        detect_rc4_downgrade(records),
    )
