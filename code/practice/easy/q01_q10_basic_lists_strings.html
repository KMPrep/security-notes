# Q1 - Two Sum
# ---------------------------------------------------------------------------
ANSWER_Q1 = """
Problem:
  Given alert severity scores and a target, return two indices whose scores
  add to the target.

Naive approach:
  Try every pair of indices. This is simple and needs O(1) extra space, but
  costs O(n^2) time.

Optimal approach:
  Use a hash map from score -> first index seen. For each score x, check
  whether target - x was already seen. This gives average O(n) time and O(n)
  space.

Alternative:
  Sort (score, index) pairs and use two pointers. That is O(n log n) and useful
  when deterministic pair ordering matters, but the hash map is faster.

Security meaning:
  Pairing alert scores is a toy version of correlating two weak signals whose
  combined score crosses a response threshold.
"""


def two_sum_naive(scores: List[int], target: int) -> Optional[Tuple[int, int]]:
    for i in range(len(scores)):
        for j in range(i + 1, len(scores)):
            if scores[i] + scores[j] == target:
                return i, j
    return None


def two_sum(scores: List[int], target: int) -> Optional[Tuple[int, int]]:
    seen: Dict[int, int] = {}
    for index, score in enumerate(scores):
        needed = target - score
        if needed in seen:
            return seen[needed], index
        if score not in seen:
            seen[score] = index
    return None


def test_q1() -> None:
    assert two_sum([4, 7, 1, 9], 10) == (2, 3)
    assert two_sum([5, 5], 10) == (0, 1)
    assert two_sum([], 10) is None
    assert two_sum([1, 2, 3], 100) is None

# Q2 - Find Duplicate IPs
# ---------------------------------------------------------------------------
ANSWER_Q2 = """
Problem:
  Given IP addresses, return IPs that appear more than once.

Naive approach:
  For each IP, scan the list to count how many times it appears. This is
  O(n^2) time.

Optimal approach:
  Use two sets: one for IPs already seen and one for duplicates. Membership is
  average O(1), so the scan is O(n). A list preserves the order in which an IP
  first becomes a duplicate.

Security meaning:
  Repeated IPs may indicate repeated access, noisy scanners, brute force
  sources, or duplicated log ingestion.
"""


def find_duplicate_ips_naive(ips: List[str]) -> List[str]:
    duplicates: List[str] = []
    for ip in ips:
        if not ip or ip in duplicates:
            continue
        if ips.count(ip) > 1:
            duplicates.append(ip)
    return duplicates


def find_duplicate_ips(ips: Iterable[str]) -> List[str]:
    seen = set()
    duplicates = set()
    ordered_duplicates: List[str] = []
    for raw_ip in ips:
        ip = raw_ip.strip() if isinstance(raw_ip, str) else raw_ip
        if not ip:
            continue
        if ip in seen and ip not in duplicates:
            duplicates.add(ip)
            ordered_duplicates.append(ip)
        seen.add(ip)
    return ordered_duplicates


def test_q2() -> None:
    assert find_duplicate_ips(["1.1.1.1", "2.2.2.2", "1.1.1.1"]) == ["1.1.1.1"]
    assert find_duplicate_ips(["", "3.3.3.3", "3.3.3.3", "3.3.3.3"]) == ["3.3.3.3"]
    assert find_duplicate_ips([]) == []

# Q3 - Count Login Attempts by User
# ---------------------------------------------------------------------------
ANSWER_Q3 = """
Problem:
  Given login records with usernames, count attempts per user.

Naive approach:
  Build a unique user list, then count every user's records with another scan.
  This is O(n * u), where u is unique users.

Optimal approach:
  Use Counter, a hash map specialized for counts. One pass over the records is
  O(n) time and O(u) space.

Security meaning:
  Per-user attempt counts are a base metric for brute force detection, account
  takeover triage, and lockout policy tuning.
"""


def count_login_attempts_by_user_naive(records: List[Dict[str, str]]) -> Dict[str, int]:
    users: List[str] = []
    for record in records:
        user = record.get("user")
        if user and user not in users:
            users.append(user)

    result: Dict[str, int] = {}
    for user in users:
        result[user] = sum(1 for record in records if record.get("user") == user)
    return result


def count_login_attempts_by_user(records: Iterable[Dict[str, str]]) -> Dict[str, int]:
    counts: Counter[str] = Counter()
    for record in records:
        user = record.get("user")
        if user:
            counts[user] += 1
    return dict(counts)


def test_q3() -> None:
    records = [{"user": "alice"}, {"user": "bob"}, {"user": "alice"}, {}]
    assert count_login_attempts_by_user(records) == {"alice": 2, "bob": 1}
    assert count_login_attempts_by_user([]) == {}

# Q4 - Most Frequent Username
# ---------------------------------------------------------------------------
ANSWER_Q4 = """
Problem:
  Return the username with the highest number of authentication events.

Naive approach:
  Count and sort all users by count. This is O(n + u log u).

Optimal approach:
  Use a Counter, then scan the counts once to find the max. This avoids sorting
  and runs in O(n + u), with O(u) space. Ties are resolved lexicographically for
  deterministic output.

Security meaning:
  The most active user may be a high-value target, service account, test user,
  or abused account.
"""


def most_frequent_username_naive(records: List[Dict[str, str]]) -> Optional[str]:
    counts = count_login_attempts_by_user(records)
    if not counts:
        return None
    return sorted(counts.items(), key=lambda item: (-item[1], item[0]))[0][0]


def most_frequent_username(records: Iterable[Dict[str, str]]) -> Optional[str]:
    counts = Counter(record.get("user") for record in records if record.get("user"))
    if not counts:
        return None

    best_user: Optional[str] = None
    best_count = -1
    for user, count in counts.items():
        if count > best_count or (count == best_count and (best_user is None or user < best_user)):
            best_user = user
            best_count = count
    return best_user


def test_q4() -> None:
    records = [{"user": "carol"}, {"user": "bob"}, {"user": "bob"}, {"user": "alice"}, {"user": "alice"}]
    assert most_frequent_username(records) == "alice"
    assert most_frequent_username([]) is None

# Q5 - First Repeated Event ID
# ---------------------------------------------------------------------------
ANSWER_Q5 = """
Problem:
  Given event IDs, return the first ID whose second occurrence appears first.

Naive approach:
  For each position, scan earlier IDs to see whether this ID has appeared.
  This is O(n^2).

Optimal approach:
  Keep a set of IDs already seen. The first ID found in the set is the answer.
  This is O(n) time and O(u) space.

Security meaning:
  Repeated event IDs can reveal duplicate ingestion, replayed events, or broken
  event generation.
"""


def first_repeated_event_id_naive(event_ids: List[str]) -> Optional[str]:
    for index, event_id in enumerate(event_ids):
        if event_id in event_ids[:index]:
            return event_id
    return None


def first_repeated_event_id(event_ids: Iterable[str]) -> Optional[str]:
    seen = set()
    for event_id in event_ids:
        if event_id in seen:
            return event_id
        seen.add(event_id)
    return None


def test_q5() -> None:
    assert first_repeated_event_id(["e1", "e2", "e1", "e2"]) == "e1"
    assert first_repeated_event_id(["e1", "e2", "e3"]) is None
    assert first_repeated_event_id([]) is None

# Q6 - Unique Failed Login Users
# ---------------------------------------------------------------------------
ANSWER_Q6 = """
Problem:
  Given failed login records, return unique users who failed authentication.

Naive approach:
  Keep a list and use linear membership checks before appending. This is
  O(n * u).

Optimal approach:
  Use a set for unique users. Insertion and membership are average O(1), so the
  scan is O(n). Return a set because the problem asks for unique users.

Security meaning:
  Unique failed users show the breadth of a password spray or credential
  stuffing attempt.
"""


def unique_failed_login_users_naive(records: List[Dict[str, str]]) -> set[str]:
    users: List[str] = []
    for record in records:
        user = record.get("user")
        event = record.get("event", "LOGIN_FAILED")
        if user and event == "LOGIN_FAILED" and user not in users:
            users.append(user)
    return set(users)


def unique_failed_login_users(records: Iterable[Dict[str, str]]) -> set[str]:
    return {
        record["user"]
        for record in records
        if record.get("user") and record.get("event", "LOGIN_FAILED") == "LOGIN_FAILED"
    }


def test_q6() -> None:
    records = [
        {"event": "LOGIN_FAILED", "user": "alice"},
        {"event": "LOGIN_SUCCESS", "user": "alice"},
        {"event": "LOGIN_FAILED", "user": "bob"},
        {"event": "LOGIN_FAILED", "user": "alice"},
    ]
    assert unique_failed_login_users(records) == {"alice", "bob"}
    assert unique_failed_login_users([]) == set()

# Q7 - Count Word Frequency in Security Notes
# ---------------------------------------------------------------------------
ANSWER_Q7 = """
Problem:
  Count the frequency of each word in a text blob.

Naive approach:
  Lowercase the text and split on whitespace. This is fast but treats
  'alert,' and 'alert' as different words.

Optimal approach:
  Use a regex tokenizer plus Counter. Regex gives a consistent definition of a
  word, and Counter gives O(n) counting over tokens.

Security meaning:
  Word frequency helps summarize incident notes, cluster common indicators,
  and spot repeated attacker tooling names.
"""


def count_word_frequency_naive(text: str) -> Dict[str, int]:
    counts: Dict[str, int] = {}
    for word in text.lower().split():
        counts[word] = counts.get(word, 0) + 1
    return counts


def count_word_frequency(text: str) -> Dict[str, int]:
    words = re.findall(r"[A-Za-z0-9_']+", text.lower())
    return dict(Counter(words))


def test_q7() -> None:
    assert count_word_frequency("Alert alert, Login!") == {"alert": 2, "login": 1}
    assert count_word_frequency("") == {}

# Q8 - Reverse a List of Alerts
# ---------------------------------------------------------------------------
ANSWER_Q8 = """
Problem:
  Reverse alert IDs without using a built-in reverse.

Naive approach:
  Insert each item at the front of a new list. It is simple but O(n^2) because
  front insertion shifts existing elements.

Optimal approach:
  Copy the list and use two pointers to swap from both ends. This is O(n) time
  and O(n) space for a non-mutating result. If mutation is allowed, the same
  two-pointer idea uses O(1) extra space.

Security meaning:
  Reversing event order is common when analysts switch between oldest-first
  and newest-first timelines.
"""


def reverse_alerts_naive(alert_ids: List[str]) -> List[str]:
    reversed_ids: List[str] = []
    for alert_id in alert_ids:
        reversed_ids.insert(0, alert_id)
    return reversed_ids


def reverse_alerts(alert_ids: List[str]) -> List[str]:
    result = list(alert_ids)
    left = 0
    right = len(result) - 1
    while left < right:
        result[left], result[right] = result[right], result[left]
        left += 1
        right -= 1
    return result


def test_q8() -> None:
    assert reverse_alerts(["a1", "a2", "a3"]) == ["a3", "a2", "a1"]
    assert reverse_alerts([]) == []
    assert reverse_alerts(["only"]) == ["only"]

# Q9 - Validate IPv4 Format - Basic
# ---------------------------------------------------------------------------
ANSWER_Q9 = """
Problem:
  Check whether a string has four dot-separated numeric parts.

Naive approach:
  Split on dots and check each piece with isdigit().

Optimal approach:
  The same split-based check is optimal for this basic requirement: it scans
  the string once, uses O(1) space for four parts, and exits early on failure.
  For strict production validation, use the standard ipaddress module to also
  enforce 0-255 ranges and canonical formatting.

Security meaning:
  Basic IP validation prevents malformed indicators from polluting allowlists,
  blocklists, and log enrichment pipelines.
"""


def validate_ipv4_basic_naive(value: str) -> bool:
    parts = value.split(".")
    return len(parts) == 4 and all(part.isdigit() for part in parts)


def validate_ipv4_basic(value: str) -> bool:
    if not isinstance(value, str):
        return False
    parts = value.strip().split(".")
    if len(parts) != 4:
        return False
    for part in parts:
        if part == "" or not part.isdigit():
            return False
    return True


def test_q9() -> None:
    assert validate_ipv4_basic("192.168.1.10") is True
    assert validate_ipv4_basic("192.168.1") is False
    assert validate_ipv4_basic("192.168.one.10") is False
    assert validate_ipv4_basic("999.999.999.999") is True

# Q10 - Normalize Email Address
# ---------------------------------------------------------------------------
ANSWER_Q10 = """
Problem:
  Lowercase and trim whitespace from an email address.

Naive approach:
  Call strip(), then lower().

Optimal approach:
  The built-in string methods are already optimal for this requirement:
  O(n) time and O(n) space for the normalized copy. If internationalized
  usernames matter, clarify whether casefold() is desired.

Security meaning:
  Normalization avoids missing the same account written with different casing
  in identity, audit, and phishing investigations.
"""


def normalize_email_naive(email: str) -> str:
    return email.strip().lower()


def normalize_email(email: str) -> str:
    return email.strip().lower()


def test_q10() -> None:
    assert normalize_email("  Alice@Example.COM ") == "alice@example.com"
    assert normalize_email("") == ""
