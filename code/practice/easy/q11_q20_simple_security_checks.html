from collections import Counter
from pathlib import PurePosixPath
from typing import Dict, Iterable, List, Optional, Sequence, Tuple

# Q11 - Extract Domain from Email
# ---------------------------------------------------------------------------
ANSWER_Q11 = """
Problem:
  Given user@example.com, return example.com.

Naive approach:
  Split on '@' and return the second piece. This is easy but does not reject
  malformed emails with zero or multiple '@' characters.

Optimal approach:
  Strip whitespace, split from the right once with rsplit('@', 1), and require
  both local and domain parts. This is O(n) time and O(n) space.

Alternative:
  Use email parsing libraries when accepting full RFC-style addresses such as
  'Alice <alice@example.com>'.

Security meaning:
  Domain extraction supports phishing triage, vendor allowlists, and grouping
  identity events by organization.
"""


def extract_domain_naive(email: str) -> Optional[str]:
    parts = email.split("@")
    if len(parts) < 2:
        return None
    return parts[1].strip().lower()


def extract_domain(email: str) -> Optional[str]:
    normalized = email.strip().lower()
    local, separator, domain = normalized.rpartition("@")
    if not separator or not local or not domain:
        return None
    return domain


def test_q11() -> None:
    assert extract_domain("User@Example.COM") == "example.com"
    assert extract_domain("missing-at") is None
    assert extract_domain("@example.com") is None

# Q12 - Count File Extensions
# ---------------------------------------------------------------------------
ANSWER_Q12 = """
Problem:
  Count how many .exe, .dll, .txt, .log, and other extensions exist.

Naive approach:
  For every file, run a chain of endswith checks for each known extension.
  This is O(n * k * m), where k is known extension count and m is filename
  length.

Optimal approach:
  Extract the suffix once, normalize it, and count with Counter. Known
  extensions stay separate; everything else is counted as 'other'.

Security meaning:
  Extension counts give quick visibility into executable-heavy downloads,
  suspicious attachments, and unexpected log/config artifacts.
"""


KNOWN_EXTENSIONS = {".exe", ".dll", ".txt", ".log"}


def count_file_extensions_naive(filenames: Iterable[str]) -> Dict[str, int]:
    counts = {".exe": 0, ".dll": 0, ".txt": 0, ".log": 0, "other": 0}
    for filename in filenames:
        lower_name = filename.lower()
        if lower_name.endswith(".exe"):
            counts[".exe"] += 1
        elif lower_name.endswith(".dll"):
            counts[".dll"] += 1
        elif lower_name.endswith(".txt"):
            counts[".txt"] += 1
        elif lower_name.endswith(".log"):
            counts[".log"] += 1
        else:
            counts["other"] += 1
    return counts


def count_file_extensions(filenames: Iterable[str]) -> Dict[str, int]:
    counts = Counter({".exe": 0, ".dll": 0, ".txt": 0, ".log": 0, "other": 0})
    for filename in filenames:
        suffix = PurePosixPath(str(filename).strip()).suffix.lower()
        key = suffix if suffix in KNOWN_EXTENSIONS else "other"
        counts[key] += 1
    return dict(counts)


def test_q12() -> None:
    result = count_file_extensions(["a.exe", "b.DLL", "readme.txt", "app.log", "archive.zip", "noext"])
    assert result == {".exe": 1, ".dll": 1, ".txt": 1, ".log": 1, "other": 2}
    assert count_file_extensions([]) == {".exe": 0, ".dll": 0, ".txt": 0, ".log": 0, "other": 0}

# Q13 - Find Longest Username
# ---------------------------------------------------------------------------
ANSWER_Q13 = """
Problem:
  Return the longest username.

Naive approach:
  Sort usernames by length descending and return the first item. Sorting costs
  O(n log n).

Optimal approach:
  Scan once while tracking the longest username. This is O(n) time and O(1)
  extra space. Keep first seen on ties for stable behavior.

Security meaning:
  Very long usernames can reveal malformed input, abuse attempts, or identity
  normalization problems.
"""


def find_longest_username_naive(usernames: List[str]) -> Optional[str]:
    valid = [user for user in usernames if user]
    if not valid:
        return None
    return sorted(valid, key=len, reverse=True)[0]


def find_longest_username(usernames: Iterable[str]) -> Optional[str]:
    longest: Optional[str] = None
    for username in usernames:
        if not username:
            continue
        if longest is None or len(username) > len(longest):
            longest = username
    return longest


def test_q13() -> None:
    assert find_longest_username(["bob", "alice", "christopher"]) == "christopher"
    assert find_longest_username(["same", "ties"]) == "same"
    assert find_longest_username([]) is None

# Q14 - Find Shortest Token
# ---------------------------------------------------------------------------
ANSWER_Q14 = """
Problem:
  Return the shortest API token.

Naive approach:
  Sort all tokens by length and return the first. This costs O(n log n).

Optimal approach:
  Scan once while tracking the shortest token. This is O(n) time and O(1)
  space. Empty tokens can be included because they are important security
  findings.

Security meaning:
  Short or empty tokens may indicate weak secrets, test credentials, or
  truncated ingestion.
"""


def find_shortest_token_naive(tokens: List[str]) -> Optional[str]:
    if not tokens:
        return None
    return sorted(tokens, key=len)[0]


def find_shortest_token(tokens: Iterable[str]) -> Optional[str]:
    shortest: Optional[str] = None
    for token in tokens:
        if shortest is None or len(token) < len(shortest):
            shortest = token
    return shortest


def test_q14() -> None:
    assert find_shortest_token(["abcdef", "xy", "12345"]) == "xy"
    assert find_shortest_token(["", "abc"]) == ""
    assert find_shortest_token([]) is None

# Q15 - Detect Empty Passwords
# ---------------------------------------------------------------------------
ANSWER_Q15 = """
Problem:
  Given username-password pairs, return users whose password is empty.

Naive approach:
  Loop through tuple pairs and append users where password == ''.

Optimal approach:
  For a list of dictionaries, scan once and filter records where password is
  exactly empty after optional whitespace stripping. This is O(n) time. If the
  input is a dictionary of user -> password, use dictionary iteration.

Security meaning:
  Empty passwords are critical credential hygiene failures and should be
  remediated immediately.
"""


def detect_empty_passwords_naive(pairs: Iterable[Tuple[str, str]]) -> List[str]:
    users: List[str] = []
    for username, password in pairs:
        if password == "":
            users.append(username)
    return users


def detect_empty_passwords(records: Iterable[Dict[str, str]], strip_whitespace: bool = False) -> List[str]:
    users: List[str] = []
    for record in records:
        user = record.get("user")
        password = record.get("password")
        if user is None or password is None:
            continue
        value = password.strip() if strip_whitespace else password
        if value == "":
            users.append(user)
    return users


def test_q15() -> None:
    records = [{"user": "alice", "password": ""}, {"user": "bob", "password": "secret"}]
    assert detect_empty_passwords(records) == ["alice"]
    assert detect_empty_passwords([{"user": "carol", "password": "   "}], strip_whitespace=True) == ["carol"]

# Q16 - Find Suspicious Ports
# ---------------------------------------------------------------------------
ANSWER_Q16 = """
Problem:
  Return destination ports that are in a suspicious list such as 22, 23, 3389,
  and 445.

Naive approach:
  For each port, scan a list of suspicious ports. This is O(n * k).

Optimal approach:
  Convert suspicious ports to a set and do O(1) average membership checks.
  Preserve original order in the output.

Security meaning:
  These ports often expose SSH, Telnet, RDP, or SMB and are high-value targets
  when open broadly.
"""


DEFAULT_SUSPICIOUS_PORTS = {22, 23, 3389, 445}


def find_suspicious_ports_naive(ports: Iterable[int], suspicious_ports: Sequence[int] = (22, 23, 3389, 445)) -> List[int]:
    result: List[int] = []
    for port in ports:
        if port in suspicious_ports:
            result.append(port)
    return result


def find_suspicious_ports(ports: Iterable[int], suspicious_ports: Iterable[int] = DEFAULT_SUSPICIOUS_PORTS) -> List[int]:
    suspicious = set(suspicious_ports)
    return [port for port in ports if port in suspicious]


def test_q16() -> None:
    assert find_suspicious_ports([80, 22, 443, 3389, 8080]) == [22, 3389]
    assert find_suspicious_ports([]) == []

# Q17 - Count HTTP Status Codes
# ---------------------------------------------------------------------------
ANSWER_Q17 = """
Problem:
  Count the occurrence of each HTTP status code.

Naive approach:
  Build a unique list of codes and call count for each one. This is O(n * u).

Optimal approach:
  Use Counter to count codes in one pass. This is O(n) time and O(u) space.

Security meaning:
  Status code distribution highlights scans, broken endpoints, auth failures,
  and server-side instability.
"""


def count_http_status_codes_naive(status_codes: List[int]) -> Dict[int, int]:
    counts: Dict[int, int] = {}
    for code in status_codes:
        if code not in counts:
            counts[code] = status_codes.count(code)
    return counts


def count_http_status_codes(status_codes: Iterable[int]) -> Dict[int, int]:
    return dict(Counter(status_codes))


def test_q17() -> None:
    assert count_http_status_codes([200, 200, 404, 500]) == {200: 2, 404: 1, 500: 1}
    assert count_http_status_codes([]) == {}

# Q18 - Find All 5xx Responses
# ---------------------------------------------------------------------------
ANSWER_Q18 = """
Problem:
  Return web access records with status code between 500 and 599.

Naive approach:
  Convert each status code to a string and check whether it starts with '5'.
  This works for normal HTTP codes but is loose for invalid values.

Optimal approach:
  Use a numeric range check, 500 <= status <= 599. This is O(n) time and avoids
  string conversion.

Security meaning:
  5xx responses can indicate successful probing, exploit attempts, overloaded
  services, or broken upstream dependencies.
"""


def find_5xx_responses_naive(records: Iterable[Dict[str, int]]) -> List[Dict[str, int]]:
    return [record for record in records if str(record.get("status", "")).startswith("5")]


def find_5xx_responses(records: Iterable[Dict[str, int]]) -> List[Dict[str, int]]:
    result: List[Dict[str, int]] = []
    for record in records:
        status = record.get("status")
        if isinstance(status, int) and 500 <= status <= 599:
            result.append(record)
    return result


def test_q18() -> None:
    records = [{"status": 200}, {"status": 500}, {"status": 503}, {"status": 404}, {"status": "500"}]
    assert find_5xx_responses(records) == [{"status": 500}, {"status": 503}]

# Q19 - Find All Admin Users
# ---------------------------------------------------------------------------
ANSWER_Q19 = """
Problem:
  Return usernames that contain 'admin'.

Naive approach:
  Check substring membership exactly as written. This misses 'Admin' and
  'ADMIN' unless case sensitivity is required.

Optimal approach:
  Normalize usernames and the keyword to lowercase, then filter in one pass.
  This is O(total characters) time.

Security meaning:
  Admin-like accounts deserve tighter monitoring, MFA checks, and privilege
  review.
"""


def find_admin_users_naive(usernames: Iterable[str]) -> List[str]:
    return [username for username in usernames if "admin" in username]


def find_admin_users(usernames: Iterable[str]) -> List[str]:
    return [username for username in usernames if "admin" in username.lower()]


def test_q19() -> None:
    assert find_admin_users(["alice", "admin", "DomainAdmin", "bob"]) == ["admin", "DomainAdmin"]
    assert find_admin_users([]) == []

# Q20 - Check Password Length
# ---------------------------------------------------------------------------
ANSWER_Q20 = """
Problem:
  Return whether each password has at least 12 characters.

Naive approach:
  Loop over passwords and append True or False from len(password) >= 12.

Optimal approach:
  The direct length check is already optimal: O(n) over password records, with
  O(n) output. If passwords are keyed by user, return a dictionary so the result
  remains attributable.

Security meaning:
  Length is not a full password policy, but it is a strong baseline against
  brute force and password guessing.
"""


def check_password_length_naive(passwords: Iterable[str]) -> List[bool]:
    result: List[bool] = []
    for password in passwords:
        result.append(len(password) >= 12)
    return result


def check_password_length(passwords: Iterable[str], min_length: int = 12) -> List[bool]:
    return [len(password) >= min_length for password in passwords]


def test_q20() -> None:
    assert check_password_length(["short", "longenough123"]) == [False, True]
    assert check_password_length([]) == []
