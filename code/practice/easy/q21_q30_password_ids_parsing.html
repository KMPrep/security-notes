# Q21 - Mask API Key
# ---------------------------------------------------------------------------
ANSWER_Q21 = """
Problem:
  Return only the first 4 and last 4 characters of an API key, masking the
  middle.

Naive approach:
  Loop through characters and replace indexes 4 through len(key)-5 with '*'.

Optimal approach:
  Use slicing: key[:4] + '*' * masked_len + key[-4:]. This is O(n) time for the
  returned string and very readable. For short keys, mask the entire key to
  avoid leaking too much.

Security meaning:
  Masking lets analysts debug which secret was used without exposing the full
  credential in logs, tickets, or screenshots.
"""


def mask_api_key_naive(api_key: str) -> str:
    if len(api_key) <= 8:
        return "*" * len(api_key)
    chars = list(api_key)
    for index in range(4, len(chars) - 4):
        chars[index] = "*"
    return "".join(chars)


def mask_api_key(api_key: str) -> str:
    if len(api_key) <= 8:
        return "*" * len(api_key)
    return f"{api_key[:4]}{'*' * (len(api_key) - 8)}{api_key[-4:]}"


def test_q21() -> None:
    assert mask_api_key("AKIAEXAMPLE123456789") == "AKIA************6789"
    assert mask_api_key("short") == "*****"
    assert mask_api_key("") == ""

# Q22 - Count Uppercase Letters in Password
# ---------------------------------------------------------------------------
ANSWER_Q22 = """
Problem:
  Count uppercase letters in a password.

Naive approach:
  Use ASCII range checks, 'A' <= char <= 'Z'.

Optimal approach:
  Use str.isupper() so Unicode uppercase letters are handled according to
  Python's character rules. This is O(n) time and O(1) space.

Security meaning:
  Uppercase count is one dimension of password complexity checks, though modern
  guidance favors length and breached-password checks.
"""


def count_uppercase_letters_naive(password: str) -> int:
    count = 0
    for char in password:
        if "A" <= char <= "Z":
            count += 1
    return count


def count_uppercase_letters(password: str) -> int:
    return sum(1 for char in password if char.isupper())


def test_q22() -> None:
    assert count_uppercase_letters("PassWORD123") == 5
    assert count_uppercase_letters("lower") == 0

# Q23 - Count Digits in Password
# ---------------------------------------------------------------------------
ANSWER_Q23 = """
Problem:
  Count digits in a password.

Naive approach:
  Compare each character against '0' through '9'.

Optimal approach:
  Use str.isdigit() in a generator expression. This remains O(n) time and O(1)
  space. If policy requires ASCII-only digits, use char in string.digits.

Security meaning:
  Digit count is a simple password policy signal and can help explain why a
  password fails validation.
"""


def count_digits_in_password_naive(password: str) -> int:
    count = 0
    for char in password:
        if "0" <= char <= "9":
            count += 1
    return count


def count_digits_in_password(password: str) -> int:
    return sum(1 for char in password if char.isdigit())


def test_q23() -> None:
    assert count_digits_in_password("a1b2c3") == 3
    assert count_digits_in_password("abcdef") == 0

# Q24 - Simple Password Policy Check
# ---------------------------------------------------------------------------
ANSWER_Q24 = """
Problem:
  Check whether a password has uppercase, lowercase, digit, special character,
  and minimum length.

Naive approach:
  Run separate passes for each requirement. It is clear but scans the password
  multiple times.

Optimal approach:
  Scan once and update boolean flags. This is O(n) time and O(1) space.

Alternative:
  Regex lookaheads can be compact, but boolean flags are easier to explain in
  clarity and to customize in production.

Security meaning:
  This enforces baseline password complexity. It should be combined with MFA,
  rate limiting, and breached-password screening.
"""


def simple_password_policy_check_naive(password: str, min_length: int = 12) -> bool:
    has_upper = any(char.isupper() for char in password)
    has_lower = any(char.islower() for char in password)
    has_digit = any(char.isdigit() for char in password)
    has_special = any(not char.isalnum() for char in password)
    return len(password) >= min_length and has_upper and has_lower and has_digit and has_special


def simple_password_policy_check(password: str, min_length: int = 12) -> bool:
    has_upper = False
    has_lower = False
    has_digit = False
    has_special = False

    for char in password:
        if char.isupper():
            has_upper = True
        elif char.islower():
            has_lower = True
        elif char.isdigit():
            has_digit = True
        else:
            has_special = True

    return len(password) >= min_length and has_upper and has_lower and has_digit and has_special


def test_q24() -> None:
    assert simple_password_policy_check("StrongerPass1!") is True
    assert simple_password_policy_check("short1!") is False
    assert simple_password_policy_check("NoSpecial1234") is False

# Q25 - Remove Duplicate Alert IDs
# ---------------------------------------------------------------------------
ANSWER_Q25 = """
Problem:
  Return alert IDs in original order without duplicates.

Naive approach:
  Build a result list and check list membership before appending. Membership is
  O(n), so this can become O(n^2).

Optimal approach:
  Use a set for membership and a list for output order. This is O(n) time and
  O(u) space.

Security meaning:
  De-duplicating alert IDs prevents repeated triage, repeated notifications,
  and inflated metrics.
"""


def remove_duplicate_alert_ids_naive(alert_ids: Iterable[str]) -> List[str]:
    result: List[str] = []
    for alert_id in alert_ids:
        if alert_id not in result:
            result.append(alert_id)
    return result


def remove_duplicate_alert_ids(alert_ids: Iterable[str]) -> List[str]:
    seen = set()
    result: List[str] = []
    for alert_id in alert_ids:
        if alert_id not in seen:
            seen.add(alert_id)
            result.append(alert_id)
    return result


def test_q25() -> None:
    assert remove_duplicate_alert_ids(["a1", "a2", "a1", "a3"]) == ["a1", "a2", "a3"]
    assert remove_duplicate_alert_ids([]) == []

# Q26 - Find Missing Event ID
# ---------------------------------------------------------------------------
ANSWER_Q26 = """
Problem:
  Given event IDs from 1..n with one missing, find the missing ID.

Naive approach:
  For each number from 1..n, check whether it appears in the list. This is
  O(n^2) with list membership.

Optimal approach:
  Use the arithmetic series sum n*(n+1)//2 minus the observed sum. This is O(n)
  time and O(1) space.

Alternative:
  Use XOR to avoid large integer sums in fixed-width languages.

Security meaning:
  Missing event IDs may indicate log loss, ingestion gaps, tampering, or dropped
  events in a queue.
"""


def find_missing_event_id_naive(event_ids: Sequence[int], n: int) -> Optional[int]:
    for expected in range(1, n + 1):
        if expected not in event_ids:
            return expected
    return None


def find_missing_event_id(event_ids: Iterable[int], n: int) -> int:
    expected_sum = n * (n + 1) // 2
    return expected_sum - sum(event_ids)


def test_q26() -> None:
    assert find_missing_event_id([1, 2, 4, 5], 5) == 3
    assert find_missing_event_id([2, 3, 4], 4) == 1

# Q27 - Merge Two Sorted Lists of Timestamps
# ---------------------------------------------------------------------------
ANSWER_Q27 = """
Problem:
  Merge two sorted timestamp lists into one sorted list.

Naive approach:
  Concatenate both lists and sort the result. This is O((a+b) log(a+b)).

Optimal approach:
  Use two pointers, one per sorted list, and append the smaller timestamp each
  step. This is O(a+b) time and O(a+b) output space.

Security meaning:
  Timeline merging is common when correlating identity, endpoint, network, and
  cloud logs from different sources.
"""


def merge_sorted_timestamps_naive(left: List[str], right: List[str]) -> List[str]:
    return sorted(left + right)


def merge_sorted_timestamps(left: Sequence[str], right: Sequence[str]) -> List[str]:
    merged: List[str] = []
    i = 0
    j = 0
    while i < len(left) and j < len(right):
        if left[i] <= right[j]:
            merged.append(left[i])
            i += 1
        else:
            merged.append(right[j])
            j += 1
    merged.extend(left[i:])
    merged.extend(right[j:])
    return merged


def test_q27() -> None:
    assert merge_sorted_timestamps(["2026-01-01T00:00:01Z"], ["2026-01-01T00:00:00Z"]) == [
        "2026-01-01T00:00:00Z",
        "2026-01-01T00:00:01Z",
    ]
    assert merge_sorted_timestamps([], ["t1"]) == ["t1"]

# Q28 - Find Maximum Failed Attempts
# ---------------------------------------------------------------------------
ANSWER_Q28 = """
Problem:
  Given a dictionary of users to failed attempts, return the user with the
  highest count.

Naive approach:
  Sort all dictionary items by count descending. This costs O(u log u).

Optimal approach:
  Scan dictionary items once and track the best user/count. This is O(u) time
  and O(1) extra space. Ties are resolved lexicographically for deterministic
  output.

Security meaning:
  The max failed-attempt account is a strong candidate for lockout review,
  targeted attack investigation, or password reset.
"""


def find_maximum_failed_attempts_naive(failed_attempts: Dict[str, int]) -> Optional[str]:
    if not failed_attempts:
        return None
    return sorted(failed_attempts.items(), key=lambda item: (-item[1], item[0]))[0][0]


def find_maximum_failed_attempts(failed_attempts: Dict[str, int]) -> Optional[str]:
    best_user: Optional[str] = None
    best_count = -1
    for user, count in failed_attempts.items():
        if count > best_count or (count == best_count and (best_user is None or user < best_user)):
            best_user = user
            best_count = count
    return best_user


def test_q28() -> None:
    assert find_maximum_failed_attempts({"bob": 5, "alice": 5, "carol": 2}) == "alice"
    assert find_maximum_failed_attempts({}) is None

# Q29 - Convert CSV Line to Fields
# ---------------------------------------------------------------------------
ANSWER_Q29 = """
Problem:
  Given a simple comma-separated log line, split it into fields.

Naive approach:
  Use line.split(','). This is fine for simple CSV but fails for quoted commas.

Optimal approach:
  Use Python's csv module, which correctly handles quoted values, escaped
  quotes, and empty fields. This is O(n) time for the line length.

Security meaning:
  Correct CSV parsing prevents malformed fields from shifting columns and
  corrupting alert or asset data.
"""


def convert_csv_line_to_fields_naive(line: str) -> List[str]:
    return [field.strip() for field in line.split(",")]


def convert_csv_line_to_fields(line: str) -> List[str]:
    reader = csv.reader(StringIO(line))
    return next(reader, [])


def test_q29() -> None:
    assert convert_csv_line_to_fields("alice,10.0.0.1,login") == ["alice", "10.0.0.1", "login"]
    assert convert_csv_line_to_fields('"alice, admin",10.0.0.1') == ["alice, admin", "10.0.0.1"]
    assert convert_csv_line_to_fields("") == []

# Q30 - Detect Localhost IP
# ---------------------------------------------------------------------------
ANSWER_Q30 = """
Problem:
  Check whether an IP is 127.0.0.1 or ::1.

Naive approach:
  Compare the string directly against both localhost values.

Optimal approach:
  Trim whitespace and check membership in a set of loopback literals. Set
  membership is O(1). For broader loopback detection, use the ipaddress module
  and check is_loopback.

Security meaning:
  Localhost traffic can indicate local testing, SSRF behavior, or services that
  should not be exposed remotely.
"""


LOCALHOST_LITERALS = {"127.0.0.1", "::1"}


def detect_localhost_ip_naive(ip_address: str) -> bool:
    return ip_address == "127.0.0.1" or ip_address == "::1"


def detect_localhost_ip(ip_address: str) -> bool:
    return ip_address.strip().lower() in LOCALHOST_LITERALS


def test_q30() -> None:
    assert detect_localhost_ip("127.0.0.1") is True
    assert detect_localhost_ip(" ::1 ") is True
    assert detect_localhost_ip("127.0.0.2") is False
