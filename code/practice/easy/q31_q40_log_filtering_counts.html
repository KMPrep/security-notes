"""
Google Security Engineer - Easy Questions: Q31-Q40
Topic: Log filtering, extraction, and counting.
"""

from collections import Counter
import re
from typing import Dict, Iterable, List, Optional


IPV4_LOOKING_RE = re.compile(r"\b(?:\d{1,3}\.){3}\d{1,3}\b")
ACCESS_REQUEST_RE = re.compile(r'"[A-Z]+ (?P<endpoint>\S+) HTTP/[0-9.]+"')


# ---------------------------------------------------------------------------
# Q31 - Count Log Lines by Severity
# ---------------------------------------------------------------------------
ANSWER_Q31 = """
Problem:
  Given log lines containing INFO, WARN, or ERROR, count each severity.

Naive approach:
  For each severity, scan every line and count substring matches. This is
  O(n * s), where s is severity count.

Optimal approach:
  Scan once and use Counter. Check severity as a token to avoid matching words
  that merely contain the letters. This is O(n) time and O(s) space.

Security meaning:
  Severity distribution helps spot outages, noisy controls, and sudden spikes
  in errors or warnings during attacks.
"""


SEVERITIES = {"INFO", "WARN", "ERROR"}


def count_log_lines_by_severity_naive(lines: Iterable[str]) -> Dict[str, int]:
    materialized = list(lines)
    return {
        "INFO": sum(1 for line in materialized if "INFO" in line),
        "WARN": sum(1 for line in materialized if "WARN" in line),
        "ERROR": sum(1 for line in materialized if "ERROR" in line),
    }


def count_log_lines_by_severity(lines: Iterable[str]) -> Dict[str, int]:
    counts = Counter({"INFO": 0, "WARN": 0, "ERROR": 0})
    for line in lines:
        for token in re.findall(r"\b[A-Z]+\b", line):
            if token in SEVERITIES:
                counts[token] += 1
                break
    return dict(counts)


def test_q31() -> None:
    lines = ["INFO LOGIN_SUCCESS", "WARN LOGIN_FAILED", "ERROR LOGIN_FAILED", "no severity"]
    assert count_log_lines_by_severity(lines) == {"INFO": 1, "WARN": 1, "ERROR": 1}
    assert count_log_lines_by_severity([]) == {"INFO": 0, "WARN": 0, "ERROR": 0}

# Q32 - Extract IP from Log Line
# ---------------------------------------------------------------------------
ANSWER_Q32 = """
Problem:
  Extract the first IPv4-looking value from a log line.

Naive approach:
  Split the line by spaces and dots, then manually test tokens. This is brittle
  around punctuation.

Optimal approach:
  Use a compiled regex for IPv4-looking values. It is O(n) over the line and
  handles punctuation around the IP. For strict validation, use ipaddress after
  extraction.

Security meaning:
  IP extraction is the first step for enrichment, geo lookup, blocklisting, and
  request attribution.
"""


def extract_ip_from_log_line_naive(line: str) -> Optional[str]:
    for token in line.split():
        cleaned = token.strip("[](),")
        parts = cleaned.split(".")
        if len(parts) == 4 and all(part.isdigit() for part in parts):
            return cleaned
    return None


def extract_ip_from_log_line(line: str) -> Optional[str]:
    match = IPV4_LOOKING_RE.search(line)
    return match.group(0) if match else None


def test_q32() -> None:
    assert extract_ip_from_log_line("src=10.0.1.10 action=login") == "10.0.1.10"
    assert extract_ip_from_log_line("no ip here") is None

# Q33 - Extract Timestamp from Log Line
# ---------------------------------------------------------------------------
ANSWER_Q33 = """
Problem:
  Given log lines beginning with timestamps, extract the timestamp.

Naive approach:
  Split on whitespace and return the first token.

Optimal approach:
  For ISO-style logs, the first token extraction is optimal: O(k), where k is
  the timestamp token length. Add validation if timestamp format matters.

Security meaning:
  Timestamps anchor incident timelines and are required for correlation across
  data sources.
"""


def extract_timestamp_from_log_line_naive(line: str) -> Optional[str]:
    parts = line.split()
    return parts[0] if parts else None


def extract_timestamp_from_log_line(line: str) -> Optional[str]:
    stripped = line.strip()
    if not stripped:
        return None
    return stripped.split(maxsplit=1)[0]


def test_q33() -> None:
    assert extract_timestamp_from_log_line("2026-05-29T08:01:11Z INFO login") == "2026-05-29T08:01:11Z"
    assert extract_timestamp_from_log_line("   ") is None

# Q34 - Count Failed Logins
# ---------------------------------------------------------------------------
ANSWER_Q34 = """
Problem:
  Count logs containing LOGIN_FAILED.

Naive approach:
  Use a loop and increment a counter when the substring appears.

Optimal approach:
  The single scan is already optimal. Use token/word matching if false matches
  such as LOGIN_FAILED_EXTRA should not count.

Security meaning:
  Failed login volume is a core signal for brute force, password spray, and
  credential stuffing.
"""


def count_failed_logins_naive(lines: Iterable[str]) -> int:
    count = 0
    for line in lines:
        if "LOGIN_FAILED" in line:
            count += 1
    return count


def count_failed_logins(lines: Iterable[str]) -> int:
    return sum(1 for line in lines if re.search(r"\bLOGIN_FAILED\b", line))


def test_q34() -> None:
    assert count_failed_logins(["LOGIN_FAILED user=a", "LOGIN_SUCCESS user=a"]) == 1
    assert count_failed_logins([]) == 0

# Q36 - Find Logs for a Specific User
# ---------------------------------------------------------------------------
ANSWER_Q36 = """
Problem:
  Return all log lines containing a specific username.

Naive approach:
  Use simple substring checks. This may match 'ann' inside 'joann'.

Optimal approach:
  Compile a regex for user=<username> or a word-boundary match and scan once.
  For repeated lookups, build an inverted index user -> lines.

Security meaning:
  User-centric filtering is essential for account compromise investigation and
  audit review.
"""


def find_logs_for_specific_user_naive(lines: Iterable[str], username: str) -> List[str]:
    return [line for line in lines if username in line]


def find_logs_for_specific_user(lines: Iterable[str], username: str) -> List[str]:
    pattern = re.compile(rf"\buser={re.escape(username)}\b|\b{re.escape(username)}\b")
    return [line for line in lines if pattern.search(line)]


def test_q36() -> None:
    lines = ["INFO user=alice LOGIN_SUCCESS", "INFO user=bob LOGIN_FAILED", "INFO user=malice note"]
    assert find_logs_for_specific_user(lines, "alice") == ["INFO user=alice LOGIN_SUCCESS"]
    assert find_logs_for_specific_user(lines, "missing") == []

# Q37 - Find Logs for a Specific IP
# ---------------------------------------------------------------------------
ANSWER_Q37 = """
Problem:
  Return all log lines containing a specific IP address.

Naive approach:
  Use substring matching. It is simple but can match 10.0.0.1 inside
  10.0.0.10.

Optimal approach:
  Use a regex with boundaries around the escaped IP. For repeated searches,
  build an IP -> lines index.

Security meaning:
  IP-centric filtering supports source investigation, firewall decisions, and
  enrichment workflows.
"""


def find_logs_for_specific_ip_naive(lines: Iterable[str], ip_address: str) -> List[str]:
    return [line for line in lines if ip_address in line]


def find_logs_for_specific_ip(lines: Iterable[str], ip_address: str) -> List[str]:
    pattern = re.compile(rf"(?<![\d.]){re.escape(ip_address)}(?![\d.])")
    return [line for line in lines if pattern.search(line)]


def test_q37() -> None:
    lines = ["src=10.0.0.1 ok", "src=10.0.0.10 ok", "src=203.0.113.5 denied"]
    assert find_logs_for_specific_ip(lines, "10.0.0.1") == ["src=10.0.0.1 ok"]
    assert find_logs_for_specific_ip(lines, "8.8.8.8") == []

# Q38 - Count Requests Per Endpoint
# ---------------------------------------------------------------------------
ANSWER_Q38 = """
Problem:
  Given HTTP logs, count requests for each endpoint path.

Naive approach:
  Split each line and assume the endpoint is always at a fixed index. This is
  fragile if the format changes.

Optimal approach:
  Use a compiled regex for the quoted HTTP request, then Counter for endpoint
  counts. This is O(n) over log lines.

Security meaning:
  Endpoint counts reveal abused routes, scan targets, and heavily used APIs.
"""


def count_requests_per_endpoint_naive(lines: Iterable[str]) -> Dict[str, int]:
    counts: Dict[str, int] = {}
    for line in lines:
        parts = line.split()
        if len(parts) > 6:
            endpoint = parts[6]
            counts[endpoint] = counts.get(endpoint, 0) + 1
    return counts


def count_requests_per_endpoint(lines: Iterable[str]) -> Dict[str, int]:
    counts: Counter[str] = Counter()
    for line in lines:
        match = ACCESS_REQUEST_RE.search(line)
        if match:
            counts[match.group("endpoint")] += 1
    return dict(counts)


def test_q38() -> None:
    lines = [
        '10.0.0.1 - - [date] "GET /home HTTP/1.1" 200 1 "-" "ua"',
        '10.0.0.1 - - [date] "POST /login HTTP/1.1" 200 1 "-" "ua"',
        '10.0.0.2 - - [date] "GET /home HTTP/1.1" 200 1 "-" "ua"',
    ]
    assert count_requests_per_endpoint(lines) == {"/home": 2, "/login": 1}

# Q39 - Count Requests Per Source IP
# ---------------------------------------------------------------------------
ANSWER_Q39 = """
Problem:
  Given access logs, count requests per source IP.

Naive approach:
  Extract unique source IPs, then count each one with repeated scans.

Optimal approach:
  Source IP is the first token in common access logs. Scan once and count with
  Counter. This is O(n) time and O(u) space.

Security meaning:
  Per-source volume identifies scanners, brute force sources, noisy NATs, and
  candidates for rate limiting.
"""


def count_requests_per_source_ip_naive(lines: List[str]) -> Dict[str, int]:
    ips = []
    for line in lines:
        ip = line.split(maxsplit=1)[0] if line.strip() else None
        if ip and ip not in ips:
            ips.append(ip)
    return {ip: sum(1 for line in lines if line.startswith(ip + " ")) for ip in ips}


def count_requests_per_source_ip(lines: Iterable[str]) -> Dict[str, int]:
    counts: Counter[str] = Counter()
    for line in lines:
        stripped = line.strip()
        if not stripped:
            continue
        ip = stripped.split(maxsplit=1)[0]
        if IPV4_LOOKING_RE.fullmatch(ip):
            counts[ip] += 1
    return dict(counts)


def test_q39() -> None:
    lines = ["10.0.0.1 - - request", "10.0.0.2 - - request", "10.0.0.1 - - request", ""]
    assert count_requests_per_source_ip(lines) == {"10.0.0.1": 2, "10.0.0.2": 1}

# Q40 - Find Most Talkative IP - Basic
# ---------------------------------------------------------------------------
ANSWER_Q40 = """
Problem:
  Given source IPs in logs, return the IP with the highest number of requests.

Naive approach:
  Count per source IP, sort all counts, and return the first. Sorting costs
  O(u log u).

Optimal approach:
  Use Counter and a linear scan for the maximum. This is O(n + u) time and
  O(u) space. Ties are resolved lexicographically for deterministic output.

Security meaning:
  The most talkative IP is a first triage target for blocking, rate limiting,
  bot detection, or proxy/NAT investigation.
"""


def find_most_talkative_ip_basic_naive(lines: List[str]) -> Optional[str]:
    counts = count_requests_per_source_ip(lines)
    if not counts:
        return None
    return sorted(counts.items(), key=lambda item: (-item[1], item[0]))[0][0]


def find_most_talkative_ip_basic(lines: Iterable[str]) -> Optional[str]:
    counts = count_requests_per_source_ip(lines)
    if not counts:
        return None
    best_ip: Optional[str] = None
    best_count = -1
    for ip, count in counts.items():
        if count > best_count or (count == best_count and (best_ip is None or ip < best_ip)):
            best_ip = ip
            best_count = count
    return best_ip


def test_q40() -> None:
    lines = ["10.0.0.2 - - request", "10.0.0.1 - - request", "10.0.0.1 - - request"]
    assert find_most_talkative_ip_basic(lines) == "10.0.0.1"
    assert find_most_talkative_ip_basic([]) is None
