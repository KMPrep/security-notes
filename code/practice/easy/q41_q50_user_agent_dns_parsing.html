from collections import Counter

# Q41 - Detect User-Agent Missing
# ---------------------------------------------------------------------------
ANSWER_Q41 = """
Problem:
  Return HTTP request records where user-agent is empty.

Naive approach:
  Loop through records and check if record['user_agent'] == ''.

Optimal approach:
  Treat missing, None, empty, and whitespace-only user-agent values as missing.
  A single pass is O(n) time. The output list stores the matching records.

Security meaning:
  Missing user-agents are common in scripts, bots, broken clients, and some
  scanners.
"""


def detect_user_agent_missing_naive(records: Iterable[Dict[str, str]]) -> List[Dict[str, str]]:
    result: List[Dict[str, str]] = []
    for record in records:
        if record.get("user_agent") == "":
            result.append(record)
    return result


def detect_user_agent_missing(records: Iterable[Dict[str, Optional[str]]]) -> List[Dict[str, Optional[str]]]:
    result: List[Dict[str, Optional[str]]] = []
    for record in records:
        user_agent = record.get("user_agent")
        if user_agent is None or user_agent.strip() == "":
            result.append(record)
    return result


def test_q41() -> None:
    records = [{"user_agent": "Mozilla"}, {"user_agent": ""}, {}, {"user_agent": "   "}]
    assert detect_user_agent_missing(records) == [{"user_agent": ""}, {}, {"user_agent": "   "}]

# Q42 - Detect Suspicious User-Agent Keyword
# ---------------------------------------------------------------------------
ANSWER_Q42 = """
Problem:
  Return logs where user-agent contains sqlmap, nikto, curl, or
  python-requests.

Naive approach:
  Lowercase each user-agent and check every keyword with a loop.

Optimal approach:
  Compile one case-insensitive regex from the keyword list and scan each
  user-agent once. This keeps matching logic centralized and easy to extend.

Security meaning:
  These user-agents often identify scanners, automation, exploit tools, and
  scripted probing.
"""


SUSPICIOUS_UA_KEYWORDS = ("sqlmap", "nikto", "curl", "python-requests")
SUSPICIOUS_UA_RE = re.compile("|".join(re.escape(word) for word in SUSPICIOUS_UA_KEYWORDS), re.IGNORECASE)


def detect_suspicious_user_agent_keyword_naive(records: Iterable[Dict[str, str]]) -> List[Dict[str, str]]:
    result: List[Dict[str, str]] = []
    for record in records:
        user_agent = record.get("user_agent", "").lower()
        if any(keyword in user_agent for keyword in SUSPICIOUS_UA_KEYWORDS):
            result.append(record)
    return result


def detect_suspicious_user_agent_keyword(records: Iterable[Dict[str, str]]) -> List[Dict[str, str]]:
    return [record for record in records if SUSPICIOUS_UA_RE.search(record.get("user_agent", ""))]


def test_q42() -> None:
    records = [{"user_agent": "Mozilla"}, {"user_agent": "sqlmap"}, {"user_agent": "Python-Requests/2.0"}]
    assert detect_suspicious_user_agent_keyword(records) == [{"user_agent": "sqlmap"}, {"user_agent": "Python-Requests/2.0"}]

# Q43 - Count Authentication Events by Day
# ---------------------------------------------------------------------------
ANSWER_Q43 = """
Problem:
  Given timestamps, count authentication events per date.

Naive approach:
  Build a unique date list, then count each date with repeated scans.

Optimal approach:
  Extract the date component once per timestamp and count with Counter. This is
  O(n) time and O(d) space, where d is unique days.

Security meaning:
  Daily authentication counts make unusual login spikes visible and support
  investigation timelines.
"""


def count_authentication_events_by_day_naive(timestamps: List[str]) -> Dict[str, int]:
    days: List[str] = []
    for timestamp in timestamps:
        day = timestamp[:10]
        if day and day not in days:
            days.append(day)
    return {day: sum(1 for timestamp in timestamps if timestamp.startswith(day)) for day in days}


def count_authentication_events_by_day(timestamps: Iterable[str]) -> Dict[str, int]:
    counts: Counter[str] = Counter()
    for timestamp in timestamps:
        timestamp = timestamp.strip()
        if len(timestamp) >= 10:
            counts[timestamp[:10]] += 1
    return dict(counts)


def test_q43() -> None:
    timestamps = ["2026-05-29T08:00:00Z", "2026-05-29T09:00:00Z", "2026-05-30T08:00:00Z"]
    assert count_authentication_events_by_day(timestamps) == {"2026-05-29": 2, "2026-05-30": 1}

# Q44 - Count Events Per Host
# ---------------------------------------------------------------------------
ANSWER_Q44 = """
Problem:
  Given hostnames in logs, count events per hostname.

Naive approach:
  Build a unique hostname list and count each host with another scan.

Optimal approach:
  Use Counter keyed by hostname. This is O(n) time and O(h) space, where h is
  unique hosts.

Security meaning:
  Host event counts identify noisy endpoints, impacted assets, and collection
  gaps.
"""


def count_events_per_host_naive(records: List[Dict[str, str]]) -> Dict[str, int]:
    hosts: List[str] = []
    for record in records:
        host = record.get("host")
        if host and host not in hosts:
            hosts.append(host)
    return {host: sum(1 for record in records if record.get("host") == host) for host in hosts}


def count_events_per_host(records: Iterable[Dict[str, str]]) -> Dict[str, int]:
    return dict(Counter(record["host"] for record in records if record.get("host")))


def test_q44() -> None:
    records = [{"host": "laptop-01"}, {"host": "laptop-01"}, {"host": "server-01"}, {}]
    assert count_events_per_host(records) == {"laptop-01": 2, "server-01": 1}

# Q45 - Find Error Logs
# ---------------------------------------------------------------------------
ANSWER_Q45 = """
Problem:
  Return all log lines containing the word ERROR.

Naive approach:
  Use substring matching. It can match words like NOERROR.

Optimal approach:
  Use a word-boundary regex for ERROR and scan once. This is O(n) over log
  text.

Security meaning:
  Error logs often surface failed controls, exploit attempts, and service
  failures during an incident.
"""


ERROR_RE = re.compile(r"\bERROR\b")


def find_error_logs_naive(lines: Iterable[str]) -> List[str]:
    return [line for line in lines if "ERROR" in line]


def find_error_logs(lines: Iterable[str]) -> List[str]:
    return [line for line in lines if ERROR_RE.search(line)]


def test_q45() -> None:
    assert find_error_logs(["INFO ok", "ERROR failed", "NOERROR text"]) == ["ERROR failed"]

# Q46 - Find Warnings and Errors
# ---------------------------------------------------------------------------
ANSWER_Q46 = """
Problem:
  Return all log lines containing either WARN or ERROR.

Naive approach:
  Check 'WARN' in line or 'ERROR' in line.

Optimal approach:
  Use a compiled word-boundary regex for both severities. This is still a
  single pass but avoids false substring matches.

Security meaning:
  WARN and ERROR lines are a compact triage view for reliability and security
  anomalies.
"""


WARN_OR_ERROR_RE = re.compile(r"\b(?:WARN|ERROR)\b")


def find_warnings_and_errors_naive(lines: Iterable[str]) -> List[str]:
    return [line for line in lines if "WARN" in line or "ERROR" in line]


def find_warnings_and_errors(lines: Iterable[str]) -> List[str]:
    return [line for line in lines if WARN_OR_ERROR_RE.search(line)]


def test_q46() -> None:
    assert find_warnings_and_errors(["INFO ok", "WARN slow", "ERROR failed"]) == ["WARN slow", "ERROR failed"]

# Q47 - Count DNS Queries Per Domain
# ---------------------------------------------------------------------------
ANSWER_Q47 = """
Problem:
  Given DNS query logs, count queries per domain.

Naive approach:
  Build unique domains and use repeated scans to count each one.

Optimal approach:
  Use Counter keyed by normalized domain. This is O(n) time and O(d) space.
  For raw key-value logs, parse query=<domain> first.

Security meaning:
  DNS query frequency highlights common destinations, suspicious domains, and
  beaconing candidates.
"""


def count_dns_queries_per_domain_naive(records: List[Dict[str, str]]) -> Dict[str, int]:
    domains: List[str] = []
    for record in records:
        domain = record.get("query")
        if domain and domain not in domains:
            domains.append(domain)
    return {domain: sum(1 for record in records if record.get("query") == domain) for domain in domains}


def count_dns_queries_per_domain(records: Iterable[Dict[str, str]]) -> Dict[str, int]:
    counts: Counter[str] = Counter()
    for record in records:
        domain = record.get("query") or record.get("domain")
        if domain:
            counts[domain.strip().lower().rstrip(".")] += 1
    return dict(counts)


def test_q47() -> None:
    records = [{"query": "Example.com"}, {"query": "example.com."}, {"query": "accounts.google.com"}]
    assert count_dns_queries_per_domain(records) == {"example.com": 2, "accounts.google.com": 1}

# Q48 - Find Suspicious File Downloads
# ---------------------------------------------------------------------------
ANSWER_Q48 = """
Problem:
  Return downloaded filenames ending in .exe, .bat, .ps1, .vbs, or .dll.

Naive approach:
  Check each filename against each suspicious suffix with nested loops.

Optimal approach:
  Extract the suffix once and check membership in a set. This is O(n) over
  filenames plus filename length for suffix extraction.

Security meaning:
  Executable and script downloads can indicate malware staging, lateral
  movement tooling, or risky user behavior.
"""


SUSPICIOUS_DOWNLOAD_EXTENSIONS = {".exe", ".bat", ".ps1", ".vbs", ".dll"}


def find_suspicious_file_downloads_naive(filenames: Iterable[str]) -> List[str]:
    result: List[str] = []
    for filename in filenames:
        lower_name = filename.lower()
        for suffix in SUSPICIOUS_DOWNLOAD_EXTENSIONS:
            if lower_name.endswith(suffix):
                result.append(filename)
                break
    return result


def find_suspicious_file_downloads(filenames: Iterable[str]) -> List[str]:
    result: List[str] = []
    for filename in filenames:
        suffix = PurePosixPath(filename.strip()).suffix.lower()
        if suffix in SUSPICIOUS_DOWNLOAD_EXTENSIONS:
            result.append(filename)
    return result


def test_q48() -> None:
    files = ["invoice.pdf", "tool.exe", "script.PS1", "library.dll", "archive.zip"]
    assert find_suspicious_file_downloads(files) == ["tool.exe", "script.PS1", "library.dll"]

# Q49 - Parse Key-Value Logs
# ---------------------------------------------------------------------------
ANSWER_Q49 = """
Problem:
  Given 'user=alice ip=1.2.3.4 action=login', return a dictionary.

Naive approach:
  Split on spaces, then split each token on '='. This fails when quoted values
  contain spaces.

Optimal approach:
  Use shlex.split to respect quotes, then split each token once on '='. This is
  O(n) over the log line and handles common shell-like key-value logs.

Security meaning:
  Key-value parsing turns raw logs into structured fields for correlation,
  detection, and enrichment.
"""


def parse_key_value_logs_naive(line: str) -> Dict[str, str]:
    result: Dict[str, str] = {}
    for token in line.split():
        if "=" in token:
            key, value = token.split("=", 1)
            result[key] = value
    return result


def parse_key_value_logs(line: str) -> Dict[str, str]:
    result: Dict[str, str] = {}
    for token in shlex.split(line):
        if "=" not in token:
            continue
        key, value = token.split("=", 1)
        if key:
            result[key] = value
    return result


def test_q49() -> None:
    assert parse_key_value_logs("user=alice ip=1.2.3.4 action=login") == {
        "user": "alice",
        "ip": "1.2.3.4",
        "action": "login",
    }
    assert parse_key_value_logs('user=alice reason="bad password"') == {"user": "alice", "reason": "bad password"}

# Q50 - Convert Log Line to JSON Object
# ---------------------------------------------------------------------------
ANSWER_Q50 = """
Problem:
  Parse a simple log line into a JSON-like dictionary.

Naive approach:
  Split the line on spaces and assign fixed positions manually.

Optimal approach:
  Parse the first three tokens as timestamp, severity, and event, then parse
  the remaining key-value fields with the Q49 parser. A dictionary is the right
  structure because it gives named field lookup.

Security meaning:
  Structured JSON-like events are easier to query, alert on, and pass through
  SIEM or data lake pipelines.
"""


def convert_log_line_to_json_object_naive(line: str) -> Dict[str, str]:
    parts = line.split()
    if len(parts) < 3:
        return {}
    result = {"timestamp": parts[0], "severity": parts[1], "event": parts[2]}
    for token in parts[3:]:
        if "=" in token:
            key, value = token.split("=", 1)
            result[key] = value
    return result


def convert_log_line_to_json_object(line: str) -> Dict[str, str]:
    parts = line.strip().split(maxsplit=3)
    if len(parts) < 3:
        return {}

    result = {"timestamp": parts[0], "severity": parts[1], "event": parts[2]}
    if len(parts) == 4:
        result.update(parse_key_value_logs(parts[3]))
    return result


def test_q50() -> None:
    line = "2026-05-29T08:01:11Z INFO LOGIN_SUCCESS user=alice ip=10.0.1.10"
    assert convert_log_line_to_json_object(line) == {
        "timestamp": "2026-05-29T08:01:11Z",
        "severity": "INFO",
        "event": "LOGIN_SUCCESS",
        "user": "alice",
        "ip": "10.0.1.10",
    }
    assert convert_log_line_to_json_object("bad") == {}
