# ---------------------------------------------------------------------------
# Q51 - Caesar Cipher Encrypt
# ---------------------------------------------------------------------------
ANSWER_Q51 = """
Problem:
  Given text and shift value, return Caesar-encrypted text.

Naive approach:
  For each character, calculate its shifted alphabet index manually and append
  to a list.

Optimal approach:
  Build a translation table with str.maketrans and call text.translate(). This
  keeps the loop in optimized Python internals and is O(n) time.

Alternative:
  Character-by-character logic is easier to customize when preserving only a
  specific alphabet or supporting non-English alphabets.

Security meaning:
  Caesar cipher is not secure encryption, but it teaches encoding, decoding,
  and reversible transformations often seen in obfuscation.
"""


def caesar_cipher_encrypt_naive(text: str, shift: int) -> str:
    result = []
    shift %= 26
    for char in text:
        if "a" <= char <= "z":
            result.append(chr((ord(char) - ord("a") + shift) % 26 + ord("a")))
        elif "A" <= char <= "Z":
            result.append(chr((ord(char) - ord("A") + shift) % 26 + ord("A")))
        else:
            result.append(char)
    return "".join(result)


def _caesar_translation_table(shift: int) -> Dict[int, int]:
    shift %= 26
    lower = string.ascii_lowercase
    upper = string.ascii_uppercase
    shifted_lower = lower[shift:] + lower[:shift]
    shifted_upper = upper[shift:] + upper[:shift]
    return str.maketrans(lower + upper, shifted_lower + shifted_upper)


def caesar_cipher_encrypt(text: str, shift: int) -> str:
    return text.translate(_caesar_translation_table(shift))


def test_q51() -> None:
    assert caesar_cipher_encrypt("Abc xyz!", 2) == "Cde zab!"
    assert caesar_cipher_encrypt("abc", 26) == "abc"

# Q52 - Caesar Cipher Decrypt
# ---------------------------------------------------------------------------
ANSWER_Q52 = """
Problem:
  Given Caesar-encrypted text and shift, return plaintext.

Naive approach:
  Repeat the encrypt logic with a negative shift.

Optimal approach:
  Reuse the translation-table encrypt function with -shift. This avoids
  duplicate logic and remains O(n).

Security meaning:
  Decryption mirrors encryption and helps analysts reverse weak obfuscation in
  malware strings or toy challenges.
"""


def caesar_cipher_decrypt_naive(text: str, shift: int) -> str:
    return caesar_cipher_encrypt_naive(text, -shift)


def caesar_cipher_decrypt(text: str, shift: int) -> str:
    return caesar_cipher_encrypt(text, -shift)


def test_q52() -> None:
    assert caesar_cipher_decrypt("Cde zab!", 2) == "Abc xyz!"
    assert caesar_cipher_decrypt(caesar_cipher_encrypt("Security", 7), 7) == "Security"

# Q53 - ROT13 Encoder
# ---------------------------------------------------------------------------
ANSWER_Q53 = """
Problem:
  Implement ROT13 transformation.

Naive approach:
  Treat ROT13 as Caesar shift 13 and transform one character at a time.

Optimal approach:
  Use Python's codecs ROT13 implementation or a fixed translation table. ROT13
  is symmetric, so encoding twice returns the original text.

Security meaning:
  ROT13 is weak obfuscation, not encryption. Analysts still see it in puzzles,
  samples, and simple hiding attempts.
"""


def rot13_encoder_naive(text: str) -> str:
    return caesar_cipher_encrypt_naive(text, 13)


def rot13_encoder(text: str) -> str:
    return codecs.encode(text, "rot_13")


def test_q53() -> None:
    assert rot13_encoder("hello") == "uryyb"
    assert rot13_encoder(rot13_encoder("Security123")) == "Security123"

# Q54 - Base64 Decode String
# ---------------------------------------------------------------------------
ANSWER_Q54 = """
Problem:
  Given a Base64 string, decode it.

Naive approach:
  Call base64.b64decode and assume input is valid.

Optimal approach:
  Use base64.b64decode(..., validate=True), then decode bytes with an explicit
  character encoding. Return None for invalid Base64 or invalid text.

Security meaning:
  Base64 decoding is common in malware analysis, HTTP payload inspection, and
  secret scanning.
"""


def base64_decode_string_naive(encoded: str) -> str:
    return base64.b64decode(encoded).decode("utf-8")


def base64_decode_string(encoded: str) -> Optional[str]:
    try:
        decoded = base64.b64decode(encoded, validate=True)
        return decoded.decode("utf-8")
    except (binascii.Error, UnicodeDecodeError):
        return None


def test_q54() -> None:
    assert base64_decode_string("YWxpY2U=") == "alice"
    assert base64_decode_string("not base64!") is None

# Q55 - Base64 Encode String
# ---------------------------------------------------------------------------
ANSWER_Q55 = """
Problem:
  Given a string, encode it in Base64.

Naive approach:
  Convert to bytes and call base64.b64encode.

Optimal approach:
  The standard library is the optimal production approach. Use explicit UTF-8
  encoding before Base64 and decode the result back to a string.

Security meaning:
  Encoding is not encryption. Base64 protects transport formatting, not
  confidentiality.
"""


def base64_encode_string_naive(text: str) -> str:
    return base64.b64encode(text.encode()).decode()


def base64_encode_string(text: str) -> str:
    return base64.b64encode(text.encode("utf-8")).decode("ascii")


def test_q55() -> None:
    assert base64_encode_string("alice") == "YWxpY2U="
    assert base64_decode_string(base64_encode_string("hello")) == "hello"

# Q56 - Hex to ASCII
# ---------------------------------------------------------------------------
ANSWER_Q56 = """
Problem:
  Convert a hexadecimal string into ASCII text.

Naive approach:
  Process two hex characters at a time, convert each pair with int(pair, 16),
  and call chr().

Optimal approach:
  Use bytes.fromhex(), then decode as ASCII. This handles whitespace between
  hex bytes and validates input. Return None for invalid hex or non-ASCII bytes.

Security meaning:
  Hex decoding is useful for payload inspection, malware strings, and encoded
  indicators.
"""


def hex_to_ascii_naive(hex_string: str) -> str:
    result = []
    compact = hex_string.replace(" ", "")
    for index in range(0, len(compact), 2):
        result.append(chr(int(compact[index:index + 2], 16)))
    return "".join(result)


def hex_to_ascii(hex_string: str) -> Optional[str]:
    try:
        return bytes.fromhex(hex_string).decode("ascii")
    except (ValueError, UnicodeDecodeError):
        return None


def test_q56() -> None:
    assert hex_to_ascii("68656c6c6f") == "hello"
    assert hex_to_ascii("68 69") == "hi"
    assert hex_to_ascii("zz") is None

# Q57 - ASCII to Hex
# ---------------------------------------------------------------------------
ANSWER_Q57 = """
Problem:
  Convert ASCII text into hexadecimal representation.

Naive approach:
  Loop through characters and format ord(char) as two hex digits.

Optimal approach:
  Encode the string to ASCII bytes and call bytes.hex(). This is O(n) and
  validates that the input is ASCII.

Security meaning:
  Hex encoding is common for binary-safe representation of payloads and
  indicators.
"""


def ascii_to_hex_naive(text: str) -> str:
    return "".join(format(ord(char), "02x") for char in text)


def ascii_to_hex(text: str) -> Optional[str]:
    try:
        return text.encode("ascii").hex()
    except UnicodeEncodeError:
        return None


def test_q57() -> None:
    assert ascii_to_hex("hello") == "68656c6c6f"
    assert ascii_to_hex("hi!") == "686921"

# Q58 - XOR Two Equal-Length Strings
# ---------------------------------------------------------------------------
ANSWER_Q58 = """
Problem:
  Given two byte strings, return their XOR result.

Naive approach:
  Loop by index, XOR each pair of bytes, and append to a bytearray.

Optimal approach:
  Use zip over the two byte strings and construct bytes from XOR results. This
  is O(n) time and O(n) output space. Raise an error on unequal lengths because
  silent truncation would hide data loss.

Security meaning:
  XOR appears in stream ciphers, malware obfuscation, and simple payload
  encodings.
"""


def xor_two_equal_length_strings_naive(left: bytes, right: bytes) -> bytes:
    if len(left) != len(right):
        raise ValueError("inputs must have equal length")
    result = bytearray()
    for index in range(len(left)):
        result.append(left[index] ^ right[index])
    return bytes(result)


def xor_two_equal_length_strings(left: bytes, right: bytes) -> bytes:
    if len(left) != len(right):
        raise ValueError("inputs must have equal length")
    return bytes(left_byte ^ right_byte for left_byte, right_byte in zip(left, right))


def test_q58() -> None:
    assert xor_two_equal_length_strings(b"\x0f\xf0", b"\xf0\x0f") == b"\xff\xff"
    try:
        xor_two_equal_length_strings(b"a", b"ab")
    except ValueError:
        pass
    else:
        raise AssertionError("expected ValueError")

# Q59 - Check Hash Length
# ---------------------------------------------------------------------------
ANSWER_Q59 = """
Problem:
  Identify whether a hash string looks like MD5, SHA-1, or SHA-256 by length.

Naive approach:
  Compare string length to 32, 40, and 64.

Optimal approach:
  Normalize casing, verify all characters are hex, then use a length-to-name
  dictionary for O(1) classification.

Security meaning:
  Hash type guessing helps triage indicators, but length alone is not proof of
  the algorithm.
"""


HASH_LENGTHS = {32: "MD5", 40: "SHA-1", 64: "SHA-256"}
HEX_CHARS = set(string.hexdigits)


def check_hash_length_naive(hash_string: str) -> str:
    if len(hash_string) == 32:
        return "MD5"
    if len(hash_string) == 40:
        return "SHA-1"
    if len(hash_string) == 64:
        return "SHA-256"
    return "unknown"


def check_hash_length(hash_string: str) -> str:
    value = hash_string.strip()
    if not value or any(char not in HEX_CHARS for char in value):
        return "unknown"
    return HASH_LENGTHS.get(len(value), "unknown")


def test_q59() -> None:
    assert check_hash_length("a" * 32) == "MD5"
    assert check_hash_length("b" * 40) == "SHA-1"
    assert check_hash_length("c" * 64) == "SHA-256"
    assert check_hash_length("z" * 32) == "unknown"

# Q60 - Detect Weak Hash Algorithm Name
# ---------------------------------------------------------------------------
ANSWER_Q60 = """
Problem:
  Given algorithm names, flag md5 and sha1 as weak.

Naive approach:
  Lowercase each name and compare with 'md5' or 'sha1'.

Optimal approach:
  Normalize names by lowercasing and removing separators, then check membership
  in a set of weak algorithms. Set lookup is O(1) average time.

Security meaning:
  MD5 and SHA-1 are weak for collision resistance and should not be used for
  new security-sensitive designs.
"""


WEAK_HASH_ALGORITHMS = {"md5", "sha1"}


def detect_weak_hash_algorithm_name_naive(algorithms: Iterable[str]) -> Dict[str, bool]:
    result: Dict[str, bool] = {}
    for algorithm in algorithms:
        lower_name = algorithm.lower()
        result[algorithm] = lower_name == "md5" or lower_name == "sha1"
    return result


def _normalize_algorithm_name(name: str) -> str:
    return name.strip().lower().replace("-", "").replace("_", "")


def detect_weak_hash_algorithm_name(algorithms: Iterable[str]) -> Dict[str, bool]:
    return {algorithm: _normalize_algorithm_name(algorithm) in WEAK_HASH_ALGORITHMS for algorithm in algorithms}


def test_q60() -> None:
    assert detect_weak_hash_algorithm_name(["MD5", "sha-1", "SHA-256"]) == {
        "MD5": True,
        "sha-1": True,
        "SHA-256": False,
    }
