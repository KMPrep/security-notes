Topic: Certificate dates, cloud configuration checks, secret keyword scans, and HTTPS.

from datetime import date, datetime, timedelta
import re
from typing import Dict, Iterable, List, Optional, Set
from urllib.parse import urlparse


def _parse_date(value: object) -> Optional[date]:
    if isinstance(value, date) and not isinstance(value, datetime):
        return value
    if isinstance(value, datetime):
        return value.date()
    if isinstance(value, str):
        try:
            return datetime.strptime(value.strip(), "%Y-%m-%d").date()
        except ValueError:
            return None
    return None

# Q61 - Find Expired Certificates
# ---------------------------------------------------------------------------
ANSWER_Q61 = """
Problem:
  Given certificate expiry dates, return those already expired.

Naive approach:
  Compare date strings directly. This works only if all dates are valid
  YYYY-MM-DD strings.

Optimal approach:
  Parse expiry values into date objects and compare with today's date. This is
  O(n) time and skips malformed records safely.

Security meaning:
  Expired certificates can cause outages and can weaken user trust in secure
  connections.
"""


def find_expired_certificates_naive(certificates: Iterable[Dict[str, str]], today: str) -> List[Dict[str, str]]:
    return [certificate for certificate in certificates if certificate.get("expires_on", "") < today]


def find_expired_certificates(certificates: Iterable[Dict[str, object]], today: Optional[date] = None) -> List[Dict[str, object]]:
    current_day = today or date.today()
    expired: List[Dict[str, object]] = []
    for certificate in certificates:
        expiry = _parse_date(certificate.get("expires_on"))
        if expiry is not None and expiry < current_day:
            expired.append(certificate)
    return expired


def test_q61() -> None:
    today = date(2026, 5, 30)
    certs = [{"name": "old", "expires_on": "2026-05-29"}, {"name": "new", "expires_on": "2026-06-01"}]
    assert find_expired_certificates(certs, today) == [{"name": "old", "expires_on": "2026-05-29"}]

# Q62 - Find Certificates Expiring Soon
# ---------------------------------------------------------------------------
ANSWER_Q62 = """
Problem:
  Return certificates expiring within the next 30 days.

Naive approach:
  Convert dates to strings and compare them lexicographically. This is fragile
  when formats differ.

Optimal approach:
  Parse dates and compare against the inclusive window [today, today + 30 days].
  This is O(n) time.

Security meaning:
  Expiring-soon certificates should be renewed before they become outages or
  incident noise.
"""


def find_certificates_expiring_soon_naive(certificates: Iterable[Dict[str, str]], today: str, cutoff: str) -> List[Dict[str, str]]:
    return [
        certificate
        for certificate in certificates
        if today <= certificate.get("expires_on", "") <= cutoff
    ]


def find_certificates_expiring_soon(
    certificates: Iterable[Dict[str, object]],
    today: Optional[date] = None,
    days: int = 30,
) -> List[Dict[str, object]]:
    current_day = today or date.today()
    cutoff = current_day + timedelta(days=days)
    expiring: List[Dict[str, object]] = []
    for certificate in certificates:
        expiry = _parse_date(certificate.get("expires_on"))
        if expiry is not None and current_day <= expiry <= cutoff:
            expiring.append(certificate)
    return expiring


def test_q62() -> None:
    today = date(2026, 5, 30)
    certs = [
        {"name": "expired", "expires_on": "2026-05-29"},
        {"name": "soon", "expires_on": "2026-06-15"},
        {"name": "later", "expires_on": "2026-08-01"},
    ]
    assert find_certificates_expiring_soon(certs, today) == [{"name": "soon", "expires_on": "2026-06-15"}]

# Q63 - Check Open Ports Against Allowlist
# ---------------------------------------------------------------------------
ANSWER_Q63 = """
Problem:
  Given open ports and allowed ports, return ports not in the allowlist.

Naive approach:
  For each open port, scan the allowed port list. This is O(n * a).

Optimal approach:
  Convert allowed ports to a set and check membership in O(1) average time.
  Preserve input order in the output.

Security meaning:
  Non-allowlisted ports can indicate exposed services, drift, or unauthorized
  deployments.
"""


def check_open_ports_against_allowlist_naive(open_ports: Iterable[int], allowed_ports: List[int]) -> List[int]:
    return [port for port in open_ports if port not in allowed_ports]


def check_open_ports_against_allowlist(open_ports: Iterable[int], allowed_ports: Iterable[int]) -> List[int]:
    allowed: Set[int] = set(allowed_ports)
    return [port for port in open_ports if port not in allowed]


def test_q63() -> None:
    assert check_open_ports_against_allowlist([22, 80, 443, 3389], [80, 443]) == [22, 3389]
    assert check_open_ports_against_allowlist([], [80]) == []

# Q64 - Check User Against Admin List
# ---------------------------------------------------------------------------
ANSWER_Q64 = """
Problem:
  Given a username and admin list, return whether the user is admin.

Naive approach:
  Scan the admin list linearly and compare each value.

Optimal approach:
  Convert the admin list to a set for O(1) average membership checks. Normalize
  casing if usernames are case-insensitive in the environment.

Security meaning:
  Fast admin checks are useful for authorization decisions and privileged user
  monitoring.
"""


def check_user_against_admin_list_naive(username: str, admins: Iterable[str]) -> bool:
    for admin in admins:
        if admin == username:
            return True
    return False


def check_user_against_admin_list(username: str, admins: Iterable[str], case_sensitive: bool = False) -> bool:
    if case_sensitive:
        return username in set(admins)
    normalized_admins = {admin.casefold() for admin in admins}
    return username.casefold() in normalized_admins


def test_q64() -> None:
    assert check_user_against_admin_list("Alice", ["alice", "root"]) is True
    assert check_user_against_admin_list("bob", ["alice", "root"]) is False

# Q65 - Detect Public S3 Bucket Flag
# ---------------------------------------------------------------------------
ANSWER_Q65 = """
Problem:
  Given bucket metadata, return buckets marked public.

Naive approach:
  Loop through records and check if public == True.

Optimal approach:
  Single-pass filter over dictionaries. If metadata has multiple public flags,
  treat any explicit public flag as risky and return the whole bucket record as
  evidence.

Security meaning:
  Public buckets are a common cause of accidental data exposure.
"""


def detect_public_s3_bucket_flag_naive(buckets: Iterable[Dict[str, object]]) -> List[Dict[str, object]]:
    return [bucket for bucket in buckets if bucket.get("public") is True]


def detect_public_s3_bucket_flag(buckets: Iterable[Dict[str, object]]) -> List[Dict[str, object]]:
    public_buckets: List[Dict[str, object]] = []
    for bucket in buckets:
        if bucket.get("public") is True or bucket.get("is_public") is True:
            public_buckets.append(bucket)
    return public_buckets


def test_q65() -> None:
    buckets = [{"name": "private", "public": False}, {"name": "public", "public": True}]
    assert detect_public_s3_bucket_flag(buckets) == [{"name": "public", "public": True}]

# Q66 - Find IAM Users Without MFA
# ---------------------------------------------------------------------------
ANSWER_Q66 = """
Problem:
  Given IAM user records, return users where mfa_enabled is false.

Naive approach:
  Loop and append records with mfa_enabled == False.

Optimal approach:
  Single-pass filter. Treat only explicit False as non-compliant, and decide
  separately whether missing MFA fields should be flagged.

Security meaning:
  Privileged or cloud users without MFA are high-risk targets for credential
  theft and account takeover.
"""


def find_iam_users_without_mfa_naive(users: Iterable[Dict[str, object]]) -> List[Dict[str, object]]:
    return [user for user in users if user.get("mfa_enabled") is False]


def find_iam_users_without_mfa(users: Iterable[Dict[str, object]], flag_missing: bool = False) -> List[Dict[str, object]]:
    result: List[Dict[str, object]] = []
    for user in users:
        mfa_enabled = user.get("mfa_enabled")
        if mfa_enabled is False or (flag_missing and mfa_enabled is None):
            result.append(user)
    return result


def test_q66() -> None:
    users = [{"user": "alice", "mfa_enabled": True}, {"user": "bob", "mfa_enabled": False}, {"user": "carol"}]
    assert find_iam_users_without_mfa(users) == [{"user": "bob", "mfa_enabled": False}]
    assert find_iam_users_without_mfa(users, flag_missing=True) == [
        {"user": "bob", "mfa_enabled": False},
        {"user": "carol"},
    ]

# Q67 - Find Security Groups Open to Internet
# ---------------------------------------------------------------------------
ANSWER_Q67 = """
Problem:
  Given security group rules, return those with CIDR 0.0.0.0/0.

Naive approach:
  Loop and compare cidr exactly to '0.0.0.0/0'.

Optimal approach:
  Use a set of internet-open CIDR values for O(1) membership. Include ::/0 if
  IPv6 should also be considered internet-open.

Security meaning:
  Internet-open security group rules can expose admin ports, databases, and
  internal services to attackers.
"""


INTERNET_OPEN_CIDRS = {"0.0.0.0/0", "::/0"}


def find_security_groups_open_to_internet_naive(rules: Iterable[Dict[str, str]]) -> List[Dict[str, str]]:
    return [rule for rule in rules if rule.get("cidr") == "0.0.0.0/0"]


def find_security_groups_open_to_internet(rules: Iterable[Dict[str, str]], include_ipv6: bool = True) -> List[Dict[str, str]]:
    open_cidrs = INTERNET_OPEN_CIDRS if include_ipv6 else {"0.0.0.0/0"}
    return [rule for rule in rules if rule.get("cidr") in open_cidrs]


def test_q67() -> None:
    rules = [{"sg": "web", "cidr": "0.0.0.0/0"}, {"sg": "db", "cidr": "10.0.0.0/8"}]
    assert find_security_groups_open_to_internet(rules) == [{"sg": "web", "cidr": "0.0.0.0/0"}]

# Q68 - Detect Hardcoded Password Keyword
# ---------------------------------------------------------------------------
ANSWER_Q68 = """
Problem:
  Return source code lines containing password=, passwd=, or pwd=.

Naive approach:
  Lowercase each line and check each keyword with substring membership.

Optimal approach:
  Use one compiled case-insensitive regex with optional whitespace around '='.
  This catches 'password = value' as well as 'password=value'.

Security meaning:
  Hardcoded passwords are a direct secret leakage risk and should move to a
  secret manager.
"""


PASSWORD_KEYWORD_RE = re.compile(r"\b(?:password|passwd|pwd)\s*=", re.IGNORECASE)
PASSWORD_KEYWORDS = ("password=", "passwd=", "pwd=")


def detect_hardcoded_password_keyword_naive(lines: Iterable[str]) -> List[str]:
    result: List[str] = []
    for line in lines:
        lower_line = line.lower()
        if any(keyword in lower_line for keyword in PASSWORD_KEYWORDS):
            result.append(line)
    return result


def detect_hardcoded_password_keyword(lines: Iterable[str]) -> List[str]:
    return [line for line in lines if PASSWORD_KEYWORD_RE.search(line)]


def test_q68() -> None:
    lines = ['password = "Welcome123"', "normal_config = true", "pwd=secret"]
    assert detect_hardcoded_password_keyword(lines) == ['password = "Welcome123"', "pwd=secret"]

# Q69 - Detect Hardcoded API Key Keyword
# ---------------------------------------------------------------------------
ANSWER_Q69 = """
Problem:
  Return code lines containing api_key, secret_key, access_token, or private_key.

Naive approach:
  Lowercase each line and check each keyword with substring membership.

Optimal approach:
  Compile one regex for all keywords. A regex keeps matching logic central and
  makes it easy to extend with word boundaries or assignment checks.

Security meaning:
  API keys, private keys, and access tokens in source code can lead to cloud

API_SECRET_RE = re.compile(r"\b(?:api_key|secret_key|access_token|private_key)\b", re.IGNORECASE)

    result: List[str] = []

            result.append(line)
    return result


def detect_hardcoded_api_key_keyword(lines: Iterable[str]) -> List[str]:

def test_q69() -> None:

# Q70 - Check If URL Uses HTTPS
# ---------------------------------------------------------------------------
ANSWER_Q70 = """
Problem:
  Given a URL, return whether it starts with https://.

Naive approach:
  Call url.startswith('https://').

Optimal approach:
  Strip whitespace and parse the URL with urlparse, then check that the scheme
  is https. This is clearer when URLs may have leading whitespace or mixed
  case.

Security meaning:
  HTTPS is required to protect confidentiality and integrity in transit, though
  it does not prove the destination is safe.
"""


def check_if_url_uses_https_naive(url: str) -> bool:
    return url.startswith("https://")


def check_if_url_uses_https(url: str) -> bool:
    parsed = urlparse(url.strip())
    return parsed.scheme.lower() == "https"


def test_q70() -> None:
    assert check_if_url_uses_https("https://example.com") is True
    assert check_if_url_uses_https(" HTTP://example.com") is False
    assert check_if_url_uses_https("  HTTPS://example.com") is True
