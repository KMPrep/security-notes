"""
=============================================================================
Q10 [Hard] — Suppress Duplicate Alerts with Escalation
=============================================================================

PROBLEM
-------
A detection rule fires repeatedly for the same condition (e.g. brute force
continuing for 10 minutes).  Emitting an alert on every match floods analysts.

Requirements:
  1. Suppress duplicate alerts within a configurable suppression window.
  2. Track the suppressed count so the severity can be escalated when the
     same alert fires many times.
  3. Emit a "re-open" alert if the same pattern fires after the window expires.
  4. Emit a "resolved" notification when the pattern stops for resolve_after_seconds.
  5. Different alert keys (rule_name + group_by fields) have independent windows.

CLARIFYING QUESTIONS
--------------------
1. How is "same alert" defined?
   → Same (rule_name, entity_key) pair  — where entity_key is a tuple of
     the relevant field values (e.g. rule=brute_force, src=10.0.0.1).
2. Suppression window: rolling or fixed?
   → Fixed window from first alert; after expiry, next occurrence re-opens.
3. What does "escalation" mean here?
   → If suppress_count reaches escalation_threshold, bump severity one level.
4. Should suppressed alerts be recorded for audit?
   → Yes; maintain a suppressed_count and last_suppressed_ts.
5. What is "resolved"?
   → The condition has not fired for resolve_after_seconds.
6. Memory bound?
   → Evict state for keys that have been resolved.
7. Thread safety?
   → Single-threaded; note locking requirements for multi-thread.

=============================================================================
APPROACH 1 — NAIVE (dict of suppress windows, no escalation)
=============================================================================
Simple dict: key → expiry_timestamp.
If current_ts < expiry → suppress. Else → pass-through + reset expiry.
  No suppressed_count, no escalation, no resolve notifications.
  O(1) per alert.

=============================================================================
APPROACH 2 — FULL DEDUPLICATION ENGINE (RECOMMENDED)
=============================================================================
State per (rule, entity_key):
  - status: "open" | "suppressed" | "resolved"
  - window_start_ts
  - window_end_ts   = window_start_ts + suppress_seconds
  - suppress_count  (incremented for each duplicate within window)
  - last_seen_ts
  - severity        (current, may have been escalated)

Transitions:
  RESOLVED → OPEN on new match (re-open)
  OPEN → SUPPRESSED after first alert in window is emitted
  SUPPRESSED → SUPPRESSED + increment count (no alert, unless escalation triggers)
  SUPPRESSED → check escalation threshold → emit escalated alert if crossed
  Any → RESOLVED if now - last_seen_ts > resolve_after_seconds

  Time : O(1) per alert
  Space: O(active keys)

=============================================================================
"""

from __future__ import annotations

from collections import defaultdict
from dataclasses import dataclass, field
from datetime import datetime
from enum import Enum
from typing import Any, Dict, List, Optional, Set, Tuple


# ---------------------------------------------------------------------------
# Severity ladder
# ---------------------------------------------------------------------------
SEVERITY_LADDER = ["info", "low", "medium", "high", "critical"]

def _escalate(current: str) -> str:
    idx = SEVERITY_LADDER.index(current.lower()) if current.lower() in SEVERITY_LADDER else 1
    return SEVERITY_LADDER[min(idx + 1, len(SEVERITY_LADDER) - 1)]


# ---------------------------------------------------------------------------
# Alert and Notification Types
# ---------------------------------------------------------------------------

class AlertAction(Enum):
    EMIT = "emit"              # New or re-opened alert; send to analyst
    SUPPRESS = "suppress"      # Duplicate within window; drop silently
    ESCALATE = "escalate"      # Threshold crossed; re-emit with higher severity
    RESOLVE = "resolve"        # Pattern stopped; notify resolution
    REOPEN = "reopen"          # Was resolved; same pattern fires again


@dataclass
class AlertDecision:
    action: AlertAction
    rule_name: str
    entity_key: Tuple
    severity: str
    suppress_count: int = 0
    original_ts: float = 0.0
    current_ts: float = 0.0
    message: str = ""


# ---------------------------------------------------------------------------
# APPROACH 1 — Naive (simple expiry window, no escalation)
# ---------------------------------------------------------------------------

class NaiveSuppressor:
    """
    Simplest possible deduplication: suppress identical alerts within
    suppress_seconds of each other.
    No escalation, no resolve, no count tracking.
    """

    def __init__(self, suppress_seconds: float = 300.0) -> None:
        self.suppress_seconds = suppress_seconds
        # (rule_name, entity_key) → expiry_ts
        self._windows: Dict[Tuple, float] = {}

    def process(self, rule_name: str, entity_key: Tuple, ts: float,
                severity: str) -> AlertDecision:
        key = (rule_name, entity_key)
        expiry = self._windows.get(key, 0.0)

        if ts < expiry:
            return AlertDecision(AlertAction.SUPPRESS, rule_name, entity_key,
                                 severity, current_ts=ts)
        # New or re-opened
        self._windows[key] = ts + self.suppress_seconds
        return AlertDecision(AlertAction.EMIT, rule_name, entity_key,
                             severity, current_ts=ts)


# ---------------------------------------------------------------------------
# APPROACH 2 — Full Deduplication Engine with Escalation (RECOMMENDED)
# ---------------------------------------------------------------------------

@dataclass
class _KeyState:
    rule_name: str
    entity_key: Tuple
    status: str = "resolved"       # open | suppressed | resolved
    original_severity: str = "info"
    current_severity: str = "info"
    window_start_ts: float = 0.0
    window_end_ts: float = 0.0
    last_seen_ts: float = 0.0
    suppress_count: int = 0         # duplicates suppressed in current window
    total_count: int = 0            # total times seen (all windows)
    escalation_emitted: Set[int] = field(default_factory=set)  # escalation thresholds hit


class AlertDeduplicator:
    """
    Full deduplication + escalation + resolve engine.

    suppress_seconds   : window length for deduplication
    escalation_thresholds : list of suppress_counts that trigger escalation
                             e.g. [5, 20, 100] → escalate at 5th, 20th, 100th dup
    resolve_after_seconds : idle time before sending resolve notification
    """

    def __init__(
        self,
        suppress_seconds: float = 300.0,
        escalation_thresholds: List[int] = None,
        resolve_after_seconds: float = 1800.0,
    ) -> None:
        self.suppress_seconds = suppress_seconds
        self.escalation_thresholds = sorted(escalation_thresholds or [10, 50, 200])
        self.resolve_after = resolve_after_seconds
        self._states: Dict[Tuple, _KeyState] = {}

    def _get_state(self, rule_name: str, entity_key: Tuple, severity: str) -> _KeyState:
        k = (rule_name, entity_key)
        if k not in self._states:
            self._states[k] = _KeyState(
                rule_name=rule_name, entity_key=entity_key,
                original_severity=severity, current_severity=severity,
            )
        return self._states[k]

    def process(self, rule_name: str, entity_key: Tuple, ts: float,
                severity: str = "medium") -> AlertDecision:
        """
        Process one alert firing.  Returns the AlertDecision (what to do).
        """
        state = self._get_state(rule_name, entity_key, severity)
        state.last_seen_ts = ts
        state.total_count += 1

        # --- Case 1: resolved / never seen → open new alert ---
        if state.status == "resolved":
            action = AlertAction.EMIT if state.total_count == 1 else AlertAction.REOPEN
            state.status = "suppressed"  # after first emit, further dups are suppressed
            state.window_start_ts = ts
            state.window_end_ts = ts + self.suppress_seconds
            state.suppress_count = 0
            state.escalation_emitted = set()
            state.original_severity = severity
            state.current_severity = severity
            msg = "New alert" if action == AlertAction.EMIT else "Alert re-opened after quiet period"
            return AlertDecision(action, rule_name, entity_key, severity,
                                 current_ts=ts, message=msg)

        # --- Case 2: within active window ---
        if ts <= state.window_end_ts:
            state.suppress_count += 1

            # Check escalation thresholds
            for threshold in self.escalation_thresholds:
                if (state.suppress_count >= threshold
                        and threshold not in state.escalation_emitted):
                    state.escalation_emitted.add(threshold)
                    state.current_severity = _escalate(state.current_severity)
                    return AlertDecision(
                        AlertAction.ESCALATE, rule_name, entity_key,
                        state.current_severity,
                        suppress_count=state.suppress_count,
                        current_ts=ts,
                        message=f"Escalated after {state.suppress_count} duplicates",
                    )

            return AlertDecision(AlertAction.SUPPRESS, rule_name, entity_key,
                                 state.current_severity,
                                 suppress_count=state.suppress_count,
                                 current_ts=ts)

        # --- Case 3: window expired → start new window, emit (re-fire) ---
        state.window_start_ts = ts
        state.window_end_ts = ts + self.suppress_seconds
        state.suppress_count = 0
        state.escalation_emitted = set()
        # Keep escalated severity if it was raised; could optionally reset here
        return AlertDecision(AlertAction.EMIT, rule_name, entity_key,
                             state.current_severity,
                             current_ts=ts,
                             message="Alert re-fired after window expiry")

    def check_resolutions(self, now: float) -> List[AlertDecision]:
        """
        Call periodically (e.g. every minute) to emit resolve notifications
        for keys that have been quiet for resolve_after_seconds.
        """
        resolved = []
        for key, state in self._states.items():
            if (state.status != "resolved"
                    and now - state.last_seen_ts >= self.resolve_after):
                state.status = "resolved"
                resolved.append(AlertDecision(
                    AlertAction.RESOLVE, state.rule_name, state.entity_key,
                    state.current_severity,
                    suppress_count=state.suppress_count,
                    current_ts=now,
                    message=f"Resolved after {now - state.last_seen_ts:.0f}s quiet",
                ))
        return resolved

    def stats(self) -> Dict[Tuple, dict]:
        return {
            (s.rule_name, s.entity_key): {
                "status": s.status,
                "total_count": s.total_count,
                "suppress_count": s.suppress_count,
                "current_severity": s.current_severity,
            }
            for s in self._states.values()
        }

# Test Cases
# ---------------------------------------------------------------------------

def test_naive_suppress():
    sup = NaiveSuppressor(suppress_seconds=60)
    d1 = sup.process("r1", ("10.0.0.1",), 0.0, "medium")
    assert d1.action == AlertAction.EMIT

    d2 = sup.process("r1", ("10.0.0.1",), 30.0, "medium")
    assert d2.action == AlertAction.SUPPRESS

    d3 = sup.process("r1", ("10.0.0.1",), 70.0, "medium")
    assert d3.action == AlertAction.EMIT  # window expired
    print("NaiveSuppressor: passed")


def test_emit_on_first():
    eng = AlertDeduplicator(suppress_seconds=300, escalation_thresholds=[5, 20])
    d = eng.process("brute_force", ("alice",), 0.0, "medium")
    assert d.action == AlertAction.EMIT
    print("First alert emitted: passed")


def test_suppress_duplicates():
    eng = AlertDeduplicator(suppress_seconds=300, escalation_thresholds=[5, 20])
    eng.process("r", ("alice",), 0.0, "medium")
    for i in range(1, 4):
        d = eng.process("r", ("alice",), float(i * 10), "medium")
        assert d.action == AlertAction.SUPPRESS, f"Expected SUPPRESS at i={i}: {d}"
    print("Suppress duplicates: passed")


def test_escalation():
    eng = AlertDeduplicator(suppress_seconds=3600, escalation_thresholds=[3, 10])
    eng.process("r", ("bob",), 0.0, "low")
    escalated = None
    for i in range(1, 15):
        d = eng.process("r", ("bob",), float(i * 10), "low")
        if d.action == AlertAction.ESCALATE:
            escalated = d
            print(f"  Escalated at count={d.suppress_count} new_sev={d.severity}")
    assert escalated is not None, "Expected at least one escalation"
    print("Escalation: passed")


def test_window_expiry_refire():
    eng = AlertDeduplicator(suppress_seconds=60)
    eng.process("r", ("host-1",), 0.0, "high")
    eng.process("r", ("host-1",), 30.0, "high")  # suppress

    d = eng.process("r", ("host-1",), 100.0, "high")  # past window
    assert d.action == AlertAction.EMIT, f"Expected EMIT after window expiry: {d}"
    print("Window expiry refire: passed")


def test_resolve_notification():
    eng = AlertDeduplicator(suppress_seconds=60, resolve_after_seconds=120)
    eng.process("r", ("ip-x",), 0.0, "high")
    eng.process("r", ("ip-x",), 30.0, "high")

    # Nothing resolved yet at t=100
    resolved = eng.check_resolutions(100.0)
    assert len(resolved) == 0

    # At t=200, 200s idle → resolve
    resolved = eng.check_resolutions(200.0)
    assert len(resolved) == 1
    assert resolved[0].action == AlertAction.RESOLVE
    print("Resolve notification: passed")


def test_reopen_after_resolve():
    eng = AlertDeduplicator(suppress_seconds=60, resolve_after_seconds=120)
    eng.process("r", ("user-a",), 0.0, "medium")
    eng.check_resolutions(200.0)  # mark resolved

    d = eng.process("r", ("user-a",), 300.0, "medium")  # fires again after resolve
    assert d.action == AlertAction.REOPEN, f"Expected REOPEN: {d}"
    print("Reopen after resolve: passed")


def test_independent_keys():
    """Different entity keys should not interfere."""
    eng = AlertDeduplicator(suppress_seconds=3600)
    d1 = eng.process("r", ("10.0.0.1",), 0.0, "low")
    d2 = eng.process("r", ("10.0.0.2",), 0.0, "low")
    assert d1.action == AlertAction.EMIT
    assert d2.action == AlertAction.EMIT  # independent key
    print("Independent keys: passed")

"""
THREAT MODEL
  Without suppression, a single persistent attacker (e.g. long-running brute
  force) can generate thousands of identical alerts per hour, causing:
  - Alert fatigue: analysts ignore all alerts
  - PagerDuty flooding: on-call engineers wake up to 500 pages
  - SOAR playbook loops: automated responses trigger repeatedly, wasting resources

  Suppression with escalation gives the best of both worlds:
  - Analyst sees ONE alert when the attack starts.
  - After N duplicates (5, 20, 100), the alert is escalated to ensure it's
    not buried during high-volume attacks.
  - When the attack stops, a RESOLVE notification confirms clean state.

ESCALATION THRESHOLDS — DESIGN GUIDANCE
  - threshold=5: brief burst (automated scan that ran for a few seconds)
  - threshold=50: moderate persistence (minutes-long brute force)
  - threshold=500: severe persistence (hours-long campaign)
  Use these to route to different teams: medium → Tier 1, high → Tier 2, critical → Tier 3.

FAILURE MODES
  - Suppression window too long: a new attacker from a different vector is
    suppressed because a previous window is still open for the same (rule, entity).
    Mitigate: include more specific fields in entity_key (e.g. add src_ip).
  - Suppression window too short: burst attackers fire many windows rapidly;
    each window emits a new alert → partial flooding.
    Mitigate: use exponential backoff for window length after repeated re-fires.
  - Missing resolve: if the process crashes before check_resolutions runs,
    resolve notifications are never sent.
    Mitigate: persist state to durable store; recover on restart.

PRODUCTION HARDENING
  - Store _states in Redis with TTL = suppress_seconds + resolve_after_seconds.
  - Run check_resolutions in a scheduled job (every 60s).
  - Expose a /suppress-status API for SOC dashboards.
  - Log every SUPPRESS action with suppress_count for post-incident auditing.
"""
