"""
=============================================================================
Q17 [Hard] — Container Escape Indicators
=============================================================================

PROBLEM
-------
Detect process and system call events that suggest a container escape attempt.
Container escapes exploit the gap between the container isolation boundary
and the host kernel.  Indicators include:

  1. nsenter / unshare / chroot calls — namespace manipulation
  2. Mount syscall with unusual targets (/proc, /sys, /dev, host paths)
  3. Process executing from /proc/*/fd/* or /proc/*/mem (memfd tricks)
  4. CAP_SYS_ADMIN capability usage
  5. Writing to /proc/sysrq-trigger, /proc/kcore, /proc/kmem
  6. Child process PID 1 in a different PID namespace (pid_ns change)
  7. Docker socket access: open("/var/run/docker.sock", ...)
  8. cgroup release_agent write (CVE-2022-0492 pattern)
  9. Unusual parent for sensitive files (e.g. /etc/crontab written by containerd-shim)
 10. runc binary execution from inside a container (CVE-2019-5736 pattern)

Input: a stream of process/syscall events with fields:
  pid, ppid, container_id, exe, cmdline, syscall, path, ts, uid, ns_pid

CLARIFYING QUESTIONS
--------------------
1. Are events enriched with container_id?
   → Yes; container_id="" means host.
2. Are we on the host (eBPF) or inside the container?
   → Assume host-level eBPF sensor (Falco-style).
3. Do we need to correlate across multiple events (e.g. fork → exec → write)?
   → Yes for cgroup escape pattern; stateful chain detection needed.
4. What is the baseline for "unusual"?
   → Configurable allowlist of known-good processes that may trigger these.
5. PID namespace change detection?
   → Compare ns_pid of process vs ns_pid of parent; alert if they differ
     AND event comes from a container (container_id != "").

=============================================================================
APPROACH 1 — NAIVE (per-event keyword match + path check)
=============================================================================
Simple list of indicators; match event fields against patterns.
  O(I) per event  (I = number of indicators, ≈ 20)
  No stateful correlation.

=============================================================================
APPROACH 2 — INDEXED DISPATCH + STATEFUL CHAIN DETECTION (RECOMMENDED)
=============================================================================
  - Dict of (exe_basename / syscall / path_prefix) → handler
  - Chain state machine: track per-container multi-step escape sequences
    (e.g. nsenter → mount → exec)
  O(1) per event for direct rules; O(1) per event for chain state.

=============================================================================
"""

from __future__ import annotations

import os
from collections import defaultdict
from dataclasses import dataclass, field
from datetime import datetime
from enum import Enum
from typing import Dict, FrozenSet, List, Optional, Set

# ---------------------------------------------------------------------------
# Known-bad patterns
# ---------------------------------------------------------------------------

_ESCAPE_EXES: FrozenSet[str] = frozenset({
    "nsenter", "unshare", "chroot", "pivot_root",
    "runc", "containerd-shim", "docker",  # runc/docker should not run *inside* containers
})

_ESCAPE_PATHS: FrozenSet[str] = frozenset({
    "/proc/sysrq-trigger",
    "/proc/kcore",
    "/proc/kmem",
    "/proc/sys/kernel/core_pattern",
    "/var/run/docker.sock",
    "/run/docker.sock",
    "/run/containerd/containerd.sock",
})

_ESCAPE_PATH_PREFIXES: FrozenSet[str] = frozenset({
    "/proc/",       # accessing arbitrary /proc paths from container
    "/sys/",        # sysfs manipulation
    "/dev/",        # device access
    "/host",        # host filesystem mounts
    "/mnt/host",
    "/var/lib/docker",
    "/run/snapd",
})

_CGROUP_PATHS: FrozenSet[str] = frozenset({
    "/sys/fs/cgroup",
    "/cgroup",
})

_DANGEROUS_SYSCALLS: FrozenSet[str] = frozenset({
    "mount", "umount", "umount2",
    "pivot_root", "chroot",
    "unshare", "setns",
    "ptrace",
    "process_vm_readv", "process_vm_writev",
    "perf_event_open",
    "bpf",
    "kexec_load", "kexec_file_load",
})


# ---------------------------------------------------------------------------
# Data Types
# ---------------------------------------------------------------------------

@dataclass
class ContainerEvent:
    pid: int
    ppid: int
    container_id: str    # "" = host
    exe: str
    cmdline: str
    syscall: str
    path: str            # file path accessed (for open/write/mount)
    ts: float
    uid: int
    ns_pid: int          # PID in the container's PID namespace
    event_type: str      # process | syscall | file
    raw: dict = field(default_factory=dict)


@dataclass
class EscapeAlert:
    alert_type: str
    severity: str
    container_id: str
    pid: int
    exe: str
    ts: float
    detail: str
    indicator: str       # which indicator triggered this


def _parse_ts(ts) -> Optional[float]:
    if isinstance(ts, (int, float)):
        return float(ts)
    try:
        return datetime.fromisoformat(str(ts).replace("Z", "+00:00")).timestamp()
    except (ValueError, TypeError):
        return None


def _parse_event(raw: dict) -> Optional[ContainerEvent]:
    ts = _parse_ts(raw.get("ts"))
    if ts is None:
        return None
    return ContainerEvent(
        pid=int(raw.get("pid", 0)),
        ppid=int(raw.get("ppid", 0)),
        container_id=raw.get("container_id", ""),
        exe=raw.get("exe", ""),
        cmdline=raw.get("cmdline", ""),
        syscall=raw.get("syscall", ""),
        path=raw.get("path", ""),
        ts=ts,
        uid=int(raw.get("uid", 0)),
        ns_pid=int(raw.get("ns_pid", 0)),
        event_type=raw.get("event_type", "process"),
        raw=raw,
    )


# ---------------------------------------------------------------------------
# Helper functions
# ---------------------------------------------------------------------------

def _exe_basename(exe: str) -> str:
    return os.path.basename(exe).lower()


def _path_is_dangerous(path: str) -> Optional[str]:
    """Return the matching dangerous path or prefix, or None."""
    if path in _ESCAPE_PATHS:
        return path
    for prefix in _ESCAPE_PATH_PREFIXES:
        if path.startswith(prefix):
            return prefix
    return None


# ---------------------------------------------------------------------------
# APPROACH 1 — Naive per-event check
# ---------------------------------------------------------------------------

class NaiveEscapeDetector:
    """
    Stateless per-event escape indicator detector.
    O(I) per event.
    """

    def __init__(self, allowlist: Optional[Set[str]] = None) -> None:
        self.allowlist = allowlist or set()
        self.malformed = 0

    def process(self, raw: dict) -> List[EscapeAlert]:
        evt = _parse_event(raw)
        if evt is None:
            self.malformed += 1
            return []
        if _exe_basename(evt.exe) in self.allowlist or evt.container_id == "":
            # Not from inside a container — only flag container-scoped events
            return []

        alerts = []
        base_exe = _exe_basename(evt.exe)

        # Rule 1: suspicious exe inside container
        if base_exe in _ESCAPE_EXES:
            alerts.append(EscapeAlert(
                alert_type="suspicious_exe_in_container",
                severity="high",
                container_id=evt.container_id,
                pid=evt.pid, exe=evt.exe, ts=evt.ts,
                detail=f"{base_exe!r} executed inside container",
                indicator="exe_in_container",
            ))

        # Rule 2: dangerous syscall from container
        if evt.syscall in _DANGEROUS_SYSCALLS:
            alerts.append(EscapeAlert(
                alert_type="dangerous_syscall",
                severity="critical",
                container_id=evt.container_id,
                pid=evt.pid, exe=evt.exe, ts=evt.ts,
                detail=f"syscall {evt.syscall!r} from container process {base_exe!r}",
                indicator="dangerous_syscall",
            ))

        # Rule 3: path access
        matched_path = _path_is_dangerous(evt.path)
        if matched_path and evt.event_type in ("file", "syscall"):
            severity = "critical" if evt.path in _ESCAPE_PATHS else "high"
            alerts.append(EscapeAlert(
                alert_type="dangerous_path_access",
                severity=severity,
                container_id=evt.container_id,
                pid=evt.pid, exe=evt.exe, ts=evt.ts,
                detail=f"Container {evt.container_id!r} accessed {evt.path!r}",
                indicator="path_access",
            ))

        # Rule 4: docker socket access
        if "/docker.sock" in evt.path or "/containerd.sock" in evt.path:
            alerts.append(EscapeAlert(
                alert_type="docker_socket_access",
                severity="critical",
                container_id=evt.container_id,
                pid=evt.pid, exe=evt.exe, ts=evt.ts,
                detail=f"Container accessed Docker/containerd socket {evt.path!r}",
                indicator="docker_socket",
            ))

        return alerts


# ---------------------------------------------------------------------------
# APPROACH 2 — Stateful chain detector (RECOMMENDED)
# ---------------------------------------------------------------------------

class EscapeChainState(Enum):
    IDLE = 0
    MOUNT_OBSERVED = 1       # suspicious mount
    CGROUP_WRITE = 2         # cgroup release_agent written
    CHAIN_COMPLETE = 3       # full escape chain observed


@dataclass
class _ContainerChain:
    state: EscapeChainState = EscapeChainState.IDLE
    mount_ts: Optional[float] = None
    mount_path: str = ""
    write_ts: Optional[float] = None
    chain_window_s: float = 30.0


class ContainerEscapeDetector:
    """
    Full container escape detector with:
    1. Per-event indicator checks (delegates to NaiveEscapeDetector)
    2. Stateful cgroup escape chain detection (CVE-2022-0492 / runc)
    3. PID namespace change detection
    """

    def __init__(
        self,
        allowlist: Optional[Set[str]] = None,
        chain_window_s: float = 30.0,
    ) -> None:
        self._naive = NaiveEscapeDetector(allowlist=allowlist)
        self.allowlist = allowlist or set()
        self.chain_window = chain_window_s
        # container_id → chain state
        self._chains: Dict[str, _ContainerChain] = defaultdict(
            lambda: _ContainerChain(chain_window_s=chain_window_s))
        # container_id → set of ns_pids seen (for PID ns change)
        self._ns_pids: Dict[str, Set[int]] = defaultdict(set)
        self.malformed = 0

    def process(self, raw: dict) -> List[EscapeAlert]:
        evt = _parse_event(raw)
        if evt is None:
            self.malformed += 1
            return []

        alerts: List[EscapeAlert] = []

        # Per-event naive checks (only inside containers)
        alerts.extend(self._naive.process(raw))

        cid = evt.container_id
        if not cid:
            return alerts  # host events — skip chain tracking

        # --- PID namespace change detection ---
        if evt.ns_pid:
            ns_set = self._ns_pids[cid]
            if ns_set and evt.ns_pid not in ns_set and evt.ns_pid == 1:
                # ns_pid=1 from a process that is not the original PID 1 → namespace change
                alerts.append(EscapeAlert(
                    alert_type="pid_ns_change",
                    severity="critical",
                    container_id=cid, pid=evt.pid, exe=evt.exe, ts=evt.ts,
                    detail=f"PID namespace change detected (ns_pid=1 seen for secondary process)",
                    indicator="pid_ns_change",
                ))
            ns_set.add(evt.ns_pid)

        # --- Cgroup release_agent escape chain ---
        chain = self._chains[cid]
        if chain.state == EscapeChainState.IDLE:
            # Step 1: mount of cgroup (or /proc) inside container
            if evt.syscall == "mount" and any(
                    evt.path.startswith(p) for p in _CGROUP_PATHS):
                chain.state = EscapeChainState.MOUNT_OBSERVED
                chain.mount_ts = evt.ts
                chain.mount_path = evt.path
        elif chain.state == EscapeChainState.MOUNT_OBSERVED:
            # Step 2: write to release_agent
            if evt.path.endswith("release_agent") and evt.event_type == "file":
                # Check timeout
                if chain.mount_ts and evt.ts - chain.mount_ts <= chain.chain_window_s:
                    chain.state = EscapeChainState.CHAIN_COMPLETE
                    chain.write_ts = evt.ts
                    alerts.append(EscapeAlert(
                        alert_type="cgroup_release_agent_escape",
                        severity="critical",
                        container_id=cid, pid=evt.pid, exe=evt.exe, ts=evt.ts,
                        detail=(f"cgroup release_agent escape chain: "
                                f"mount {chain.mount_path!r} → write release_agent "
                                f"within {evt.ts - chain.mount_ts:.1f}s"),
                        indicator="cgroup_escape_chain",
                    ))
                else:
                    chain.state = EscapeChainState.IDLE
            elif chain.mount_ts and evt.ts - chain.mount_ts > chain.chain_window_s:
                chain.state = EscapeChainState.IDLE  # expired

        return alerts

    def active_containers(self) -> int:
        return len(self._chains)

def _ev(cid, exe, syscall="", path="", ns_pid=0, event_type="process",
        ts=1000.0, uid=0) -> dict:
    return {
        "pid": 1234, "ppid": 1, "container_id": cid,
        "exe": exe, "cmdline": f"{exe} -arg",
        "syscall": syscall, "path": path,
        "ts": ts, "uid": uid, "ns_pid": ns_pid,
        "event_type": event_type,
    }


def test_suspicious_exe():
    det = ContainerEscapeDetector()
    alerts = det.process(_ev("ctr-abc", "/usr/bin/nsenter"))
    assert any(a.alert_type == "suspicious_exe_in_container" for a in alerts)
    print("Suspicious exe: passed")


def test_dangerous_syscall():
    det = ContainerEscapeDetector()
    alerts = det.process(_ev("ctr-abc", "/bin/sh", syscall="mount",
                              path="/sys/fs/cgroup", event_type="syscall"))
    assert any(a.alert_type == "dangerous_syscall" for a in alerts)
    print("Dangerous syscall: passed")


def test_docker_socket_access():
    det = ContainerEscapeDetector()
    alerts = det.process(_ev("ctr-abc", "/usr/bin/curl",
                              path="/var/run/docker.sock", event_type="file"))
    assert any(a.alert_type == "docker_socket_access" for a in alerts)
    print("Docker socket access: passed")


def test_cgroup_escape_chain():
    det = ContainerEscapeDetector(chain_window_s=30)
    # Step 1: mount cgroup
    det.process(_ev("ctr-xyz", "/bin/sh", syscall="mount",
                     path="/sys/fs/cgroup/memory", ts=1000.0, event_type="syscall"))
    # Step 2: write release_agent
    alerts = det.process(_ev("ctr-xyz", "/bin/sh",
                               path="/sys/fs/cgroup/memory/release_agent",
                               event_type="file", ts=1010.0))
    assert any(a.alert_type == "cgroup_release_agent_escape" for a in alerts), \
        f"Expected cgroup escape; got {[a.alert_type for a in alerts]}"
    print("Cgroup escape chain: passed")


def test_cgroup_chain_timeout():
    det = ContainerEscapeDetector(chain_window_s=5)
    det.process(_ev("ctr-xyz", "/bin/sh", syscall="mount",
                     path="/sys/fs/cgroup/memory", ts=1000.0, event_type="syscall"))
    # Write comes 60s later — chain should expire
    alerts = det.process(_ev("ctr-xyz", "/bin/sh",
                               path="/sys/fs/cgroup/memory/release_agent",
                               event_type="file", ts=1060.0))
    assert not any(a.alert_type == "cgroup_release_agent_escape" for a in alerts)
    print("Cgroup chain timeout: passed")


def test_host_events_skipped():
    det = ContainerEscapeDetector()
    # container_id="" → host, not a container escape
    alerts = det.process(_ev("", "/usr/bin/nsenter", syscall="mount"))
    assert len(alerts) == 0
    print("Host events skipped: passed")


def test_malformed():
    det = ContainerEscapeDetector()
    r = det.process({"pid": 1})  # missing ts
    assert r == []
    assert det.malformed == 1
    print("Malformed: passed")

# SECURITY ANGLE

"""
THREAT MODEL
  Container escapes break the isolation boundary between workloads and the host.
  Once an attacker escapes to the host, they have:
  - Access to all other container filesystems via /var/lib/docker or /var/lib/containerd
  - Node credentials (kubeconfig, cloud metadata IMDS token)
  - Ability to tamper with the container runtime / kubelet

PROMINENT ESCAPES
  CVE-2019-5736 (runc overwrite): attacker overwrites the host runc binary
    via /proc/self/exe while runc is executing — requires CAP_SYS_PTRACE.
  CVE-2020-15257 (containerd shim): exploits the containerd shim API
    exposed via an abstract UNIX socket — mounted inside containers.
  CVE-2022-0492 (cgroup release_agent): writable cgroup v1 mount inside
    container → write a payload to release_agent → triggered on cgroup cleanup.
  Docker socket bind mount: if the Docker socket is mounted into a container,
    attacker runs `docker run -v /:/host alpine chroot /host`.

DEFENCES
  - Seccomp profiles: block dangerous syscalls (mount, unshare, ptrace).
  - AppArmor / SELinux: MAC policies prevent access to host paths.
  - Read-only filesystem: prevents writing to release_agent.
  - Drop ALL capabilities; add only what the workload needs.
  - Never mount Docker socket or host paths into containers.
  - Runtime security: Falco (eBPF rules) for real-time syscall monitoring.

DETECTION GAPS
  - Kernel exploits (0-days) may not trigger any of these indicators.
  - If attacker has CAP_SYS_PTRACE they can manipulate the monitoring process.
  - Rootkits can hide processes from /proc — monitor at the eBPF layer (not /proc).
"""
