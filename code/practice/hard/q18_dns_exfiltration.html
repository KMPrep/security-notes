"""
=============================================================================
Q18 [Hard] — DNS Data Exfiltration Detection
=============================================================================

PROBLEM
-------
Attackers encode stolen data in DNS query labels and slowly drip-query
a C2-controlled domain.  Detect DNS-based covert channel / exfiltration:

  1. High query rate to a single domain (burst indicator)
  2. Unusually long subdomains (encoded payload)
  3. High Shannon entropy in subdomain labels (Base32/Base64/hex encoding)
  4. Many unique subdomains under the same parent domain (enumeration)
  5. Query type abuse: TXT/NULL/ANY records used for data transfer
  6. NXDOMAIN flood (probing phase before registering payload domains)
  7. Low-volume "slow drip" exfiltration spread over many minutes

Input: stream of DNS query events with fields:
  src_ip, domain, qtype, ts, response_code, answer_count

CLARIFYING QUESTIONS
--------------------
1. Do we have full recursive query logs or only stub-resolver logs?
   → Full resolver logs.
2. Can we afford per-query per-domain state?
   → Yes; O(D) domains tracked, evict idle ones.
3. What encoding schemes do attackers use?
   → Base32 (RFC 4648), Base64url, hex.  All have high entropy.
4. What is a "suspicious" label length?
   → Label >40 chars; total FQDN >200 chars are strong signals.
5. Should we ignore well-known CDN/cloud domains?
   → Accept an allowlist; default allowlist includes common CDNs.

=============================================================================
APPROACH 1 — NAIVE (per-query checks, no state)
=============================================================================
  Compute entropy of subdomain; compare label length; check qtype.
  O(L) per query (L = FQDN length).
  Misses slow-drip and burst patterns.

=============================================================================
APPROACH 2 — STATEFUL PER-DOMAIN TRACKER + SLIDING WINDOWS (RECOMMENDED)
=============================================================================
  Per parent-domain state:
    - Sliding window query counter (burst)
    - Sliding window set of unique subdomain labels (enumeration)
    - Max entropy seen; running average label length
    - NXDOMAIN counter
  O(1) amortised per query.

=============================================================================
"""

from __future__ import annotations

import math
import re
from collections import defaultdict, deque
from dataclasses import dataclass, field
from datetime import datetime
from typing import Dict, FrozenSet, List, Optional, Set

# ---------------------------------------------------------------------------
# Well-known allowlist
# ---------------------------------------------------------------------------

_ALLOWLIST_TLDS: FrozenSet[str] = frozenset({
    "googleapis.com", "gstatic.com", "akamaihd.net", "cloudfront.net",
    "fastly.net", "windowsupdate.com", "apple.com", "microsoft.com",
    "amazon.com", "s3.amazonaws.com",
})

_SUSPICIOUS_QTYPES: FrozenSet[str] = frozenset({"TXT", "NULL", "ANY", "CNAME"})

_LONG_LABEL_THRESHOLD = 40
_ENTROPY_THRESHOLD = 3.5  # bits per character; random Base32/hex ≈ 4.5+
_BURST_THRESHOLD = 50     # queries per window
_BURST_WINDOW_S = 60.0
_UNIQUE_LABELS_THRESHOLD = 100
_UNIQUE_LABELS_WINDOW_S = 300.0
_NXDOMAIN_THRESHOLD = 20
_NXDOMAIN_WINDOW_S = 60.0


# ---------------------------------------------------------------------------
# Data Types
# ---------------------------------------------------------------------------

@dataclass
class DnsEvent:
    src_ip: str
    domain: str          # full FQDN
    qtype: str
    ts: float
    response_code: str   # NOERROR | NXDOMAIN | SERVFAIL
    answer_count: int = 0


@dataclass
class DnsAlert:
    alert_type: str
    severity: str
    src_ip: str
    domain: str
    parent_domain: str
    ts: float
    detail: str


def _parse_ts(ts) -> Optional[float]:
    if isinstance(ts, (int, float)):
        return float(ts)
    try:
        return datetime.fromisoformat(str(ts).replace("Z", "+00:00")).timestamp()
    except (ValueError, TypeError):
        return None


def _parse_event(raw: dict) -> Optional[DnsEvent]:
    ts = _parse_ts(raw.get("ts"))
    if ts is None:
        return None
    return DnsEvent(
        src_ip=raw.get("src_ip", ""),
        domain=raw.get("domain", "").lower().rstrip("."),
        qtype=raw.get("qtype", "A").upper(),
        ts=ts,
        response_code=raw.get("response_code", "NOERROR").upper(),
        answer_count=int(raw.get("answer_count", 0)),
    )


# ---------------------------------------------------------------------------
# Entropy calculation
# ---------------------------------------------------------------------------

def _shannon_entropy(s: str) -> float:
    """Shannon entropy (bits per character) for string s."""
    if not s:
        return 0.0
    freq: Dict[str, int] = {}
    for ch in s:
        freq[ch] = freq.get(ch, 0) + 1
    n = len(s)
    return -sum((c / n) * math.log2(c / n) for c in freq.values())


# ---------------------------------------------------------------------------
# Domain decomposition
# ---------------------------------------------------------------------------

_TLD_LIKE_RE = re.compile(r"[a-z]+$")

def _parent_domain(fqdn: str, depth: int = 2) -> str:
    """
    Return the registrable domain (last `depth` labels).
    E.g. "foo.bar.evil.com" → "evil.com"  (depth=2)
    """
    parts = fqdn.split(".")
    if len(parts) <= depth:
        return fqdn
    return ".".join(parts[-depth:])


def _subdomain(fqdn: str, depth: int = 2) -> str:
    """Return everything before the registrable domain."""
    parts = fqdn.split(".")
    if len(parts) <= depth:
        return ""
    return ".".join(parts[:-depth])


def _max_label_len(fqdn: str) -> int:
    return max((len(l) for l in fqdn.split(".") if l), default=0)


# ---------------------------------------------------------------------------
# APPROACH 1 — Naive stateless detector
# ---------------------------------------------------------------------------

class NaiveDnsExfilDetector:
    """
    Per-query stateless checks.  O(L) per query.
    """

    def __init__(
        self,
        allowlist: Optional[Set[str]] = None,
        entropy_threshold: float = _ENTROPY_THRESHOLD,
        label_length_threshold: int = _LONG_LABEL_THRESHOLD,
    ) -> None:
        self.allowlist = (allowlist or set()) | _ALLOWLIST_TLDS
        self.entropy_threshold = entropy_threshold
        self.label_threshold = label_length_threshold
        self.malformed = 0

    def process(self, raw: dict) -> List[DnsAlert]:
        evt = _parse_event(raw)
        if evt is None:
            self.malformed += 1
            return []
        parent = _parent_domain(evt.domain)
        if parent in self.allowlist:
            return []

        alerts = []
        subdomain = _subdomain(evt.domain)

        # Check entropy
        if subdomain:
            entropy = _shannon_entropy(subdomain.replace(".", ""))
            if entropy >= self.entropy_threshold:
                alerts.append(DnsAlert(
                    alert_type="high_entropy_subdomain",
                    severity="high",
                    src_ip=evt.src_ip,
                    domain=evt.domain,
                    parent_domain=parent,
                    ts=evt.ts,
                    detail=f"Subdomain entropy={entropy:.2f} (threshold={self.entropy_threshold})",
                ))

        # Check label length
        max_len = _max_label_len(evt.domain)
        if max_len >= self.label_threshold:
            alerts.append(DnsAlert(
                alert_type="long_dns_label",
                severity="medium",
                src_ip=evt.src_ip,
                domain=evt.domain,
                parent_domain=parent,
                ts=evt.ts,
                detail=f"Max label length={max_len} chars",
            ))

        # Suspicious qtype
        if evt.qtype in _SUSPICIOUS_QTYPES:
            alerts.append(DnsAlert(
                alert_type="suspicious_qtype",
                severity="medium",
                src_ip=evt.src_ip,
                domain=evt.domain,
                parent_domain=parent,
                ts=evt.ts,
                detail=f"Suspicious query type {evt.qtype!r}",
            ))

        return alerts

# APPROACH 2 — Full stateful detector
# ---------------------------------------------------------------------------

@dataclass
class _DomainState:
    # burst window
    query_times: deque = field(default_factory=deque)
    # unique labels window: deque of (ts, label)
    label_times: deque = field(default_factory=deque)
    label_set: Set[str] = field(default_factory=set)
    # NXDOMAIN burst
    nxdomain_times: deque = field(default_factory=deque)
    # flags
    burst_alerted: bool = False
    enum_alerted: bool = False
    nxdomain_alerted: bool = False
    # stats
    total_queries: int = 0
    max_entropy: float = 0.0


class DnsExfilDetector:
    """
    Stateful DNS exfiltration detector with per-parent-domain tracking.
    """

    def __init__(
        self,
        allowlist: Optional[Set[str]] = None,
        burst_threshold: int = _BURST_THRESHOLD,
        burst_window_s: float = _BURST_WINDOW_S,
        unique_labels_threshold: int = _UNIQUE_LABELS_THRESHOLD,
        unique_labels_window_s: float = _UNIQUE_LABELS_WINDOW_S,
        nxdomain_threshold: int = _NXDOMAIN_THRESHOLD,
        nxdomain_window_s: float = _NXDOMAIN_WINDOW_S,
        entropy_threshold: float = _ENTROPY_THRESHOLD,
        label_length_threshold: int = _LONG_LABEL_THRESHOLD,
    ) -> None:
        self.allowlist = (allowlist or set()) | _ALLOWLIST_TLDS
        self.burst_threshold = burst_threshold
        self.burst_window = burst_window_s
        self.unique_threshold = unique_labels_threshold
        self.unique_window = unique_labels_window_s
        self.nxdomain_threshold = nxdomain_threshold
        self.nxdomain_window = nxdomain_window_s
        self._naive = NaiveDnsExfilDetector(
            allowlist=allowlist,
            entropy_threshold=entropy_threshold,
            label_length_threshold=label_length_threshold,
        )
        self._state: Dict[str, _DomainState] = defaultdict(_DomainState)
        self.malformed = 0

    def process(self, raw: dict) -> List[DnsAlert]:
        evt = _parse_event(raw)
        if evt is None:
            self.malformed += 1
            return []
        parent = _parent_domain(evt.domain)
        if parent in self.allowlist:
            return []

        alerts: List[DnsAlert] = []

        # Per-event naive checks
        alerts.extend(self._naive.process(raw))

        st = self._state[parent]
        st.total_queries += 1
        now = evt.ts

        # --- Burst detection ---
        cutoff = now - self.burst_window
        while st.query_times and st.query_times[0] < cutoff:
            st.query_times.popleft()
        st.query_times.append(now)
        if len(st.query_times) > self.burst_threshold:
            if not st.burst_alerted:
                st.burst_alerted = True
                alerts.append(DnsAlert(
                    alert_type="dns_query_burst",
                    severity="high",
                    src_ip=evt.src_ip,
                    domain=evt.domain,
                    parent_domain=parent,
                    ts=now,
                    detail=(f"{len(st.query_times)} queries to {parent!r} "
                            f"in {self.burst_window}s"),
                ))
        else:
            st.burst_alerted = False

        # --- Unique subdomain enumeration ---
        subdomain = _subdomain(evt.domain)
        if subdomain:
            cutoff_u = now - self.unique_window
            # Evict old labels
            while st.label_times and st.label_times[0][0] < cutoff_u:
                _, old_label = st.label_times.popleft()
                # Rebuild set (simple approach, bounded by window size)
                st.label_set = {l for _, l in st.label_times}
            st.label_times.append((now, subdomain))
            st.label_set.add(subdomain)
            if len(st.label_set) > self.unique_threshold:
                if not st.enum_alerted:
                    st.enum_alerted = True
                    alerts.append(DnsAlert(
                        alert_type="subdomain_enumeration",
                        severity="high",
                        src_ip=evt.src_ip,
                        domain=evt.domain,
                        parent_domain=parent,
                        ts=now,
                        detail=(f"{len(st.label_set)} unique subdomains under {parent!r} "
                                f"in {self.unique_window}s"),
                    ))

        # --- NXDOMAIN flood ---
        if evt.response_code == "NXDOMAIN":
            cutoff_nx = now - self.nxdomain_window
            while st.nxdomain_times and st.nxdomain_times[0] < cutoff_nx:
                st.nxdomain_times.popleft()
            st.nxdomain_times.append(now)
            if len(st.nxdomain_times) > self.nxdomain_threshold:
                if not st.nxdomain_alerted:
                    st.nxdomain_alerted = True
                    alerts.append(DnsAlert(
                        alert_type="nxdomain_flood",
                        severity="medium",
                        src_ip=evt.src_ip,
                        domain=evt.domain,
                        parent_domain=parent,
                        ts=now,
                        detail=(f"{len(st.nxdomain_times)} NXDOMAINs for {parent!r} "
                                f"in {self.nxdomain_window}s"),
                    ))

        return alerts

    def evict_idle(self, before_ts: float) -> int:
        """Remove per-domain state that hasn't seen queries since before_ts."""
        stale = [p for p, st in self._state.items()
                 if st.query_times and st.query_times[-1] < before_ts]
        for p in stale:
            del self._state[p]
        return len(stale)

def _raw_dns(src, domain, qtype="A", ts=1000.0, rc="NOERROR") -> dict:
    return {"src_ip": src, "domain": domain, "qtype": qtype,
            "ts": ts, "response_code": rc, "answer_count": 1}


def test_high_entropy_subdomain():
    det = DnsExfilDetector(entropy_threshold=3.5)
    # "aabbccddee...1234" — high entropy Base32-like
    label = "MFRA" * 12  # 48 chars of high-entropy
    alerts = det.process(_raw_dns("10.0.0.1", f"{label.lower()}.evil.com"))
    assert any(a.alert_type == "high_entropy_subdomain" for a in alerts), \
        f"Expected high_entropy; got {[a.alert_type for a in alerts]}"
    print("High entropy subdomain: passed")


def test_long_label():
    det = DnsExfilDetector(label_length_threshold=40)
    label = "x" * 50
    alerts = det.process(_raw_dns("10.0.0.1", f"{label}.evil.com"))
    assert any(a.alert_type == "long_dns_label" for a in alerts)
    print("Long label: passed")


def test_burst():
    det = DnsExfilDetector(burst_threshold=5, burst_window_s=60)
    all_alerts = []
    for i in range(7):
        all_alerts += det.process(_raw_dns("10.0.0.1", f"sub{i}.evil.com", ts=float(1000 + i)))
    assert any(a.alert_type == "dns_query_burst" for a in all_alerts)
    print("Burst: passed")


def test_nxdomain_flood():
    det = DnsExfilDetector(nxdomain_threshold=3, nxdomain_window_s=60)
    all_alerts = []
    for i in range(5):
        all_alerts += det.process(_raw_dns("10.0.0.1", f"rand{i}.evil.com",
                                            ts=float(1000 + i), rc="NXDOMAIN"))
    assert any(a.alert_type == "nxdomain_flood" for a in all_alerts)
    print("NXDOMAIN flood: passed")


def test_allowlist():
    det = DnsExfilDetector()
    label = "x" * 50
    alerts = det.process(_raw_dns("10.0.0.1", f"{label}.googleapis.com"))
    assert len(alerts) == 0
    print("Allowlist: passed")


def test_suspicious_qtype():
    det = DnsExfilDetector()
    alerts = det.process(_raw_dns("10.0.0.1", "evil.com", qtype="TXT"))
    assert any(a.alert_type == "suspicious_qtype" for a in alerts)
    print("Suspicious qtype: passed")


def test_subdomain_enumeration():
    det = DnsExfilDetector(unique_labels_threshold=3, unique_labels_window_s=120)
    all_alerts = []
    for i in range(5):
        all_alerts += det.process(_raw_dns("10.0.0.1", f"label{i}.evil.com", ts=float(1000 + i)))
    assert any(a.alert_type == "subdomain_enumeration" for a in all_alerts)
    print("Subdomain enumeration: passed")


def test_malformed():
    det = DnsExfilDetector()
    r = det.process({"src_ip": "x"})  # missing ts
    assert r == []
    assert det.malformed == 1
    print("Malformed: passed")

"""
THREAT MODEL
  DNS tunneling / DNS covert channel lets attackers exfiltrate data through
  recursive DNS infrastructure that is usually unrestricted by perimeter firewalls.

TECHNIQUES
  iodine, dnscat2, dns2tcp: encode data as hex/Base32 in A/AAAA subdomains.
  TXT record abuse: C2 replies with commands in TXT records.
  CNAME chains: multiple redirections to delay detection.
  Slow drip: one query every 30s → low volume but persistent.

INDICATORS
  - Label entropy > 3.5 bits/char  (random Base32 ≈ 4.5)
  - Max label length > 40 chars
  - > 50 unique subdomains in 5 minutes
  - TXT / NULL qtype from endpoints
  - High NXDOMAIN rate to a single eTLD

HARDENING
  - DNS RPZ (Response Policy Zones): block known C2 domains.
  - Internal recursive resolver only: block direct external DNS on 53/tcp+udp.
  - DoH/DoT inspection: decrypt and inspect DNS-over-HTTPS if used.
  - Query rate limiting per source IP at the resolver.
  - Monitor EDR telemetry: processes making unusual DNS calls (curl, ncat, iodine).

FALSE POSITIVES
  - Legitimate CDN domains with long labels (mitigated by allowlist).
  - DKIM/SPF TXT records query.
  - Dynamic DNS update flood (TSIG).
"""
