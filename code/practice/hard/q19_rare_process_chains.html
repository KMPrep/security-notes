"""
=============================================================================
Q19 [Hard] — Rare Parent-Child Process Chain Detection
=============================================================================

PROBLEM
-------
Attackers abuse legitimate binaries by spawning them from unusual parents.
Classic examples:
  - Word.exe → cmd.exe → powershell.exe (macro execution)
  - nginx → /bin/sh (web shell)
  - sshd → python (interactive shell from web server)
  - cron → wget (download-and-execute)
  - explorer.exe → wscript.exe → powershell.exe (phishing chain)

Given a stream of process-start events, detect parent-child relationships
that have not been seen before (or are below a frequency threshold) and
that match known-suspicious patterns.

Approach requires:
  a) Counting historical (parent, child) pair frequencies.
  b) Flagging pairs below a rarity threshold.
  c) Applying high-confidence pattern rules (unconditionally suspicious).
  d) Multi-hop chain detection (depth-first ancestor traversal).

Input event fields: pid, ppid, exe, parent_exe, cmdline, host, ts, user

CLARIFYING QUESTIONS
--------------------
1. Do we have access to a training corpus (normal process pairs)?
   → Yes; feed historical events first; then flag new pairs below threshold.
2. Are exe paths normalised (lowercase, no drive letter noise)?
   → Normalise to basename in the detector.
3. What chain depth to check?
   → Configurable; default 3 hops (grandparent → parent → child).
4. Should we track per-host or globally?
   → Global model is fine; per-host reduces FP for specialised servers.
5. Rarity threshold: absolute count or frequency?
   → Count; pairs seen < N times overall are flagged as rare.

=============================================================================
APPROACH 1 — NAIVE (hardcoded rule table)
=============================================================================
  Dict of {parent_exe: Set[suspicious_children]} patterns.
  O(1) per event.  Does not learn; cannot detect novel pairs.

=============================================================================
APPROACH 2 — FREQUENCY-BASED RARITY MODEL (RECOMMENDED)
=============================================================================
  - defaultdict(int) of (parent_basename, child_basename) → count
  - Pairs below `rarity_threshold` are flagged
  - Chain tracking: store pid→exe, pid→ppid for lineage walk
  - Hard rules overlay (unconditionally suspicious, e.g. doc → shell)
  O(1) per event; O(D) memory per host for pid→ppid map.

=============================================================================
"""

from __future__ import annotations

import os
from collections import defaultdict, deque
from dataclasses import dataclass, field
from datetime import datetime
from typing import Dict, FrozenSet, List, Optional, Set, Tuple

# ---------------------------------------------------------------------------
# Hard-rule suspicious pairs (parent_base → set of child_bases)
# ---------------------------------------------------------------------------

# These combos are ALWAYS suspicious regardless of frequency
_SUSPICIOUS_CHAINS: Dict[str, FrozenSet[str]] = {
    "winword.exe":     frozenset({"cmd.exe", "powershell.exe", "wscript.exe", "cscript.exe", "mshta.exe", "regsvr32.exe"}),
    "excel.exe":       frozenset({"cmd.exe", "powershell.exe", "wscript.exe", "cscript.exe"}),
    "outlook.exe":     frozenset({"cmd.exe", "powershell.exe", "wscript.exe"}),
    "nginx":           frozenset({"sh", "bash", "dash", "python", "python3", "perl", "ruby"}),
    "apache2":         frozenset({"sh", "bash", "python", "perl"}),
    "httpd":           frozenset({"sh", "bash", "python", "perl"}),
    "tomcat":          frozenset({"sh", "bash", "python", "perl", "java"}),
    "sshd":            frozenset({"python", "python3", "perl", "ruby", "nc", "ncat", "nmap"}),
    "cron":            frozenset({"wget", "curl", "nc", "ncat", "python", "bash"}),
    "explorer.exe":    frozenset({"wscript.exe", "cscript.exe", "mshta.exe"}),
    "lsass.exe":       frozenset({"cmd.exe", "powershell.exe"}),  # credential dumping artefact
    "services.exe":    frozenset({"cmd.exe", "powershell.exe", "net.exe"}),
}

# Parents that are themselves unusual launchers
_NEVER_SHOULD_EXEC_SHELL: FrozenSet[str] = frozenset({
    "mysqld", "postgres", "redis-server", "mongod",
    "java",  # should not exec raw shells
})


# ---------------------------------------------------------------------------
# Data Types
# ---------------------------------------------------------------------------

@dataclass
class ProcessEvent:
    pid: int
    ppid: int
    exe: str
    parent_exe: str
    cmdline: str
    host: str
    ts: float
    user: str


@dataclass
class ChainAlert:
    alert_type: str
    severity: str
    host: str
    pid: int
    exe: str
    parent_exe: str
    ts: float
    detail: str
    chain: List[str] = field(default_factory=list)  # ancestor chain for context


def _parse_ts(ts) -> Optional[float]:
    if isinstance(ts, (int, float)):
        return float(ts)
    try:
        return datetime.fromisoformat(str(ts).replace("Z", "+00:00")).timestamp()
    except (ValueError, TypeError):
        return None


def _parse_event(raw: dict) -> Optional[ProcessEvent]:
    ts = _parse_ts(raw.get("ts"))
    if ts is None:
        return None
    return ProcessEvent(
        pid=int(raw.get("pid", 0)),
        ppid=int(raw.get("ppid", 0)),
        exe=raw.get("exe", ""),
        parent_exe=raw.get("parent_exe", ""),
        cmdline=raw.get("cmdline", ""),
        host=raw.get("host", "localhost"),
        ts=ts,
        user=raw.get("user", ""),
    )


def _base(exe: str) -> str:
    return os.path.basename(exe).lower()


# ---------------------------------------------------------------------------
# APPROACH 1 — Naive hard-rule detector
# ---------------------------------------------------------------------------

class NaiveProcessChainDetector:
    """
    Checks parent→child against a static suspicious table.
    O(1) per event.
    """

    def __init__(self, allowlist: Optional[Set[Tuple[str, str]]] = None) -> None:
        self.allowlist = allowlist or set()
        self.malformed = 0

    def process(self, raw: dict) -> List[ChainAlert]:
        evt = _parse_event(raw)
        if evt is None:
            self.malformed += 1
            return []
        p_base = _base(evt.parent_exe)
        c_base = _base(evt.exe)
        if (p_base, c_base) in self.allowlist:
            return []

        alerts = []
        if p_base in _SUSPICIOUS_CHAINS and c_base in _SUSPICIOUS_CHAINS[p_base]:
            alerts.append(ChainAlert(
                alert_type="suspicious_parent_child",
                severity="high",
                host=evt.host,
                pid=evt.pid, exe=evt.exe, parent_exe=evt.parent_exe,
                ts=evt.ts,
                detail=f"Suspicious: {p_base!r} spawned {c_base!r}",
            ))
        if p_base in _NEVER_SHOULD_EXEC_SHELL and c_base in {"sh", "bash", "dash", "zsh",
                                                               "cmd.exe", "powershell.exe"}:
            alerts.append(ChainAlert(
                alert_type="db_server_spawned_shell",
                severity="critical",
                host=evt.host,
                pid=evt.pid, exe=evt.exe, parent_exe=evt.parent_exe,
                ts=evt.ts,
                detail=f"DB/service {p_base!r} spawned shell {c_base!r}",
            ))
        return alerts


# ---------------------------------------------------------------------------
# APPROACH 2 — Frequency-based rarity model
# ---------------------------------------------------------------------------

@dataclass
class _ProcNode:
    exe: str
    ppid: int
    ts: float


class ProcessChainDetector:
    """
    Full stateful detector.
    - Learns (parent, child) pair frequencies.
    - Flags rare pairs below rarity_threshold.
    - Performs multi-hop chain walk.
    - Overlays hard rules.
    """

    def __init__(
        self,
        rarity_threshold: int = 5,
        max_chain_depth: int = 3,
        allowlist: Optional[Set[Tuple[str, str]]] = None,
        pid_ttl_s: float = 3600.0,
    ) -> None:
        self.rarity_threshold = rarity_threshold
        self.max_chain_depth = max_chain_depth
        self.allowlist = allowlist or set()
        self.pid_ttl = pid_ttl_s
        # (parent_base, child_base) → count
        self._pair_freq: Dict[Tuple[str, str], int] = defaultdict(int)
        # host → (pid → _ProcNode)
        self._proc_table: Dict[str, Dict[int, _ProcNode]] = defaultdict(dict)
        self._naive = NaiveProcessChainDetector(allowlist=allowlist)
        self.malformed = 0

    # -----------------------------------------------------------------------
    # Internal helpers
    # -----------------------------------------------------------------------

    def _get_chain(self, host: str, ppid: int, depth: int) -> List[str]:
        """Walk ancestor tree up to `depth` hops; return list of exe basenames."""
        chain = []
        current_ppid = ppid
        seen_pids: Set[int] = set()
        for _ in range(depth):
            if current_ppid in seen_pids or current_ppid == 0:
                break
            seen_pids.add(current_ppid)
            node = self._proc_table[host].get(current_ppid)
            if node is None:
                break
            chain.append(_base(node.exe))
            current_ppid = node.ppid
        return chain

    def _evict_stale_pids(self, host: str, now: float) -> None:
        """Remove PID entries older than pid_ttl (approximate; evict on access)."""
        table = self._proc_table[host]
        stale = [pid for pid, node in table.items()
                 if now - node.ts > self.pid_ttl]
        for pid in stale:
            del table[pid]

    # -----------------------------------------------------------------------

    def feed_baseline(self, raw: dict) -> None:
        """Ingest historical events to build frequency model (no alerting)."""
        evt = _parse_event(raw)
        if evt is None:
            return
        p_base = _base(evt.parent_exe)
        c_base = _base(evt.exe)
        self._pair_freq[(p_base, c_base)] += 1
        self._proc_table[evt.host][evt.pid] = _ProcNode(evt.exe, evt.ppid, evt.ts)

    def process(self, raw: dict) -> List[ChainAlert]:
        evt = _parse_event(raw)
        if evt is None:
            self.malformed += 1
            return []

        p_base = _base(evt.parent_exe)
        c_base = _base(evt.exe)

        # Store in process table for lineage
        self._proc_table[evt.host][evt.pid] = _ProcNode(evt.exe, evt.ppid, evt.ts)

        # Evict stale PIDs periodically (cheap: only when a new event arrives)
        if evt.pid % 50 == 0:
            self._evict_stale_pids(evt.host, evt.ts)

        alerts: List[ChainAlert] = []

        # Allowlist check
        if (p_base, c_base) not in self.allowlist:
            # Hard rule check first
            alerts.extend(self._naive.process(raw))

            # Frequency-based rarity
            pair = (p_base, c_base)
            count = self._pair_freq.get(pair, 0)
            self._pair_freq[pair] = count + 1

            if count < self.rarity_threshold and p_base and c_base:
                ancestor_chain = self._get_chain(evt.host, evt.ppid, self.max_chain_depth)
                alerts.append(ChainAlert(
                    alert_type="rare_parent_child",
                    severity="medium",
                    host=evt.host,
                    pid=evt.pid, exe=evt.exe, parent_exe=evt.parent_exe,
                    ts=evt.ts,
                    detail=(f"Rare pair: {p_base!r} → {c_base!r} "
                            f"(seen {count} times; threshold {self.rarity_threshold})"),
                    chain=ancestor_chain,
                ))

        return alerts

def _ev(exe, parent_exe, pid=100, ppid=50, host="host1", ts=1000.0, user="root") -> dict:
    return {"pid": pid, "ppid": ppid, "exe": exe, "parent_exe": parent_exe,
            "cmdline": f"{exe} -x", "host": host, "ts": ts, "user": user}


def test_hard_rule_office_macro():
    det = NaiveProcessChainDetector()
    alerts = det.process(_ev("/c/Program Files/winword.exe", "/windows/system32/cmd.exe"))
    # Note: exe is the child, parent_exe is the parent
    # Actually need child=cmd, parent=winword
    det2 = NaiveProcessChainDetector()
    alerts2 = det2.process(_ev(
        "/windows/system32/cmd.exe",
        "/c/Program Files/Microsoft Office/winword.exe"
    ))
    assert any(a.alert_type == "suspicious_parent_child" for a in alerts2), \
        f"Expected alert; got {[a.alert_type for a in alerts2]}"
    print("Office macro hard rule: passed")


def test_db_shell_spawn():
    det = NaiveProcessChainDetector()
    alerts = det.process(_ev("/bin/bash", "/usr/sbin/mysqld"))
    assert any(a.alert_type == "db_server_spawned_shell" for a in alerts)
    print("DB shell spawn: passed")


def test_webserver_shell():
    det = ProcessChainDetector(rarity_threshold=5)
    alerts = det.process(_ev("/bin/sh", "/usr/sbin/nginx"))
    assert any(a.alert_type == "suspicious_parent_child" for a in alerts)
    print("Webserver shell: passed")


def test_rare_pair_flagged():
    det = ProcessChainDetector(rarity_threshold=3)
    # Feed 2 baseline events
    for i in range(2):
        det.feed_baseline(_ev("/usr/bin/python3", "/usr/bin/cron",
                               pid=100 + i, ts=float(900 + i)))
    # Third event should be flagged as rare (count was 2, below threshold 3)
    alerts = det.process(_ev("/usr/bin/python3", "/usr/bin/cron", pid=200, ts=1000.0))
    # rare_parent_child fires when count < threshold (0, 1, 2 → rare; 3+ → not)
    # After 2 baseline feeds, count=2 < 3, so should still flag
    assert any(a.alert_type == "rare_parent_child" for a in alerts), \
        f"Expected rare_parent_child; got {[a.alert_type for a in alerts]}"
    print("Rare pair flagged: passed")


def test_common_pair_not_flagged():
    det = ProcessChainDetector(rarity_threshold=3)
    # Feed 5 baseline events (above threshold)
    for i in range(5):
        det.feed_baseline(_ev("/bin/ls", "/bin/bash", pid=100 + i, ts=float(900 + i)))
    # 6th event - pair has count=5 → not rare
    alerts = det.process(_ev("/bin/ls", "/bin/bash", pid=200, ts=1000.0))
    rare = [a for a in alerts if a.alert_type == "rare_parent_child"]
    assert len(rare) == 0
    print("Common pair not flagged: passed")


def test_chain_walk():
    det = ProcessChainDetector(rarity_threshold=100, max_chain_depth=3)
    # Build chain: pid1(bash) → pid2(python) → pid3(nc)
    det.feed_baseline(_ev("/bin/bash", "/usr/sbin/sshd", pid=1, ppid=0, ts=900.0))
    det.feed_baseline(_ev("/usr/bin/python3", "/bin/bash", pid=2, ppid=1, ts=901.0))
    alerts = det.process(_ev("/bin/nc", "/usr/bin/python3", pid=3, ppid=2, ts=1000.0))
    # Find the rare_parent_child alert and check chain
    chain_alerts = [a for a in alerts if a.alert_type == "rare_parent_child"]
    assert chain_alerts, f"Expected rare_parent_child; got {[a.alert_type for a in alerts]}"
    assert "python3" in chain_alerts[0].chain or "bash" in chain_alerts[0].chain
    print("Chain walk: passed")


def test_malformed():
    det = ProcessChainDetector()
    r = det.process({"pid": 1})  # missing ts
    assert r == []
    assert det.malformed == 1
    print("Malformed: passed")

"""
THREAT MODEL
  Living-off-the-land attacks (LotL) use legitimate binaries (cmd, powershell,
  wscript, cscript, mshta, regsvr32) to avoid signature-based AV.
  The key signal is the parent process: a web server should never spawn bash.

COMMON ATTACK CHAINS
  Macro execution:    winword → wscript → powershell → (download C2 agent)
  Web shell:          nginx/apache → /bin/sh  → curl → malware
  Cron download:      cron → wget → bash script
  LOLBins Windows:    msiexec → cmd → regsvr32 (register malicious COM DLL)
  Container escape:   containerd-shim → nsenter → chroot

HARDENING / COMPLEMENTARY DEFENCES
  - Process execution whitelisting (AppLocker / SRP / SELinux domain transitions)
  - EDR with parent-child baseline (CrowdStrike, Defender for Endpoint)
  - Seccomp / sandbox: block exec syscall from web server worker processes
  - Immutable container image: container rootfs mounted read-only

FALSE POSITIVES
  - Admin scripts that legitimately run shells from cron or services
  - Development environments (IDE spawning arbitrary build tools)
  - Mitigation: per-host allowlists; environment tags (dev vs prod)
"""
