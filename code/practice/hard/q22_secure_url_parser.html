"""
=============================================================================
Q22 [Hard] — Secure URL Parser and SSRF Prevention
=============================================================================

PROBLEM
-------
Build a secure URL parser that:
  1. Parses a URL into scheme, host, port, path, query, fragment.
  2. Normalises the host (strip trailing dots, lowercase, punycode-decode).
  3. Resolves numeric / encoded IP representations.
  4. Blocks SSRF-dangerous targets:
       a. Private IPv4 ranges (10/8, 172.16/12, 192.168/16, 127/8, 169.254/16)
       b. Link-local IPv6 (::1, fe80::/10)
       c. Cloud metadata endpoints (169.254.169.254, fd00:ec2::254)
       d. Loopback via alternate forms: 0177.0.0.1 (octal), 0x7f.0.0.1 (hex),
          2130706433 (decimal), 127.1 (partial notation)
  5. Reject schemes other than http/https (or configurable allowlist).
  6. Detect open-redirect indicators (URL in path/query that redirects out).

CLARIFYING QUESTIONS
--------------------
1. Do we do DNS resolution to catch SSRF via hostname?
   → Not in this function (network call); caller handles DNS; we validate
     IPs that are already resolved or embedded in the URL directly.
2. What to do with IPv6 in URL?
   → Validate bracket notation [::1]; expand compressed IPv6 for matching.
3. Allowlist of safe external hosts?
   → Accept a set of allowed host suffixes.
4. Should we handle redirects?
   → Detect suspicious URL-in-query patterns; actual redirect following is
     caller responsibility.
5. What about credentials in URL (user:pass@host)?
   → Strip; return warning in parse errors.

=============================================================================
APPROACH 1 — NAIVE (urllib.parse + simple blocklist)
=============================================================================
  Use Python urllib.parse.urlparse.
  Check hostname against a hardcoded IP blocklist.
  Misses octal/hex/decimal IP forms.
  O(1) per URL.

=============================================================================
APPROACH 2 — FULL NORMALISER + CANONICAL IP RESOLVER (RECOMMENDED)
=============================================================================
  1. Parse with urllib.parse.
  2. Decode percent-encoding of host.
  3. Resolve numeric IP forms (octal, hex, decimal, mixed).
  4. Validate against CIDR blocklist using ipaddress module.
  5. Detect open-redirect patterns.
  O(1) per URL.

=============================================================================
"""

from __future__ import annotations

import ipaddress
import re
import urllib.parse
from dataclasses import dataclass, field
from typing import FrozenSet, List, Optional, Set, Tuple

# ---------------------------------------------------------------------------
# Blocked CIDR ranges for SSRF
# ---------------------------------------------------------------------------

_BLOCKED_IPV4_NETWORKS = [
    ipaddress.IPv4Network("127.0.0.0/8"),       # loopback
    ipaddress.IPv4Network("10.0.0.0/8"),         # RFC 1918
    ipaddress.IPv4Network("172.16.0.0/12"),      # RFC 1918
    ipaddress.IPv4Network("192.168.0.0/16"),     # RFC 1918
    ipaddress.IPv4Network("169.254.0.0/16"),     # link-local / cloud metadata
    ipaddress.IPv4Network("0.0.0.0/8"),          # "this" network
    ipaddress.IPv4Network("100.64.0.0/10"),      # shared address space (RFC 6598)
    ipaddress.IPv4Network("192.0.0.0/24"),       # IETF protocol assignments
    ipaddress.IPv4Network("198.18.0.0/15"),      # benchmark testing
    ipaddress.IPv4Network("198.51.100.0/24"),    # documentation
    ipaddress.IPv4Network("203.0.113.0/24"),     # documentation
    ipaddress.IPv4Network("240.0.0.0/4"),        # reserved
    ipaddress.IPv4Network("255.255.255.255/32"), # broadcast
]

_BLOCKED_IPV6_NETWORKS = [
    ipaddress.IPv6Network("::1/128"),            # loopback
    ipaddress.IPv6Network("fc00::/7"),           # unique local
    ipaddress.IPv6Network("fe80::/10"),          # link-local
    ipaddress.IPv6Network("::ffff:0:0/96"),      # IPv4-mapped
    ipaddress.IPv6Network("::/128"),             # unspecified
    ipaddress.IPv6Network("fd00:ec2::/32"),      # AWS metadata IPv6
]

_ALLOWED_SCHEMES: FrozenSet[str] = frozenset({"http", "https"})

# Detect URL-in-query that could be an open redirect
_URL_IN_VALUE_RE = re.compile(r'https?://', re.IGNORECASE)

# Detect numeric IP representations
_PURE_DECIMAL_RE = re.compile(r'^\d+$')
_OCTAL_DOTTED_RE = re.compile(r'^0\d+(\.\d+)*$')
_HEX_DOTTED_RE = re.compile(r'^0x[0-9a-fA-F]+(\.\d+)*$')
_HEX_PURE_RE = re.compile(r'^0x[0-9a-fA-F]+$')


# ---------------------------------------------------------------------------
# Data Types
# ---------------------------------------------------------------------------

@dataclass
class ParsedURL:
    original: str
    scheme: str = ""
    host: str = ""          # normalised hostname (lowercase, no trailing dot)
    port: Optional[int] = None
    path: str = ""
    query: str = ""
    fragment: str = ""
    resolved_ip: Optional[str] = None   # canonical IP string if host is an IP
    is_safe: bool = True
    warnings: List[str] = field(default_factory=list)
    errors: List[str] = field(default_factory=list)


# ---------------------------------------------------------------------------
# IP Normalisation Helpers
# ---------------------------------------------------------------------------

def _try_parse_ipv4(raw: str) -> Optional[ipaddress.IPv4Address]:
    """
    Try to parse raw string as IPv4, handling:
      - Standard dotted-decimal
      - Pure decimal (integer): 2130706433 → 127.0.0.1
      - Hex: 0x7f000001 → 127.0.0.1
      - Octal parts: 0177.0.0.1 → 127.0.0.1
      - Mixed partial notation: 127.1 → 127.0.0.1
    Returns IPv4Address or None.
    """
    raw = raw.strip()
    # Standard dotted first
    try:
        return ipaddress.IPv4Address(raw)
    except ValueError:
        pass

    # Pure decimal (integer form)
    if _PURE_DECIMAL_RE.match(raw):
        try:
            return ipaddress.IPv4Address(int(raw))
        except ValueError:
            pass

    # Hex (0x...)
    if _HEX_PURE_RE.match(raw):
        try:
            return ipaddress.IPv4Address(int(raw, 16))
        except ValueError:
            pass

    # Dotted forms — parse each octet allowing octal (0...) and hex (0x...)
    parts = raw.split(".")
    if len(parts) in (1, 2, 3, 4):
        try:
            values = []
            for part in parts:
                if part.startswith("0x") or part.startswith("0X"):
                    values.append(int(part, 16))
                elif part.startswith("0") and len(part) > 1:
                    values.append(int(part, 8))
                else:
                    values.append(int(part))
            # Handle partial notation (fewer than 4 parts)
            if len(values) == 4:
                ip_int = (values[0] << 24) | (values[1] << 16) | (values[2] << 8) | values[3]
            elif len(values) == 3:
                ip_int = (values[0] << 24) | (values[1] << 16) | values[2]
            elif len(values) == 2:
                ip_int = (values[0] << 24) | values[1]
            else:
                ip_int = values[0]
            return ipaddress.IPv4Address(ip_int)
        except (ValueError, OverflowError):
            pass
    return None


def _try_parse_ipv6(raw: str) -> Optional[ipaddress.IPv6Address]:
    """Parse IPv6 host from bracket notation or bare."""
    raw = raw.strip("[]")
    # Remove zone ID
    raw = raw.split("%")[0]
    try:
        return ipaddress.IPv6Address(raw)
    except ValueError:
        return None


def _is_blocked_ip(ip: ipaddress.IPv4Address | ipaddress.IPv6Address) -> bool:
    if isinstance(ip, ipaddress.IPv4Address):
        return any(ip in net for net in _BLOCKED_IPV4_NETWORKS)
    else:
        return any(ip in net for net in _BLOCKED_IPV6_NETWORKS)


# ---------------------------------------------------------------------------
# APPROACH 1 — Naive parser
# ---------------------------------------------------------------------------

class NaiveURLParser:
    """
    Uses urllib.parse; basic private-IP blocklist.
    Misses obfuscated IP forms.
    """

    def parse(self, url: str, allowed_schemes: Optional[Set[str]] = None) -> ParsedURL:
        schemes = allowed_schemes or _ALLOWED_SCHEMES
        result = ParsedURL(original=url)
        try:
            parsed = urllib.parse.urlparse(url)
            result.scheme = parsed.scheme.lower()
            result.host = (parsed.hostname or "").lower().rstrip(".")
            result.port = parsed.port
            result.path = parsed.path
            result.query = parsed.query
            result.fragment = parsed.fragment

            if result.scheme not in schemes:
                result.is_safe = False
                result.errors.append(f"Scheme {result.scheme!r} not allowed")

            # Basic IP check
            try:
                ip = ipaddress.ip_address(result.host)
                result.resolved_ip = str(ip)
                if _is_blocked_ip(ip):
                    result.is_safe = False
                    result.errors.append(f"Blocked IP: {result.resolved_ip}")
            except ValueError:
                pass  # not an IP literal; caller does DNS

        except Exception as e:
            result.is_safe = False
            result.errors.append(f"Parse error: {e}")
        return result

# APPROACH 2 — Full secure parser
# ---------------------------------------------------------------------------

class SecureURLParser:
    """
    Full secure URL parser with obfuscated-IP detection and SSRF prevention.
    """

    def __init__(
        self,
        allowed_schemes: Optional[Set[str]] = None,
        allowed_host_suffixes: Optional[Set[str]] = None,
        max_url_length: int = 8192,
    ) -> None:
        self.allowed_schemes = allowed_schemes or _ALLOWED_SCHEMES
        self.allowed_suffixes = allowed_host_suffixes or set()
        self.max_length = max_url_length

    def parse(self, url: str) -> ParsedURL:
        result = ParsedURL(original=url)

        # --- Length guard ---
        if len(url) > self.max_length:
            result.is_safe = False
            result.errors.append(f"URL exceeds max length {self.max_length}")
            return result

        # --- Basic parse ---
        try:
            parsed = urllib.parse.urlparse(url)
        except Exception as e:
            result.is_safe = False
            result.errors.append(f"Parse error: {e}")
            return result

        result.scheme = (parsed.scheme or "").lower()
        result.path = parsed.path
        result.query = parsed.query
        result.fragment = parsed.fragment

        # --- Scheme check ---
        if result.scheme not in self.allowed_schemes:
            result.is_safe = False
            result.errors.append(f"Scheme {result.scheme!r} not in allowlist")

        # --- Credentials warning ---
        if parsed.username or parsed.password:
            result.warnings.append("URL contains embedded credentials — stripped")

        # --- Host normalisation ---
        raw_host = parsed.netloc
        # Strip credentials
        if "@" in raw_host:
            raw_host = raw_host.rsplit("@", 1)[1]
        # Strip port from netloc
        if raw_host.startswith("["):
            # IPv6 bracket notation
            close = raw_host.find("]")
            host_part = raw_host[:close + 1]
            port_part = raw_host[close + 1:]
        else:
            if ":" in raw_host:
                host_part, _, port_part = raw_host.rpartition(":")
            else:
                host_part, port_part = raw_host, ""

        # Decode percent-encoding in host
        host_decoded = urllib.parse.unquote(host_part).strip()
        result.host = host_decoded.lower().rstrip(".")

        # Port
        if port_part.lstrip(":").isdigit():
            result.port = int(port_part.lstrip(":"))
        else:
            result.port = {"http": 80, "https": 443}.get(result.scheme)

        # --- IP resolution ---
        resolved = _try_parse_ipv4(result.host)
        if resolved:
            result.resolved_ip = str(resolved)
            if _is_blocked_ip(resolved):
                result.is_safe = False
                result.errors.append(f"SSRF: blocked IP {result.resolved_ip} (from {result.host!r})")
        else:
            resolved6 = _try_parse_ipv6(result.host)
            if resolved6:
                result.resolved_ip = str(resolved6)
                if _is_blocked_ip(resolved6):
                    result.is_safe = False
                    result.errors.append(f"SSRF: blocked IPv6 {result.resolved_ip}")

        # --- Allowed suffix check ---
        if self.allowed_suffixes and result.resolved_ip is None:
            # Only apply allowlist if host is not an IP
            if not any(result.host == s or result.host.endswith("." + s)
                       for s in self.allowed_suffixes):
                result.warnings.append(f"Host {result.host!r} not in allowed suffixes")

        # --- Open redirect detection ---
        for key, val in urllib.parse.parse_qsl(result.query):
            if _URL_IN_VALUE_RE.search(val):
                result.warnings.append(
                    f"Potential open-redirect: query param {key!r} contains URL {val[:80]!r}"
                )

        return result

def test_safe_url():
    p = SecureURLParser()
    r = p.parse("https://example.com/path?q=hello")
    assert r.is_safe
    assert r.host == "example.com"
    print("Safe URL: passed")


def test_loopback_blocked():
    p = SecureURLParser()
    for url in ["http://127.0.0.1/admin", "http://localhost/", "https://127.1/"]:
        r = p.parse(url)
        # localhost not an IP literal; 127.0.0.1 and 127.1 should be blocked
        if "127" in url or url == "https://127.1/":
            assert not r.is_safe or r.resolved_ip is None, f"{url} should be blocked"
    print("Loopback blocked: passed")


def test_octal_ip_blocked():
    p = SecureURLParser()
    r = p.parse("http://0177.0.0.1/admin")   # octal for 127.0.0.1
    assert not r.is_safe, f"Expected blocked; got is_safe={r.is_safe}, resolved={r.resolved_ip}"
    assert r.resolved_ip == "127.0.0.1"
    print("Octal IP blocked: passed")


def test_hex_ip_blocked():
    p = SecureURLParser()
    r = p.parse("http://0x7f000001/admin")   # hex for 127.0.0.1
    assert not r.is_safe
    assert r.resolved_ip == "127.0.0.1"
    print("Hex IP blocked: passed")


def test_decimal_ip_blocked():
    p = SecureURLParser()
    r = p.parse("http://2130706433/admin")   # decimal for 127.0.0.1
    assert not r.is_safe
    assert r.resolved_ip == "127.0.0.1"
    print("Decimal IP blocked: passed")


def test_private_rfc1918():
    p = SecureURLParser()
    for url in ["http://10.0.0.1/", "http://192.168.1.1/", "http://172.16.0.1/"]:
        r = p.parse(url)
        assert not r.is_safe, f"{url} should be blocked"
    print("Private RFC1918 blocked: passed")


def test_metadata_endpoint():
    p = SecureURLParser()
    r = p.parse("http://169.254.169.254/latest/meta-data/")
    assert not r.is_safe
    print("Metadata endpoint blocked: passed")


def test_bad_scheme():
    p = SecureURLParser()
    r = p.parse("ftp://example.com/file")
    assert not r.is_safe
    assert "ftp" in r.errors[0]
    print("Bad scheme rejected: passed")


def test_open_redirect():
    p = SecureURLParser()
    r = p.parse("https://safe.com/redirect?url=http://evil.com/steal")
    assert any("open-redirect" in w for w in r.warnings)
    print("Open redirect: passed")


def test_credentials_warning():
    p = SecureURLParser()
    r = p.parse("https://user:pass@example.com/path")
    assert any("credentials" in w for w in r.warnings)
    print("Credentials warning: passed")


def test_ipv6_blocked():
    p = SecureURLParser()
    r = p.parse("http://[::1]/admin")
    assert not r.is_safe
    print("IPv6 loopback blocked: passed")


def test_naive_parser():
    p = NaiveURLParser()
    r = p.parse("http://127.0.0.1/admin")
    assert not r.is_safe
    print("Naive parser: passed")


# ---------------------------------------------------------------------------
# SECURITY ANGLE
# ---------------------------------------------------------------------------
"""
SSRF (Server-Side Request Forgery) ATTACK SURFACE
  An attacker supplies a URL that the server fetches on their behalf.
  Target: internal services (metadata API, admin panels, internal HTTP APIs).

BYPASS TECHNIQUES
  - Numeric forms: decimal(2130706433), octal(0177.0.0.1), hex(0x7f000001)
  - Partial notation: 127.1, 0.0.1
  - IPv6: [::1], [::ffff:127.0.0.1] (IPv4-mapped)
  - DNS rebinding: domain resolves to external IP at validation time,
    then to internal IP when the actual request is made.
  - HTTP redirects: server follows 301/302 to internal target.
  - Unicode/IDN: ⓵.⓶.⓷.⓸ → 1.2.3.4

HARDENING
  - Use an allowlist of permitted destinations (not a blocklist).
  - Pin DNS resolution: resolve once, validate, then use the IP directly.
  - Use a dedicated egress proxy with SSRF-safe policy.
  - Block HTTP redirects server-side or re-validate after each redirect.
  - Cloud: block 169.254.169.254 via IMDSv2 (require token header).
"""
