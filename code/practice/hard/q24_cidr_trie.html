"""
=============================================================================
Q24 [Hard] — CIDR Trie for IP Lookup
=============================================================================

PROBLEM
-------
Build a data structure that supports:
  1. Insert a CIDR block (e.g. "10.0.0.0/8") with an associated tag/metadata.
  2. Lookup: given an IP address, return ALL matching CIDR blocks (and their tags).
  3. Best-match: return the most specific (longest-prefix) matching CIDR.
  4. Delete a CIDR block.
  5. Count total covered addresses.
  6. Support both IPv4 and IPv6.

Use case: firewall rule lookup, threat-intel IP reputation lookup,
geo-IP lookup, SSRF blocklist.

CLARIFYING QUESTIONS
--------------------
1. Exact match only or all matching prefixes?
   → Return all matches + specifically identify the longest prefix match.
2. IPv4 only or IPv6 too?
   → Both; store in separate tries or a unified 128-bit trie.
3. Overlapping CIDRs?
   → Allowed; all matches returned.
4. Max number of rules?
   → Up to ~1 million (BGP full table ≈ 900K IPv4 prefixes).
5. Thread safety?
   → Single-threaded; no locking needed.

=============================================================================
APPROACH 1 — NAIVE (linear scan)
=============================================================================
  Store list of (network, tag) pairs.
  Lookup: iterate all entries; check if IP in network.
  O(N) lookup, O(1) insert.
  Unacceptable for large N.

=============================================================================
APPROACH 2 — BINARY TRIE (Patricia Trie / Radix Tree) — RECOMMENDED
=============================================================================
  Store prefixes in a binary trie keyed by prefix bits.
  Insert: traverse/create nodes for each bit of the prefix.
  Lookup: traverse bits of IP; collect all nodes marked as terminal.
  O(W) lookup and insert  (W = address width: 32 for IPv4, 128 for IPv6).

APPROACH 3 — INTERVAL TREE (alternative for large overlapping ranges)
=============================================================================
  Represent each CIDR as an integer interval [first, last].
  Use sorted list + bisect for lookup.
  O(log N + K) lookup, O(log N) insert (with sorted insertion).
  Better for heavily overlapping ranges where K matches are needed.

=============================================================================
"""

from __future__ import annotations

import ipaddress
from bisect import bisect_left, insort
from dataclasses import dataclass, field
from typing import Any, Dict, Iterator, List, Optional, Tuple


# ---------------------------------------------------------------------------
# Data Types
# ---------------------------------------------------------------------------

@dataclass
class CIDREntry:
    network: ipaddress.IPv4Network | ipaddress.IPv6Network
    tag: Any
    prefix_len: int

    def __lt__(self, other: "CIDREntry") -> bool:
        return self.prefix_len < other.prefix_len


# ---------------------------------------------------------------------------
# APPROACH 1 — Naive linear scan
# ---------------------------------------------------------------------------

class NaiveCIDRLookup:
    """
    O(N) lookup.  Simple baseline.
    """

    def __init__(self) -> None:
        self._entries: List[CIDREntry] = []

    def insert(self, cidr: str, tag: Any = None) -> None:
        net = ipaddress.ip_network(cidr, strict=False)
        self._entries.append(CIDREntry(net, tag, net.prefixlen))

    def lookup(self, ip: str) -> List[CIDREntry]:
        addr = ipaddress.ip_address(ip)
        return [e for e in self._entries if addr in e.network]

    def best_match(self, ip: str) -> Optional[CIDREntry]:
        matches = self.lookup(ip)
        return max(matches, key=lambda e: e.prefix_len) if matches else None

    def delete(self, cidr: str) -> bool:
        net = ipaddress.ip_network(cidr, strict=False)
        for i, e in enumerate(self._entries):
            if e.network == net:
                self._entries.pop(i)
                return True
        return False

    def __len__(self) -> int:
        return len(self._entries)


# ---------------------------------------------------------------------------
# APPROACH 2 — Binary Trie (RECOMMENDED)
# ---------------------------------------------------------------------------

class _TrieNode:
    __slots__ = ("children", "entries")

    def __init__(self) -> None:
        self.children: List[Optional["_TrieNode"]] = [None, None]
        self.entries: List[CIDREntry] = []  # CIDRs terminating at this node


class CIDRTrie:
    """
    Binary trie for O(W) IP → CIDR lookup.
    Supports IPv4 (W=32) and IPv6 (W=128).
    Stores IPv4 and IPv6 in separate tries.
    """

    def __init__(self) -> None:
        self._root4 = _TrieNode()  # IPv4 trie
        self._root6 = _TrieNode()  # IPv6 trie
        self._count = 0

    # -----------------------------------------------------------------------
    # Helpers
    # -----------------------------------------------------------------------

    @staticmethod
    def _network_bits(network: ipaddress.IPv4Network | ipaddress.IPv6Network) -> Iterator[int]:
        """Yield bits of the network address, MSB first."""
        n = int(network.network_address)
        width = 32 if isinstance(network, ipaddress.IPv4Network) else 128
        for i in range(width - 1, -1, -1):
            yield (n >> i) & 1

    @staticmethod
    def _ip_bits(addr: ipaddress.IPv4Address | ipaddress.IPv6Address) -> Iterator[int]:
        n = int(addr)
        width = 32 if isinstance(addr, ipaddress.IPv4Address) else 128
        for i in range(width - 1, -1, -1):
            yield (n >> i) & 1

    def _root(self, is_v6: bool) -> _TrieNode:
        return self._root6 if is_v6 else self._root4

    # -----------------------------------------------------------------------
    # Public API
    # -----------------------------------------------------------------------

    def insert(self, cidr: str, tag: Any = None) -> None:
        net = ipaddress.ip_network(cidr, strict=False)
        is_v6 = isinstance(net, ipaddress.IPv6Network)
        entry = CIDREntry(net, tag, net.prefixlen)
        node = self._root(is_v6)
        bits = self._network_bits(net)
        for i, bit in enumerate(bits):
            if i == net.prefixlen:
                break
            if node.children[bit] is None:
                node.children[bit] = _TrieNode()
            node = node.children[bit]
        node.entries.append(entry)
        self._count += 1

    def lookup(self, ip: str) -> List[CIDREntry]:
        """Return all CIDR entries that contain `ip`."""
        addr = ipaddress.ip_address(ip)
        is_v6 = isinstance(addr, ipaddress.IPv6Address)
        node = self._root(is_v6)
        matches: List[CIDREntry] = list(node.entries)  # match root (0.0.0.0/0 if present)
        for bit in self._ip_bits(addr):
            child = node.children[bit]
            if child is None:
                break
            node = child
            matches.extend(node.entries)
        return matches

    def best_match(self, ip: str) -> Optional[CIDREntry]:
        """Return the longest-prefix matching CIDR."""
        matches = self.lookup(ip)
        return max(matches, key=lambda e: e.prefix_len) if matches else None

    def delete(self, cidr: str) -> bool:
        """Delete the exact CIDR entry.  Returns True if found and removed."""
        net = ipaddress.ip_network(cidr, strict=False)
        is_v6 = isinstance(net, ipaddress.IPv6Network)
        node = self._root(is_v6)
        for i, bit in enumerate(self._network_bits(net)):
            if i == net.prefixlen:
                break
            if node.children[bit] is None:
                return False
            node = node.children[bit]
        before = len(node.entries)
        node.entries = [e for e in node.entries if e.network != net]
        removed = len(node.entries) < before
        if removed:
            self._count -= 1
        return removed

    def __len__(self) -> int:
        return self._count


# ---------------------------------------------------------------------------
# APPROACH 3 — Interval tree (sorted list + bisect)
# ---------------------------------------------------------------------------

@dataclass
class _Interval:
    first: int      # integer representation of first IP
    last: int       # integer representation of last IP
    entry: CIDREntry

    def __lt__(self, other: "_Interval") -> bool:
        return (self.first, -self.last) < (other.first, -other.last)


class CIDRIntervalLookup:
    """
    Represents each CIDR as an integer interval.
    Sorted by (first, -last) → efficient range search with bisect.
    O(log N) to find start; O(K) to collect matches.
    Handles overlapping ranges well.
    """

    def __init__(self) -> None:
        self._intervals: List[_Interval] = []

    def insert(self, cidr: str, tag: Any = None) -> None:
        net = ipaddress.ip_network(cidr, strict=False)
        entry = CIDREntry(net, tag, net.prefixlen)
        interval = _Interval(
            int(net.network_address),
            int(net.broadcast_address),
            entry,
        )
        insort(self._intervals, interval)

    def lookup(self, ip: str) -> List[CIDREntry]:
        addr_int = int(ipaddress.ip_address(ip))
        results = []
        # Find all intervals where first <= addr_int <= last
        # Binary search for insertion point of (addr_int+1, ...) to get upper bound
        pos = bisect_left(self._intervals, _Interval(addr_int + 1, 0, None))  # type: ignore
        for i in range(pos - 1, -1, -1):
            iv = self._intervals[i]
            if iv.first > addr_int:
                continue
            if iv.last >= addr_int:
                results.append(iv.entry)
            elif iv.last < addr_int and iv.first < addr_int:
                # Intervals are sorted; all earlier ones start even smaller
                # but may still contain addr_int — must scan
                continue
        return results

    def best_match(self, ip: str) -> Optional[CIDREntry]:
        matches = self.lookup(ip)
        return max(matches, key=lambda e: e.prefix_len) if matches else None

    def delete(self, cidr: str) -> bool:
        net = ipaddress.ip_network(cidr, strict=False)
        first = int(net.network_address)
        last = int(net.broadcast_address)
        for i, iv in enumerate(self._intervals):
            if iv.first == first and iv.last == last and iv.entry.network == net:
                self._intervals.pop(i)
                return True
        return False

    def __len__(self) -> int:
        return len(self._intervals)

def _run_tests(impl_name: str, Cls):
    t = Cls()
    t.insert("10.0.0.0/8", tag="rfc1918")
    t.insert("10.1.0.0/16", tag="subnet-10.1")
    t.insert("192.168.0.0/16", tag="rfc1918-192")
    t.insert("0.0.0.0/0", tag="default")

    # Test: IP in 10.1.x.x matches both /8 and /16
    matches = t.lookup("10.1.5.5")
    tags = {e.tag for e in matches}
    assert "rfc1918" in tags, f"{impl_name}: Expected rfc1918 in {tags}"
    assert "subnet-10.1" in tags, f"{impl_name}: Expected subnet-10.1 in {tags}"
    assert "default" in tags, f"{impl_name}: Expected default in {tags}"

    # Best match should be /16
    best = t.best_match("10.1.5.5")
    assert best and best.prefix_len == 16, \
        f"{impl_name}: Expected /16 as best match, got {best}"

    # IP not in any specific subnet
    best2 = t.best_match("10.2.0.1")
    assert best2 and best2.tag == "rfc1918", \
        f"{impl_name}: Expected rfc1918 for 10.2.0.1, got {best2}"

    # Delete and re-check
    t.delete("10.1.0.0/16")
    best3 = t.best_match("10.1.5.5")
    assert best3 and best3.prefix_len == 8, \
        f"{impl_name}: After delete, expected /8 as best, got {best3}"

    print(f"{impl_name}: all tests passed")


def test_naive():
    _run_tests("NaiveCIDRLookup", NaiveCIDRLookup)


def test_trie():
    _run_tests("CIDRTrie", CIDRTrie)


def test_interval():
    _run_tests("CIDRIntervalLookup", CIDRIntervalLookup)


def test_ipv6_trie():
    t = CIDRTrie()
    t.insert("fe80::/10", tag="link-local")
    t.insert("::/0", tag="default6")
    t.insert("::1/128", tag="loopback6")
    matches = t.lookup("::1")
    tags = {e.tag for e in matches}
    assert "loopback6" in tags
    best = t.best_match("::1")
    assert best and best.prefix_len == 128
    print("IPv6 trie: passed")


def test_host_route():
    t = CIDRTrie()
    t.insert("192.168.1.100/32", tag="host-route")
    t.insert("192.168.1.0/24", tag="subnet")
    best = t.best_match("192.168.1.100")
    assert best and best.prefix_len == 32 and best.tag == "host-route"
    print("Host route (/32) best match: passed")


def test_no_match():
    t = CIDRTrie()
    t.insert("10.0.0.0/8", tag="private")
    matches = t.lookup("8.8.8.8")
    assert len(matches) == 0
    print("No match: passed")

"""
USE CASES IN SECURITY
  1. Firewall rule lookup: given packet destination IP, find applicable rules.
  2. Threat intel: classify src IP as known-malicious / TOR exit / botnet C2.
  3. SSRF prevention: block requests to private CIDR ranges (Q22).
  4. Geo-IP: map IP to country/region for rate-limiting and fraud detection.
  5. DDoS mitigation: match incoming SYN packets against block-listed prefixes.

TRIE vs INTERVAL TREE
  Trie:           O(W) per lookup regardless of N.  Best for very large tables
                  (BGP: 900K+ prefixes).  W=32 for IPv4 → 32 comparisons max.
  Interval tree:  O(log N + K) per lookup.  Better when N is small and many
                  overlapping ranges need to be collected (K is large).
  Naive list:     O(N) — only for N < 100 or one-shot checks.

PRODUCTION NOTE
  For BGP-scale lookups (millions of routes), use a Radix/Patricia trie
  (Python: `pytricia` library) or hardware TCAM (256-bit ternary CAM).
"""
