"""
=============================================================================
Q25 [Hard] — Longest Prefix Match for Firewall Rules
=============================================================================

PROBLEM
-------
Implement a firewall rule engine that:
  1. Stores rules of the form: (src_cidr, dst_cidr, proto, dst_port, action)
     where action ∈ {ALLOW, DENY} and fields may be wildcards (*).
  2. Given a packet (src_ip, dst_ip, proto, dst_port), returns the FIRST
     matching rule in priority order (most specific match wins; for equal
     specificity, insertion order wins).
  3. Default action when no rule matches is configurable (default: DENY).
  4. Support deletion and listing of rules.
  5. Fast lookup for high-throughput scenarios.

Specificity ordering (tiebreaker):
  - More specific CIDR (longer prefix) > less specific
  - Specific port > wildcard port
  - Specific protocol > wildcard protocol

CLARIFYING QUESTIONS
--------------------
1. Port range or single port?
   → Single port or wildcard; port ranges can be added as an extension.
2. Stateful or stateless firewall?
   → Stateless; each packet evaluated independently.
3. IPv4 + IPv6?
   → IPv4 focus; structure supports IPv6 extension.
4. Priority ties broken by insertion order?
   → Yes; earlier-inserted rule wins on equal specificity.
5. Up to how many rules?
   → Up to ~100K rules; O(W) per lookup is required.

=============================================================================
APPROACH 1 — NAIVE (linear scan, first-match)
=============================================================================
  Store rules in a list sorted by priority.
  Scan until a match is found.
  O(R) lookup, O(1) insert.
  Acceptable for small rule sets (< 500 rules).

=============================================================================
APPROACH 2 — DUAL TRIE LOOKUP + RULE PRIORITY SORT (RECOMMENDED)
=============================================================================
  - Two CIDR tries (src, dst) for IP matching.
  - For each lookup, find candidate rules from both tries.
  - Intersect candidates; return highest-priority matching rule.
  Alternatively: use CIDRTrie from Q24 for IP matching; supplement with
  port/proto hash.
  O(W) for CIDR lookup + O(K) to find best matching complete rule
  (K = candidates sharing the matched CIDR).

=============================================================================
"""

from __future__ import annotations

import ipaddress
from dataclasses import dataclass, field
from enum import Enum
from typing import Any, Dict, List, Optional, Set, Tuple

from q24_cidr_trie import CIDRTrie, CIDREntry


# ---------------------------------------------------------------------------
# Data Types
# ---------------------------------------------------------------------------

class Action(Enum):
    ALLOW = "ALLOW"
    DENY = "DENY"


@dataclass
class FirewallRule:
    rule_id: int
    src_cidr: str      # "10.0.0.0/8" or "*" (any)
    dst_cidr: str
    proto: str         # "tcp" | "udp" | "icmp" | "*"
    dst_port: str      # "443" | "*"
    action: Action
    comment: str = ""

    # Priority score: higher is more specific (evaluated first)
    @property
    def specificity(self) -> int:
        score = 0
        if self.src_cidr != "*":
            score += ipaddress.ip_network(self.src_cidr, strict=False).prefixlen
        if self.dst_cidr != "*":
            score += ipaddress.ip_network(self.dst_cidr, strict=False).prefixlen
        if self.proto != "*":
            score += 8
        if self.dst_port != "*":
            score += 8
        return score

    def matches(self, src_ip: str, dst_ip: str, proto: str, dst_port: str) -> bool:
        # IP checks
        if self.src_cidr != "*":
            net = ipaddress.ip_network(self.src_cidr, strict=False)
            if ipaddress.ip_address(src_ip) not in net:
                return False
        if self.dst_cidr != "*":
            net = ipaddress.ip_network(self.dst_cidr, strict=False)
            if ipaddress.ip_address(dst_ip) not in net:
                return False
        # Proto and port
        if self.proto != "*" and self.proto.lower() != proto.lower():
            return False
        if self.dst_port != "*" and self.dst_port != str(dst_port):
            return False
        return True


@dataclass
class Verdict:
    action: Action
    rule: Optional[FirewallRule]
    default: bool = False  # True if default policy applied


# ---------------------------------------------------------------------------
# APPROACH 1 — Naive linear scan
# ---------------------------------------------------------------------------

class NaiveFirewall:
    """O(R) lookup; rules pre-sorted by decreasing specificity."""

    def __init__(self, default_action: Action = Action.DENY) -> None:
        self.default_action = default_action
        self._rules: List[FirewallRule] = []
        self._next_id = 1

    def add_rule(self, src_cidr: str, dst_cidr: str, proto: str,
                 dst_port: str, action: Action, comment: str = "") -> int:
        rule = FirewallRule(
            rule_id=self._next_id, src_cidr=src_cidr, dst_cidr=dst_cidr,
            proto=proto, dst_port=dst_port, action=action, comment=comment,
        )
        self._next_id += 1
        self._rules.append(rule)
        # Keep sorted: highest specificity first; ties by rule_id (insertion order)
        self._rules.sort(key=lambda r: (-r.specificity, r.rule_id))
        return rule.rule_id

    def lookup(self, src_ip: str, dst_ip: str, proto: str, dst_port: str) -> Verdict:
        for rule in self._rules:
            if rule.matches(src_ip, dst_ip, proto, dst_port):
                return Verdict(action=rule.action, rule=rule)
        return Verdict(action=self.default_action, rule=None, default=True)

    def delete_rule(self, rule_id: int) -> bool:
        before = len(self._rules)
        self._rules = [r for r in self._rules if r.rule_id != rule_id]
        return len(self._rules) < before

    def list_rules(self) -> List[FirewallRule]:
        return list(self._rules)

    def __len__(self) -> int:
        return len(self._rules)


# ---------------------------------------------------------------------------
# APPROACH 2 — Dual CIDR trie + priority-sorted candidate lookup
# ---------------------------------------------------------------------------

class Firewall:
    """
    Trie-accelerated firewall.

    Index: 
      - `_dst_trie`: CIDRTrie; each entry's `tag` is a set of rule_ids.
      - Rules stored in `_rules` dict by rule_id.
      - On lookup: find candidate rules from dst_trie, filter by src, proto, port.
      - Pick highest-specificity rule among candidates.

    O(W) for trie lookup + O(K) rule evaluation, where K = rules per CIDR.
    """

    def __init__(self, default_action: Action = Action.DENY) -> None:
        self.default_action = default_action
        self._rules: Dict[int, FirewallRule] = {}
        self._dst_trie: CIDRTrie = CIDRTrie()
        # Wildcard-dst rules must be checked separately
        self._wildcard_dst_rules: List[int] = []
        self._next_id = 1

    def add_rule(
        self,
        src_cidr: str = "*",
        dst_cidr: str = "*",
        proto: str = "*",
        dst_port: str = "*",
        action: Action = Action.DENY,
        comment: str = "",
    ) -> int:
        rid = self._next_id
        self._next_id += 1
        rule = FirewallRule(
            rule_id=rid, src_cidr=src_cidr, dst_cidr=dst_cidr,
            proto=proto, dst_port=dst_port, action=action, comment=comment,
        )
        self._rules[rid] = rule

        if dst_cidr == "*":
            self._wildcard_dst_rules.append(rid)
        else:
            # Insert into trie with tag=rule_id
            self._dst_trie.insert(dst_cidr, tag=rid)

        return rid

    def lookup(self, src_ip: str, dst_ip: str, proto: str, dst_port: str) -> Verdict:
        # Gather candidate rule IDs via dst trie
        candidate_ids: Set[int] = set()
        for entry in self._dst_trie.lookup(dst_ip):
            candidate_ids.add(entry.tag)
        candidate_ids.update(self._wildcard_dst_rules)

        # Evaluate candidates
        best_rule: Optional[FirewallRule] = None
        for rid in candidate_ids:
            rule = self._rules.get(rid)
            if rule and rule.matches(src_ip, dst_ip, proto, dst_port):
                if best_rule is None:
                    best_rule = rule
                elif rule.specificity > best_rule.specificity:
                    best_rule = rule
                elif rule.specificity == best_rule.specificity and rule.rule_id < best_rule.rule_id:
                    best_rule = rule  # earlier insertion wins ties

        if best_rule:
            return Verdict(action=best_rule.action, rule=best_rule)
        return Verdict(action=self.default_action, rule=None, default=True)

    def delete_rule(self, rule_id: int) -> bool:
        if rule_id not in self._rules:
            return False
        rule = self._rules.pop(rule_id)
        if rule.dst_cidr == "*":
            self._wildcard_dst_rules.remove(rule_id)
        else:
            self._dst_trie.delete(rule.dst_cidr)
        return True

    def list_rules(self) -> List[FirewallRule]:
        return sorted(self._rules.values(), key=lambda r: (-r.specificity, r.rule_id))

    def __len__(self) -> int:
        return len(self._rules)

def _test_impl(name: str, Cls):
    fw = Cls(default_action=Action.DENY)

    # Allow HTTPS to DMZ subnet
    fw.add_rule(src_cidr="*", dst_cidr="192.168.10.0/24",
                proto="tcp", dst_port="443", action=Action.ALLOW,
                comment="allow-https-dmz")
    # Deny known-bad source
    fw.add_rule(src_cidr="10.200.0.0/16", dst_cidr="*",
                proto="*", dst_port="*", action=Action.DENY,
                comment="block-bad-src")
    # Allow SSH from admin range to specific host
    fw.add_rule(src_cidr="10.1.0.0/24", dst_cidr="192.168.10.50/32",
                proto="tcp", dst_port="22", action=Action.ALLOW,
                comment="admin-ssh")

    # Test 1: HTTPS to DMZ → ALLOW
    v = fw.lookup("1.2.3.4", "192.168.10.99", "tcp", "443")
    assert v.action == Action.ALLOW, f"{name}: Expected ALLOW; got {v.action}"

    # Test 2: HTTP to DMZ (port 80 not in rules) → DENY (default)
    v = fw.lookup("1.2.3.4", "192.168.10.99", "tcp", "80")
    assert v.action == Action.DENY and v.default, f"{name}: Expected default DENY"

    # Test 3: Bad source to anywhere → DENY (explicit rule)
    v = fw.lookup("10.200.1.5", "8.8.8.8", "tcp", "80")
    assert v.action == Action.DENY, f"{name}: Expected DENY for bad src"

    # Test 4: Admin SSH to specific host → ALLOW (most specific)
    v = fw.lookup("10.1.0.10", "192.168.10.50", "tcp", "22")
    assert v.action == Action.ALLOW, f"{name}: Expected ALLOW for admin SSH"

    # Test 5: Delete the HTTPS rule → DENY
    rid = fw.list_rules()[0].rule_id
    # Find the https rule
    https_rule = next((r for r in fw.list_rules() if r.comment == "allow-https-dmz"), None)
    if https_rule:
        fw.delete_rule(https_rule.rule_id)
    v = fw.lookup("1.2.3.4", "192.168.10.99", "tcp", "443")
    assert v.action == Action.DENY, f"{name}: After delete, expected DENY"

    print(f"{name}: all tests passed")


def test_naive():
    _test_impl("NaiveFirewall", NaiveFirewall)


def test_trie_firewall():
    _test_impl("Firewall", Firewall)


def test_default_allow():
    fw = Firewall(default_action=Action.ALLOW)
    fw.add_rule(src_cidr="10.0.0.0/8", dst_cidr="*",
                proto="tcp", dst_port="22", action=Action.DENY)
    v = fw.lookup("8.8.8.8", "1.1.1.1", "tcp", "80")
    assert v.action == Action.ALLOW and v.default
    v2 = fw.lookup("10.1.2.3", "1.1.1.1", "tcp", "22")
    assert v2.action == Action.DENY
    print("Default ALLOW + explicit DENY: passed")


def test_most_specific_wins():
    fw = Firewall(default_action=Action.DENY)
    fw.add_rule(dst_cidr="192.168.0.0/16", proto="*", dst_port="*", action=Action.DENY)
    fw.add_rule(dst_cidr="192.168.1.0/24", proto="tcp", dst_port="80", action=Action.ALLOW)
    v = fw.lookup("1.2.3.4", "192.168.1.50", "tcp", "80")
    assert v.action == Action.ALLOW, "More specific rule should win"
    print("Most specific wins: passed")

"""
FIREWALL RULE ORDERING
  Security mis-configurations often arise from rule ordering bugs:
  - DENY rule after ALLOW rule: the DENY never fires.
  - Over-broad ALLOW before specific DENY: attackers exploit the gap.

LONGEST-PREFIX-MATCH SIGNIFICANCE
  Network firewalls (iptables, nftables, cloud SGs) apply rules in order.
  Firewalls using TCAM hardware use LPM to match the "most specific" rule.
  If an ALLOW /32 is in the table alongside a DENY /8, the /32 wins for
  that specific host — this is intentional for overrides.

COMMON MISCONFIGURATIONS
  - Allow * → * on internal segments (flat network).
  - Allow * → 0.0.0.0/0 on egress → exfiltration risk.
  - Shadowed rules: a rule that can never match because an earlier rule
    always matches first (Q26 detects this).

CLOUD SECURITY GROUPS
  AWS SGs: stateful; implicit DENY at end; rules evaluated as a set (most
  permissive wins — counter-intuitive).
  GCP Firewall: priority-ordered; lower number = higher priority.
"""
