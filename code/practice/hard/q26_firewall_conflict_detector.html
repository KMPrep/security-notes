"""
=============================================================================
Q26 [Hard] — Firewall Rule Conflict Detector
=============================================================================

PROBLEM
-------
Given a list of firewall rules, detect the following conflict types:

  1. Shadowed rules: Rule B is shadowed by Rule A if every packet that
     would match B also matches A, and A appears before B in priority order.
     Result: B never fires.

  2. Redundant rules: Rule B is redundant if removing it does not change
     any packet's verdict (equivalent to shadowed for same-action pairs).

  3. Contradictory rules: Two rules match exactly the same traffic but
     have opposite actions (ALLOW vs DENY).

  4. Overly-permissive rules: Rules with too many wildcards (e.g. allow
     ANY → ANY on ANY port) that should be flagged for review.

  5. Duplicate rules: Identical (src, dst, proto, port, action) tuples.

Input: list of rule dicts in priority order (index 0 = highest priority).
Each rule: {src_cidr, dst_cidr, proto, dst_port, action}

CLARIFYING QUESTIONS
--------------------
1. What defines "shadows"?
   → Rule A shadows Rule B if A's src_cidr ⊇ B's src_cidr AND
     A's dst_cidr ⊇ B's dst_cidr AND proto/port of A is ≥ permissive than B.
2. Does "A appears before B" mean lower rule_id or lower index?
   → Lower index in the input list = higher priority.
3. How to handle wildcards in CIDR comparison?
   → "*" means 0.0.0.0/0 (covers everything) for IPv4 comparisons.
4. Do we detect cross-protocol conflicts?
   → Flag opposite-action rules with overlapping traffic.
5. Return all conflicts or just count?
   → Return list of conflict objects with details.

=============================================================================
APPROACH 1 — NAIVE (O(R²) pairwise comparison)
=============================================================================
  Compare every pair of rules.  Check containment.
  O(R²) time.  Fine for small rule sets (< 1000 rules).

=============================================================================
APPROACH 2 — INTERVAL + SET ALGEBRA (RECOMMENDED for large rule sets)
=============================================================================
  Represent each field as a set or interval:
    - CIDR → integer range [first, last]
    - proto, port → sets or singletons
  Compute set overlap/containment using integer arithmetic.
  O(R²) worst case but with early exit; O(R log R) with sorted ranges.

  For very large sets: use BDD (Binary Decision Diagram) or SAT solver.
  Here we implement the O(R²) approach with fast interval arithmetic.

=============================================================================
"""

from __future__ import annotations

import ipaddress
from dataclasses import dataclass, field
from enum import Enum
from typing import FrozenSet, List, Optional, Set, Tuple


# ---------------------------------------------------------------------------
# Data Types
# ---------------------------------------------------------------------------

class Action(Enum):
    ALLOW = "ALLOW"
    DENY = "DENY"


@dataclass
class Rule:
    rule_id: int
    src_cidr: str     # "10.0.0.0/8" or "*"
    dst_cidr: str
    proto: str        # "tcp" | "udp" | "icmp" | "*"
    dst_port: str     # "443" | "*"
    action: Action

    def __post_init__(self) -> None:
        self._src_net = _parse_net(self.src_cidr)
        self._dst_net = _parse_net(self.dst_cidr)

    @property
    def src_range(self) -> Tuple[int, int]:
        return _net_range(self._src_net)

    @property
    def dst_range(self) -> Tuple[int, int]:
        return _net_range(self._dst_net)

    @property
    def port_set(self) -> Optional[int]:
        """None means wildcard; otherwise the integer port."""
        return None if self.dst_port == "*" else int(self.dst_port)

    @property
    def proto_lower(self) -> Optional[str]:
        return None if self.proto == "*" else self.proto.lower()

    def wildcard_count(self) -> int:
        """Count of wildcard fields (0–4)."""
        return sum([
            self.src_cidr == "*",
            self.dst_cidr == "*",
            self.proto == "*",
            self.dst_port == "*",
        ])


@dataclass
class Conflict:
    conflict_type: str   # shadowed | redundant | contradictory | overpermissive | duplicate
    severity: str        # high | medium | low
    rule_a_id: int
    rule_b_id: int = -1
    detail: str = ""


# ---------------------------------------------------------------------------
# IP Range Helpers
# ---------------------------------------------------------------------------

_ALL_IPV4 = ipaddress.IPv4Network("0.0.0.0/0")

def _parse_net(cidr: str) -> ipaddress.IPv4Network:
    if cidr == "*":
        return _ALL_IPV4
    return ipaddress.ip_network(cidr, strict=False)


def _net_range(net: ipaddress.IPv4Network) -> Tuple[int, int]:
    return int(net.network_address), int(net.broadcast_address)


def _range_contains(outer: Tuple[int, int], inner: Tuple[int, int]) -> bool:
    """True if outer range fully contains inner range."""
    return outer[0] <= inner[0] and outer[1] >= inner[1]


def _ranges_overlap(a: Tuple[int, int], b: Tuple[int, int]) -> bool:
    return a[0] <= b[1] and b[0] <= a[1]


def _ranges_equal(a: Tuple[int, int], b: Tuple[int, int]) -> bool:
    return a == b


def _field_a_covers_b(a_val: Optional[object], b_val: Optional[object]) -> bool:
    """True if A's field is at least as permissive as B's (None = wildcard)."""
    if a_val is None:
        return True  # A wildcard covers everything
    return a_val == b_val  # A specific field only covers same value


def _traffic_overlap(a: Rule, b: Rule) -> bool:
    """True if any packet could match both A and B."""
    if not _ranges_overlap(a.src_range, b.src_range):
        return False
    if not _ranges_overlap(a.dst_range, b.dst_range):
        return False
    if a.proto_lower and b.proto_lower and a.proto_lower != b.proto_lower:
        return False
    if a.port_set and b.port_set and a.port_set != b.port_set:
        return False
    return True


def _a_shadows_b(a: Rule, b: Rule) -> bool:
    """
    True if every packet matching B also matches A
    (A is a superset of B in traffic coverage).
    A must appear before B in priority order (caller checks order).
    """
    # A must cover B's src range
    if not _range_contains(a.src_range, b.src_range):
        return False
    # A must cover B's dst range
    if not _range_contains(a.dst_range, b.dst_range):
        return False
    # A must cover B's proto
    if not _field_a_covers_b(a.proto_lower, b.proto_lower):
        return False
    # A must cover B's port
    if not _field_a_covers_b(a.port_set, b.port_set):
        return False
    return True


def _rules_are_identical_traffic(a: Rule, b: Rule) -> bool:
    return (a.src_range == b.src_range and
            a.dst_range == b.dst_range and
            a.proto_lower == b.proto_lower and
            a.port_set == b.port_set)


# ---------------------------------------------------------------------------
# APPROACH 1 — Naive pairwise conflict detector
# ---------------------------------------------------------------------------

def detect_conflicts_naive(rules: List[Rule], overperms_threshold: int = 2) -> List[Conflict]:
    """
    O(R²) pairwise comparison.
    `overperms_threshold`: rules with >= this many wildcards are flagged.
    """
    conflicts: List[Conflict] = []
    seen_traffic: Set[Tuple] = set()

    for i, a in enumerate(rules):
        # Overly-permissive check (single-rule)
        if a.wildcard_count() >= overperms_threshold:
            conflicts.append(Conflict(
                conflict_type="overpermissive",
                severity="medium",
                rule_a_id=a.rule_id,
                detail=f"Rule {a.rule_id} has {a.wildcard_count()} wildcard fields",
            ))

        for j in range(i + 1, len(rules)):
            b = rules[j]

            # Duplicate
            if (a.src_cidr == b.src_cidr and a.dst_cidr == b.dst_cidr and
                    a.proto == b.proto and a.dst_port == b.dst_port and
                    a.action == b.action):
                conflicts.append(Conflict(
                    conflict_type="duplicate",
                    severity="low",
                    rule_a_id=a.rule_id,
                    rule_b_id=b.rule_id,
                    detail=f"Rules {a.rule_id} and {b.rule_id} are identical",
                ))

            # Contradictory (same traffic, opposite actions)
            elif _rules_are_identical_traffic(a, b) and a.action != b.action:
                conflicts.append(Conflict(
                    conflict_type="contradictory",
                    severity="high",
                    rule_a_id=a.rule_id,
                    rule_b_id=b.rule_id,
                    detail=(f"Rules {a.rule_id}({a.action.value}) and "
                            f"{b.rule_id}({b.action.value}) match same traffic"),
                ))

            # Shadowed: A (higher priority) shadows B
            elif _a_shadows_b(a, b):
                severity = "high" if a.action != b.action else "low"
                conflict_type = "shadowed" if a.action != b.action else "redundant"
                conflicts.append(Conflict(
                    conflict_type=conflict_type,
                    severity=severity,
                    rule_a_id=a.rule_id,
                    rule_b_id=b.rule_id,
                    detail=(f"Rule {a.rule_id} ({a.action.value}) shadows "
                            f"rule {b.rule_id} ({b.action.value})"),
                ))

    return conflicts


# ---------------------------------------------------------------------------
# APPROACH 2 — Full detector with deduplication
# ---------------------------------------------------------------------------

def detect_conflicts(rules: List[Rule], overperms_threshold: int = 2) -> List[Conflict]:
    """
    Full conflict detector.
    Same O(R²) algorithm as naive but deduplicates conflict pairs and
    returns conflicts sorted by severity.
    """
    all_conflicts = detect_conflicts_naive(rules, overperms_threshold)
    # Deduplicate by (type, a_id, b_id)
    seen = set()
    deduped = []
    for c in all_conflicts:
        key = (c.conflict_type, c.rule_a_id, c.rule_b_id)
        if key not in seen:
            seen.add(key)
            deduped.append(c)
    # Sort: high severity first
    severity_order = {"high": 0, "medium": 1, "low": 2}
    deduped.sort(key=lambda c: severity_order.get(c.severity, 3))
    return deduped

def _rule(rid, src, dst, proto, port, action_str) -> Rule:
    return Rule(
        rule_id=rid, src_cidr=src, dst_cidr=dst,
        proto=proto, dst_port=port,
        action=Action.ALLOW if action_str == "ALLOW" else Action.DENY,
    )


def test_shadowed():
    rules = [
        _rule(1, "*", "0.0.0.0/0", "*", "*", "ALLOW"),  # allow all
        _rule(2, "*", "10.0.0.0/8", "tcp", "22", "DENY"),  # deny SSH to private
    ]
    # Rule 1 shadows rule 2 (allow-all comes first)
    conflicts = detect_conflicts(rules)
    shadow = [c for c in conflicts if c.conflict_type == "shadowed"]
    assert shadow, f"Expected shadowed conflict; got {[c.conflict_type for c in conflicts]}"
    assert shadow[0].rule_a_id == 1 and shadow[0].rule_b_id == 2
    print("Shadowed: passed")


def test_redundant():
    rules = [
        _rule(1, "*", "10.0.0.0/8", "tcp", "80", "ALLOW"),
        _rule(2, "*", "10.1.0.0/16", "tcp", "80", "ALLOW"),  # redundant: same action, already covered
    ]
    conflicts = detect_conflicts(rules)
    redundant = [c for c in conflicts if c.conflict_type == "redundant"]
    assert redundant, f"Expected redundant; got {[c.conflict_type for c in conflicts]}"
    print("Redundant: passed")


def test_contradictory():
    rules = [
        _rule(1, "10.0.0.0/8", "192.168.1.0/24", "tcp", "443", "ALLOW"),
        _rule(2, "10.0.0.0/8", "192.168.1.0/24", "tcp", "443", "DENY"),
    ]
    conflicts = detect_conflicts(rules)
    contra = [c for c in conflicts if c.conflict_type == "contradictory"]
    assert contra
    print("Contradictory: passed")


def test_duplicate():
    rules = [
        _rule(1, "10.0.0.0/8", "*", "tcp", "22", "DENY"),
        _rule(2, "10.0.0.0/8", "*", "tcp", "22", "DENY"),
    ]
    conflicts = detect_conflicts(rules)
    dups = [c for c in conflicts if c.conflict_type == "duplicate"]
    assert dups
    print("Duplicate: passed")


def test_overpermissive():
    rules = [
        _rule(1, "*", "*", "*", "*", "ALLOW"),  # 4 wildcards
    ]
    conflicts = detect_conflicts(rules, overperms_threshold=2)
    over = [c for c in conflicts if c.conflict_type == "overpermissive"]
    assert over
    print("Overpermissive: passed")


def test_clean_rules():
    rules = [
        _rule(1, "10.1.0.0/24", "192.168.10.0/24", "tcp", "443", "ALLOW"),
        _rule(2, "10.2.0.0/24", "192.168.10.0/24", "tcp", "443", "ALLOW"),
    ]
    conflicts = detect_conflicts(rules)
    real = [c for c in conflicts if c.conflict_type not in ("overpermissive",)]
    assert len(real) == 0
    print("Clean rules: passed")

"""
FIREWALL RULE CONFLICTS IN PRACTICE

Shadowing:
  Most critical bug. An ALLOW * /* rule early in the chain means all
  subsequent DENY rules are dead code.  Common in AWS security groups
  where rules are evaluated "most-permissive wins".

Redundant rules:
  Performance and maintenance burden; not a security risk by itself.

Contradictory rules:
  Often a sign of policy drift — two engineers added conflicting rules.
  The one with higher priority wins; the other is dead code.

Overly-permissive:
  "Allow * on *" is the classic misconfiguration.  Principle of least
  privilege demands explicit allow lists.

AUDITING TOOLS
  - iptables-save | iptables-xml for Linux.
  - AWS Config + Macie rules for SG analysis.
  - Firewall Audit Wizard (Tufin, AlgoSec, FireMon) for enterprise NGFWs.
  - This detector is a simplified version of their core conflict-detection logic.
"""
