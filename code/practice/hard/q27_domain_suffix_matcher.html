"""
=============================================================================
Q27 [Hard] — Build a Domain Suffix Matcher
=============================================================================

PROBLEM
-------
Build a data structure that, given a hostname, efficiently determines:
  1. Whether the hostname belongs to a known domain suffix (e.g. ".google.com")
  2. The longest matching suffix
  3. Public Suffix List (PSL)-style eTLD+1 extraction
  4. Support for wildcard rules: "*.compute.internal" matches any subdomain
  5. Exception rules: !specific.compute.internal → exclude from wildcard

Use cases:
  - Domain allowlist / blocklist enforcement
  - DNS SSRF: detect if a host resolves to a blocked domain family
  - Phishing detection: extract eTLD+1 to compare against known brands
  - Cookie scope enforcement (same-origin checks)

CLARIFYING QUESTIONS
--------------------
1. Case sensitivity?
   → All lower-case; normalise input.
2. Wildcard rules?
   → "*.evil.com" matches "sub.evil.com" but NOT "evil.com" itself.
3. Exception rules?
   → "!allowed.evil.com" exempts that specific domain from a wildcard match.
4. Partial match or full suffix?
   → Must match from the right; "oogle.com" does NOT match "google.com".
5. Max domain depth?
   → Domains can have up to 128 labels; structure handles arbitrary depth.

=============================================================================
APPROACH 1 — NAIVE (set of suffixes + linear scan)
=============================================================================
  Store suffixes in a set; strip leftmost label and check iteratively.
  O(L²) worst case per lookup (L = labels in hostname).

=============================================================================
APPROACH 2 — REVERSED-LABEL TRIE (RECOMMENDED)
=============================================================================
  Store domain labels in a trie, root = TLD, next level = SLD, etc.
  Insert reversed label sequence.
  Lookup: traverse labels from right to left; collect match.
  O(L) insert and lookup (L = number of labels).

=============================================================================
"""

from __future__ import annotations

from dataclasses import dataclass, field
from typing import Any, Dict, List, Optional, Set, Tuple


# ---------------------------------------------------------------------------
# Data Types
# ---------------------------------------------------------------------------

@dataclass
class MatchResult:
    hostname: str
    matched_suffix: str        # longest matching suffix
    tag: Any = None
    is_wildcard_match: bool = False
    is_exception: bool = False
    etld_plus_one: str = ""    # registrable domain (eTLD+1)


# ---------------------------------------------------------------------------
# Helpers
# ---------------------------------------------------------------------------

def _normalise(domain: str) -> str:
    return domain.strip().lower().rstrip(".")


def _labels(domain: str) -> List[str]:
    """Split domain into labels, normalised."""
    return _normalise(domain).split(".")


# ---------------------------------------------------------------------------
# APPROACH 1 — Naive set-based matcher
# ---------------------------------------------------------------------------

class NaiveSuffixMatcher:
    """
    O(L²) per lookup.  No wildcard support.
    """

    def __init__(self) -> None:
        self._suffixes: Dict[str, Any] = {}  # suffix → tag

    def add(self, suffix: str, tag: Any = None) -> None:
        self._suffixes[_normalise(suffix)] = tag

    def match(self, hostname: str) -> Optional[MatchResult]:
        host = _normalise(hostname)
        parts = host.split(".")
        best: Optional[str] = None
        for i in range(len(parts)):
            candidate = ".".join(parts[i:])
            if candidate in self._suffixes:
                if best is None or len(candidate) > len(best):
                    best = candidate
        if best is None:
            return None
        tag = self._suffixes[best]
        labels = host.split(".")
        suffix_parts = best.split(".")
        etld1 = ".".join(labels[-(len(suffix_parts) + 1):]) if len(labels) > len(suffix_parts) else host
        return MatchResult(hostname=host, matched_suffix=best, tag=tag, etld_plus_one=etld1)

    def __len__(self) -> int:
        return len(self._suffixes)


# ---------------------------------------------------------------------------
# APPROACH 2 — Reversed-label trie
# ---------------------------------------------------------------------------

@dataclass
class _TrieNode:
    children: Dict[str, "_TrieNode"] = field(default_factory=dict)
    is_suffix: bool = False       # this node represents a complete suffix
    is_wildcard: bool = False     # wildcard rule: any one label matches
    is_exception: bool = False    # exception rule: exclude this from wildcard parent
    tag: Any = None


class DomainSuffixMatcher:
    """
    Reversed-label trie for O(L) domain suffix matching.
    Supports:
      - Exact suffixes: "google.com"
      - Wildcard: "*.compute.internal"
      - Exceptions: "!allowed.compute.internal"
    """

    def __init__(self) -> None:
        self._root = _TrieNode()
        self._count = 0

    def add(self, suffix: str, tag: Any = None) -> None:
        """
        Add a suffix rule.
        Prefix with "!" to add an exception rule.
        Use "*" as a label to add a wildcard rule (e.g. "*.evil.com").
        """
        is_exception = suffix.startswith("!")
        if is_exception:
            suffix = suffix[1:]
        norm = _normalise(suffix)
        parts = norm.split(".")
        # Reverse labels: TLD first in trie
        node = self._root
        for label in reversed(parts):
            if label not in node.children:
                node.children[label] = _TrieNode()
            node = node.children[label]
        if is_exception:
            node.is_exception = True
        elif "*" in parts:
            # Mark the parent of "*" as wildcard
            node.is_wildcard = True
        else:
            node.is_suffix = True
        node.tag = tag
        self._count += 1

    def match(self, hostname: str) -> Optional[MatchResult]:
        """
        Find longest matching suffix for hostname.
        Returns MatchResult or None.
        """
        host = _normalise(hostname)
        parts = host.split(".")

        node = self._root
        matched_depth = 0
        best_depth = 0
        best_node: Optional[_TrieNode] = None
        is_wildcard_match = False
        is_exception = False

        # Traverse labels from right to left (TLD → subdomain)
        for i, label in enumerate(reversed(parts)):
            # Check for exception rule first
            exc_node = node.children.get(label)
            if exc_node and exc_node.is_exception:
                is_exception = True
                return MatchResult(
                    hostname=host,
                    matched_suffix=".".join(reversed(parts[-i-1:])),
                    tag=exc_node.tag,
                    is_exception=True,
                    etld_plus_one=host,
                )

            # Check exact label match
            next_node = node.children.get(label)
            if next_node:
                matched_depth += 1
                node = next_node
                if node.is_suffix:
                    best_depth = matched_depth
                    best_node = node
                    is_wildcard_match = False
            else:
                # Check wildcard at this level
                wildcard_node = node.children.get("*")
                if wildcard_node:
                    matched_depth += 1
                    best_depth = matched_depth
                    best_node = wildcard_node
                    is_wildcard_match = True
                    node = wildcard_node
                else:
                    break

        if best_node is None:
            return None

        # Build matched suffix from the right
        suffix_parts = parts[-best_depth:]
        matched = ".".join(suffix_parts)

        # eTLD+1: one label to the left of the suffix
        if len(parts) > best_depth:
            etld1 = ".".join(parts[-(best_depth + 1):])
        else:
            etld1 = host

        return MatchResult(
            hostname=host,
            matched_suffix=matched,
            tag=best_node.tag,
            is_wildcard_match=is_wildcard_match,
            is_exception=is_exception,
            etld_plus_one=etld1,
        )

    def contains(self, suffix: str) -> bool:
        return self.match(_normalise(suffix)) is not None

    def __len__(self) -> int:
        return self._count


# ---------------------------------------------------------------------------
# Test Cases
# ---------------------------------------------------------------------------

def test_exact_match():
    m = DomainSuffixMatcher()
    m.add("evil.com", tag="malware")
    r = m.match("sub.evil.com")
    assert r and r.matched_suffix == "evil.com" and r.tag == "malware"
    assert r.etld_plus_one == "sub.evil.com"
    print("Exact match: passed")


def test_longest_match():
    m = DomainSuffixMatcher()
    m.add("com", tag="generic-tld")
    m.add("evil.com", tag="malicious")
    r = m.match("sub.evil.com")
    assert r and r.matched_suffix == "evil.com", \
        f"Expected evil.com; got {r.matched_suffix if r else None}"
    print("Longest match: passed")


def test_no_match():
    m = DomainSuffixMatcher()
    m.add("evil.com")
    r = m.match("google.com")
    assert r is None
    print("No match: passed")


def test_wildcard():
    m = DomainSuffixMatcher()
    m.add("*.compute.internal", tag="internal")
    r = m.match("host1.compute.internal")
    assert r and r.is_wildcard_match and r.tag == "internal"
    # exact "compute.internal" should NOT match wildcard
    r2 = m.match("compute.internal")
    # Wildcard only matches if there is an extra label
    assert r2 is None or not r2.is_wildcard_match
    print("Wildcard: passed")


def test_etld_plus_one():
    m = DomainSuffixMatcher()
    m.add("co.uk", tag="uk-registrar")
    r = m.match("www.brand.co.uk")
    assert r and r.etld_plus_one == "brand.co.uk", \
        f"Expected brand.co.uk; got {r.etld_plus_one if r else None}"
    print("eTLD+1: passed")


def test_naive_match():
    m = NaiveSuffixMatcher()
    m.add("evil.com", tag="malware")
    r = m.match("sub.evil.com")
    assert r and r.matched_suffix == "evil.com"
    print("Naive match: passed")


def test_partial_label_no_match():
    m = DomainSuffixMatcher()
    m.add("evil.com")
    # "notevil.com" should NOT match "evil.com" (label boundary enforced)
    r = m.match("notevil.com")
    assert r is None
    print("Partial label no match: passed")


def test_subdomain_of_suffix():
    m = DomainSuffixMatcher()
    m.add("malware-cdn.net", tag="cdn")
    r1 = m.match("payload.malware-cdn.net")
    r2 = m.match("malware-cdn.net")
    assert r1 and r1.matched_suffix == "malware-cdn.net"
    assert r2 and r2.matched_suffix == "malware-cdn.net"
    print("Subdomain and apex match: passed")

"""
SECURITY USE CASES

1. DNS RPZ (Response Policy Zone)
   Match queried domain against threat-intel domain list.
   Return NXDOMAIN or sinkhole IP for matches.
   DomainSuffixMatcher enables "*.badactor.com" blocking.

2. eTLD+1 Extraction (Cookie Scope / CORS)
   Browsers use the Public Suffix List to determine cookie scope.
   "evil.com" and "app.evil.com" share the eTLD+1 "evil.com".
   Correct eTLD+1 extraction prevents cross-subdomain cookie leakage.

3. Phishing Detection
   Check if eTLD+1 of email "From" domain matches a brand impersonation list.
   E.g. "paypa1.com" → not in brandlist but visually similar (Q28).

4. SSRF via DNS
   An attacker registers "169.254.169.254.attacker.com" → resolves to metadata IP.
   Suffix matching alone doesn't help; must also validate the resolved IP (Q22/Q23).

TRIE ADVANTAGES OVER HASH SET
  - O(L) lookup vs O(L) hash compute — comparable, but trie enables
    longest-prefix matching and wildcard patterns.
  - Memory: shared prefixes stored once (space efficient for many TLD suffixes).
"""
