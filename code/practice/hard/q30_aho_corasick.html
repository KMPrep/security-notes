"""
=============================================================================
Q30 [Hard] — Multi-Pattern String Matching (Aho-Corasick)
=============================================================================

PROBLEM
-------
Build a multi-pattern string matcher for IOC (Indicator of Compromise)
scanning.  Given a large text/bytes buffer and a dictionary of patterns
(malware hashes, domain names, file paths, C2 strings), find all occurrences
of all patterns in one pass.

Requirements:
  1. Index N patterns; find all matches in text of length M.
  2. O(N × L + M + Z) total — L = avg pattern length, Z = match count.
  3. Case-insensitive matching option.
  4. Return (offset, pattern_name, matched_text) per match.
  5. Support adding new patterns after build (rebuild automaton).

CLARIFYING QUESTIONS
--------------------
1. Pattern types?
   → Exact strings only; regex handled in Q31.
2. Overlapping matches?
   → Yes; report all occurrences including overlaps.
3. Byte or text?
   → Both; encode text patterns to bytes.
4. Max patterns and text size?
   → Up to 100K patterns, text up to 100MB; must be streaming-friendly.
5. Unicode?
   → ASCII / Latin-1 focus; Unicode support via UTF-8 bytes.

=============================================================================
APPROACH 1 — NAIVE (O(N × M) linear scan for each pattern)
=============================================================================
  For each pattern: scan full text using str.find() or bytes.find().
  O(N × M) — too slow for large sets.

=============================================================================
APPROACH 2 — AHO-CORASICK AUTOMATON (RECOMMENDED)
=============================================================================
  1. Build a trie from all patterns.
  2. BFS to compute failure links (suffix links) and output links.
  3. Run text through automaton: O(M + Z) after O(sum(|pattern|)) build.
  Total: O(N × L + M + Z).

=============================================================================
"""

from __future__ import annotations

from collections import deque
from dataclasses import dataclass, field
from typing import Any, Dict, Generator, Iterator, List, Optional, Set, Tuple


# ---------------------------------------------------------------------------
# Data Types
# ---------------------------------------------------------------------------

@dataclass
class PatternMatch:
    offset: int           # start offset in text (end - len + 1 is start)
    end_offset: int       # one past last byte
    pattern_id: str       # user-provided pattern name
    matched_text: bytes


# ---------------------------------------------------------------------------
# Helpers
# ---------------------------------------------------------------------------

def _to_bytes(s: Any, case_insensitive: bool = False) -> bytes:
    if isinstance(s, bytes):
        return s.lower() if case_insensitive else s
    return str(s).lower().encode("utf-8") if case_insensitive else str(s).encode("utf-8")


# ---------------------------------------------------------------------------
# APPROACH 1 — Naive
# ---------------------------------------------------------------------------

class NaiveMultiPatternMatcher:
    """O(N × M) per search; simple and correct for small N."""

    def __init__(self, case_insensitive: bool = False) -> None:
        self._patterns: Dict[str, bytes] = {}
        self.case_insensitive = case_insensitive

    def add_pattern(self, pattern_id: str, pattern: Any) -> None:
        self._patterns[pattern_id] = _to_bytes(pattern, self.case_insensitive)

    def search(self, text: Any) -> List[PatternMatch]:
        data = _to_bytes(text, self.case_insensitive)
        results: List[PatternMatch] = []
        for pid, pat in self._patterns.items():
            start = 0
            while True:
                idx = data.find(pat, start)
                if idx == -1:
                    break
                results.append(PatternMatch(
                    offset=idx, end_offset=idx + len(pat),
                    pattern_id=pid, matched_text=pat,
                ))
                start = idx + 1
        results.sort(key=lambda m: m.offset)
        return results

    def __len__(self) -> int:
        return len(self._patterns)


# ---------------------------------------------------------------------------
# APPROACH 2 — Aho-Corasick Automaton
# ---------------------------------------------------------------------------

class _ACNode:
    __slots__ = ("children", "fail", "output")

    def __init__(self) -> None:
        self.children: Dict[int, "_ACNode"] = {}
        self.fail: Optional["_ACNode"] = None
        self.output: List[Tuple[str, bytes]] = []  # (pattern_id, pattern_bytes)


class AhoCorasick:
    """
    Aho-Corasick multi-pattern matcher.
    Build: O(sum of pattern lengths).
    Search: O(M + Z) where M = text length, Z = number of matches.
    """

    def __init__(self, case_insensitive: bool = False) -> None:
        self.case_insensitive = case_insensitive
        self._root = _ACNode()
        self._built = False
        self._patterns: Dict[str, bytes] = {}

    def add_pattern(self, pattern_id: str, pattern: Any) -> None:
        pat = _to_bytes(pattern, self.case_insensitive)
        self._patterns[pattern_id] = pat
        self._built = False
        # Insert into trie
        node = self._root
        for byte in pat:
            if byte not in node.children:
                node.children[byte] = _ACNode()
            node = node.children[byte]
        node.output.append((pattern_id, pat))

    def build(self) -> None:
        """Compute failure links and output links via BFS."""
        self._root.fail = self._root
        q: deque[_ACNode] = deque()

        # Level 1: root's direct children fail back to root
        for child in self._root.children.values():
            child.fail = self._root
            q.append(child)

        while q:
            curr = q.popleft()
            for byte, child in curr.children.items():
                # Failure link for child
                fail = curr.fail
                while fail is not self._root and byte not in fail.children:
                    fail = fail.fail
                child.fail = fail.children.get(byte, self._root)
                if child.fail is child:
                    child.fail = self._root
                # Propagate output from fail link
                child.output = child.output + child.fail.output
                q.append(child)

        self._built = True

    def search(self, text: Any) -> List[PatternMatch]:
        if not self._built:
            self.build()
        data = _to_bytes(text, self.case_insensitive)
        results: List[PatternMatch] = []
        node = self._root

        for i, byte in enumerate(data):
            # Follow failure links until match or root
            while node is not self._root and byte not in node.children:
                node = node.fail
            node = node.children.get(byte, self._root)

            # Collect all outputs at this node
            for pid, pat in node.output:
                start = i - len(pat) + 1
                results.append(PatternMatch(
                    offset=start,
                    end_offset=i + 1,
                    pattern_id=pid,
                    matched_text=pat,
                ))

        results.sort(key=lambda m: (m.offset, m.pattern_id))
        return results

    def search_iter(self, text: Any) -> Iterator[PatternMatch]:
        """Generator version for streaming large files."""
        if not self._built:
            self.build()
        data = _to_bytes(text, self.case_insensitive)
        node = self._root

        for i, byte in enumerate(data):
            while node is not self._root and byte not in node.children:
                node = node.fail
            node = node.children.get(byte, self._root)

            for pid, pat in node.output:
                start = i - len(pat) + 1
                yield PatternMatch(
                    offset=start,
                    end_offset=i + 1,
                    pattern_id=pid,
                    matched_text=pat,
                )

    def __len__(self) -> int:
        return len(self._patterns)

# Test Cases

def test_basic():
    ac = AhoCorasick()
    ac.add_pattern("p1", "he")
    ac.add_pattern("p2", "she")
    ac.add_pattern("p3", "his")
    ac.add_pattern("p4", "hers")
    ac.build()
    text = b"ushers"
    results = ac.search(text)
    pids = {r.pattern_id for r in results}
    assert "p2" in pids, "Expected 'she' match"
    assert "p4" in pids, "Expected 'hers' match"
    assert "p1" in pids, "Expected 'he' match"
    print("Basic Aho-Corasick: passed")


def test_case_insensitive():
    ac = AhoCorasick(case_insensitive=True)
    ac.add_pattern("evil", "EVIL.EXE")
    results = ac.search(b"launching evil.exe process")
    assert results and results[0].pattern_id == "evil"
    print("Case insensitive: passed")


def test_overlapping():
    ac = AhoCorasick()
    ac.add_pattern("a", "ab")
    ac.add_pattern("b", "abc")
    ac.add_pattern("c", "bc")
    text = b"xabcx"
    results = ac.search(text)
    pids = {r.pattern_id for r in results}
    assert {"a", "b", "c"} == pids
    print("Overlapping: passed")


def test_no_match():
    ac = AhoCorasick()
    ac.add_pattern("ioc", "malware.dll")
    results = ac.search(b"clean file contents")
    assert not results
    print("No match: passed")


def test_multiple_occurrences():
    ac = AhoCorasick()
    ac.add_pattern("token", "admin")
    text = b"admin logged in; admin changed password"
    results = ac.search(text)
    assert len(results) == 2
    assert results[0].offset == 0
    assert results[1].offset == 17
    print("Multiple occurrences: passed")


def test_naive_matcher():
    m = NaiveMultiPatternMatcher(case_insensitive=True)
    m.add_pattern("c2", "beacon.callhome.net")
    m.add_pattern("hash", "d41d8cd98f00b204e9800998ecf8427e")
    results = m.search(b"DNS query to beacon.callhome.net resolved")
    assert results and results[0].pattern_id == "c2"
    print("Naive matcher: passed")


def test_ioc_scan():
    """Simulate scanning a memory dump for IOCs."""
    ac = AhoCorasick(case_insensitive=True)
    iocs = {
        "c2_domain": b"evilc2.example.com",
        "registry_key": b"HKEY_LOCAL_MACHINE\\SOFTWARE\\malicious",
        "mutex": b"Global\\m_mutex_1337",
        "user_agent": b"python-requests/2.0",
    }
    for ioc_id, ioc in iocs.items():
        ac.add_pattern(ioc_id, ioc)

    # Simulated memory dump
    dump = (
        b"\x00" * 100 +
        b"GET / HTTP/1.1\r\nUser-Agent: python-requests/2.0\r\n" +
        b"\x00" * 200 +
        b"connecting to evilc2.example.com:443\x00" +
        b"\x00" * 50
    )

    results = ac.search(dump)
    found = {r.pattern_id for r in results}
    assert "c2_domain" in found
    assert "user_agent" in found
    print(f"IOC scan: passed (found: {found})")

"""
AHO-CORASICK IN SECURITY

AV ENGINES
  Aho-Corasick (or variants like Wu-Manber, SBOM) is used by almost every
  antivirus engine to scan files for signature strings in a single pass.
  ClamAV uses it internally for its byte-string signatures.

NETWORK IDS (SNORT/SURICATA)
  Pattern matching in fast_pattern (Snort) and the multi-pattern matcher
  (MPM) in Suricata uses AC + SIMD acceleration (Hyperscan library).

ADVANTAGES OVER REGEX
  - For large pattern sets (10K+ exact strings), AC is far faster than regex.
  - No catastrophic backtracking risk (Q31).
  - Deterministic O(M) scan after O(N*L) build.

LIMITATION
  - Exact string matching only; no fuzzy matching or wildcards.
  - Combine with Q29 (YARA) for hex patterns and regexes.
  - Combine with Q31 for regex-based IOCs.

STREAMING / LARGE FILES
  The `search_iter()` generator can feed bytes in chunks.
  Must track state (current node) between chunks for true streaming.
  This implementation processes the full buffer at once (for simplicity).
"""
