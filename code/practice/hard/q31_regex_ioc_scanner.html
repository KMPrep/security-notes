"""
=============================================================================
Q31 [Hard] — Regex-Based IOC Scanner (ReDoS-Safe)
=============================================================================

PROBLEM
-------
Build an IOC (Indicator of Compromise) scanner that:
  1. Detects IPs, domains, URLs, file hashes (MD5/SHA1/SHA256),
     CVE IDs, email addresses, registry keys, and custom patterns.
  2. Compiles all regex patterns at startup.
  3. Applies ReDoS (Regular Expression Denial of Service) detection
     heuristics to reject unsafe patterns before compilation.
  4. Applies scan timeout to bound worst-case execution time.
  5. Returns structured IOC objects with type, value, and context.

CLARIFYING QUESTIONS
--------------------
1. What types of IOCs?
   → IPv4, domain, URL, MD5, SHA1, SHA256, CVE, email, file path,
     registry key, custom user-supplied patterns.
2. ReDoS prevention?
   → Static analysis heuristics (nested quantifiers, catastrophic patterns).
     Optionally: timeout via threading.
3. Input: file path or raw text?
   → Raw text or bytes; caller loads file.
4. Deduplicate results?
   → Yes; return unique IOC values per type.
5. Normalise output?
   → Yes; lowercase domains, normalise IPs, uppercase hashes.

=============================================================================
APPROACH 1 — NAIVE (compile and apply one regex at a time)
=============================================================================
  Sequential: compile each pattern, scan text, collect results.
  O(P × M) per scan (P = patterns, M = text length).
  No ReDoS detection.

=============================================================================
APPROACH 2 — PRE-COMPILED REGISTRY + REDOS FILTER (RECOMMENDED)
=============================================================================
  1. Define IOC patterns as a registry with metadata.
  2. On load: run ReDoS heuristics; reject unsafe patterns.
  3. Batch patterns into categories; findall per category.
  4. Apply optional timeout via threading for custom patterns.
  O(P × M) but with early rejection of dangerous patterns.

=============================================================================
"""

from __future__ import annotations

import re
import threading
from dataclasses import dataclass, field
from typing import Any, Dict, FrozenSet, Iterator, List, Optional, Set, Tuple

# ---------------------------------------------------------------------------
# ReDoS Heuristics
# ---------------------------------------------------------------------------
# Patterns that indicate potential catastrophic backtracking:
#   (a+)+  or  (a*)*  or  (a|aa)+  or  (a+b+)+

_REDOS_PATTERNS = [
    re.compile(r"\([^)]*[+*][^)]*\)[+*]"),              # (x+)+ or (x*)* style
    re.compile(r"\([^)]*\|[^)]*\)[+*]"),                # (a|b)+ with overlap possible
    re.compile(r"\([^)]*[+*][^)]*\)\{[0-9,]+\}"),       # (x+){n,m}
    re.compile(r"\w[+*]\w[+*].*\)"),                    # nested quantifiers general
]


def is_redos_safe(pattern: str) -> Tuple[bool, str]:
    """
    Return (is_safe, reason).
    Uses simple heuristics; not a complete analysis.
    """
    for hp in _REDOS_PATTERNS:
        if hp.search(pattern):
            return False, f"Potential ReDoS: nested quantifier near '{hp.pattern}'"
    # Reject extremely long patterns
    if len(pattern) > 2000:
        return False, "Pattern too long (> 2000 chars)"
    return True, ""


# ---------------------------------------------------------------------------
# IOC Regex Registry
# ---------------------------------------------------------------------------

@dataclass
class IOCPattern:
    ioc_type: str
    pattern: str
    compiled: Optional[re.Pattern] = field(default=None, repr=False)
    flags: int = re.IGNORECASE
    normalise: Optional[Any] = None   # callable: str → str


def _normalise_domain(v: str) -> str:
    return v.strip().lower().rstrip(".")

def _normalise_ip(v: str) -> str:
    return v.strip()

def _normalise_hash(v: str) -> str:
    return v.strip().upper()

def _normalise_cve(v: str) -> str:
    return v.strip().upper()

def _normalise_url(v: str) -> str:
    return v.strip()

def _normalise_email(v: str) -> str:
    return v.strip().lower()


# Built-in IOC pattern definitions (safe, pre-vetted)
_BUILTIN_PATTERNS: List[IOCPattern] = [
    IOCPattern(
        ioc_type="ipv4",
        pattern=r"\b(?:(?:25[0-5]|2[0-4]\d|[01]?\d\d?)\.){3}(?:25[0-5]|2[0-4]\d|[01]?\d\d?)\b",
        normalise=_normalise_ip,
    ),
    IOCPattern(
        ioc_type="domain",
        pattern=r"\b(?:[a-z0-9](?:[a-z0-9\-]{0,61}[a-z0-9])?\.)+(?:com|net|org|io|gov|edu|co|uk|ru|cn|de|fr|jp|br|in|info|biz|xyz|top|club|online|site|tech)\b",
        normalise=_normalise_domain,
    ),
    IOCPattern(
        ioc_type="url",
        pattern=r"https?://[^\s<>\"']{8,512}",
        normalise=_normalise_url,
        flags=re.IGNORECASE,
    ),
    IOCPattern(
        ioc_type="md5",
        pattern=r"\b[a-f0-9]{32}\b",
        normalise=_normalise_hash,
        flags=re.IGNORECASE,
    ),
    IOCPattern(
        ioc_type="sha1",
        pattern=r"\b[a-f0-9]{40}\b",
        normalise=_normalise_hash,
        flags=re.IGNORECASE,
    ),
    IOCPattern(
        ioc_type="sha256",
        pattern=r"\b[a-f0-9]{64}\b",
        normalise=_normalise_hash,
        flags=re.IGNORECASE,
    ),
    IOCPattern(
        ioc_type="cve",
        pattern=r"\bCVE-\d{4}-\d{4,7}\b",
        normalise=_normalise_cve,
        flags=re.IGNORECASE,
    ),
    IOCPattern(
        ioc_type="email",
        pattern=r"\b[a-z0-9._%+\-]+@[a-z0-9.\-]+\.[a-z]{2,}\b",
        normalise=_normalise_email,
    ),
    IOCPattern(
        ioc_type="windows_path",
        pattern=r"\b[A-Z]:\\(?:[^\\\s<>\":|?*\x00-\x1f]{1,255}\\)*[^\\\s<>\":|?*\x00-\x1f]{1,255}\b",
        flags=re.IGNORECASE,
    ),
    IOCPattern(
        ioc_type="registry_key",
        pattern=r"\b(?:HKEY_LOCAL_MACHINE|HKEY_CURRENT_USER|HKEY_CLASSES_ROOT|HKLM|HKCU)\\[^\s\"]{4,200}\b",
        flags=re.IGNORECASE,
    ),
]


# ---------------------------------------------------------------------------
# Data Types
# ---------------------------------------------------------------------------

@dataclass
class IOC:
    ioc_type: str
    value: str
    raw_value: str
    context: str = ""    # surrounding text snippet


# ---------------------------------------------------------------------------
# Scanner
# ---------------------------------------------------------------------------

class _TimeoutError(Exception):
    pass


def _run_with_timeout(func, args, timeout_s: float):
    """Run func(*args) with a timeout; raise _TimeoutError on expiry."""
    result = [None]
    exc = [None]

    def target():
        try:
            result[0] = func(*args)
        except Exception as e:
            exc[0] = e

    t = threading.Thread(target=target, daemon=True)
    t.start()
    t.join(timeout_s)
    if t.is_alive():
        raise _TimeoutError(f"Pattern scan timed out after {timeout_s}s")
    if exc[0]:
        raise exc[0]
    return result[0]


class IOCScanner:
    """
    Compiled IOC scanner with ReDoS filtering.
    """

    def __init__(
        self,
        include_types: Optional[Set[str]] = None,
        exclude_types: Optional[Set[str]] = None,
        deduplicate: bool = True,
        context_chars: int = 40,
        scan_timeout_s: float = 5.0,
    ) -> None:
        self.deduplicate = deduplicate
        self.context_chars = context_chars
        self.scan_timeout_s = scan_timeout_s
        self._patterns: List[IOCPattern] = []

        for p in _BUILTIN_PATTERNS:
            if include_types and p.ioc_type not in include_types:
                continue
            if exclude_types and p.ioc_type in exclude_types:
                continue
            try:
                p.compiled = re.compile(p.pattern, p.flags)
                self._patterns.append(p)
            except re.error:
                pass  # Skip invalid builtins (shouldn't happen)

    def add_custom_pattern(self, ioc_type: str, pattern: str,
                           flags: int = re.IGNORECASE) -> Tuple[bool, str]:
        """Add a user-supplied pattern; runs ReDoS check first."""
        safe, reason = is_redos_safe(pattern)
        if not safe:
            return False, reason
        try:
            compiled = re.compile(pattern, flags)
        except re.error as e:
            return False, f"Regex compile error: {e}"
        self._patterns.append(IOCPattern(
            ioc_type=ioc_type, pattern=pattern,
            compiled=compiled, flags=flags,
        ))
        return True, "ok"

    def scan(self, text: str) -> List[IOC]:
        results: List[IOC] = []
        seen: Set[Tuple[str, str]] = set()

        for ioc_pat in self._patterns:
            if ioc_pat.compiled is None:
                continue

            def _search(pattern=ioc_pat.compiled, data=text):
                return list(pattern.finditer(data))

            try:
                matches = _run_with_timeout(_search, (), self.scan_timeout_s)
            except _TimeoutError:
                # Pattern timed out; skip it
                continue

            for m in (matches or []):
                raw = m.group(0)
                value = ioc_pat.normalise(raw) if ioc_pat.normalise else raw
                key = (ioc_pat.ioc_type, value)
                if self.deduplicate and key in seen:
                    continue
                seen.add(key)
                # Extract context
                start = max(0, m.start() - self.context_chars)
                end = min(len(text), m.end() + self.context_chars)
                ctx = text[start:end].replace("\n", " ")
                results.append(IOC(
                    ioc_type=ioc_pat.ioc_type, value=value,
                    raw_value=raw, context=ctx,
                ))

        return results

    def scan_by_type(self, text: str) -> Dict[str, List[str]]:
        """Return dict of ioc_type → list of unique values."""
        iocs = self.scan(text)
        result: Dict[str, List[str]] = {}
        for ioc in iocs:
            result.setdefault(ioc.ioc_type, []).append(ioc.value)
        return result


# ---------------------------------------------------------------------------
# APPROACH 1 — Naive (sequential, no ReDoS check, no dedup)
# ---------------------------------------------------------------------------

class NaiveIOCScanner:
    """Simple scanner; no ReDoS checking or deduplication."""

    def __init__(self) -> None:
        self._patterns = [(p.ioc_type, re.compile(p.pattern, p.flags))
                          for p in _BUILTIN_PATTERNS]

    def scan(self, text: str) -> List[IOC]:
        results = []
        for ioc_type, pat in self._patterns:
            for m in pat.finditer(text):
                results.append(IOC(ioc_type=ioc_type, value=m.group(0),
                                   raw_value=m.group(0)))
        return results

def test_ipv4():
    s = IOCScanner(include_types={"ipv4"})
    result = s.scan_by_type(SAMPLE_LOG)
    assert "192.168.1.50" in result.get("ipv4", [])
    print(f"IPv4: passed → {result.get('ipv4', [])}")


def test_url():
    s = IOCScanner(include_types={"url"})

    assert any("evilc2.net" in u for u in result.get("url", []))
    print(f"URL: passed → {result.get('url', [])}")

def test_redos_rejection():
    s = IOCScanner()
    ok, reason = s.add_custom_pattern("test", r"(a+)+")
    assert not ok, "Expected ReDoS pattern to be rejected"
    print(f"ReDoS rejection: passed → {reason}")


def test_custom_pattern():
    s = IOCScanner()
    ok, _ = s.add_custom_pattern("c2_beacon", r"\bbeacon_token=[a-f0-9]{32}\b")
    assert ok
    result = s.scan("beacon_token=deadbeef00112233445566778899aabb")
    assert result and result[0].ioc_type == "c2_beacon"

"""
REDOS ATTACKS
  ReDoS exploits catastrophic backtracking in regex engines.
  Pattern (a+)+ on "aaaaaaaaaaaaaab" causes exponential backtracking.
  Impact: CPU exhaustion; has caused outages at Cloudflare (2019) and Stack
  Overflow (2016).

SAFE REGEX PRACTICES
  1. Avoid nested quantifiers: (x+)+ → linear (anchored) alternatives.
  2. Use possessive quantifiers or atomic groups where available (not in Python re).
  3. Use the 're2' library (linear-time matching) instead of 're' for
     untrusted patterns.
  4. Apply timeouts on all user-supplied regex evaluation.
  5. Validate patterns with heuristics before compilation.

IOC SCANNER CONTEXT
  Threat-intel feeds produce millions of IOC regexes.
  Always sanitise and rate-limit pattern loading.
  Combine with Aho-Corasick (Q30) for exact-string IOCs; reserve regex
  for complex patterns where exact matching is insufficient.
"""
