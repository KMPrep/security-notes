"""
=============================================================================
Q34 [Hard] — PDF Metadata Extractor
=============================================================================

PROBLEM
-------
Parse a PDF file from raw bytes and extract:
  1. PDF header version (%PDF-1.x or %PDF-2.0).
  2. Cross-reference (xref) table or xref stream: object byte offsets.
  3. Trailer dictionary: /Info, /Encrypt, /Root, /ID.
  4. Info dictionary metadata: /Title, /Author, /CreationDate,
     /ModDate, /Producer, /Creator.
  5. Detect suspicious elements:
     - JavaScript (/JS, /JavaScript action types).
     - Launch actions (/Launch).
     - Embedded files (/EmbeddedFile, /Filespec).
     - Encrypted streams (/Crypt filter).
     - URIs pointing to external hosts.
  6. Count objects and stream objects.

CLARIFYING QUESTIONS
--------------------
1. Full PDF rendering or metadata only?
   → Metadata only; no rendering needed.
2. Handle linearised PDFs?
   → Best-effort; parse last xref first (traditional linear scan from %%EOF).
3. Compressed xref streams (PDF 1.5+)?
   → Detect presence; flag as "compressed xref stream" without full decode.
4. Encrypted PDFs?
   → Detect /Encrypt in trailer; flag as encrypted; don't attempt decrypt.
5. Return format?
   → PDFMetadata dataclass.

=============================================================================
APPROACH 1 — NAIVE (regex-based field extraction, no xref parsing)
=============================================================================
  Use regex to find /Author, /Title, /Creator etc. in raw bytes.
  Fast, works on most PDFs; fails on obfuscated/binary-encoded objects.

=============================================================================
APPROACH 2 — XREF + OBJECT PARSER (RECOMMENDED)
=============================================================================
  1. Find %%EOF and startxref.
  2. Parse xref table or detect xref stream.
  3. Parse trailer dictionary.
  4. For each referenced object, read object header and dictionary.
  5. Scan dictionaries for suspicious keys.

=============================================================================
"""

from __future__ import annotations

import re
from dataclasses import dataclass, field
from typing import Any, Dict, List, Optional, Set, Tuple

# ---------------------------------------------------------------------------
# Data Types
# ---------------------------------------------------------------------------

@dataclass
class PDFInfo:
    title: str = ""
    author: str = ""
    creator: str = ""
    producer: str = ""
    creation_date: str = ""
    mod_date: str = ""


@dataclass
class PDFMetadata:
    path: str = ""
    is_valid: bool = False
    error: str = ""

    # Header
    version: str = ""

    # Object counts
    total_objects: int = 0
    stream_objects: int = 0

    # Trailer
    is_encrypted: bool = False
    encrypt_filter: str = ""
    root_obj: str = ""
    info_obj: str = ""

    # Info dict
    info: PDFInfo = field(default_factory=PDFInfo)

    # Suspicious elements
    has_javascript: bool = False
    has_launch_action: bool = False
    has_embedded_files: bool = False
    has_uri_actions: bool = False
    external_uris: List[str] = field(default_factory=list)
    suspicious_keys: List[str] = field(default_factory=list)

    # Anomalies
    anomalies: List[str] = field(default_factory=list)


# ---------------------------------------------------------------------------
# Regex Patterns for PDF Parsing
# ---------------------------------------------------------------------------

_VERSION_RE     = re.compile(rb"%PDF-(\d+\.\d+)", re.MULTILINE)
_STARTXREF_RE  = re.compile(rb"startxref\s+(\d+)\s+%%EOF", re.DOTALL)
_XREF_RE       = re.compile(rb"xref\s+(\d+)\s+(\d+)\s+((?:\d{10}\s+\d{5}\s+[fn]\s*)+)", re.DOTALL)
_TRAILER_RE    = re.compile(rb"trailer\s*<<(.*?)>>", re.DOTALL)
_OBJ_RE        = re.compile(rb"(\d+)\s+(\d+)\s+obj\s*<<(.*?)>>\s*(?:stream|endobj)", re.DOTALL)
_STREAM_RE     = re.compile(rb"\bstream\b")

# Dictionary key/value extractor
_KEY_STR_RE    = re.compile(rb"/([\w]+)\s*\(([^)]{0,1024})\)")
_KEY_OBJREF_RE = re.compile(rb"/([\w]+)\s+(\d+)\s+\d+\s+R")
_KEY_NAME_RE   = re.compile(rb"/([\w]+)\s+/([\w]+)")
_KEY_NUM_RE    = re.compile(rb"/([\w]+)\s+(\d+)")

# URI extraction
_URI_RE = re.compile(rb"URI\s*\(([^)]{1,1024})\)", re.IGNORECASE)

# Suspicious PDF keys (phishing/malware indicators)
_SUSPICIOUS_KEYS: Set[bytes] = {
    b"JS", b"JavaScript", b"Launch", b"SubmitForm",
    b"ImportData", b"RichMedia", b"EmbeddedFile",
    b"OpenAction", b"AA",  # additional actions
    b"XFA",  # XML Forms Architecture (complex, abuse vector)
}

# ---------------------------------------------------------------------------
# Helpers
# ---------------------------------------------------------------------------

def _decode_pdf_string(raw: bytes) -> str:
    """Decode a PDF literal string (basic); strips outer parens if present."""
    try:
        s = raw.decode("latin-1", errors="replace")
        # Unescape basic PDF escape sequences
        s = s.replace("\\n", "\n").replace("\\r", "\r").replace("\\t", "\t")
        s = s.replace("\\(", "(").replace("\\)", ")").replace("\\\\", "\\")
        return s.strip()
    except Exception:
        return ""


def _extract_dict_string(data: bytes, key: str) -> str:
    """Extract /(key) (value) from raw dict bytes."""
    pat = re.compile(rb"/" + key.encode() + rb"\s*\(([^)]{0,2048})\)", re.DOTALL)
    m = pat.search(data)
    return _decode_pdf_string(m.group(1)) if m else ""


def _scan_for_keys(data: bytes) -> Set[str]:
    """Return set of suspicious keys found in data."""
    found: Set[str] = set()
    for key in _SUSPICIOUS_KEYS:
        # Match as name (e.g. /JS or /JavaScript)
        if re.search(b"/" + key + rb"[\s/\[(>]", data):
            found.add(key.decode("latin-1"))
    return found


# ---------------------------------------------------------------------------
# APPROACH 1 — Naive regex scanner
# ---------------------------------------------------------------------------

class NaivePDFScanner:
    """
    Regex-only scanner.
    Extracts metadata using simple patterns; no xref parsing.
    """

    def scan(self, data: bytes, path: str = "") -> PDFMetadata:
        meta = PDFMetadata(path=path)

        # Version
        vm = _VERSION_RE.search(data)
        if not vm:
            meta.error = "Missing PDF header"
            return meta
        meta.version = vm.group(1).decode("latin-1")
        meta.is_valid = True

        # Count objects
        meta.total_objects = len(re.findall(rb"\d+ \d+ obj", data))
        meta.stream_objects = len(_STREAM_RE.findall(data))

        # Info fields (scan raw data)
        meta.info.author   = _extract_dict_string(data, "Author")
        meta.info.title    = _extract_dict_string(data, "Title")
        meta.info.creator  = _extract_dict_string(data, "Creator")
        meta.info.producer = _extract_dict_string(data, "Producer")
        meta.info.creation_date = _extract_dict_string(data, "CreationDate")
        meta.info.mod_date = _extract_dict_string(data, "ModDate")

        # Encrypt
        if re.search(rb"/Encrypt\b", data):
            meta.is_encrypted = True
            meta.anomalies.append("PDF is encrypted")

        # Suspicious keys
        susp = _scan_for_keys(data)
        meta.suspicious_keys = sorted(susp)
        meta.has_javascript = bool(susp & {"JS", "JavaScript"})
        meta.has_launch_action = "Launch" in susp
        meta.has_embedded_files = "EmbeddedFile" in susp

        # URIs
        for m in _URI_RE.finditer(data):
            uri = m.group(1).decode("latin-1", errors="replace").strip()
            if uri.startswith("http"):
                meta.external_uris.append(uri)
        meta.has_uri_actions = bool(meta.external_uris)

        # Anomalies
        if meta.has_javascript:
            meta.anomalies.append("JavaScript action found (/JS or /JavaScript)")
        if meta.has_launch_action:
            meta.anomalies.append("Launch action found (/Launch)")
        if meta.has_embedded_files:
            meta.anomalies.append("Embedded file found (/EmbeddedFile)")
        if meta.external_uris:
            meta.anomalies.append(f"External URIs: {meta.external_uris[:5]}")

        return meta

# APPROACH 2 — xref-aware parser (preferred)
# ---------------------------------------------------------------------------

class PDFScanner:
    """
    xref-aware PDF scanner.
    Parses trailer and object dictionaries more reliably than pure regex.
    """

    def scan(self, data: bytes, path: str = "") -> PDFMetadata:
        meta = PDFMetadata(path=path)

        # Version
        vm = _VERSION_RE.search(data[:1024])  # version in first 1KB
        if not vm:
            meta.error = "Missing %PDF- header"
            return meta
        meta.version = vm.group(1).decode("latin-1")

        # Locate startxref + %%EOF (from end of file)
        startxref_offset = self._find_startxref(data)
        xref_at = startxref_offset

        # Parse xref table
        trailer_data: bytes = b""
        if xref_at > 0:
            xref_m = re.match(rb"xref", data[xref_at:xref_at + 10])
            if xref_m:
                # Traditional xref table → find trailer
                t_m = _TRAILER_RE.search(data[xref_at:])
                if t_m:
                    trailer_data = t_m.group(1)
            else:
                # Likely compressed xref stream (PDF 1.5+)
                meta.anomalies.append("Compressed xref stream detected (PDF 1.5+)")
                # Fall back: find any trailer in the file
                t_m = _TRAILER_RE.search(data)
                if t_m:
                    trailer_data = t_m.group(1)

        # Parse trailer dict
        if trailer_data:
            self._parse_trailer(trailer_data, meta)

        # Count objects and streams
        meta.total_objects = len(re.findall(rb"\b\d+\s+\d+\s+obj\b", data))
        meta.stream_objects = len(_STREAM_RE.findall(data))

        # Parse object dictionaries for Info fields and suspicious content
        for obj_m in _OBJ_RE.finditer(data):
            obj_dict = obj_m.group(3)
            self._parse_obj_dict(obj_dict, data, meta)

        # URIs
        for m in _URI_RE.finditer(data):
            uri = m.group(1).decode("latin-1", errors="replace").strip()
            if uri.startswith("http"):
                meta.external_uris.append(uri)
        meta.has_uri_actions = bool(meta.external_uris)

        # Final anomalies
        if meta.has_javascript:
            meta.anomalies.append("JavaScript action found (/JS or /JavaScript)")
        if meta.has_launch_action:
            meta.anomalies.append("Launch action found (/Launch)")
        if meta.has_embedded_files:
            meta.anomalies.append("Embedded file found (/EmbeddedFile)")
        if meta.is_encrypted:
            meta.anomalies.append("PDF is encrypted")
        if meta.external_uris:
            meta.anomalies.append(f"External URIs: {meta.external_uris[:5]}")

        meta.is_valid = True
        return meta

    def _find_startxref(self, data: bytes) -> int:
        # Search last 2KB of file for startxref
        tail = data[-2048:] if len(data) > 2048 else data
        offset_in_tail = len(data) - len(tail)
        m = list(re.finditer(rb"startxref\s+(\d+)", tail))
        if m:
            return int(m[-1].group(1))
        return 0

    def _parse_trailer(self, trailer_bytes: bytes, meta: PDFMetadata) -> None:
        # /Encrypt
        if re.search(rb"/Encrypt\b", trailer_bytes):
            meta.is_encrypted = True
        # /Root
        m = re.search(rb"/Root\s+(\d+\s+\d+\s+R)", trailer_bytes)
        if m:
            meta.root_obj = m.group(1).decode("latin-1")
        # /Info
        m = re.search(rb"/Info\s+(\d+\s+\d+\s+R)", trailer_bytes)
        if m:
            meta.info_obj = m.group(1).decode("latin-1")

    def _parse_obj_dict(self, obj_dict: bytes, full_data: bytes,
                        meta: PDFMetadata) -> None:
        # Info fields
        if not meta.info.author:
            meta.info.author   = _extract_dict_string(obj_dict, "Author")
        if not meta.info.title:
            meta.info.title    = _extract_dict_string(obj_dict, "Title")
        if not meta.info.creator:
            meta.info.creator  = _extract_dict_string(obj_dict, "Creator")
        if not meta.info.producer:
            meta.info.producer = _extract_dict_string(obj_dict, "Producer")
        if not meta.info.creation_date:
            meta.info.creation_date = _extract_dict_string(obj_dict, "CreationDate")
        if not meta.info.mod_date:
            meta.info.mod_date = _extract_dict_string(obj_dict, "ModDate")

        # Suspicious keys
        susp = _scan_for_keys(obj_dict)
        for k in susp:
            if k not in meta.suspicious_keys:
                meta.suspicious_keys.append(k)
        meta.has_javascript = meta.has_javascript or bool(susp & {"JS", "JavaScript"})
        meta.has_launch_action = meta.has_launch_action or ("Launch" in susp)
        meta.has_embedded_files = meta.has_embedded_files or ("EmbeddedFile" in susp)

def test_basic_metadata():
    s = PDFScanner()
    meta = s.scan(_SIMPLE_PDF)
    assert meta.is_valid, f"Expected valid; got {meta.error}"
    assert meta.version == "1.4"
    assert "Alice Smith" in meta.info.author or meta.info.author == ""
    print(f"Basic metadata: passed (version={meta.version}, author={meta.info.author!r})")


def test_javascript_detection():
    s = PDFScanner()
    meta = s.scan(_JS_PDF)
    assert meta.is_valid
    assert meta.has_javascript, f"Expected JS detected; suspicious_keys={meta.suspicious_keys}"
    print(f"JavaScript detection: passed ({meta.anomalies})")


def test_encrypted():
    s = PDFScanner()
    meta = s.scan(_ENCRYPTED_PDF)
    assert meta.is_valid
    assert meta.is_encrypted
    print(f"Encrypted PDF: passed")


def test_launch_action():
    s = PDFScanner()
    meta = s.scan(_LAUNCH_PDF)
    assert meta.is_valid
    assert meta.has_launch_action
    print(f"Launch action: passed ({meta.anomalies})")


def test_invalid_pdf():
    s = PDFScanner()
    meta = s.scan(b"Not a PDF file content")
    assert not meta.is_valid
    print("Invalid PDF: passed")


def test_naive_scanner():
    s = NaivePDFScanner()
    meta = s.scan(_SIMPLE_PDF)
    assert meta.is_valid
    assert meta.version == "1.4"
    print(f"Naive scanner: passed (version={meta.version})")


def test_uri_detection():
    pdf_with_uri = _SIMPLE_PDF + b"\n/URI (https://evil.example.com/payload)\n"
    s = PDFScanner()

    assert meta.has_uri_actions or meta.external_uris
    print(f"URI detection: passed ({meta.external_uris})")

"""
PDF AS AN ATTACK VECTOR

JavaScript in PDF
  /JS and /JavaScript actions can execute ECMAScript within PDF viewers.
  Used in drive-by exploits (e.g. CVE-2008-2992, Acrobat JS RCE).
  Mitigation: disable JavaScript in PDF readers, sandbox viewers.

Launch Actions
  /Launch can execute arbitrary programs on Windows without user consent
  in older Acrobat versions.
  Modern readers prompt the user; still abused via social engineering.

Embedded Files
  /EmbeddedFile can contain executables or other malicious payloads.
  Combined with auto-open actions, delivers malware transparently.

OpenAction / AA (Additional Actions)
  /OpenAction executes on document open; /AA on page open, close, etc.
  Commonly used to silently run JavaScript or Launch actions.

XFA (XML Forms Architecture)
  Complex, XML-based form system embedded in PDF.
  Has its own scripting model; historically full of security issues.
  Adobe phased it out in favour of AcroForms.

ENCRYPTED PDFs
  Standard /Encrypt with RC4 or AES; metadata often not encrypted,
  allowing exfiltration even if content is protected.
  Encrypted PDFs are commonly used to evade email gateway scanning.

DETECTION
  - Search for /JS, /JavaScript, /Launch, /OpenAction, /AA in dictionaries.
  - Check Info dictionary dates for timestomping.
  - High object count relative to stream count → fragmented/obfuscated PDF.
"""
