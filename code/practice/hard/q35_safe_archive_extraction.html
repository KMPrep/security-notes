"""
=============================================================================
Q35 [Hard] — Safe Archive Extraction (Zip-Slip Prevention)
=============================================================================

PROBLEM
-------
Write code to extract ZIP (and tar.gz) archives safely, preventing:
  1. Zip-Slip: path traversal via entries like "../../etc/passwd".
  2. Zip bomb: excessive decompressed size or entry count.
  3. Absolute paths: entries starting with "/" or a drive letter.
  4. Symlink attacks: symlinks pointing outside the target directory.
  5. Executable bit preservation that could create attack surface.

Also implement a dry-run mode that validates without extracting.

CLARIFYING QUESTIONS
--------------------
1. ZIP only or also tar/tar.gz?
   → Both; common in malware delivery and supply-chain attacks.
2. Symlinks allowed?
   → Only symlinks staying within the target directory; external → reject.
3. Maximum decompressed size?
   → Configurable; default 500 MB.
4. Maximum entry count?
   → Configurable; default 10,000 entries.
5. What to do on violation?
   → Raise ExtractionError with detail; don't partially extract.

=============================================================================
APPROACH 1 — NAIVE (zipfile.extractall without validation)
=============================================================================
  zipfile.ZipFile.extractall() is inherently unsafe; some Python versions
  strip ".." but absolute paths are implementation-dependent.
  Never use without validation.

=============================================================================
APPROACH 2 — SAFE EXTRACTION WITH ENTRY VALIDATION (RECOMMENDED)
=============================================================================
  1. Open archive; iterate entries before extracting anything.
  2. Resolve target path; verify it is inside the destination directory.
  3. Check total size against bomb limit.
  4. For symlinks: verify symlink target is also inside destination.
  5. Extract one entry at a time; abort and clean up on any violation.

=============================================================================
"""

from __future__ import annotations

import os
import shutil
import tarfile
import tempfile
import zipfile
from dataclasses import dataclass, field
from typing import List, Optional


# ---------------------------------------------------------------------------
# Data Types
# ---------------------------------------------------------------------------

@dataclass
class ExtractionSummary:
    archive_path: str
    destination: str
    entries_extracted: int = 0
    total_bytes: int = 0
    rejected_entries: List[str] = field(default_factory=list)
    warnings: List[str] = field(default_factory=list)


class ExtractionError(Exception):
    pass


# ---------------------------------------------------------------------------
# Validation Helper
# ---------------------------------------------------------------------------

def _safe_path(base_dir: str, member_path: str) -> Optional[str]:
    """
    Resolve member_path relative to base_dir.
    Returns the absolute resolved path if it is inside base_dir, else None.
    """
    # Normalise base
    base = os.path.realpath(base_dir)
    # Strip leading slashes and drive letters from member path
    # (handles absolute entries like "/etc/passwd" or "C:\\evil")
    sanitised = os.path.normpath(member_path)
    # Remove any leading separator after normalisation
    while sanitised.startswith(os.sep) or (len(sanitised) > 1 and sanitised[1] == ":"):
        sanitised = sanitised.lstrip(os.sep)
        if len(sanitised) > 1 and sanitised[1] == ":":
            sanitised = sanitised[2:].lstrip(os.sep)

    target = os.path.realpath(os.path.join(base, sanitised))
    if not target.startswith(base + os.sep) and target != base:
        return None
    return target


# ---------------------------------------------------------------------------
# APPROACH 1 — Naive (unsafe, for illustration only)
# ---------------------------------------------------------------------------

def extract_zip_naive(archive_path: str, dest: str) -> None:
    """
    UNSAFE: zipfile.extractall() without entry validation.
    Vulnerable to zip-slip on older Python versions and absolute paths.
    NEVER USE IN PRODUCTION.
    """
    with zipfile.ZipFile(archive_path, "r") as zf:
        zf.extractall(dest)


# ---------------------------------------------------------------------------
# APPROACH 2 — Safe ZIP extractor
# ---------------------------------------------------------------------------

def extract_zip_safe(
    archive_path: str,
    dest: str,
    max_decompressed_bytes: int = 500 * 1024 * 1024,  # 500 MB
    max_entries: int = 10_000,
    max_ratio: float = 100.0,
    dry_run: bool = False,
) -> ExtractionSummary:
    """
    Safe ZIP extraction.
    Validates all entries before extracting any.
    Raises ExtractionError on policy violation.
    """
    summary = ExtractionSummary(archive_path=archive_path, destination=dest)

    if not os.path.isfile(archive_path):
        raise ExtractionError(f"Archive not found: {archive_path}")

    dest = os.path.realpath(dest)

    with zipfile.ZipFile(archive_path, "r") as zf:
        infos = zf.infolist()

        if len(infos) > max_entries:
            raise ExtractionError(
                f"Archive has {len(infos)} entries, exceeds max {max_entries}"
            )

        total_uncompressed = sum(e.file_size for e in infos)
        total_compressed   = sum(e.compress_size for e in infos) or 1
        ratio = total_uncompressed / total_compressed

        if total_uncompressed > max_decompressed_bytes:
            raise ExtractionError(
                f"Total decompressed size {total_uncompressed} exceeds "
                f"limit {max_decompressed_bytes} (zip bomb?)"
            )

        if ratio > max_ratio:
            raise ExtractionError(
                f"Compression ratio {ratio:.1f}:1 exceeds limit {max_ratio}:1 "
                f"(zip bomb?)"
            )

        # Validate each entry path
        for info in infos:
            safe = _safe_path(dest, info.filename)
            if safe is None:
                raise ExtractionError(
                    f"Path traversal detected: {info.filename!r}"
                )
            # Detect symlinks (external_attr on Unix stores file mode)
            if info.create_system == 3:  # Unix
                mode = (info.external_attr >> 16) & 0xFFFF
                if (mode & 0xA000) == 0xA000:  # S_ISLNK
                    # We'd need to read the link target; flag for review
                    summary.warnings.append(
                        f"Symlink entry detected: {info.filename!r} — skipping"
                    )
                    summary.rejected_entries.append(info.filename)
                    continue

        if dry_run:
            summary.entries_extracted = len(infos) - len(summary.rejected_entries)
            summary.total_bytes = total_uncompressed
            return summary

        # Create destination
        os.makedirs(dest, exist_ok=True)

        # Extract validated entries
        rejected = set(summary.rejected_entries)
        for info in infos:
            if info.filename in rejected:
                continue
            safe = _safe_path(dest, info.filename)
            if safe is None:
                continue  # Already validated above; defensive
            # Directory entry
            if info.filename.endswith("/"):
                os.makedirs(safe, exist_ok=True)
                continue
            os.makedirs(os.path.dirname(safe), exist_ok=True)
            with zf.open(info) as src, open(safe, "wb") as dst:
                shutil.copyfileobj(src, dst)
            summary.entries_extracted += 1
            summary.total_bytes += info.file_size

    return summary


# ---------------------------------------------------------------------------
# APPROACH 2b — Safe tar.gz extractor
# ---------------------------------------------------------------------------

def extract_tar_safe(
    archive_path: str,
    dest: str,
    max_decompressed_bytes: int = 500 * 1024 * 1024,
    max_entries: int = 10_000,
    dry_run: bool = False,
) -> ExtractionSummary:
    """
    Safe tar/tar.gz extraction.
    Rejects absolute paths, path traversal, devices, and symlinks.
    """
    summary = ExtractionSummary(archive_path=archive_path, destination=dest)
    dest = os.path.realpath(dest)

    with tarfile.open(archive_path, "r:*") as tf:
        members = tf.getmembers()

        if len(members) > max_entries:
            raise ExtractionError(
                f"Archive has {len(members)} entries, exceeds max {max_entries}"
            )

        total_size = sum(m.size for m in members if m.isfile())
        if total_size > max_decompressed_bytes:
            raise ExtractionError(
                f"Total decompressed size {total_size} exceeds limit {max_decompressed_bytes}"
            )

        safe_members = []
        for member in members:
            # Reject absolute
            if os.path.isabs(member.name):
                raise ExtractionError(f"Absolute path in archive: {member.name!r}")
            # Reject traversal
            safe = _safe_path(dest, member.name)
            if safe is None:
                raise ExtractionError(f"Path traversal: {member.name!r}")
            # Reject device files
            if member.isdev():
                raise ExtractionError(f"Device file in archive: {member.name!r}")
            # Warn on symlinks
            if member.issym() or member.islnk():
                link_target = os.path.realpath(
                    os.path.join(os.path.dirname(safe), member.linkname)
                )
                if not link_target.startswith(dest):
                    raise ExtractionError(
                        f"Symlink {member.name!r} → {member.linkname!r} escapes dest"
                    )
            safe_members.append(member)

        if dry_run:
            summary.entries_extracted = len(safe_members)
            summary.total_bytes = total_size
            return summary

        os.makedirs(dest, exist_ok=True)
        for member in safe_members:
            tf.extract(member, path=dest, set_attrs=False)
            summary.entries_extracted += 1
            summary.total_bytes += member.size

    return summary

def _make_zip(entries: dict, out_path: str) -> None:
    """entries: {name: content_bytes}"""
    with zipfile.ZipFile(out_path, "w", zipfile.ZIP_DEFLATED) as zf:
        for name, data in entries.items():
            zf.writestr(name, data)


def test_normal_extraction():
    with tempfile.TemporaryDirectory() as tmp:
        arc = os.path.join(tmp, "test.zip")
        dest = os.path.join(tmp, "out")
        _make_zip({
            "readme.txt": b"hello",
            "subdir/config.json": b'{"key":"value"}',
        }, arc)
        s = extract_zip_safe(arc, dest)
        assert s.entries_extracted == 2
        assert os.path.isfile(os.path.join(dest, "readme.txt"))
        print("Normal extraction: passed")


def test_path_traversal_rejected():
    with tempfile.TemporaryDirectory() as tmp:
        arc = os.path.join(tmp, "evil.zip")
        dest = os.path.join(tmp, "out")
        _make_zip({"../../evil.txt": b"pwned"}, arc)
        try:
            extract_zip_safe(arc, dest)
            assert False, "Expected ExtractionError"
        except ExtractionError as e:
            assert "traversal" in str(e).lower() or "Path traversal" in str(e)
            print(f"Path traversal rejected: passed ({e})")


def test_zip_bomb_size():
    with tempfile.TemporaryDirectory() as tmp:
        arc = os.path.join(tmp, "bomb.zip")
        dest = os.path.join(tmp, "out")
        # Create a highly compressible entry that exceeds limit
        _make_zip({"zeros.bin": b"\x00" * (2 * 1024 * 1024)}, arc)
        try:
            extract_zip_safe(arc, dest, max_decompressed_bytes=1024)
            assert False, "Expected ExtractionError"
        except ExtractionError as e:
            assert "zip bomb" in str(e).lower() or "decompressed size" in str(e).lower() or "exceeds" in str(e).lower()
            print(f"Zip bomb size rejected: passed ({e})")


def test_too_many_entries():
    with tempfile.TemporaryDirectory() as tmp:
        arc = os.path.join(tmp, "many.zip")
        dest = os.path.join(tmp, "out")
        entries = {f"file{i}.txt": b"x" for i in range(50)}
        _make_zip(entries, arc)
        try:
            extract_zip_safe(arc, dest, max_entries=10)
            assert False, "Expected ExtractionError"
        except ExtractionError as e:
            assert "entries" in str(e)
            print(f"Too many entries rejected: passed ({e})")


def test_dry_run():
    with tempfile.TemporaryDirectory() as tmp:
        arc = os.path.join(tmp, "test.zip")
        dest = os.path.join(tmp, "out")
        _make_zip({"a.txt": b"data"}, arc)
        s = extract_zip_safe(arc, dest, dry_run=True)
        assert s.entries_extracted == 1
        assert not os.path.isdir(dest), "Dry run should not create files"
        print("Dry run: passed")


# ---------------------------------------------------------------------------
# SECURITY ANGLE
# ---------------------------------------------------------------------------
"""
ZIP-SLIP (CVE-2018-1002220 and related)
  A zip/tar archive contains an entry like:
    "../../../../etc/cron.d/evil"
  When extracted naively, this writes outside the destination directory.
  Affected: Java's ZipInputStream, Ruby gems, npm packages, Go archives.
  Fix: Resolve the absolute target path; verify it starts with dest/.

ZIP BOMB
  A ZIP entry stores ~50 bytes compressed that expand to 4 GB
  (e.g. all-zero content; deflate ratio ~1000:1).
  Nested zips (e.g. 42.zip) exponentially amplify this.
  Fix: Check total_uncompressed before extraction; limit ratio.

SYMLINK ATTACK
  A symlink entry points to /etc or C:\\Windows.
  Subsequent extraction of a real file into that symlink overwrites
  the target on the host.
  Fix: Validate symlink targets are inside dest; or reject symlinks entirely.

SAFE ALTERNATIVES
  Python's tarfile.extractall() since Python 3.12 has a filter=
  parameter (filter="data" is the safest).
  Always use it: tarfile.open(...).extractall(path, filter="data").
"""
