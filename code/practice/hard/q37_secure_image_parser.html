"""
=============================================================================
Q37 [Hard] — Secure Image Parser Design
=============================================================================

PROBLEM
-------
Implement safe parsing for a simplified image format with:
  - Magic bytes header (4 bytes)
  - Width and height fields (2 bytes each, unsigned)
  - Bit-depth field (1 byte)
  - Payload length field (4 bytes, unsigned)
  - Pixel payload (payload_length bytes)

Security requirements:
  1. Validate magic bytes before any allocation.
  2. Reject images exceeding MAX_DIMENSION or MAX_PIXELS.
  3. Verify payload_length == width * height * bytes_per_pixel.
  4. Reject if declared payload_length does not match actual remaining bytes.
  5. Never use eval() or exec() for parsing.
  6. Reject negative or zero dimensions.

CLARIFYING QUESTIONS
--------------------
1. What is the format specification?
   → Simplified: MAGIC(4) | W(2) | H(2) | BIT_DEPTH(1) | PAYLOAD_LEN(4) | DATA
2. Supported bit depths?
   → 8 (grayscale), 24 (RGB), 32 (RGBA).
3. Maximum allowed image size?
   → Configurable; default 16384x16384.
4. What happens on malformed input?
   → Raise ImageParseError with specific reason.
5. Streaming or in-memory?
   → In-memory bytes for simplicity; streaming discussed as extension.

=============================================================================
APPROACH 1 — NAIVE (index slicing without bounds checks)
=============================================================================
  Direct slicing: width = data[4:6].  No length checks; IndexError possible.
  Trusted payload_length from header without cross-validation.
  Allocates width*height array before verifying data is present.

=============================================================================
APPROACH 2 — SAFE BOUNDED PARSER WITH _Buf WRAPPER (OPTIMAL)
=============================================================================
  1. _Buf class tracks position; raises on any over-read.
  2. Validate magic first; fail fast.
  3. Cross-validate: payload_length == W * H * (bit_depth // 8).
  4. Verify remaining bytes match declared length before allocating.
  5. Unpack struct fields with explicit format strings.

=============================================================================
"""

from __future__ import annotations

import struct
from dataclasses import dataclass
from typing import Optional


# ---------------------------------------------------------------------------
# Constants
# ---------------------------------------------------------------------------

MAGIC           = b"\x89IMG"
MAX_DIMENSION   = 16_384          # max width or height
MAX_PIXELS      = 64_000_000      # max width * height
VALID_BIT_DEPTH = frozenset([8, 24, 32])


# ---------------------------------------------------------------------------
# Data Types
# ---------------------------------------------------------------------------

class ImageParseError(Exception):
    pass


@dataclass
class ImageHeader:
    width: int
    height: int
    bit_depth: int
    bytes_per_pixel: int
    payload_length: int


@dataclass
class ParsedImage:
    header: ImageHeader
    pixels: bytes    # raw pixel bytes


# ---------------------------------------------------------------------------
# Safe buffer wrapper (reused pattern from Q32/Q33)
# ---------------------------------------------------------------------------

class _Buf:
    __slots__ = ("_data", "_pos", "_size")

    def __init__(self, data: bytes):
        self._data = data
        self._pos  = 0
        self._size = len(data)

    @property
    def remaining(self) -> int:
        return self._size - self._pos

    def read(self, n: int) -> bytes:
        if n < 0:
            raise ImageParseError("Cannot read negative bytes")
        if n > self.remaining:
            raise ImageParseError(
                f"Buffer overread: need {n} bytes at offset {self._pos}, "
                f"have {self.remaining}"
            )
        chunk = self._data[self._pos : self._pos + n]
        self._pos += n
        return chunk

    def unpack(self, fmt: str) -> tuple:
        size = struct.calcsize(fmt)
        raw  = self.read(size)
        return struct.unpack(fmt, raw)

    def peek(self, n: int) -> bytes:
        if n > self.remaining:
            raise ImageParseError(f"Buffer peek overread: need {n}, have {self.remaining}")
        return self._data[self._pos : self._pos + n]


# ---------------------------------------------------------------------------
# APPROACH 1 — Naive (unsafe)
# ---------------------------------------------------------------------------

class NaiveImageParser:
    """Vulnerable to buffer over-read, integer overflow, and allocation bombs."""

    def parse(self, data: bytes) -> ParsedImage:
        # No magic check
        width   = struct.unpack_from(">H", data, 4)[0]
        height  = struct.unpack_from(">H", data, 6)[0]
        bdepth  = data[8]
        paylen  = struct.unpack_from(">I", data, 9)[0]
        # Trusts paylen; could allocate huge buffer
        pixels  = data[13 : 13 + paylen]
        header  = ImageHeader(width, height, bdepth, bdepth // 8, paylen)
        return ParsedImage(header=header, pixels=pixels)


# ---------------------------------------------------------------------------
# APPROACH 2 — Safe image parser
# ---------------------------------------------------------------------------

class ImageParser:
    """
    Secure image parser.
    Validates before any allocation; raises ImageParseError on all anomalies.
    """

    def __init__(
        self,
        max_dimension: int = MAX_DIMENSION,
        max_pixels: int = MAX_PIXELS,
    ):
        self._max_dim    = max_dimension
        self._max_pixels = max_pixels

    def parse(self, data: bytes) -> ParsedImage:
        if not isinstance(data, (bytes, bytearray)):
            raise ImageParseError("Input must be bytes")

        buf = _Buf(data)

        # 1. Magic check (fast fail)
        magic = buf.read(4)
        if magic != MAGIC:
            raise ImageParseError(
                f"Bad magic: expected {MAGIC!r}, got {magic!r}"
            )

        # 2. Dimensions
        (width,)  = buf.unpack(">H")   # 2 bytes unsigned
        (height,) = buf.unpack(">H")   # 2 bytes unsigned

        if width == 0 or height == 0:
            raise ImageParseError(
                f"Zero dimension: width={width}, height={height}"
            )
        if width > self._max_dim or height > self._max_dim:
            raise ImageParseError(
                f"Dimension exceeds limit: {width}x{height} > {self._max_dim}"
            )
        pixel_count = width * height
        if pixel_count > self._max_pixels:
            raise ImageParseError(
                f"Pixel count {pixel_count:,} exceeds max {self._max_pixels:,}"
            )

        # 3. Bit depth
        (bit_depth,) = buf.unpack("B")  # 1 byte
        if bit_depth not in VALID_BIT_DEPTH:
            raise ImageParseError(
                f"Invalid bit depth: {bit_depth}. Expected one of {sorted(VALID_BIT_DEPTH)}"
            )
        bytes_per_pixel = bit_depth // 8

        # 4. Declared payload length
        (declared_len,) = buf.unpack(">I")   # 4 bytes unsigned
        expected_len    = pixel_count * bytes_per_pixel

        if declared_len != expected_len:
            raise ImageParseError(
                f"Payload length mismatch: declared={declared_len}, "
                f"expected={expected_len} ({width}x{height}x{bytes_per_pixel})"
            )

        # 5. Verify actual bytes available before allocating
        if buf.remaining < declared_len:
            raise ImageParseError(
                f"Truncated payload: have {buf.remaining} bytes, "
                f"need {declared_len}"
            )
        if buf.remaining > declared_len:
            raise ImageParseError(
                f"Trailing garbage: {buf.remaining - declared_len} extra bytes "
                f"after payload"
            )

        # 6. Read pixel data
        pixels = buf.read(declared_len)

        header = ImageHeader(
            width=width,
            height=height,
            bit_depth=bit_depth,
            bytes_per_pixel=bytes_per_pixel,
            payload_length=declared_len,
        )
        return ParsedImage(header=header, pixels=pixels)


# ---------------------------------------------------------------------------
# Builder helper for tests
# ---------------------------------------------------------------------------

def _build_image(
    width: int,
    height: int,
    bit_depth: int = 24,
    payload_override: Optional[bytes] = None,
    magic_override: Optional[bytes] = None,
    paylen_override: Optional[int] = None,
) -> bytes:
    magic   = magic_override if magic_override is not None else MAGIC
    bpp     = bit_depth // 8
    payload = payload_override if payload_override is not None else (
        bytes(range(256)) * ((width * height * bpp + 255) // 256)
    )[:width * height * bpp]
    paylen  = paylen_override if paylen_override is not None else len(payload)
    return (
        magic
        + struct.pack(">HHB", width, height, bit_depth)
        + struct.pack(">I", paylen)
        + payload
    )


# ---------------------------------------------------------------------------
# Test Cases
# ---------------------------------------------------------------------------

def test_valid_image():
    parser = ImageParser()
    data = _build_image(4, 3, bit_depth=24)
    img = parser.parse(data)
    assert img.header.width == 4
    assert img.header.height == 3
    assert img.header.bit_depth == 24
    assert len(img.pixels) == 4 * 3 * 3
    print("Valid 24-bit image: passed")


def test_bad_magic():
    parser = ImageParser()
    data = _build_image(4, 4)
    bad  = b"\x00BAD" + data[4:]
    try:
        parser.parse(bad)
        assert False
    except ImageParseError as e:
        assert "magic" in str(e).lower()
        print(f"Bad magic rejected: passed ({e})")


def test_zero_dimension():
    parser = ImageParser()
    data = _build_image(0, 100)
    try:
        parser.parse(data)
        assert False
    except ImageParseError as e:
        assert "dimension" in str(e).lower() or "zero" in str(e).lower()
        print(f"Zero dimension rejected: passed ({e})")


def test_payload_mismatch():
    parser = ImageParser()
    # Claim paylen of 100 but actual = 4*4*3 = 48
    data = _build_image(4, 4, paylen_override=100)
    try:
        parser.parse(data)
        assert False
    except ImageParseError as e:
        assert "mismatch" in str(e).lower() or "payload" in str(e).lower()
        print(f"Payload length mismatch: passed ({e})")


def test_truncated_payload():
    parser = ImageParser()
    data = _build_image(10, 10, bit_depth=24)
    # Cut off last 50 bytes
    truncated = data[:-50]
    try:
        parser.parse(truncated)
        assert False
    except ImageParseError as e:
        assert "truncated" in str(e).lower() or "mismatch" in str(e).lower() or "payload" in str(e).lower()
        print(f"Truncated payload rejected: passed ({e})")


def test_invalid_bit_depth():
    parser = ImageParser()
    data = _build_image(4, 4, bit_depth=16)  # 16-bit not supported
    try:
        parser.parse(data)
        assert False
    except ImageParseError as e:
        assert "bit depth" in str(e).lower()
        print(f"Invalid bit depth rejected: passed ({e})")


def test_oversized_image():
    parser = ImageParser(max_dimension=100)
    data = _build_image(200, 200)
    try:
        parser.parse(data)
        assert False
    except ImageParseError as e:
        assert "dimension" in str(e).lower() or "exceeds" in str(e).lower()
        print(f"Oversized image rejected: passed ({e})")


def test_empty_input():
    parser = ImageParser()
    try:
        parser.parse(b"")
        assert False
    except ImageParseError as e:
        print(f"Empty input rejected: passed ({e})")


# ---------------------------------------------------------------------------
# SECURITY ANGLE
# ---------------------------------------------------------------------------
"""
PARSER VULNERABILITIES IN IMAGE FORMATS
----------------------------------------
Real-world CVEs in image parsers:
  • libpng heap overflow (CVE-2011-2501): bad chunk length caused malloc of
    wrong size, then over-write.
  • libjpeg-turbo integer overflow (CVE-2021-46822): width*height overflow
    when both are near UINT16_MAX.
  • WebP/ImageIO heap buffer overflow (CVE-2023-4863): exploited via crafted
    WebP file; critical severity; used in 0-day campaigns.

KEY DEFENCES ILLUSTRATED HERE
------------------------------
1. Magic bytes first: fail before any parsing, avoid wasted work.
2. Integer cross-validation: declared_len vs computed expected_len.
   Prevents allocation bombs and TOCTOU between length field and data.
3. Bounds-checked buffer (_Buf): every read asserts availability,
   preventing any buffer overread.
4. Dimension limits: prevent pixel_count integer overflow and huge allocations.
5. Exact length check: trailing garbage could indicate zip-in-image polyglot
   or parser confusion attacks.
6. No eval/exec: all parsing via struct.unpack with typed format strings.

INTEGER OVERFLOW NOTE
---------------------
  width=65535, height=65535 → pixel_count = 4,294,836,225 > UINT32_MAX
  Python ints are arbitrary precision so no overflow, but max_pixels guards
  against absurd allocation.  In C/C++ you must use checked multiplication.
"""
