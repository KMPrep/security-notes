"""
=============================================================================
Q39 [Hard] — Validate Binary File Magic Numbers
=============================================================================

PROBLEM
-------
Given binary file bytes and a declared filename/extension, classify the
true type by magic number and reject mismatches between extension and content.

Threat: an attacker uploads "invoice.pdf" that is actually a PE executable,
or "image.jpg" that is a ZIP archive (polyglot).

Requirements:
  1. Identify file type from the first N bytes (magic bytes / file signature).
  2. Validate the declared extension matches the detected type.
  3. Detect polyglot files: content that is valid as multiple types.
  4. Return a structured result with type, confidence, and mismatch flag.
  5. Handle truncated files (< magic_length bytes) gracefully.

CLARIFYING QUESTIONS
--------------------
1. Which file types must be supported?
   → Common upload types: PDF, ZIP, PE, ELF, PNG, JPEG, GIF, DOCX/XLSX/PPTX,
     WebP, 7z, RAR, TAR, GZIP, BZIP2, XZ, MP4, BMP.
2. Polyglot detection depth?
   → Check against all known signatures; report all matches.
3. Case sensitivity for extensions?
   → Case-insensitive; normalize to lowercase.
4. Truncated input?
   → Classify as UNKNOWN; do not raise exceptions.
5. Performance?
   → O(1) per check (fixed-length prefix comparison).

=============================================================================
APPROACH 1 — NAIVE (file extension lookup only)
=============================================================================
  Map extension → MIME type.
  Problem: completely bypassable; "evil.exe" renamed to "evil.pdf" passes.

=============================================================================
APPROACH 2 — MAGIC BYTE DATABASE + EXTENSION CROSS-VALIDATION (OPTIMAL)
=============================================================================
  1. Build a lookup table: signature bytes → FileType.
  2. Scan the header bytes against all signatures (sorted by length,
     longest first to avoid false positives from shared prefixes).
  3. Cross-validate detected type(s) against declared extension.
  4. Report all detected types for polyglot detection.

=============================================================================
"""

from __future__ import annotations

import os
from dataclasses import dataclass, field
from typing import Dict, FrozenSet, List, Optional, Set, Tuple


# ---------------------------------------------------------------------------
# Data Types
# ---------------------------------------------------------------------------

@dataclass(frozen=True)
class FileType:
    name: str                        # human-readable name
    extensions: FrozenSet[str]       # allowed lowercase extensions
    description: str = ""


@dataclass
class MagicRule:
    signature: bytes                 # byte pattern to match
    offset: int                      # byte offset in file
    file_type: FileType


@dataclass
class MagicResult:
    declared_name: str
    declared_ext: str
    detected_types: List[FileType]
    is_mismatch: bool
    is_polyglot: bool
    is_unknown: bool
    is_truncated: bool
    mismatch_reason: str = ""

    def __str__(self) -> str:
        types = [t.name for t in self.detected_types] or ["UNKNOWN"]
        flags = []
        if self.is_mismatch:   flags.append("MISMATCH")
        if self.is_polyglot:   flags.append("POLYGLOT")
        if self.is_truncated:  flags.append("TRUNCATED")
        flag_str = f" [{', '.join(flags)}]" if flags else ""
        return (
            f"{self.declared_name!r}: declared={self.declared_ext!r}, "
            f"detected={types}{flag_str}"
        )


# ---------------------------------------------------------------------------
# Magic Number Database
# ---------------------------------------------------------------------------

def _ft(name: str, exts: List[str], desc: str = "") -> FileType:
    return FileType(name=name, extensions=frozenset(exts), description=desc)


_FT_PDF    = _ft("PDF",       [".pdf"],              "Portable Document Format")
_FT_PE     = _ft("PE",        [".exe", ".dll", ".sys", ".ocx", ".drv"], "Windows PE executable")
_FT_ELF    = _ft("ELF",       [".elf", ".so", ".out", ""],              "ELF binary")
_FT_PNG    = _ft("PNG",       [".png"],              "PNG image")
_FT_JPEG   = _ft("JPEG",      [".jpg", ".jpeg"],     "JPEG image")
_FT_GIF87  = _ft("GIF87",     [".gif"],              "GIF87 image")
_FT_GIF89  = _ft("GIF89",     [".gif"],              "GIF89 image")
_FT_ZIP    = _ft("ZIP",       [".zip", ".jar", ".war", ".apk", ".docx",
                                ".xlsx", ".pptx", ".odt", ".ods", ".odp"],
                               "ZIP archive (also OOXML/JAR)")
_FT_GZIP   = _ft("GZIP",      [".gz", ".tgz"],       "GZIP compressed")
_FT_BZIP2  = _ft("BZIP2",     [".bz2"],              "BZIP2 compressed")
_FT_XZ     = _ft("XZ",        [".xz"],               "XZ/LZMA compressed")
_FT_7Z     = _ft("7Z",        [".7z"],               "7-Zip archive")
_FT_RAR4   = _ft("RAR4",      [".rar"],              "RAR archive v4")
_FT_RAR5   = _ft("RAR5",      [".rar"],              "RAR archive v5")
_FT_BMP    = _ft("BMP",       [".bmp"],              "BMP image")
_FT_WEBP   = _ft("WEBP",      [".webp"],             "WebP image")
_FT_MP4    = _ft("MP4",       [".mp4", ".m4v", ".m4a"], "MP4/MPEG-4")
_FT_TAR    = _ft("TAR",       [".tar"],              "TAR archive")
_FT_CLASS  = _ft("JAVACLASS", [".class"],            "Java class file")
_FT_DEX    = _ft("DEX",       [".dex"],              "Dalvik executable")
_FT_MACHO  = _ft("MACHO",     [".dylib", ".macho"],  "Mach-O binary")
_FT_MACHO_FAT = _ft("MACHOFAT", [".dylib", ".macho"], "Mach-O fat binary")


# Rules: (signature, offset, file_type)
# Order matters for display but matching is exhaustive.
_MAGIC_RULES: List[MagicRule] = [
    MagicRule(b"%PDF-",          0, _FT_PDF),
    MagicRule(b"MZ",             0, _FT_PE),
    MagicRule(b"\x7fELF",        0, _FT_ELF),
    MagicRule(b"\x89PNG\r\n\x1a\n", 0, _FT_PNG),
    MagicRule(b"\xff\xd8\xff",   0, _FT_JPEG),
    MagicRule(b"GIF87a",         0, _FT_GIF87),
    MagicRule(b"GIF89a",         0, _FT_GIF89),
    MagicRule(b"PK\x03\x04",     0, _FT_ZIP),
    MagicRule(b"\x1f\x8b",       0, _FT_GZIP),
    MagicRule(b"BZh",            0, _FT_BZIP2),
    MagicRule(b"\xfd7zXZ\x00",   0, _FT_XZ),
    MagicRule(b"7z\xbc\xaf'\x1c", 0, _FT_7Z),
    MagicRule(b"Rar!\x1a\x07\x00", 0, _FT_RAR4),
    MagicRule(b"Rar!\x1a\x07\x01\x00", 0, _FT_RAR5),
    MagicRule(b"BM",             0, _FT_BMP),
    # WebP: RIFF....WEBP
    MagicRule(b"RIFF",           0, _FT_WEBP),    # further check needed
    # TAR: magic at offset 257
    MagicRule(b"ustar",        257, _FT_TAR),
    MagicRule(b"\xca\xfe\xba\xbe", 0, _FT_CLASS), # Java class (also Mach-O fat)
    MagicRule(b"\xde\xad\xbe\xef", 0, _FT_MACHO),
    MagicRule(b"\xfe\xed\xfa\xce", 0, _FT_MACHO),
    MagicRule(b"\xfe\xed\xfa\xcf", 0, _FT_MACHO),
    MagicRule(b"\xce\xfa\xed\xfe", 0, _FT_MACHO),
    MagicRule(b"\xcf\xfa\xed\xfe", 0, _FT_MACHO),
    MagicRule(b"dex\n",          0, _FT_DEX),
]

# Pre-compute minimum bytes needed for any rule
_MAX_OFFSET_NEEDED = max(r.offset + len(r.signature) for r in _MAGIC_RULES)


# ---------------------------------------------------------------------------
# APPROACH 1 — Naive
# ---------------------------------------------------------------------------

class NaiveMagicValidator:
    """Extension-only validation — completely bypassable."""

    _EXT_MAP: Dict[str, str] = {
        ".pdf": "PDF", ".exe": "PE", ".png": "PNG",
        ".jpg": "JPEG", ".jpeg": "JPEG", ".zip": "ZIP",
    }

    def validate(self, filename: str, data: bytes) -> bool:
        ext = os.path.splitext(filename)[1].lower()
        return ext in self._EXT_MAP  # No content check at all


# ---------------------------------------------------------------------------
# APPROACH 2 — Magic byte validator
# ---------------------------------------------------------------------------

class MagicValidator:
    """
    Classifies content by magic bytes and validates against declared extension.
    Time: O(R * S) where R = rule count, S = max signature length (both tiny).
    Space: O(1) reading only the first ~280 bytes of the file.
    """

    def __init__(self, rules: Optional[List[MagicRule]] = None):
        self._rules = rules or _MAGIC_RULES

    def validate(self, filename: str, data: bytes) -> MagicResult:
        declared_ext = os.path.splitext(filename)[1].lower()
        # Ensure at least 1 character for no-extension case
        if not declared_ext:
            declared_ext = ""

        is_truncated = len(data) < 8

        detected = self._detect_types(data)

        is_unknown = len(detected) == 0

        # Polyglot: more than one distinct type detected
        is_polyglot = len(set(t.name for t in detected)) > 1

        # Mismatch check
        is_mismatch = False
        mismatch_reason = ""
        if not is_unknown and not is_truncated:
            allowed_exts: Set[str] = set()
            for ft in detected:
                allowed_exts.update(ft.extensions)
            if declared_ext not in allowed_exts:
                is_mismatch = True
                detected_names = list(dict.fromkeys(t.name for t in detected))
                mismatch_reason = (
                    f"declared ext {declared_ext!r} not in allowed "
                    f"{sorted(allowed_exts)} for detected type(s) {detected_names}"
                )

        return MagicResult(
            declared_name=filename,
            declared_ext=declared_ext,
            detected_types=list(dict.fromkeys(detected)),  # dedup preserving order
            is_mismatch=is_mismatch,
            is_polyglot=is_polyglot,
            is_unknown=is_unknown,
            is_truncated=is_truncated,
            mismatch_reason=mismatch_reason,
        )

    def _detect_types(self, data: bytes) -> List[FileType]:
        found: List[FileType] = []
        for rule in self._rules:
            end = rule.offset + len(rule.signature)
            if len(data) < end:
                continue
            if data[rule.offset : end] == rule.signature:
                # Special case: WEBP needs "WEBP" at offset 8
                if rule.file_type is _FT_WEBP:
                    if len(data) < 12 or data[8:12] != b"WEBP":
                        continue
                found.append(rule.file_type)
        return found


# ---------------------------------------------------------------------------
# Test Cases
# ---------------------------------------------------------------------------

def _pdf_header() -> bytes:
    return b"%PDF-1.4 fake content" + b"\x00" * 300


def _pe_header() -> bytes:
    return b"MZ" + b"\x00" * 300


def _png_header() -> bytes:
    return b"\x89PNG\r\n\x1a\n" + b"\x00" * 50


def _zip_header() -> bytes:
    return b"PK\x03\x04" + b"\x00" * 50


def test_valid_pdf():
    v = MagicValidator()
    r = v.validate("report.pdf", _pdf_header())
    assert not r.is_mismatch, str(r)
    assert not r.is_unknown
    assert r.detected_types[0].name == "PDF"
    print(f"Valid PDF: {r}")


def test_exe_disguised_as_pdf():
    v = MagicValidator()
    r = v.validate("invoice.pdf", _pe_header())
    assert r.is_mismatch, f"Expected mismatch: {r}"
    assert "PE" in [t.name for t in r.detected_types]
    print(f"EXE-as-PDF mismatch detected: {r}")


def test_polyglot_zip_pdf():
    # A polyglot: starts with PDF magic but also has PK header appended
    # (or a zip-within-pdf).  Simulate by combining magic bytes at different
    # offsets — here we just test the framework.
    data = _pdf_header()[:5] + b"PK\x03\x04" + b"\x00" * 50
    # This won't trigger both rules as the offset for ZIP is 0, not 5.
    # Real polyglots are more subtle; test framework handles the case.
    v = MagicValidator()
    r = v.validate("poly.pdf", data)
    print(f"Polyglot test: {r} (is_polyglot={r.is_polyglot})")


def test_unknown_type():
    v = MagicValidator()
    r = v.validate("data.bin", b"\xDE\xAD\x00\x00" + b"\x00" * 20)
    # 0xDEAD0000 doesn't match any rule
    print(f"Unknown type: {r} (is_unknown={r.is_unknown})")


def test_truncated_file():
    v = MagicValidator()
    r = v.validate("small.png", b"\x89PNG")  # Only 4 bytes
    assert r.is_truncated
    print(f"Truncated file: {r}")


def test_zip_disguised_as_docx():
    # DOCX is a ZIP; this should NOT be a mismatch
    v = MagicValidator()
    r = v.validate("document.docx", _zip_header())
    assert not r.is_mismatch, str(r)
    print(f"ZIP-as-DOCX (valid): {r}")


def test_elf_no_extension():
    v = MagicValidator()
    r = v.validate("binary", b"\x7fELF" + b"\x00" * 50)
    # ELF allows empty extension (executables have no extension on Linux)
    print(f"ELF binary: {r} (is_mismatch={r.is_mismatch})")


def test_jpeg_extension_mismatch():
    v = MagicValidator()
    r = v.validate("photo.png", b"\xff\xd8\xff" + b"\x00" * 50)
    assert r.is_mismatch
    print(f"JPEG-as-PNG mismatch detected: {r}")

"""
FILE TYPE CONFUSION ATTACKS
-----------------------------
1. Polyglot files (e.g. GIFAR):
   A file that is simultaneously a valid GIF image and a valid JAR/ZIP archive.
   Used to bypass upload filters: upload as GIF, execute as Java JAR in browser.
   CVE-2007-5232 (Firefox) exploited GIFAR to steal cookies via same-origin.

2. Extension mismatch (content-sniffing):
   Browser MIME sniffing (pre-IE8): serve "image/jpeg" that is actually HTML;
   browser executes it.  Mitigation: X-Content-Type-Options: nosniff.

3. Double extension:
   "evil.php.jpg" — web server strips .jpg, executes .php.

4. Null byte injection:
   "evil.php\x00.jpg" — C-string functions stop at null; PHP (old) executes .php.

DEFENCE IN DEPTH
----------------
1. Magic byte validation (this solution): detect actual type regardless of name.
2. Allowlist-only upload types: reject anything not in explicit allowlist.
3. Rename on upload: overwrite attacker-supplied filename entirely.
4. Serve uploads from separate domain (sandbox): even if type is wrong,
   it can't access the main app's cookies.
5. Anti-virus scanning: detect known malicious content signatures.
6. Content-Disposition: attachment: force download rather than inline render.
"""
