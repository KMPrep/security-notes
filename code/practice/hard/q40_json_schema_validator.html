"""
=============================================================================
Q40 [Hard] — Build Safe JSON Schema Validator
=============================================================================

PROBLEM
-------
Validate security-event JSON against a schema that defines:
  - Required fields and their types
  - Allowed enum values
  - Numeric range constraints
  - String length constraints
  - Nested object and array schemas
  - No arbitrary code execution (no eval, no exec)

Use case: validate incoming SIEM events before ingestion.

CLARIFYING QUESTIONS
--------------------
1. Schema format? JSON Schema standard or custom?
   → Custom lightweight subset (avoids jsonschema dependency):
     type, required, properties, enum, minLength, maxLength,
     minimum, maximum, items (for arrays), additionalProperties.
2. Partial schema (extra keys allowed)?
   → additionalProperties: false rejects unknown keys.
3. Error behaviour: fail-fast or collect all errors?
   → Collect all errors; return full report.
4. Recursion depth limit?
   → Yes; default 10 to prevent DoS via deeply nested schema.
5. Security events can be large?
   → Cap at MAX_EVENT_BYTES before parsing.

=============================================================================
APPROACH 1 — NAIVE (isinstance checks only)
=============================================================================
  Manual per-field isinstance + range check scattered in validation code.
  Problem: not reusable, easy to forget a check, no schema-driven approach.

=============================================================================
APPROACH 2 — SCHEMA-DRIVEN RECURSIVE VALIDATOR (OPTIMAL)
=============================================================================
  1. Schema as a Python dict (equivalent to JSON Schema subset).
  2. Validate(data, schema) recursively.
  3. Collect all violations; return ValidationResult.
  4. Depth limit to prevent infinite recursion on circular schemas.

=============================================================================
"""

from __future__ import annotations

import json
import re
from dataclasses import dataclass, field
from typing import Any, Dict, List, Optional, Tuple


# ---------------------------------------------------------------------------
# Data Types
# ---------------------------------------------------------------------------

MAX_EVENT_BYTES  = 1 * 1024 * 1024  # 1 MB
MAX_DEPTH        = 10
MAX_STRING_LEN   = 65_536           # default max string length


@dataclass
class ValidationError:
    path: str
    message: str

    def __str__(self) -> str:
        return f"{self.path}: {self.message}"


@dataclass
class ValidationResult:
    valid: bool
    errors: List[ValidationError] = field(default_factory=list)

    def add(self, path: str, msg: str) -> None:
        self.valid = False
        self.errors.append(ValidationError(path=path, message=msg))

    def __str__(self) -> str:
        if self.valid:
            return "OK"
        return "; ".join(str(e) for e in self.errors)


# Type mapping
_TYPE_MAP: Dict[str, type] = {
    "string":  str,
    "integer": int,
    "number":  (int, float),  # type: ignore[assignment]
    "boolean": bool,
    "array":   list,
    "object":  dict,
    "null":    type(None),
}


# ---------------------------------------------------------------------------
# APPROACH 1 — Naive (manual per-field checks)
# ---------------------------------------------------------------------------

def validate_security_event_naive(event: dict) -> Tuple[bool, List[str]]:
    """Hardcoded checks; not reusable or schema-driven."""
    errors: List[str] = []
    if "event_id" not in event:
        errors.append("Missing event_id")
    if "timestamp" not in event:
        errors.append("Missing timestamp")
    if "severity" in event and event["severity"] not in ("low", "medium", "high", "critical"):
        errors.append("Invalid severity")
    return len(errors) == 0, errors


# ---------------------------------------------------------------------------
# APPROACH 2 — Schema-driven recursive validator
# ---------------------------------------------------------------------------

class SchemaValidator:
    """
    Validates a data object against a schema dict.

    Schema keys supported:
      type           : "string"|"integer"|"number"|"boolean"|"array"|"object"|"null"
      required       : list of required property names
      properties     : {name: sub-schema}
      additionalProperties : bool (default True)
      enum           : list of allowed values
      minLength      : minimum string length
      maxLength      : maximum string length
      minimum        : minimum numeric value (inclusive)
      maximum        : maximum numeric value (inclusive)
      items          : sub-schema for array elements
      minItems       : minimum array length
      maxItems       : maximum array length
      pattern        : regex pattern for string values
      nullable       : bool — allow null in addition to declared type
    """

    def validate(self, data: Any, schema: dict, path: str = "$") -> ValidationResult:
        result = ValidationResult(valid=True)
        self._validate(data, schema, path, result, depth=0)
        return result

    def validate_json_bytes(self, raw: bytes, schema: dict) -> ValidationResult:
        result = ValidationResult(valid=True)
        if len(raw) > MAX_EVENT_BYTES:
            result.add("$", f"Input too large: {len(raw)} bytes > {MAX_EVENT_BYTES}")
            return result
        try:
            data = json.loads(raw)
        except json.JSONDecodeError as e:
            result.add("$", f"Invalid JSON: {e}")
            return result
        self._validate(data, schema, "$", result, depth=0)
        return result

    # ------------------------------------------------------------------

    def _validate(
        self,
        data: Any,
        schema: dict,
        path: str,
        result: ValidationResult,
        depth: int,
    ) -> None:
        if depth > MAX_DEPTH:
            result.add(path, "Schema depth limit exceeded")
            return

        nullable = schema.get("nullable", False)

        # Handle null
        if data is None:
            if nullable or schema.get("type") == "null":
                return
            result.add(path, "Value is null but nullable=false")
            return

        # Type check
        schema_type = schema.get("type")
        if schema_type:
            expected = _TYPE_MAP.get(schema_type)
            if expected is None:
                result.add(path, f"Unknown schema type: {schema_type!r}")
                return
            # bool is subclass of int in Python; check bool explicitly
            if schema_type == "integer" and isinstance(data, bool):
                result.add(path, f"Expected integer, got boolean")
                return
            if schema_type == "number":
                if not isinstance(data, (int, float)) or isinstance(data, bool):
                    result.add(path, f"Expected number, got {type(data).__name__}")
                    return
            elif not isinstance(data, expected):  # type: ignore[arg-type]
                result.add(
                    path,
                    f"Expected {schema_type}, got {type(data).__name__}",
                )
                return

        # Enum check
        enum_values = schema.get("enum")
        if enum_values is not None:
            if data not in enum_values:
                result.add(path, f"Value {data!r} not in enum {enum_values}")

        # String checks
        if isinstance(data, str):
            min_len = schema.get("minLength", 0)
            max_len = schema.get("maxLength", MAX_STRING_LEN)
            if len(data) < min_len:
                result.add(path, f"String length {len(data)} < minLength {min_len}")
            if len(data) > max_len:
                result.add(path, f"String length {len(data)} > maxLength {max_len}")
            pattern = schema.get("pattern")
            if pattern:
                try:
                    if not re.fullmatch(pattern, data):
                        result.add(path, f"String {data!r} does not match pattern {pattern!r}")
                except re.error as e:
                    result.add(path, f"Invalid regex pattern {pattern!r}: {e}")

        # Numeric checks
        if isinstance(data, (int, float)) and not isinstance(data, bool):
            minimum = schema.get("minimum")
            maximum = schema.get("maximum")
            if minimum is not None and data < minimum:
                result.add(path, f"Value {data} < minimum {minimum}")
            if maximum is not None and data > maximum:
                result.add(path, f"Value {data} > maximum {maximum}")

        # Object checks
        if isinstance(data, dict):
            # Required fields
            for req_field in schema.get("required", []):
                if req_field not in data:
                    result.add(path, f"Missing required field: {req_field!r}")

            # Validate properties
            props_schema: dict = schema.get("properties", {})
            for key, value in data.items():
                child_path = f"{path}.{key}"
                if key in props_schema:
                    self._validate(value, props_schema[key], child_path, result, depth + 1)
                elif schema.get("additionalProperties") is False:
                    result.add(child_path, f"Additional property not allowed: {key!r}")

        # Array checks
        if isinstance(data, list):
            min_items = schema.get("minItems", 0)
            max_items = schema.get("maxItems", 2**31)
            if len(data) < min_items:
                result.add(path, f"Array length {len(data)} < minItems {min_items}")
            if len(data) > max_items:
                result.add(path, f"Array length {len(data)} > maxItems {max_items}")
            items_schema = schema.get("items")
            if items_schema:
                for i, item in enumerate(data):
                    self._validate(item, items_schema, f"{path}[{i}]", result, depth + 1)


# ---------------------------------------------------------------------------
# Example schema: security event
# ---------------------------------------------------------------------------

SECURITY_EVENT_SCHEMA: dict = {
    "type": "object",
    "required": ["event_id", "timestamp", "severity", "source_ip", "event_type"],
    "additionalProperties": False,
    "properties": {
        "event_id": {
            "type": "string",
            "minLength": 1,
            "maxLength": 64,
            "pattern": r"[a-zA-Z0-9\-_]+",
        },
        "timestamp": {
            "type": "string",
            "minLength": 20,
            "maxLength": 32,
        },
        "severity": {
            "type": "string",
            "enum": ["low", "medium", "high", "critical"],
        },
        "source_ip": {
            "type": "string",
            "minLength": 7,
            "maxLength": 45,  # IPv6
        },
        "event_type": {
            "type": "string",
            "enum": [
                "auth_failure", "port_scan", "exfil_attempt",
                "privilege_escalation", "malware_detected",
            ],
        },
        "confidence": {
            "type": "number",
            "minimum": 0.0,
            "maximum": 1.0,
        },
        "tags": {
            "type": "array",
            "items": {"type": "string", "maxLength": 64},
            "maxItems": 20,
        },
        "metadata": {
            "type": "object",
        },
    },
}


# ---------------------------------------------------------------------------
# Test Cases
# ---------------------------------------------------------------------------

def test_valid_event():
    v = SchemaValidator()
    event = {
        "event_id": "evt-001",
        "timestamp": "2026-05-30T12:00:00Z",
        "severity": "high",
        "source_ip": "192.168.1.100",
        "event_type": "port_scan",
        "confidence": 0.92,
        "tags": ["recon", "network"],
    }
    r = v.validate(event, SECURITY_EVENT_SCHEMA)
    assert r.valid, str(r)
    print("Valid event: passed")


def test_missing_required():
    v = SchemaValidator()
    event = {"event_id": "evt-002", "timestamp": "2026-05-30T12:00:00Z"}
    r = v.validate(event, SECURITY_EVENT_SCHEMA)
    assert not r.valid
    missing = {e.message for e in r.errors}
    assert any("severity" in m for m in missing)
    assert any("source_ip" in m for m in missing)
    print(f"Missing required fields: passed ({[str(e) for e in r.errors]})")


def test_invalid_enum():
    v = SchemaValidator()
    event = {
        "event_id": "evt-003",
        "timestamp": "2026-05-30T12:00:00Z",
        "severity": "CRITICAL",  # wrong case
        "source_ip": "10.0.0.1",
        "event_type": "port_scan",
    }
    r = v.validate(event, SECURITY_EVENT_SCHEMA)
    assert not r.valid
    assert any("enum" in str(e) for e in r.errors)
    print(f"Invalid enum: passed")


def test_additional_property_rejected():
    v = SchemaValidator()
    event = {
        "event_id": "evt-004",
        "timestamp": "2026-05-30T12:00:00Z",
        "severity": "low",
        "source_ip": "10.0.0.1",
        "event_type": "port_scan",
        "evil_field": "injected",
    }
    r = v.validate(event, SECURITY_EVENT_SCHEMA)
    assert not r.valid
    assert any("evil_field" in str(e) for e in r.errors)
    print(f"Additional property rejected: passed")


def test_out_of_range_confidence():
    v = SchemaValidator()
    event = {
        "event_id": "evt-005",
        "timestamp": "2026-05-30T12:00:00Z",
        "severity": "medium",
        "source_ip": "10.0.0.1",
        "event_type": "port_scan",
        "confidence": 1.5,  # > 1.0
    }
    r = v.validate(event, SECURITY_EVENT_SCHEMA)
    assert not r.valid
    assert any("maximum" in str(e) for e in r.errors)
    print(f"Out-of-range confidence: passed")


def test_invalid_json_bytes():
    v = SchemaValidator()
    r = v.validate_json_bytes(b"{bad json", SECURITY_EVENT_SCHEMA)
    assert not r.valid
    assert any("JSON" in str(e) for e in r.errors)
    print("Invalid JSON bytes: passed")


def test_oversized_input():
    v = SchemaValidator()
    r = v.validate_json_bytes(b"x" * (MAX_EVENT_BYTES + 1), SECURITY_EVENT_SCHEMA)
    assert not r.valid
    assert any("too large" in str(e) for e in r.errors)
    print("Oversized input: passed")


def test_collect_all_errors():
    v = SchemaValidator()
    event = {
        # Missing: event_id, timestamp, source_ip, event_type
        "severity": "INVALID",
    }
    r = v.validate(event, SECURITY_EVENT_SCHEMA)
    assert not r.valid
    assert len(r.errors) >= 4
    print(f"Collect all errors: passed ({len(r.errors)} errors)")

"""
JSON SCHEMA VALIDATION AS A SECURITY CONTROL
---------------------------------------------
1. Mass assignment / parameter pollution:
   additionalProperties: false rejects unexpected fields, preventing
   attackers from injecting fields like "role":"admin" into event objects.

2. Type confusion:
   type validation prevents integer/string confusion exploits.
   E.g., API expects a numeric ID but attacker sends {"id": "../../config"}.

3. Prototype pollution (Node.js):
   JSON.parse() can inject __proto__ keys.
   Schema with additionalProperties:false + explicit field allowlist
   prevents __proto__ from reaching object construction.

4. Billion laughs / deeply nested JSON:
   A schema bomb: {"a":{"b":{"c":...}}} with depth 10000.
   Our MAX_DEPTH=10 and MAX_EVENT_BYTES limits prevent DoS.

5. Malicious enum bypass:
   Attackers probe for accepted values via brute force.
   Enum validation should use constant-time comparison where values
   are security-sensitive (e.g., role names).

PRODUCTION HARDENING
--------------------
  • Use jsonschema library (Draft-2020-12) for full JSON Schema support.
  • Add $ref and $defs support only if needed; they add recursion risk.
  • Log validation errors as potential attack indicators (WAF-style).
  • Never include raw user input in error messages returned to callers.
"""
