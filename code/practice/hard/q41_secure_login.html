"""
=============================================================================
Q41 [Hard] — Secure Login System Implementation
=============================================================================

PROBLEM
-------
Implement a login system with:
  1. Password hashing using bcrypt (or PBKDF2 for stdlib-only).
  2. Constant-time comparison (prevents timing attacks).
  3. Account lockout after N failed attempts within a time window.
  4. Generic error messages (no user enumeration).
  5. Structured audit logging (all attempts, successes, lockouts).
  6. Secure session token generation (cryptographically random).

CLARIFYING QUESTIONS
--------------------
1. Use external libs (bcrypt, argon2)?
   → Stdlib-only: PBKDF2-HMAC-SHA256 via hashlib; secrets for tokens.
2. Lockout: hard lock or temporary?
   → Temporary lockout: 15 minutes after 5 failures in 10 minutes.
3. Distributed (shared state)?
   → In-memory for this implementation; discuss Redis-based extension.
4. Session token format?
   → 32-byte URL-safe base64; store hash in server side.
5. Password complexity?
   → Minimum 12 characters; no other constraints (reject weak patterns).

=============================================================================
APPROACH 1 — NAIVE
=============================================================================
  Store passwords in plaintext or md5(password).
  Return "Invalid password" vs "User not found" (user enumeration).
  No rate limiting.

=============================================================================
APPROACH 2 — SECURE LOGIN (OPTIMAL)
=============================================================================
  1. PBKDF2-HMAC-SHA256 with random salt per password.
  2. hmac.compare_digest for constant-time comparison.
  3. Sliding window lockout tracker (deque per user).
  4. Generic "Invalid credentials" for all failures.
  5. secrets.token_urlsafe(32) for session tokens.
  6. Audit log with timestamps, IP, and outcome.

=============================================================================
"""

from __future__ import annotations

import hashlib
import hmac
import os
import secrets
import time
from collections import deque
from dataclasses import dataclass, field
from typing import Dict, Deque, List, Optional, Tuple


# ---------------------------------------------------------------------------
# Config
# ---------------------------------------------------------------------------

PBKDF2_ITERATIONS  = 260_000     # NIST SP 800-132 minimum for SHA-256
PBKDF2_HASH        = "sha256"
SALT_BYTES         = 32
TOKEN_BYTES        = 32
MIN_PASSWORD_LEN   = 12
MAX_PASSWORD_LEN   = 128
MAX_USERNAME_LEN   = 64

LOCKOUT_MAX_ATTEMPTS   = 5
LOCKOUT_WINDOW_S       = 600    # 10 minutes
LOCKOUT_DURATION_S     = 900    # 15 minutes


# ---------------------------------------------------------------------------
# Data Types
# ---------------------------------------------------------------------------

@dataclass
class User:
    username: str
    password_hash: str    # "pbkdf2:sha256:iterations$salt_hex$hash_hex"
    is_active: bool = True


@dataclass
class AuditEntry:
    timestamp: float
    username: str
    source_ip: str
    event: str            # "login_attempt", "login_success", "login_failure", "lockout"
    detail: str = ""


@dataclass
class LoginResult:
    success: bool
    session_token: Optional[str]
    message: str
    lockout_remaining_s: Optional[float] = None


# ---------------------------------------------------------------------------
# Password hashing
# ---------------------------------------------------------------------------

def hash_password(password: str) -> str:
    """Hash a password with PBKDF2-HMAC-SHA256 and random salt."""
    salt = os.urandom(SALT_BYTES)
    dk   = hashlib.pbkdf2_hmac(
        PBKDF2_HASH, password.encode("utf-8"), salt, PBKDF2_ITERATIONS
    )
    return f"pbkdf2:{PBKDF2_HASH}:{PBKDF2_ITERATIONS}${salt.hex()}${dk.hex()}"


def verify_password(password: str, stored_hash: str) -> bool:
    """
    Verify password against stored hash.
    Uses hmac.compare_digest for constant-time comparison.
    Always runs the full computation (even if hash format is invalid)
    to prevent timing oracle on username existence.
    """
    try:
        parts = stored_hash.split("$", 2)
        if len(parts) != 3:
            # Invalid format — do a dummy derivation so timing is consistent
            hashlib.pbkdf2_hmac(PBKDF2_HASH, password.encode(), b"\x00" * 32, 1000)
            return False
        meta, salt_hex, expected_hex = parts
        salt     = bytes.fromhex(salt_hex)
        expected = bytes.fromhex(expected_hex)
        # Parse iterations from meta: "pbkdf2:sha256:260000"
        meta_parts = meta.split(":")
        iterations = int(meta_parts[2]) if len(meta_parts) == 3 else PBKDF2_ITERATIONS
        dk = hashlib.pbkdf2_hmac(PBKDF2_HASH, password.encode("utf-8"), salt, iterations)
        return hmac.compare_digest(dk, expected)
    except Exception:
        # Catch any parse error; do dummy work then return False
        return False


# ---------------------------------------------------------------------------
# APPROACH 1 — Naive (insecure, illustrative)
# ---------------------------------------------------------------------------

class NaiveLoginSystem:
    """
    Insecure: plaintext passwords, user enumeration, no lockout.
    NEVER USE IN PRODUCTION.
    """

    def __init__(self):
        self._users: Dict[str, str] = {}   # username → plaintext password

    def register(self, username: str, password: str) -> None:
        self._users[username] = password   # plaintext storage!

    def login(self, username: str, password: str) -> Tuple[bool, str]:
        if username not in self._users:
            return False, "User not found"          # user enumeration!
        if self._users[username] != password:
            return False, "Wrong password"          # user enumeration!
        return True, "ok"


# ---------------------------------------------------------------------------
# APPROACH 2 — Secure login system
# ---------------------------------------------------------------------------

class SecureLoginSystem:
    """
    Secure login with PBKDF2, lockout, audit log, and session tokens.
    """

    def __init__(self):
        self._users: Dict[str, User] = {}
        # username → deque of failure timestamps
        self._failure_times: Dict[str, Deque[float]] = {}
        # username → lockout expiry timestamp
        self._lockout_until: Dict[str, float] = {}
        self._audit_log: List[AuditEntry] = []
        # session_token_hash → username
        self._sessions: Dict[str, str] = {}

    # --- Registration -------------------------------------------------------

    def register(self, username: str, password: str) -> None:
        """Register a new user with hashed password."""
        if not username or len(username) > MAX_USERNAME_LEN:
            raise ValueError(f"Username must be 1–{MAX_USERNAME_LEN} characters")
        if len(password) < MIN_PASSWORD_LEN:
            raise ValueError(f"Password must be at least {MIN_PASSWORD_LEN} characters")
        if len(password) > MAX_PASSWORD_LEN:
            raise ValueError(f"Password must be at most {MAX_PASSWORD_LEN} characters")
        if username in self._users:
            raise ValueError("Username already exists")
        phash = hash_password(password)
        self._users[username] = User(username=username, password_hash=phash)

    # --- Login --------------------------------------------------------------

    def login(self, username: str, password: str, source_ip: str = "0.0.0.0") -> LoginResult:
        now = time.monotonic()

        # Sanitize inputs (length cap to prevent DoS)
        username = (username or "")[:MAX_USERNAME_LEN]
        password = (password or "")[:MAX_PASSWORD_LEN]

        # Check lockout (before any DB lookup — fast path)
        remaining = self._lockout_remaining(username, now)
        if remaining > 0:
            self._audit("lockout_blocked", username, source_ip,
                        f"remaining={remaining:.0f}s")
            return LoginResult(
                success=False,
                session_token=None,
                message="Invalid credentials",   # generic
                lockout_remaining_s=remaining,
            )

        # Lookup user — always run password verification to avoid timing oracle
        user = self._users.get(username)
        if user is None:
            # Dummy verification to keep constant time
            verify_password(password, "pbkdf2:sha256:260000$" + "00" * 32 + "$" + "00" * 32)
            self._record_failure(username, now)
            self._audit("login_failure", username, source_ip, "unknown_user")
            return LoginResult(success=False, session_token=None,
                               message="Invalid credentials")

        pw_ok = verify_password(password, user.password_hash)

        if not pw_ok or not user.is_active:
            self._record_failure(username, now)
            self._audit("login_failure", username, source_ip,
                        "bad_password" if pw_ok is False else "inactive")
            return LoginResult(success=False, session_token=None,
                               message="Invalid credentials")

        # Success — clear failures, issue token
        self._failure_times.pop(username, None)
        self._lockout_until.pop(username, None)
        token = self._issue_session(username)
        self._audit("login_success", username, source_ip, "")
        return LoginResult(success=True, session_token=token,
                           message="OK")

    # --- Session management -------------------------------------------------

    def validate_session(self, token: str) -> Optional[str]:
        """Return username if token is valid, else None."""
        token_hash = hashlib.sha256(token.encode()).hexdigest()
        return self._sessions.get(token_hash)

    def logout(self, token: str) -> None:
        token_hash = hashlib.sha256(token.encode()).hexdigest()
        self._sessions.pop(token_hash, None)

    # --- Internals ----------------------------------------------------------

    def _issue_session(self, username: str) -> str:
        raw_token  = secrets.token_urlsafe(TOKEN_BYTES)
        token_hash = hashlib.sha256(raw_token.encode()).hexdigest()
        self._sessions[token_hash] = username
        return raw_token

    def _record_failure(self, username: str, now: float) -> None:
        q = self._failure_times.setdefault(username, deque())
        # Evict old failures outside the window
        while q and (now - q[0]) > LOCKOUT_WINDOW_S:
            q.popleft()
        q.append(now)
        if len(q) >= LOCKOUT_MAX_ATTEMPTS:
            self._lockout_until[username] = now + LOCKOUT_DURATION_S
            self._audit("lockout_set", username, "", f"attempts={len(q)}")

    def _lockout_remaining(self, username: str, now: float) -> float:
        until = self._lockout_until.get(username, 0.0)
        if now < until:
            return until - now
        return 0.0

    def _audit(self, event: str, username: str, ip: str, detail: str) -> None:
        self._audit_log.append(AuditEntry(
            timestamp=time.time(),
            username=username,
            source_ip=ip,
            event=event,
            detail=detail,
        ))

    @property
    def audit_log(self) -> List[AuditEntry]:
        return list(self._audit_log)

def test_successful_login():
    sys = SecureLoginSystem()
    sys.register("alice", "SuperSecret123!")
    r = sys.login("alice", "SuperSecret123!", "1.2.3.4")
    assert r.success
    assert r.session_token is not None
    assert len(r.session_token) > 20
    print(f"Successful login: passed (token={r.session_token[:8]}...)")


def test_wrong_password():
    sys = SecureLoginSystem()
    sys.register("bob", "ValidPassword99#")
    r = sys.login("bob", "WrongPassword!", "1.2.3.4")
    assert not r.success
    assert r.message == "Invalid credentials"
    print(f"Wrong password: passed (generic message)")


def test_nonexistent_user():
    sys = SecureLoginSystem()
    r = sys.login("ghost", "anything", "1.2.3.4")
    assert not r.success
    assert r.message == "Invalid credentials"
    print(f"Non-existent user: passed (no enumeration)")


def test_lockout():
    sys = SecureLoginSystem()
    sys.register("carol", "ValidPassword12#")
    for i in range(LOCKOUT_MAX_ATTEMPTS):
        r = sys.login("carol", "wrongpass", "5.5.5.5")
        assert not r.success
    # Next attempt should be locked
    r = sys.login("carol", "ValidPassword12#", "5.5.5.5")
    assert not r.success
    assert r.lockout_remaining_s is not None and r.lockout_remaining_s > 0
    print(f"Lockout triggered: passed (remaining={r.lockout_remaining_s:.0f}s)")


def test_session_validate():
    sys = SecureLoginSystem()
    sys.register("dave", "SecurePhrase99$")
    r = sys.login("dave", "SecurePhrase99$", "2.2.2.2")
    assert r.success and r.session_token
    user = sys.validate_session(r.session_token)
    assert user == "dave"
    sys.logout(r.session_token)
    assert sys.validate_session(r.session_token) is None
    print("Session validate/logout: passed")


def test_password_hashing_unique_salts():
    h1 = hash_password("SamePassword!1")
    h2 = hash_password("SamePassword!1")
    assert h1 != h2, "Same password must produce different hashes (unique salts)"
    assert verify_password("SamePassword!1", h1)
    assert verify_password("SamePassword!1", h2)
    assert not verify_password("WrongPassword1", h1)
    print("Unique salts per hash: passed")


def test_audit_log():
    sys = SecureLoginSystem()
    sys.register("eve", "AuditTestPass!1")
    sys.login("eve", "WrongPass!!", "9.9.9.9")
    sys.login("eve", "AuditTestPass!1", "9.9.9.9")
    log = sys.audit_log
    events = [e.event for e in log]
    assert "login_failure" in events
    assert "login_success" in events
    print(f"Audit log: passed ({events})")


# ---------------------------------------------------------------------------
# SECURITY ANGLE
# ---------------------------------------------------------------------------
"""
KEY SECURITY CONTROLS IN THIS IMPLEMENTATION
---------------------------------------------
1. PASSWORD HASHING
   PBKDF2-HMAC-SHA256 with 260,000 iterations + random 32-byte salt.
   Why iterations?  Makes offline brute-force ~260,000x slower.
   Why unique salt?  Prevents rainbow table attacks and shows identical
   passwords with different hashes.
   Production preference: argon2id > bcrypt > PBKDF2.

2. CONSTANT-TIME COMPARISON
   hmac.compare_digest prevents timing attacks:
   A naive "if hash1 == hash2" short-circuits on first mismatch.
   By measuring response time across many attempts, an attacker can
   reconstruct the correct hash byte by byte.

3. USER ENUMERATION
   "User not found" vs "Wrong password" lets attackers enumerate valid usernames.
   Generic "Invalid credentials" prevents this.
   Always run verify_password even for non-existent users (dummy computation).

4. ACCOUNT LOCKOUT
   Prevents online brute-force. 5 failures → 15-minute lockout.
   Note: lockout itself can be weaponized for DoS (lock out legitimate users).
   Mitigations: CAPTCHA, IP-based limits, notify user via email.

5. SESSION TOKENS
   secrets.token_urlsafe(32) = 256 bits of entropy; unguessable.
   Store only hash(token) server-side so a DB breach doesn't expose tokens.

6. AUDIT LOG
   All login events (success, failure, lockout) recorded with timestamp + IP.
   Essential for incident response and compliance (SOX, PCI-DSS, ISO 27001).
"""
