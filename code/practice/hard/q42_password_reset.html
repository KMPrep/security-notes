"""
=============================================================================
Q42 [Hard] — Password Reset System
=============================================================================

PROBLEM
-------
Implement a secure password reset flow with:
  1. Cryptographically random reset token (not predictable).
  2. Token expiry (short TTL, default 15 minutes).
  3. Single-use enforcement (token invalidated after first use).
  4. Rate limiting (prevent token flooding for a given user/IP).
  5. Constant-time token comparison (timing-attack resistance).
  6. Audit logging of all reset events.
  7. No oracle: same response for valid/invalid usernames.

CLARIFYING QUESTIONS
--------------------
1. Token delivery? Email, SMS?
   → Delivery is out of scope; we return the token to simulate sending.
2. Token length?
   → 32 bytes URL-safe base64 (43 characters); ~256 bits entropy.
3. How many reset requests per user per window?
   → 3 per 10 minutes.
4. Should password complexity be re-validated on reset?
   → Yes; re-run the same password policy as registration.
5. Is the old password required to reset?
   → No (that's a password CHANGE flow, not reset).

=============================================================================
APPROACH 1 — NAIVE
=============================================================================
  Token = str(user_id) + timestamp (guessable, no expiry check, reusable).
  Return "User not found" (enumeration), no rate limit.

=============================================================================
APPROACH 2 — SECURE RESET SYSTEM (OPTIMAL)
=============================================================================
  1. Generate token via secrets.token_urlsafe.
  2. Store token_hash (SHA-256) with expiry and used flag.
  3. Rate limit: sliding window per (username, ip).
  4. hmac.compare_digest for constant-time token comparison.
  5. Invalidate token immediately after successful use.
  6. Generic response for all outcomes.

=============================================================================
"""

from __future__ import annotations

import hashlib
import hmac
import secrets
import time
from collections import deque
from dataclasses import dataclass, field
from typing import Deque, Dict, Optional, List


# ---------------------------------------------------------------------------
# Config
# ---------------------------------------------------------------------------

TOKEN_BYTES         = 32
TOKEN_TTL_S         = 900    # 15 minutes
RATE_LIMIT_MAX      = 3
RATE_LIMIT_WINDOW_S = 600    # 10 minutes
MIN_PASSWORD_LEN    = 12
MAX_PASSWORD_LEN    = 128


# ---------------------------------------------------------------------------
# Data Types
# ---------------------------------------------------------------------------

@dataclass
class ResetToken:
    token_hash: str        # SHA-256(raw_token) stored; raw token is not stored
    username: str
    created_at: float
    expires_at: float
    used: bool = False


@dataclass
class ResetAudit:
    timestamp: float
    event: str
    username: str
    source_ip: str
    detail: str = ""


@dataclass
class ResetResult:
    success: bool
    message: str
    token: Optional[str] = None    # Only populated when success; simulates email send


# ---------------------------------------------------------------------------
# APPROACH 1 — Naive (insecure)
# ---------------------------------------------------------------------------

class NaivePasswordReset:
    """
    Insecure: predictable token, no expiry, no rate limit, user enumeration.
    NEVER USE IN PRODUCTION.
    """

    def __init__(self):
        self._users: Dict[str, str] = {}   # username → password
        self._tokens: Dict[str, str] = {}  # token → username (reusable)

    def register(self, username: str, password: str) -> None:
        self._users[username] = password

    def request_reset(self, username: str) -> str:
        if username not in self._users:
            return "User not found"         # enumeration
        token = f"reset_{username}_1234"    # predictable
        self._tokens[token] = username
        return token

    def do_reset(self, token: str, new_password: str) -> bool:
        if token in self._tokens:           # reusable token
            username = self._tokens[token]
            self._users[username] = new_password
            return True
        return False


# ---------------------------------------------------------------------------
# APPROACH 2 — Secure password reset
# ---------------------------------------------------------------------------

class PasswordResetSystem:
    """
    Secure password reset with random tokens, expiry, single-use, rate limiting.
    """

    def __init__(self):
        # username → hashed_password (simple store; see Q41 for full hashing)
        self._users: Dict[str, bytes] = {}
        # token_hash → ResetToken
        self._tokens: Dict[str, ResetToken] = {}
        # (username, ip) → deque of request timestamps
        self._rate_windows: Dict[str, Deque[float]] = {}
        self._audit: List[ResetAudit] = []

    # --- User management (simplified) --------------------------------------

    def register(self, username: str, password: str) -> None:
        self._users[username] = password.encode()

    # --- Request reset ------------------------------------------------------

    def request_reset(self, username: str, source_ip: str = "") -> ResetResult:
        now = time.monotonic()

        # Rate limit check (per username+ip)
        rate_key = f"{username}|{source_ip}"
        if not self._check_rate(rate_key, now):
            self._log("rate_limit", username, source_ip, "")
            # Generic response; do NOT reveal whether user exists
            return ResetResult(success=False, message="Too many requests")

        # Always log and return same message regardless of user existence
        user_exists = username in self._users
        if not user_exists:
            self._log("reset_request_unknown_user", username, source_ip, "")
            return ResetResult(
                success=True,
                message="If that account exists, a reset link has been sent.",
            )

        # Invalidate any previous unused tokens for this user
        for th, tok in list(self._tokens.items()):
            if tok.username == username and not tok.used:
                tok.used = True
                self._log("reset_token_superseded", username, source_ip, "")

        # Generate token
        raw_token  = secrets.token_urlsafe(TOKEN_BYTES)
        token_hash = hashlib.sha256(raw_token.encode()).hexdigest()
        self._tokens[token_hash] = ResetToken(
            token_hash=token_hash,
            username=username,
            created_at=now,
            expires_at=now + TOKEN_TTL_S,
        )

        self._log("reset_requested", username, source_ip, "token issued")
        # In production: send raw_token via email; here we return it for testing
        return ResetResult(
            success=True,
            message="If that account exists, a reset link has been sent.",
            token=raw_token,
        )

    # --- Perform reset ------------------------------------------------------

    def perform_reset(
        self, raw_token: str, new_password: str, source_ip: str = ""
    ) -> ResetResult:
        now = time.monotonic()

        # Validate new password
        if len(new_password) < MIN_PASSWORD_LEN:
            return ResetResult(
                success=False,
                message=f"Password must be at least {MIN_PASSWORD_LEN} characters",
            )
        if len(new_password) > MAX_PASSWORD_LEN:
            return ResetResult(
                success=False,
                message=f"Password must be at most {MAX_PASSWORD_LEN} characters",
            )

        # Lookup token record (by hash)
        token_hash = hashlib.sha256(raw_token.encode()).hexdigest()
        record = self._tokens.get(token_hash)

        # Always use constant-time for token comparison
        # (We compare hashes, so direct dict lookup is safe after hashing)
        if record is None:
            self._log("reset_invalid_token", "", source_ip, "not found")
            return ResetResult(success=False, message="Invalid or expired token")

        if record.used:
            self._log("reset_token_reuse", record.username, source_ip, "used")
            return ResetResult(success=False, message="Invalid or expired token")

        if now > record.expires_at:
            record.used = True  # mark expired tokens as used
            self._log("reset_expired_token", record.username, source_ip,
                      f"age={now - record.created_at:.0f}s")
            return ResetResult(success=False, message="Invalid or expired token")

        # Verify token using constant-time comparison of hashes
        expected = hashlib.sha256(raw_token.encode()).hexdigest()
        if not hmac.compare_digest(record.token_hash, expected):
            self._log("reset_token_mismatch", record.username, source_ip, "")
            return ResetResult(success=False, message="Invalid or expired token")

        # Single-use: invalidate immediately
        record.used = True

        # Update password (simplified; use PBKDF2 in production)
        self._users[record.username] = new_password.encode()

        self._log("reset_success", record.username, source_ip, "")
        return ResetResult(success=True, message="Password reset successfully")

    # --- Helpers ------------------------------------------------------------

    def _check_rate(self, key: str, now: float) -> bool:
        q = self._rate_windows.setdefault(key, deque())
        while q and (now - q[0]) > RATE_LIMIT_WINDOW_S:
            q.popleft()
        if len(q) >= RATE_LIMIT_MAX:
            return False
        q.append(now)
        return True

    def _log(self, event: str, username: str, ip: str, detail: str) -> None:
        self._audit.append(ResetAudit(
            timestamp=time.time(),
            event=event,
            username=username,
            source_ip=ip,
            detail=detail,
        ))

    @property
    def audit_log(self) -> List[ResetAudit]:
        return list(self._audit)

def test_happy_path():
    sys = PasswordResetSystem()
    sys.register("alice", "OldPassword12!")
    r1 = sys.request_reset("alice", "1.1.1.1")
    assert r1.success and r1.token
    r2 = sys.perform_reset(r1.token, "NewPassword99#", "1.1.1.1")
    assert r2.success, r2.message
    print(f"Happy path: passed")


def test_single_use_token():
    sys = PasswordResetSystem()
    sys.register("bob", "OldPassword12!")
    r1 = sys.request_reset("bob")
    assert r1.token
    sys.perform_reset(r1.token, "NewPassword12!")
    r2 = sys.perform_reset(r1.token, "AnotherPass99!")
    assert not r2.success
    assert "expired" in r2.message.lower() or "invalid" in r2.message.lower()
    print(f"Single-use token: passed")


def test_expired_token():
    sys = PasswordResetSystem()
    sys.register("carol", "ValidPass12!")
    r1 = sys.request_reset("carol")
    assert r1.token
    # Manually expire the token
    token_hash = hashlib.sha256(r1.token.encode()).hexdigest()
    import hashlib as _hl
    sys._tokens[token_hash].expires_at = time.monotonic() - 1
    r2 = sys.perform_reset(r1.token, "NewPassword12!")
    assert not r2.success
    print(f"Expired token rejected: passed")


def test_nonexistent_user_generic_response():
    sys = PasswordResetSystem()
    r = sys.request_reset("ghost", "2.2.2.2")
    assert r.success   # generic success response; no enumeration
    assert r.token is None
    print(f"Non-existent user (no enumeration): passed")


def test_rate_limit():
    sys = PasswordResetSystem()
    sys.register("dave", "ValidPassword12!")
    for i in range(RATE_LIMIT_MAX):
        r = sys.request_reset("dave", "3.3.3.3")
        assert r.success
    r = sys.request_reset("dave", "3.3.3.3")
    assert not r.success
    assert "Too many" in r.message
    print(f"Rate limit enforced: passed")


def test_token_superseded():
    """New reset request invalidates old unused token."""
    sys = PasswordResetSystem()
    sys.register("eve", "ValidPassword12!")
    r1 = sys.request_reset("eve")
    r2 = sys.request_reset("eve")
    assert r1.token and r2.token
    # Old token should now fail
    r3 = sys.perform_reset(r1.token, "NewPassword12!")
    assert not r3.success
    # New token should succeed
    r4 = sys.perform_reset(r2.token, "NewPassword12!")
    assert r4.success
    print("Token superseded by new request: passed")


def test_weak_password_rejected():
    sys = PasswordResetSystem()
    sys.register("frank", "ValidPassword12!")
    r1 = sys.request_reset("frank")
    assert r1.token
    r2 = sys.perform_reset(r1.token, "short")
    assert not r2.success
    assert "characters" in r2.message
    print(f"Weak new password rejected: passed ({r2.message})")

"""
COMMON PASSWORD RESET VULNERABILITIES
---------------------------------------
1. PREDICTABLE TOKENS
   Token = username + timestamp (e.g., MD5("alice1685000000")).
   Attacker brute-forces or predicts token → arbitrary account takeover.
   Fix: secrets.token_urlsafe(32) = 256 bits of entropy.

2. TOKEN REUSE (NO SINGLE-USE)
   Token stored permanently; attacker can replay it later if intercepted.
   Fix: Invalidate immediately after first successful use.

3. NO EXPIRY (OR LONG EXPIRY)
   Token with 24-hour TTL: if intercepted in transit, attacker has a day.
   Fix: Short TTL (15 min); force re-request if expired.

4. USER ENUMERATION
   "This email is not registered" → attacker maps valid users.
   Fix: Generic "If that account exists, a link has been sent."

5. TOKEN FLOODING / DoS
   Spam 1000 reset requests for "admin@corp.com".
   Fix: Rate limit per username+IP; invalidate previous tokens.

6. TOKEN TRANSMISSION
   Token in URL (HTTPS OK but logged in server/CDN logs).
   Token in email (secure channel; acceptable for this flow).
   Never return token in response body to the requester.

7. TIMING ATTACK ON TOKEN COMPARISON
   if stored_token == submitted_token: — short-circuits on first byte mismatch.
   Fix: hmac.compare_digest on hashes.
"""
