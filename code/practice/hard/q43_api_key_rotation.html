"""
=============================================================================
Q43 [Hard] — API Key Rotation System
=============================================================================

PROBLEM
-------
Design and implement logic for rotating API keys with zero-downtime:
  1. Generate new key while keeping the previous key active during a
     configurable overlap (grace period).
  2. Accept both "active" and "previous" keys during overlap.
  3. Mark "previous" key as deprecated (log a warning on each use).
  4. After grace period expires, invalidate the previous key.
  5. Each key has version, created_at, expires_at, and a key_id prefix
     (first 8 chars, safe to log).
  6. Verify key via constant-time HMAC.
  7. Scope keys to a service principal.

CLARIFYING QUESTIONS
--------------------
1. Key format?
   → <prefix>_<base64url random bytes>  (e.g., "sk_live_AbCd1234...")
2. Rotation strategy: immediate or dual-key?
   → Dual-key: active + previous both valid for grace period.
3. Persistence?
   → In-memory for this solution; production = database.
4. Key length?
   → 32 random bytes (256 bits) + 8-char prefix.
5. How many prior keys to retain?
   → Only one (current + one previous); older keys immediately revoked.

=============================================================================
APPROACH 1 — NAIVE (no overlap, hard cutover)
=============================================================================
  Replace active key immediately; all callers with old key fail instantly.
  No grace period; operational risk of downtime.

=============================================================================
APPROACH 2 — DUAL-KEY ROTATION WITH GRACE PERIOD (OPTIMAL)
=============================================================================
  Data structure: per-principal {active: Key, previous: Key|None}
  On rotate():
    1. Promote active → previous (set previous_expires_at = now + grace).
    2. Generate new active key.
  On verify(key):
    1. Hash key; compare to active_hash (constant-time).
    2. If no match, compare to previous_hash (if not expired).
    3. Return VerifyResult with deprecation warning if previous used.

=============================================================================
"""

from __future__ import annotations

import hashlib
import hmac
import os
import secrets
import time
from dataclasses import dataclass
from typing import Dict, Optional


# ---------------------------------------------------------------------------
# Config
# ---------------------------------------------------------------------------

KEY_PREFIX            = "sk_"
KEY_RANDOM_BYTES      = 32
DEFAULT_KEY_TTL_S     = 86_400 * 90   # 90 days
DEFAULT_GRACE_S       = 86_400 * 7    # 7-day overlap
MAX_PRINCIPAL_LEN     = 64


# ---------------------------------------------------------------------------
# Data Types
# ---------------------------------------------------------------------------

@dataclass
class ApiKey:
    key_id: str          # first 8 chars of the key (safe to log)
    key_hash: str        # SHA-256(full_key)
    version: int
    principal: str
    created_at: float
    expires_at: float
    is_deprecated: bool = False

    @property
    def is_expired(self) -> bool:
        return time.monotonic() > self.expires_at


@dataclass
class VerifyResult:
    valid: bool
    principal: str
    key_id: str
    version: int
    is_deprecated: bool = False
    deprecation_message: str = ""
    failure_reason: str = ""


@dataclass
class RotationResult:
    new_key: str          # full raw key (return once; don't store)
    new_key_id: str
    new_version: int
    previous_key_id: Optional[str]
    grace_expires_at: float


# ---------------------------------------------------------------------------
# APPROACH 1 — Naive (hard cutover)
# ---------------------------------------------------------------------------

class NaiveApiKeyManager:
    """No grace period; old keys immediately revoked on rotation."""

    def __init__(self):
        self._keys: Dict[str, str] = {}  # principal → raw key (plaintext!)

    def generate(self, principal: str) -> str:
        raw = KEY_PREFIX + secrets.token_urlsafe(KEY_RANDOM_BYTES)
        self._keys[principal] = raw          # stored in plaintext!
        return raw

    def rotate(self, principal: str) -> str:
        return self.generate(principal)      # old key immediately gone

    def verify(self, principal: str, key: str) -> bool:
        return self._keys.get(principal) == key   # timing vulnerable


# ---------------------------------------------------------------------------
# APPROACH 2 — Secure dual-key rotation
# ---------------------------------------------------------------------------

class _PrincipalState:
    __slots__ = ("active", "previous")

    def __init__(self, active: ApiKey, previous: Optional[ApiKey] = None):
        self.active   = active
        self.previous = previous


class ApiKeyManager:
    """
    Dual-key rotation system.
    Never stores raw keys; only SHA-256 hashes.
    Constant-time comparison via hmac.compare_digest.
    """

    def __init__(
        self,
        default_ttl_s: float = DEFAULT_KEY_TTL_S,
        default_grace_s: float = DEFAULT_GRACE_S,
    ):
        self._ttl_s   = default_ttl_s
        self._grace_s = default_grace_s
        self._state: Dict[str, _PrincipalState] = {}
        self._version_counter: Dict[str, int] = {}

    # --- Key generation -----------------------------------------------------

    def _generate_key(self) -> tuple[str, str, str]:
        """Returns (raw_key, key_id, key_hash)."""
        raw        = KEY_PREFIX + secrets.token_urlsafe(KEY_RANDOM_BYTES)
        key_id     = raw[:8] + "..."   # first 8 chars safe to log
        key_hash   = hashlib.sha256(raw.encode()).hexdigest()
        return raw, key_id, key_hash

    def _next_version(self, principal: str) -> int:
        v = self._version_counter.get(principal, 0) + 1
        self._version_counter[principal] = v
        return v

    # --- Issue first key ----------------------------------------------------

    def issue(self, principal: str) -> tuple[str, RotationResult]:
        """Issue the first API key for a principal."""
        if not principal or len(principal) > MAX_PRINCIPAL_LEN:
            raise ValueError("Invalid principal")
        if principal in self._state:
            raise ValueError(f"Principal {principal!r} already has a key; use rotate()")

        now     = time.monotonic()
        raw, key_id, key_hash = self._generate_key()
        version = self._next_version(principal)
        active  = ApiKey(
            key_id=key_id,
            key_hash=key_hash,
            version=version,
            principal=principal,
            created_at=now,
            expires_at=now + self._ttl_s,
        )
        self._state[principal] = _PrincipalState(active=active, previous=None)
        result = RotationResult(
            new_key=raw,
            new_key_id=key_id,
            new_version=version,
            previous_key_id=None,
            grace_expires_at=0.0,
        )
        return raw, result

    # --- Rotate key ---------------------------------------------------------

    def rotate(self, principal: str) -> tuple[str, RotationResult]:
        """
        Rotate key for principal.
        Current active becomes previous (with grace period).
        Returns (raw_new_key, RotationResult).
        """
        if principal not in self._state:
            raise ValueError(f"Principal {principal!r} not found; use issue()")

        now                     = time.monotonic()
        state                   = self._state[principal]
        old_active              = state.active
        prev_key_id             = old_active.key_id

        # Demote current active to previous
        old_active.is_deprecated = True
        old_active.expires_at    = now + self._grace_s

        # Generate new active key
        raw, key_id, key_hash = self._generate_key()
        version = self._next_version(principal)
        new_active = ApiKey(
            key_id=key_id,
            key_hash=key_hash,
            version=version,
            principal=principal,
            created_at=now,
            expires_at=now + self._ttl_s,
        )
        state.active   = new_active
        state.previous = old_active

        result = RotationResult(
            new_key=raw,
            new_key_id=key_id,
            new_version=version,
            previous_key_id=prev_key_id,
            grace_expires_at=now + self._grace_s,
        )
        return raw, result

    # --- Verify key ---------------------------------------------------------

    def verify(self, raw_key: str, principal: str) -> VerifyResult:
        """Verify a raw key against stored hashes (constant-time)."""
        not_found = VerifyResult(
            valid=False, principal=principal, key_id="",
            version=0, failure_reason="invalid_key",
        )

        if not raw_key or not principal:
            return not_found

        state = self._state.get(principal)
        if state is None:
            # Dummy work to maintain timing parity
            _ = hmac.compare_digest("a", "b")
            return not_found

        submitted_hash = hashlib.sha256(raw_key.encode()).hexdigest()
        now = time.monotonic()

        # Check active key
        active_match = hmac.compare_digest(state.active.key_hash, submitted_hash)
        if active_match and not state.active.is_expired:
            return VerifyResult(
                valid=True,
                principal=principal,
                key_id=state.active.key_id,
                version=state.active.version,
            )

        # Check previous key (if exists and within grace period)
        prev_match = False
        if state.previous is not None:
            prev_match = hmac.compare_digest(state.previous.key_hash, submitted_hash)

        if prev_match and state.previous is not None and not state.previous.is_expired:
            return VerifyResult(
                valid=True,
                principal=principal,
                key_id=state.previous.key_id,
                version=state.previous.version,
                is_deprecated=True,
                deprecation_message=(
                    f"API key version {state.previous.version} is deprecated. "
                    f"Please rotate to version {state.active.version}."
                ),
            )

        return VerifyResult(
            valid=False, principal=principal, key_id="",
            version=0, failure_reason="invalid_or_expired",
        )

    # --- Revoke -------------------------------------------------------------

    def revoke(self, principal: str) -> None:
        """Immediately revoke all keys for a principal."""
        self._state.pop(principal, None)
        self._version_counter.pop(principal, None)

# Test Cases

def test_issue_and_verify():
    mgr = ApiKeyManager()
    raw, result = mgr.issue("service-a")
    assert result.new_version == 1
    vr = mgr.verify(raw, "service-a")
    assert vr.valid
    assert not vr.is_deprecated
    print(f"Issue and verify: passed (key_id={result.new_key_id})")


def test_rotation_grace_period():
    mgr = ApiKeyManager(default_grace_s=3600)
    raw_v1, _ = mgr.issue("service-b")
    raw_v2, r = mgr.rotate("service-b")
    assert r.new_version == 2

    # Both keys should be valid during grace period
    vr1 = mgr.verify(raw_v1, "service-b")
    vr2 = mgr.verify(raw_v2, "service-b")
    assert vr1.valid and vr1.is_deprecated
    assert vr2.valid and not vr2.is_deprecated
    print(f"Grace period: passed (v1 deprecated={vr1.is_deprecated})")


def test_old_key_invalid_after_grace():
    mgr = ApiKeyManager(default_grace_s=-1)  # Already expired
    raw_v1, _ = mgr.issue("service-c")
    raw_v2, _ = mgr.rotate("service-c")

    vr1 = mgr.verify(raw_v1, "service-c")
    assert not vr1.valid
    vr2 = mgr.verify(raw_v2, "service-c")
    assert vr2.valid
    print(f"Old key expired after grace: passed")


def test_revoke():
    mgr = ApiKeyManager()
    raw, _ = mgr.issue("service-d")
    mgr.revoke("service-d")
    vr = mgr.verify(raw, "service-d")
    assert not vr.valid
    print("Revoke: passed")


def test_wrong_principal():
    mgr = ApiKeyManager()
    raw, _ = mgr.issue("svc-x")
    mgr.issue("svc-y")
    vr = mgr.verify(raw, "svc-y")   # key for svc-x used against svc-y
    assert not vr.valid
    print("Wrong principal rejected: passed")


def test_key_uniqueness():
    mgr = ApiKeyManager()
    raw1, _ = mgr.issue("svc-u")
    mgr.rotate("svc-u")
    raw2, _ = mgr.rotate("svc-u")
    assert raw1 != raw2
    print("Key uniqueness: passed")


def test_constant_time_no_timing_leak():
    """Verify hmac.compare_digest is used (structural test)."""
    import inspect
    src = inspect.getsource(ApiKeyManager.verify)
    assert "compare_digest" in src
    print("Constant-time comparison verified in source: passed")


# ---------------------------------------------------------------------------
# SECURITY ANGLE

"""
API KEY ROTATION — SECURITY RATIONALE
---------------------------------------
1. WHY ROTATE?
   Keys are long-lived credentials. If a key leaks (log exposure, git commit,
   MITM on non-TLS endpoint), rotation limits the blast radius to the period
   before rotation was noticed.

2. DUAL-KEY GRACE PERIOD
   Hard cutover breaks clients that haven't updated yet.
   Grace period (7 days) lets operators update all callers before the old key
   expires, achieving zero-downtime rotation.

3. NEVER STORE RAW KEYS
   Store only SHA-256(raw_key). A DB breach exposes only hashes.
   Attackers can't use hashes directly as API keys.
   Similar to storing password hashes (but bcrypt/Argon2 not needed here
   since API keys are already 256-bit random, not low-entropy user passwords).

4. KEY SCOPE / PRINCIPAL BINDING
   Each key is bound to a principal (service name).
   Prevents key cross-use: a leaked key for service-A can't authenticate as B.

5. CONSTANT-TIME COMPARISON
   Comparing keys with == leaks timing information.
   hmac.compare_digest compares all bytes in constant time.

6. KEY_ID PREFIX (safe to log)
   First 8 chars safe to appear in logs (too short to derive full key).
   Allows correlating audit logs with specific key versions without
   exposing the secret.

7. AUDIT EVENTS TO LOG
   key_issued, key_rotated, key_verified, key_deprecated_used, key_revoked.
   These support forensic investigation after a suspected compromise.
"""
