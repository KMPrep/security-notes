"""
=============================================================================
Q45 [Hard] — Secure Webhook Verification
=============================================================================

PROBLEM
-------
Implement webhook signature validation with:
  1. HMAC-SHA256 signature verification (constant-time).
  2. Timestamp tolerance: reject webhooks older than MAX_AGE_S seconds
     (prevents replay attacks).
  3. Replay protection: track nonces (or timestamp+sig) within the tolerance
     window so the same payload can't be delivered twice.
  4. Support multiple signing key versions (key rotation).
  5. Return a structured result with failure reason.

CLARIFYING QUESTIONS
--------------------
1. Signature format?
   → As used by Stripe/GitHub: "t=<timestamp>,v1=<hex-sig>"
     The signed payload is: "<timestamp>.<raw_body>"
2. Multiple signatures in header?
   → Yes; iterate all v1= values and accept if any matches.
3. Replay window?
   → Default 300 seconds (5 minutes); configurable.
4. Key rotation?
   → Accept a list of valid keys; verify against all in constant time.
5. Tolerance for clock skew?
   → ±MAX_AGE_S; reject if |now - timestamp| > MAX_AGE_S.

=============================================================================
APPROACH 1 — NAIVE (no timestamp, no replay protection)
=============================================================================
  HMAC compare via ==; no timestamp check; single key; replayable.

=============================================================================
APPROACH 2 — STRIPE-STYLE WEBHOOK VERIFIER (OPTIMAL)
=============================================================================
  1. Parse header: extract timestamp and all v1 signatures.
  2. Check timestamp freshness.
  3. Compute expected = HMAC-SHA256(key, f"{timestamp}.{body}").
  4. Constant-time compare against each v1 signature.
  5. Track (timestamp, sig) pairs in a deque/set for replay detection.

=============================================================================
"""

from __future__ import annotations

import hashlib
import hmac
import time
from collections import deque
from dataclasses import dataclass
from typing import Deque, List, Optional, Set, Tuple


# ---------------------------------------------------------------------------
# Config
# ---------------------------------------------------------------------------

MAX_AGE_S      = 300      # 5 minutes
REPLAY_TTL_S   = 600      # keep nonces for 10 minutes


# ---------------------------------------------------------------------------
# Data Types
# ---------------------------------------------------------------------------

@dataclass
class WebhookVerifyResult:
    valid: bool
    timestamp: Optional[int] = None
    key_version: Optional[int] = None    # index of matching key
    failure_reason: str = ""

    def __bool__(self) -> bool:
        return self.valid


# ---------------------------------------------------------------------------
# APPROACH 1 — Naive
# ---------------------------------------------------------------------------

class NaiveWebhookVerifier:
    """Vulnerable to replay and timing attacks."""

    def __init__(self, secret: str):
        self._secret = secret

    def verify(self, body: bytes, signature: str) -> bool:
        expected = hmac.new(
            self._secret.encode(), body, hashlib.sha256
        ).hexdigest()
        return signature == expected   # timing-vulnerable == comparison


# ---------------------------------------------------------------------------
# APPROACH 2 — Secure webhook verifier
# ---------------------------------------------------------------------------

class WebhookVerifier:
    """
    Stripe-style webhook signature verification.
    Header format: "t=<unix_timestamp>,v1=<hex_sig>[,v1=<hex_sig2>]"
    Signed string: f"{timestamp}.{body_bytes}"
    """

    def __init__(
        self,
        secrets: List[str],           # current + previous signing secrets
        max_age_s: int = MAX_AGE_S,
        replay_ttl_s: int = REPLAY_TTL_S,
    ):
        if not secrets:
            raise ValueError("At least one signing secret is required")
        self._secrets    = [s.encode() for s in secrets]
        self._max_age_s  = max_age_s
        self._replay_ttl = replay_ttl_s
        # Replay nonce store: set of (timestamp, hex_sig)
        self._seen: Set[Tuple[int, str]] = set()
        # Ordered by insertion time for cleanup: deque of (time.time(), nonce)
        self._seen_queue: Deque[Tuple[float, Tuple[int, str]]] = deque()

    def verify(self, body: bytes, header: str) -> WebhookVerifyResult:
        now = int(time.time())

        # Parse header
        timestamp, v1_sigs = self._parse_header(header)
        if timestamp is None or not v1_sigs:
            return WebhookVerifyResult(
                valid=False, failure_reason="invalid_header"
            )

        # Timestamp freshness check
        age = abs(now - timestamp)
        if age > self._max_age_s:
            return WebhookVerifyResult(
                valid=False,
                timestamp=timestamp,
                failure_reason=f"timestamp_too_old: age={age}s",
            )

        # Compute signed_payload
        signed_payload = f"{timestamp}.".encode() + body

        # Try each key against each v1 signature
        matched_key_idx  = None
        matched_sig      = None

        for key_idx, secret in enumerate(self._secrets):
            expected_hex = hmac.new(secret, signed_payload, hashlib.sha256).hexdigest()
            for v1_sig in v1_sigs:
                if hmac.compare_digest(expected_hex, v1_sig):
                    matched_key_idx = key_idx
                    matched_sig     = v1_sig
                    break
            if matched_key_idx is not None:
                break

        if matched_sig is None:
            return WebhookVerifyResult(
                valid=False,
                timestamp=timestamp,
                failure_reason="signature_mismatch",
            )

        # Replay check
        nonce = (timestamp, matched_sig)
        self._evict_old_nonces(time.monotonic())
        if nonce in self._seen:
            return WebhookVerifyResult(
                valid=False,
                timestamp=timestamp,
                failure_reason="replay_detected",
            )

        self._seen.add(nonce)
        self._seen_queue.append((time.monotonic(), nonce))

        return WebhookVerifyResult(
            valid=True,
            timestamp=timestamp,
            key_version=matched_key_idx,
        )

    # --- Helpers ------------------------------------------------------------

    @staticmethod
    def _parse_header(header: str) -> Tuple[Optional[int], List[str]]:
        """
        Parse "t=<ts>,v1=<sig>[,v1=<sig2>]"
        Returns (timestamp_int, list_of_v1_sigs) or (None, []) on error.
        """
        timestamp = None
        v1_sigs: List[str] = []
        for part in header.split(","):
            part = part.strip()
            if part.startswith("t="):
                try:
                    timestamp = int(part[2:])
                except ValueError:
                    return None, []
            elif part.startswith("v1="):
                sig = part[3:]
                # Validate hex
                try:
                    bytes.fromhex(sig)
                    v1_sigs.append(sig)
                except ValueError:
                    pass
        return timestamp, v1_sigs

    def _evict_old_nonces(self, now_mono: float) -> None:
        while self._seen_queue:
            insert_time, nonce = self._seen_queue[0]
            if now_mono - insert_time > self._replay_ttl:
                self._seen_queue.popleft()
                self._seen.discard(nonce)
            else:
                break

    @staticmethod
    def build_header(body: bytes, secret: str, timestamp: Optional[int] = None) -> str:
        """Helper to build a valid webhook header (used in tests / sending)."""
        if timestamp is None:
            timestamp = int(time.time())
        signed_payload  = f"{timestamp}.".encode() + body
        sig = hmac.new(secret.encode(), signed_payload, hashlib.sha256).hexdigest()
        return f"t={timestamp},v1={sig}"


# ---------------------------------------------------------------------------
# Test Cases
# ---------------------------------------------------------------------------

def test_valid_webhook():
    secret  = "whsec_test_supersecret_key"
    body    = b'{"event":"payment.completed","amount":100}'
    verifier = WebhookVerifier([secret])
    header   = WebhookVerifier.build_header(body, secret)
    result   = verifier.verify(body, header)
    assert result.valid, f"Expected valid: {result.failure_reason}"
    assert result.key_version == 0
    print(f"Valid webhook: passed (ts={result.timestamp})")


def test_signature_mismatch():
    secret  = "correct_secret"
    body    = b'{"event":"test"}'
    verifier = WebhookVerifier([secret])
    header   = WebhookVerifier.build_header(body, "wrong_secret")
    result   = verifier.verify(body, header)
    assert not result.valid
    assert result.failure_reason == "signature_mismatch"
    print(f"Signature mismatch: passed")


def test_expired_timestamp():
    secret  = "my_secret"
    body    = b'{"event":"old"}'
    old_ts  = int(time.time()) - 600    # 10 minutes ago
    header  = WebhookVerifier.build_header(body, secret, timestamp=old_ts)
    verifier = WebhookVerifier([secret], max_age_s=300)
    result   = verifier.verify(body, header)
    assert not result.valid
    assert "too_old" in result.failure_reason
    print(f"Expired timestamp: passed ({result.failure_reason})")


def test_replay_detection():
    secret  = "replay_secret"
    body    = b'{"event":"charge"}'
    verifier = WebhookVerifier([secret])
    header   = WebhookVerifier.build_header(body, secret)
    r1 = verifier.verify(body, header)
    assert r1.valid
    r2 = verifier.verify(body, header)   # Same header — replay
    assert not r2.valid
    assert r2.failure_reason == "replay_detected"
    print(f"Replay detection: passed")


def test_key_rotation():
    old_secret = "old_key"
    new_secret = "new_key"
    body = b'{"event":"subscription.created"}'
    # Signed with old key
    header = WebhookVerifier.build_header(body, old_secret)
    # Verifier accepts both
    verifier = WebhookVerifier([new_secret, old_secret])
    result   = verifier.verify(body, header)
    assert result.valid
    assert result.key_version == 1   # matched old_key at index 1
    print(f"Key rotation: passed (matched key_version={result.key_version})")


def test_invalid_header():
    verifier = WebhookVerifier(["secret"])
    result   = verifier.verify(b"body", "not_a_valid_header")
    assert not result.valid
    assert result.failure_reason == "invalid_header"
    print(f"Invalid header: passed")


def test_body_tampering():
    secret   = "tamper_secret"
    body     = b'{"amount":100}'
    header   = WebhookVerifier.build_header(body, secret)
    tampered = b'{"amount":99999}'
    verifier = WebhookVerifier([secret])
    result   = verifier.verify(tampered, header)
    assert not result.valid
    print(f"Body tampering detected: passed")

"""
WEBHOOK SECURITY THREAT MODEL
--------------------------------
Webhooks are server-to-server callbacks.  Threats:
  1. SPOOFED WEBHOOKS: attacker sends a fake webhook to trigger actions
     (e.g., fake "payment.completed" to release goods without payment).
     Fix: HMAC signature verification.

  2. REPLAY ATTACKS: attacker captures a legitimate webhook and resends it.
     E.g., replay "order.shipped" triggers duplicate fulfillment.
     Fix: Timestamp freshness (5-minute window) + nonce tracking.

  3. TIMING ATTACKS ON SIGNATURE COMPARISON:
     if sig == expected: leaks timing information.
     Fix: hmac.compare_digest.

  4. MAN-IN-THE-MIDDLE:
     If delivered over HTTP, payload and signature visible.
     Fix: Enforce HTTPS on webhook endpoint; validate TLS certificate.

  5. CLOCK SKEW:
     Sender and receiver clocks differ by more than tolerance.
     Fix: Use NTP; 5-minute window accommodates typical skew.

STRIPE WEBHOOK HEADER FORMAT (REAL-WORLD REFERENCE)
-----------------------------------------------------
  Stripe-Signature: t=1492774577,v1=5257a869...,v0=6ffbb59b...
  v1 = HMAC-SHA256(secret, f"{t}.{raw_body}")
  v0 = legacy MD5 (deprecated); our implementation only validates v1.

PRODUCTION CHECKLIST
--------------------
  ✓ Always verify signature before processing
  ✓ Use constant-time comparison
  ✓ Check timestamp freshness
  ✓ Deduplicate by event ID (webhook_id field) not just timestamp+sig
  ✓ Log all verification failures with body hash (not body) for forensics
  ✓ Respond with 200 quickly; process async to avoid timeout retries
"""
