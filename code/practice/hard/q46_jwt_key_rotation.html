"""
=============================================================================
Q46 [Hard] — JWT Key Rotation Validator
=============================================================================

PROBLEM
-------
Validate JWTs using a JWKS (JSON Web Key Set) with:
  1. Key ID (kid) lookup from the JWKS.
  2. In-memory JWKS cache with configurable TTL.
  3. Algorithm enforcement (reject "none", restrict to RS256/ES256).
  4. Claim validation: exp, nbf, iss, aud.
  5. Graceful key-not-found handling (don't crash; try cache refresh once).
  6. Stub JWKS fetching (no actual HTTP; simulated for testing).

CLARIFYING QUESTIONS
--------------------
1. Which algorithms?
   → RS256 and ES256 only; reject "none", HS256 with public key, and others.
2. What claims must be validated?
   → exp (required), nbf (if present), iss (if configured), aud (if configured).
3. JWKS refresh?
   → Cache with TTL; on kid-not-found, refresh once before failing.
4. External HTTP calls?
   → Abstract behind a JwksFetcher interface; stub in tests.
5. Actual crypto?
   → Use cryptography library if available; otherwise stub validation for
     stdlib-only mode and note production requires it.

=============================================================================
APPROACH 1 — NAIVE (no kid check, accepts "none" algorithm)
=============================================================================
  Decode JWT with no signature verification or algorithm enforcement.
  Vulnerable to: algorithm confusion, unsigned tokens, expired tokens.

=============================================================================
APPROACH 2 — JWKS-BACKED VALIDATOR WITH CACHE AND ALGORITHM ENFORCEMENT
=============================================================================
  1. Decode header without verification → extract kid + alg.
  2. Enforce algorithm allowlist.
  3. Lookup key from cache; refresh if needed.
  4. Verify signature (using cryptography lib; stubbed here).
  5. Validate exp, nbf, iss, aud.
  6. Return structured result.

=============================================================================
"""

from __future__ import annotations

import base64
import hashlib
import json
import time
from dataclasses import dataclass, field
from typing import Any, Callable, Dict, List, Optional, Set


# ---------------------------------------------------------------------------
# Config
# ---------------------------------------------------------------------------

ALLOWED_ALGORITHMS: Set[str] = {"RS256", "ES256"}
JWKS_CACHE_TTL_S   = 3600      # 1 hour
MAX_CLOCK_SKEW_S   = 30        # allow 30s clock skew on exp/nbf
TOKEN_MAX_AGE_S    = 86400     # reject tokens older than 24h even if not expired


# ---------------------------------------------------------------------------
# Data Types
# ---------------------------------------------------------------------------

@dataclass
class JwksKey:
    kid: str
    alg: str
    kty: str           # "RSA" or "EC"
    use: str           # "sig"
    # Raw key material (simplified; real JWKS has n, e, x, y etc.)
    raw: Dict[str, Any] = field(default_factory=dict)


@dataclass
class JwtClaims:
    sub: Optional[str] = None
    iss: Optional[str] = None
    aud: Optional[Any] = None    # str or list
    exp: Optional[int] = None
    nbf: Optional[int] = None
    iat: Optional[int] = None
    jti: Optional[str] = None
    extra: Dict[str, Any] = field(default_factory=dict)


@dataclass
class JwtValidationResult:
    valid: bool
    claims: Optional[JwtClaims] = None
    kid: Optional[str] = None
    alg: Optional[str] = None
    failure_reason: str = ""


class JwtValidationError(Exception):
    pass


# ---------------------------------------------------------------------------
# JWT utility (stdlib-only; no external library)
# ---------------------------------------------------------------------------

def _b64url_decode(s: str) -> bytes:
    """Decode base64url without padding."""
    s += "=" * (-len(s) % 4)
    return base64.urlsafe_b64decode(s)


def _decode_part(s: str) -> dict:
    return json.loads(_b64url_decode(s))


def _decode_header_no_verify(token: str) -> Dict[str, Any]:
    """Decode JWT header without signature verification."""
    parts = token.split(".")
    if len(parts) != 3:
        raise JwtValidationError("Invalid JWT format: expected 3 parts")
    return _decode_part(parts[0])


def _decode_payload_no_verify(token: str) -> Dict[str, Any]:
    parts = token.split(".")
    return _decode_part(parts[1])


# ---------------------------------------------------------------------------
# JWKS Fetcher interface + stub
# ---------------------------------------------------------------------------

class JwksFetcher:
    """Abstract interface for fetching JWKS."""

    def fetch(self, jwks_uri: str) -> List[JwksKey]:
        raise NotImplementedError


class StubJwksFetcher(JwksFetcher):
    """Returns a pre-configured list of keys; used in tests."""

    def __init__(self, keys: List[JwksKey], fail: bool = False):
        self._keys = keys
        self._fail = fail

    def fetch(self, jwks_uri: str) -> List[JwksKey]:
        if self._fail:
            raise RuntimeError("JWKS fetch failed")
        return list(self._keys)


# ---------------------------------------------------------------------------
# JWKS Cache
# ---------------------------------------------------------------------------

class JwksCache:
    def __init__(self, fetcher: JwksFetcher, ttl_s: float = JWKS_CACHE_TTL_S):
        self._fetcher  = fetcher
        self._ttl_s    = ttl_s
        self._keys: Dict[str, JwksKey] = {}
        self._fetched_at: float = 0.0

    def get_key(self, jwks_uri: str, kid: str) -> Optional[JwksKey]:
        if self._is_stale():
            self._refresh(jwks_uri)
        key = self._keys.get(kid)
        if key is None and not self._is_fresh_enough():
            # Try refresh once
            self._refresh(jwks_uri)
            key = self._keys.get(kid)
        return key

    def _is_stale(self) -> bool:
        return (time.monotonic() - self._fetched_at) > self._ttl_s

    def _is_fresh_enough(self) -> bool:
        return (time.monotonic() - self._fetched_at) < 10.0

    def _refresh(self, jwks_uri: str) -> None:
        try:
            keys = self._fetcher.fetch(jwks_uri)
            self._keys = {k.kid: k for k in keys}
            self._fetched_at = time.monotonic()
        except Exception:
            pass  # Keep stale cache on error


# ---------------------------------------------------------------------------
# APPROACH 1 — Naive (no verification)
# ---------------------------------------------------------------------------

class NaiveJwtValidator:
    """
    Decodes JWT with no verification — vulnerable to all JWT attacks.
    NEVER USE IN PRODUCTION.
    """

    def validate(self, token: str) -> Dict[str, Any]:
        parts = token.split(".")
        if len(parts) != 3:
            return {}
        return json.loads(_b64url_decode(parts[1]))


# ---------------------------------------------------------------------------
# APPROACH 2 — Secure JWT validator
# ---------------------------------------------------------------------------

class JwtValidator:
    """
    Validates JWT using JWKS with:
      - Algorithm enforcement
      - kid lookup + cache with refresh
      - Claim validation (exp, nbf, iss, aud)
      - Signature verification (stubbed; real impl uses cryptography library)
    """

    def __init__(
        self,
        jwks_uri: str,
        cache: JwksCache,
        allowed_algorithms: Optional[Set[str]] = None,
        expected_issuer: Optional[str] = None,
        expected_audience: Optional[str] = None,
        verify_signature: Optional[Callable[[str, JwksKey], bool]] = None,
        max_clock_skew_s: int = MAX_CLOCK_SKEW_S,
    ):
        self._jwks_uri     = jwks_uri
        self._cache        = cache
        self._allowed_algs = allowed_algorithms or ALLOWED_ALGORITHMS
        self._issuer       = expected_issuer
        self._audience     = expected_audience
        # Signature verifier: injected for testability;
        # production: use cryptography lib
        self._verify_sig   = verify_signature or self._stub_verify
        self._skew         = max_clock_skew_s

    def validate(self, token: str) -> JwtValidationResult:
        try:
            return self._validate_internal(token)
        except JwtValidationError as e:
            return JwtValidationResult(valid=False, failure_reason=str(e))
        except Exception as e:
            return JwtValidationResult(valid=False, failure_reason=f"internal: {e}")

    def _validate_internal(self, token: str) -> JwtValidationResult:
        # 1. Parse structure
        parts = token.split(".")
        if len(parts) != 3:
            raise JwtValidationError("invalid_jwt_format")

        # 2. Decode header
        try:
            header = _decode_part(parts[0])
        except Exception:
            raise JwtValidationError("header_decode_error")

        alg = header.get("alg", "")
        kid = header.get("kid", "")

        # 3. Algorithm enforcement
        if not alg:
            raise JwtValidationError("missing_algorithm")
        if alg.lower() == "none":
            raise JwtValidationError("algorithm_none_rejected")
        if alg not in self._allowed_algs:
            raise JwtValidationError(f"algorithm_not_allowed: {alg}")
        if not kid:
            raise JwtValidationError("missing_kid")

        # 4. Key lookup
        key = self._cache.get_key(self._jwks_uri, kid)
        if key is None:
            raise JwtValidationError(f"unknown_kid: {kid}")

        # 5. Algorithm consistency: JWKS key alg must match header alg
        if key.alg and key.alg != alg:
            raise JwtValidationError(
                f"algorithm_mismatch: header={alg}, jwks={key.alg}"
            )

        # 6. Signature verification
        if not self._verify_sig(token, key):
            raise JwtValidationError("signature_invalid")

        # 7. Decode payload
        try:
            payload = _decode_part(parts[1])
        except Exception:
            raise JwtValidationError("payload_decode_error")

        # 8. Claim validation
        now = int(time.time())

        exp = payload.get("exp")
        if exp is None:
            raise JwtValidationError("missing_exp_claim")
        if not isinstance(exp, int):
            raise JwtValidationError("invalid_exp_type")
        if now > exp + self._skew:
            raise JwtValidationError(f"token_expired: exp={exp}, now={now}")

        nbf = payload.get("nbf")
        if nbf is not None:
            if not isinstance(nbf, int):
                raise JwtValidationError("invalid_nbf_type")
            if now < nbf - self._skew:
                raise JwtValidationError(f"token_not_yet_valid: nbf={nbf}")

        iat = payload.get("iat")
        if iat is not None and isinstance(iat, int):
            if now - iat > TOKEN_MAX_AGE_S:
                raise JwtValidationError(f"token_too_old: iat={iat}")

        iss = payload.get("iss")
        if self._issuer and iss != self._issuer:
            raise JwtValidationError(f"issuer_mismatch: got={iss!r}")

        aud = payload.get("aud")
        if self._audience:
            aud_list = aud if isinstance(aud, list) else [aud]
            if self._audience not in aud_list:
                raise JwtValidationError(f"audience_mismatch: got={aud!r}")

        claims = JwtClaims(
            sub=payload.get("sub"),
            iss=iss,
            aud=aud,
            exp=exp,
            nbf=nbf,
            iat=iat,
            jti=payload.get("jti"),
            extra={k: v for k, v in payload.items()
                   if k not in ("sub", "iss", "aud", "exp", "nbf", "iat", "jti")},
        )
        return JwtValidationResult(valid=True, claims=claims, kid=kid, alg=alg)

    @staticmethod
    def _stub_verify(token: str, key: JwksKey) -> bool:
        """
        Stub: in production use cryptography.hazmat.primitives.asymmetric
        to verify the RS256 or ES256 signature.
        Here we trust a special test marker in key.raw["_stub_valid"].
        """
        return key.raw.get("_stub_valid", True)


# ---------------------------------------------------------------------------
# JWT builder for tests (minimal; no real signing)

def _b64url_encode(data: bytes) -> str:
    return base64.urlsafe_b64encode(data).rstrip(b"=").decode()


def _build_test_token(
    kid: str,
    alg: str = "RS256",
    exp_offset: int = 3600,
    extra_claims: Optional[Dict[str, Any]] = None,
    iss: str = "https://auth.example.com",
    aud: str = "api.example.com",
) -> str:
    """Build a minimal JWT for testing (signature is placeholder)."""
    now = int(time.time())
    header  = {"alg": alg, "kid": kid, "typ": "JWT"}
    payload = {
        "sub": "user123", "iss": iss, "aud": aud,
        "exp": now + exp_offset, "nbf": now - 5, "iat": now,
        "jti": "test-jti-001",
    }
    if extra_claims:
        payload.update(extra_claims)
    h = _b64url_encode(json.dumps(header).encode())
    p = _b64url_encode(json.dumps(payload).encode())
    # Fake signature (stub verifier will accept it)
    s = _b64url_encode(b"fake_signature_placeholder")
    return f"{h}.{p}.{s}"

def _make_validator(
    kid: str = "key-001",
    alg: str = "RS256",
    stub_valid: bool = True,
    issuer: str = "https://auth.example.com",
    audience: str = "api.example.com",
) -> JwtValidator:
    key = JwksKey(kid=kid, alg=alg, kty="RSA", use="sig",
                  raw={"_stub_valid": stub_valid})
    fetcher = StubJwksFetcher([key])
    cache   = JwksCache(fetcher, ttl_s=3600)
    return JwtValidator(
        jwks_uri="https://auth.example.com/.well-known/jwks.json",
        cache=cache,
        expected_issuer=issuer,
        expected_audience=audience,
    )


def test_valid_token():
    v     = _make_validator()
    token = _build_test_token("key-001")
    r     = v.validate(token)
    assert r.valid, r.failure_reason
    assert r.claims.sub == "user123"
    print(f"Valid token: passed")


def test_algorithm_none_rejected():
    v     = _make_validator()
    token = _build_test_token("key-001", alg="none")
    r     = v.validate(token)
    assert not r.valid
    assert "none" in r.failure_reason
    print(f"Algorithm none rejected: passed ({r.failure_reason})")


def test_algorithm_hs256_rejected():
    v     = _make_validator()

    assert "not_allowed" in r.failure_reason
    print(f"HS256 rejected: passed")

    v     = _make_validator()

# ---------------------------------------------------------------------------
# SECURITY ANGLE
# ---------------------------------------------------------------------------
"""
JWT ATTACK VECTORS
--------------------
1. ALGORITHM CONFUSION (CVE-2015-9235 and variants):
   Attacker changes header "alg":"RS256" → "alg":"HS256".
   Server using the public RSA key as an HMAC secret will verify correctly.
   Fix: enforce allowed algorithm list server-side; ignore client header alg.

2. "ALG":"NONE" BYPASS:
   JWT spec (early) allowed "none" to skip signing.
   Many libraries accepted unsigned tokens.
   Fix: reject "none" before any other processing.

3. KID INJECTION:
   "kid":"../../dev/null" or "kid":"; DROP TABLE keys;"
   If kid is used in a file path or SQL query without sanitisation.
   Fix: lookup kid in a safe in-memory map; never use in filesystem paths.

4. EXP MANIPULATION:
   Attacker removes or sets exp = year 9999.
   Fix: require exp; enforce TOKEN_MAX_AGE_S even if exp is far in future.

5. JWKS SPOOFING (kid-matching attack):
   Attacker embeds their own public key in the JWT header as a "jku" URL.
   Fix: never follow jku/x5u URLs from the token header; use pinned JWKS URI.

KEY ROTATION HANDLING
---------------------
  JWKS supports multiple keys simultaneously (each with a unique kid).
  Rotate: add new key to JWKS, update signing service, wait TTL,
  remove old key. Clients transparently pick up the new kid.
  Our cache refreshes on kid-not-found to handle rotation latency.
"""
