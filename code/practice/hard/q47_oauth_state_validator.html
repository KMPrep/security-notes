"""
=============================================================================
Q47 [Hard] — OAuth State Parameter Validator
=============================================================================

PROBLEM
-------
Implement OAuth 2.0 Authorization Code flow callback validation with:
  1. State parameter generation (cryptographically random, bound to session).
  2. State validation: verify state matches session-stored value.
  3. Nonce validation (OIDC).
  4. Redirect URI validation against allowlist.
  5. Expiry: state tokens expire after MAX_AGE_S.
  6. Single-use: state invalidated after first use.
  7. PKCE (code_challenge/code_verifier) verification.

CLARIFYING QUESTIONS
--------------------
1. State format?
   → Random 32-byte URL-safe base64 + HMAC for integrity.
2. State bound to session?
   → Yes; store in server-side session by session_id.
3. PKCE method?
   → S256 only (code_challenge = BASE64URL(SHA256(code_verifier))).
4. Redirect URI matching?
   → Exact match against pre-registered allowlist; no wildcard.
5. Multiple providers?
   → Yes; state binds to a provider (iss) as well.

=============================================================================
APPROACH 1 — NAIVE
=============================================================================
  State = random token; no HMAC, no expiry, no single-use, wildcard redirect.

=============================================================================
APPROACH 2 — FULL OAUTH CALLBACK VALIDATOR (OPTIMAL)
=============================================================================
  1. generate_state() → (state_token, state_id) stored in server session.
  2. validate_callback(session_id, state, code, redirect_uri) validates all.
  3. PKCE: store code_challenge on auth request; verify on callback.
  4. Nonce: store nonce on auth request; verify in ID token claims.

=============================================================================
"""

from __future__ import annotations

import base64
import hashlib
import hmac
import os
import secrets
import time
from dataclasses import dataclass, field
from typing import Dict, List, Optional, Set


# ---------------------------------------------------------------------------
# Config
# ---------------------------------------------------------------------------

STATE_BYTES          = 32
STATE_MAX_AGE_S      = 600    # 10 minutes
PKCE_VERIFIER_MIN    = 43     # RFC 7636: 43–128 chars
PKCE_VERIFIER_MAX    = 128
NONCE_BYTES          = 16


# ---------------------------------------------------------------------------
# Data Types
# ---------------------------------------------------------------------------

@dataclass
class OAuthSession:
    session_id: str
    state: str
    state_hmac: str
    code_challenge: Optional[str]        # BASE64URL(SHA256(verifier))
    nonce: Optional[str]
    redirect_uri: str
    provider: str
    created_at: float
    used: bool = False


@dataclass
class OAuthCallbackResult:
    valid: bool
    session_id: str
    nonce: Optional[str] = None
    failure_reason: str = ""


# ---------------------------------------------------------------------------
# APPROACH 1 — Naive
# ---------------------------------------------------------------------------

class NaiveOAuthValidator:
    """Stores state as plaintext; no expiry, no PKCE, wildcard redirect."""

    def __init__(self):
        self._pending: Dict[str, str] = {}   # session_id → state

    def generate_state(self, session_id: str) -> str:
        state = secrets.token_urlsafe(16)
        self._pending[session_id] = state
        return state

    def validate(self, session_id: str, state: str, redirect_uri: str) -> bool:
        return self._pending.get(session_id) == state  # timing-vulnerable


# ---------------------------------------------------------------------------
# APPROACH 2 — Secure OAuth validator
# ---------------------------------------------------------------------------

_HMAC_KEY = os.urandom(32)   # Server-side secret for HMAC integrity on state


class OAuthStateManager:
    """
    Manages OAuth 2.0 Authorization Code flow with PKCE and nonce.
    """

    def __init__(
        self,
        allowed_redirect_uris: List[str],
        state_max_age_s: int = STATE_MAX_AGE_S,
        hmac_key: Optional[bytes] = None,
    ):
        self._allowed_uris: Set[str] = set(allowed_redirect_uris)
        self._max_age_s    = state_max_age_s
        self._hmac_key     = hmac_key or _HMAC_KEY
        # session_id → OAuthSession
        self._sessions: Dict[str, OAuthSession] = {}

    # --- Begin authorization flow -------------------------------------------

    def begin_authorization(
        self,
        session_id: str,
        redirect_uri: str,
        provider: str = "default",
        use_pkce: bool = True,
        use_nonce: bool = True,
    ) -> Dict[str, str]:
        """
        Generates state (and optionally code_challenge + nonce).
        Returns dict of query params to include in the authorization URL.
        """
        if redirect_uri not in self._allowed_uris:
            raise ValueError(f"Redirect URI not in allowlist: {redirect_uri!r}")

        raw_state  = secrets.token_urlsafe(STATE_BYTES)
        state_hmac = self._compute_hmac(raw_state, session_id)

        code_verifier  = None
        code_challenge = None
        if use_pkce:
            code_verifier  = secrets.token_urlsafe(PKCE_VERIFIER_MIN)[:PKCE_VERIFIER_MAX]
            code_challenge = self._s256(code_verifier)

        nonce = secrets.token_hex(NONCE_BYTES) if use_nonce else None

        session = OAuthSession(
            session_id=session_id,
            state=raw_state,
            state_hmac=state_hmac,
            code_challenge=code_challenge,
            nonce=nonce,
            redirect_uri=redirect_uri,
            provider=provider,
            created_at=time.monotonic(),
        )
        self._sessions[session_id] = session

        params: Dict[str, str] = {
            "state": raw_state,
            "redirect_uri": redirect_uri,
        }
        if code_challenge:
            params["code_challenge"] = code_challenge
            params["code_challenge_method"] = "S256"
        if nonce:
            params["nonce"] = nonce
        # code_verifier is stored server-side; NOT sent in redirect
        if code_verifier:
            # In a real implementation, store code_verifier in the session
            # and send it in the token exchange step.
            session.code_challenge = code_challenge  # already set above
            params["_code_verifier_for_test"] = code_verifier
        return params

    # --- Validate callback ---------------------------------------------------

    def validate_callback(
        self,
        session_id: str,
        returned_state: str,
        redirect_uri: str,
        code_verifier: Optional[str] = None,
    ) -> OAuthCallbackResult:
        """
        Validate the OAuth callback.
        """
        fail = lambda reason: OAuthCallbackResult(
            valid=False, session_id=session_id, failure_reason=reason
        )

        session = self._sessions.get(session_id)
        if session is None:
            return fail("session_not_found")

        if session.used:
            return fail("state_already_used")

        # Expiry check
        age = time.monotonic() - session.created_at
        if age > self._max_age_s:
            del self._sessions[session_id]
            return fail(f"state_expired: age={age:.0f}s")

        # Redirect URI exact match
        if redirect_uri != session.redirect_uri:
            return fail(f"redirect_uri_mismatch")

        # State comparison (constant-time via HMAC)
        expected_hmac = self._compute_hmac(returned_state, session_id)
        if not hmac.compare_digest(expected_hmac, session.state_hmac):
            return fail("state_mismatch")

        # Double check raw state
        if not hmac.compare_digest(returned_state, session.state):
            return fail("state_raw_mismatch")

        # PKCE verification
        if session.code_challenge is not None:
            if code_verifier is None:
                return fail("pkce_verifier_missing")
            if not (PKCE_VERIFIER_MIN <= len(code_verifier) <= PKCE_VERIFIER_MAX):
                return fail(f"pkce_verifier_length: {len(code_verifier)}")
            computed_challenge = self._s256(code_verifier)
            if not hmac.compare_digest(computed_challenge, session.code_challenge):
                return fail("pkce_challenge_mismatch")

        # Mark used (single-use)
        session.used = True

        return OAuthCallbackResult(
            valid=True,
            session_id=session_id,
            nonce=session.nonce,
        )

    # --- Helpers ------------------------------------------------------------

    def _compute_hmac(self, state: str, session_id: str) -> str:
        msg = f"{session_id}:{state}".encode()
        return hmac.new(self._hmac_key, msg, hashlib.sha256).hexdigest()

    @staticmethod
    def _s256(verifier: str) -> str:
        digest = hashlib.sha256(verifier.encode("ascii")).digest()
        return base64.urlsafe_b64encode(digest).rstrip(b"=").decode()

ALLOWED_URIS = [
    "https://app.example.com/callback",
    "https://app.example.com/auth/callback",
]


def test_valid_flow():
    mgr  = OAuthStateManager(ALLOWED_URIS)
    sess = "session-abc123"
    redirect = "https://app.example.com/callback"

    params = mgr.begin_authorization(sess, redirect)
    r = mgr.validate_callback(
        session_id=sess,
        returned_state=params["state"],
        redirect_uri=redirect,
        code_verifier=params.get("_code_verifier_for_test"),
    )
    assert r.valid, r.failure_reason
    assert r.nonce is not None
    print(f"Valid OAuth flow: passed (nonce={r.nonce[:8]}...)")


def test_csrf_state_mismatch():
    mgr  = OAuthStateManager(ALLOWED_URIS)
    sess = "session-csrf"
    redirect = "https://app.example.com/callback"
    mgr.begin_authorization(sess, redirect)
    r = mgr.validate_callback(
        session_id=sess,
        returned_state="attacker_forged_state",
        redirect_uri=redirect,
    )
    assert not r.valid
    assert "mismatch" in r.failure_reason or "challenge" in r.failure_reason
    print(f"CSRF state rejected: passed ({r.failure_reason})")


def test_redirect_uri_mismatch():
    mgr  = OAuthStateManager(ALLOWED_URIS)
    sess = "session-redirect"
    redirect = "https://app.example.com/callback"
    params = mgr.begin_authorization(sess, redirect, use_pkce=False, use_nonce=False)
    r = mgr.validate_callback(
        session_id=sess,
        returned_state=params["state"],
        redirect_uri="https://evil.attacker.com/steal",
    )
    assert not r.valid
    assert "redirect_uri" in r.failure_reason
    print(f"Redirect URI mismatch: passed")


def test_disallowed_redirect_uri():
    mgr = OAuthStateManager(ALLOWED_URIS)
    try:
        mgr.begin_authorization("s1", "https://evil.com/phish", use_pkce=False)
        assert False
    except ValueError as e:
        print(f"Disallowed redirect URI: passed ({e})")


def test_single_use_state():
    mgr  = OAuthStateManager(ALLOWED_URIS)
    sess = "session-single"
    redirect = "https://app.example.com/callback"
    params = mgr.begin_authorization(sess, redirect, use_pkce=False, use_nonce=False)
    mgr.validate_callback(sess, params["state"], redirect)
    r2 = mgr.validate_callback(sess, params["state"], redirect)
    assert not r2.valid
    assert r2.failure_reason == "state_already_used"
    print(f"Single-use state: passed")


def test_expired_state():
    mgr  = OAuthStateManager(ALLOWED_URIS, state_max_age_s=0)
    sess = "session-expire"
    redirect = "https://app.example.com/callback"
    params = mgr.begin_authorization(sess, redirect, use_pkce=False, use_nonce=False)
    import time; time.sleep(0.01)
    r = mgr.validate_callback(sess, params["state"], redirect)
    assert not r.valid
    assert "expired" in r.failure_reason
    print(f"Expired state: passed ({r.failure_reason})")


def test_pkce_verifier_missing():
    mgr  = OAuthStateManager(ALLOWED_URIS)
    sess = "session-pkce"
    redirect = "https://app.example.com/callback"
    params = mgr.begin_authorization(sess, redirect, use_pkce=True, use_nonce=False)
    r = mgr.validate_callback(
        session_id=sess,
        returned_state=params["state"],
        redirect_uri=redirect,
        code_verifier=None,  # Missing verifier
    )
    assert not r.valid
    assert "pkce" in r.failure_reason
    print(f"PKCE verifier missing: passed")


# ---------------------------------------------------------------------------
# SECURITY ANGLE
# ---------------------------------------------------------------------------
"""
OAUTH CSRF AND STATE PARAMETER
--------------------------------
The OAuth state parameter is the CSRF token for the authorization flow.

CSRF ATTACK WITHOUT STATE
--------------------------
1. Attacker starts their own OAuth flow; gets an authorization code.
2. Instead of using it, embeds the callback URL in the victim's browser
   (via img src, XSS, link).
3. Victim's browser sends the callback to the app → app links the
   attacker's OAuth account to the victim's session.
   Result: attacker controls the victim's account.
4. Fix: bind state to session; validate on callback.

OPEN REDIRECT IN REDIRECT_URI
-------------------------------
If redirect_uri is not strictly validated:
  redirect_uri=https://app.example.com/callback/../phish
  redirect_uri=https://app.example.com%0a.attacker.com/steal
Attacker receives authorization code → exchanges for access token.
Fix: exact match against pre-registered allowlist; no path-relative matching.

PKCE (PROOF KEY FOR CODE EXCHANGE — RFC 7636)
---------------------------------------------
Protects public clients (SPAs, mobile apps) from authorization code
interception:
1. App generates random code_verifier (43–128 chars).
2. Sends code_challenge = BASE64URL(SHA256(verifier)) in auth request.
3. On token exchange, sends verifier.
4. Server verifies challenge = S256(verifier) before issuing token.
Without PKCE: malicious app on same device intercepts the redirect;
exchanges code for tokens. With PKCE: can't compute verifier from challenge.

NONCE (OIDC REPLAY PREVENTION)
-------------------------------
Nonce embedded in ID token prevents replay:
  Attacker captures an ID token, replays it to the app.
  If nonce is checked, replay fails (nonce already consumed).
"""
