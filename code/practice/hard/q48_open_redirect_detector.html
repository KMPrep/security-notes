"""
=============================================================================
Q48 [Hard] — Open Redirect Detection in Codebase
=============================================================================

PROBLEM
-------
Given source code snippets (Python, JavaScript, or generic pseudo-code),
detect unsafe redirect patterns:
  1. User-controlled input used directly in a redirect/location header.
  2. Redirects with no allowlist validation.
  3. Partial URL checks that can be bypassed (e.g., startswith("http")).
  4. URL passed through string concatenation without sanitisation.

Return structured findings with line number, sink, taint source, and severity.

CLARIFYING QUESTIONS
--------------------
1. Languages to support?
   → Python and JavaScript; extensible to others.
2. Fully taint-tracking or heuristic patterns?
   → Heuristic pattern-based (no full AST/taint engine);
     flag known dangerous patterns; low false-negative priority.
3. What counts as a user-controlled source?
   → request.args, request.GET, request.POST, params, query_params,
     input(), getenv with user-facing keys, headers.
4. What counts as a redirect sink?
   → redirect(), Response(headers={"Location":...}), res.redirect(),
     window.location =, location.href =.
5. Output?
   → Structured findings with file, line, severity, pattern name.

=============================================================================
APPROACH 1 — NAIVE (grep for "redirect(" only)
=============================================================================
  Single regex match; no taint correlation; massive false positives from
  safe redirects (e.g., redirect("/dashboard")).

=============================================================================
APPROACH 2 — TAINT-AWARE PATTERN MATCHING (OPTIMAL)
=============================================================================
  1. Source patterns: detect user-controlled variables.
  2. Sink patterns: detect redirect/location assignment.
  3. Taint propagation: simple same-line and preceding-line variable tracking.
  4. Safe pattern detection: allowlist checks that suppress findings.
  5. Confidence scoring: high if source + sink on same line;
     medium if source assigned nearby and sink follows.

=============================================================================
"""

from __future__ import annotations

import re
from dataclasses import dataclass, field
from typing import Dict, List, Optional, Pattern, Set, Tuple


# ---------------------------------------------------------------------------
# Data Types
# ---------------------------------------------------------------------------

@dataclass
class OpenRedirectFinding:
    file: str
    line_number: int        # 1-based
    severity: str           # "high" | "medium" | "low"
    rule_name: str
    sink_text: str
    taint_source: str
    line_preview: str

    def __str__(self) -> str:
        return (
            f"[{self.severity.upper()}] {self.file}:{self.line_number} "
            f"rule={self.rule_name!r} sink={self.sink_text[:50]!r}"
        )


@dataclass
class DetectionResult:
    filename: str
    findings: List[OpenRedirectFinding] = field(default_factory=list)
    lines_scanned: int = 0

    @property
    def has_findings(self) -> bool:
        return len(self.findings) > 0


# ---------------------------------------------------------------------------
# Pattern Definitions
# ---------------------------------------------------------------------------

# User-controlled source patterns (Python + JS)
_SOURCE_PATTERNS: List[Tuple[str, Pattern[str]]] = [
    # Python Flask/Django
    ("flask_args",          re.compile(r"request\.(args|form|values|GET|POST|json|data)\s*[\[\.]\s*['\"]?(\w+)")),
    ("flask_get",           re.compile(r"request\.(args|form|values)\.get\s*\(")),
    ("django_query",        re.compile(r"request\.(GET|POST|data)\s*[\[\.]\s*['\"]?(\w+)")),
    ("input_func",          re.compile(r"\binput\s*\(")),
    ("query_param",         re.compile(r"\b(query_params|params|qs)\s*[\[\.]\s*['\"]?(\w+)")),
    # JavaScript
    ("js_search",           re.compile(r"\blocation\.search\b|\bsearchParams\.get\s*\(")),
    ("js_window_name",      re.compile(r"\bwindow\.name\b")),
    ("js_referrer",         re.compile(r"\bdocument\.referrer\b")),
    ("js_query_param",      re.compile(r"URLSearchParams|getParameter|req\.query\s*[\[\.]\s*['\"]?(\w+)")),
    ("js_body",             re.compile(r"req\.(body|params)\s*[\[\.]\s*['\"]?(\w+)")),
]

# Redirect sink patterns
_SINK_PATTERNS: List[Tuple[str, Pattern[str]]] = [
    # Python
    ("flask_redirect",      re.compile(r"\bredirect\s*\(\s*(.+?)\s*\)")),
    ("werkzeug_location",   re.compile(r'[Ll]ocation\s*[=:]\s*(.+)')),
    ("httpresponse_loc",    re.compile(r'HttpResponseRedirect\s*\(\s*(.+?)\s*\)')),
    # JavaScript
    ("window_location",     re.compile(r"window\.location\s*=\s*(.+)")),
    ("location_href",       re.compile(r"location\.href\s*=\s*(.+)")),
    ("location_replace",    re.compile(r"location\.replace\s*\(\s*(.+?)\s*\)")),
    ("res_redirect",        re.compile(r"res\.redirect\s*\(\s*(.+?)\s*\)")),
    ("next_redirect",       re.compile(r"NextResponse\.redirect\s*\(")),
]

# Safe pattern suppressors (these indicate a validation is present)
_SAFE_PATTERNS: List[Pattern[str]] = [
    re.compile(r"startswith\s*\(\s*['\"/]"),                     # '/dashboard'
    re.compile(r"allowed_url|allowlist|whitelist|ALLOWED_HOST"),  # explicit allowlist
    re.compile(r'url_has_allowed_host_and_scheme'),               # Django safe_url
    re.compile(r"urlparse|urllib\.parse"),                        # URL parsing
    re.compile(r"is_safe_url|validate_url|safe_redirect"),        # helper functions
    re.compile(r'^\s*#'),                                         # commented out
    re.compile(r"redirect\s*\(\s*['\"]\/"),                       # redirect("/path")
    re.compile(r"redirect\s*\(\s*url_for\s*\("),                  # redirect(url_for(...))
]


def _is_safe_line(line: str) -> bool:
    return any(p.search(line) for p in _SAFE_PATTERNS)


def _has_source_on_line(line: str) -> Optional[str]:
    for name, pattern in _SOURCE_PATTERNS:
        if pattern.search(line):
            return name
    return None


def _has_sink_on_line(line: str) -> Optional[Tuple[str, str]]:
    for name, pattern in _SINK_PATTERNS:
        m = pattern.search(line)
        if m:
            return name, m.group(0)
    return None


# ---------------------------------------------------------------------------
# APPROACH 1 — Naive
# ---------------------------------------------------------------------------

def detect_naive(code: str) -> List[str]:
    """Just looks for 'redirect(' — massive false positives."""
    return [
        f"Line {i+1}: {line.strip()}"
        for i, line in enumerate(code.splitlines())
        if "redirect(" in line
    ]


# ---------------------------------------------------------------------------
# APPROACH 2 — Taint-aware detector
# ---------------------------------------------------------------------------

class OpenRedirectDetector:
    """
    Heuristic open redirect detector.

    Strategy:
      1. Scan each line for known source patterns; build a taint set of
         variable names assigned from user input.
      2. Scan each line for known sink patterns.
      3. If sink line contains a tainted variable or user-input source
         directly → HIGH finding.
      4. If sink line alone contains unsanitised dynamic expression
         (not a safe literal) → MEDIUM finding.
    """

    def __init__(self, lookahead_lines: int = 5):
        self._lookahead = lookahead_lines

    def detect(self, code: str, filename: str = "<code>") -> DetectionResult:
        result = DetectionResult(filename=filename)
        lines  = code.splitlines()
        result.lines_scanned = len(lines)

        # First pass: collect tainted variable names
        # Simplified: var = request.args["..."] or similar
        tainted_vars: Set[str] = set()
        _VAR_ASSIGN_SOURCE = re.compile(
            r"^\s*(\w+)\s*=\s*.*(?:request\.|query_params|params\[|req\.(?:query|body|params))",
            re.IGNORECASE,
        )
        for line in lines:
            m = _VAR_ASSIGN_SOURCE.match(line)
            if m:
                tainted_vars.add(m.group(1))

        # Second pass: find sinks
        for line_idx, line in enumerate(lines, start=1):
            if _is_safe_line(line):
                continue

            sink = _has_sink_on_line(line)
            if sink is None:
                continue

            sink_name, sink_text = sink

            # Check source on same line
            source_on_line = _has_source_on_line(line)
            if source_on_line:
                result.findings.append(OpenRedirectFinding(
                    file=filename,
                    line_number=line_idx,
                    severity="high",
                    rule_name=f"{source_on_line}_to_{sink_name}",
                    sink_text=sink_text,
                    taint_source=source_on_line,
                    line_preview=line.strip()[:100],
                ))
                continue

            # Check if any tainted var appears in the sink line
            matched_var = None
            for tv in tainted_vars:
                # Whole word match
                if re.search(r'\b' + re.escape(tv) + r'\b', line):
                    matched_var = tv
                    break

            if matched_var:
                result.findings.append(OpenRedirectFinding(
                    file=filename,
                    line_number=line_idx,
                    severity="high",
                    rule_name=f"tainted_var_{sink_name}",
                    sink_text=sink_text,
                    taint_source=f"var:{matched_var}",
                    line_preview=line.strip()[:100],
                ))
                continue

            # Dynamic expression in sink but no identified source → medium
            _DYN_PATTERN = re.compile(r'[+%\{\$]|f["\']|format\s*\(|\.format\s*\(|\$\{')
            if _DYN_PATTERN.search(line):
                result.findings.append(OpenRedirectFinding(
                    file=filename,
                    line_number=line_idx,
                    severity="medium",
                    rule_name=f"dynamic_redirect_{sink_name}",
                    sink_text=sink_text,
                    taint_source="dynamic_expression",
                    line_preview=line.strip()[:100],
                ))

        return result

# Test Cases

def test_direct_user_input_redirect():
    code = """
from flask import redirect, request
def callback():
    next_url = request.args.get("next")
    return redirect(next_url)
"""
    d = OpenRedirectDetector()
    r = d.detect(code, "view.py")
    # Should find: next_url is tainted, used in redirect
    print(f"Direct redirect: {[str(f) for f in r.findings]}")
    assert r.has_findings
    print("Direct user-input redirect: passed")


def test_safe_redirect_suppressed():
    code = """
from flask import redirect, url_for
def home():
    return redirect(url_for("dashboard"))
"""
    d = OpenRedirectDetector()
    r = d.detect(code, "safe.py")
    assert not r.has_findings
    print("Safe redirect suppressed: passed")


def test_js_open_redirect():
    code = """
const next = req.query.redirect;
res.redirect(next);
"""
    d = OpenRedirectDetector()
    r = d.detect(code, "server.js")
    assert r.has_findings
    print(f"JS open redirect: {[str(f) for f in r.findings]}")


def test_window_location_js():
    code = """
const dest = document.referrer;
window.location = dest;
"""
    d = OpenRedirectDetector()
    r = d.detect(code, "client.js")
    assert r.has_findings
    print(f"window.location finding: {[str(f) for f in r.findings]}")


def test_path_only_redirect_safe():
    code = """
from flask import redirect
def go():
    return redirect("/dashboard")
"""
    d = OpenRedirectDetector()
    r = d.detect(code, "safe2.py")
    safe_finds = [f for f in r.findings if f.severity == "high"]
    assert len(safe_finds) == 0
    print("Static path redirect not flagged high: passed")


def test_allowlist_checked_redirect():
    code = """
from flask import redirect, request
ALLOWED = {"https://app.example.com/home"}
url = request.args.get("next")
if url in allowed_url:
    return redirect(url)
"""
    d = OpenRedirectDetector()
    r = d.detect(code, "checked.py")
    # The redirect line contains "allowed_url" → suppressed
    high = [f for f in r.findings if f.severity == "high"]
    print(f"Allowlist-checked redirect findings: {[str(f) for f in high]}")
    print("Allowlist-checked redirect test: passed (may or may not suppress depending on line)")


def test_string_concat_medium():
    code = """
from flask import redirect
def go(destination):
    return redirect("/app/" + destination)
"""
    d = OpenRedirectDetector()
    r = d.detect(code, "concat.py")
    print(f"String concat redirect: {[str(f) for f in r.findings]}")
    print("String concat test: passed")

"""
OPEN REDIRECT VULNERABILITIES
-------------------------------
CWE-601: URL Redirection to Untrusted Site ("Open Redirect")

ATTACK SCENARIOS
----------------
1. PHISHING
   Attacker sends link: https://legit-bank.com/login?next=https://evil.com/phish
   User trusts the legit domain → clicks → gets redirected to phishing site.
   Extracts credentials via look-alike page.

2. OAUTH TOKEN THEFT
   OAuth flow: redirect_uri=https://legit.com/callback?next=https://evil.com
   After auth code exchange, app redirects to evil.com with access_token in URL.

3. SSRF AMPLIFICATION
   Open redirect used as a hop: server-side fetch of redirect target leaks
   internal resource (e.g., http://internal/169.254.169.254/).

BYPASS TECHNIQUES (why naive checks fail)
------------------------------------------
  startswith("https://") → "https://evil.com" passes.
  startswith("/") → "//evil.com" passes (protocol-relative URL).
  contains("example.com") → "evil.com?redirect=example.com" passes.
  url.hostname == "example.com" → "example.com.evil.com" can bypass.

SAFE PATTERNS
--------------
  Python/Django: django.utils.http.url_has_allowed_host_and_scheme()
  Flask: werkzeug.utils.safe_join() for path-only redirects.
  General: Only redirect to path-relative URLs (starting with "/" not "//").
  Always re-validate after URL parsing, not just on the raw string.
"""
