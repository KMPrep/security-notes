"""
=============================================================================
Q49 [Hard] — SQL Injection Static Pattern Detector
=============================================================================

PROBLEM
-------
Build a static analyzer that detects SQL injection vulnerabilities:
  1. String concatenation to build SQL queries.
  2. %-style or .format() interpolation directly into SQL strings.
  3. f-string SQL queries with variable interpolation.
  4. Taint analysis: user-controlled variable flows into SQL string.
  5. Flag parameterized queries as safe.

Languages: Python, JavaScript (Node/MySQL), Java-like pseudo-code.

CLARIFYING QUESTIONS
--------------------
1. Languages to support?
   → Python, JavaScript, Java; heuristic approach (no full AST).
2. What sources are user-controlled?
   → request.args/form/body/query, input(), sys.argv, headers.
3. What are SQL sinks?
   → cursor.execute, db.query, conn.execute, db.run, executeQuery.
4. What suppresses findings?
   → Parameterized queries (cursor.execute(sql, (param,))),
     ORM calls (Model.objects.filter), prepared statements.
5. Output?
   → Structured findings with file, line, severity, pattern name.

=============================================================================
APPROACH 1 — NAIVE (grep for "execute")
=============================================================================
  Flag every `execute(` line. High false positives; misses .format() patterns.

=============================================================================
APPROACH 2 — TAINT-AWARE PATTERN DETECTOR (OPTIMAL)
=============================================================================
  Pass 1: collect tainted variables (assigned from user input).
  Pass 2: detect SQL string construction patterns.
  Pass 3: detect SQL sinks; correlate with tainted strings.
  Safe pattern suppressors: parameterized query detection.

=============================================================================
"""

from __future__ import annotations

import re
from dataclasses import dataclass, field
from typing import Dict, List, Optional, Set, Tuple


# ---------------------------------------------------------------------------
# Data Types
# ---------------------------------------------------------------------------

@dataclass
class SqliFinding:
    file: str
    line_number: int
    severity: str          # "high" | "medium" | "low"
    rule_name: str
    evidence: str
    line_preview: str

    def __str__(self) -> str:
        return (
            f"[{self.severity.upper()}] {self.file}:{self.line_number} "
            f"rule={self.rule_name!r}: {self.evidence[:60]}"
        )


@dataclass
class ScanResult:
    filename: str
    findings: List[SqliFinding] = field(default_factory=list)
    lines_scanned: int = 0

    @property
    def has_findings(self) -> bool:
        return bool(self.findings)


# ---------------------------------------------------------------------------
# Pattern Definitions
# ---------------------------------------------------------------------------

# User-controlled source patterns
_SOURCE_PATTERNS: List[Tuple[str, re.Pattern]] = [
    ("flask_args",   re.compile(r"request\.(args|form|values|GET|POST|json)(\[|\.get\s*\()")),
    ("django_req",   re.compile(r"request\.(GET|POST|data)\s*[\[.]")),
    ("input_func",   re.compile(r"\binput\s*\(")),
    ("sys_argv",     re.compile(r"\bsys\.argv\b")),
    ("req_query",    re.compile(r"req\.(query|body|params)\s*[\[.]")),
    ("req_param",    re.compile(r"request\.getParameter\s*\(")),
    ("env_user",     re.compile(r"os\.environ\.get\s*\(['\"]USER|os\.getenv\s*\(['\"]USER")),
]

# SQL sinks
_SQL_SINKS: List[Tuple[str, re.Pattern]] = [
    ("cursor_execute",  re.compile(r"\bcursor\.execute\s*\(")),
    ("conn_execute",    re.compile(r"\bconn\.execute\s*\(")),
    ("db_execute",      re.compile(r"\bdb\.execute\s*\(")),
    ("db_query",        re.compile(r"\bdb\.(query|run)\s*\(")),
    ("execute_query",   re.compile(r"\bexecuteQuery\s*\(")),
    ("execute_update",  re.compile(r"\bexecuteUpdate\s*\(")),
    ("raw_query",       re.compile(r"\.raw\s*\(")),
    ("stmt_execute",    re.compile(r"\bstmt\.execute\s*\(")),
]

# SQL string construction patterns (dangerous)
_SQL_CONCAT_PATTERNS: List[Tuple[str, re.Pattern]] = [
    ("sql_fstring",     re.compile(r'\bf["\'].*\b(SELECT|INSERT|UPDATE|DELETE|WHERE|FROM)\b.*\{')),
    ("sql_format",      re.compile(r'["\'].*\b(SELECT|INSERT|UPDATE|DELETE|WHERE|FROM)\b.*["\'].*\.format\s*\(')),
    ("sql_percent",     re.compile(r'["\'].*\b(SELECT|INSERT|UPDATE|DELETE|WHERE|FROM)\b.*%\s*(?:\w|[(\[])')),
    ("sql_concat_plus", re.compile(r'["\'].*\b(SELECT|INSERT|UPDATE|DELETE|WHERE|FROM)\b.*["\'].*\+')),
    ("sql_concat_var",  re.compile(r'\+\s*\w+\s*\+.*(?:SELECT|INSERT|UPDATE|DELETE|WHERE)')),
]

# Parameterized query suppressors
_SAFE_PATTERNS: List[re.Pattern] = [
    re.compile(r"cursor\.execute\s*\(\s*\w+\s*,\s*[\[(]"),   # cursor.execute(sql, (x,))
    re.compile(r"\.filter\s*\("),                              # ORM filter
    re.compile(r"\.objects\.\w+\s*\("),                       # Django ORM
    re.compile(r"prepare\s*\("),                               # prepared statement
    re.compile(r"sqlalchemy\b"),                               # SQLAlchemy ORM
    re.compile(r"#\s*(nosec|noqa|safe)"),                     # inline suppression
    re.compile(r"^\s*#"),                                      # commented line
]


def _line_is_safe(line: str) -> bool:
    return any(p.search(line) for p in _SAFE_PATTERNS)


def _source_on_line(line: str) -> Optional[str]:
    for name, pat in _SOURCE_PATTERNS:
        if pat.search(line):
            return name
    return None


def _sql_concat_on_line(line: str) -> Optional[Tuple[str, str]]:
    for name, pat in _SQL_CONCAT_PATTERNS:
        m = pat.search(line)
        if m:
            return name, m.group(0)
    return None


def _sink_on_line(line: str) -> Optional[Tuple[str, str]]:
    for name, pat in _SQL_SINKS:
        m = pat.search(line)
        if m:
            return name, m.group(0)
    return None


# ---------------------------------------------------------------------------
# APPROACH 1 — Naive
# ---------------------------------------------------------------------------

def detect_naive(code: str) -> List[str]:
    results = []
    for i, line in enumerate(code.splitlines(), 1):
        if "execute(" in line:
            results.append(f"Line {i}: {line.strip()}")
    return results


# ---------------------------------------------------------------------------
# APPROACH 2 — Taint-aware detector
# ---------------------------------------------------------------------------

class SqlInjectionDetector:
    """
    Multi-pass heuristic SQL injection detector.

    Pass 1: taint variables assigned from user-controlled sources.
    Pass 2: flag SQL string construction patterns (fstring, %, .format, +).
    Pass 3: flag SQL sink calls containing tainted vars or dynamic SQL.
    """

    def __init__(self, lookahead: int = 8):
        self._lookahead = lookahead
        self._tainted_sql_vars: Set[str] = set()

    def detect(self, code: str, filename: str = "<code>") -> ScanResult:
        self._tainted_sql_vars = set()
        result = ScanResult(filename=filename)
        lines  = code.splitlines()
        result.lines_scanned = len(lines)

        # ---- Pass 1: Collect tainted variable names ----
        _VAR_FROM_SOURCE = re.compile(
            r"^\s*(\w+)\s*=\s*.*(?:request\.|req\.|input\(|sys\.argv|getParameter\s*\()",
            re.IGNORECASE,
        )
        tainted: Set[str] = set()
        for line in lines:
            m = _VAR_FROM_SOURCE.match(line)
            if m:
                tainted.add(m.group(1))

        # ---- Pass 2: Track SQL variables built from user input ----
        _SQL_VAR_ASSIGN = re.compile(r"^\s*(\w+)\s*=\s*.*(?:SELECT|INSERT|UPDATE|DELETE|WHERE|FROM)",
                                     re.IGNORECASE)
        sql_vars: Dict[str, int] = {}    # var_name → line_idx
        for i, line in enumerate(lines):
            m = _SQL_VAR_ASSIGN.match(line)
            if m:
                var = m.group(1)
                sql_vars[var] = i
                # If any tainted var appears in the assignment → tainted sql_var
                for tv in tainted:
                    if re.search(r'\b' + re.escape(tv) + r'\b', line):
                        self._tainted_sql_vars.add(var)

        # ---- Pass 3: Find sink calls ----
        for line_idx, line in enumerate(lines, start=1):
            if _line_is_safe(line):
                continue

            # Direct source in sink (highest severity)
            src  = _source_on_line(line)
            sink = _sink_on_line(line)
            if src and sink:
                result.findings.append(SqliFinding(
                    file=filename,
                    line_number=line_idx,
                    severity="high",
                    rule_name=f"{src}_in_{sink[0]}",
                    evidence=line.strip()[:80],
                    line_preview=line.strip()[:100],
                ))
                continue

            # SQL concat pattern on same line as sink
            sql_pat = _sql_concat_on_line(line)
            if sql_pat and sink:
                result.findings.append(SqliFinding(
                    file=filename,
                    line_number=line_idx,
                    severity="high",
                    rule_name=f"{sql_pat[0]}_in_{sink[0]}",
                    evidence=line.strip()[:80],
                    line_preview=line.strip()[:100],
                ))
                continue

            # Tainted SQL variable in sink
            if sink:
                for tsv in self._tainted_sql_vars:
                    if re.search(r'\b' + re.escape(tsv) + r'\b', line):
                        result.findings.append(SqliFinding(
                            file=filename,
                            line_number=line_idx,
                            severity="high",
                            rule_name=f"tainted_sql_var_{sink[0]}",
                            evidence=f"tainted_var={tsv}",
                            line_preview=line.strip()[:100],
                        ))
                        break
                else:
                    # Sink with any sql variable (possibly tainted)
                    for sv in sql_vars:
                        if re.search(r'\b' + re.escape(sv) + r'\b', line):
                            result.findings.append(SqliFinding(
                                file=filename,
                                line_number=line_idx,
                                severity="medium",
                                rule_name=f"sql_var_{sink[0]}",
                                evidence=f"sql_var={sv}",
                                line_preview=line.strip()[:100],
                            ))
                            break

            # SQL concat/fstring without a sink yet → flag as medium
            elif sql_pat and (src or any(
                re.search(r'\b' + re.escape(tv) + r'\b', line) for tv in tainted
            )):
                result.findings.append(SqliFinding(
                    file=filename,
                    line_number=line_idx,
                    severity="medium",
                    rule_name=f"sql_construction_{sql_pat[0]}",
                    evidence=sql_pat[1][:60],
                    line_preview=line.strip()[:100],
                ))

        return result

def test_fstring_injection():
    code = """
from flask import request
import sqlite3

conn = sqlite3.connect("users.db")
cursor = conn.cursor()

username = request.args.get("username")
query = f"SELECT * FROM users WHERE name = '{username}'"
cursor.execute(query)
"""
    d = SqlInjectionDetector()
    r = d.detect(code, "views.py")
    print(f"fstring findings: {[str(f) for f in r.findings]}")
    assert r.has_findings
    high = [f for f in r.findings if f.severity == "high"]
    assert high
    print("fstring SQLi: passed")


def test_parameterized_suppressed():
    code = """
import sqlite3
conn = sqlite3.connect("db.sqlite")
cursor = conn.cursor()
username = "alice"
cursor.execute("SELECT * FROM users WHERE name = ?", (username,))
"""
    d = SqlInjectionDetector()
    r = d.detect(code, "safe.py")
    assert not r.has_findings
    print("Parameterized query not flagged: passed")

"""
SQL INJECTION — OWASP #1 / CWE-89
------------------------------------
Root Cause: user input treated as SQL syntax instead of data.

CLASSIC ATTACK
--------------
  query = "SELECT * FROM users WHERE name='" + username + "'"
  username = "' OR '1'='1"
  Result: SELECT * FROM users WHERE name='' OR '1'='1'
  → Returns all rows.

  username = "'; DROP TABLE users;--"
  Result: SELECT * FROM users WHERE name=''; DROP TABLE users;--
  → Destroys the table.

BLIND INJECTION
---------------
Boolean: ?id=1 AND 1=1 (true) vs ?id=1 AND 1=2 (false) → infer data.
Time-based: ?id=1; WAITFOR DELAY '0:0:5'-- → confirm injection via delay.
Out-of-band: DNS exfiltration via DNS lookup on injected host name.

DEFENSES (in order of priority)
---------------------------------
1. Parameterized queries / prepared statements — gold standard.
2. ORM with proper escaping (avoid raw() unless absolutely needed).
3. Stored procedures (if not constructed dynamically inside them).
4. Input validation (allowlist, not blocklist — bypass with Unicode, comments).
5. Least privilege DB account (can't DROP even if injected).
6. WAF as defense-in-depth (not primary defense; bypassable).
"""
