"""
=============================================================================
Q51 [Hard] — Path Traversal Static Pattern Detector
=============================================================================

PROBLEM
-------
Detect file operations using unsanitized user-controlled paths:
  1. open(), os.path.join, os.makedirs, shutil.copy with user input.
  2. "../" sequences not stripped or normalized.
  3. Missing os.path.realpath / os.path.abspath + prefix check.
  4. JavaScript: fs.readFile, path.join, res.sendFile with user input.

CLARIFYING QUESTIONS
--------------------
1. Languages? → Python + JavaScript.
2. What makes a path safe?
   → realpath/abspath computed, then checked that result starts with
     the allowed base directory.
   → Flask: send_from_directory (internally validates).
3. Output? → Structured findings.

=============================================================================
APPROACH 1 — NAIVE (grep for "open(" only)
=============================================================================
  Misses path.join, shutil, JavaScript fs calls; massive false positives.

=============================================================================
APPROACH 2 — TAINT-AWARE PATH TRAVERSAL DETECTOR (OPTIMAL)
=============================================================================
  1. Collect tainted variables from user-controlled sources.
  2. Detect file-operation sinks.
  3. Check if path sanitization patterns are present nearby.
  4. Flag if tainted var reaches sink without sanitization.

=============================================================================
"""

from __future__ import annotations

import re
from dataclasses import dataclass, field
from typing import List, Optional, Set, Tuple


# ---------------------------------------------------------------------------
# Data Types
# ---------------------------------------------------------------------------

@dataclass
class PathTraversalFinding:
    file: str
    line_number: int
    severity: str
    rule_name: str
    evidence: str
    line_preview: str

    def __str__(self) -> str:
        return (
            f"[{self.severity.upper()}] {self.file}:{self.line_number} "
            f"rule={self.rule_name!r}: {self.evidence[:60]}"
        )


@dataclass
class PathScanResult:
    filename: str
    findings: List[PathTraversalFinding] = field(default_factory=list)
    lines_scanned: int = 0

    @property
    def has_findings(self) -> bool:
        return bool(self.findings)


# ---------------------------------------------------------------------------
# Patterns
# ---------------------------------------------------------------------------

_SOURCE_PATTERNS: List[Tuple[str, re.Pattern]] = [
    ("flask_args",   re.compile(r"request\.(args|form|values|GET|POST)(\[|\.get\s*\()")),
    ("input_func",   re.compile(r"\binput\s*\(")),
    ("sys_argv",     re.compile(r"\bsys\.argv\b")),
    ("req_query",    re.compile(r"req\.(query|body|params)\s*[\[.]")),
    ("req_param",    re.compile(r"request\.getParameter\s*\(")),
]

_FILE_SINKS: List[Tuple[str, re.Pattern]] = [
    ("open_builtin",      re.compile(r"\bopen\s*\(")),
    ("os_path_join",      re.compile(r"os\.path\.join\s*\(")),
    ("os_makedirs",       re.compile(r"os\.makedirs\s*\(")),
    ("shutil_copy",       re.compile(r"shutil\.(copy|move|rmtree)\s*\(")),
    ("pathlib_open",      re.compile(r"Path\s*\(.*\)\s*\.(open|read_text|write_text)\s*\(")),
    ("send_file",         re.compile(r"\bsend_file\s*\(")),
    ("fs_read",           re.compile(r"fs\.(readFile|createReadStream|readFileSync)\s*\(")),
    ("path_join_js",      re.compile(r"path\.join\s*\(")),
    ("res_send_file",     re.compile(r"res\.(sendFile|download)\s*\(")),
]

# Path-safe patterns: realpath + startswith, or send_from_directory
_SAFE_PATTERNS: List[re.Pattern] = [
    re.compile(r"os\.path\.realpath\s*\("),
    re.compile(r"os\.path\.abspath\s*\("),
    re.compile(r"send_from_directory\s*\("),         # Flask safe file serve
    re.compile(r"\.startswith\s*\("),                 # prefix check
    re.compile(r"secure_filename\s*\("),              # werkzeug secure_filename
    re.compile(r"path\.resolve\s*\("),                # JS path.resolve
    re.compile(r"normalize\s*\("),
    re.compile(r"^\s*#"),
    re.compile(r"#\s*(nosec|noqa|safe)"),
]

# Explicit traversal pattern in literal
_TRAVERSAL_LITERAL = re.compile(r"\.\./|\.\.\\\\")

_VAR_FROM_SOURCE = re.compile(
    r"^\s*(\w+)\s*=\s*.*(?:request\.|req\.|input\(|sys\.argv|getParameter\s*\()",
    re.IGNORECASE,
)


def _line_is_safe(line: str) -> bool:
    return any(p.search(line) for p in _SAFE_PATTERNS)


def _source_on_line(line: str) -> Optional[str]:
    for name, pat in _SOURCE_PATTERNS:
        if pat.search(line):
            return name
    return None


def _sink_on_line(line: str) -> Optional[Tuple[str, str]]:
    for name, pat in _FILE_SINKS:
        m = pat.search(line)
        if m:
            return name, m.group(0)
    return None


# ---------------------------------------------------------------------------
# Context Window: check if sanitization appears in nearby lines
# ---------------------------------------------------------------------------

def _has_sanitization_nearby(lines: List[str], center: int, window: int = 5) -> bool:
    start = max(0, center - window)
    end   = min(len(lines), center + window + 1)
    for line in lines[start:end]:
        if any(p.search(line) for p in _SAFE_PATTERNS):
            return True
    return False


# ---------------------------------------------------------------------------
# APPROACH 1 — Naive
# ---------------------------------------------------------------------------

def detect_naive(code: str) -> List[str]:
    return [
        f"Line {i+1}: {ln.strip()}"
        for i, ln in enumerate(code.splitlines())
        if "open(" in ln or "readFile(" in ln
    ]


# ---------------------------------------------------------------------------
# APPROACH 2 — Taint-aware detector
# ---------------------------------------------------------------------------

class PathTraversalDetector:
    """
    Detects path traversal vulnerabilities using taint-aware heuristics.
    """

    def detect(self, code: str, filename: str = "<code>") -> PathScanResult:
        result = PathScanResult(filename=filename)
        lines  = code.splitlines()
        result.lines_scanned = len(lines)

        # Collect tainted vars
        tainted: Set[str] = set()
        for line in lines:
            m = _VAR_FROM_SOURCE.match(line)
            if m:
                tainted.add(m.group(1))

        # Collect path-var assignments using tainted sources
        path_vars: Set[str] = set()
        _PATH_VAR = re.compile(
            r"^\s*(\w+)\s*=\s*.*(?:os\.path\.join|path\.join|Path\s*\()",
            re.IGNORECASE,
        )
        for line in lines:
            m = _PATH_VAR.match(line)
            if m:
                var = m.group(1)
                if any(re.search(r'\b' + re.escape(tv) + r'\b', line) for tv in tainted):
                    path_vars.add(var)

        # Find sinks
        for line_idx, line in enumerate(lines, start=1):
            if _line_is_safe(line):
                continue

            # Explicit ../  in any line → high regardless
            if _TRAVERSAL_LITERAL.search(line):
                sink = _sink_on_line(line)
                result.findings.append(PathTraversalFinding(
                    file=filename, line_number=line_idx,
                    severity="high",
                    rule_name="traversal_literal_in_path",
                    evidence=line.strip()[:80],
                    line_preview=line.strip()[:100],
                ))
                continue

            sink = _sink_on_line(line)
            if sink is None:
                continue
            sink_name, sink_text = sink

            # Direct source on sink line
            src = _source_on_line(line)
            if src and not _has_sanitization_nearby(lines, line_idx - 1):
                result.findings.append(PathTraversalFinding(
                    file=filename, line_number=line_idx,
                    severity="high",
                    rule_name=f"{src}_to_{sink_name}",
                    evidence=line.strip()[:80],
                    line_preview=line.strip()[:100],
                ))
                continue

            # Tainted var in sink
            matched_tv = next(
                (tv for tv in tainted
                 if re.search(r'\b' + re.escape(tv) + r'\b', line)),
                None
            )
            if matched_tv and not _has_sanitization_nearby(lines, line_idx - 1):
                result.findings.append(PathTraversalFinding(
                    file=filename, line_number=line_idx,
                    severity="high",
                    rule_name=f"tainted_path_{sink_name}",
                    evidence=f"var={matched_tv}",
                    line_preview=line.strip()[:100],
                ))
                continue

            # Tainted path var in sink
            matched_pv = next(
                (pv for pv in path_vars
                 if re.search(r'\b' + re.escape(pv) + r'\b', line)),
                None
            )
            if matched_pv and not _has_sanitization_nearby(lines, line_idx - 1):
                result.findings.append(PathTraversalFinding(
                    file=filename, line_number=line_idx,
                    severity="high",
                    rule_name=f"tainted_path_var_{sink_name}",
                    evidence=f"path_var={matched_pv}",
                    line_preview=line.strip()[:100],
                ))

        return result

def test_direct_path_traversal():
    code = """
from flask import request
filename = request.args.get("file")
with open("/uploads/" + filename) as f:
    data = f.read()
"""
    d = PathTraversalDetector()
    r = d.detect(code, "download.py")
    print(f"Direct path traversal: {[str(f) for f in r.findings]}")
    assert r.has_findings
    high = [f for f in r.findings if f.severity == "high"]
    assert high
    print("Direct path traversal: passed")


def test_send_from_directory_safe():
    code = """
from flask import send_from_directory, request
filename = request.args.get("file")
return send_from_directory("/uploads", filename)
"""
    d = PathTraversalDetector()
    r = d.detect(code, "safe.py")
    high = [f for f in r.findings if f.severity == "high"]
    # send_from_directory is in safe patterns (suppresses findings on that line)
    print(f"send_from_directory findings: {[str(f) for f in high]}")
    print("send_from_directory test: passed")


def test_realpath_check_safe():
    code = """
import os
from flask import request
BASE = "/uploads"
filename = request.args.get("file")
full_path = os.path.join(BASE, filename)
safe_path = os.path.realpath(full_path)
if safe_path.startswith(BASE):
    with open(safe_path) as f:
        return f.read()
"""
    d = PathTraversalDetector()
    r = d.detect(code, "realpath.py")
    high = [f for f in r.findings if f.severity == "high"]
    # realpath + startswith present in nearby lines
    print(f"realpath check findings: {[str(f) for f in r.findings]}")
    print("realpath check test: passed (sanitization nearby)")

# SECURITY ANGLE

"""
PATH TRAVERSAL — CWE-22 / OWASP A1
-------------------------------------
Root Cause: User-controlled filename not normalized; "../" sequences walk
up directory tree to access files outside the intended directory.

CLASSIC ATTACKS
----------------
  GET /download?file=../../etc/passwd
  GET /download?file=%2e%2e%2f%2e%2e%2fetc%2fpasswd  (URL-encoded)
  GET /download?file=....//....//etc/passwd           (double-slash bypass)
  Windows: ?file=..\\..\windows\win.ini

EVASION TECHNIQUES
-------------------
  %2e%2e%2f → URL decode → ../
  ..%2f, .%2e/, %%32%65 → double-encoding
  Null byte injection: ?file=../../etc/passwd%00.jpg (in some frameworks)
  Unicode: U+2025 ONE LEADER (dots), U+FE52 SMALL FULL STOP

SAFE IMPLEMENTATION (Python)
-----------------------------
  import os

  BASE_DIR = "/safe/uploads"

  def safe_open(filename):
      # 1. Reject null bytes
      if '\\x00' in filename:
          raise ValueError("null byte in filename")
      # 2. Compute absolute, resolved path
      abs_path = os.path.realpath(os.path.join(BASE_DIR, filename))
      # 3. Verify within base directory
      if not abs_path.startswith(BASE_DIR + os.sep):
          raise ValueError(f"path escape: {abs_path}")
      return open(abs_path)

NOTE: os.path.join("/uploads", "/etc/passwd") → "/etc/passwd"
  → Always join relative filename, not full user-supplied path.
  → Strip leading "/" from user input before joining.
"""
