"""
=============================================================================
Q53 [Hard] — Race Condition in Token Revocation
=============================================================================

PROBLEM
-------
Implement thread-safe token validation against a revocation list:
  1. Tokens are issued with an expiry.
  2. Tokens can be revoked at any time (stored in a revocation set).
  3. Concurrent requests may try to use the same token simultaneously.
  4. Revocation must be atomic; validation must not race with revocation.
  5. Support bulk revocation (revoke all tokens for a given user).

CLARIFYING QUESTIONS
--------------------
1. Token format? → Opaque random string (not JWT for simplicity).
2. Persistence? → In-memory (distributed cache in production).
3. Race conditions to prevent?
   → Check-then-act (TOCTOU): validate passes, then token gets revoked
     between check and use.
   → Double revocation (idempotent).
   → Concurrent issue + immediate revoke.
4. How to detect TOCTOU? → Lock around check+use; or atomic pop.
5. Thread model? → Python threading; use threading.RLock.

=============================================================================
APPROACH 1 — NAIVE (no locking)
=============================================================================
  Separate set for active tokens; check `in` then pop.
  TOCTOU: thread A checks (valid), thread B revokes, thread A uses.

=============================================================================
APPROACH 2 — ATOMIC CHECK-AND-USE WITH LOCKING (OPTIMAL)
=============================================================================
  1. tokens: dict[token → TokenRecord].
  2. revoked: set[token].
  3. user_tokens: dict[user_id → set[token]] for bulk revocation.
  4. RLock protects all mutations.
  5. validate_and_use(): checks expiry + revocation inside lock.
     Returns a short-lived "claim" — caller must not cache it.
  6. revoke(): adds to revoked set + removes from user_tokens inside lock.
  7. revoke_all_for_user(): bulk revoke inside single lock acquisition.

=============================================================================
"""

from __future__ import annotations

import secrets
import threading
import time
from dataclasses import dataclass, field
from typing import Dict, Optional, Set


# ---------------------------------------------------------------------------
# Config
# ---------------------------------------------------------------------------

DEFAULT_TOKEN_TTL_S = 3600   # 1 hour
TOKEN_BYTES         = 32


# ---------------------------------------------------------------------------
# Data Types
# ---------------------------------------------------------------------------

@dataclass
class TokenRecord:
    token: str
    user_id: str
    issued_at: float
    expires_at: float
    scope: str = "default"


@dataclass
class ValidationResult:
    valid: bool
    user_id: str = ""
    scope: str = ""
    reason: str = ""


# ---------------------------------------------------------------------------
# APPROACH 1 — Naive (TOCTOU vulnerable)
# ---------------------------------------------------------------------------

class NaiveTokenStore:
    """
    No locking. Susceptible to TOCTOU race between check and revoke.
    """

    def __init__(self):
        self._tokens: Dict[str, TokenRecord] = {}
        self._revoked: Set[str] = set()

    def issue(self, user_id: str, ttl_s: float = DEFAULT_TOKEN_TTL_S) -> str:
        token = secrets.token_urlsafe(TOKEN_BYTES)
        now   = time.monotonic()
        self._tokens[token] = TokenRecord(
            token=token, user_id=user_id,
            issued_at=now, expires_at=now + ttl_s,
        )
        return token

    def validate(self, token: str) -> bool:
        # TOCTOU: another thread may revoke between these two checks
        if token in self._revoked:
            return False
        rec = self._tokens.get(token)
        if rec is None or time.monotonic() > rec.expires_at:
            return False
        return True

    def revoke(self, token: str) -> None:
        self._revoked.add(token)
        self._tokens.pop(token, None)


# ---------------------------------------------------------------------------
# APPROACH 2 — Thread-safe token store
# ---------------------------------------------------------------------------

class TokenRevocationStore:
    """
    Thread-safe token store with atomic revocation and validation.

    Key design:
      - Single RLock protects all state.
      - validate_and_use() performs check inside lock (no TOCTOU).
      - revoke() and revoke_all_for_user() are atomic.
      - Lazy expiry cleanup via _purge_expired() on issue.
    """

    def __init__(self):
        self._lock        = threading.RLock()
        self._tokens:      Dict[str, TokenRecord] = {}
        self._revoked:     Set[str]               = set()
        self._user_tokens: Dict[str, Set[str]]    = {}

    # --- Issue ---------------------------------------------------------------

    def issue(
        self,
        user_id: str,
        ttl_s: float = DEFAULT_TOKEN_TTL_S,
        scope: str   = "default",
    ) -> str:
        token = secrets.token_urlsafe(TOKEN_BYTES)
        now   = time.monotonic()
        record = TokenRecord(
            token=token, user_id=user_id,
            issued_at=now, expires_at=now + ttl_s,
            scope=scope,
        )
        with self._lock:
            self._tokens[token] = record
            self._user_tokens.setdefault(user_id, set()).add(token)
            self._purge_expired_locked()
        return token

    # --- Atomic validate + mark-used ----------------------------------------

    def validate_and_use(self, token: str) -> ValidationResult:
        """
        Atomically validates and consumes a single-use token.
        If the token is valid, it is NOT removed (multi-use).
        Revocation and expiry are checked inside the lock.
        """
        with self._lock:
            if token in self._revoked:
                return ValidationResult(valid=False, reason="revoked")
            rec = self._tokens.get(token)
            if rec is None:
                return ValidationResult(valid=False, reason="not_found")
            if time.monotonic() > rec.expires_at:
                # Lazy cleanup
                self._remove_token_locked(token)
                return ValidationResult(valid=False, reason="expired")
            return ValidationResult(
                valid=True,
                user_id=rec.user_id,
                scope=rec.scope,
            )

    # --- Revoke -------------------------------------------------------------

    def revoke(self, token: str) -> bool:
        """Revoke a specific token. Returns True if it was active."""
        with self._lock:
            was_active = token in self._tokens and token not in self._revoked
            self._revoked.add(token)
            self._remove_token_locked(token)
            return was_active

    def revoke_all_for_user(self, user_id: str) -> int:
        """Bulk-revoke all tokens for a user. Returns count revoked."""
        with self._lock:
            tokens = list(self._user_tokens.get(user_id, set()))
            for t in tokens:
                self._revoked.add(t)
                self._remove_token_locked(t)
            return len(tokens)

    # --- Internal helpers ---------------------------------------------------

    def _remove_token_locked(self, token: str) -> None:
        """Must be called with self._lock held."""
        rec = self._tokens.pop(token, None)
        if rec:
            user_set = self._user_tokens.get(rec.user_id)
            if user_set:
                user_set.discard(token)

    def _purge_expired_locked(self) -> None:
        """Remove expired tokens. Must hold lock."""
        now     = time.monotonic()
        expired = [t for t, r in self._tokens.items() if now > r.expires_at]
        for t in expired:
            self._remove_token_locked(t)

    @property
    def active_count(self) -> int:
        with self._lock:
            return len(self._tokens)


# ---------------------------------------------------------------------------
# Test Cases
# ---------------------------------------------------------------------------

def test_basic_issue_validate():
    store = TokenRevocationStore()
    token = store.issue("user1")
    r = store.validate_and_use(token)
    assert r.valid, r.reason
    assert r.user_id == "user1"
    print("Basic issue+validate: passed")


def test_revoke_makes_token_invalid():
    store = TokenRevocationStore()
    token = store.issue("user2")
    assert store.validate_and_use(token).valid
    store.revoke(token)
    r = store.validate_and_use(token)
    assert not r.valid
    assert r.reason == "revoked"
    print("Revoke makes token invalid: passed")


def test_expired_token():
    store = TokenRevocationStore()
    token = store.issue("user3", ttl_s=0.01)
    time.sleep(0.05)
    r = store.validate_and_use(token)
    assert not r.valid
    assert r.reason == "expired"
    print("Expired token: passed")


def test_bulk_revoke():
    store = TokenRevocationStore()
    tokens = [store.issue("user4") for _ in range(5)]
    count = store.revoke_all_for_user("user4")
    assert count == 5
    for t in tokens:
        assert not store.validate_and_use(t).valid
    print("Bulk revoke: passed")


def test_concurrent_revoke_validate():
    """
    Stress test: concurrent threads validate and revoke the same token.
    No panics or data corruption expected.
    """
    store   = TokenRevocationStore()
    token   = store.issue("user5", ttl_s=60)
    results = []
    errors  = []

    def validator():
        try:
            r = store.validate_and_use(token)
            results.append(r.valid)
        except Exception as e:
            errors.append(str(e))

    def revoker():
        try:
            store.revoke(token)
        except Exception as e:
            errors.append(str(e))

    threads = [threading.Thread(target=validator) for _ in range(10)]
    threads += [threading.Thread(target=revoker)  for _ in range(3)]

    for t in threads:
        t.start()
    for t in threads:
        t.join()

    assert not errors, f"Errors during concurrent access: {errors}"
    # At most 1 valid result expected (token revoked after first revoke)
    valid_count = results.count(True)
    print(f"Concurrent revoke+validate: valid_count={valid_count}, errors={errors}")
    print("Concurrent stress test: passed")


def test_double_revoke_idempotent():
    store = TokenRevocationStore()
    token = store.issue("user6")
    assert store.revoke(token) == True
    assert store.revoke(token) == False  # already revoked; not active
    print("Double revoke idempotent: passed")


def test_unknown_token():
    store = TokenRevocationStore()
    r = store.validate_and_use("not_a_real_token_xyz")
    assert not r.valid
    assert r.reason == "not_found"
    print("Unknown token: passed")


# ---------------------------------------------------------------------------
# SECURITY ANGLE
# ---------------------------------------------------------------------------
"""
TOCTOU (TIME-OF-CHECK TO TIME-OF-USE) IN TOKEN REVOCATION
-----------------------------------------------------------
Race Condition:
  Thread A: check(token) → valid       ← check happens here
  Thread B: revoke(token)              ← revoke happens here
  Thread A: use(token)                 ← use happens AFTER revocation

  Result: Thread A uses a revoked token.
  Fix: wrap check + use in the same lock acquisition.

REAL-WORLD EXAMPLES
-------------------
  JWT revocation with in-memory blocklist:
    check-then-act without locking allows ~1ms window.
  Session invalidation on logout:
    Load balancer may route next request to a node with stale session cache.
    Fix: use distributed revocation store (Redis SET with TTL).

DISTRIBUTED TOKEN REVOCATION PATTERNS
---------------------------------------
1. Blocklist (pull model): all validators query Redis/DynamoDB for revocation.
   Latency: O(1) lookup per request.
   Consistency: eventual (new nodes may not see revocation immediately).

2. Short-lived tokens + introspection:
   Tokens expire quickly (5 min); revocation only needs to last until expiry.
   OAuth 2 Token Introspection (RFC 7662): real-time status check.

3. Version counter:
   Each user has a version number. Token encodes version.
   Revocation = increment user version. Validate token.version == current.

4. Event stream:
   Revocation published to Kafka; consumers update local cache.
   Window = Kafka consumer lag (typically <1s).

LOCK GRANULARITY
-----------------
  Global lock (this implementation): simple; bottleneck at high throughput.
  Per-user lock: better concurrency; manage lock map carefully.
  Lock-free (CAS on atomic set): most scalable; harder to implement correctly.
"""
