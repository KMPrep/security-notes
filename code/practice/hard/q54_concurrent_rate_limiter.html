"""
=============================================================================
Q54 [Hard] — Concurrent Rate Limiter
=============================================================================

PROBLEM
-------
Implement a thread-safe rate limiter supporting many concurrent users:
  1. Fixed window: allow N requests per window period per key.
  2. Sliding window (token bucket): smoother enforcement.
  3. Thread-safe: multiple threads may call check() for the same key.
  4. Memory-bounded: evict idle windows to avoid OOM.
  5. Support per-key limits (different limits for different API tiers).

CLARIFYING QUESTIONS
--------------------
1. Algorithm? → Fixed window + sliding window log (two approaches).
2. Distributed? → Single-node (in-memory); Redis extension noted.
3. Keys? → Typically user_id or IP address.
4. Granularity? → Seconds.
5. Thread model? → Python threading; per-key locking for throughput.

=============================================================================
APPROACH 1 — NAIVE (global lock, dict counter)
=============================================================================
  Single global dict + global lock. All threads serialize on every call.
  O(1) per request but contention kills throughput under load.

=============================================================================
APPROACH 2A — FIXED WINDOW WITH PER-KEY LOCKING (OPTIMAL FOR SIMPLICITY)
=============================================================================
  Per-key Lock (lazy-created) + fixed window counter.
  Only locks the specific key; other keys can proceed concurrently.
  Memory bounded: prune idle keys periodically.

APPROACH 2B — SLIDING WINDOW LOG
=============================================================================
  Sliding window: store timestamps in a deque per key.
  Evict timestamps older than window.
  Count remaining to check limit.
  More accurate (no burst at window boundary); O(N) memory per key.

=============================================================================
"""

from __future__ import annotations

import threading
import time
from collections import defaultdict, deque
from dataclasses import dataclass, field
from typing import DefaultDict, Dict, Deque, Optional, Tuple


# ---------------------------------------------------------------------------
# Config
# ---------------------------------------------------------------------------

DEFAULT_LIMIT      = 100      # requests per window
DEFAULT_WINDOW_S   = 60       # 1-minute window
CLEANUP_INTERVAL_S = 300      # clean up idle keys every 5 min


# ---------------------------------------------------------------------------
# Data Types
# ---------------------------------------------------------------------------

@dataclass
class RateLimit:
    limit: int           # max requests per window
    window_s: float      # window duration in seconds


@dataclass
class LimitResult:
    allowed: bool
    remaining: int
    reset_at: float     # unix timestamp when window resets
    retry_after: float  # seconds to wait if not allowed


# ---------------------------------------------------------------------------
# APPROACH 1 — Naive (global lock)
# ---------------------------------------------------------------------------

class NaiveRateLimiter:
    """Single global lock; all keys serialize. Not suitable for production."""

    def __init__(self, limit: int = DEFAULT_LIMIT, window_s: float = DEFAULT_WINDOW_S):
        self._limit   = limit
        self._window  = window_s
        self._lock    = threading.Lock()
        self._windows: Dict[str, Tuple[int, float]] = {}  # key → (count, window_start)

    def check(self, key: str) -> bool:
        with self._lock:
            now = time.monotonic()
            count, win_start = self._windows.get(key, (0, now))
            if now - win_start >= self._window:
                count, win_start = 0, now
            if count >= self._limit:
                return False
            self._windows[key] = (count + 1, win_start)
            return True


# ---------------------------------------------------------------------------
# APPROACH 2A — Fixed Window with per-key locking
# ---------------------------------------------------------------------------

@dataclass
class _FixedWindow:
    count: int
    window_start: float
    lock: threading.Lock = field(default_factory=threading.Lock)


class FixedWindowRateLimiter:
    """
    Per-key lock for high concurrency.
    A meta-lock only guards the _windows dict itself (brief).
    """

    def __init__(
        self,
        default_limit: RateLimit = None,
        per_key_limits: Optional[Dict[str, RateLimit]] = None,
    ):
        self._default   = default_limit or RateLimit(DEFAULT_LIMIT, DEFAULT_WINDOW_S)
        self._key_limits: Dict[str, RateLimit] = per_key_limits or {}
        self._meta_lock = threading.Lock()
        self._windows:   Dict[str, _FixedWindow] = {}
        self._last_clean = time.monotonic()

    def _get_limit(self, key: str) -> RateLimit:
        return self._key_limits.get(key, self._default)

    def _get_window(self, key: str) -> _FixedWindow:
        with self._meta_lock:
            if key not in self._windows:
                self._windows[key] = _FixedWindow(
                    count=0,
                    window_start=time.monotonic(),
                )
            return self._windows[key]

    def check(self, key: str) -> LimitResult:
        rl  = self._get_limit(key)
        win = self._get_window(key)
        now = time.monotonic()

        with win.lock:
            # Reset window if expired
            if now - win.window_start >= rl.window_s:
                win.count        = 0
                win.window_start = now

            reset_at = win.window_start + rl.window_s

            if win.count >= rl.limit:
                return LimitResult(
                    allowed=False,
                    remaining=0,
                    reset_at=reset_at,
                    retry_after=max(0.0, reset_at - now),
                )

            win.count += 1
            return LimitResult(
                allowed=True,
                remaining=rl.limit - win.count,
                reset_at=reset_at,
                retry_after=0.0,
            )

    def cleanup_idle(self, idle_threshold_s: float = 600.0) -> int:
        """Remove keys with windows older than idle_threshold_s."""
        now    = time.monotonic()
        purged = 0
        with self._meta_lock:
            stale = [
                k for k, w in self._windows.items()
                if now - w.window_start > idle_threshold_s
            ]
            for k in stale:
                del self._windows[k]
                purged += 1
        return purged

    @property
    def tracked_keys(self) -> int:
        with self._meta_lock:
            return len(self._windows)


# ---------------------------------------------------------------------------
# APPROACH 2B — Sliding Window Log
# ---------------------------------------------------------------------------

class SlidingWindowRateLimiter:
    """
    Sliding window: track individual request timestamps per key.
    More accurate (no burst at window boundary).
    Memory: O(limit) per key.
    """

    def __init__(
        self,
        default_limit: RateLimit = None,
        per_key_limits: Optional[Dict[str, RateLimit]] = None,
    ):
        self._default    = default_limit or RateLimit(DEFAULT_LIMIT, DEFAULT_WINDOW_S)
        self._key_limits = per_key_limits or {}
        self._meta_lock  = threading.Lock()
        self._logs:       Dict[str, Tuple[Deque[float], threading.Lock]] = {}

    def _get_limit(self, key: str) -> RateLimit:
        return self._key_limits.get(key, self._default)

    def _get_log(self, key: str) -> Tuple[Deque[float], threading.Lock]:
        with self._meta_lock:
            if key not in self._logs:
                self._logs[key] = (deque(), threading.Lock())
            return self._logs[key]

    def check(self, key: str) -> LimitResult:
        rl       = self._get_limit(key)
        log, lk  = self._get_log(key)
        now      = time.monotonic()
        cutoff   = now - rl.window_s

        with lk:
            # Evict timestamps outside the window
            while log and log[0] <= cutoff:
                log.popleft()

            count = len(log)
            if count >= rl.limit:
                oldest   = log[0] if log else now
                reset_at = oldest + rl.window_s
                return LimitResult(
                    allowed=False,
                    remaining=0,
                    reset_at=reset_at,
                    retry_after=max(0.0, reset_at - now),
                )

            log.append(now)
            return LimitResult(
                allowed=True,
                remaining=rl.limit - len(log),
                reset_at=now + rl.window_s,
                retry_after=0.0,
            )

    def cleanup_idle(self, idle_threshold_s: float = 600.0) -> int:
        now    = time.monotonic()
        purged = 0
        with self._meta_lock:
            stale = [
                k for k, (log, _) in self._logs.items()
                if not log or (now - log[-1] > idle_threshold_s)
            ]
            for k in stale:
                del self._logs[k]
                purged += 1
        return purged

def test_fixed_window_allows_within_limit():
    rl  = FixedWindowRateLimiter(default_limit=RateLimit(5, 60))
    for i in range(5):
        r = rl.check("user1")
        assert r.allowed, f"Request {i+1} should be allowed"
    print("Fixed window allows within limit: passed")


def test_fixed_window_blocks_over_limit():
    rl = FixedWindowRateLimiter(default_limit=RateLimit(3, 60))
    for _ in range(3):
        rl.check("user2")
    r = rl.check("user2")
    assert not r.allowed
    assert r.remaining == 0
    assert r.retry_after > 0
    print("Fixed window blocks over limit: passed")


def test_fixed_window_resets():
    rl = FixedWindowRateLimiter(default_limit=RateLimit(2, 0.05))
    rl.check("user3"); rl.check("user3")
    time.sleep(0.1)
    r = rl.check("user3")
    assert r.allowed
    print("Fixed window resets after expiry: passed")


def test_sliding_window_within_limit():
    rl = SlidingWindowRateLimiter(default_limit=RateLimit(5, 60))
    for i in range(5):
        r = rl.check("ip1")
        assert r.allowed, f"Request {i+1} should pass"
    print("Sliding window allows within limit: passed")


def test_sliding_window_blocks():
    rl = SlidingWindowRateLimiter(default_limit=RateLimit(3, 60))
    for _ in range(3):
        rl.check("ip2")
    r = rl.check("ip2")
    assert not r.allowed
    print("Sliding window blocks over limit: passed")


def test_sliding_window_old_requests_evicted():
    rl = SlidingWindowRateLimiter(default_limit=RateLimit(3, 0.1))
    for _ in range(3):
        rl.check("ip3")
    time.sleep(0.15)
    r = rl.check("ip3")
    assert r.allowed
    print("Sliding window evicts old timestamps: passed")


def test_per_key_limits():
    per_key = {"premium": RateLimit(1000, 60), "free": RateLimit(10, 60)}
    rl = FixedWindowRateLimiter(per_key_limits=per_key)
    # Premium: 1000 allowed
    for _ in range(100):
        assert rl.check("premium").allowed
    # Free: blocked at 11
    for _ in range(10):
        rl.check("free")
    r = rl.check("free")
    assert not r.allowed
    print("Per-key limits: passed")


def test_concurrent_requests():
    """Multiple threads hammering the same key."""
    rl      = FixedWindowRateLimiter(default_limit=RateLimit(50, 60))
    results = []
    lock    = threading.Lock()
    errors  = []

    def worker():
        try:
            r = rl.check("shared_key")
            with lock:
                results.append(r.allowed)
        except Exception as e:
            with lock:
                errors.append(str(e))

    threads = [threading.Thread(target=worker) for _ in range(100)]
    for t in threads:
        t.start()
    for t in threads:
        t.join()

    assert not errors, f"Errors: {errors}"
    allowed = results.count(True)
    denied  = results.count(False)
    # Should allow exactly 50
    assert allowed == 50, f"Expected 50 allowed, got {allowed}"
    assert denied  == 50, f"Expected 50 denied, got {denied}"
    print(f"Concurrent: allowed={allowed}, denied={denied} — passed")


def test_cleanup_idle_keys():
    rl = FixedWindowRateLimiter(default_limit=RateLimit(10, 0.01))
    for i in range(5):
        rl.check(f"temp_key_{i}")
    time.sleep(0.05)
    purged = rl.cleanup_idle(idle_threshold_s=0.02)
    assert purged == 5
    print(f"Cleanup idle keys: purged={purged} — passed")

"""
RATE LIMITING AS A SECURITY CONTROL
--------------------------------------
Rate limiting defends against:
  1. Brute force login attacks (credential stuffing, password spray).
  2. API abuse (scraping, enumeration, DoS).
  3. SMS/email abuse (OTP flooding, account enumeration via timing).

BURST ATTACK AT WINDOW BOUNDARY (fixed window weakness)
---------------------------------------------------------
  Window: 0–60s, limit=100.
  At t=59s: attacker sends 100 requests → all allowed.
  At t=60s: new window → attacker sends 100 more → 200 requests in 2 seconds!
  Fix: sliding window or token bucket eliminates this boundary burst.

TOKEN BUCKET (alternative)
---------------------------
  Tokens refill at rate r per second; bucket capacity = burst_limit.
  Each request consumes 1 token; block if bucket empty.
  Allows controlled bursting; rate limiting is smooth.

DISTRIBUTED RATE LIMITING
--------------------------
  Single-node in-memory won't work behind a load balancer.
  Use Redis with atomic INCR + EXPIRE:
    val = INCR(key)
    if val == 1: EXPIRE(key, window_s)
    if val > limit: reject

  Or Redis sliding window with ZADD + ZREMRANGEBYSCORE + ZCARD.

KEY SELECTION (SECURITY IMPLICATIONS)
--------------------------------------
  Rate limit by: IP (bypassed by rotating IPs), user_id (bypassed by
  creating accounts), API key, fingerprint combo.
  Best: layer multiple signals (IP + user + device fingerprint).

RATE LIMIT BYPASS ATTACKS
--------------------------
  X-Forwarded-For spoofing: if server trusts that header for IP.
    Fix: use client IP from TCP connection, not headers.
  IPv6 rotation: /48 prefix = 2^80 addresses; limit by /64 or /48.
  Tor/VPN: many exit nodes; use CAPTCHA fallback.
"""
