"""
=============================================================================
Q55 [Hard] — Concurrent Alert Deduplication Cache
=============================================================================

PROBLEM
-------
Implement a thread-safe cache to suppress duplicate security alerts:
  1. Multiple workers process alert streams concurrently.
  2. An alert is considered a "duplicate" if an identical (or similar)
     alert for the same key was seen within a suppression window.
  3. First occurrence is passed through; subsequent duplicates are suppressed.
  4. After the window expires, the next occurrence is again passed through.
  5. Memory-bounded: evict expired entries to avoid OOM.
  6. Support composite dedup keys (alert_type + source_ip + rule_id).

CLARIFYING QUESTIONS
--------------------
1. What defines "same" alert?
   → Composite key: (alert_type, source_ip, rule_id).
2. Suppression window? → Configurable per alert type; default 5 minutes.
3. Thread model? → Python threading; per-key locking preferred.
4. Persistence? → In-memory only; distributed cache in production.
5. Output of check()? → (is_new: bool, first_seen: float).

=============================================================================
APPROACH 1 — NAIVE (global lock, dict of timestamps)
=============================================================================
  Single global lock + dict. All workers serialize. Under high alert volume
  this becomes a bottleneck.

=============================================================================
APPROACH 2 — PER-SHARD LOCKING (OPTIMAL)
=============================================================================
  Shard the dedup dict into N buckets by hash(key) % N.
  Each shard has its own lock → only threads hitting the same shard block.
  Background cleanup thread evicts expired entries periodically.

=============================================================================
"""

from __future__ import annotations

import hashlib
import threading
import time
from dataclasses import dataclass, field
from typing import Dict, List, Optional, Tuple


# ---------------------------------------------------------------------------
# Config
# ---------------------------------------------------------------------------

DEFAULT_WINDOW_S  = 300     # 5-minute suppression window
DEFAULT_SHARDS    = 16      # number of lock shards
CLEANUP_INTERVAL  = 60.0    # seconds between cleanup sweeps


# ---------------------------------------------------------------------------
# Data Types
# ---------------------------------------------------------------------------

@dataclass
class AlertKey:
    alert_type: str
    source_ip: str
    rule_id: str

    def to_str(self) -> str:
        return f"{self.alert_type}|{self.source_ip}|{self.rule_id}"


@dataclass
class DeduplicationResult:
    is_new: bool
    key_str: str
    first_seen: float       # monotonic timestamp of first occurrence
    suppressed_count: int   # how many duplicates suppressed in this window


@dataclass
class _DedupEntry:
    first_seen: float
    expires_at: float
    suppressed: int = 0


# ---------------------------------------------------------------------------
# APPROACH 1 — Naive
# ---------------------------------------------------------------------------

class NaiveAlertDedup:
    """Single global lock; all threads serialize."""

    def __init__(self, window_s: float = DEFAULT_WINDOW_S):
        self._window = window_s
        self._lock   = threading.Lock()
        self._seen:  Dict[str, float] = {}   # key → first_seen

    def check(self, key: AlertKey) -> DeduplicationResult:
        k   = key.to_str()
        now = time.monotonic()
        with self._lock:
            first = self._seen.get(k)
            if first is not None and now - first < self._window:
                return DeduplicationResult(
                    is_new=False, key_str=k, first_seen=first, suppressed_count=0
                )
            self._seen[k] = now
            return DeduplicationResult(
                is_new=True, key_str=k, first_seen=now, suppressed_count=0
            )


# ---------------------------------------------------------------------------
# APPROACH 2 — Sharded dedup cache
# ---------------------------------------------------------------------------

class _Shard:
    def __init__(self):
        self.lock    = threading.Lock()
        self.entries: Dict[str, _DedupEntry] = {}


class ShardedAlertDedup:
    """
    Thread-safe alert dedup cache with per-shard locking.

    Algorithm:
      1. Hash the key to a shard index.
      2. Lock only that shard.
      3. Check if entry exists and is within suppression window.
         → Yes: increment suppressed_count; return is_new=False.
         → No:  create new entry; return is_new=True.
      4. Background cleanup thread removes expired entries.
    """

    def __init__(
        self,
        window_s: float   = DEFAULT_WINDOW_S,
        num_shards: int   = DEFAULT_SHARDS,
        per_type_windows: Optional[Dict[str, float]] = None,
    ):
        self._window      = window_s
        self._per_type    = per_type_windows or {}
        self._shards      = [_Shard() for _ in range(num_shards)]
        self._num_shards  = num_shards
        self._stop_event  = threading.Event()
        self._cleanup_thr = threading.Thread(
            target=self._cleanup_loop, daemon=True, name="dedup-cleanup"
        )
        self._cleanup_thr.start()

    # --- Public API ---------------------------------------------------------

    def check(self, key: AlertKey) -> DeduplicationResult:
        k_str  = key.to_str()
        shard  = self._get_shard(k_str)
        window = self._per_type.get(key.alert_type, self._window)
        now    = time.monotonic()

        with shard.lock:
            entry = shard.entries.get(k_str)

            if entry is not None and now < entry.expires_at:
                entry.suppressed += 1
                return DeduplicationResult(
                    is_new=False,
                    key_str=k_str,
                    first_seen=entry.first_seen,
                    suppressed_count=entry.suppressed,
                )

            # New or expired → reset
            new_entry = _DedupEntry(
                first_seen=now,
                expires_at=now + window,
                suppressed=0,
            )
            shard.entries[k_str] = new_entry
            return DeduplicationResult(
                is_new=True,
                key_str=k_str,
                first_seen=now,
                suppressed_count=0,
            )

    def force_expire(self, key: AlertKey) -> bool:
        """Manually expire a key so the next occurrence is treated as new."""
        k_str = key.to_str()
        shard = self._get_shard(k_str)
        with shard.lock:
            if k_str in shard.entries:
                del shard.entries[k_str]
                return True
            return False

    def stats(self) -> Dict[str, int]:
        total = sum(len(s.entries) for s in self._shards)
        return {"tracked_keys": total, "num_shards": self._num_shards}

    def stop(self) -> None:
        self._stop_event.set()

    # --- Internal -----------------------------------------------------------

    def _get_shard(self, key_str: str) -> _Shard:
        idx = int(hashlib.md5(key_str.encode(), usedforsecurity=False).hexdigest(), 16)
        return self._shards[idx % self._num_shards]

    def _cleanup_loop(self) -> None:
        while not self._stop_event.wait(CLEANUP_INTERVAL):
            self._purge_expired()

    def _purge_expired(self) -> int:
        now    = time.monotonic()
        purged = 0
        for shard in self._shards:
            with shard.lock:
                expired = [k for k, e in shard.entries.items() if now >= e.expires_at]
                for k in expired:
                    del shard.entries[k]
                    purged += 1
        return purged

# Test Cases

def _key(t: str = "sql_injection", ip: str = "1.2.3.4", rule: str = "R001") -> AlertKey:
    return AlertKey(alert_type=t, source_ip=ip, rule_id=rule)


def test_first_alert_passes():
    d = ShardedAlertDedup(window_s=60)
    r = d.check(_key())
    assert r.is_new
    assert r.suppressed_count == 0
    print("First alert passes: passed")
    d.stop()


def test_duplicate_suppressed():
    d = ShardedAlertDedup(window_s=60)
    r1 = d.check(_key())
    r2 = d.check(_key())
    r3 = d.check(_key())
    assert r1.is_new
    assert not r2.is_new
    assert r3.suppressed_count == 2
    print("Duplicate suppressed: passed")
    d.stop()


def test_different_keys_independent():
    d  = ShardedAlertDedup(window_s=60)
    r1 = d.check(_key(ip="1.1.1.1"))
    r2 = d.check(_key(ip="2.2.2.2"))
    assert r1.is_new and r2.is_new
    print("Different keys independent: passed")
    d.stop()


def test_expiry_allows_new():
    d  = ShardedAlertDedup(window_s=0.05)
    r1 = d.check(_key())
    assert r1.is_new
    time.sleep(0.1)
    r2 = d.check(_key())
    assert r2.is_new
    print("Expiry allows new alert: passed")
    d.stop()


def test_force_expire():
    d  = ShardedAlertDedup(window_s=60)
    k  = _key()
    d.check(k)
    assert not d.check(k).is_new
    d.force_expire(k)
    assert d.check(k).is_new
    print("Force expire: passed")
    d.stop()


def test_concurrent_workers():
    """Many threads dedup the same key; only the first should be is_new."""
    d       = ShardedAlertDedup(window_s=60)
    k       = _key()
    new_cnt = []
    dup_cnt = []
    lock    = threading.Lock()
    errors  = []

    def worker():
        try:
            r = d.check(k)
            with lock:
                if r.is_new:
                    new_cnt.append(1)
                else:
                    dup_cnt.append(1)
        except Exception as e:
            with lock:
                errors.append(str(e))

    threads = [threading.Thread(target=worker) for _ in range(50)]
    for t in threads:
        t.start()
    for t in threads:
        t.join()

    assert not errors
    # Exactly one thread should see is_new=True
    assert len(new_cnt) == 1, f"Expected 1 new, got {len(new_cnt)}"
    assert len(dup_cnt) == 49
    print(f"Concurrent: new={len(new_cnt)}, dup={len(dup_cnt)} — passed")
    d.stop()


def test_per_type_window():
    d = ShardedAlertDedup(
        window_s=60,
        per_type_windows={"brute_force": 0.05},
    )
    k = _key(t="brute_force")
    d.check(k)
    assert not d.check(k).is_new
    time.sleep(0.1)
    assert d.check(k).is_new    # expired; new window
    print("Per-type window: passed")
    d.stop()

"""
ALERT DEDUPLICATION IN SECURITY OPERATIONS
-------------------------------------------
Why it matters:
  A single attacker action (e.g., port scan, brute force) triggers
  thousands of identical alerts within seconds. Without dedup:
  - SIEM queue overflows.
  - Analysts become desensitized (alert fatigue).
  - Critical new alerts buried under noise.

SUPPRESSION WINDOW TUNING
--------------------------
  Too short (< 1 min): duplicate storms overwhelm the analyst.
  Too long (> 24h): new occurrences of the same attack are silently dropped.
  Best practice: per-rule, per-severity tuning.
  Critical rules: short windows (1–5 min) or no suppression.
  Noisy info rules: long windows (1–4 h).

DEDUP KEY DESIGN
-----------------
  Granularity matters:
    Too broad:  ("sql_injection", *, *)  → ALL SQLi from any source suppressed.
    Too narrow: (type, ip, rule, port, payload_hash) → nearly no dedup.
  Good default: (alert_type, source_ip, rule_id) with a 5-min window.
  For credential stuffing: (alert_type, target_account) instead of source_ip
    to dedup across IP rotation.

DISTRIBUTED DEDUP
-----------------
  Redis SET NX + EXPIRE:
    result = redis.set(key, 1, nx=True, ex=window_s)
    is_new = (result == True)
  This is atomic (SET NX = set if not exists).
  For count tracking: INCR + EXPIRE on first write.
"""
