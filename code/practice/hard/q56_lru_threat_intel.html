"""
=============================================================================
Q56 [Hard] — LRU Cache for Threat Intelligence Lookups
=============================================================================

PROBLEM
-------
Implement an LRU (Least Recently Used) cache for IOC (Indicator of
Compromise) lookups:
  1. Stores threat intelligence results for IPs, domains, file hashes.
  2. Fixed maximum capacity; evicts least recently used entry when full.
  3. Thread-safe: concurrent readers and writers.
  4. Optional TTL per entry: entries expire even if not evicted.
  5. Hit/miss statistics for monitoring cache effectiveness.
  6. Negative caching: cache "clean" results too.

CLARIFYING QUESTIONS
--------------------
1. Key type? → String (IP, domain, SHA256 hash).
2. Value type? → ThreatResult dataclass.
3. Capacity? → Configurable; default 10,000 entries.
4. TTL? → Per-entry configurable; default 15 minutes.
5. Thread model? → Python threading; single RLock (simple) vs read-write lock.

=============================================================================
APPROACH 1 — NAIVE (unbounded dict, no LRU, not thread-safe)
=============================================================================
  Plain dict; grows unbounded; no eviction; no TTL; not thread-safe.

=============================================================================
APPROACH 2 — LRU CACHE WITH OrderedDict + RLock (OPTIMAL)
=============================================================================
  Use collections.OrderedDict for O(1) LRU eviction:
    - move_to_end(key) on every access.
    - popitem(last=False) to evict the oldest.
  Single RLock protects all mutations.
  TTL stored per entry; expired entries treated as cache miss.

=============================================================================
"""

from __future__ import annotations

import threading
import time
from collections import OrderedDict
from dataclasses import dataclass, field
from enum import Enum
from typing import Dict, Optional


# ---------------------------------------------------------------------------
# Config
# ---------------------------------------------------------------------------

DEFAULT_CAPACITY  = 10_000
DEFAULT_TTL_S     = 900        # 15 minutes
NEGATIVE_TTL_S    = 300        # 5-minute TTL for "clean" results


# ---------------------------------------------------------------------------
# Data Types
# ---------------------------------------------------------------------------

class ThreatLevel(Enum):
    CLEAN     = "clean"
    LOW       = "low"
    MEDIUM    = "medium"
    HIGH      = "high"
    CRITICAL  = "critical"


@dataclass
class ThreatResult:
    ioc: str
    threat_level: ThreatLevel
    categories: list     # e.g., ["malware", "c2"]
    confidence: float    # 0.0–1.0
    source: str          # e.g., "virustotal", "internal"
    last_seen: Optional[str] = None

    @property
    def is_malicious(self) -> bool:
        return self.threat_level not in (ThreatLevel.CLEAN,)


@dataclass
class _CacheEntry:
    result: ThreatResult
    expires_at: float          # monotonic timestamp


@dataclass
class CacheStats:
    hits: int   = 0
    misses: int = 0
    evictions: int = 0
    expirations: int = 0

    @property
    def hit_rate(self) -> float:
        total = self.hits + self.misses
        return self.hits / total if total else 0.0


# ---------------------------------------------------------------------------
# APPROACH 1 — Naive
# ---------------------------------------------------------------------------

class NaiveThreatCache:
    """Unbounded dict; no TTL; no LRU; not thread-safe."""

    def __init__(self):
        self._cache: Dict[str, ThreatResult] = {}

    def get(self, ioc: str) -> Optional[ThreatResult]:
        return self._cache.get(ioc)

    def put(self, result: ThreatResult) -> None:
        self._cache[result.ioc] = result


# ---------------------------------------------------------------------------
# APPROACH 2 — LRU cache with TTL
# ---------------------------------------------------------------------------

class ThreatIntelLRUCache:
    """
    Thread-safe LRU cache for threat intelligence with TTL.

    Data structure:
      OrderedDict[str, _CacheEntry]
        Keys ordered by recency (most recently used at the end).
        move_to_end()  → O(1) access update.
        popitem(last=False) → O(1) evict LRU.

    Locking:
      Single RLock covers all mutations and reads.
      (A read-write lock would improve read concurrency but adds complexity.)
    """

    def __init__(
        self,
        capacity: int    = DEFAULT_CAPACITY,
        default_ttl_s: float = DEFAULT_TTL_S,
        negative_ttl_s: float = NEGATIVE_TTL_S,
    ):
        if capacity <= 0:
            raise ValueError(f"capacity must be positive, got {capacity}")
        self._capacity       = capacity
        self._default_ttl    = default_ttl_s
        self._negative_ttl   = negative_ttl_s
        self._lock           = threading.RLock()
        self._cache: OrderedDict[str, _CacheEntry] = OrderedDict()
        self._stats          = CacheStats()

    # --- Read ---------------------------------------------------------------

    def get(self, ioc: str) -> Optional[ThreatResult]:
        """
        Returns cached ThreatResult or None on miss/expiry.
        Moves accessed entry to most-recently-used position.
        """
        with self._lock:
            entry = self._cache.get(ioc)
            if entry is None:
                self._stats.misses += 1
                return None

            # Check TTL
            if time.monotonic() > entry.expires_at:
                del self._cache[ioc]
                self._stats.misses     += 1
                self._stats.expirations += 1
                return None

            # LRU update: move to end (most recent)
            self._cache.move_to_end(ioc)
            self._stats.hits += 1
            return entry.result

    # --- Write --------------------------------------------------------------

    def put(
        self,
        result: ThreatResult,
        ttl_s: Optional[float] = None,
    ) -> None:
        """
        Insert or update a cache entry.
        Negative results (CLEAN) get a shorter TTL.
        """
        if ttl_s is None:
            ttl_s = (
                self._negative_ttl
                if result.threat_level == ThreatLevel.CLEAN
                else self._default_ttl
            )

        with self._lock:
            now = time.monotonic()
            if result.ioc in self._cache:
                # Update existing; move to most-recent
                self._cache[result.ioc] = _CacheEntry(result=result, expires_at=now + ttl_s)
                self._cache.move_to_end(result.ioc)
            else:
                # Evict LRU if at capacity
                if len(self._cache) >= self._capacity:
                    self._cache.popitem(last=False)
                    self._stats.evictions += 1
                self._cache[result.ioc] = _CacheEntry(result=result, expires_at=now + ttl_s)

    # --- Invalidation -------------------------------------------------------

    def invalidate(self, ioc: str) -> bool:
        """Remove a specific entry. Returns True if it was present."""
        with self._lock:
            if ioc in self._cache:
                del self._cache[ioc]
                return True
            return False

    def clear(self) -> None:
        with self._lock:
            self._cache.clear()

    # --- Stats & introspection ----------------------------------------------

    @property
    def size(self) -> int:
        with self._lock:
            return len(self._cache)

    @property
    def stats(self) -> CacheStats:
        with self._lock:
            return CacheStats(
                hits=self._stats.hits,
                misses=self._stats.misses,
                evictions=self._stats.evictions,
                expirations=self._stats.expirations,
            )

    def purge_expired(self) -> int:
        """Remove all expired entries. Returns count purged."""
        now    = time.monotonic()
        purged = 0
        with self._lock:
            expired = [k for k, e in self._cache.items() if now > e.expires_at]
            for k in expired:
                del self._cache[k]
                purged += 1
            self._stats.expirations += purged
        return purged

def _make_result(ioc: str, level: ThreatLevel = ThreatLevel.HIGH) -> ThreatResult:
    return ThreatResult(
        ioc=ioc,
        threat_level=level,
        categories=["malware"],
        confidence=0.9,
        source="test",
    )

# Test Cases
# ---------------------------------------------------------------------------

def test_basic_put_get():
    cache = ThreatIntelLRUCache(capacity=100)
    r = _make_result("192.168.1.1")
    cache.put(r)
    got = cache.get("192.168.1.1")
    assert got is not None
    assert got.threat_level == ThreatLevel.HIGH
    print("Basic put/get: passed")


def test_miss_on_unknown():
    cache = ThreatIntelLRUCache(capacity=100)
    assert cache.get("10.0.0.1") is None
    assert cache.stats.misses == 1
    print("Miss on unknown: passed")


def test_lru_eviction():
    cache = ThreatIntelLRUCache(capacity=3)
    for i in range(3):
        cache.put(_make_result(f"10.0.0.{i}"))

    # Access 10.0.0.0 and 10.0.0.1 (making 10.0.0.2 the LRU)
    cache.get("10.0.0.0")
    cache.get("10.0.0.1")

    # Insert 4th → should evict 10.0.0.2 (LRU)
    cache.put(_make_result("10.0.0.3"))
    assert cache.get("10.0.0.2") is None     # evicted
    assert cache.get("10.0.0.0") is not None # still present
    assert cache.stats.evictions == 1
    print("LRU eviction: passed")


def test_ttl_expiry():
    cache = ThreatIntelLRUCache(capacity=100, default_ttl_s=0.05)
    cache.put(_make_result("evil.com"))
    assert cache.get("evil.com") is not None
    time.sleep(0.1)
    assert cache.get("evil.com") is None
    assert cache.stats.expirations == 1
    print("TTL expiry: passed")


def test_negative_caching():
    cache = ThreatIntelLRUCache(capacity=100, negative_ttl_s=0.05)
    clean = ThreatResult(
        ioc="safe.com", threat_level=ThreatLevel.CLEAN,
        categories=[], confidence=1.0, source="test",
    )
    cache.put(clean)
    assert cache.get("safe.com") is not None
    time.sleep(0.1)
    assert cache.get("safe.com") is None    # negative TTL expired
    print("Negative caching TTL: passed")


def test_invalidate():
    cache = ThreatIntelLRUCache(capacity=100)
    cache.put(_make_result("badhost.com"))
    assert cache.invalidate("badhost.com") == True
    assert cache.get("badhost.com") is None
    assert cache.invalidate("badhost.com") == False
    print("Invalidate: passed")


def test_concurrent_reads_writes():
    cache   = ThreatIntelLRUCache(capacity=200)
    errors  = []
    results = []
    lock    = threading.Lock()

    def writer(n):
        try:
            for i in range(10):
                cache.put(_make_result(f"10.{n}.0.{i}"))
        except Exception as e:
            with lock:
                errors.append(str(e))

    def reader(n):
        try:
            for i in range(10):
                val = cache.get(f"10.{n}.0.{i}")
                with lock:
                    results.append(val)
        except Exception as e:
            with lock:
                errors.append(str(e))

    threads = (
        [threading.Thread(target=writer, args=(n,)) for n in range(5)] +
        [threading.Thread(target=reader, args=(n,)) for n in range(5)]
    )
    for t in threads:
        t.start()
    for t in threads:
        t.join()

    assert not errors, f"Errors: {errors}"
    print(f"Concurrent r/w: size={cache.size}, stats={cache.stats} — passed")


def test_hit_rate():
    cache = ThreatIntelLRUCache(capacity=100)
    for i in range(5):
        cache.put(_make_result(f"ip-{i}"))
    for i in range(5):
        cache.get(f"ip-{i}")        # 5 hits
    for i in range(5):
        cache.get(f"ip-miss-{i}")   # 5 misses
    s = cache.stats
    assert s.hit_rate == 0.5
    print(f"Hit rate: {s.hit_rate:.0%} — passed")

"""
THREAT INTELLIGENCE CACHING
-----------------------------
Why cache TI lookups?
  External TI APIs (VirusTotal, Shodan, AbuseIPDB) have rate limits
  and latency (100–500ms). Caching reduces:
    - API cost (fewer calls).
    - Latency (sub-millisecond cache hit vs 200ms API call).
    - Single point of failure dependency on external service.

CACHE POISONING RISKS
----------------------
  Attacker controls IOC store → injects "clean" verdict for known-malicious IPs.
  Defenses:
    - Sign TI data with HMAC before caching; verify on read.
    - TTL-bound: even poisoned entries expire.
    - Multiple TI sources; cross-check before caching.

NEGATIVE CACHING TRADEOFFS
---------------------------
  Cache "clean" verdicts with shorter TTL.
  Pro: avoids re-querying safe traffic (reduces noise).
  Con: newly malicious IOC stays "clean" until TTL expires.
  Mitigation: shorter TTL for CLEAN; push invalidation on TI feed updates.

LRU VS LFU FOR TI LOOKUPS
---------------------------
  LRU: assumes recent = likely to be queried again.
  LFU: assumes frequent = important to keep.
  For TI: LRU is better since active threats trend in bursts.
  LFU risks keeping stale popular IOCs while evicting new threats.

DISTRIBUTED TI CACHE
---------------------
  Redis with TTL-aware keys:
    SET ioc:192.168.1.1 <json_result> EX 900
  Consistent hashing for multi-node Redis.
  Read-through cache: on miss, query TI API; write result back.
"""
