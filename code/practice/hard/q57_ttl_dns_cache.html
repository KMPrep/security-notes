"""
=============================================================================
Q57 [Hard] — TTL Cache for DNS Reputation
=============================================================================

PROBLEM
-------
Implement a TTL cache where each DNS reputation result expires after a
configured time-to-live:
  1. Lookup domain → returns cached ReputationResult or None on miss.
  2. Each entry has its own TTL (set either globally or from the DNS response).
  3. Expired entries are treated as misses; auto-purged in the background.
  4. Thread-safe for concurrent DNS resolvers.
  5. Negative caching: NXDOMAIN/clean results cached with shorter TTL.
  6. Size limit: cap total entries; evict soonest-to-expire on overflow.

CLARIFYING QUESTIONS
--------------------
1. Key? → Domain name (lowercased, stripped).
2. TTL source? → Configurable default; can be overridden per entry.
3. Thread model? → threading module.
4. Eviction on overflow? → Min-heap by expires_at (smallest TTL first).
5. Background cleanup? → Daemon thread running every N seconds.

=============================================================================
APPROACH 1 — NAIVE (dict + time check, no eviction, not thread-safe)
=============================================================================
  Simple dict; TTL checked on access only; no background cleanup; unbounded.

=============================================================================
APPROACH 2 — HEAP-BASED TTL CACHE WITH LAZY EXPIRY (OPTIMAL)
=============================================================================
  1. dict[domain → _Entry] for O(1) lookup.
  2. min-heap[(expires_at, domain)] for O(log N) overflow eviction.
  3. Lazy expiry: check TTL on every get(); also background thread purges.
  4. Overflow: before inserting, heappop the min-expiry entry.
  5. RLock protects all state.

=============================================================================
"""

from __future__ import annotations

import heapq
import threading
import time
from dataclasses import dataclass, field
from enum import Enum
from typing import Dict, List, Optional, Tuple


# ---------------------------------------------------------------------------
# Config
# ---------------------------------------------------------------------------

DEFAULT_CAPACITY      = 50_000
DEFAULT_TTL_S         = 300.0    # 5 minutes
NEGATIVE_TTL_S        = 60.0     # 1 minute for NXDOMAIN / clean
CLEANUP_INTERVAL_S    = 30.0     # background cleanup frequency


# ---------------------------------------------------------------------------
# Data Types
# ---------------------------------------------------------------------------

class DnsReputation(Enum):
    UNKNOWN   = "unknown"
    CLEAN     = "clean"
    SUSPICIOUS= "suspicious"
    MALICIOUS = "malicious"
    NXDOMAIN  = "nxdomain"


@dataclass
class ReputationResult:
    domain: str
    reputation: DnsReputation
    categories: List[str]
    confidence: float           # 0.0–1.0
    ttl_s: Optional[float] = None   # override cache TTL from DNS response


@dataclass
class _Entry:
    result: ReputationResult
    expires_at: float    # monotonic timestamp


@dataclass
class CacheStats:
    hits: int       = 0
    misses: int     = 0
    evictions: int  = 0
    expirations: int = 0

    @property
    def hit_rate(self) -> float:
        total = self.hits + self.misses
        return self.hits / total if total else 0.0


# ---------------------------------------------------------------------------
# APPROACH 1 — Naive
# ---------------------------------------------------------------------------

class NaiveDnsReputationCache:
    """Not thread-safe; unbounded; TTL only checked on access."""

    def __init__(self, default_ttl_s: float = DEFAULT_TTL_S):
        self._ttl  = default_ttl_s
        self._data: Dict[str, _Entry] = {}

    def get(self, domain: str) -> Optional[ReputationResult]:
        e = self._data.get(domain.lower())
        if e and time.monotonic() <= e.expires_at:
            return e.result
        return None

    def put(self, result: ReputationResult) -> None:
        ttl = result.ttl_s or self._ttl
        self._data[result.domain.lower()] = _Entry(result, time.monotonic() + ttl)


# ---------------------------------------------------------------------------
# APPROACH 2 — Heap-based TTL cache
# ---------------------------------------------------------------------------

class DnsReputationCache:
    """
    Thread-safe DNS reputation TTL cache.

    Internals:
      _cache:  dict[domain → _Entry]      — O(1) lookup
      _heap:   min-heap[(expires_at, domain)]  — overflow eviction

    Overflow policy: evict the soonest-expiring entry (weakest TTL).
    This retains long-lived, highly confident results while shedding
    entries that were about to expire anyway.
    """

    def __init__(
        self,
        capacity:       int   = DEFAULT_CAPACITY,
        default_ttl_s:  float = DEFAULT_TTL_S,
        negative_ttl_s: float = NEGATIVE_TTL_S,
        cleanup_interval_s: float = CLEANUP_INTERVAL_S,
    ):
        if capacity <= 0:
            raise ValueError("capacity must be positive")
        self._capacity       = capacity
        self._default_ttl    = default_ttl_s
        self._negative_ttl   = negative_ttl_s
        self._lock           = threading.RLock()
        self._cache:  Dict[str, _Entry]            = {}
        self._heap:   List[Tuple[float, str]]      = []   # (expires_at, domain)
        self._stats          = CacheStats()
        self._stop           = threading.Event()
        self._bg             = threading.Thread(
            target=self._cleanup_loop, daemon=True, name="dns-ttl-cleanup"
        )
        self._bg.start()

    # --- Public API ---------------------------------------------------------

    def get(self, domain: str) -> Optional[ReputationResult]:
        key = domain.strip().lower()
        with self._lock:
            entry = self._cache.get(key)
            if entry is None:
                self._stats.misses += 1
                return None
            if time.monotonic() > entry.expires_at:
                del self._cache[key]
                self._stats.misses      += 1
                self._stats.expirations += 1
                return None
            self._stats.hits += 1
            return entry.result

    def put(self, result: ReputationResult) -> None:
        key = result.domain.strip().lower()
        ttl = result.ttl_s

        if ttl is None:
            if result.reputation in (DnsReputation.CLEAN, DnsReputation.NXDOMAIN,
                                      DnsReputation.UNKNOWN):
                ttl = self._negative_ttl
            else:
                ttl = self._default_ttl

        now        = time.monotonic()
        expires_at = now + ttl
        entry      = _Entry(result=result, expires_at=expires_at)

        with self._lock:
            if key not in self._cache and len(self._cache) >= self._capacity:
                self._evict_one_locked()

            self._cache[key] = entry
            heapq.heappush(self._heap, (expires_at, key))

    def invalidate(self, domain: str) -> bool:
        key = domain.strip().lower()
        with self._lock:
            if key in self._cache:
                del self._cache[key]
                return True
            return False

    def purge_expired(self) -> int:
        now    = time.monotonic()
        purged = 0
        with self._lock:
            # Drain heap of already-expired or stale references
            while self._heap and self._heap[0][0] <= now:
                exp, key = heapq.heappop(self._heap)
                entry = self._cache.get(key)
                # Only delete if this heap entry matches the current entry
                if entry is not None and entry.expires_at <= now:
                    del self._cache[key]
                    purged += 1
            self._stats.expirations += purged
        return purged

    def stop(self) -> None:
        self._stop.set()

    @property
    def size(self) -> int:
        with self._lock:
            return len(self._cache)

    @property
    def stats(self) -> CacheStats:
        with self._lock:
            return CacheStats(
                hits=self._stats.hits,
                misses=self._stats.misses,
                evictions=self._stats.evictions,
                expirations=self._stats.expirations,
            )

    # --- Internal -----------------------------------------------------------

    def _evict_one_locked(self) -> None:
        """Evict the soonest-expiring entry. Must hold lock."""
        while self._heap:
            exp, key = heapq.heappop(self._heap)
            if key in self._cache:
                del self._cache[key]
                self._stats.evictions += 1
                return

    def _cleanup_loop(self) -> None:
        while not self._stop.wait(CLEANUP_INTERVAL_S):
            self.purge_expired()

# Test helpers

def _result(
    domain: str,
    rep: DnsReputation = DnsReputation.MALICIOUS,
    ttl: Optional[float] = None,
) -> ReputationResult:
    return ReputationResult(
        domain=domain, reputation=rep,
        categories=["malware"], confidence=0.95, ttl_s=ttl,
    )


# ---------------------------------------------------------------------------
# Test Cases
# ---------------------------------------------------------------------------

def test_basic_put_get():
    cache = DnsReputationCache(capacity=100)
    cache.put(_result("evil.example.com"))
    r = cache.get("evil.example.com")
    assert r is not None
    assert r.reputation == DnsReputation.MALICIOUS
    print("Basic put/get: passed")
    cache.stop()


def test_miss_on_unknown():
    cache = DnsReputationCache(capacity=100)
    assert cache.get("unknown.domain") is None
    assert cache.stats.misses == 1
    print("Miss on unknown: passed")
    cache.stop()


def test_ttl_expiry():
    cache = DnsReputationCache(capacity=100, default_ttl_s=0.05)
    cache.put(_result("short.ttl.com"))
    assert cache.get("short.ttl.com") is not None
    time.sleep(0.1)
    assert cache.get("short.ttl.com") is None
    print("TTL expiry: passed")
    cache.stop()


def test_negative_ttl():
    cache = DnsReputationCache(capacity=100, negative_ttl_s=0.05)
    cache.put(_result("safe.com", rep=DnsReputation.CLEAN))
    assert cache.get("safe.com") is not None
    time.sleep(0.1)
    assert cache.get("safe.com") is None
    print("Negative TTL (clean): passed")
    cache.stop()


def test_per_entry_ttl_override():
    cache = DnsReputationCache(capacity=100, default_ttl_s=60)
    cache.put(_result("short.custom.com", ttl=0.05))
    time.sleep(0.1)
    assert cache.get("short.custom.com") is None
    print("Per-entry TTL override: passed")
    cache.stop()


def test_overflow_eviction():
    cache = DnsReputationCache(capacity=3)
    for i in range(3):
        cache.put(_result(f"d{i}.com", ttl=300))
    assert cache.size == 3
    cache.put(_result("d3.com", ttl=300))
    assert cache.size == 3   # still at capacity; one was evicted
    assert cache.stats.evictions == 1
    print("Overflow eviction: passed")
    cache.stop()


def test_case_insensitive():
    cache = DnsReputationCache(capacity=100)
    cache.put(_result("Evil.EXAMPLE.COM"))
    assert cache.get("evil.example.com") is not None
    assert cache.get("EVIL.EXAMPLE.COM") is not None
    print("Case-insensitive lookup: passed")
    cache.stop()


def test_invalidate():
    cache = DnsReputationCache(capacity=100)
    cache.put(_result("bad.com"))
    assert cache.invalidate("bad.com") == True
    assert cache.get("bad.com") is None
    print("Invalidate: passed")
    cache.stop()


def test_concurrent_access():
    cache   = DnsReputationCache(capacity=500)
    errors  = []
    lock    = threading.Lock()

    def worker(n):
        try:
            domain = f"domain{n % 20}.com"
            cache.put(_result(domain, ttl=60))
            cache.get(domain)
        except Exception as e:
            with lock:
                errors.append(str(e))

    threads = [threading.Thread(target=worker, args=(i,)) for i in range(100)]
    for t in threads:
        t.start()
    for t in threads:
        t.join()

    assert not errors, f"Errors: {errors}"
    s = cache.stats
    print(f"Concurrent: hits={s.hits}, misses={s.misses} — passed")
    cache.stop()

"""
DNS REPUTATION CACHING IN SECURITY
-------------------------------------
Use cases:
  1. Block malicious domains at DNS resolver level (RPZ — Response Policy Zones).
  2. Inline TI lookup on every DNS query (cache hit = ~0ms; miss = 50–200ms).
  3. Detect DGA (Domain Generation Algorithm) domains: entropy-high, new.

TTL POLICY DESIGN
-----------------
  Malicious domain:  Long TTL (5–15 min) — these rarely flip to clean.
  NXDOMAIN:          Short TTL (1–5 min) — may be registered soon (DGA).
  Clean domain:      Medium TTL (5–10 min) — may become malicious (takeover).
  Infrastructure IPs: Longer TTL (1h) — IPs change slowly.

DNS CACHE POISONING (UNRELATED TO TI CACHE)
--------------------------------------------
  Attacker injects forged DNS responses into resolver cache.
  Result: victim's browser goes to attacker's IP for "bank.com".
  Defenses:
    - DNSSEC: cryptographically signs DNS records.
    - DNS-over-HTTPS / DNS-over-TLS: prevents interception.
    - Random source port + query ID entropy (Kaminsky-resistant resolvers).

STALE WHILE REVALIDATE (PRODUCTION PATTERN)
--------------------------------------------
  Serve stale cache result while asynchronously re-fetching from TI API.
  Prevents cold-start latency spikes; safe because:
    - TI verdicts rarely reverse instantly.
    - TTL expiry is a soft deadline.
  Dangerous if: you use stale results to authorize (not just detect).
"""
