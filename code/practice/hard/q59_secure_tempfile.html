"""
=============================================================================
Q59 [Hard] — Secure Temporary File Handling
=============================================================================

PROBLEM
-------
Write code that safely creates, writes, and deletes temporary files
without race or symlink attacks:
  1. Create temp files in a secure directory (not world-writable).
  2. Prevent TOCTOU race between check and create.
  3. Prevent symlink attacks (attacker replaces temp path with symlink
     before the program opens it).
  4. Ensure cleanup even if an exception occurs.
  5. Support temp directories (for multi-file operations).
  6. Restrict permissions: 0o600 (owner read/write only).

CLARIFYING QUESTIONS
--------------------
1. OS? → Unix/Linux primary; Windows noted.
2. Temp content? → Arbitrary bytes or text.
3. Persistence? → All temp files deleted after use.
4. Multi-file? → Support temp directory context manager.
5. Symlink protection? → O_NOFOLLOW on open; verify realpath.

=============================================================================
APPROACH 1 — NAIVE (tempfile path with open(), predictable names)
=============================================================================
  /tmp/myapp_<pid> → predictable name; attacker creates symlink before us.
  TOCTOU: check path doesn't exist → sleep → create (race window).

=============================================================================
APPROACH 2 — SECURE TEMPFILE WITH MKSTEMP + O_NOFOLLOW (OPTIMAL)
=============================================================================
  1. tempfile.mkstemp(): atomic file creation; returns fd + path.
     No TOCTOU window; OS creates the file exclusively.
  2. O_NOFOLLOW on re-open (if needed): fails if path is a symlink.
  3. Restrict permissions to 0o600 immediately after creation.
  4. Use tempfile.mkdtemp() for temp directories.
  5. Context managers ensure cleanup on all exit paths.
  6. Verify realpath stays within expected temp dir.

=============================================================================
"""

from __future__ import annotations

import os
import stat
import tempfile
from contextlib import contextmanager
from dataclasses import dataclass
from typing import Iterator, Optional


# ---------------------------------------------------------------------------
# Config
# ---------------------------------------------------------------------------

# Mode: owner read+write only; no group/world access
_SECURE_FILE_MODE = 0o600
_SECURE_DIR_MODE  = 0o700

# Maximum temp file size (to prevent runaway writes)
MAX_TEMP_SIZE_BYTES = 10 * 1024 * 1024   # 10 MB


# ---------------------------------------------------------------------------
# Data Types
# ---------------------------------------------------------------------------

@dataclass
class TempFileInfo:
    path: str
    fd: int           # OS file descriptor from mkstemp (already opened)

    def __enter__(self) -> "TempFileInfo":
        return self

    def __exit__(self, *args) -> None:
        self.close_and_delete()

    def close_and_delete(self) -> None:
        # Close FD if still open
        try:
            os.close(self.fd)
        except OSError:
            pass
        _secure_delete(self.path)


class SecureTempError(Exception):
    pass


# ---------------------------------------------------------------------------
# APPROACH 1 — Naive (DO NOT USE)
# ---------------------------------------------------------------------------

def naive_write_temp(data: bytes, filename: str = "tmp_data.bin") -> str:
    """
    INSECURE: predictable /tmp path + TOCTOU + no permission restriction.
    Included only to illustrate what NOT to do.
    """
    path = f"/tmp/{filename}"
    with open(path, "wb") as f:   # creates with umask permissions
        f.write(data)
    return path


# ---------------------------------------------------------------------------
# Secure helpers
# ---------------------------------------------------------------------------

def _secure_delete(path: str) -> None:
    """Delete a file, ignoring errors if already gone."""
    try:
        os.unlink(path)
    except FileNotFoundError:
        pass
    except PermissionError:
        pass


def _verify_within_dir(path: str, expected_dir: str) -> None:
    """
    Resolve symlinks in path; ensure result is under expected_dir.
    Raises SecureTempError if path escapes the directory.
    """
    real = os.path.realpath(path)
    base = os.path.realpath(expected_dir)
    if not real.startswith(base + os.sep) and real != base:
        raise SecureTempError(
            f"Path escape detected: {path!r} resolves to {real!r}, "
            f"outside {base!r}"
        )


def _restrict_permissions(path: str, mode: int = _SECURE_FILE_MODE) -> None:
    """Set file permissions; removes group and world bits."""
    os.chmod(path, mode)


# ---------------------------------------------------------------------------
# APPROACH 2A — Secure single temp file
# ---------------------------------------------------------------------------

@contextmanager
def secure_temp_file(
    suffix: str = "",
    prefix: str = "sec_",
    dir: Optional[str] = None,
    text: bool = False,
) -> Iterator[TempFileInfo]:
    """
    Context manager that provides a secure temporary file.
    File is created atomically with mkstemp; deleted on exit.

    Usage:
        with secure_temp_file(suffix=".bin") as tmp:
            os.write(tmp.fd, b"secret data")
            # path is tmp.path
    """
    tmp_dir = dir or tempfile.gettempdir()
    fd, path = tempfile.mkstemp(suffix=suffix, prefix=prefix, dir=tmp_dir, text=text)

    try:
        # Immediately restrict permissions
        _restrict_permissions(path, _SECURE_FILE_MODE)

        # Verify realpath stays within expected directory
        _verify_within_dir(path, tmp_dir)

        yield TempFileInfo(fd=fd, path=path)
    finally:
        # Close FD if caller didn't
        try:
            os.close(fd)
        except OSError:
            pass
        _secure_delete(path)


@contextmanager
def secure_temp_dir(
    suffix: str = "",
    prefix: str = "sec_dir_",
    base_dir: Optional[str] = None,
) -> Iterator[str]:
    """
    Context manager for a secure temporary directory.
    Directory is removed with all contents on exit.
    """
    import shutil
    tmp_base = base_dir or tempfile.gettempdir()
    tmp_dir  = tempfile.mkdtemp(suffix=suffix, prefix=prefix, dir=tmp_base)

    try:
        _restrict_permissions(tmp_dir, _SECURE_DIR_MODE)
        _verify_within_dir(tmp_dir, tmp_base)
        yield tmp_dir
    finally:
        shutil.rmtree(tmp_dir, ignore_errors=True)


# ---------------------------------------------------------------------------
# APPROACH 2B — Write helper with size limit
# ---------------------------------------------------------------------------

def write_secure_temp(
    data: bytes,
    suffix: str = ".tmp",
    prefix: str = "sec_",
) -> str:
    """
    Write bytes to a secure temp file; returns path.
    Caller is responsible for deleting the file.
    Use secure_temp_file() context manager for automatic cleanup.
    """
    if len(data) > MAX_TEMP_SIZE_BYTES:
        raise SecureTempError(
            f"Data size {len(data)} exceeds MAX_TEMP_SIZE_BYTES {MAX_TEMP_SIZE_BYTES}"
        )

    fd, path = tempfile.mkstemp(suffix=suffix, prefix=prefix)
    try:
        _restrict_permissions(path, _SECURE_FILE_MODE)
        _verify_within_dir(path, tempfile.gettempdir())
        os.write(fd, data)
    except Exception:
        os.close(fd)
        _secure_delete(path)
        raise
    finally:
        try:
            os.close(fd)
        except OSError:
            pass
    return path

# Test Cases

def test_secure_file_created():
    with secure_temp_file(suffix=".test") as tmp:
        os.write(tmp.fd, b"hello secure world")
        # Verify file exists and has correct permissions
        assert os.path.exists(tmp.path)
        mode = stat.S_IMODE(os.stat(tmp.path).st_mode)
        assert mode == _SECURE_FILE_MODE, f"Expected 0o600, got {oct(mode)}"
        path_snapshot = tmp.path
    # File should be deleted after context exit
    assert not os.path.exists(path_snapshot)
    print("Secure temp file created with 0o600 and deleted: passed")


def test_write_and_read_back():
    payload = b"sensitive_payload_xyz_" + os.urandom(32)
    with secure_temp_file() as tmp:
        os.write(tmp.fd, payload)
        # Re-open for reading (fd may be write-only on some systems)
        with open(tmp.path, "rb") as f:
            data = f.read()
    assert data == payload
    print("Write and read back: passed")


def test_cleanup_on_exception():
    path_snapshot = None
    try:
        with secure_temp_file() as tmp:
            path_snapshot = tmp.path
            assert os.path.exists(path_snapshot)
            raise ValueError("simulated error")
    except ValueError:
        pass
    assert path_snapshot is not None
    assert not os.path.exists(path_snapshot)
    print("Cleanup on exception: passed")


def test_size_limit_enforced():
    big = os.urandom(MAX_TEMP_SIZE_BYTES + 1)
    try:
        write_secure_temp(big)
        assert False, "Should have raised"
    except SecureTempError as e:
        assert "MAX_TEMP_SIZE_BYTES" in str(e)
    print("Size limit enforced: passed")


def test_secure_temp_dir():
    import os
    contents = None
    with secure_temp_dir(prefix="test_dir_") as tmp_dir:
        assert os.path.isdir(tmp_dir)
        mode = stat.S_IMODE(os.stat(tmp_dir).st_mode)
        assert mode == _SECURE_DIR_MODE, f"Expected 0o700, got {oct(mode)}"
        # Write files inside
        fname = os.path.join(tmp_dir, "secret.bin")
        with open(fname, "wb") as f:
            f.write(b"temp_secret")
        assert os.path.exists(fname)
        contents = fname
    # Directory and all contents removed
    assert not os.path.exists(tmp_dir)
    assert not os.path.exists(contents)
    print("Secure temp directory created and cleaned up: passed")


def test_path_stays_in_tmp():
    """Verify realpath doesn't escape the tmp directory."""
    path = write_secure_temp(b"data", suffix=".bin")
    tmp_dir = tempfile.gettempdir()
    real = os.path.realpath(path)
    assert real.startswith(os.path.realpath(tmp_dir))
    _secure_delete(path)
    print("Path stays within temp dir: passed")


def test_write_secure_temp_returns_valid_path():
    path = write_secure_temp(b"test_data_123")
    try:
        assert os.path.exists(path)
        with open(path, "rb") as f:
            assert f.read() == b"test_data_123"
        mode = stat.S_IMODE(os.stat(path).st_mode)
        assert mode == _SECURE_FILE_MODE
    finally:
        _secure_delete(path)
    print("write_secure_temp returns valid path: passed")

"""
SECURE TEMPORARY FILE VULNERABILITIES
---------------------------------------
1. TOCTOU (TIME-OF-CHECK TO TIME-OF-USE) / TMPRACE
---------------------------------------------------
  if not os.path.exists("/tmp/myapp.tmp"):   ← check
      # attacker creates symlink here
      open("/tmp/myapp.tmp", "w").write(data)  ← use (writes through symlink)

  Result: writes data to attacker-controlled target file.
  Fix: mkstemp() — O_CREAT|O_EXCL atomically. No race window.

2. SYMLINK ATTACK
-----------------
  Attacker creates: /tmp/myapp.tmp → /etc/cron.d/evil
  Program writes to /tmp/myapp.tmp → overwrites /etc/cron.d/evil.
  Fix: O_NOFOLLOW when re-opening (fails if path is a symlink);
       mkstemp() already creates exclusive non-symlink file.
       Verify realpath after open.

3. PREDICTABLE FILENAME
-----------------------
  /tmp/myapp_<pid>.tmp → attacker knows the PID range.
  mkstemp() uses 6+ random chars (e.g., /tmp/tmpXXXXXX).
  Additional fix: use /tmp/user_<uid>/ directory (sticky bit UID isolation).

4. PERMISSIONS LEAK
-------------------
  mkstemp() uses umask; if umask=000, file is 0o666.
  Fix: immediately chmod to 0o600 after creation.
  Alternatively: temporarily set umask to 0o077 during creation.

5. SENSITIVE DATA REMANENCE
----------------------------
  Temp files with sensitive data deleted but content still on disk.
  For high-security: overwrite with zeros before deletion.
  Note: SSD / filesystem journaling may preserve data regardless.

UNIX TMPDIR ISOLATION (HARDENING)
-----------------------------------
  /tmp has sticky bit (1777): each user can only delete their own files.
  Better: private per-service /run/myapp/ directory (0o700, owned by service).
  Container: mount /tmp as tmpfs (in-memory; no disk remanence).

WINDOWS NOTE
-------------
  tempfile.mkstemp() works on Windows but O_NOFOLLOW is not available.
  Use GetTempFileName() Win32 API for equivalent atomic creation.
  NTFS junction points = Windows symlinks; similar attack vector.
"""
