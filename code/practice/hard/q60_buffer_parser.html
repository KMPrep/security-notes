"""
=============================================================================
Q60 [Hard] — Memory-Safe Buffer Parser
=============================================================================

PROBLEM
-------
Given a binary packet format, parse fields safely:
  1. Check bounds before every read.
  2. Reject malformed input early (wrong magic, truncated, bad length field).
  3. Prevent integer overflow on length fields.
  4. Return structured parsed result or detailed error.
  5. Support variable-length fields (length-prefixed strings, TLV).

Packet format (custom binary protocol):
  ┌──────────────────────────────────────────────────────────────────┐
  │ Magic     4 bytes   0xDEADBEEF                                   │
  │ Version   1 byte    1 or 2                                       │
  │ Flags     1 byte    bitmask                                      │
  │ Sequence  4 bytes   big-endian uint32                            │
  │ Length    4 bytes   big-endian uint32 (total payload length)     │
  │ Payload   N bytes   TLV records until end                        │
  │ Checksum  4 bytes   big-endian uint32 (CRC32 of header+payload)  │
  └──────────────────────────────────────────────────────────────────┘

TLV (Type-Length-Value) inside payload:
  Type    2 bytes  uint16
  Length  2 bytes  uint16
  Value   Length bytes

CLARIFYING QUESTIONS
--------------------
1. Endianness? → Big-endian (network byte order).
2. Max packet size? → Configurable; default 64 KB.
3. Max TLV value length? → 4 KB per field.
4. What to do on error? → Return ParseError; do not raise exception.
5. Checksum algorithm? → CRC32.

=============================================================================
APPROACH 1 — NAIVE (unguarded struct.unpack, no bounds check)
=============================================================================
  Direct struct.unpack without checking buffer length.
  struct.error on truncated input; no meaningful error messages.

=============================================================================
APPROACH 2 — SAFE _Buf WRAPPER + INCREMENTAL PARSER (OPTIMAL)
=============================================================================
  _Buf class with cursor:
    - read(n): checks remaining >= n before slicing; raises BufUnderflow.
    - peek(n): same but doesn't advance cursor.
    - remaining: bytes left.
  Parser uses _Buf; catches BufUnderflow; returns ParseError.
  Validates: magic, version, length <= max, payload fits, checksum.

=============================================================================
"""

from __future__ import annotations

import struct
import zlib
from dataclasses import dataclass, field
from typing import Dict, List, Optional, Union


# ---------------------------------------------------------------------------
# Config
# ---------------------------------------------------------------------------

MAGIC              = 0xDEADBEEF
SUPPORTED_VERSIONS = {1, 2}
HEADER_SIZE        = 14       # magic(4) + version(1) + flags(1) + seq(4) + length(4)
CHECKSUM_SIZE      = 4
MIN_PACKET_SIZE    = HEADER_SIZE + CHECKSUM_SIZE
MAX_PACKET_SIZE    = 64 * 1024          # 64 KB
MAX_TLV_VALUE_LEN  = 4 * 1024          # 4 KB per TLV field
MAX_TLV_COUNT      = 256

# TLV type constants (illustrative)
TLV_SOURCE_IP  = 0x0001
TLV_DEST_IP    = 0x0002
TLV_PAYLOAD    = 0x0003
TLV_TIMESTAMP  = 0x0004
TLV_USERNAME   = 0x0005


# ---------------------------------------------------------------------------
# Data Types
# ---------------------------------------------------------------------------

@dataclass
class TlvField:
    type_id: int
    value: bytes

    @property
    def as_str(self) -> str:
        return self.value.decode("utf-8", errors="replace")

    @property
    def as_uint32(self) -> Optional[int]:
        if len(self.value) == 4:
            return struct.unpack_from(">I", self.value)[0]
        return None


@dataclass
class ParsedPacket:
    magic: int
    version: int
    flags: int
    sequence: int
    payload_length: int
    tlv_fields: List[TlvField]
    checksum: int
    checksum_valid: bool

    @property
    def flag_encrypted(self) -> bool:
        return bool(self.flags & 0x01)

    @property
    def flag_compressed(self) -> bool:
        return bool(self.flags & 0x02)


@dataclass
class ParseError:
    reason: str
    offset: int = 0

    def __str__(self) -> str:
        return f"ParseError at offset {self.offset}: {self.reason}"


ParseResult = Union[ParsedPacket, ParseError]


# ---------------------------------------------------------------------------
# _Buf — safe buffer reader
# ---------------------------------------------------------------------------

class BufUnderflow(Exception):
    def __init__(self, offset: int, needed: int, remaining: int):
        self.offset    = offset
        self.needed    = needed
        self.remaining = remaining
        super().__init__(
            f"underflow at offset {offset}: need {needed}, have {remaining}"
        )


class _Buf:
    """
    Cursor-based safe buffer reader.
    All read operations check bounds before accessing memory.
    """

    __slots__ = ("_data", "_pos", "_end")

    def __init__(self, data: bytes):
        self._data = data
        self._pos  = 0
        self._end  = len(data)

    @property
    def pos(self) -> int:
        return self._pos

    @property
    def remaining(self) -> int:
        return self._end - self._pos

    def read(self, n: int) -> bytes:
        if n < 0:
            raise ValueError(f"read size must be non-negative, got {n}")
        if self.remaining < n:
            raise BufUnderflow(self._pos, n, self.remaining)
        chunk      = self._data[self._pos: self._pos + n]
        self._pos += n
        return chunk

    def read_u8(self) -> int:
        return struct.unpack_from(">B", self.read(1))[0]

    def read_u16(self) -> int:
        return struct.unpack_from(">H", self.read(2))[0]

    def read_u32(self) -> int:
        return struct.unpack_from(">I", self.read(4))[0]

    def peek(self, n: int) -> bytes:
        if self.remaining < n:
            raise BufUnderflow(self._pos, n, self.remaining)
        return self._data[self._pos: self._pos + n]

    def slice(self, start: int, end: int) -> bytes:
        """Return bytes[start:end] without moving cursor."""
        if end > self._end or start < 0 or start > end:
            raise BufUnderflow(start, end - start, self._end - start)
        return self._data[start:end]


# ---------------------------------------------------------------------------
# APPROACH 1 — Naive (no bounds checking)
# ---------------------------------------------------------------------------

def parse_naive(data: bytes) -> Optional[Dict]:
    """Vulnerable to truncated input (struct.error crash)."""
    try:
        magic, version, flags, seq, length = struct.unpack_from(">IBBII", data, 0)
        payload = data[14: 14 + length]
        checksum = struct.unpack_from(">I", data, 14 + length)[0]
        return {"magic": magic, "version": version, "payload": payload}
    except struct.error:
        return None


# ---------------------------------------------------------------------------
# APPROACH 2 — Safe incremental parser
# ---------------------------------------------------------------------------

def parse_packet(data: bytes) -> ParseResult:
    """
    Parse a binary packet safely with bounds checking at every step.
    Returns ParsedPacket on success or ParseError on any validation failure.
    """
    # --- 0. Input size guards ---
    if not isinstance(data, (bytes, bytearray)):
        return ParseError("input must be bytes", 0)
    if len(data) > MAX_PACKET_SIZE:
        return ParseError(f"packet too large: {len(data)} > {MAX_PACKET_SIZE}", 0)
    if len(data) < MIN_PACKET_SIZE:
        return ParseError(f"packet too short: {len(data)} < {MIN_PACKET_SIZE}", 0)

    buf = _Buf(data)

    # --- 1. Header ---
    try:
        magic    = buf.read_u32()
        version  = buf.read_u8()
        flags    = buf.read_u8()
        sequence = buf.read_u32()
        payload_length = buf.read_u32()
    except BufUnderflow as e:
        return ParseError(f"truncated header: {e}", e.offset)

    # --- 2. Validate magic ---
    if magic != MAGIC:
        return ParseError(f"invalid magic: 0x{magic:08X} (expected 0x{MAGIC:08X})", 0)

    # --- 3. Validate version ---
    if version not in SUPPORTED_VERSIONS:
        return ParseError(f"unsupported version: {version}", 5)

    # --- 4. Validate length field (overflow guard) ---
    if payload_length > MAX_PACKET_SIZE:
        return ParseError(f"payload_length too large: {payload_length}", 10)

    expected_total = HEADER_SIZE + payload_length + CHECKSUM_SIZE
    if len(data) < expected_total:
        return ParseError(
            f"buffer too short: have {len(data)}, need {expected_total}", 14
        )
    if len(data) > expected_total:
        return ParseError(
            f"trailing bytes: have {len(data)}, expected {expected_total}", 14
        )

    # --- 5. Read payload ---
    try:
        payload_bytes = buf.read(payload_length)
    except BufUnderflow as e:
        return ParseError(f"truncated payload: {e}", e.offset)

    # --- 6. Read and verify checksum ---
    try:
        checksum = buf.read_u32()
    except BufUnderflow as e:
        return ParseError(f"truncated checksum: {e}", e.offset)

    # Checksum covers header + payload (not the checksum field itself)
    crc_data  = data[:HEADER_SIZE + payload_length]
    computed  = zlib.crc32(crc_data) & 0xFFFFFFFF
    checksum_valid = (computed == checksum)

    # --- 7. Parse TLV fields ---
    tlv_fields: List[TlvField] = []
    tlv_buf = _Buf(payload_bytes)

    while tlv_buf.remaining > 0:
        if tlv_buf.remaining < 4:
            return ParseError(
                f"truncated TLV header: {tlv_buf.remaining} bytes left",
                HEADER_SIZE + tlv_buf.pos,
            )

        try:
            tlv_type   = tlv_buf.read_u16()
            tlv_length = tlv_buf.read_u16()
        except BufUnderflow as e:
            return ParseError(f"truncated TLV type/length: {e}", e.offset)

        # Validate TLV length
        if tlv_length > MAX_TLV_VALUE_LEN:
            return ParseError(
                f"TLV[type={tlv_type}] length too large: {tlv_length} > {MAX_TLV_VALUE_LEN}",
                HEADER_SIZE + tlv_buf.pos,
            )

        if tlv_buf.remaining < tlv_length:
            return ParseError(
                f"TLV[type={tlv_type}] value truncated: need {tlv_length}, "
                f"have {tlv_buf.remaining}",
                HEADER_SIZE + tlv_buf.pos,
            )

        try:
            value = tlv_buf.read(tlv_length)
        except BufUnderflow as e:
            return ParseError(f"TLV value underflow: {e}", e.offset)

        tlv_fields.append(TlvField(type_id=tlv_type, value=value))

        if len(tlv_fields) > MAX_TLV_COUNT:
            return ParseError(f"too many TLV fields: > {MAX_TLV_COUNT}", HEADER_SIZE)

    return ParsedPacket(
        magic=magic,
        version=version,
        flags=flags,
        sequence=sequence,
        payload_length=payload_length,
        tlv_fields=tlv_fields,
        checksum=checksum,
        checksum_valid=checksum_valid,
    )


# ---------------------------------------------------------------------------
# Packet builder (for testing)
# ---------------------------------------------------------------------------

def build_packet(
    version: int         = 1,
    flags: int           = 0,
    sequence: int        = 1,
    tlv_fields: Optional[List[TlvField]] = None,
) -> bytes:
    """Build a valid packet for testing."""
    payload = b""
    for tlv in (tlv_fields or []):
        payload += struct.pack(">HH", tlv.type_id, len(tlv.value)) + tlv.value

    header = struct.pack(">IBBII", MAGIC, version, flags, sequence, len(payload))
    crc    = zlib.crc32(header + payload) & 0xFFFFFFFF
    checksum_bytes = struct.pack(">I", crc)
    return header + payload + checksum_bytes


# ---------------------------------------------------------------------------
# Test Cases
# ---------------------------------------------------------------------------

def test_valid_packet():
    tlvs = [
        TlvField(TLV_SOURCE_IP, b"\xc0\xa8\x01\x01"),
        TlvField(TLV_USERNAME,  b"alice"),
    ]
    pkt = build_packet(version=1, sequence=42, tlv_fields=tlvs)
    r   = parse_packet(pkt)
    assert isinstance(r, ParsedPacket), str(r)
    assert r.magic    == MAGIC
    assert r.version  == 1
    assert r.sequence == 42
    assert r.checksum_valid
    assert len(r.tlv_fields) == 2
    assert r.tlv_fields[1].as_str == "alice"
    print("Valid packet parsed: passed")


def test_bad_magic():
    pkt = build_packet()
    bad = b"\x00\x00\x00\x00" + pkt[4:]
    r   = parse_packet(bad)
    assert isinstance(r, ParseError)
    assert "magic" in r.reason
    print(f"Bad magic rejected: passed ({r})")


def test_unsupported_version():
    pkt = build_packet(version=99)
    # Build manually to bypass builder's version check
    pkt = bytes([pkt[0], pkt[1], pkt[2], pkt[3], 99]) + pkt[5:]
    r   = parse_packet(pkt)
    assert isinstance(r, ParseError)
    assert "version" in r.reason
    print(f"Unsupported version: passed ({r})")


def test_truncated_header():
    pkt = build_packet()
    r   = parse_packet(pkt[:6])   # cut off mid-header
    assert isinstance(r, ParseError)
    print(f"Truncated header: passed ({r})")


def test_length_field_too_large():
    pkt = bytearray(build_packet())
    # Override length field (bytes 10–13) with huge value
    struct.pack_into(">I", pkt, 10, MAX_PACKET_SIZE + 1)
    r = parse_packet(bytes(pkt))
    assert isinstance(r, ParseError)
    assert "too large" in r.reason
    print(f"Length too large: passed ({r})")


def test_corrupted_checksum():
    pkt      = bytearray(build_packet())
    pkt[-1] ^= 0xFF    # flip last byte of checksum
    r = parse_packet(bytes(pkt))
    assert isinstance(r, ParsedPacket)
    assert not r.checksum_valid
    print("Corrupted checksum detected: passed")


def test_truncated_tlv():
    tlvs = [TlvField(TLV_PAYLOAD, b"hello")]
    pkt  = bytearray(build_packet(tlv_fields=tlvs))
    # Remove last 3 bytes of the TLV value
    trimmed = pkt[:-3 - CHECKSUM_SIZE]  # remove checksum + part of TLV value
    r = parse_packet(bytes(trimmed))
    assert isinstance(r, ParseError)
    print(f"Truncated TLV: passed ({r})")


def test_tlv_length_too_large():
    # Build a packet where TLV length field exceeds MAX_TLV_VALUE_LEN
    payload = struct.pack(">HH", TLV_PAYLOAD, MAX_TLV_VALUE_LEN + 1)
    payload += b"\x00" * (MAX_TLV_VALUE_LEN + 1)
    header   = struct.pack(">IBBII", MAGIC, 1, 0, 1, len(payload))
    crc      = zlib.crc32(header + payload) & 0xFFFFFFFF
    pkt      = header + payload + struct.pack(">I", crc)
    r        = parse_packet(pkt)
    assert isinstance(r, ParseError)
    assert "too large" in r.reason
    print(f"TLV length too large: passed ({r})")


def test_empty_payload():
    pkt = build_packet(version=2, flags=0x01, tlv_fields=[])
    r   = parse_packet(pkt)
    assert isinstance(r, ParsedPacket)
    assert r.version == 2
    assert r.flag_encrypted
    assert r.tlv_fields == []
    print("Empty payload (v2 encrypted flag): passed")


def test_packet_too_short():
    r = parse_packet(b"\xDE\xAD")
    assert isinstance(r, ParseError)
    assert "too short" in r.reason
    print(f"Packet too short: passed ({r})")


def test_packet_too_large():
    r = parse_packet(bytes(MAX_PACKET_SIZE + 1))
    assert isinstance(r, ParseError)
    assert "too large" in r.reason
    print(f"Packet too large: passed ({r})")

"""
BINARY PARSER SECURITY
-----------------------
Binary parsers are a common source of CVEs in:
  - Network protocol stacks (Heartbleed in TLS; EternalBlue in SMB)
  - File format parsers (PDF, PNG, ZIP, ELF, PE)
  - IPC deserialization (D-Bus, protobuf, gRPC framing)

COMMON VULNERABILITIES
-----------------------
1. BUFFER OVERFLOW (CWE-120)
   Read N bytes based on untrusted length field; N > buffer size.
   Fix: always validate length <= remaining bytes before read.

2. INTEGER OVERFLOW IN LENGTH CALCULATION (CWE-190)
   length_field * element_size overflows to small number.
   if (n * sizeof(T) < BUFSIZE)  → n = 2^29, sizeof(T) = 8 → overflows to 0.
   Fix: use checked arithmetic; validate each field independently.

3. OFF-BY-ONE (CWE-193)
   Parse until pos < length instead of pos + record_size <= length.
   Fix: check remaining >= record_size before each read.

4. MISSING MAGIC / VERSION CHECK
   Parser proceeds on garbage input; undefined behavior.
   Fix: validate magic and version as the very first step.

5. CHECKSUM NOT VERIFIED (or verified after use)
   Parser uses data before verifying integrity.
   Fix: verify checksum before processing any field values.

HEARTBLEED (CVE-2014-0160) — CLASSIC EXAMPLE
----------------------------------------------
  OpenSSL TLS heartbeat extension:
    Client: type=1, length=64, "hello"
    Server: echo back length bytes from client message.
  Attacker: type=1, length=65535, "A"
  Server: read 65535 bytes starting from "A" → leaks heap contents.
  Fix: check payload_length <= actual received bytes before echo.

_Buf PATTERN ADVANTAGES
------------------------
  - All reads go through a single bounds-checking method.
  - Cursor tracks position; no manual offset arithmetic (common bug source).
  - BufUnderflow exception carries offset for debugging.
  - Easy to fuzz: any bytes input → either ParsedPacket or ParseError; no crash.
"""
