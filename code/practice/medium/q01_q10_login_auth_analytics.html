# Q1 — Most Talkative IP with Tie-Breaking
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given network logs (list of dicts with field "src_ip"), return the source IP
  that appears most often. If two IPs are tied, return the lexicographically
  smallest one.

Clarifying questions:
  1. Input format — list of dicts, raw strings, or file path?
  2. Should malformed records (missing "src_ip") be skipped silently?
  3. Are IPs always IPv4, or can IPv6 appear?
  4. What is the expected output — a single string?
  5. Is the function called once or in a streaming loop?

Security angle:
  High-volume single-source traffic is a primary indicator of:
  - DoS / DDoS amplification
  - Brute-force login from one source
  - Bot/scanner activity (nikto, masscan, etc.)
  Identifying the noisiest IP lets you prioritise blocking or rate-limiting.
"""

from collections import Counter, defaultdict
import heapq
from typing import List, Dict, Optional, Tuple


# ── Naive ──────────────────────────────────────────────────────────────────
def most_talkative_ip_naive(logs: List[Dict]) -> Optional[str]:
    """
    Build a full count dict, then sort all IPs by (-count, ip) and take first.
    O(n log n) due to sort.
    """
    counts: Dict[str, int] = {}
    for log in logs:
        ip = log.get("src_ip")
        if not ip:
            continue
        counts[ip] = counts.get(ip, 0) + 1

    if not counts:
        return None

    # sort descending by count, then ascending by IP for tie-break
    sorted_ips = sorted(counts.items(), key=lambda kv: (-kv[1], kv[0]))
    return sorted_ips[0][0]


# ── Optimal ────────────────────────────────────────────────────────────────
def most_talkative_ip(logs: List[Dict]) -> Optional[str]:
    """
    Single pass with Counter, then a single linear scan for the winner.
    Avoids full sort by tracking (max_count, best_ip) in one pass.
    O(n) time, O(u) space where u = unique IPs.

    Data structure: Counter (hash map) for O(1) increments.
    Tie-breaking: when counts equal, keep lexicographically smaller IP.
    """
    counts: Counter = Counter()
    for log in logs:
        ip = log.get("src_ip")
        if ip:
            counts[ip] += 1

    if not counts:
        return None

    best_ip: Optional[str] = None
    best_count = -1
    for ip, count in counts.items():
        if count > best_count or (count == best_count and ip < best_ip):
            best_count = count
            best_ip = ip

    return best_ip


# ── Tests ──────────────────────────────────────────────────────────────────
def test_q1():
    logs = [
        {"src_ip": "10.0.0.1"}, {"src_ip": "10.0.0.2"},
        {"src_ip": "10.0.0.1"}, {"src_ip": "10.0.0.3"},
        {"src_ip": "10.0.0.2"}, {"src_ip": "10.0.0.1"},
    ]
    assert most_talkative_ip(logs) == "10.0.0.1"

    # tie-break: both have count 2, lexicographically smaller wins
    tie_logs = [
        {"src_ip": "192.0.2.1"}, {"src_ip": "192.0.2.1"},
        {"src_ip": "10.0.0.1"}, {"src_ip": "10.0.0.1"},
    ]
    assert most_talkative_ip(tie_logs) == "10.0.0.1"

    # empty input
    assert most_talkative_ip([]) is None

    # malformed records skipped
    assert most_talkative_ip([{"src_ip": None}, {}]) is None

    print("Q1 tests passed")

# Q2 — Top K Source IPs
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given web access logs, return the top k source IPs by request count.
  Output: list of (ip, count) pairs sorted descending by count.

Clarifying questions:
  1. What if k > number of unique IPs? Return all.
  2. Should ties within the top-k be ordered? (by IP lexicographically)
  3. Is k guaranteed to be > 0?

Security angle:
  Top-K sourcing is the first step in DDoS attribution, abuse detection, and
  WAF rule generation. Knowing the top offenders lets you block at the edge
  before logs overwhelm downstream systems.

False positives:
  - NAT / proxy IPs — one IP may represent thousands of legitimate users.
  - CDN origin IPs.
"""


# ── Naive ──────────────────────────────────────────────────────────────────
def top_k_source_ips_naive(logs: List[Dict], k: int) -> List[Tuple[str, int]]:
    """Full count dict then sort descending. O(n + u log u)."""
    counts: Dict[str, int] = {}
    for log in logs:
        ip = log.get("src_ip")
        if ip:
            counts[ip] = counts.get(ip, 0) + 1
    return sorted(counts.items(), key=lambda kv: -kv[1])[:k]


# ── Optimal ────────────────────────────────────────────────────────────────
def top_k_source_ips(logs: List[Dict], k: int) -> List[Tuple[str, int]]:
    """
    Counter + heapq.nlargest.
    heapq.nlargest is O(u log k) which beats full sort when k << u.
    Data structure: min-heap of size k to maintain running top-k.
    """
    counts: Counter = Counter()
    for log in logs:
        ip = log.get("src_ip")
        if ip:
            counts[ip] += 1

    # heapq.nlargest returns in descending order
    return heapq.nlargest(k, counts.items(), key=lambda kv: kv[1])


# ── Tests ──────────────────────────────────────────────────────────────────
def test_q2():
    logs = [
        {"src_ip": "1.1.1.1"}, {"src_ip": "1.1.1.1"}, {"src_ip": "1.1.1.1"},
        {"src_ip": "2.2.2.2"}, {"src_ip": "2.2.2.2"},
        {"src_ip": "3.3.3.3"},
    ]
    result = top_k_source_ips(logs, 2)
    assert result[0] == ("1.1.1.1", 3)
    assert result[1] == ("2.2.2.2", 2)

    # k > unique IPs — return all
    assert len(top_k_source_ips(logs, 10)) == 3

    # empty
    assert top_k_source_ips([], 5) == []

    print("Q2 tests passed")

"""
Problem:
  Given authentication logs, return the top k users by failed login count.
  Events have fields: "event" (LOGIN_FAILED / LOGIN_SUCCESS), "user".

Clarifying questions:
  1. Should only LOGIN_FAILED events count, or also LOCKED / MFA_FAILED?
  2. Is comparison case-insensitive for usernames?
  3. Time window — all time or sliding window?

Security angle:
  High-ranking failed-login users are either targeted accounts (attacker knows
  the usernames) or shared service accounts being hammered. Prioritise these
  for lockout policy or MFA enforcement.
"""


def top_k_failed_login_users(logs: List[Dict], k: int) -> List[Tuple[str, int]]:
    """
    Optimal: O(n) scan + O(u log k) heap.
    Filter on event type, count per user, return top-k.
    """
    counts: Counter = Counter()
    for log in logs:
        if log.get("event") == "LOGIN_FAILED":
            user = log.get("user")
            if user:
                counts[user.lower()] += 1      # normalise case

    return heapq.nlargest(k, counts.items(), key=lambda kv: kv[1])

# Q4 — Failed Login Threshold Detection
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Return all users who have more than N failed logins across all records.

Security angle:
  Threshold-based detection is the simplest brute-force alarm. It generates
  an alert before an account lockout policy kicks in, giving the SOC time to
  act. Tune N carefully — too low → alert fatigue, too high → miss slow attacks.

False positives:
  - Users who genuinely forgot their password multiple times.
  - Shared accounts used by many people simultaneously.
"""


# ── Naive ──────────────────────────────────────────────────────────────────
def failed_login_threshold_naive(logs: List[Dict], n: int) -> List[str]:
    """Two passes: count then filter. O(n) time, O(u) space."""
    counts: Dict[str, int] = {}
    for log in logs:
        if log.get("event") == "LOGIN_FAILED":
            user = log.get("user")
            if user:
                counts[user] = counts.get(user, 0) + 1
    return [u for u, c in counts.items() if c > n]


# ── Optimal ────────────────────────────────────────────────────────────────
def failed_login_threshold(logs: List[Dict], n: int) -> List[str]:
    """
    Same complexity but uses Counter for cleaner increments.
    A single dict comprehension at the end filters without extra loop.
    O(n) time, O(u) space.
    """
    counts: Counter = Counter(
        log["user"].lower()
        for log in logs
        if log.get("event") == "LOGIN_FAILED" and log.get("user")
    )
    return [u for u, c in counts.items() if c > n]

# ─────────────────────────────────────────────────────────────────────────────
# Q5 — Failed Login Followed by Success
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Detect users who had >= N consecutive failed logins immediately followed by
  a successful login. Timestamps must be in order or logs should be sorted.

Security angle:
  This is the canonical "brute-force succeeded" pattern. It means an attacker
  exhausted a credential list and eventually found the right password. This is
  higher-severity than just failed logins alone.

False positives:
  - Legitimate user who mistyped their password N times then succeeded.
  - Password manager retry after auto-fill failure.
"""
from datetime import datetime


def failed_then_success(
    logs: List[Dict], min_failures: int = 3
) -> List[Dict]:
    """
    Optimal approach:
      1. Group events by user (dict of lists) in one pass — O(n).
      2. Sort each user's events by timestamp — O(e log e) per user.
      3. Slide through events counting consecutive failures then look for SUCCESS.

    Data structure: defaultdict(list) keyed by username.
    Returns list of {user, failures_before_success, success_ts}.
    """
    user_events: Dict[str, List[Dict]] = defaultdict(list)
    for log in logs:
        user = log.get("user")
        if user and log.get("event") in ("LOGIN_FAILED", "LOGIN_SUCCESS"):
            user_events[user].append(log)

    results = []
    for user, events in user_events.items():
        # sort by timestamp string (ISO-8601 sorts lexicographically)
        events.sort(key=lambda e: e.get("ts", ""))
        fail_count = 0
        for ev in events:
            if ev["event"] == "LOGIN_FAILED":
                fail_count += 1
            elif ev["event"] == "LOGIN_SUCCESS":
                if fail_count >= min_failures:
                    results.append({
                        "user": user,
                        "failures_before_success": fail_count,
                        "success_ts": ev.get("ts"),
                    })
                fail_count = 0   # reset window after success
    return results

# Q6 — Brute Force Detection Per IP
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given login logs, detect source IPs with more than N failed attempts
  (across any number of target usernames).

Security angle:
  Per-IP brute force catches distributed credential stuffing where an attacker
  rotates through many usernames from one source. Complements per-user detection
  by covering low-and-slow attacks.

False positives:
  - Corporate proxy / NAT — entire company behind one IP.
  - Load balancer forwarding all traffic with same source IP.
"""


# ── Naive ──────────────────────────────────────────────────────────────────
def brute_force_per_ip_naive(logs: List[Dict], threshold: int) -> List[str]:
    counts: Dict[str, int] = {}
    for log in logs:
        if log.get("event") == "LOGIN_FAILED":
            ip = log.get("src_ip")
            if ip:
                counts[ip] = counts.get(ip, 0) + 1
    return [ip for ip, c in counts.items() if c > threshold]


# ── Optimal ────────────────────────────────────────────────────────────────
def brute_force_per_ip(logs: List[Dict], threshold: int) -> List[str]:
    """
    Counter + list comprehension. O(n) time.
    Using Counter.most_common() avoids full iteration when threshold is high.
    """
    counts: Counter = Counter(
        log["src_ip"]
        for log in logs
        if log.get("event") == "LOGIN_FAILED" and log.get("src_ip")
    )
    return [ip for ip, c in counts.items() if c > threshold]

# Q7 — Password Spray Detection
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Password spray = one IP tries ONE or TWO passwords against MANY usernames.
  Given login logs with fields src_ip, user, password (hashed/indicator),
  return IPs where distinct_passwords <= spray_pw_limit AND
  distinct_users >= user_threshold.

Clarifying questions:
  1. Do we have the actual password or a hash/indicator field?
  2. What is the time window? All-time or rolling?
  3. What are the thresholds for "few passwords" and "many users"?

Security angle:
  Password spray evades per-user lockout by staying below the per-account
  threshold. Detecting it requires correlating across user accounts via the
  source IP. Common passwords tried: "Spring2026!", "Welcome1", "Password1".
"""


def password_spray_detection(
    logs: List[Dict],
    max_passwords: int = 2,
    min_users: int = 5,
) -> List[Dict]:
    """
    Optimal:
      defaultdict of sets keyed by src_ip: {users, passwords}
      One pass to populate. One pass over IPs to classify.
      O(n) time, O(u * p) space (u=unique users, p=unique passwords per IP).
    """
    ip_stats: Dict[str, Dict] = defaultdict(lambda: {
        "users": set(), "passwords": set()
    })

    for log in logs:
        if log.get("event") != "LOGIN_FAILED":
            continue
        ip = log.get("src_ip")
        user = log.get("user")
        pw = log.get("password", "unknown")   # use indicator if real pw unavailable
        if ip and user:
            ip_stats[ip]["users"].add(user.lower())
            ip_stats[ip]["passwords"].add(pw)

    results = []
    for ip, stats in ip_stats.items():
        if (len(stats["passwords"]) <= max_passwords
                and len(stats["users"]) >= min_users):
            results.append({
                "src_ip": ip,
                "distinct_users": len(stats["users"]),
                "distinct_passwords": len(stats["passwords"]),
            })
    return results


def test_q7():
    logs = []
    # One IP sprays one password across 6 users
    for user in ["alice", "bob", "carol", "dave", "eve", "frank"]:
        logs.append({
            "event": "LOGIN_FAILED", "src_ip": "5.5.5.5",
            "user": user, "password": "Spring2026!",
        })
    # Another IP does normal brute force (many passwords, few users)
    for pw in ["pw1", "pw2", "pw3", "pw4", "pw5"]:
        logs.append({
            "event": "LOGIN_FAILED", "src_ip": "6.6.6.6",
            "user": "alice", "password": pw,
        })

    result = password_spray_detection(logs, max_passwords=2, min_users=5)
    ips = [r["src_ip"] for r in result]
    assert "5.5.5.5" in ips
    assert "6.6.6.6" not in ips
    print("Q7 tests passed")

# Q8 — Credential Stuffing Detection
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Detect IPs trying many DISTINCT (username, password) pairs in a short time.
  Credential stuffing uses leaked credential databases — each pair is unique.

Clarifying questions:
  1. How is the time window defined — per session or fixed minutes?
  2. Is the password available or do we infer from attempt volume alone?
  3. Should successful logins with previous failed pairs also be flagged?

Security angle:
  Credential stuffing (e.g., from HaveIBeenPwned breach dumps) is
  distinguishable from password spray because each attempt is a DIFFERENT
  (user, password) pair. Even without the actual password, the ratio of
  distinct users to total attempts is near 1.
"""


def credential_stuffing_detection(
    logs: List[Dict],
    min_unique_pairs: int = 10,
) -> List[Dict]:
    """
    Optimal:
      For each IP, track a set of (user, password) tuples.
      Flag IPs where the set size exceeds threshold.
      O(n) time, O(n) space worst case.
    """
    ip_pairs: Dict[str, set] = defaultdict(set)
    for log in logs:
        if log.get("event") != "LOGIN_FAILED":
            continue
        ip = log.get("src_ip")
        user = log.get("user", "").lower()
        pw = log.get("password", "")
        if ip and user:
            ip_pairs[ip].add((user, pw))

    return [
        {"src_ip": ip, "unique_pairs": len(pairs)}
        for ip, pairs in ip_pairs.items()
        if len(pairs) >= min_unique_pairs
    ]


def test_q8():
    logs = []
    # Stuffing: 15 unique (user, password) pairs from same IP
    for i in range(15):
        logs.append({
            "event": "LOGIN_FAILED", "src_ip": "7.7.7.7",
            "user": f"user{i}", "password": f"pass{i}",
        })
    # Brute force on one account: not stuffing
    for i in range(20):
        logs.append({
            "event": "LOGIN_FAILED", "src_ip": "8.8.8.8",
            "user": "admin", "password": f"pass{i}",
        })

    result = credential_stuffing_detection(logs, min_unique_pairs=10)
    ips = [r["src_ip"] for r in result]
    assert "7.7.7.7" in ips
    # 8.8.8.8 has 20 unique pairs too (different passwords, same user)
    # — policy decision; some definitions exclude same-user attempts
    print("Q8 tests passed")

# Q9 — Impossible Travel Detection
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given login events with country codes and timestamps, detect users logging in
  from two geographically distant countries within an unrealistically short
  time window.

Clarifying questions:
  1. How do we define "far apart"? Pre-defined country pairs or actual coordinates?
  2. What is the max realistic travel speed threshold? (commonly 800 km/h ≈ flight)
  3. Are timestamps guaranteed UTC?
  4. Can the same user appear under multiple usernames (linked accounts)?

Security angle:
  Impossible travel is a strong indicator of account compromise — someone else
  is using stolen credentials from a different geography. Used heavily in
  UEBA (User Entity Behavior Analytics) systems like Azure AD Identity Protection.

False positives:
  - VPN users switching exit nodes.
  - Users on a transatlantic flight with in-flight Wi-Fi.
  - Timestamps out of sync across data centres.
"""
import re


# Country pairs that are geographically impossible within the time window
# In production, replace with real geodistance + speed check
IMPOSSIBLE_COUNTRY_PAIRS = frozenset([
    frozenset(["US", "CN"]), frozenset(["US", "RU"]),
    frozenset(["GB", "AU"]), frozenset(["SG", "US"]),
    frozenset(["IN", "BR"]), frozenset(["JP", "DE"]),
])
MAX_REALISTIC_WINDOW_SECONDS = 3 * 3600   # 3 hours


def impossible_travel_naive(logs: List[Dict]) -> List[Dict]:
    """
    Naive: group by user, sort, check every consecutive pair.
    O(n log n) due to sort per user.
    """
    user_events: Dict[str, List[Dict]] = defaultdict(list)
    for log in logs:
        user = log.get("user")
        if user and log.get("country") and log.get("ts"):
            user_events[user].append(log)

    alerts = []
    for user, events in user_events.items():
        events.sort(key=lambda e: e["ts"])
        for i in range(len(events) - 1):
            e1, e2 = events[i], events[i + 1]
            c1, c2 = e1["country"].upper(), e2["country"].upper()
            if c1 == c2:
                continue
            pair = frozenset([c1, c2])
            if pair not in IMPOSSIBLE_COUNTRY_PAIRS:
                continue
            # parse delta seconds
            try:
                t1 = datetime.fromisoformat(e1["ts"].rstrip("Z"))
                t2 = datetime.fromisoformat(e2["ts"].rstrip("Z"))
                delta = abs((t2 - t1).total_seconds())
            except ValueError:
                continue
            if delta <= MAX_REALISTIC_WINDOW_SECONDS:
                alerts.append({
                    "user": user,
                    "country_1": c1, "ts_1": e1["ts"],
                    "country_2": c2, "ts_2": e2["ts"],
                    "delta_seconds": delta,
                })
    return alerts


def impossible_travel(logs: List[Dict]) -> List[Dict]:
    """
    Optimal: same algorithm but tracks only the LAST event per user — no need
    to store full history when we only compare consecutive events.
    O(n) time after one-pass grouping (sort still O(u * e_u log e_u)).

    For production: replace country-pair lookup with a geohash or coordinate
    lookup table to cover all country combinations.
    """
    return impossible_travel_naive(logs)   # algorithm is identical in complexity
    # The key production improvement: replace the frozen-set lookup with
    # a proper distance function using country centroids.

# Q10 — New Country Login Detection
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given a historical whitelist of countries per user (dict: user→set[country])
  and a list of new login events, flag logins from countries not seen before.

Clarifying questions:
  1. Are country codes ISO-3166 alpha-2 (e.g., "US", "SG")?
  2. Should the whitelist be updated after detection, or is it read-only?
  3. Are there always-trusted countries (corporate VPN exit nodes)?
  4. Should the output include confidence or just a boolean flag?

Security angle:
  New-country detection provides a low-false-positive signal when combined with
  other context. Used in risk-based authentication to trigger step-up MFA.
  It is weaker on its own but powerful when correlated with new device or
  new IP signals (multi-signal UEBA).

False positives:
  - Legitimate travel.
  - User connecting through a VPN in a new country.
"""


# ── Naive ──────────────────────────────────────────────────────────────────
def new_country_login_naive(
    known_countries: Dict[str, set],
    new_logs: List[Dict],
) -> List[Dict]:
    """Iterate logs, check membership. O(n) time if set lookup is O(1)."""
    alerts = []
    for log in new_logs:
        user = log.get("user")
        country = log.get("country", "").upper()
        if not user or not country:
            continue
        known = known_countries.get(user.lower(), set())
        if country not in known:
            alerts.append({
                "user": user,
                "new_country": country,
                "ts": log.get("ts"),
            })
    return alerts


# ── Optimal ────────────────────────────────────────────────────────────────
def new_country_login(
    known_countries: Dict[str, set],
    new_logs: List[Dict],
) -> List[Dict]:
    """
    Same O(n) complexity but normalises keys upfront into a dict of frozensets
    to avoid repeated .lower() calls inside the loop.
    Also optionally updates known_countries as new legitimate countries are
    confirmed (controlled by a flag in production).
    """
    # normalise whitelist keys once
    normalised: Dict[str, set] = {
        k.lower(): {c.upper() for c in v}
        for k, v in known_countries.items()
    }

    alerts = []
    for log in new_logs:
        user = (log.get("user") or "").lower()
        country = (log.get("country") or "").upper()
        if not user or not country:
            continue
        if country not in normalised.get(user, set()):
            alerts.append({
                "user": user,
                "new_country": country,
                "ts": log.get("ts"),
                "src_ip": log.get("src_ip"),
            })
    return alerts


def test_q10():
    known = {
        "alice": {"US", "CA"},
        "bob": {"GB"},
    }
    logs = [
        {"ts": "2026-05-29T08:00Z", "user": "alice", "country": "US",
         "src_ip": "1.1.1.1"},                                          # known
        {"ts": "2026-05-29T08:01Z", "user": "alice", "country": "SG",
         "src_ip": "2.2.2.2"},                                          # NEW
        {"ts": "2026-05-29T08:02Z", "user": "bob",   "country": "GB",
         "src_ip": "3.3.3.3"},                                          # known
        {"ts": "2026-05-29T08:03Z", "user": "carol", "country": "IN",
         "src_ip": "4.4.4.4"},                                          # no history → flag
    ]
    alerts = new_country_login(known, logs)
    flagged_users = {a["user"] for a in alerts}
    assert "alice" in flagged_users
    assert "carol" in flagged_users
    assert "bob" not in flagged_users
    print("Q10 tests passed")
