# Q101 — Secure Command Execution Wrapper
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Implement a safe command execution wrapper that:
  1. Only allows a predefined allowlist of commands.
  2. Passes arguments as a list (never shell=True).
  3. Strips or rejects dangerous characters in arguments.
  4. Returns (stdout, stderr, returncode) tuple.

Clarifying questions:
  1. Should the allowlist be a config file or hardcoded?
  2. Should we capture stdout+stderr together or separately?
  3. What is the timeout?

Security angle:
  Shell injection (CWE-78) occurs when user input is interpolated into a shell
  command. The fix is using subprocess with shell=False and an argument list.
  An allowlist of commands prevents arbitrary binary execution even if an
  injection bypass is found. This is the pattern used in hardened CI/CD runners.
"""

import subprocess

_ALLOWED_COMMANDS: frozenset = frozenset({"cmd", "python", "whoami", "hostname"})
_DANGEROUS_ARG_RE = re.compile(r"[;&|`$<>!\\]")


def safe_exec(
    command: str,
    args: List[str],
    allowed_commands: Optional[frozenset] = None,
    timeout: int = 10,
) -> Tuple[str, str, int]:
    """
    Executes command safely with allowlist and argument sanitisation.
    Returns (stdout, stderr, returncode).
    Raises ValueError if command not allowed or args contain dangerous chars.
    """
    if allowed_commands is None:
        allowed_commands = _ALLOWED_COMMANDS

    if command not in allowed_commands:
        raise ValueError(f"Command '{command}' not in allowlist")

    for arg in args:
        if _DANGEROUS_ARG_RE.search(arg):
            raise ValueError(f"Dangerous character in argument: {arg!r}")

    result = subprocess.run(
        [command] + args,
        shell=False,
        capture_output=True,
        text=True,
        timeout=timeout,
    )
    return result.stdout, result.stderr, result.returncode


def test_q101():
    import sys
    # Allowed: run python with --version (works cross-platform)
    stdout, stderr, rc = safe_exec("python", ["--version"])
    assert rc == 0

    # Disallowed command
    try:
        safe_exec("rm", ["/tmp/test"])
        assert False, "Should have raised"
    except ValueError as e:
        assert "allowlist" in str(e)

    # Dangerous argument
    try:
        safe_exec("python", ["hello; rm -rf /"])
        assert False, "Should have raised"
    except ValueError as e:
        assert "Dangerous" in str(e)

    print("Q101 tests passed")

# Q102 — Validate JWT Algorithm
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Implement a JWT header decoder that validates the algorithm field.
  Reject tokens that use "none", symmetric algorithms on asymmetric endpoints,
  or algorithms not in an explicit allowlist.

Clarifying questions:
  1. Are we validating the header only, or the full JWT signature?
  2. What algorithms are acceptable? (default: RS256, ES256, PS256)
  3. Should we flag alg:none with an error or silently reject?

Security angle:
  The CVE-2015-9235 "alg:none" attack allows bypassing JWT signature
  verification by setting the algorithm to "none" — the server then accepts
  any token without verifying. Other attack: downgrade from RS256 to HS256,
  then sign with the server's public key as the HMAC secret.
  The fix: maintain an explicit server-side allowlist of accepted algorithms.
  Never read the algorithm from the token header without validation.
"""

_SAFE_JWT_ALGORITHMS: frozenset = frozenset({"RS256", "RS384", "RS512", "ES256", "ES384", "PS256"})


def decode_jwt_header(token: str) -> Dict:
    """
    Decodes the JWT header (base64url) and returns the parsed dict.
    Raises ValueError if header cannot be decoded.
    """
    try:
        parts = token.split(".")
        if len(parts) != 3:
            raise ValueError("Not a valid JWT (expected 3 parts)")
        header_b64 = parts[0]
        # Add padding
        padding = 4 - len(header_b64) % 4
        header_b64 += "=" * (padding % 4)
        header_json = base64.urlsafe_b64decode(header_b64).decode("utf-8")
        return json.loads(header_json)
    except Exception as exc:
        raise ValueError(f"Failed to decode JWT header: {exc}") from exc


def validate_jwt_algorithm(
    token: str,
    allowed_algorithms: Optional[frozenset] = None,
) -> Dict:
    """
    Returns {"valid": bool, "alg": str, "reason": str}.
    """
    if allowed_algorithms is None:
        allowed_algorithms = _SAFE_JWT_ALGORITHMS

    try:
        header = decode_jwt_header(token)
    except ValueError as e:
        return {"valid": False, "alg": None, "reason": str(e)}

    alg = header.get("alg") or "none"

    if alg.lower() == "none":
        return {"valid": False, "alg": alg, "reason": "Algorithm 'none' rejected"}
    if alg not in allowed_algorithms:
        return {"valid": False, "alg": alg, "reason": f"Algorithm '{alg}' not in allowlist"}

    return {"valid": True, "alg": alg, "reason": "OK"}


def _make_jwt(header_dict: dict) -> str:
    """Helper: create a JWT token string with given header (no real signature)."""
    header_json = json.dumps(header_dict).encode()
    header_b64 = base64.urlsafe_b64encode(header_json).rstrip(b"=").decode()
    return f"{header_b64}.payload.signature"


def test_q102():
    # Valid RS256 token
    tok_rs256 = _make_jwt({"alg": "RS256", "typ": "JWT"})
    r = validate_jwt_algorithm(tok_rs256)
    assert r["valid"] is True

    # alg:none attack
    tok_none = _make_jwt({"alg": "none", "typ": "JWT"})
    r = validate_jwt_algorithm(tok_none)
    assert r["valid"] is False
    assert "none" in r["reason"]

    # HS256 not in allowlist
    tok_hs256 = _make_jwt({"alg": "HS256", "typ": "JWT"})
    r = validate_jwt_algorithm(tok_hs256)
    assert r["valid"] is False

    print("Q102 tests passed")

# Q103 — Validate JWT Expiry and Audience
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Decode and validate the JWT payload (exp, nbf, aud claims) without
  verifying the signature (treating signature verification as a separate step).

Clarifying questions:
  1. Should we enforce strict aud matching (list vs string)?
  2. What is the acceptable clock skew? (default: 30 seconds)
  3. Should we reject JWTs with no exp claim?

Security angle:
  Missing claim validation is a common JWT implementation flaw:
  - No exp check → tokens valid forever after compromise.
  - No aud check → tokens issued for service A accepted by service B.
  These are listed in jwt.io's "Critical vulnerabilities in JWT libraries" (2015).
  Every JWT consumer must validate: exp, nbf, aud, iss (and sub as needed).
"""


def decode_jwt_payload(token: str) -> Dict:
    """Decode the JWT payload claims (middle part)."""
    try:
        parts = token.split(".")
        if len(parts) != 3:
            raise ValueError("Invalid JWT structure")
        payload_b64 = parts[1]
        padding = 4 - len(payload_b64) % 4
        payload_b64 += "=" * (padding % 4)
        payload_json = base64.urlsafe_b64decode(payload_b64).decode("utf-8")
        return json.loads(payload_json)
    except Exception as exc:
        raise ValueError(f"Failed to decode JWT payload: {exc}") from exc


def validate_jwt_claims(
    token: str,
    expected_audience: Optional[str] = None,
    clock_skew_seconds: int = 30,
    require_exp: bool = True,
    now: Optional[float] = None,
) -> Dict:
    """
    Validates exp, nbf, and aud claims.
    Returns {"valid": bool, "reason": str, "claims": dict}.
    """
    now = now if now is not None else time.time()
    try:
        claims = decode_jwt_payload(token)
    except ValueError as e:
        return {"valid": False, "reason": str(e), "claims": {}}

    # exp check
    exp = claims.get("exp")
    if require_exp and exp is None:
        return {"valid": False, "reason": "Missing 'exp' claim", "claims": claims}
    if exp is not None and now > exp + clock_skew_seconds:
        return {"valid": False, "reason": "Token expired", "claims": claims}

    # nbf check
    nbf = claims.get("nbf")
    if nbf is not None and now < nbf - clock_skew_seconds:
        return {"valid": False, "reason": "Token not yet valid (nbf)", "claims": claims}

    # aud check
    if expected_audience is not None:
        aud = claims.get("aud")
        aud_list = [aud] if isinstance(aud, str) else (aud or [])
        if expected_audience not in aud_list:
            return {"valid": False, "reason": f"Audience mismatch: expected {expected_audience!r}", "claims": claims}

    return {"valid": True, "reason": "OK", "claims": claims}


def _make_jwt_with_payload(header: dict, payload: dict) -> str:
    def b64(d: dict) -> str:
        return base64.urlsafe_b64encode(json.dumps(d).encode()).rstrip(b"=").decode()
    return f"{b64(header)}.{b64(payload)}.sig"


def test_q103():
    now = time.time()

    # Valid token
    tok_ok = _make_jwt_with_payload(
        {"alg": "RS256"}, {"sub": "u1", "exp": int(now + 300), "aud": "api.example.com"}
    )
    r = validate_jwt_claims(tok_ok, expected_audience="api.example.com", now=now)
    assert r["valid"] is True

    # Expired
    tok_exp = _make_jwt_with_payload(
        {"alg": "RS256"}, {"sub": "u1", "exp": int(now - 400)}
    )
    r = validate_jwt_claims(tok_exp, now=now)
    assert r["valid"] is False and "expired" in r["reason"].lower()

    # Wrong audience
    tok_aud = _make_jwt_with_payload(
        {"alg": "RS256"}, {"sub": "u1", "exp": int(now + 300), "aud": "other.service"}
    )
    r = validate_jwt_claims(tok_aud, expected_audience="api.example.com", now=now)
    assert r["valid"] is False and "Audience" in r["reason"]

    print("Q103 tests passed")

# Q104 — Detect Hardcoded Secrets in Code
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Scan a code snippet or file for hardcoded secrets:
  - AWS access keys (AKIA...).
  - Password assignments.
  - API keys / tokens in variable names.
  - Private key PEM blocks.

Clarifying questions:
  1. Should we scan line-by-line or as full text?
  2. What is the false positive strategy for test fixtures?
  3. Should we compute entropy of string values to reduce false positives?

Security angle:
  Hardcoded secrets (CWE-798) are the most common cause of credential leakage
  on GitHub. GitHub secret scanning, truffleHog, and Gitleaks all use this
  approach. Detection must balance high sensitivity with low false positive rate.
  Entropy threshold (Shannon > 4.5 bits/char) reduces false positives
  from dictionary-like values.
"""

import math

_SECRET_PATTERNS: List[Tuple[str, re.Pattern]] = [
    ("aws_access_key", re.compile(r"\bAKIA[0-9A-Z]{16}\b")),
    ("aws_secret_key", re.compile(r"""(?:aws_secret|secret_access_key)\s*=\s*["'][A-Za-z0-9/+]{40}["']""", re.IGNORECASE)),
    ("password_assign", re.compile(r"""(?:password|passwd|pwd)\s*=\s*["'][^"']{6,}["']""", re.IGNORECASE)),
    ("api_key_assign",  re.compile(r"""(?:api_key|apikey|token|secret)\s*=\s*["'][A-Za-z0-9._\-]{16,}["']""", re.IGNORECASE)),
    ("private_key_pem", re.compile(r"-----BEGIN (RSA |EC )?PRIVATE KEY-----")),
]


def _shannon_entropy(s: str) -> float:
    if not s:
        return 0.0
    freq = {c: s.count(c) / len(s) for c in set(s)}
    return -sum(p * math.log2(p) for p in freq.values())


def detect_hardcoded_secrets(code: str, min_entropy: float = 3.0) -> List[Dict]:
    findings = []
    for name, pattern in _SECRET_PATTERNS:
        for match in pattern.finditer(code):
            snippet = match.group(0)
            # Extract the value portion for entropy check
            val_match = re.search(r"""["']([^"']+)["']""", snippet)
            value = val_match.group(1) if val_match else snippet
            if _shannon_entropy(value) >= min_entropy:
                findings.append({
                    "type": name,
                    "snippet": snippet[:100],
                    "entropy": round(_shannon_entropy(value), 2),
                })
    return findings


def test_q104():
    code_with_secrets = """
AWS_ACCESS_KEY_ID = "AKIAIOSFODNN7EXAMPLE"
password = "SuperSecret123!"
api_key = "sk-abcdefghijklmnopqrstuvwx12345678"
"""
    code_clean = """
password = os.environ.get("DB_PASSWORD")
api_key = config.get("API_KEY")
"""
    results_bad = detect_hardcoded_secrets(code_with_secrets)
    results_clean = detect_hardcoded_secrets(code_clean)
    assert len(results_bad) >= 2
    assert len(results_clean) == 0
    print("Q104 tests passed")

# Q105 — Build Simple Secret Scanner (File-Level)
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Extend Q104 to scan a list of (filename, content) pairs and return a
  report grouped by file with line numbers.

Clarifying questions:
  1. Should we skip binary files?
  2. Should we support .gitignore-style exclusions?
  3. Should we return only the first match per file or all?

Security angle:
  Secret scanners like truffleHog and Gitleaks scan git history line by line.
  Providing line numbers is critical for remediation. Scanning pre-commit
  (before secrets enter git history) is the best control.
"""


def scan_files_for_secrets(
    files: List[Tuple[str, str]],
    min_entropy: float = 3.0,
) -> Dict[str, List[Dict]]:
    """
    Returns {filename: [finding, ...]} for files with secrets.
    """
    report: Dict[str, List[Dict]] = {}
    for filename, content in files:
        file_findings = []
        for lineno, line in enumerate(content.splitlines(), start=1):
            findings = detect_hardcoded_secrets(line, min_entropy=min_entropy)
            for f in findings:
                file_findings.append({**f, "line": lineno})
        if file_findings:
            report[filename] = file_findings
    return report


def test_q105():
    files = [
        ("config.py", 'api_key = "sk-abcdefghijklmnopqrstu12345"\nDEBUG = True\n'),
        ("settings.py", 'DB_HOST = "localhost"\nDB_PORT = 5432\n'),
    ]
    report = scan_files_for_secrets(files)
    assert "config.py" in report
    assert report["config.py"][0]["line"] == 1
    assert "settings.py" not in report
    print("Q105 tests passed")

# Q106 — Build Allowlist-Based Input Validator
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Build a generic input validator that checks user-supplied values against
  configurable allowlist rules (regex pattern, min/max length, allowed chars).
  Return a validation result with specific error messages.

Clarifying questions:
  1. Should we validate nested JSON objects or flat key-value pairs?
  2. Should we support multiple rules per field?
  3. Should we sanitise (strip) or reject?

Security angle:
  Input validation (OWASP A03) is the first layer of defence against injection.
  Allowlist validation (only permit known-good values) is stronger than
  denylist (block known-bad). This pattern is used in API gateway request
  validation and WAF rule configuration.
"""


def validate_input(
    value: str,
    field_name: str,
    allowed_pattern: Optional[str] = None,
    min_length: int = 0,
    max_length: int = 256,
    allowed_chars: Optional[str] = None,
) -> Dict:
    """
    Returns {"valid": bool, "errors": [str]}.
    """
    errors = []

    if not isinstance(value, str):
        return {"valid": False, "errors": [f"{field_name}: must be a string"]}

    length = len(value)
    if length < min_length:
        errors.append(f"{field_name}: too short (min {min_length})")
    if length > max_length:
        errors.append(f"{field_name}: too long (max {max_length})")

    if allowed_pattern:
        if not re.fullmatch(allowed_pattern, value):
            errors.append(f"{field_name}: does not match required pattern")

    if allowed_chars:
        bad = set(value) - set(allowed_chars)
        if bad:
            errors.append(f"{field_name}: contains disallowed characters: {sorted(bad)}")

    return {"valid": len(errors) == 0, "errors": errors}


def test_q106():
    r = validate_input("alice123", "username", allowed_pattern=r"[a-z0-9]{3,20}", min_length=3, max_length=20)
    assert r["valid"] is True

    r = validate_input("al", "username", allowed_pattern=r"[a-z0-9]{3,20}", min_length=3)
    assert r["valid"] is False
    assert any("short" in e for e in r["errors"])

    r = validate_input("alice'; DROP TABLE users--", "username",
                       allowed_chars="abcdefghijklmnopqrstuvwxyz0123456789")
    assert r["valid"] is False

    print("Q106 tests passed")

# Q107 — CSRF Token Validator
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Implement CSRF token generation and validation using HMAC-SHA256.
  Tokens are bound to the session ID so they can't be reused across sessions.

Clarifying questions:
  1. Should the token be per-request (nonce) or per-session?
  2. How long should tokens be valid? (default: per-session)
  3. Should we use the double-submit cookie pattern or synchroniser token?

Security angle:
  CSRF (OWASP A01) exploits the browser's automatic inclusion of cookies.
  HMAC-based tokens are unforgeable without the server's secret key.
  Binding the token to the session ID prevents an attacker from using their own
  valid token against another user's session (token substitution attack).
"""

_CSRF_SECRET = b"csrf-secret-key-change-in-production"


def generate_csrf_token(session_id: str, secret: bytes = _CSRF_SECRET) -> str:
    """Generate a CSRF token bound to the session ID."""
    mac = hmac.new(secret, session_id.encode(), hashlib.sha256)
    return mac.hexdigest()


def validate_csrf_token(
    session_id: str,
    token: str,
    secret: bytes = _CSRF_SECRET,
) -> bool:
    """Validate CSRF token using constant-time comparison."""
    expected = generate_csrf_token(session_id, secret)
    return hmac.compare_digest(expected, token)


def test_q107():
    sid = "session-abc-123"
    token = generate_csrf_token(sid)
    assert validate_csrf_token(sid, token) is True
    assert validate_csrf_token(sid, "tampered") is False
    assert validate_csrf_token("other-session", token) is False
    print("Q107 tests passed")


# ─────────────────────────────────────────────────────────────────────────────
# Q108 — Secure Cookie Configuration Checker
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given a Set-Cookie header string, check that it has all required security
  attributes: Secure, HttpOnly, SameSite=Strict or Lax, and an appropriate
  Max-Age or Expires.

Clarifying questions:
  1. Should we check all cookies or just session cookies?
  2. Is SameSite=None acceptable if Secure is also set? (yes, for cross-origin)
  3. Should we flag overly long Max-Age values?

Security angle:
  Cookie security attributes prevent a range of attacks:
  - Secure: prevents transmission over plain HTTP (MitM interception).
  - HttpOnly: prevents JavaScript access (XSS cookie theft).
  - SameSite=Strict/Lax: prevents CSRF.
  Missing any of these is a misconfiguration (CWE-614, CWE-1004, CWE-1275).
"""


def check_cookie_security(set_cookie_header: str) -> Dict:
    """
    Returns {"secure": bool, "httponly": bool, "samesite": str|None, "issues": [str]}.
    """
    issues = []
    attrs = [a.strip().lower() for a in set_cookie_header.split(";")]

    secure = "secure" in attrs
    httponly = "httponly" in attrs
    samesite = next((a.split("=")[1].strip() for a in attrs if a.startswith("samesite=")), None)
    has_expiry = any(a.startswith("max-age=") or a.startswith("expires=") for a in attrs)

    if not secure:
        issues.append("Missing Secure flag")
    if not httponly:
        issues.append("Missing HttpOnly flag")
    if samesite is None:
        issues.append("Missing SameSite attribute")
    elif samesite == "none" and not secure:
        issues.append("SameSite=None requires Secure flag")
    if not has_expiry:
        issues.append("Missing Max-Age or Expires (session cookie — consider explicit expiry)")

    return {
        "secure": secure,
        "httponly": httponly,
        "samesite": samesite,
        "has_expiry": has_expiry,
        "issues": issues,
    }


def test_q108():
    good = "sessionid=abc123; Secure; HttpOnly; SameSite=Strict; Max-Age=3600"
    bad  = "sessionid=abc123; Max-Age=3600"

    r_good = check_cookie_security(good)
    assert r_good["issues"] == []

    r_bad = check_cookie_security(bad)
    assert any("Secure" in i for i in r_bad["issues"])
    assert any("HttpOnly" in i for i in r_bad["issues"])
    assert any("SameSite" in i for i in r_bad["issues"])
    print("Q108 tests passed")


# ─────────────────────────────────────────────────────────────────────────────
# Q109 — Detect Missing Security Headers
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given an HTTP response headers dict, check for the presence and correctness
  of required security headers:
  - Strict-Transport-Security (HSTS)
  - X-Content-Type-Options: nosniff
  - X-Frame-Options: DENY or SAMEORIGIN
  - Content-Security-Policy (any non-empty value)
  - Referrer-Policy

Clarifying questions:
  1. Should we validate header values or just presence?
  2. Should we flag deprecated headers (X-XSS-Protection)?
  3. Is Permissions-Policy required?

Security angle:
  Missing security headers (OWASP A05 misconfiguration) are flagged by every
  security scanner (e.g., SecurityHeaders.com). HSTS prevents SSL stripping,
  X-Content-Type-Options prevents MIME sniffing attacks, CSP reduces XSS impact,
  X-Frame-Options prevents clickjacking.
"""

_REQUIRED_HEADERS: Dict[str, Optional[re.Pattern]] = {
    "strict-transport-security": re.compile(r"max-age=\d+", re.IGNORECASE),
    "x-content-type-options": re.compile(r"nosniff", re.IGNORECASE),
    "x-frame-options": re.compile(r"(DENY|SAMEORIGIN)", re.IGNORECASE),
    "content-security-policy": re.compile(r".+"),
    "referrer-policy": re.compile(r".+"),
}


def check_security_headers(headers: Dict[str, str]) -> Dict:
    """
    Returns {"missing": [str], "misconfigured": [str], "ok": [str]}.
    Headers dict keys are compared case-insensitively.
    """
    normalised = {k.lower(): v for k, v in headers.items()}
    missing, misconfigured, ok = [], [], []

    for header, pattern in _REQUIRED_HEADERS.items():
        value = normalised.get(header)
        if value is None:
            missing.append(header)
        elif pattern and not pattern.search(value):
            misconfigured.append(f"{header}: {value!r}")
        else:
            ok.append(header)

    return {"missing": missing, "misconfigured": misconfigured, "ok": ok}


def test_q109():
    good_headers = {
        "Strict-Transport-Security": "max-age=31536000; includeSubDomains",
        "X-Content-Type-Options": "nosniff",
        "X-Frame-Options": "DENY",
        "Content-Security-Policy": "default-src 'self'",
        "Referrer-Policy": "strict-origin-when-cross-origin",
    }
    bad_headers = {
        "X-Content-Type-Options": "nosniff",
        "X-Frame-Options": "ALLOWALL",   # bad value
    }

    r_good = check_security_headers(good_headers)
    assert r_good["missing"] == []
    assert r_good["misconfigured"] == []

    r_bad = check_security_headers(bad_headers)
    assert "strict-transport-security" in r_bad["missing"]
    assert any("x-frame-options" in m for m in r_bad["misconfigured"])
    print("Q109 tests passed")


# ─────────────────────────────────────────────────────────────────────────────
# Q110 — Validate CORS Configuration
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Validate an Access-Control-Allow-Origin header value against a server-side
  allowlist of trusted origins. Return whether the CORS response is safe.
  Flag: wildcard + credentials, or reflective origin without validation.

Clarifying questions:
  1. Is the server reflecting the request Origin header back? (dangerous)
  2. Does the response also include Access-Control-Allow-Credentials: true?
  3. Are subdomains included in the allowlist?

Security angle:
  CORS misconfiguration (PortSwigger Web Security Academy top topic):
  1. Wildcard (*) with credentials is forbidden by spec but some libraries
     bypass it — credential leakage.
  2. Reflective origin (echo request Origin unconditionally) allows any site
     to make credentialed cross-origin requests.
  3. Null origin allowed (for iframe/file:// attacks).
  The fix: a server-side allowlist, never reflect without validation.
"""

_NULL_ORIGIN = "null"


def validate_cors_config(
    acao_header: str,
    acac_header: Optional[str],   # Access-Control-Allow-Credentials
    request_origin: Optional[str],
    allowed_origins: Optional[Set[str]] = None,
) -> Dict:
    """
    Returns {"safe": bool, "issues": [str]}.
    acao_header: the ACAO header value the server sent.
    acac_header: the ACAC header value (e.g., "true").
    request_origin: the Origin the client sent.
    allowed_origins: server-side allowlist.
    """
    issues = []
    credentials = (acac_header or "").lower() == "true"

    if acao_header == "*":
        if credentials:
            issues.append("Wildcard ACAO with credentials=true is forbidden")
        # else wildcard without credentials is acceptable for public APIs

    elif acao_header == _NULL_ORIGIN:
        issues.append("ACAO: null is dangerous (allows iframe/file:// attacks)")

    else:
        # Check reflective origin
        if request_origin and acao_header == request_origin:
            if allowed_origins and request_origin not in allowed_origins:
                issues.append(
                    f"Origin '{request_origin}' reflected but not in allowlist (reflective CORS)"
                )

        # Check against allowlist
        if allowed_origins and acao_header not in allowed_origins:
            issues.append(f"ACAO value '{acao_header}' not in allowed origins allowlist")

    return {"safe": len(issues) == 0, "issues": issues}


def test_q110():
    allowed = {"https://example.com", "https://app.example.com"}

    # Safe: allowlisted origin, no credentials
    r = validate_cors_config("https://example.com", None, "https://example.com", allowed)
    assert r["safe"] is True

    # Unsafe: wildcard + credentials
    r = validate_cors_config("*", "true", "https://evil.com", allowed)
    assert r["safe"] is False
    assert any("Wildcard" in i for i in r["issues"])

    # Unsafe: null origin
    r = validate_cors_config("null", "true", None, allowed)
    assert r["safe"] is False

    # Unsafe: reflected origin not in allowlist
    r = validate_cors_config("https://evil.com", "true", "https://evil.com", allowed)
    assert r["safe"] is False

    print("Q110 tests passed")
