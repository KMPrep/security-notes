# Q111 — Brute Force Caesar Cipher
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given a Caesar-ciphered ciphertext (letters shifted by a fixed key 0-25),
  find all possible plaintext candidates by brute-forcing all 26 shifts.
  Optionally rank by English character frequency to find the most likely key.

Clarifying questions:
  1. Are non-letter characters preserved?
  2. Is it uppercase-only, lowercase-only, or mixed?
  3. Should we return all 26 candidates or just the best?

Security angle:
  Caesar cipher has zero computational security — key space of 26.
  Understanding brute-force attack complexity is fundamental to cryptanalysis
  and helps security engineers evaluate why modern ciphers need large key spaces.
  This maps to CWE-327 (use of broken/risky algorithm).
"""

_ENGLISH_FREQ = "etaoinshrdlucmfywgpbvkjxqz"


def caesar_encrypt(plaintext: str, shift: int) -> str:
    result = []
    for ch in plaintext:
        if ch.isalpha():
            base = ord('A') if ch.isupper() else ord('a')
            result.append(chr((ord(ch) - base + shift) % 26 + base))
        else:
            result.append(ch)
    return "".join(result)


def caesar_decrypt(ciphertext: str, shift: int) -> str:
    return caesar_encrypt(ciphertext, -shift)


def brute_force_caesar(ciphertext: str) -> List[Tuple[int, float, str]]:
    """
    Returns list of (shift, score, plaintext) for all 26 shifts, sorted by
    decreasing score (higher = more English-like).
    """
    results = []
    for shift in range(26):
        plaintext = caesar_decrypt(ciphertext, shift)
        # Score: count characters that appear in top-12 English letters
        top12 = frozenset(_ENGLISH_FREQ[:12])
        score = sum(1 for c in plaintext.lower() if c in top12) / max(len(plaintext), 1)
        results.append((shift, score, plaintext))
    results.sort(key=lambda x: -x[1])
    return results


def test_q111():
    original = (
        "The quick brown fox jumps over the lazy dog and then runs "
        "away into the night"
    )
    shift = 13
    ciphertext = caesar_encrypt(original, shift)
    candidates = brute_force_caesar(ciphertext)
    assert candidates[0][0] == shift, (
        f"Expected top shift {shift}, got {candidates[0][0]}"
    )
    assert candidates[0][2] == original
    print("Q111 tests passed")

# Q112 — Repeating-Key XOR Encrypt/Decrypt
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Implement repeating-key XOR encryption / decryption.
  XOR plaintext bytes with repeated key bytes cyclically.

Clarifying questions:
  1. Are the inputs bytes or strings?
  2. Should the output be hex-encoded or raw bytes?
  3. Is this symmetric (encrypt == decrypt)?

Security angle:
  Repeating-key XOR (Vigenère cipher variant) is broken by Kasiski analysis
  and the index of coincidence method. It was used in classic ciphers and
  still appears in CTF challenges and malware obfuscation (XOR with 1-byte key).
  Understanding its weakness motivates why block ciphers with proper key
  schedules are required.
"""


def xor_encrypt(plaintext: bytes, key: bytes) -> bytes:
    if not key:
        raise ValueError("Key must not be empty")
    return bytes(p ^ key[i % len(key)] for i, p in enumerate(plaintext))


def xor_decrypt(ciphertext: bytes, key: bytes) -> bytes:
    return xor_encrypt(ciphertext, key)   # XOR is its own inverse


def test_q112():
    plaintext = b"Attack at dawn"
    key = b"ICE"
    ciphertext = xor_encrypt(plaintext, key)
    assert ciphertext != plaintext
    recovered = xor_decrypt(ciphertext, key)
    assert recovered == plaintext
    print("Q112 tests passed")

# Q113 — Detect Single-Byte XOR Cipher
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given a hex-encoded ciphertext that was XOR-encrypted with a single byte,
  find the key byte and recover the plaintext using frequency analysis.

Clarifying questions:
  1. Is the ciphertext hex-encoded or raw bytes?
  2. Should we return the single best key or all candidates?
  3. Should the score be character frequency or chi-squared?

Security angle:
  Single-byte XOR is trivially broken (only 256 possible keys). It is still
  used in malware to obfuscate strings, configuration, and network payloads.
  Reversing this is a core malware analysis skill. The same frequency
  analysis technique is used against all substitution ciphers.
"""


def _score_english(data: bytes) -> float:
    """Score bytes for English-likeness using character frequency."""
    freq = {c: data.count(c) for c in range(256)}
    target = {ord(c): f for c, f in zip(_ENGLISH_FREQ, range(26, 0, -1))}
    score = sum(freq.get(b, 0) * target.get(b, 0) for b in range(256))
    score += sum(freq.get(b, 0) * 5 for b in b" ")  # boost spaces
    return score


def break_single_byte_xor(hex_ciphertext: str) -> Tuple[int, str]:
    """
    Returns (key_byte, plaintext) for the best single-byte XOR key.
    """
    ciphertext = bytes.fromhex(hex_ciphertext)
    best_score = -1
    best_key = 0
    best_plain = b""

    for key_byte in range(256):
        candidate = bytes(b ^ key_byte for b in ciphertext)
        score = _score_english(candidate)
        if score > best_score:
            best_score = score
            best_key = key_byte
            best_plain = candidate

    return best_key, best_plain.decode("latin-1")


def test_q113():
    original = "Cooking MC's like a pound of bacon"
    key_byte = 88  # 'X'
    ciphertext_hex = bytes(b ^ key_byte for b in original.encode("ascii")).hex()
    found_key, plaintext = break_single_byte_xor(ciphertext_hex)
    assert found_key == key_byte
    assert plaintext == original
    print("Q113 tests passed")

# Q114 — HMAC Verification
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Implement HMAC-SHA256 generation and verification for a message.
  Demonstrate constant-time comparison to prevent timing attacks.

Clarifying questions:
  1. Should the HMAC be over raw bytes or a UTF-8 encoded string?
  2. Should the output be hex or base64?
  3. Should we support multiple algorithms?

Security angle:
  HMAC (Hash-based Message Authentication Code) provides both integrity and
  authenticity. Timing attacks against naive == comparison (where short-circuit
  evaluation leaks info) are prevented by hmac.compare_digest() which always
  takes constant time regardless of where the mismatch occurs.
  This is used in JWT signature verification, webhook signature verification
  (GitHub, Stripe), and API request signing.
"""


def generate_hmac(message: bytes, key: bytes, algorithm: str = "sha256") -> str:
    """Returns hex-encoded HMAC."""
    mac = hmac.new(key, message, algorithm)
    return mac.hexdigest()


def verify_hmac(message: bytes, key: bytes, expected_hex: str, algorithm: str = "sha256") -> bool:
    """Constant-time HMAC verification."""
    actual = generate_hmac(message, key, algorithm)
    return hmac.compare_digest(actual, expected_hex)


def test_q114():
    key = b"super-secret-key"
    msg = b"Hello, world!"
    mac = generate_hmac(msg, key)
    assert verify_hmac(msg, key, mac) is True
    assert verify_hmac(msg, key, "tampered" + mac[8:]) is False
    assert verify_hmac(b"other message", key, mac) is False
    print("Q114 tests passed")

# Q115 — Password Hash Migration
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given a list of stored password hashes in weak format (MD5), implement
  a migration function that:
  1. On next login, verifies the old hash.
  2. Re-hashes the password with PBKDF2-SHA256 (strong KDF).
  3. Updates the stored record.

Clarifying questions:
  1. Should the migration be transparent (no user re-enrollment required)?
  2. What is the PBKDF2 iteration count? (NIST 2023 recommendation: ≥600,000)
  3. Should we store a "hash_version" field?

Security angle:
  Hash migration is required when upgrading from legacy systems using MD5/SHA1
  to modern KDFs. The on-login migration pattern avoids user disruption:
  users with the old hash scheme get upgraded transparently on first login
  post-migration. LinkedIn (2012 breach) and Adobe (2013) both suffered due to
  weak password hashing. NIST SP 800-63B mandates memory-hard KDFs.
"""

_PBKDF2_ITERATIONS = 260_000


def md5_hash(password: str) -> str:
    return hashlib.md5(password.encode()).hexdigest()


def pbkdf2_hash(password: str, salt: Optional[bytes] = None) -> str:
    """Returns 'pbkdf2:<iterations>:<salt_hex>:<hash_hex>'."""
    if salt is None:
        salt = os.urandom(16)
    dk = hashlib.pbkdf2_hmac("sha256", password.encode(), salt, _PBKDF2_ITERATIONS)
    return f"pbkdf2:{_PBKDF2_ITERATIONS}:{salt.hex()}:{dk.hex()}"


def verify_pbkdf2(password: str, stored: str) -> bool:
    parts = stored.split(":")
    if len(parts) != 4 or parts[0] != "pbkdf2":
        return False
    _, iterations, salt_hex, expected_hex = parts
    salt = bytes.fromhex(salt_hex)
    dk = hashlib.pbkdf2_hmac("sha256", password.encode(), salt, int(iterations))
    return hmac.compare_digest(dk.hex(), expected_hex)


def migrate_password_on_login(
    user_record: Dict,
    provided_password: str,
) -> Tuple[bool, Dict]:
    """
    Returns (login_success, updated_record).
    If hash_version == 'md5', verify and upgrade to pbkdf2.
    """
    stored = user_record.get("password_hash", "")
    version = user_record.get("hash_version", "md5")

    if version == "md5":
        if not hmac.compare_digest(md5_hash(provided_password), stored):
            return False, user_record
        # Upgrade hash
        new_hash = pbkdf2_hash(provided_password)
        updated = {**user_record, "password_hash": new_hash, "hash_version": "pbkdf2"}
        return True, updated

    if version == "pbkdf2":
        return verify_pbkdf2(provided_password, stored), user_record

    return False, user_record


def test_q115():
    password = "correct-horse-battery"
    record = {"user": "alice", "password_hash": md5_hash(password), "hash_version": "md5"}

    # Correct password → migrates
    success, updated = migrate_password_on_login(record, password)
    assert success is True
    assert updated["hash_version"] == "pbkdf2"
    assert "pbkdf2:" in updated["password_hash"]

    # Wrong password
    success, _ = migrate_password_on_login(record, "wrong-password")
    assert success is False

    # Second login with upgraded record
    success, _ = migrate_password_on_login(updated, password)
    assert success is True
    print("Q115 tests passed")

# Q116 — Detect Reused Password Hashes
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given a user database with password hashes, detect accounts sharing the same
  hash (password reuse / credential stuffing indicator).

Clarifying questions:
  1. Should we check all hashes or only specific hash schemes?
  2. Should we also flag empty hashes?
  3. Should the output include usernames or just count?

Security angle:
  Identical hashes mean identical passwords. If one account is compromised,
  all accounts with the same hash are immediately accessible (credential
  stuffing, password spraying). For strong KDFs with per-user salts (bcrypt,
  PBKDF2), identical hashes should never occur — their presence suggests
  missing salts or use of a weak unsalted algorithm.
"""

from collections import defaultdict


def detect_reused_hashes(
    users: List[Dict],
    hash_field: str = "password_hash",
    min_count: int = 2,
) -> List[Dict]:
    """
    Returns list of {hash, user_count, users} for hashes shared by ≥ min_count users.
    O(n) time.
    """
    hash_to_users: Dict[str, List[str]] = defaultdict(list)
    for user in users:
        h = user.get(hash_field) or ""
        username = user.get("username") or user.get("user") or "unknown"
        if h:
            hash_to_users[h].append(username)

    return [
        {"hash": h[:16] + "...", "user_count": len(ul), "users": ul}
        for h, ul in hash_to_users.items()
        if len(ul) >= min_count
    ]


def test_q116():
    common_hash = md5_hash("Password123")
    users = [
        {"username": "alice", "password_hash": common_hash},
        {"username": "bob",   "password_hash": common_hash},
        {"username": "carol", "password_hash": md5_hash("unique-password-carol!")},
    ]
    results = detect_reused_hashes(users)
    assert len(results) == 1
    assert set(results[0]["users"]) == {"alice", "bob"}
    print("Q116 tests passed")

# Q117 — Detect Weak Passwords Against Dictionary
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given a list of plaintext passwords (or password candidates), flag passwords
  that appear in a dictionary (rockyou-style list) or fail complexity rules:
  - Min length < 12
  - No uppercase
  - No digit
  - No special character

Clarifying questions:
  1. Should we check against a hash of the dictionary (for secure comparison)?
  2. Is the dictionary provided as a file path or in-memory list?
  3. Should we score password strength (zxcvbn-style) or just pass/fail?

Security angle:
  NIST SP 800-63B bans passwords that appear in breach corpuses.
  Password complexity rules reduce guessability, but NIST now recommends
  length over complexity (a 16-char passphrase is better than P@ssw0rd1!).
  This function implements both dictionary check and complexity scoring.
"""

_COMPLEXITY_CHECKS = [
    (lambda p: len(p) >= 12,            "min_length_12"),
    (lambda p: any(c.isupper() for c in p), "has_uppercase"),
    (lambda p: any(c.isdigit() for c in p), "has_digit"),
    (lambda p: any(not c.isalnum() for c in p), "has_special"),
]


def check_password_strength(
    password: str,
    dictionary: Optional[frozenset] = None,
) -> Dict:
    """
    Returns {"strong": bool, "failed_checks": [str], "in_dictionary": bool}.
    """
    in_dict = password.lower() in (dictionary or frozenset())
    failed = [name for fn, name in _COMPLEXITY_CHECKS if not fn(password)]
    strong = not in_dict and not failed
    return {"strong": strong, "failed_checks": failed, "in_dictionary": in_dict}


def test_q117():
    dictionary = frozenset({"password", "123456", "qwerty", "letmein"})

    r = check_password_strength("password", dictionary)
    assert r["strong"] is False
    assert r["in_dictionary"] is True

    r = check_password_strength("abc", dictionary)
    assert r["strong"] is False
    assert "min_length_12" in r["failed_checks"]

    r = check_password_strength("Tr0ub4dor&3xAmple!", dictionary)
    assert r["strong"] is True
    print("Q117 tests passed")

# Q118 — Generate Secure Random Tokens
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Implement a secure token generator that:
  1. Uses os.urandom (CSPRNG) via the secrets module.
  2. Supports URL-safe base64 and hex formats.
  3. Returns tokens of configurable entropy (default: 256 bits).

Clarifying questions:
  1. Should the token include a version prefix for rotation tracking?
  2. Should the token length be fixed or configurable?
  3. Should we add a HMAC for token integrity verification?

Security angle:
  Token generation is the foundation of session management, password reset,
  API keys, and CSRF tokens. Using the secrets module (Python 3.6+) ensures:
  - os.urandom as the entropy source (CSPRNG).
  - No accidental seeding from time-based sources.
  A 256-bit token has 2^256 possible values — computationally infeasible to
  brute force even with a billion guesses per second for the lifetime of the universe.
"""


def generate_token_hex(bits: int = 256) -> str:
    """Generate a cryptographically secure token as hex string."""
    num_bytes = bits // 8
    return secrets.token_hex(num_bytes)


def generate_token_urlsafe(bits: int = 256) -> str:
    """Generate a cryptographically secure URL-safe base64 token."""
    num_bytes = bits // 8
    return secrets.token_urlsafe(num_bytes)


def token_entropy_bits(token: str, encoding: str = "hex") -> float:
    """Estimate entropy bits of a token based on its encoding and length."""
    if encoding == "hex":
        return len(token) * 4  # 4 bits per hex char
    if encoding == "urlsafe_b64":
        return len(token) * 6  # ~6 bits per base64 char
    return len(token) * 8


def test_q118():
    tok_hex = generate_token_hex(256)
    assert len(tok_hex) == 64   # 256 bits = 32 bytes = 64 hex chars
    assert all(c in "0123456789abcdef" for c in tok_hex)

    tok_b64 = generate_token_urlsafe(256)
    assert len(tok_b64) >= 40   # base64url is slightly shorter than hex

    # Two tokens must not be equal
    assert generate_token_hex() != generate_token_hex()

    entropy = token_entropy_bits(tok_hex, "hex")
    assert entropy >= 256
    print("Q118 tests passed")

# Q119 — Validate Certificate Expiry Chain
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given a list of PEM certificate strings (representing a chain), parse the
  Not After date from each and flag any that are expired or expiring within
  N days (default: 30 days).

Clarifying questions:
  1. Should we validate the chain order (leaf → intermediate → root)?
  2. Should we use the cryptography library or stdlib only?
  3. Should we check revocation (OCSP/CRL)?

Security angle:
  Expired certificates cause service outages and TLS connection failures.
  High-profile examples: Let's Encrypt root expiry (Sep 2021), Ericsson PKI
  outage (Dec 2021 — network outage across 11 countries due to expired cert).
  Certificate lifecycle management is a core security operations task.
  We use a regex-based parser for stdlib-only implementation.
"""

# Parse the Not After field from PEM-like cert text (simplified: DER not parsed,
# we simulate with structured text containing the date)
_NOT_AFTER_RE = re.compile(r"Not After\s*:\s*(.+)", re.IGNORECASE)
_DATE_FORMATS = ["%b %d %H:%M:%S %Y %Z", "%Y-%m-%dT%H:%M:%SZ", "%Y-%m-%d"]


def parse_not_after(cert_text: str) -> Optional[datetime.datetime]:
    """
    Parse the Not After field from a certificate text representation.
    Supports openssl-style output: 'Not After : Dec  7 00:00:00 2024 GMT'
    """
    m = _NOT_AFTER_RE.search(cert_text)
    if not m:
        return None
    date_str = m.group(1).strip()
    for fmt in _DATE_FORMATS:
        try:
            return datetime.datetime.strptime(date_str, fmt).replace(
                tzinfo=datetime.timezone.utc
            )
        except ValueError:
            continue
    return None


def validate_cert_chain_expiry(
    cert_texts: List[str],
    warn_days: int = 30,
    now: Optional[datetime.datetime] = None,
) -> List[Dict]:
    """
    Returns list of issues for certs that are expired or expiring within warn_days.
    """
    if now is None:
        now = datetime.datetime.now(tz=datetime.timezone.utc)
    issues = []
    for i, cert_text in enumerate(cert_texts):
        not_after = parse_not_after(cert_text)
        if not_after is None:
            issues.append({"index": i, "status": "parse_error", "not_after": None})
            continue
        days_left = (not_after - now).days
        if days_left < 0:
            issues.append({"index": i, "status": "expired", "not_after": not_after.isoformat(), "days_left": days_left})
        elif days_left < warn_days:
            issues.append({"index": i, "status": "expiring_soon", "not_after": not_after.isoformat(), "days_left": days_left})
    return issues


def test_q119():
    now = datetime.datetime(2026, 5, 29, 0, 0, 0, tzinfo=datetime.timezone.utc)
    certs = [
        "Not After : May 28 00:00:00 2026 GMT",   # expired (1 day ago)
        "Not After : Jun 10 00:00:00 2026 GMT",   # expiring in 12 days
        "Not After : Dec 31 00:00:00 2027 GMT",   # valid
    ]
    issues = validate_cert_chain_expiry(certs, warn_days=30, now=now)
    statuses = {iss["index"]: iss["status"] for iss in issues}
    assert statuses[0] == "expired"
    assert statuses[1] == "expiring_soon"
    assert 2 not in statuses
    print("Q119 tests passed")

# Q120 — Parse PEM Blocks
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Parse a string containing multiple PEM-encoded blocks and return each block
  as a dict with its type (CERTIFICATE, PRIVATE KEY, PUBLIC KEY, etc.)
  and base64 content.

Clarifying questions:
  1. Should we decode the DER content?
  2. Should we validate the base64 padding?
  3. Should we flag dangerous combinations (private key + certificate in same file)?

Security angle:
  PEM files are used for TLS certificates, private keys, and CRLs. Mishandling
  PEM files can lead to:
  - Private key exposure (logging/serving PEM files over HTTP).
  - Loading wrong cert type (confusing certificate chain order).
  - Mixing private keys with public certs (accidentally committing private key).
  Parsing PEM blocks correctly is a prerequisite for any PKI tooling.
"""

_PEM_RE = re.compile(
    r"-----BEGIN ([A-Z0-9 ]+)-----\s*([\s\S]*?)\s*-----END \1-----",
    re.MULTILINE,
)


def parse_pem_blocks(pem_string: str) -> List[Dict]:
    """
    Returns list of {type: str, raw_b64: str, der_bytes: bytes}.
    Raises ValueError for invalid base64.
    """
    blocks = []
    for match in _PEM_RE.finditer(pem_string):
        block_type = match.group(1).strip()
        b64_content = re.sub(r"\s+", "", match.group(2))
        try:
            # Add padding if needed
            padding = 4 - len(b64_content) % 4
            der_bytes = base64.b64decode(b64_content + "=" * (padding % 4))
        except Exception as exc:
            raise ValueError(f"Invalid base64 in PEM block '{block_type}': {exc}") from exc
        blocks.append({
            "type": block_type,
            "raw_b64": b64_content[:32] + "...",
            "der_length": len(der_bytes),
        })
    return blocks


def check_pem_security(pem_string: str) -> Dict:
    """
    Flag if PEM contains private keys mixed with certificates.
    """
    blocks = parse_pem_blocks(pem_string)
    types = {b["type"] for b in blocks}
    issues = []
    private_key_types = types & {"RSA PRIVATE KEY", "EC PRIVATE KEY", "PRIVATE KEY"}
    if private_key_types and "CERTIFICATE" in types:
        issues.append("Private key and certificate in the same file — risk of accidental exposure")
    return {"blocks": blocks, "types": sorted(types), "issues": issues}


def test_q120():
    # Minimal valid PEM (a real base64 block would be DER-encoded — use dummy content here)
    dummy_cert_der = base64.b64encode(b"\x30" * 64).decode()
    pem = f"-----BEGIN CERTIFICATE-----\n{dummy_cert_der}\n-----END CERTIFICATE-----\n"

    blocks = parse_pem_blocks(pem)
    assert len(blocks) == 1
    assert blocks[0]["type"] == "CERTIFICATE"
    assert blocks[0]["der_length"] == 64

    # Two blocks
    dummy_key_der = base64.b64encode(b"\x02" * 32).decode()
    pem_two = (
        f"-----BEGIN CERTIFICATE-----\n{dummy_cert_der}\n-----END CERTIFICATE-----\n"
        f"-----BEGIN RSA PRIVATE KEY-----\n{dummy_key_der}\n-----END RSA PRIVATE KEY-----\n"
    )
    result = check_pem_security(pem_two)
    assert len(result["blocks"]) == 2
    assert any("Private key" in i for i in result["issues"])
    print("Q120 tests passed")
