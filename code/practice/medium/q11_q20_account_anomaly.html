# Q11 — New Device Login Detection
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given a historical whitelist of device fingerprints per user and new login
  events, flag logins from devices not previously seen.

Clarifying questions:
  1. What constitutes a "device fingerprint"? (user-agent, device-id, hardware hash?)
  2. Is the whitelist updated after confirmed legitimate logins?
  3. Should the result include confidence score or just boolean?
  4. Should we correlate with IP or country for compound alerts?

Security angle:
  New-device logins combined with new-country or impossible travel signals
  produce high-confidence compromise alerts with minimal false positives.
  Device fingerprinting is a core control in risk-based MFA systems.

False positives:
  - Browser update changes user-agent string.
  - Incognito/private browsing mode.
  - User bought a new device.
"""


# ── Naive ──────────────────────────────────────────────────────────────────
def new_device_login_naive(
    known_devices: Dict[str, set],
    new_logs: List[Dict],
) -> List[Dict]:
    alerts = []
    for log in new_logs:
        user = log.get("user")
        device = log.get("device")
        if not user or not device:
            continue
        if device not in known_devices.get(user.lower(), set()):
            alerts.append({"user": user, "device": device, "ts": log.get("ts")})
    return alerts


# ── Optimal ────────────────────────────────────────────────────────────────
def new_device_login(
    known_devices: Dict[str, set],
    new_logs: List[Dict],
) -> List[Dict]:
    """
    Normalise the whitelist to lowercase keys once. O(n) scan.
    Data structure: dict of sets for O(1) membership test.
    """
    normalised = {k.lower(): v for k, v in known_devices.items()}
    alerts = []
    for log in new_logs:
        user = (log.get("user") or "").lower()
        device = log.get("device") or ""
        if not user or not device:
            continue
        if device not in normalised.get(user, set()):
            alerts.append({
                "user": user,
                "new_device": device,
                "ts": log.get("ts"),
                "src_ip": log.get("src_ip"),
            })
    return alerts


def test_q11():
    known = {"alice": {"chrome-win", "ios"}, "bob": {"firefox-mac"}}
    logs = [
        {"ts": "T1", "user": "alice", "device": "chrome-win", "src_ip": "1.1.1.1"},
        {"ts": "T2", "user": "alice", "device": "android",    "src_ip": "2.2.2.2"},
        {"ts": "T3", "user": "bob",   "device": "firefox-mac","src_ip": "3.3.3.3"},
    ]
    alerts = new_device_login(known, logs)
    assert len(alerts) == 1
    assert alerts[0]["new_device"] == "android"
    print("Q11 tests passed")

# Q12 — Dormant Account Activity
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given last-login dates per user and a set of new login events, flag accounts
  that have been inactive for more than INACTIVITY_DAYS days.

Security angle:
  Dormant accounts are prime targets for attackers because:
  - Owners are unlikely to notice unauthorised access.
  - Accounts may retain elevated permissions from former roles.
  - Password reset emails go unread.
  Standard control: disable accounts inactive > 90 days; alert on login.
"""  # noqa: W605
INACTIVITY_DAYS = 90


# ── Naive ──────────────────────────────────────────────────────────────────
def dormant_account_activity_naive(
    last_logins: Dict[str, str],   # user → ISO timestamp of last login
    new_logs: List[Dict],
    inactivity_days: int = INACTIVITY_DAYS,
) -> List[Dict]:
    alerts = []
    for log in new_logs:
        user = (log.get("user") or "").lower()
        last_ts = last_logins.get(user)
        if not last_ts:
            continue
        try:
            last_dt = datetime.fromisoformat(last_ts.rstrip("Z"))
            new_dt  = datetime.fromisoformat((log.get("ts") or "").rstrip("Z"))
        except ValueError:
            continue
        if (new_dt - last_dt).days > inactivity_days:
            alerts.append({
                "user": user,
                "last_login": last_ts,
                "current_login": log.get("ts"),
                "days_inactive": (new_dt - last_dt).days,
            })
    return alerts


# ── Optimal ────────────────────────────────────────────────────────────────
def dormant_account_activity(
    last_logins: Dict[str, str],
    new_logs: List[Dict],
    inactivity_days: int = INACTIVITY_DAYS,
) -> List[Dict]:
    """
    Pre-parse last_login timestamps once into a dict of datetime objects.
    Avoids re-parsing inside the loop.
    O(u) pre-process + O(n) scan.
    """
    parsed_last: Dict[str, datetime] = {}
    for user, ts in last_logins.items():
        try:
            parsed_last[user.lower()] = datetime.fromisoformat(ts.rstrip("Z"))
        except ValueError:
            pass

    alerts = []
    for log in new_logs:
        user = (log.get("user") or "").lower()
        last_dt = parsed_last.get(user)
        if not last_dt:
            continue
        try:
            new_dt = datetime.fromisoformat((log.get("ts") or "").rstrip("Z"))
        except ValueError:
            continue
        days_inactive = (new_dt - last_dt).days
        if days_inactive > inactivity_days:
            alerts.append({
                "user": user,
                "last_login": last_logins[user],
                "current_login": log.get("ts"),
                "days_inactive": days_inactive,
            })
    return alerts


def test_q12():
    last_logins = {
        "alice": "2026-01-01T00:00:00",   # 148 days ago from May-29
        "bob":   "2026-05-01T00:00:00",   # 28 days ago
    }
    logs = [
        {"ts": "2026-05-29T08:00:00", "user": "alice"},
        {"ts": "2026-05-29T08:00:00", "user": "bob"},
    ]
    alerts = dormant_account_activity(last_logins, logs, inactivity_days=90)
    users = {a["user"] for a in alerts}
    assert "alice" in users
    assert "bob" not in users
    print("Q12 tests passed")

# Q13 — Privileged Account Login Outside Business Hours
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given login events with role field, flag admin/privileged logins outside
  working hours (default: Mon-Fri 09:00-18:00 local time).

Security angle:
  Privileged logins at 2 AM are a strong insider-threat or compromise indicator.
  Most attackers operate in their own timezone, which differs from victim org's
  business hours. SOC analysts prioritise out-of-hours privileged activity.

Clarifying questions:
  1. Which timezone defines "business hours"?
  2. Which roles are "privileged"? Admin, root, svc-account?
  3. Should public holidays be considered?
"""

PRIVILEGED_ROLES = frozenset({"admin", "root", "superuser", "sre", "sysadmin"})
BUSINESS_HOUR_START = 9   # 09:00
BUSINESS_HOUR_END   = 18  # 18:00


def _is_outside_business_hours(ts: str) -> bool:
    try:
        dt = datetime.fromisoformat(ts.rstrip("Z"))
    except ValueError:
        return False
    return (
        dt.weekday() >= 5                        # Saturday / Sunday
        or dt.hour < BUSINESS_HOUR_START
        or dt.hour >= BUSINESS_HOUR_END
    )


# ── Naive ──────────────────────────────────────────────────────────────────
def privileged_login_outside_hours_naive(logs: List[Dict]) -> List[Dict]:
    return [
        log for log in logs
        if log.get("role", "").lower() in PRIVILEGED_ROLES
        and _is_outside_business_hours(log.get("ts", ""))
    ]


# ── Optimal ────────────────────────────────────────────────────────────────
def privileged_login_outside_hours(logs: List[Dict]) -> List[Dict]:
    """
    Same O(n) complexity. Optimal version uses generator for lazy evaluation
    and pre-compiled frozenset for O(1) role lookup.
    In production: apply timezone conversion using pytz before comparison.
    """
    alerts = []
    for log in logs:
        role = (log.get("role") or "").lower()
        if role not in PRIVILEGED_ROLES:
            continue
        if _is_outside_business_hours(log.get("ts", "")):
            alerts.append({
                "user": log.get("user"),
                "role": role,
                "ts": log.get("ts"),
                "src_ip": log.get("src_ip"),
            })
    return alerts


def test_q13():
    logs = [
        {"ts": "2026-05-29T02:30:00", "user": "alice", "role": "admin",
         "src_ip": "1.1.1.1"},   # 2:30 AM — outside hours
        {"ts": "2026-05-29T10:00:00", "user": "bob", "role": "admin",
         "src_ip": "2.2.2.2"},   # 10 AM Friday — inside hours
        {"ts": "2026-05-30T15:00:00", "user": "carol", "role": "user",
         "src_ip": "3.3.3.3"},   # regular user
        {"ts": "2026-05-31T10:00:00", "user": "dave", "role": "root",
         "src_ip": "4.4.4.4"},   # Sunday — outside hours
    ]
    alerts = privileged_login_outside_hours(logs)
    users = {a["user"] for a in alerts}
    assert "alice" in users
    assert "dave" in users
    assert "bob" not in users
    assert "carol" not in users
    print("Q13 tests passed")

# Q14 — Multiple Failed MFA Attempts
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given MFA event logs (event types: MFA_FAILED, MFA_SUCCESS), detect users
  with more than N MFA failures.

Security angle:
  Repeated MFA failures signal either a targeted attack where the attacker
  has the password but not the second factor, or a misconfigured authenticator
  app. Locking MFA after N failures prevents brute force of OTP codes.
"""


def multiple_failed_mfa(logs: List[Dict], threshold: int = 3) -> List[str]:
    """
    Optimal: Counter filtered in one pass.
    O(n) time, O(u) space.
    """
    counts: Counter = Counter(
        (log.get("user") or "").lower()
        for log in logs
        if log.get("event") == "MFA_FAILED" and log.get("user")
    )
    return [u for u, c in counts.items() if c > threshold]


def test_q14():
    logs = [
        {"event": "MFA_FAILED", "user": "alice"},
        {"event": "MFA_FAILED", "user": "alice"},
        {"event": "MFA_FAILED", "user": "alice"},
        {"event": "MFA_FAILED", "user": "alice"},
        {"event": "MFA_FAILED", "user": "bob"},
        {"event": "MFA_SUCCESS", "user": "alice"},
    ]
    result = multiple_failed_mfa(logs, threshold=3)
    assert "alice" in result
    assert "bob" not in result
    print("Q14 tests passed")

# Q15 — MFA Push Fatigue Pattern
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  MFA push fatigue = attacker sends dozens of push notifications hoping the user
  accidentally approves one. Detect users receiving >= N push requests within
  T seconds.

Clarifying questions:
  1. Is the window per-session or global rolling?
  2. Should the window reset on approval?
  3. What is T (seconds)?

Security angle:
  Push bombing / MFA fatigue is how attackers bypassed Okta, Cisco, and
  Uber's MFA in 2022. Mitigation: number matching, additional context in push,
  hard limit on push frequency.
"""
from collections import deque


def mfa_push_fatigue_naive(
    logs: List[Dict],
    push_threshold: int = 3,
    window_seconds: int = 60,
) -> List[Dict]:
    """
    Naive: group by user, sort, then slide through timestamps.
    O(n log n) due to sort.
    """
    user_ts: Dict[str, List[str]] = defaultdict(list)
    for log in logs:
        if log.get("event") == "MFA_PUSH" and log.get("user"):
            user_ts[log["user"]].append(log["ts"])

    alerts = []
    for user, timestamps in user_ts.items():
        timestamps.sort()
        dq: deque = deque()
        for ts in timestamps:
            try:
                t = datetime.fromisoformat(ts.rstrip("Z"))
            except ValueError:
                continue
            dq.append(t)
            # evict events outside the window
            while dq and (t - dq[0]).total_seconds() > window_seconds:
                dq.popleft()
            if len(dq) >= push_threshold:
                alerts.append({
                    "user": user,
                    "push_count": len(dq),
                    "window_start": dq[0].isoformat(),
                    "window_end": t.isoformat(),
                })
                break   # report once per burst
    return alerts


def mfa_push_fatigue(
    logs: List[Dict],
    push_threshold: int = 3,
    window_seconds: int = 60,
) -> List[Dict]:
    """
    Optimal: same sliding-window deque, but processes events already in time
    order if logs are pre-sorted (common in streaming pipelines).
    Data structure: deque for O(1) append and popleft.
    O(n) amortised when input is sorted; O(n log n) with sort step.
    """
    return mfa_push_fatigue_naive(logs, push_threshold, window_seconds)


def test_q15():
    logs = [
        {"ts": "2026-05-29T08:04:00Z", "event": "MFA_PUSH", "user": "eve"},
        {"ts": "2026-05-29T08:04:10Z", "event": "MFA_PUSH", "user": "eve"},
        {"ts": "2026-05-29T08:04:20Z", "event": "MFA_PUSH", "user": "eve"},
        {"ts": "2026-05-29T08:10:00Z", "event": "MFA_PUSH", "user": "bob"},
    ]
    alerts = mfa_push_fatigue(logs, push_threshold=3, window_seconds=60)
    users = {a["user"] for a in alerts}
    assert "eve" in users
    assert "bob" not in users
    print("Q15 tests passed")

# Q16 — Detect Account Lockout Burst
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given account status logs with event ACCOUNT_LOCKED and src_ip, detect IPs
  (or /24 subnets) that cause more than N lockouts.

Security angle:
  Lockout bursts from one IP confirm an active brute-force attack. Detecting
  at the subnet level catches distributed spray across /24 that would evade
  per-IP rules. Trigger: block the subnet, alert SOC, notify affected users.
"""


def account_lockout_burst(
    logs: List[Dict],
    threshold: int = 3,
    by_subnet: bool = False,
) -> List[Dict]:
    """
    Optimal: Counter keyed by IP or /24 prefix.
    O(n) time.
    """
    def key(ip: str) -> str:
        if by_subnet:
            parts = ip.split(".")
            if len(parts) == 4:
                return ".".join(parts[:3]) + ".0/24"
        return ip

    counts: Counter = Counter()
    for log in logs:
        if log.get("event") != "ACCOUNT_LOCKED":
            continue
        ip = log.get("src_ip")
        if ip:
            counts[key(ip)] += 1

    return [
        {"ip_or_subnet": k, "lockout_count": c}
        for k, c in counts.items() if c > threshold
    ]


def test_q16():
    logs = [
        {"event": "ACCOUNT_LOCKED", "src_ip": "10.0.0.1", "user": "alice"},
        {"event": "ACCOUNT_LOCKED", "src_ip": "10.0.0.2", "user": "bob"},
        {"event": "ACCOUNT_LOCKED", "src_ip": "10.0.0.3", "user": "carol"},
        {"event": "ACCOUNT_LOCKED", "src_ip": "10.0.0.4", "user": "dave"},
        {"event": "LOGIN_FAILED",   "src_ip": "9.9.9.9",  "user": "eve"},
    ]
    # per-IP: none exceed threshold of 1
    result_ip = account_lockout_burst(logs, threshold=1, by_subnet=False)
    # per-/24: 10.0.0.0/24 has 4 lockouts
    result_subnet = account_lockout_burst(logs, threshold=3, by_subnet=True)
    subnets = [r["ip_or_subnet"] for r in result_subnet]
    assert "10.0.0.0/24" in subnets
    print("Q16 tests passed")

# Q17 — Detect High Error Rate Endpoint
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given HTTP access logs, compute error rate per endpoint (path).
  Return endpoints where error_count / total_count > threshold.

Clarifying questions:
  1. What counts as an error? 4xx, 5xx, or just 5xx?
  2. Is the path normalised (strip query string)?
  3. Minimum total requests to qualify (avoid noise from rarely hit endpoints)?

Security angle:
  High error rates on login endpoints may indicate brute force.
  High 5xx rates can indicate a DoS attack or injection causing backend crashes.
  High 404 rates on API paths indicate scanning or broken client integrations.
"""


def _is_error(status: int) -> bool:
    return status >= 400


def high_error_rate_endpoints_naive(
    logs: List[Dict],
    error_threshold: float = 0.5,
    min_requests: int = 5,
) -> List[Dict]:
    """Naive: two dicts (total, error) then filter."""
    totals: Dict[str, int] = defaultdict(int)
    errors: Dict[str, int] = defaultdict(int)
    for log in logs:
        path = log.get("path", "").split("?")[0]  # strip query string
        status = log.get("status", 0)
        if not path:
            continue
        totals[path] += 1
        if _is_error(status):
            errors[path] += 1

    return [
        {
            "path": path,
            "error_rate": errors[path] / totals[path],
            "total": totals[path],
            "errors": errors[path],
        }
        for path in totals
        if totals[path] >= min_requests
        and errors[path] / totals[path] > error_threshold
    ]


def high_error_rate_endpoints(
    logs: List[Dict],
    error_threshold: float = 0.5,
    min_requests: int = 5,
) -> List[Dict]:
    """
    Optimal: single-pass using one dict of (total, errors) tuples.
    Reduces dict lookups by storing both counts together.
    O(n) time, O(p) space where p = unique paths.
    """
    path_stats: Dict[str, List[int]] = defaultdict(lambda: [0, 0])  # [total, errors]
    for log in logs:
        path = (log.get("path") or "").split("?")[0]
        status = log.get("status", 0)
        if not path:
            continue
        path_stats[path][0] += 1
        if _is_error(status):
            path_stats[path][1] += 1

    results = []
    for path, (total, err_count) in path_stats.items():
        if total >= min_requests and err_count / total > error_threshold:
            results.append({
                "path": path,
                "error_rate": round(err_count / total, 3),
                "total": total,
                "errors": err_count,
            })
    results.sort(key=lambda r: -r["error_rate"])
    return results


def test_q17():
    logs = [
        {"path": "/login", "status": 401},
        {"path": "/login", "status": 401},
        {"path": "/login", "status": 401},
        {"path": "/login", "status": 200},
        {"path": "/login", "status": 200},
        {"path": "/home",  "status": 200},
        {"path": "/home",  "status": 200},
        {"path": "/home",  "status": 200},
        {"path": "/home",  "status": 200},
        {"path": "/home",  "status": 200},
    ]
    results = high_error_rate_endpoints(logs, error_threshold=0.5, min_requests=5)
    paths = [r["path"] for r in results]
    assert "/login" in paths
    assert "/home" not in paths
    print("Q17 tests passed")

# Q18 — Detect 404 Scanning
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Detect IPs generating many 404 responses across DISTINCT paths.
  A single IP hitting 10+ unique non-existent paths in a short session is
  almost certainly a scanner (nikto, dirbuster, gobuster, etc.).

Security angle:
  Directory/file enumeration is reconnaissance phase of web attacks.
  Tools like nikto, gobuster, and ffuf send thousands of requests to discover
  hidden admin panels, backup files, and exposed configuration files.

False positives:
  - Broken links on a popular page causing legitimate 404s.
  - Search engine crawlers with stale sitemaps.
"""


def detect_404_scanning_naive(
    logs: List[Dict],
    threshold: int = 5,
) -> List[Dict]:
    """Naive: two dicts — one for distinct paths, one for total 404s."""
    ip_paths: Dict[str, set] = defaultdict(set)
    for log in logs:
        if log.get("status") == 404:
            ip = log.get("src_ip")
            path = (log.get("path") or "").split("?")[0]
            if ip and path:
                ip_paths[ip].add(path)

    return [
        {"src_ip": ip, "distinct_404_paths": len(paths)}
        for ip, paths in ip_paths.items()
        if len(paths) > threshold
    ]


def detect_404_scanning(
    logs: List[Dict],
    threshold: int = 5,
) -> List[Dict]:
    """
    Optimal: dict of sets, single pass.
    Sets naturally deduplicate paths. O(n) time, O(i*p) space.
    """
    ip_paths: Dict[str, set] = defaultdict(set)
    for log in logs:
        if log.get("status") != 404:
            continue
        ip = log.get("src_ip")
        path = (log.get("path") or "").split("?")[0]
        if ip and path:
            ip_paths[ip].add(path)

    return sorted(
        [
            {"src_ip": ip, "distinct_404_paths": len(p)}
            for ip, p in ip_paths.items()
            if len(p) > threshold
        ],
        key=lambda r: -r["distinct_404_paths"],
    )


def test_q18():
    logs = []
    # Scanner: 10 distinct paths
    for i in range(10):
        logs.append({"src_ip": "192.0.2.55", "path": f"/path{i}", "status": 404})
    # Normal user: 1 broken link, repeated
    for _ in range(3):
        logs.append({"src_ip": "1.1.1.1", "path": "/broken", "status": 404})

    result = detect_404_scanning(logs, threshold=5)
    ips = [r["src_ip"] for r in result]
    assert "192.0.2.55" in ips
    assert "1.1.1.1" not in ips
    print("Q18 tests passed")


# ─────────────────────────────────────────────────────────────────────────────
# Q19 — Detect Directory Traversal Attempts
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Find HTTP requests where the URL or parameters contain path traversal
  patterns: ../, ..(backslash), %2e%2e, %2f, and encoded variants.

Security angle:
  Directory traversal allows attackers to read files outside the web root
  (e.g., /etc/passwd, /proc/self/environ, Windows/system.ini).
  CVE-2021-41773 (Apache) and CVE-2021-42013 are famous real-world examples.

False positives:
  - File sharing applications with legitimate .. in paths.
  - Content management systems with deep relative path structures.
"""

# Traversal signatures covering encoded variants
TRAVERSAL_PATTERNS = re.compile(
    r'(\.\./|\.\.\\|%2e%2e%2f|%2e%2e/|\.%2e/|%2e\./|%252e%252e)',
    re.IGNORECASE,
)


def detect_directory_traversal(logs: List[Dict]) -> List[Dict]:
    """
    Regex-based detection. Decode URL encoding first to catch double-encoded
    variants (%252e → %2e → .).
    O(n * r) time where r = regex cost per string.
    """
    alerts = []
    for log in logs:
        raw_path = log.get("path") or log.get("url") or ""
        # Decode once to catch %2e, twice to catch %252e
        decoded_once  = urllib.parse.unquote(raw_path)
        decoded_twice = urllib.parse.unquote(decoded_once)

        for candidate in (raw_path, decoded_once, decoded_twice):
            if TRAVERSAL_PATTERNS.search(candidate):
                alerts.append({
                    "src_ip": log.get("src_ip"),
                    "path": raw_path,
                    "decoded": decoded_twice,
                    "ts": log.get("ts"),
                })
                break
    return alerts


def test_q19():
    logs = [
        {"src_ip": "1.1.1.1", "path": "/files/../../etc/passwd"},
        {"src_ip": "2.2.2.2", "path": "/files/%2e%2e%2fetc%2fpasswd"},
        {"src_ip": "3.3.3.3", "path": "/files/%252e%252e/etc/passwd"},
        {"src_ip": "4.4.4.4", "path": "/files/report.pdf"},             # benign
    ]
    results = detect_directory_traversal(logs)
    ips = {r["src_ip"] for r in results}
    assert "1.1.1.1" in ips
    assert "2.2.2.2" in ips
    assert "3.3.3.3" in ips
    assert "4.4.4.4" not in ips
    print("Q19 tests passed")


# ─────────────────────────────────────────────────────────────────────────────
# Q20 — Detect SQL Injection Keywords
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Flag HTTP requests containing common SQL injection patterns in the URL,
  query parameters, or body. Patterns: ' OR 1=1, UNION SELECT, --, ;DROP, etc.

Security angle:
  SQLi is OWASP Top 10 #3. Even failed attempts should be logged and the
  source IP rate-limited or blocked. WAFs use signature lists similar to
  this approach (ModSecurity CRS, AWS WAF SQLi rules).

False positives:
  - Security testing / vulnerability scanners.
  - Legitimate SQL in code review tools or developer portals.
"""

SQLI_PATTERNS = re.compile(
    r"(\bunion\b.{0,20}\bselect\b"
    r"|\bor\b.{0,10}1\s*=\s*1"
    r"|--[\s;'\"]?"
    r"|\binsert\s+into\b"
    r"|\bdrop\s+table\b"
    r"|;\s*select\b"
    r"|'\s*or\s*'1'\s*=\s*'1"
    r"|sleep\s*\(\s*\d"
    r"|benchmark\s*\("
    r"|\bexec\s*\()",
    re.IGNORECASE,
)


def detect_sqli_naive(logs: List[Dict]) -> List[Dict]:
    """Simple regex scan over path + query string."""
    return [
        log for log in logs
        if SQLI_PATTERNS.search(
            urllib.parse.unquote(log.get("path") or "")
        )
    ]


def detect_sqli(logs: List[Dict]) -> List[Dict]:
    """
    Optimal: decode URL once, scan with compiled regex.
    Returns enriched alerts with matched pattern.
    O(n) time with compiled regex being O(1) per match attempt.
    """
    alerts = []
    for log in logs:
        raw = log.get("path") or log.get("url") or ""
        decoded = urllib.parse.unquote(raw)
        m = SQLI_PATTERNS.search(decoded)
        if m:
            alerts.append({
                "src_ip": log.get("src_ip"),
                "path": raw,
                "matched_pattern": m.group(0)[:50],
                "ts": log.get("ts"),
            })
    return alerts


def test_q20():
    logs = [
        {"src_ip": "1.1.1.1", "path": "/product?id=1 UNION SELECT password FROM users"},
        {"src_ip": "2.2.2.2", "path": "/login?user=admin'--"},
        {"src_ip": "3.3.3.3", "path": "/search?q=hello+world"},           # benign
        {"src_ip": "4.4.4.4", "path": "/api?q=1%20OR%201%3D1"},            # URL-encoded
    ]
    results = detect_sqli(logs)
    ips = {r["src_ip"] for r in results}
    assert "1.1.1.1" in ips
    assert "2.2.2.2" in ips
    assert "3.3.3.3" not in ips
    assert "4.4.4.4" in ips
    print("Q20 tests passed")
