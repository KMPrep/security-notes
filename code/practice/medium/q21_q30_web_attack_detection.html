# Q21 — Detect XSS Payloads
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Flag HTTP requests containing reflected/stored XSS patterns:
  <script>, javascript:, onerror=, onload=, and their URL/HTML-encoded forms.

Clarifying questions:
  1. Are we scanning only URL path+query, or also request body?
  2. Should we handle double-encoded payloads (e.g., %253cscript%253e)?
  3. Should the result include the matched pattern for analyst review?
  4. WAF mode (block) or IDS mode (log only)?

Security angle:
  XSS is OWASP Top 10 #3. Reflected XSS can steal session cookies, redirect
  users, or execute arbitrary JS in the victim's browser context. Stored XSS
  is persistent and even more dangerous. These signatures match
  the most common payload prefixes seen in real-world scans.

False positives:
  - Legitimate HTML editors or content management form submissions.
  - Security scanners used by your own pentest team.
  - Rich-text fields in ticketing systems.
"""

XSS_PATTERNS = re.compile(
    r"(<script[\s>]"
    r"|javascript\s*:"
    r"|on(?:error|load|click|mouseover|focus|blur)\s*="
    r"|<img[^>]+src\s*="
    r"|<svg[^>]+on\w+\s*="
    r"|expression\s*\("
    r"|&#x?[0-9a-f]+;)",          # HTML entity encoding
    re.IGNORECASE,
)


def detect_xss_naive(logs: List[Dict]) -> List[Dict]:
    """Scan decoded URL string with compiled regex."""
    return [
        log for log in logs
        if XSS_PATTERNS.search(urllib.parse.unquote(log.get("path") or ""))
    ]


def detect_xss(logs: List[Dict]) -> List[Dict]:
    """
    Optimal: decode URL up to twice (catches %253c -> %3c -> <), then scan.
    Returns enriched alert with matched token.
    O(n) time per log entry.
    """
    alerts = []
    for log in logs:
        raw = log.get("path") or log.get("url") or ""
        decoded = urllib.parse.unquote(urllib.parse.unquote(raw))
        m = XSS_PATTERNS.search(decoded)
        if m:
            alerts.append({
                "src_ip": log.get("src_ip"),
                "path": raw,
                "matched": m.group(0)[:60],
                "ts": log.get("ts"),
            })
    return alerts


def test_q21():
    logs = [
        {"src_ip": "1.1.1.1", "path": '/search?q=<script>alert(1)</script>'},
        {"src_ip": "2.2.2.2", "path": '/page?url=javascript:void(0)'},
        {"src_ip": "3.3.3.3", "path": '/img?src=x&onerror=alert(1)'},
        {"src_ip": "4.4.4.4", "path": '/search?q=%3Cscript%3Ealert(1)%3C/script%3E'},  # URL-encoded
        {"src_ip": "5.5.5.5", "path": '/search?q=hello+world'},                         # benign
    ]
    results = detect_xss(logs)
    ips = {r["src_ip"] for r in results}
    assert "1.1.1.1" in ips
    assert "2.2.2.2" in ips
    assert "3.3.3.3" in ips
    assert "4.4.4.4" in ips
    assert "5.5.5.5" not in ips
    print("Q21 tests passed")

# Q22 — Detect Command Injection Patterns
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Find requests containing OS command injection characters in query params:
  ;, &&, ||, |, backticks (`), $(), newlines injected into parameters.

Clarifying questions:
  1. Are we checking URL params only, or also POST body fields?
  2. Should we decode the URL before checking?
  3. Is there a specific parameter name that is higher-risk (e.g., "cmd", "exec")?

Security angle:
  Command injection allows attackers to run OS commands on the server.
  Common targets: ping utilities, log viewers, report generators that
  shell out to system tools without sanitising user input.
  OWASP Top 10 #3 (Injection category).

False positives:
  - Legitimate shell operators in developer-facing APIs (CI/CD tooling).
  - URLs in blog posts or documentation pages containing code snippets.
"""

CMDI_PATTERNS = re.compile(
    r"(;\s*\w"                   # ; followed by command
    r"|\|\s*\w"                  # pipe to command
    r"|&&\s*\w"                  # AND chaining
    r"|\|\|\s*\w"                # OR chaining
    r"|`[^`]+`"                  # backtick execution
    r"|\$\s*\("                  # $() subshell
    r"|%0[aA]"                   # URL-encoded newline
    r"|%7C"                      # URL-encoded |
    r"|\bwhoami\b|\bcat\s+/etc"  # common payloads
    r"|\bls\s+-"                 # ls flags
    r"|/bin/(?:sh|bash|zsh))",
    re.IGNORECASE,
)


def detect_command_injection(logs: List[Dict]) -> List[Dict]:
    """
    Optimal: decode URL first, match with compiled regex.
    O(n) time.
    """
    alerts = []
    for log in logs:
        raw = log.get("path") or log.get("url") or ""
        decoded = urllib.parse.unquote(raw)
        m = CMDI_PATTERNS.search(decoded)
        if m:
            alerts.append({
                "src_ip": log.get("src_ip"),
                "path": raw,
                "matched": m.group(0)[:60],
                "ts": log.get("ts"),
            })
    return alerts


def test_q22():
    logs = [
        {"src_ip": "1.1.1.1", "path": "/ping?host=127.0.0.1;cat /etc/passwd"},
        {"src_ip": "2.2.2.2", "path": "/tool?cmd=ls+-la%7C%7Cwhoami"},
        {"src_ip": "3.3.3.3", "path": "/report?name=sales&&rm+-rf+/"},
        {"src_ip": "4.4.4.4", "path": "/search?q=normal+query"},           # benign
    ]
    results = detect_command_injection(logs)
    ips = {r["src_ip"] for r in results}
    assert "1.1.1.1" in ips
    assert "2.2.2.2" in ips
    assert "3.3.3.3" in ips
    assert "4.4.4.4" not in ips
    print("Q22 tests passed")

# Q23 — Detect Suspicious User Agents
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Detect known scanner/tool user agents and return counts by source IP.
  Scanners: nikto, sqlmap, nmap, masscan, zgrab, dirbuster, gobuster, etc.

Clarifying questions:
  1. Is the user agent field always present?
  2. Should partial matches count (e.g., UA containing "sqlmap" anywhere)?
  3. Should we also flag missing/empty user agents?
  4. Return per-IP counts or per-UA-type counts?

Security angle:
  Automated scanners announce themselves in the user agent by default.
  Many operators never change the default. Catching these in real-time
  lets you rate-limit or block scanners before they complete reconnaissance.
  Combine with 404-burst detection for higher confidence.
"""

SCANNER_UA_PATTERNS = re.compile(
    r"(nikto|sqlmap|nmap|masscan|zgrab|dirbuster|gobuster|ffuf"
    r"|wfuzz|burpsuite|nessus|openvas|acunetix|nuclei"
    r"|python-requests|go-http-client|curl/[0-9]|wget/[0-9]"
    r"|java/[0-9]|libwww-perl|scrapy|mechanize)",
    re.IGNORECASE,
)


def detect_suspicious_user_agents_naive(
    logs: List[Dict],
) -> List[Dict]:
    """Scan each UA, count per IP. O(n)."""
    ip_counts: Dict[str, int] = defaultdict(int)
    for log in logs:
        ua = log.get("user_agent") or log.get("ua") or ""
        if SCANNER_UA_PATTERNS.search(ua):
            ip = log.get("src_ip", "unknown")
            ip_counts[ip] += 1

    return [{"src_ip": ip, "scanner_requests": c} for ip, c in ip_counts.items()]


def detect_suspicious_user_agents(
    logs: List[Dict],
) -> List[Dict]:
    """
    Optimal: Counter keyed by src_ip for scanner-matching logs only.
    Returns sorted list descending by request count.
    O(n) time, O(i) space where i = unique scanner IPs.
    """
    counts: Counter = Counter()
    ua_seen: Dict[str, Set[str]] = defaultdict(set)

    for log in logs:
        ua = log.get("user_agent") or log.get("ua") or ""
        ip = log.get("src_ip", "unknown")
        if SCANNER_UA_PATTERNS.search(ua):
            counts[ip] += 1
            m = SCANNER_UA_PATTERNS.search(ua)
            if m:
                ua_seen[ip].add(m.group(0).lower())

    return sorted(
        [
            {
                "src_ip": ip,
                "scanner_requests": c,
                "tools_seen": sorted(ua_seen[ip]),
            }
            for ip, c in counts.items()
        ],
        key=lambda r: -r["scanner_requests"],
    )


def test_q23():
    logs = [
        {"src_ip": "192.0.2.55", "ua": "nikto/2.1.6"},
        {"src_ip": "192.0.2.55", "ua": "nikto/2.1.6"},
        {"src_ip": "192.0.2.55", "ua": "nikto/2.1.6"},
        {"src_ip": "10.0.0.1",   "ua": "sqlmap/1.7"},
        {"src_ip": "10.0.0.1",   "ua": "sqlmap/1.7"},
        {"src_ip": "1.1.1.1",    "ua": "Mozilla/5.0 (Windows NT 10.0)"},  # benign
    ]
    results = detect_suspicious_user_agents(logs)
    ips = {r["src_ip"] for r in results}
    assert "192.0.2.55" in ips
    assert "10.0.0.1" in ips
    assert "1.1.1.1" not in ips
    assert results[0]["src_ip"] == "192.0.2.55"   # highest count first
    print("Q23 tests passed")

# Q24 — Detect Excessive POST Requests
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Find source IPs sending unusually many HTTP POST requests above a threshold.

Clarifying questions:
  1. Is the threshold absolute count or requests-per-minute?
  2. Should we scope to specific endpoints (e.g., /login, /api)?
  3. Should GET-heavy IPs also be checked (for comparison)?

Security angle:
  Excessive POSTs to /login indicate brute force.
  Excessive POSTs to /api/transfer indicate automated fraud.
  POSTs to /upload may indicate exfiltration or malware upload attempts.
  Rate-limiting POST specifically is a cheap, high-value control.

False positives:
  - Internal monitoring systems doing health checks.
  - Legitimate batch API clients.
"""


def detect_excessive_post_naive(
    logs: List[Dict],
    threshold: int = 100,
) -> List[str]:
    counts: Dict[str, int] = defaultdict(int)
    for log in logs:
        if log.get("method", "").upper() == "POST":
            ip = log.get("src_ip")
            if ip:
                counts[ip] += 1
    return [ip for ip, c in counts.items() if c > threshold]


def detect_excessive_post(
    logs: List[Dict],
    threshold: int = 100,
    path_filter: Optional[str] = None,
) -> List[Dict]:
    """
    Optimal: Counter with optional path-prefix filter.
    Returns enriched result with count and sample paths.
    O(n) time.
    """
    counts: Counter = Counter()
    ip_paths: Dict[str, Set[str]] = defaultdict(set)

    for log in logs:
        if log.get("method", "").upper() != "POST":
            continue
        ip = log.get("src_ip")
        path = (log.get("path") or "").split("?")[0]
        if not ip:
            continue
        if path_filter and not path.startswith(path_filter):
            continue
        counts[ip] += 1
        ip_paths[ip].add(path)

    return sorted(
        [
            {
                "src_ip": ip,
                "post_count": c,
                "distinct_paths": len(ip_paths[ip]),
            }
            for ip, c in counts.items()
            if c > threshold
        ],
        key=lambda r: -r["post_count"],
    )


def test_q24():
    logs = []
    # Attacker hammers /login via POST
    for _ in range(150):
        logs.append({"src_ip": "5.5.5.5", "method": "POST", "path": "/login"})
    # Normal user: a few POSTs
    for _ in range(5):
        logs.append({"src_ip": "1.1.1.1", "method": "POST", "path": "/submit"})
    # GET requests — should not count
    for _ in range(200):
        logs.append({"src_ip": "2.2.2.2", "method": "GET", "path": "/home"})

    results = detect_excessive_post(logs, threshold=100)
    ips = {r["src_ip"] for r in results}
    assert "5.5.5.5" in ips
    assert "1.1.1.1" not in ips
    assert "2.2.2.2" not in ips
    print("Q24 tests passed")

# Q25 — Detect Large Uploads
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given upload logs with byte sizes, detect individual uploads above a threshold
  or per-IP cumulative uploads above a threshold.

Clarifying questions:
  1. Is the threshold per-request, per-session, or per-day?
  2. Should encrypted uploads (HTTPS with unknown content type) also flag?
  3. What is the normal upload size for this application?

Security angle:
  Large uploads may indicate:
  - Data exfiltration (dumping a database or file archive).
  - Malware upload to a compromised server.
  - DoS via intentionally large payloads (HTTP body floods).
  Baseline: flag uploads > 10 MB per request or > 100 MB cumulative per IP.
"""

DEFAULT_SINGLE_UPLOAD_THRESHOLD = 10 * 1024 * 1024   # 10 MB
DEFAULT_CUMULATIVE_THRESHOLD    = 100 * 1024 * 1024  # 100 MB


def detect_large_uploads(
    logs: List[Dict],
    single_threshold: int = DEFAULT_SINGLE_UPLOAD_THRESHOLD,
    cumulative_threshold: int = DEFAULT_CUMULATIVE_THRESHOLD,
) -> List[Dict]:
    """
    Optimal: single pass — flag single large requests immediately and
    accumulate per-IP bytes to detect cumulative exfiltration.
    O(n) time, O(i) space.
    """
    ip_totals: Dict[str, int] = defaultdict(int)
    single_alerts: List[Dict] = []

    for log in logs:
        if log.get("method", "").upper() not in ("POST", "PUT"):
            continue
        ip = log.get("src_ip")
        size = log.get("bytes", 0)
        if not ip or not isinstance(size, int):
            continue

        ip_totals[ip] += size

        if size > single_threshold:
            single_alerts.append({
                "type": "single_upload",
                "src_ip": ip,
                "bytes": size,
                "path": log.get("path"),
                "ts": log.get("ts"),
            })

    cumulative_alerts = [
        {
            "type": "cumulative_upload",
            "src_ip": ip,
            "total_bytes": total,
        }
        for ip, total in ip_totals.items()
        if total > cumulative_threshold
    ]

    return single_alerts + cumulative_alerts


def test_q25():
    logs = [
        # Large single upload
        {"src_ip": "1.1.1.1", "method": "POST", "path": "/upload",
         "bytes": 15 * 1024 * 1024, "ts": "T1"},
        # Small uploads that accumulate
        *[{"src_ip": "2.2.2.2", "method": "POST", "path": "/upload",
           "bytes": 20 * 1024 * 1024, "ts": "T2"} for _ in range(6)],
        # Normal small upload
        {"src_ip": "3.3.3.3", "method": "POST", "path": "/upload",
         "bytes": 500 * 1024, "ts": "T3"},
    ]
    results = detect_large_uploads(logs)
    types = [(r["src_ip"], r["type"]) for r in results]
    assert ("1.1.1.1", "single_upload") in types
    assert ("2.2.2.2", "cumulative_upload") in types
    assert all(r["src_ip"] != "3.3.3.3" for r in results)
    print("Q25 tests passed")


# ─────────────────────────────────────────────────────────────────────────────
# Q26 — Detect Data Exfiltration by Volume
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given outbound network flow logs (src_host, dst_ip, bytes_out), find hosts
  sending more than X MB to external destinations.

Clarifying questions:
  1. How do we classify an IP as "external" vs internal? (RFC-1918 private ranges)
  2. Is the threshold per destination, per hour, or total?
  3. Should we distinguish HTTPS (port 443) from other protocols?

Security angle:
  Large outbound transfers to non-corporate destinations are the hallmark of
  data exfiltration. Attackers often stage data on a compromised host, compress
  it, and push it to a cloud storage endpoint or C2 server.
  Compare baseline: a normal workstation sends < 1 GB/day outbound.
"""
import ipaddress

PRIVATE_RANGES = [
    ipaddress.ip_network("10.0.0.0/8"),
    ipaddress.ip_network("172.16.0.0/12"),
    ipaddress.ip_network("192.168.0.0/16"),
    ipaddress.ip_network("127.0.0.0/8"),
]


def _is_internal(ip: str) -> bool:
    try:
        addr = ipaddress.ip_address(ip)
        return any(addr in net for net in PRIVATE_RANGES)
    except ValueError:
        return False


def detect_exfiltration_by_volume_naive(
    flows: List[Dict],
    threshold_bytes: int = 100 * 1024 * 1024,
) -> List[Dict]:
    """Naive: accumulate per-host then filter."""
    host_bytes: Dict[str, int] = defaultdict(int)
    for flow in flows:
        dst = flow.get("dst_ip", "")
        if not _is_internal(dst):
            host_bytes[flow.get("src_host", "")] += flow.get("bytes_out", 0)
    return [
        {"src_host": h, "total_bytes_out": b}
        for h, b in host_bytes.items()
        if b > threshold_bytes
    ]


def detect_exfiltration_by_volume(
    flows: List[Dict],
    threshold_bytes: int = 100 * 1024 * 1024,
) -> List[Dict]:
    """
    Optimal: same algorithm. Key improvement is using ipaddress stdlib
    to correctly classify RFC-1918 ranges (handles all edge cases).
    O(n) time, O(h) space where h = unique source hosts.
    """
    host_bytes: Counter = Counter()
    for flow in flows:
        dst = flow.get("dst_ip", "")
        if _is_internal(dst):
            continue
        host = flow.get("src_host")
        if host:
            host_bytes[host] += flow.get("bytes_out", 0)

    return sorted(
        [
            {"src_host": h, "total_bytes_out": b, "total_mb": round(b / 1024 / 1024, 1)}
            for h, b in host_bytes.items()
            if b > threshold_bytes
        ],
        key=lambda r: -r["total_bytes_out"],
    )


def test_q26():
    MB = 1024 * 1024
    flows = [
        # Exfiltrating host: 150 MB to external IP
        {"src_host": "laptop-99", "dst_ip": "203.0.113.50", "bytes_out": 150 * MB},
        # Normal internal traffic
        {"src_host": "laptop-99", "dst_ip": "10.0.0.1",     "bytes_out": 500 * MB},
        # Small external transfer — benign
        {"src_host": "laptop-01", "dst_ip": "8.8.8.8",      "bytes_out": 1 * MB},
    ]
    results = detect_exfiltration_by_volume(flows, threshold_bytes=100 * MB)
    hosts = {r["src_host"] for r in results}
    assert "laptop-99" in hosts
    assert "laptop-01" not in hosts
    print("Q26 tests passed")

# Q27 — Detect Beaconing Pattern
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given outbound connection timestamps per host, detect hosts communicating at
  suspiciously regular intervals (C2 beaconing). A low standard deviation of
  inter-connection intervals is the signature.

Clarifying questions:
  1. What interval regularity threshold do we use?
  2. Minimum number of connections to classify as beaconing?
  3. Should we scope to specific destination IPs?

Security angle:
  Malware beacons home at fixed intervals (e.g., every 60s) to receive
  commands. This is detectable by computing the std-deviation of time deltas:
  legitimate browsing is irregular; beaconing is near-zero variance.
  Used by FireEye, Darktrace, and Vectra for C2 detection.
"""


def _stdev(values: List[float]) -> float:
    n = len(values)
    if n < 2:
        return float("inf")
    mean = sum(values) / n
    variance = sum((x - mean) ** 2 for x in values) / n
    return math.sqrt(variance)


def detect_beaconing_naive(
    logs: List[Dict],
    min_connections: int = 5,
    max_stdev_seconds: float = 5.0,
) -> List[Dict]:
    """
    Naive: group by host, sort timestamps, compute deltas, check stdev.
    O(n log n) due to sort.
    """
    host_ts: Dict[str, List[datetime]] = defaultdict(list)
    for log in logs:
        host = log.get("host") or log.get("src_host")
        ts = log.get("ts")
        if host and ts:
            try:
                host_ts[host].append(datetime.fromisoformat(ts.rstrip("Z")))
            except ValueError:
                pass

    results = []
    for host, times in host_ts.items():
        times.sort()
        if len(times) < min_connections:
            continue
        deltas = [(times[i + 1] - times[i]).total_seconds()
                  for i in range(len(times) - 1)]
        sd = _stdev(deltas)
        if sd <= max_stdev_seconds:
            results.append({
                "host": host,
                "connections": len(times),
                "avg_interval_s": round(sum(deltas) / len(deltas), 1),
                "stdev_s": round(sd, 2),
            })
    return results


def detect_beaconing(
    logs: List[Dict],
    min_connections: int = 5,
    max_stdev_seconds: float = 5.0,
) -> List[Dict]:
    """
    Optimal: same algorithm. Use sorted insertion (bisect) if logs arrive
    out of order at scale. Here we sort once per host.
    O(n log n) is acceptable; at production scale use a streaming approach
    with a sliding histogram over inter-arrival times.
    """
    return detect_beaconing_naive(logs, min_connections, max_stdev_seconds)


def test_q27():
    # Beaconing host: connects every 60s with small jitter
    base = datetime(2026, 5, 29, 8, 0, 0)
    beacon_logs = [
        {"host": "infected-01", "ts": (base + timedelta(seconds=i * 60)).isoformat()}
        for i in range(10)
    ]
    # Normal host: irregular intervals
    normal_logs = [
        {"host": "laptop-01", "ts": (base + timedelta(seconds=t)).isoformat()}
        for t in [0, 5, 45, 120, 300, 500]
    ]
    results = detect_beaconing(beacon_logs + normal_logs, min_connections=5, max_stdev_seconds=5.0)
    hosts = {r["host"] for r in results}
    assert "infected-01" in hosts
    assert "laptop-01" not in hosts
    print("Q27 tests passed")


# ─────────────────────────────────────────────────────────────────────────────
# Q28 — Detect Rare Domain Access
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given DNS query logs, return domains that were accessed by only ONE host.
  Rare domains are a strong indicator of targeted malware C2 traffic.

Clarifying questions:
  1. Should sub-domains be normalised to the registered domain (eTLD+1)?
  2. What is the observation window?
  3. Should known CDN/cloud domains be excluded from the rarity check?

Security angle:
  DGA (Domain Generation Algorithm) malware and targeted APT tooling often
  communicate with domains no other host in the network contacts. A domain
  seen by exactly one host is anomalous. Combine with entropy score for
  higher confidence.

False positives:
  - Legitimate internal tools contacting their own unique update server.
  - First-ever access to a new SaaS tool before others adopt it.
"""


def detect_rare_domains_naive(logs: List[Dict]) -> List[str]:
    """Simple dict of sets. O(n)."""
    domain_hosts: Dict[str, set] = defaultdict(set)
    for log in logs:
        domain = (log.get("query") or "").lower()
        host = log.get("host")
        if domain and host:
            domain_hosts[domain].add(host)
    return [d for d, hosts in domain_hosts.items() if len(hosts) == 1]


def detect_rare_domains(
    logs: List[Dict],
    max_hosts: int = 1,
) -> List[Dict]:
    """
    Optimal: defaultdict of sets, configurable max_hosts threshold.
    Returns domains with host count <= max_hosts.
    O(n) time, O(d*h) space.
    """
    domain_hosts: Dict[str, set] = defaultdict(set)
    for log in logs:
        domain = (log.get("query") or "").lower()
        host = log.get("host")
        if domain and host:
            domain_hosts[domain].add(host)

    return [
        {"domain": d, "seen_by_hosts": list(hosts)}
        for d, hosts in domain_hosts.items()
        if len(hosts) <= max_hosts
    ]

# ─────────────────────────────────────────────────────────────────────────────
# Q29 — Detect DGA-Like Domains
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Flag DNS queries where the hostname label has high Shannon entropy or
  contains a high ratio of consonant clusters (typical of DGA output).

Clarifying questions:
  1. Which label do we score — the leftmost subdomain, or the full hostname?
  2. What entropy threshold? (typical: > 3.5 bits/char for DGA)
  3. Should we also check label length? (DGA labels are often 12-20 chars)
  4. Should we whitelist known CDN/cloud domains?

Security angle:
  DGA malware generates thousands of random-looking domain names per day.
  Only the "active" domain resolves — the rest are sinkholes. Entropy analysis
  catches these without needing a blacklist. Used in academic and enterprise
  DGA detection pipelines.

False positives:
  - Long random-looking hostnames in legitimate CDN URLs.
  - Hash-based filenames embedded in subdomains (e.g., npm CDN URLs).
"""

VOWELS = set("aeiou")


def _shannon_entropy(s: str) -> float:
    """Compute Shannon entropy in bits per character."""
    if not s:
        return 0.0
    freq = Counter(s.lower())
    total = len(s)
    return -sum((c / total) * math.log2(c / total) for c in freq.values())


def _consonant_ratio(label: str) -> float:
    alpha = [c for c in label.lower() if c.isalpha()]
    if not alpha:
        return 0.0
    consonants = [c for c in alpha if c not in VOWELS]
    return len(consonants) / len(alpha)


def detect_dga_domains_naive(
    logs: List[Dict],
    entropy_threshold: float = 3.5,
    min_label_len: int = 10,
) -> List[Dict]:
    """Simple entropy check on the first label."""
    results = []
    for log in logs:
        domain = (log.get("query") or "").lower().rstrip(".")
        label = domain.split(".")[0]
        if len(label) < min_label_len:
            continue
        entropy = _shannon_entropy(label)
        if entropy >= entropy_threshold:
            results.append({"domain": domain, "label": label, "entropy": round(entropy, 2)})
    return results


def detect_dga_domains(
    logs: List[Dict],
    entropy_threshold: float = 3.5,
    consonant_ratio_threshold: float = 0.75,
    min_label_len: int = 10,
) -> List[Dict]:
    """
    Optimal: combines entropy + consonant ratio for better precision.
    Either signal alone produces false positives; both together reduce them.
    O(n * L) time where L = max label length (treated as O(n) in practice).

    Data structure: none beyond list comprehension — no grouping needed.
    """
    results = []
    for log in logs:
        domain = (log.get("query") or "").lower().rstrip(".")
        label = domain.split(".")[0]
        if len(label) < min_label_len:
            continue
        entropy = _shannon_entropy(label)
        c_ratio = _consonant_ratio(label)
        if entropy >= entropy_threshold or c_ratio >= consonant_ratio_threshold:
            results.append({
                "domain": domain,
                "label": label,
                "entropy": round(entropy, 2),
                "consonant_ratio": round(c_ratio, 2),
                "host": log.get("host"),
                "ts": log.get("ts"),
            })
    return sorted(results, key=lambda r: -r["entropy"])


def test_q29():
    logs = [
        {"host": "laptop-03", "query": "a8f7sd9f7sdf9sdf7sd.example.net"},  # DGA
        {"host": "laptop-03", "query": "b9f8sd7f6sdf6sd8fs.example.net"},   # DGA
        {"host": "laptop-01", "query": "accounts.google.com"},               # benign
        {"host": "laptop-01", "query": "login.microsoftonline.com"},         # benign
    ]
    results = detect_dga_domains(logs, entropy_threshold=3.5, min_label_len=8)
    domains = {r["domain"] for r in results}
    assert "a8f7sd9f7sdf9sdf7sd.example.net" in domains
    assert "b9f8sd7f6sdf6sd8fs.example.net" in domains
    assert "accounts.google.com" not in domains
    print("Q29 tests passed")

# Q30 — Detect Newly Seen Domains
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given a historical set of known domains (whitelist) and new DNS query logs,
  return domains not seen in the historical baseline.

Clarifying questions:
  1. Should we normalise to registered domain (eTLD+1) or use full FQDN?
  2. How recent is "historical"? Last 30 days? 90 days?
  3. Should the new domains be added to the whitelist after confirmation?
  4. Should common CDN and cloud provider domains be pre-whitelisted?

Security angle:
  Newly registered / first-seen domains are high-risk:
  - C2 domains set up minutes before an attack.
  - Phishing lookalike domains.
  - Malware update domains not yet in threat intel feeds.
  "New" does not mean malicious, but it warrants investigation.

False positives:
  - Legitimate new SaaS tool adopted by one employee.
  - First time visiting a partner site.
"""


def detect_newly_seen_domains_naive(
    historical_domains: Set[str],
    new_logs: List[Dict],
) -> List[str]:
    """Simple set difference. O(n)."""
    return [
        log["query"]
        for log in new_logs
        if log.get("query") and log["query"].lower() not in historical_domains
    ]


def detect_newly_seen_domains(
    historical_domains: Set[str],
    new_logs: List[Dict],
) -> List[Dict]:
    """
    Optimal: normalise historical set once to lowercase; single scan of logs.
    Returns enriched alerts with host and timestamp.
    O(n) time for scan, O(d) space for normalised set.

    Data structure: frozenset for O(1) membership test and immutability.
    """
    known = frozenset(d.lower() for d in historical_domains)

    seen_new: Dict[str, Dict] = {}   # deduplicate same domain from many hosts
    for log in new_logs:
        domain = (log.get("query") or "").lower().rstrip(".")
        if not domain or domain in known:
            continue
        if domain not in seen_new:
            seen_new[domain] = {
                "domain": domain,
                "first_seen_ts": log.get("ts"),
                "first_host": log.get("host"),
            }

    return list(seen_new.values())


def test_q30():
    historical = {"accounts.google.com", "login.microsoftonline.com",
                  "github.com", "pypi.org"}
    logs = [
        {"host": "laptop-01", "query": "accounts.google.com",    "ts": "T1"},
        {"host": "laptop-03", "query": "a8f7sd9f7sdf9sdf7sd.example.net", "ts": "T2"},
        {"host": "laptop-03", "query": "evil-c2.example.org",    "ts": "T3"},
        {"host": "laptop-03", "query": "evil-c2.example.org",    "ts": "T4"},  # duplicate
    ]
    results = detect_newly_seen_domains(historical, logs)
    domains = {r["domain"] for r in results}
    assert "a8f7sd9f7sdf9sdf7sd.example.net" in domains
    assert "evil-c2.example.org" in domains
    assert "accounts.google.com" not in domains
    assert len([r for r in results if r["domain"] == "evil-c2.example.org"]) == 1  # deduped
    print("Q30 tests passed")
