# Q41 — Parse Apache Access Logs
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Parse Apache Combined Log Format lines into structured dicts.
  Format: 127.0.0.1 - frank [10/Oct/2000:13:55:36 -0700] "GET /apache HTTP/1.1" 200 2326

Clarifying questions:
  1. Are log lines guaranteed to be in Combined Log Format, or may they vary?
  2. Should malformed lines be skipped silently or raise an error?
  3. Is the referrer/agent field needed for the use case?

Security angle:
  Parsing is the foundation for all web log analytics. Malformed lines can
  indicate log injection attempts — an attacker inserting newlines into a
  URL to inject fake log entries. A robust parser skips or flags unparseable
  lines rather than crashing.
"""

APACHE_LOG_RE = re.compile(
    r'(?P<src_ip>\S+)'           # remote host
    r'\s+\S+'                    # ident (usually -)
    r'\s+(?P<user>\S+)'          # auth user
    r'\s+\[(?P<ts>[^\]]+)\]'     # timestamp [dd/Mon/yyyy:HH:MM:SS ±HHMM]
    r'\s+"(?P<request>[^"]*)"'   # request line
    r'\s+(?P<status>\d+)'        # status code
    r'\s+(?P<bytes>\S+)'         # bytes sent (- if none)
    r'(?:\s+"(?P<referrer>[^"]*)")?' # optional referrer
    r'(?:\s+"(?P<ua>[^"]*)")?'   # optional user agent
)


def parse_apache_log_naive(line: str) -> Optional[Dict]:
    """Simple regex match and split."""
    m = APACHE_LOG_RE.match(line)
    if not m:
        return None
    d = m.groupdict()
    d["status"] = int(d["status"])
    d["bytes"] = int(d["bytes"]) if d["bytes"] != "-" else 0
    return d


def parse_apache_logs(lines: List[str]) -> List[Dict]:
    """
    Optimal: compile regex once (done at module level), parse all lines,
    silently skip malformed entries.
    O(n) time where n = number of lines.
    """
    records = []
    for line in lines:
        m = APACHE_LOG_RE.match(line.strip())
        if not m:
            continue
        d = m.groupdict()
        d["status"] = int(d["status"])
        d["bytes"] = int(d["bytes"]) if d["bytes"] != "-" else 0
        # Split request into method, path, protocol
        parts = d["request"].split(" ")
        d["method"] = parts[0] if len(parts) >= 1 else ""
        d["path"]   = parts[1] if len(parts) >= 2 else ""
        records.append(d)
    return records


def test_q41():
    lines = [
        '127.0.0.1 - frank [10/Oct/2000:13:55:36 -0700] "GET /apache_pb.gif HTTP/1.1" 200 2326',
        '10.0.0.1 - - [11/Oct/2000:09:12:00 +0000] "POST /login HTTP/1.1" 401 512 "-" "Mozilla/5.0"',
        "this is a malformed line",
    ]
    records = parse_apache_logs(lines)
    assert len(records) == 2
    assert records[0]["src_ip"] == "127.0.0.1"
    assert records[0]["method"] == "GET"
    assert records[0]["status"] == 200
    assert records[1]["status"] == 401
    assert records[1]["method"] == "POST"
    print("Q41 tests passed")

# Q42 — Parse NGINX Logs
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Parse NGINX default log format lines. NGINX default format:
  $remote_addr - $remote_user [$time_local] "$request" $status $body_bytes_sent
  "$http_referer" "$http_user_agent"

Clarifying questions:
  1. Is the log format the default NGINX format or a custom one?
  2. Should we parse combined (with referrer/UA) or main format only?
  3. Should the request line be broken into method/path/protocol?

Security angle:
  Same as Apache — parsing is foundational. NGINX logs can also carry
  real user IPs if X-Forwarded-For is configured via $proxy_add_x_forwarded_for.
  Injected newlines in URLs can corrupt log files — parsers should handle
  unexpected tokens gracefully.
"""

# NGINX default format is identical in structure to Apache Combined Log Format
NGINX_LOG_RE = APACHE_LOG_RE   # Reuse the same regex


def parse_nginx_logs(lines: List[str]) -> List[Dict]:
    """
    Optimal: reuse the same compiled regex as Apache — both formats are
    structurally identical in their common configuration.
    O(n) time.
    """
    records = []
    for line in lines:
        m = NGINX_LOG_RE.match(line.strip())
        if not m:
            continue
        d = m.groupdict()
        d["status"] = int(d["status"])
        d["bytes"] = int(d["bytes"]) if d["bytes"] != "-" else 0
        parts = d["request"].split(" ")
        d["method"] = parts[0] if len(parts) >= 1 else ""
        d["path"]   = parts[1] if len(parts) >= 2 else ""
        records.append(d)
    return records


def test_q42():
    lines = [
        '192.168.1.5 - - [29/May/2026:08:00:01 +0000] "GET /index.html HTTP/2.0" 200 4096 "-" "curl/7.85"',
        '192.168.1.6 - - [29/May/2026:08:00:02 +0000] "DELETE /admin HTTP/1.1" 403 0 "-" "sqlmap/1.7"',
    ]
    records = parse_nginx_logs(lines)
    assert len(records) == 2
    assert records[0]["method"] == "GET"
    assert records[1]["status"] == 403
    assert records[1]["ua"] == "sqlmap/1.7"
    print("Q42 tests passed")

# Q43 — Parse CloudTrail JSON Events
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  CloudTrail delivers JSON files with a top-level "Records" array.
  Parse and return the list of event dicts with key fields: eventTime,
  eventName, eventSource, userIdentity, sourceIPAddress, errorCode.

Clarifying questions:
  1. Are we processing a single JSON blob or a stream of JSON Lines?
  2. Should nested fields (userIdentity.principalId) be flattened?
  3. Are errors (errorCode present) worth flagging immediately?

Security angle:
  CloudTrail is the audit log for all AWS control-plane actions. Parsing it
  correctly is the prerequisite for every AWS detection (Q44–Q50).
  Key fields: errorCode (failed API calls), eventName (what happened),
  userIdentity (who), sourceIPAddress (from where).
"""


def parse_cloudtrail_json(raw: str) -> List[Dict]:
    """
    Optimal: parse JSON, extract 'Records' list, normalise key fields.
    Handles both single-record and multi-record responses.
    O(n) time where n = number of records.
    """
    try:
        data = json.loads(raw)
    except json.JSONDecodeError:
        return []

    records = data.get("Records", [])
    if not isinstance(records, list):
        return []

    normalised = []
    for rec in records:
        normalised.append({
            "ts":           rec.get("eventTime"),
            "event_name":  rec.get("eventName"),
            "event_source": rec.get("eventSource"),
            "principal":   (rec.get("userIdentity") or {}).get("arn")
                           or (rec.get("userIdentity") or {}).get("principalId"),
            "principal_type": (rec.get("userIdentity") or {}).get("type"),
            "src_ip":      rec.get("sourceIPAddress"),
            "region":      rec.get("awsRegion"),
            "error_code":  rec.get("errorCode"),
            "error_msg":   rec.get("errorMessage"),
            "raw":         rec,
        })
    return normalised


def test_q43():
    sample = json.dumps({
        "Records": [
            {
                "eventTime": "2026-05-29T08:00:00Z",
                "eventName": "DescribeInstances",
                "eventSource": "ec2.amazonaws.com",
                "awsRegion": "us-east-1",
                "sourceIPAddress": "203.0.113.1",
                "userIdentity": {"type": "IAMUser", "arn": "arn:aws:iam::123:user/alice"},
            },
            {
                "eventTime": "2026-05-29T08:01:00Z",
                "eventName": "GetSecretValue",
                "eventSource": "secretsmanager.amazonaws.com",
                "awsRegion": "us-east-1",
                "sourceIPAddress": "203.0.113.2",
                "userIdentity": {"type": "IAMUser", "arn": "arn:aws:iam::123:user/bob"},
                "errorCode": "AccessDenied",
                "errorMessage": "User is not authorized",
            },
        ]
    })
    records = parse_cloudtrail_json(sample)
    assert len(records) == 2
    assert records[0]["event_name"] == "DescribeInstances"
    assert records[1]["error_code"] == "AccessDenied"
    assert records[1]["principal"] == "arn:aws:iam::123:user/bob"
    print("Q43 tests passed")

# Q44 — Find Failed AWS API Calls
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  From parsed CloudTrail events, return those with a non-null errorCode.

Clarifying questions:
  1. Should we distinguish AccessDenied from throttling errors?
  2. Are repeated AccessDenied on sensitive APIs (GetSecretValue, AssumeRole)
     higher priority?
  3. Should we group by (principal, error_code)?

Security angle:
  Repeated AccessDenied events on high-value APIs indicate:
  - Privilege escalation probing (trying forbidden actions).
  - Compromised low-privilege credentials testing their boundaries.
  - Automated scanners enumerating IAM permissions.
  AWS GuardDuty 'PenTest' and 'UnauthorizedAccess' findings use this signal.
"""

SENSITIVE_APIS = frozenset({
    "GetSecretValue", "GetParameter", "GetParameters",
    "AssumeRole", "AssumeRoleWithSAML", "AssumeRoleWithWebIdentity",
    "CreateUser", "AttachUserPolicy", "PutUserPolicy",
    "CreateAccessKey", "UpdateAccessKey",
    "DeleteTrail", "StopLogging", "DeleteLogGroup",
    "PutBucketPolicy", "PutBucketAcl",
})


def find_failed_aws_calls(
    events: List[Dict],
    sensitive_only: bool = False,
) -> List[Dict]:
    """
    Optimal: single pass filter on error_code presence.
    Optional sensitive_only flag restricts to high-value API names.
    O(n) time.
    """
    results = []
    for e in events:
        if not e.get("error_code"):
            continue
        if sensitive_only and e.get("event_name") not in SENSITIVE_APIS:
            continue
        results.append(e)
    return results


def test_q44():
    events = [
        {"event_name": "DescribeInstances",  "principal": "alice", "error_code": None},
        {"event_name": "GetSecretValue",     "principal": "bob",   "error_code": "AccessDenied"},
        {"event_name": "AssumeRole",         "principal": "eve",   "error_code": "AccessDenied"},
        {"event_name": "ListBuckets",        "principal": "eve",   "error_code": "AccessDenied"},
    ]
    # All failures
    all_failed = find_failed_aws_calls(events)
    assert len(all_failed) == 3
    # Sensitive only
    sensitive_failed = find_failed_aws_calls(events, sensitive_only=True)
    assert len(sensitive_failed) == 2
    names = {e["event_name"] for e in sensitive_failed}
    assert "GetSecretValue" in names
    assert "AssumeRole" in names
    assert "ListBuckets" not in names
    print("Q44 tests passed")

# Q45 — Count CloudTrail Events by Principal
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Aggregate CloudTrail events and return per-principal event counts,
  sorted descending. Useful for spotting unusually active principals.

Clarifying questions:
  1. Should we count all events or only failed ones?
  2. Should service roles (e.g., Lambda execution roles) be excluded?
  3. What time window should we aggregate over?

Security angle:
  An IAM principal with 10x more API calls than baseline may indicate:
  - Compromised credential being used by an attacker for reconnaissance.
  - Over-privileged automation that is misbehaving.
  - An insider abusing their access.
"""


def count_events_by_principal(events: List[Dict]) -> List[Dict]:
    """
    Optimal: Counter over principal field. O(n) time.
    """
    counts: Counter = Counter()
    for e in events:
        p = e.get("principal") or "unknown"
        counts[p] += 1

    return [
        {"principal": p, "event_count": c}
        for p, c in counts.most_common()
    ]


def test_q45():
    events = [
        {"principal": "alice", "event_name": "DescribeInstances"},
        {"principal": "alice", "event_name": "ListBuckets"},
        {"principal": "alice", "event_name": "GetObject"},
        {"principal": "bob",   "event_name": "DescribeInstances"},
        {"principal": "eve",   "event_name": "GetSecretValue"},
        {"principal": "eve",   "event_name": "GetSecretValue"},
        {"principal": "eve",   "event_name": "AssumeRole"},
    ]
    results = count_events_by_principal(events)
    assert results[0]["principal"] in ("eve", "alice")   # both have 3
    totals = {r["principal"]: r["event_count"] for r in results}
    assert totals["alice"] == 3
    assert totals["bob"] == 1
    assert totals["eve"] == 3
    print("Q45 tests passed")

# Q46 — Detect Root Account Usage
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Flag any CloudTrail event made by the AWS root account.

Clarifying questions:
  1. Is the root account identifiable by type="Root" in userIdentity?
  2. Should we also flag root access key usage?
  3. Should any root usage immediately page on-call?

Security angle:
  AWS best practice: the root account should NEVER be used for routine
  operations. Root has unrestricted access and cannot be restricted by IAM
  policies. Any root activity is high-severity and could indicate:
  - Account takeover.
  - An attacker who obtained the root password.
  - An insider bypassing controls.
  CIS AWS Benchmark 1.1: Avoid the use of the root account.
"""


def detect_root_account_usage(events: List[Dict]) -> List[Dict]:
    """
    Optimal: O(1) check on principal_type field per event.
    Supports both normalized events (from parse_cloudtrail_json) and raw dicts.
    O(n) time.
    """
    alerts = []
    for e in events:
        ptype = (
            e.get("principal_type")
            or (e.get("userIdentity") or {}).get("type")
            or ""
        )
        if ptype == "Root":
            alerts.append({
                "severity": "CRITICAL",
                "event_name": e.get("event_name") or e.get("eventName"),
                "ts":         e.get("ts") or e.get("eventTime"),
                "src_ip":     e.get("src_ip") or e.get("sourceIPAddress"),
                "region":     e.get("region") or e.get("awsRegion"),
            })
    return alerts


def test_q46():
    events = [
        {"event_name": "ConsoleLogin",    "principal_type": "Root",    "ts": "T1", "src_ip": "1.1.1.1"},
        {"event_name": "DeleteTrail",     "principal_type": "Root",    "ts": "T2", "src_ip": "1.1.1.1"},
        {"event_name": "DescribeInstances","principal_type": "IAMUser", "ts": "T3", "src_ip": "2.2.2.2"},
    ]
    alerts = detect_root_account_usage(events)
    assert len(alerts) == 2
    names = {a["event_name"] for a in alerts}
    assert "ConsoleLogin" in names
    assert "DeleteTrail" in names
    print("Q46 tests passed")

# Q47 — Detect Console Login Without MFA
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Flag ConsoleLogin events where additionalEventData.MFAUsed == "No".

Clarifying questions:
  1. Is this alert for all users or only privileged/admin users?
  2. Should federated logins (SSO) also be checked?
  3. Is there a grace period for newly provisioned users?

Security angle:
  MFA prevents account takeover even when passwords are phished or leaked.
  Console logins without MFA are a compliance violation (CIS AWS 1.10,
  SOC2, ISO 27001). An attacker who phishes an IAM user's password can
  log in without MFA if it is not enforced at the policy level.
"""


def detect_console_login_without_mfa(events: List[Dict]) -> List[Dict]:
    """
    Optimal: filter on eventName == ConsoleLogin AND MFAUsed != 'Yes'.
    additionalEventData is nested in the raw CloudTrail record.
    O(n) time.
    """
    alerts = []
    for e in events:
        name = e.get("event_name") or e.get("eventName") or ""
        if name != "ConsoleLogin":
            continue
        raw = e.get("raw") or e
        add_data = raw.get("additionalEventData") or {}
        mfa_used = add_data.get("MFAUsed", "No")
        if mfa_used != "Yes":
            alerts.append({
                "severity": "HIGH",
                "principal": e.get("principal") or (raw.get("userIdentity") or {}).get("arn"),
                "ts":        e.get("ts") or raw.get("eventTime"),
                "src_ip":    e.get("src_ip") or raw.get("sourceIPAddress"),
            })
    return alerts


def test_q47():
    events = [
        {
            "event_name": "ConsoleLogin", "principal": "alice", "ts": "T1", "src_ip": "1.1.1.1",
            "raw": {"additionalEventData": {"MFAUsed": "No"}},
        },
        {
            "event_name": "ConsoleLogin", "principal": "bob", "ts": "T2", "src_ip": "2.2.2.2",
            "raw": {"additionalEventData": {"MFAUsed": "Yes"}},
        },
        {
            "event_name": "DescribeInstances", "principal": "alice", "ts": "T3",
            "raw": {},
        },
    ]
    alerts = detect_console_login_without_mfa(events)
    assert len(alerts) == 1
    assert alerts[0]["principal"] == "alice"
    print("Q47 tests passed")

# Q48 — Detect Security Group Opened to Internet
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Flag CloudTrail AuthorizeSecurityGroupIngress events where the CIDR is
  0.0.0.0/0 or ::/0 (open to all internet traffic).

Clarifying questions:
  1. Should any port opened to 0.0.0.0/0 trigger an alert, or only sensitive ports?
  2. Should "update" events (ModifyNetworkInterfaceAttribute) also be checked?
  3. Are there exceptions for load balancer security groups?

Security angle:
  An overly permissive security group can expose internal services to the
  internet (databases, admin consoles, internal APIs). This is a common
  misconfiguration that attackers exploit within minutes of it going live.
  CIS AWS Benchmark 4.1-4.4, AWS Config rule restricted-ssh.
"""

OPEN_CIDR = frozenset({"0.0.0.0/0", "::/0"})


def detect_sg_opened_to_internet(events: List[Dict]) -> List[Dict]:
    """
    Optimal: filter on eventName + walk requestParameters.ipPermissions
    to find any ingress rule with an open CIDR.
    O(n * p) where p = permissions per event (small constant).
    """
    alerts = []
    for e in events:
        name = e.get("event_name") or e.get("eventName") or ""
        if name not in ("AuthorizeSecurityGroupIngress", "ModifySecurityGroupRules"):
            continue
        raw = e.get("raw") or e
        params = raw.get("requestParameters") or {}
        ip_perms = params.get("ipPermissions") or {}
        items = ip_perms.get("items") or []

        for perm in items:
            port_from = perm.get("fromPort")
            port_to   = perm.get("toPort")
            for ip_range in (perm.get("ipRanges") or {}).get("items") or []:
                cidr = ip_range.get("cidrIp") or ip_range.get("cidrIpv6") or ""
                if cidr in OPEN_CIDR:
                    alerts.append({
                        "severity": "HIGH",
                        "event_name": name,
                        "principal": e.get("principal"),
                        "sg_id": params.get("groupId"),
                        "cidr": cidr,
                        "port_range": f"{port_from}-{port_to}",
                        "ts": e.get("ts") or raw.get("eventTime"),
                    })
                    break
    return alerts


def test_q48():
    events = [
        {
            "event_name": "AuthorizeSecurityGroupIngress",
            "principal": "admin",
            "ts": "T1",
            "raw": {
                "requestParameters": {
                    "groupId": "sg-12345",
                    "ipPermissions": {
                        "items": [{
                            "fromPort": 22, "toPort": 22,
                            "ipRanges": {"items": [{"cidrIp": "0.0.0.0/0"}]},
                        }]
                    }
                }
            },
        },
        {
            "event_name": "AuthorizeSecurityGroupIngress",
            "principal": "admin",
            "ts": "T2",
            "raw": {
                "requestParameters": {
                    "groupId": "sg-99999",
                    "ipPermissions": {
                        "items": [{
                            "fromPort": 443, "toPort": 443,
                            "ipRanges": {"items": [{"cidrIp": "10.0.0.0/8"}]},
                        }]
                    }
                }
            },
        },
    ]
    alerts = detect_sg_opened_to_internet(events)
    assert len(alerts) == 1
    assert alerts[0]["cidr"] == "0.0.0.0/0"
    assert alerts[0]["sg_id"] == "sg-12345"
    print("Q48 tests passed")

# Q49 — Detect CloudTrail StopLogging
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Flag any StopLogging or DeleteTrail event in CloudTrail.
  These events indicate an attacker trying to cover their tracks.

Clarifying questions:
  1. Should UpdateTrail (changing log destination) also alert?
  2. Should we alert immediately or batch with a short window?
  3. Is there a legitimate use case for StopLogging? (e.g., account deprecation)

Security angle:
  Attackers who gain sufficient IAM privileges will attempt to disable
  logging before proceeding with destructive actions. This is a classic
  defence evasion technique (MITRE ATT&CK T1562.008).
  AWS GuardDuty: Stealth:IAMUser/CloudTrailLoggingDisabled.
  Any StopLogging or DeleteTrail should be treated as CRITICAL.
"""

ANTI_LOGGING_EVENTS = frozenset({
    "StopLogging",
    "DeleteTrail",
    "UpdateTrail",
    "PutEventSelectors",          # can reduce what is logged
    "DeleteLogGroup",
    "DeleteLogStream",
})


def detect_cloudtrail_stop_logging(events: List[Dict]) -> List[Dict]:
    """
    Optimal: frozenset lookup per event. O(n) time.
    """
    alerts = []
    for e in events:
        name = e.get("event_name") or e.get("eventName") or ""
        if name in ANTI_LOGGING_EVENTS:
            alerts.append({
                "severity": "CRITICAL",
                "event_name": name,
                "principal": e.get("principal"),
                "ts":        e.get("ts") or e.get("eventTime"),
                "src_ip":    e.get("src_ip") or e.get("sourceIPAddress"),
            })
    return alerts


def test_q49():
    events = [
        {"event_name": "StopLogging",       "principal": "attacker", "ts": "T1", "src_ip": "5.5.5.5"},
        {"event_name": "DeleteTrail",        "principal": "attacker", "ts": "T2", "src_ip": "5.5.5.5"},
        {"event_name": "DescribeInstances",  "principal": "alice",    "ts": "T3", "src_ip": "1.1.1.1"},
    ]
    alerts = detect_cloudtrail_stop_logging(events)
    names = {a["event_name"] for a in alerts}
    assert "StopLogging" in names
    assert "DeleteTrail" in names
    assert "DescribeInstances" not in names
    print("Q49 tests passed")


# ─────────────────────────────────────────────────────────────────────────────
# Q50 — Detect S3 Bucket Public Policy Change
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Flag PutBucketPolicy or PutBucketAcl CloudTrail events that grant public
  access (Principal == "*" in the policy, or canned ACL "public-read" / "public-read-write").

Clarifying questions:
  1. Should we parse the full bucket policy JSON to confirm the Statement grants '*'?
  2. Should "authenticated users" (arn:aws:iam::*) also be flagged?
  3. Are Block Public Access settings captured in these events?

Security angle:
  Public S3 buckets have caused some of the largest data breaches in history
  (Capital One, GoDaddy, FBi/DHS). Detecting PutBucketPolicy that grants
  public access in real time is essential for any AWS environment.
  AWS Config rule s3-bucket-public-read-prohibited catches this at config
  evaluation time; CloudTrail detection catches it at the moment of change.
"""

PUBLIC_ACLS = frozenset({"public-read", "public-read-write", "authenticated-read"})


def _policy_grants_public(policy_json_str: str) -> bool:
    """Return True if any Statement has Principal == '*' or {'AWS': '*'}."""
    try:
        policy = json.loads(policy_json_str)
    except (json.JSONDecodeError, TypeError):
        return False
    for stmt in policy.get("Statement") or []:
        principal = stmt.get("Principal")
        if principal == "*":
            return True
        if isinstance(principal, dict):
            aws = principal.get("AWS") or principal.get("Service")
            if aws == "*" or aws == ["*"]:
                return True
    return False


def detect_s3_public_policy_change(events: List[Dict]) -> List[Dict]:
    """
    Optimal: filter on eventName, then inspect requestParameters for
    public ACL or parse embedded policy JSON.
    O(n) time; JSON parsing is bounded by policy size.
    """
    BUCKET_EVENTS = frozenset({"PutBucketPolicy", "PutBucketAcl"})
    alerts = []

    for e in events:
        name = e.get("event_name") or e.get("eventName") or ""
        if name not in BUCKET_EVENTS:
            continue
        raw = e.get("raw") or e
        params = raw.get("requestParameters") or {}
        bucket = params.get("bucketName") or params.get("BucketName")

        # Check canned ACL
        acl = (params.get("AccessControlPolicy") or {}).get("CannedACL") or \
              params.get("x-amz-acl") or ""
        if acl.lower() in PUBLIC_ACLS:
            alerts.append({
                "severity": "CRITICAL",
                "event_name": name,
                "bucket": bucket,
                "principal": e.get("principal"),
                "reason": f"public_acl:{acl}",
                "ts": e.get("ts"),
            })
            continue

        # Check bucket policy for wildcard principal
        policy_str = params.get("bucketPolicy") or params.get("policy") or ""
        if isinstance(policy_str, dict):
            policy_str = json.dumps(policy_str)
        if _policy_grants_public(policy_str):
            alerts.append({
                "severity": "CRITICAL",
                "event_name": name,
                "bucket": bucket,
                "principal": e.get("principal"),
                "reason": "policy_principal_wildcard",
                "ts": e.get("ts"),
            })

    return alerts


def test_q50():
    public_policy = json.dumps({
        "Version": "2012-10-17",
        "Statement": [{"Effect": "Allow", "Principal": "*", "Action": "s3:GetObject",
                       "Resource": "arn:aws:s3:::my-bucket/*"}],
    })
    events = [
        {
            "event_name": "PutBucketPolicy", "principal": "admin", "ts": "T1",
            "raw": {"requestParameters": {"bucketName": "my-bucket", "bucketPolicy": public_policy}},
        },
        {
            "event_name": "PutBucketAcl", "principal": "alice", "ts": "T2",
            "raw": {"requestParameters": {"bucketName": "my-bucket",
                                          "AccessControlPolicy": {"CannedACL": "public-read"}}},
        },
        {
            "event_name": "PutBucketPolicy", "principal": "bob", "ts": "T3",
            "raw": {"requestParameters": {"bucketName": "private-bucket",
                                          "bucketPolicy": json.dumps({
                                              "Statement": [{"Principal": {"AWS": "arn:aws:iam::123:root"}}]
                                          })}},
        },
    ]
    alerts = detect_s3_public_policy_change(events)
    assert len(alerts) == 2
    reasons = {a["reason"] for a in alerts}
    assert "policy_principal_wildcard" in reasons
    assert any("public_acl" in r for r in reasons)
    print("Q50 tests passed")
