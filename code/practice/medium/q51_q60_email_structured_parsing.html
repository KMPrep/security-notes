# Q51 — Parse Email Headers
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Parse a raw email header block (multi-line string) into a dict of header
  name → list of values. Handle folded headers (continuation lines starting
  with whitespace per RFC 5322).

Clarifying questions:
  1. Should we keep all values for multi-valued headers (e.g., multiple Received)?
  2. Should header names be case-normalised?
  3. Are MIME-encoded words (=?UTF-8?B?...?=) decoded?

Security angle:
  Email headers are the primary data source for phishing detection.
  Key headers: From, Reply-To, Return-Path, Received chain, DKIM-Signature,
  Authentication-Results. Mismatches between these headers reveal spoofing.
"""


def parse_email_headers(raw_headers: str) -> Dict[str, List[str]]:
    """
    Optimal: unfold continuation lines first, then split on 'Name: value' pattern.
    Header names are lowercased for consistent lookup.
    O(n) time where n = characters in header block.
    """
    # Unfold: join continuation lines (lines starting with whitespace)
    unfolded = re.sub(r"\r?\n[ \t]+", " ", raw_headers)

    headers: Dict[str, List[str]] = defaultdict(list)
    for line in unfolded.splitlines():
        if not line.strip():
            continue
        # Split on first colon only
        colon_pos = line.find(":")
        if colon_pos == -1:
            continue
        name = line[:colon_pos].strip().lower()
        value = line[colon_pos + 1:].strip()
        if name:
            headers[name].append(value)

    return dict(headers)


def test_q51():
    raw = (
        "From: Alice <alice@example.com>\r\n"
        "To: Bob <bob@example.com>\r\n"
        "Subject: Hello\r\n"
        "Received: from smtp1.example.com\r\n"
        "  by mx.example.com with ESMTP\r\n"
        "Received: from client.example.com\r\n"
        "  by smtp1.example.com\r\n"
        "DKIM-Signature: v=1; a=rsa-sha256; d=example.com\r\n"
    )
    headers = parse_email_headers(raw)
    assert headers["from"] == ["Alice <alice@example.com>"]
    assert len(headers["received"]) == 2
    assert "dkim-signature" in headers
    print("Q51 tests passed")

# Q52 — Count Email Received Hops
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given parsed email headers, count the number of mail relay hops (Received
  headers). Flag emails with an unusually high hop count (possible relay
  abuse or spam routing).

Clarifying questions:
  1. What is the normal hop count? (typical: 3-6 for legitimate email)
  2. Should we also extract the relay IP from each Received header?
  3. Should the first hop (from MTA to inbox) be excluded?

Security angle:
  Spammers and phishing campaigns often route email through many intermediate
  relays to obscure the true origin. A high hop count (>10) is suspicious.
  Open relay abuse chains can be detected by inspecting each Received line
  for internal vs external IPs.
"""


def count_email_hops(headers: Dict[str, List[str]], alert_threshold: int = 8) -> Dict:
    """
    Optimal: simply count the 'received' list length.
    O(h) time where h = number of Received headers.
    """
    received = headers.get("received") or []
    hop_count = len(received)
    return {
        "hop_count": hop_count,
        "alert": hop_count > alert_threshold,
    }


def test_q52():
    # 10 Received headers
    headers = {"received": [f"from relay{i}.example.com" for i in range(10)]}
    result = count_email_hops(headers, alert_threshold=8)
    assert result["hop_count"] == 10
    assert result["alert"] is True

    # Normal 4 hops
    headers2 = {"received": ["a", "b", "c", "d"]}
    result2 = count_email_hops(headers2, alert_threshold=8)
    assert result2["alert"] is False
    print("Q52 tests passed")

# Q53 — Detect SPF / DKIM / DMARC Failure
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Parse the Authentication-Results header and flag emails where SPF, DKIM,
  or DMARC did not pass.

Clarifying questions:
  1. Should partial passes (softfail, neutral) be flagged?
  2. Is DMARC alignment (strict vs relaxed) checked?
  3. Should we distinguish "fail" from "none" (not configured)?

Security angle:
  SPF, DKIM, and DMARC are the three layers of email authentication.
  Failing all three is a strong phishing indicator.
  Failing DMARC alone (with SPF pass) can indicate a lookalike domain.
  These checks are the basis for email security gateways and phishing
  detection systems at Google, Microsoft, and Proofpoint.
"""

AUTH_RESULT_RE = re.compile(
    r"\b(spf|dkim|dmarc)\s*=\s*(\S+)",
    re.IGNORECASE,
)


def detect_auth_failure(headers: Dict[str, List[str]]) -> Dict:
    """
    Optimal: scan Authentication-Results header(s) with one regex.
    O(v) time where v = total value length of Authentication-Results headers.
    """
    auth_results_values = headers.get("authentication-results") or []
    combined = " ".join(auth_results_values).lower()

    results = {}
    for m in AUTH_RESULT_RE.finditer(combined):
        mechanism = m.group(1).lower()
        result = m.group(2).lower().rstrip(";,")
        results[mechanism] = result

    failed = {
        k: v for k, v in results.items()
        if v not in ("pass",)
    }
    return {
        "spf":   results.get("spf", "none"),
        "dkim":  results.get("dkim", "none"),
        "dmarc": results.get("dmarc", "none"),
        "failures": failed,
        "alert": len(failed) > 0,
    }


def test_q53():
    headers = {
        "authentication-results": [
            "mx.example.com; spf=fail smtp.mailfrom=attacker@evil.com; "
            "dkim=none; dmarc=fail header.from=evil.com"
        ]
    }
    result = detect_auth_failure(headers)
    assert result["spf"] == "fail"
    assert result["dmarc"] == "fail"
    assert result["alert"] is True

    # All pass
    headers2 = {
        "authentication-results": [
            "mx.example.com; spf=pass; dkim=pass; dmarc=pass"
        ]
    }
    result2 = detect_auth_failure(headers2)
    assert result2["alert"] is False
    print("Q53 tests passed")

# Q54 — Extract Sender Domain from Header
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Extract the domain from the From: header email address.
  Handle both "Name <user@domain.com>" and "user@domain.com" formats.

Clarifying questions:
  1. Should we return the domain for multiple From: addresses?
  2. Should we normalise to lowercase?
  3. What if the From header is absent or malformed?

Security angle:
  The From: domain is compared against:
  - DMARC policy (header alignment).
  - SPF envelope sender.
  - Company domain allowlist (to catch lookalike domains like paypa1.com).
"""

EMAIL_DOMAIN_RE = re.compile(
    r"@([\w.-]+\.[a-zA-Z]{2,})",
    re.IGNORECASE,
)


def extract_sender_domain(headers: Dict[str, List[str]]) -> Optional[str]:
    """
    Optimal: regex on From header value to capture domain after @.
    O(1) — there is typically one From header.
    """
    from_values = headers.get("from") or []
    for val in from_values:
        m = EMAIL_DOMAIN_RE.search(val)
        if m:
            return m.group(1).lower()
    return None


def test_q54():
    h1 = {"from": ["Alice <alice@example.com>"]}
    assert extract_sender_domain(h1) == "example.com"

    h2 = {"from": ["bob@mail.corp.io"]}
    assert extract_sender_domain(h2) == "mail.corp.io"

    h3 = {"from": []}
    assert extract_sender_domain(h3) is None
    print("Q54 tests passed")


# ─────────────────────────────────────────────────────────────────────────────
# Q55 — Detect Mismatched From and Return-Path Domains
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Flag emails where the domain in the From: header differs from the domain
  in the Return-Path: header. This is a strong spoofing / phishing indicator.

Clarifying questions:
  1. Should subdomains be allowed? (e.g., From: @corp.com, Return-Path: @mail.corp.com)
  2. Should we compare eTLD+1 (registered domain) rather than full domain?
  3. What about mailing lists that legitimately rewrite Return-Path?

Security angle:
  Phishing emails spoof the From: to look legitimate but use a different
  Return-Path: that is under attacker control (for bounces). This mismatch
  is a core signal in email security gateways.
  Also called "envelope sender" vs "header sender" mismatch.
"""


def _extract_domain_from_header_value(value: str) -> Optional[str]:
    m = EMAIL_DOMAIN_RE.search(value)
    return m.group(1).lower() if m else None


def _registered_domain(domain: str) -> str:
    """Naive eTLD+1: return last two labels."""
    parts = domain.split(".")
    return ".".join(parts[-2:]) if len(parts) >= 2 else domain


def detect_from_returnpath_mismatch(
    headers: Dict[str, List[str]],
    strict: bool = False,
) -> bool:
    """
    Optimal: extract domains from both headers, compare.
    strict=False compares registered domains (eTLD+1) only.
    strict=True compares full domains.
    O(1) time.
    """
    from_vals = headers.get("from") or []
    rp_vals   = headers.get("return-path") or []

    if not from_vals or not rp_vals:
        return False

    from_domain = _extract_domain_from_header_value(from_vals[0])
    rp_domain   = _extract_domain_from_header_value(rp_vals[0])

    if not from_domain or not rp_domain:
        return False

    if strict:
        return from_domain != rp_domain
    return _registered_domain(from_domain) != _registered_domain(rp_domain)


def test_q55():
    # Mismatch: From corp.com, Return-Path attacker.net
    h1 = {
        "from":         ["CEO <ceo@corp.com>"],
        "return-path":  ["<bounce@attacker.net>"],
    }
    assert detect_from_returnpath_mismatch(h1) is True

    # Match: both corp.com
    h2 = {
        "from":         ["Alice <alice@corp.com>"],
        "return-path":  ["<bounce@mail.corp.com>"],
    }
    assert detect_from_returnpath_mismatch(h2) is False  # same eTLD+1

    # Strict mode: subdomain difference triggers alert
    assert detect_from_returnpath_mismatch(h2, strict=True) is True
    print("Q55 tests passed")

# Q56 — Parse CSV Security Events
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Parse a CSV string of security events into a list of dicts.
  Handle quoted fields, missing values, and optional header row.

Clarifying questions:
  1. Are column names in the first row?
  2. Should missing fields default to None or empty string?
  3. Should numeric fields (status, bytes) be auto-cast?

Security angle:
  Security operations teams often export SIEM alerts, firewall logs, or EDR
  events as CSV. A robust parser is the entry point for downstream detection
  logic. Injection attacks via CSV formulas (=CMD|...) are a real risk when
  these files are opened in Excel — a sanitiser should remove leading =, +, @.
"""

# Fields that should be coerced to int
INT_FIELDS = frozenset({"status", "bytes", "dst_port", "src_port", "count"})

FORMULA_INJECTION_RE = re.compile(r"^[=+\-@\t\r]")


def parse_csv_events(
    csv_text: str,
    sanitize: bool = True,
) -> List[Dict]:
    """
    Optimal: use stdlib csv.DictReader for correct quoting/escaping.
    Coerce numeric fields, optionally sanitise formula injection.
    O(n) time.
    """
    reader = csv.DictReader(io.StringIO(csv_text))
    records = []
    for row in reader:
        record: Dict = {}
        for k, v in row.items():
            if v is None:
                v = ""
            if sanitize and FORMULA_INJECTION_RE.match(v):
                v = "'" + v   # prepend apostrophe to neutralise
            k_lower = k.lower().strip()
            if k_lower in INT_FIELDS:
                try:
                    v = int(v)
                except (ValueError, TypeError):
                    pass
            record[k_lower] = v
        records.append(record)
    return records


def test_q56():
    csv_text = (
        "src_ip,dst_ip,dst_port,status,bytes\n"
        "1.1.1.1,10.0.0.1,443,200,1024\n"
        "2.2.2.2,10.0.0.1,22,401,0\n"
        '5.5.5.5,10.0.0.1,80,200,"=CMD|calc""!A1"\n'  # formula injection
    )
    records = parse_csv_events(csv_text, sanitize=True)
    assert len(records) == 3
    assert records[0]["dst_port"] == 443
    assert records[0]["bytes"] == 1024
    assert records[2]["status"] == 200             # status col is int-coerced
    # bytes field is the injected formula — sanitised (not int, starts with ')
    assert isinstance(records[2]["bytes"], str)
    assert records[2]["bytes"].startswith("'")
    assert not records[2]["src_ip"].startswith("=")
    print("Q56 tests passed")

# Q57 — Parse JSON Lines Logs
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Parse a JSON Lines (JSONL) log file where each line is a separate JSON object.
  Skip malformed lines without crashing.

Clarifying questions:
  1. Should malformed lines be returned with an error flag or silently skipped?
  2. Is there a schema to validate against?
  3. Should we handle BOM or trailing whitespace?

Security angle:
  Modern logging pipelines (Elastic, Splunk HEC, Cloud Logging) emit JSONL.
  A JSONL parser that crashes on one bad line can silently miss detections.
  Robust parsers skip bad lines, log a warning, and continue.
"""


def parse_jsonl_logs(
    jsonl_text: str,
    skip_errors: bool = True,
) -> Tuple[List[Dict], int]:
    """
    Optimal: iterate lines, attempt json.loads, track parse errors.
    Returns (records, error_count).
    O(n) time.
    """
    records = []
    errors = 0
    for line in jsonl_text.splitlines():
        line = line.strip()
        if not line:
            continue
        try:
            records.append(json.loads(line))
        except json.JSONDecodeError:
            errors += 1
            if not skip_errors:
                raise
    return records, errors


def test_q57():
    jsonl = (
        '{"ts": "T1", "event": "login", "user": "alice"}\n'
        '{"ts": "T2", "event": "logout", "user": "bob"}\n'
        'this is not json\n'
        '{"ts": "T3", "event": "delete", "user": "eve"}\n'
    )
    records, errors = parse_jsonl_logs(jsonl)
    assert len(records) == 3
    assert errors == 1
    assert records[0]["user"] == "alice"
    print("Q57 tests passed")

# Q58 — Parse Key-Value Firewall Logs
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Parse Cisco ASA / Palo Alto style key=value log lines into dicts.
  Example: "date=2026-05-29 time=08:00:00 src=1.1.1.1 dst=10.0.0.1 action=deny"

Clarifying questions:
  1. Are values ever quoted (key="multi word value")?
  2. Is the key=value separator always "=" with no spaces?
  3. Are there any repeated keys to handle?

Security angle:
  Firewall logs in KV format are the primary source for network-layer
  detection. Parsing must handle quoted values correctly to avoid splitting
  on spaces inside values (e.g., reason="Policy Violation").
"""

# Match key=value OR key="quoted value"
KV_RE = re.compile(r'(\w+)=(?:"([^"]*)"|([\S]*))')


def parse_kv_log(line: str) -> Dict[str, str]:
    """
    Optimal: compiled regex findall. Prefers quoted group over bare group.
    O(n) time where n = line length.
    """
    result = {}
    for m in KV_RE.finditer(line):
        key = m.group(1)
        value = m.group(2) if m.group(2) is not None else m.group(3)
        result[key] = value
    return result


def parse_kv_logs(lines: List[str]) -> List[Dict]:
    """Parse multiple KV log lines."""
    return [parse_kv_log(line) for line in lines if line.strip()]


def test_q58():
    lines = [
        'date=2026-05-29 time=08:00:01 src=1.1.1.1 dst=10.0.0.5 action=deny reason="Policy Violation"',
        'date=2026-05-29 time=08:00:02 src=2.2.2.2 dst=10.0.0.6 action=allow bytes=2048',
    ]
    records = parse_kv_logs(lines)
    assert records[0]["action"] == "deny"
    assert records[0]["reason"] == "Policy Violation"
    assert records[1]["bytes"] == "2048"
    assert records[1]["action"] == "allow"
    print("Q58 tests passed")

# Q59 — Normalise Mixed Timestamp Formats
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given a list of timestamp strings in mixed formats, normalise all of them
  to ISO 8601 UTC strings. Common formats:
  - 2026-05-29T08:00:00Z
  - 29/May/2026:08:00:00 +0000
  - May 29 2026 08:00:00
  - 1748505600 (Unix epoch)

Clarifying questions:
  1. Should timezone be preserved or converted to UTC?
  2. What should happen with unparseable timestamps?
  3. Should millisecond precision be kept?

Security angle:
  Log correlation (Q61–Q62) requires all timestamps to be in a comparable
  format. A timestamp normaliser is a core utility in any SIEM pipeline.
  Inconsistent timestamps are also a log tampering signal.
"""

from datetime import timezone

TIMESTAMP_FMTS = [
    "%Y-%m-%dT%H:%M:%SZ",
    "%Y-%m-%dT%H:%M:%S",
    "%Y-%m-%d %H:%M:%S",
    "%d/%b/%Y:%H:%M:%S %z",   # Apache/NGINX
    "%b %d %Y %H:%M:%S",
]


def normalise_timestamp(ts_str: str) -> Optional[str]:
    """
    Optimal: try epoch first (fast int check), then iterate format strings.
    Returns ISO 8601 UTC string or None on failure.
    O(F) time where F = number of format strings (constant).
    """
    ts_str = ts_str.strip()

    # Unix epoch (integer or float)
    try:
        epoch = float(ts_str)
        dt = datetime.fromtimestamp(epoch, tz=timezone.utc)
        return dt.strftime("%Y-%m-%dT%H:%M:%SZ")
    except ValueError:
        pass

    for fmt in TIMESTAMP_FMTS:
        try:
            dt = datetime.strptime(ts_str, fmt)
            if dt.tzinfo is None:
                dt = dt.replace(tzinfo=timezone.utc)
            else:
                dt = dt.astimezone(timezone.utc)
            return dt.strftime("%Y-%m-%dT%H:%M:%SZ")
        except ValueError:
            continue

    return None  # unparseable


def test_q59():
    samples = [
        ("2026-05-29T08:00:00Z",       "2026-05-29T08:00:00Z"),
        ("29/May/2026:08:00:00 +0000", "2026-05-29T08:00:00Z"),
        ("May 29 2026 08:00:00",       "2026-05-29T08:00:00Z"),
        ("1780041600",                  "2026-05-29T08:00:00Z"),
    ]
    for raw, expected in samples:
        result = normalise_timestamp(raw)
        assert result == expected, f"Got {result!r} for {raw!r}"
    assert normalise_timestamp("not-a-date") is None
    print("Q59 tests passed")

# Q60 — Deduplicate Logs by Event ID
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given a list of log event dicts, deduplicate by a unique event_id field,
  keeping the first occurrence and discarding subsequent duplicates.

Clarifying questions:
  1. What field is the unique identifier? (event_id, message_id, etc.)
  2. Should we keep the latest occurrence instead of the first?
  3. Should duplicates be counted and returned in a summary?

Security angle:
  Log pipelines often deliver events at-least-once (Kafka, Kinesis), leading
  to duplicate events. Without deduplication, detection rules may fire multiple
  times on the same event (alert storms, false positive counts). Deduplication
  by event ID is the standard approach in SIEM ingestion.
"""


def deduplicate_logs(
    events: List[Dict],
    id_field: str = "event_id",
    keep: str = "first",
) -> Tuple[List[Dict], int]:
    """
    Optimal: dict keyed by event_id preserves insertion order (Python 3.7+).
    For keep='last', overwrite; for keep='first', skip if already seen.
    Returns (deduped_list, duplicate_count).
    O(n) time.
    """
    seen: Dict = {}
    duplicates = 0

    for event in events:
        eid = event.get(id_field)
        if eid is None:
            seen[id(event)] = event   # no ID — keep all
        elif eid in seen:
            duplicates += 1
            if keep == "last":
                seen[eid] = event
        else:
            seen[eid] = event

    return list(seen.values()), duplicates


def test_q60():
    events = [
        {"event_id": "e1", "ts": "T1", "action": "login"},
        {"event_id": "e2", "ts": "T2", "action": "logout"},
        {"event_id": "e1", "ts": "T3", "action": "login"},   # duplicate
        {"event_id": "e3", "ts": "T4", "action": "delete"},
        {"event_id": "e2", "ts": "T5", "action": "logout"},  # duplicate
    ]
    deduped, dups = deduplicate_logs(events, keep="first")
    assert len(deduped) == 3
    assert dups == 2
    ids = [e["event_id"] for e in deduped]
    assert ids == ["e1", "e2", "e3"]

    # keep='last' should update ts
    deduped2, _ = deduplicate_logs(events, keep="last")
    id_ts = {e["event_id"]: e["ts"] for e in deduped2}
    assert id_ts["e1"] == "T3"
    assert id_ts["e2"] == "T5"
    print("Q60 tests passed")
