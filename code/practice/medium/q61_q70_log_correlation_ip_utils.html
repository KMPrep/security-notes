# Q61 — Merge Logs from Multiple Sources
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given multiple lists of log dicts (each with a "ts" timestamp field),
  merge them into a single time-sorted list.

Clarifying questions:
  1. Are all timestamps already in the same format (ISO 8601)?
  2. How many sources and how large is each log list?
  3. Should ties be broken by source priority?

Security angle:
  Log correlation requires events from different systems (firewall, auth,
  DNS) to be time-sorted. Merging is the prerequisite for building a
  coherent attack timeline. Attackers who tamper with timestamps will
  produce anomalous ordering.
"""

import heapq


def merge_logs_naive(log_sources: List[List[Dict]], ts_field: str = "ts") -> List[Dict]:
    """Concatenate all and sort by timestamp. O(N log N)."""
    combined = [e for src in log_sources for e in src]
    combined.sort(key=lambda e: e.get(ts_field) or "")
    return combined


def merge_logs(log_sources: List[List[Dict]], ts_field: str = "ts") -> List[Dict]:
    """
    Optimal: k-way merge using heapq.merge on pre-sorted per-source lists,
    or sort+merge if sources are not pre-sorted.
    If each source is already sorted (common in append-only logs),
    heapq.merge is O(N log k) vs O(N log N) for brute-force sort.
    Falls back to sort if sources are unsorted.
    O(N log k) time where k = number of sources, N = total events.
    """
    # heapq.merge requires each iterable to be sorted.
    # Sort each source first, then merge.
    sorted_sources = [
        sorted(src, key=lambda e: e.get(ts_field) or "")
        for src in log_sources
    ]
    return list(heapq.merge(*sorted_sources, key=lambda e: e.get(ts_field) or ""))


def test_q61():
    auth_logs = [
        {"src": "auth", "ts": "2026-05-29T08:00:01Z", "event": "login"},
        {"src": "auth", "ts": "2026-05-29T08:00:05Z", "event": "logout"},
    ]
    net_logs = [
        {"src": "fw",   "ts": "2026-05-29T08:00:02Z", "event": "allow"},
        {"src": "fw",   "ts": "2026-05-29T08:00:03Z", "event": "deny"},
    ]
    dns_logs = [
        {"src": "dns",  "ts": "2026-05-29T08:00:00Z", "event": "query"},
    ]
    merged = merge_logs([auth_logs, net_logs, dns_logs])
    assert len(merged) == 5
    timestamps = [e["ts"] for e in merged]
    assert timestamps == sorted(timestamps)
    assert merged[0]["event"] == "query"
    print("Q61 tests passed")

# Q62 — Correlate Auth Logs and Network Logs
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Join auth login events with network flow events on src_ip to find IPs that
  both logged in AND sent outbound traffic to external hosts in the same session.

Clarifying questions:
  1. What is the join key? (src_ip)
  2. What time window constitutes "same session"?
  3. Should we use an inner join (only IPs in both) or left join?

Security angle:
  Correlating auth + network lets you detect:
  - Data exfiltration after login.
  - Insider threat: authenticated user immediately transferring data.
  - Compromised account: login from new IP followed by bulk outbound traffic.
"""


def correlate_auth_network(
    auth_events: List[Dict],
    net_flows: List[Dict],
) -> List[Dict]:
    """
    Optimal: hash join — build a set of authenticated IPs from auth logs,
    then scan network flows for matching IPs.
    O(a + n) time where a = auth events, n = network flows.

    Data structure: dict keyed by src_ip for O(1) lookup.
    """
    # Build lookup: src_ip → auth event details
    authed: Dict[str, Dict] = {}
    for e in auth_events:
        ip = e.get("src_ip")
        if ip and e.get("result") == "success":
            authed[ip] = e

    correlations = []
    for flow in net_flows:
        ip = flow.get("src_ip")
        if ip in authed:
            correlations.append({
                "src_ip":      ip,
                "auth_user":   authed[ip].get("user"),
                "auth_ts":     authed[ip].get("ts"),
                "net_dst_ip":  flow.get("dst_ip"),
                "net_bytes":   flow.get("bytes_out"),
                "net_ts":      flow.get("ts"),
            })
    return correlations


def test_q62():
    auth = [
        {"src_ip": "10.0.0.5", "user": "alice", "result": "success", "ts": "T1"},
        {"src_ip": "10.0.0.6", "user": "bob",   "result": "failure", "ts": "T2"},
    ]
    flows = [
        {"src_ip": "10.0.0.5", "dst_ip": "203.0.113.50", "bytes_out": 50_000, "ts": "T3"},
        {"src_ip": "10.0.0.7", "dst_ip": "8.8.8.8",       "bytes_out": 1_000,  "ts": "T4"},
    ]
    results = correlate_auth_network(auth, flows)
    assert len(results) == 1
    assert results[0]["auth_user"] == "alice"
    assert results[0]["src_ip"] == "10.0.0.5"
    print("Q62 tests passed")

# Q63 — Join CloudTrail Events with IAM Metadata
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Enrich CloudTrail events by joining with an IAM metadata table on
  principal ARN, adding account ID, role name, and MFA status.

Clarifying questions:
  1. Is the join key the full ARN or just the username portion?
  2. Is the IAM metadata a dict (pre-loaded) or a second event list?
  3. Should events without a matching IAM record be kept or dropped?

Security angle:
  Raw CloudTrail events lack context: is this principal a human user, a
  service role, or a compromised credential? Enriching with IAM metadata
  lets analysts quickly assess privilege level and whether MFA was enrolled.
"""


def join_cloudtrail_iam(
    ct_events: List[Dict],
    iam_metadata: Dict[str, Dict],
) -> List[Dict]:
    """
    Optimal: IAM metadata keyed by principal ARN for O(1) lookup per event.
    O(n) time for enrichment pass.
    """
    enriched = []
    for event in ct_events:
        principal = event.get("principal") or ""
        meta = iam_metadata.get(principal) or {}
        enriched.append({
            **event,
            "account_id": meta.get("account_id"),
            "role_name":  meta.get("role_name"),
            "mfa_enrolled": meta.get("mfa_enrolled"),
            "is_admin":   meta.get("is_admin", False),
        })
    return enriched


def test_q63():
    events = [
        {"event_name": "GetSecretValue", "principal": "arn:aws:iam::123:user/alice", "ts": "T1"},
        {"event_name": "ListBuckets",    "principal": "arn:aws:iam::123:user/bob",   "ts": "T2"},
        {"event_name": "DescribeInstances", "principal": "arn:aws:iam::123:role/lambda", "ts": "T3"},
    ]
    iam = {
        "arn:aws:iam::123:user/alice": {"account_id": "123", "role_name": "admin",  "mfa_enrolled": True,  "is_admin": True},

    assert result[1]["mfa_enrolled"] is False
    assert result[2]["role_name"] is None   # no IAM metadata for lambda role
    print("Q63 tests passed")

# Q64 — Convert Flat Logs to Aggregated Summary
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given flat log events with src_ip and status_code, aggregate into a summary:
  per src_ip: total requests, distinct paths, error count, error rate.

Clarifying questions:
  1. What constitutes an "error"? (4xx, 5xx, or both?)
  2. What is the grouping key — only src_ip, or (src_ip, endpoint)?
  3. Should the output be sorted?

Security angle:
  Aggregation is the most common transformation before applying detection
  thresholds. Counting per-IP error rates is faster and more actionable
  than scanning raw logs for anomalies.
"""


def aggregate_logs(
    events: List[Dict],
    error_range: tuple = (400, 599),
) -> List[Dict]:
    """
    Optimal: single pass with per-IP accumulator dict.
    O(n) time, O(i) space where i = unique IPs.
    """
    stats: Dict[str, Dict] = defaultdict(
        lambda: {"total": 0, "errors": 0, "paths": set()}
    )

    for e in events:
        ip = e.get("src_ip") or "unknown"
        status = e.get("status") or 0
        path = (e.get("path") or "").split("?")[0]

        rec = stats[ip]
        rec["total"] += 1
        rec["paths"].add(path)
        if error_range[0] <= status <= error_range[1]:
            rec["errors"] += 1

    return sorted(
        [
            {
                "src_ip": ip,
                "total_requests": v["total"],
                "distinct_paths": len(v["paths"]),
                "error_count": v["errors"],
                "error_rate": round(v["errors"] / v["total"], 3) if v["total"] else 0,
            }
            for ip, v in stats.items()
        ],
        key=lambda r: -r["total_requests"],
    )


def test_q64():
    events = [
        {"src_ip": "1.1.1.1", "status": 200, "path": "/home"},
        {"src_ip": "1.1.1.1", "status": 404, "path": "/admin"},
        {"src_ip": "1.1.1.1", "status": 403, "path": "/secret"},
        {"src_ip": "2.2.2.2", "status": 200, "path": "/home"},
    ]
    results = aggregate_logs(events)
    by_ip = {r["src_ip"]: r for r in results}
    assert by_ip["1.1.1.1"]["total_requests"] == 3
    assert by_ip["1.1.1.1"]["error_count"] == 2
    assert abs(by_ip["1.1.1.1"]["error_rate"] - 0.667) < 0.001
    assert by_ip["2.2.2.2"]["error_count"] == 0
    print("Q64 tests passed")

# Q65 — Build IOC Matcher
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given a list of IOCs (Indicators of Compromise — IP addresses, domains,
  and file hashes), and a list of log events, return events that contain
  any matching IOC.

Clarifying questions:
  1. What IOC types are we matching? (IP, domain, MD5/SHA256)
  2. Should domain matching be exact or include subdomains?
  3. Should we return the matched IOC alongside the event?

Security angle:
  IOC matching is the backbone of threat intel integration. Every SIEM
  product applies threat intel feeds as lookup tables against log streams.
  The key insight: store IOCs in a frozenset/dict for O(1) lookup instead
  of iterating the full IOC list per event (which would be O(E * I)).
"""


def build_ioc_matcher(
    iocs: List[Dict],
) -> Dict[str, Set[str]]:
    """
    Build lookup sets per IOC type: ips, domains, hashes.
    O(i) time and space.
    """
    matcher: Dict[str, Set[str]] = {"ip": set(), "domain": set(), "hash": set()}
    for ioc in iocs:
        ioc_type = (ioc.get("type") or "").lower()
        value = (ioc.get("value") or "").lower()
        if ioc_type in matcher and value:
            matcher[ioc_type].add(value)
    # Freeze for immutability after build
    return {k: frozenset(v) for k, v in matcher.items()}


def match_iocs(
    events: List[Dict],
    matcher: Dict[str, Set[str]],
) -> List[Dict]:
    """
    Optimal: O(1) frozenset lookup per event field.
    O(E) time where E = number of events.
    """
    hits = []
    for e in events:
        matched_ioc = None
        matched_type = None

        src_ip = (e.get("src_ip") or "").lower()
        domain = (e.get("query") or e.get("domain") or "").lower()
        file_hash = (e.get("hash") or e.get("sha256") or "").lower()

        if src_ip in matcher.get("ip", frozenset()):
            matched_ioc, matched_type = src_ip, "ip"
        elif domain in matcher.get("domain", frozenset()):
            matched_ioc, matched_type = domain, "domain"
        elif file_hash in matcher.get("hash", frozenset()):
            matched_ioc, matched_type = file_hash, "hash"

        if matched_ioc:
            hits.append({**e, "ioc_match": matched_ioc, "ioc_type": matched_type})
    return hits


def test_q65():
    iocs = [
        {"type": "ip",     "value": "5.5.5.5"},
        {"type": "domain", "value": "evil-c2.example.org"},
        {"type": "hash",   "value": "aabbccdd1122334455667788990011aa"},
    ]
    events = [
        {"src_ip": "5.5.5.5",  "ts": "T1"},
        {"src_ip": "1.1.1.1",  "query": "evil-c2.example.org",          "ts": "T2"},
        {"src_ip": "2.2.2.2",  "hash": "aabbccdd1122334455667788990011aa", "ts": "T3"},
        {"src_ip": "9.9.9.9",  "ts": "T4"},  # no match
    ]
    m = build_ioc_matcher(iocs)
    results = match_iocs(events, m)
    assert len(results) == 3
    types = {r["ioc_type"] for r in results}
    assert types == {"ip", "domain", "hash"}
    assert not any(r["src_ip"] == "9.9.9.9" for r in results)
    print("Q65 tests passed")

# Q66 — Extract URLs from Text
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given a block of text (log message, email body, document), extract all URLs.

Clarifying questions:
  1. Should we extract only http/https, or also ftp, file, etc.?
  2. Should we extract URLs from inside HTML attributes (href, src)?
  3. Should we normalise to lowercase?

Security angle:
  URL extraction from log messages enables automated IOC enrichment:
  - Phishing URL detection from email body.
  - Malware download URL extraction from PCAP/proxy logs.
  - C2 URL extraction from memory dumps.
"""

URL_RE = re.compile(
    r"https?://[^\s\"'<>\])\}]+",
    re.IGNORECASE,
)


def extract_urls_naive(text: str) -> List[str]:
    return URL_RE.findall(text)


def extract_urls(text: str) -> List[str]:
    """
    Optimal: compiled regex findall. Deduplicate while preserving order.
    O(n) time where n = text length.
    """
    seen: Set[str] = set()
    results = []
    for url in URL_RE.findall(text):
        url_clean = url.rstrip(".,;:")  # strip common trailing punctuation
        if url_clean not in seen:
            seen.add(url_clean)
            results.append(url_clean)
    return results


def test_q66():
    text = (
        "User clicked http://evil.com/phish?token=abc123 and was redirected to "
        "https://legit.example.com/callback. The same URL http://evil.com/phish?token=abc123 "
        "appeared again in the next hop."
    )
    urls = extract_urls(text)
    assert "http://evil.com/phish?token=abc123" in urls
    assert "https://legit.example.com/callback" in urls
    # Deduplication
    assert len([u for u in urls if "evil.com" in u]) == 1
    print("Q66 tests passed")

# Q67 — Extract IPs from Text
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Extract all valid IPv4 and IPv6 addresses from a freeform text string.

Clarifying questions:
  1. Should we validate that the extracted IP is a valid address (not 999.999.999.999)?
  2. Should we deduplicate?
  3. Should private IPs be separated from public IPs?

Security angle:
  IP extraction is used in:
  - PCAP carving: extracting IP addresses from payload strings.
  - Email header parsing: extracting source IPs from Received headers.
  - Malware report scraping: extracting IOC IPs from threat intel PDFs.
"""

IPV4_RE = re.compile(
    r"\b(?:(?:25[0-5]|2[0-4]\d|[01]?\d\d?)\.){3}(?:25[0-5]|2[0-4]\d|[01]?\d\d?)\b"
)

IPV6_RE = re.compile(
    r"\b(?:[0-9a-fA-F]{1,4}:){7}[0-9a-fA-F]{1,4}\b"
    r"|\b(?:[0-9a-fA-F]{1,4}:){1,7}:\b"
    r"|\b::(?:[0-9a-fA-F]{1,4}:){0,6}[0-9a-fA-F]{1,4}\b"
)


def extract_ips(text: str, deduplicate: bool = True) -> Dict[str, List[str]]:
    """
    Optimal: separate compiled regexes for IPv4 and IPv6.
    Validates each match via ipaddress.ip_address.
    O(n) time.

    Data structure: two lists (v4, v6) to keep results typed.
    """
    v4: List[str] = []
    v6: List[str] = []
    seen: Set[str] = set()

    for match in IPV4_RE.finditer(text):
        ip = match.group(0)
        try:
            ipaddress.IPv4Address(ip)
            if not deduplicate or ip not in seen:
                v4.append(ip)
                seen.add(ip)
        except ValueError:
            pass

    for match in IPV6_RE.finditer(text):
        ip = match.group(0).strip(":")
        try:
            ipaddress.IPv6Address(ip)
            if not deduplicate or ip not in seen:
                v6.append(ip)
                seen.add(ip)
        except ValueError:
            pass

    return {"ipv4": v4, "ipv6": v6}


def test_q67():
    text = (
        "Connection from 192.168.1.5 to 10.0.0.1 on port 443. "
        "Also seen: 203.0.113.50 and duplicate 192.168.1.5. "
        "Invalid: 999.999.999.999"
    )
    result = extract_ips(text)
    assert "192.168.1.5" in result["ipv4"]
    assert "10.0.0.1" in result["ipv4"]
    assert "203.0.113.50" in result["ipv4"]
    # Deduplicated
    assert result["ipv4"].count("192.168.1.5") == 1
    # Invalid IP not included
    assert "999.999.999.999" not in result["ipv4"]
    print("Q67 tests passed")

# Q68 — Extract Hashes from Text
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Extract MD5, SHA1, and SHA256 hashes from a freeform text string.

Clarifying questions:
  1. Should we distinguish hash types by length?
  2. Should we validate that the hash contains only hex characters?
  3. Should hashes be returned normalised to lowercase?

Security angle:
  File hashes are the most portable IOC type — they survive IP/domain
  rotation. Hash extraction is used in:
  - Malware report parsing to build IOC feeds.
  - EDR log parsing to correlate with threat intel.
  - VirusTotal/CIRCL lookups during incident response.
"""

HASH_RE = re.compile(
    r"\b([0-9a-fA-F]{32}|[0-9a-fA-F]{40}|[0-9a-fA-F]{64})\b"
)

HASH_LEN_TYPE = {32: "md5", 40: "sha1", 64: "sha256"}


def extract_hashes(text: str) -> List[Dict[str, str]]:
    """
    Optimal: single regex that matches all three hash lengths.
    Type is inferred from length. Deduplicates.
    O(n) time.
    """
    seen: Set[str] = set()
    results = []
    for m in HASH_RE.finditer(text):
        h = m.group(0).lower()
        if h not in seen:
            seen.add(h)
            results.append({
                "hash": h,
                "type": HASH_LEN_TYPE.get(len(h), "unknown"),
            })
    return results


def test_q68():
    text = (
        "MD5: d41d8cd98f00b204e9800998ecf8427e "
        "SHA1: da39a3ee5e6b4b0d3255bfef95601890afd80709 "
        "SHA256: e3b0c44298fc1c149afbf4c8996fb92427ae41e4649b934ca495991b7852b855 "
        "Repeated: d41d8cd98f00b204e9800998ecf8427e"
    )
    results = extract_hashes(text)
    assert len(results) == 3   # deduplicated
    types = {r["type"] for r in results}
    assert types == {"md5", "sha1", "sha256"}
    print("Q68 tests passed")

# Q69 — Validate CIDR Membership
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given a list of CIDR blocks and an IP address, return whether the IP
  falls within any of the CIDR blocks.

Clarifying questions:
  1. Should we support both IPv4 and IPv6?
  2. Should overlapping CIDRs be de-duped before matching?
  3. Should we return WHICH CIDR matched?

Security angle:
  CIDR membership is used for:
  - Determining if a src_ip is internal vs external.
  - Allowlisting known corporate egress IPs.
  - Checking if a destination is in a cloud provider IP range (AWS/GCP/Azure).
  - Detecting traffic to/from geo-blocked IP ranges.
"""


def ip_in_cidrs(
    ip_str: str,
    cidrs: List[str],
) -> Optional[str]:
    """
    Optimal: use ipaddress stdlib for correct CIDR arithmetic.
    Returns matching CIDR string or None.
    O(c) time where c = number of CIDRs.
    """
    try:
        addr = ipaddress.ip_address(ip_str)
    except ValueError:
        return None

    for cidr in cidrs:
        try:
            network = ipaddress.ip_network(cidr, strict=False)
            if addr in network:
                return cidr
        except ValueError:
            continue
    return None


def filter_ips_by_cidr(
    ips: List[str],
    cidrs: List[str],
) -> Dict[str, Optional[str]]:
    """
    Check each IP against CIDR list. O(n * c) time.
    For large n, pre-build a sorted prefix tree; stdlib is fine for
    moderate-size IOC lists (< 10k CIDRs).
    """
    return {ip: ip_in_cidrs(ip, cidrs) for ip in ips}


def test_q69():
    cidrs = ["10.0.0.0/8", "192.168.0.0/16", "172.16.0.0/12"]
    assert ip_in_cidrs("10.1.2.3",     cidrs) == "10.0.0.0/8"
    assert ip_in_cidrs("192.168.5.10", cidrs) == "192.168.0.0/16"
    assert ip_in_cidrs("172.20.0.1",   cidrs) == "172.16.0.0/12"
    assert ip_in_cidrs("203.0.113.50", cidrs) is None
    print("Q69 tests passed")

# Q70 — Group IPs by /24 Subnet
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given a list of IP addresses, group them by their /24 subnet
  (i.e., first three octets) and return a dict of subnet → list of IPs.

Clarifying questions:
  1. IPv4 only or also IPv6 (/48 prefix)?
  2. Should we return sorted results?
  3. Should the subnet key include the /24 suffix?

Security angle:
  /24 grouping is used for:
  - Detecting distributed scanning from the same /24 (shared hosting attack).
  - Geo-blocking by IP range.
  - Identifying cloud provider allocations (e.g., AWS uses /18s allocated per region).
  - Correlating multiple attacker IPs that share infrastructure.
"""


def group_by_subnet24_naive(ips: List[str]) -> Dict[str, List[str]]:
    """Naive: take first three octets as key."""
    groups: Dict[str, List[str]] = defaultdict(list)
    for ip in ips:
        parts = ip.split(".")
        if len(parts) == 4:
            subnet = ".".join(parts[:3]) + ".0/24"
            groups[subnet].append(ip)
    return dict(groups)


def group_by_subnet24(ips: List[str]) -> Dict[str, List[str]]:
    """
    Optimal: use ipaddress.ip_network with supernet_of logic.
    Validates the IP, computes /24 via ipaddress for correctness.
    O(n) time.

    Data structure: defaultdict(list) keyed by subnet string.
    """
    groups: Dict[str, List[str]] = defaultdict(list)
    for ip_str in ips:
        try:
            addr = ipaddress.IPv4Address(ip_str)
            # Get the /24 network containing this address
            subnet = str(ipaddress.ip_network(f"{addr}/24", strict=False))
            groups[subnet].append(ip_str)
        except (ValueError, ipaddress.AddressValueError):
            groups["invalid"].append(ip_str)
    return dict(groups)
