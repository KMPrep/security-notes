# Q71 — Failed Logins in Rolling Window
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given a stream of login events (each with src_ip, user, result, ts),
  return users/IPs that exceeded N failed logins within a rolling T-second window.

Clarifying questions:
  1. Should we alert per-user, per-IP, or per (user, IP) pair?
  2. Should the window reset after each alert, or slide continuously?
  3. What is the expected volume? (determines if we need O(1) per event)

Security angle:
  Sliding window counting is the standard brute-force detection model used
  in WAFs, fail2ban, and SIEM correlation rules. A deque is ideal because:
  - Append is O(1).
  - Eviction of old events from the left is O(1).
  Combined complexity: O(n) total over all events.
"""


def failed_logins_in_window(
    events: List[Dict],
    threshold: int = 5,
    window_seconds: int = 60,
    group_by: str = "src_ip",
) -> List[Dict]:
    """
    Optimal: per-key deque sliding window.
    O(n) total time.
    """
    windows: Dict[str, deque] = defaultdict(deque)
    alerts: Dict[str, Dict] = {}

    for e in events:
        if e.get("result") != "failure":
            continue
        key = e.get(group_by) or "unknown"
        ts_raw = e.get("ts", "")
        try:
            ts = datetime.fromisoformat(ts_raw.rstrip("Z")).timestamp()
        except (ValueError, AttributeError):
            continue

        window = windows[key]
        window.append(ts)
        # Evict events outside the rolling window
        while window and ts - window[0] > window_seconds:
            window.popleft()

        if len(window) > threshold and key not in alerts:
            alerts[key] = {
                group_by: key,
                "failures_in_window": len(window),
                "window_seconds": window_seconds,
                "ts_first": datetime.fromtimestamp(window[0]).isoformat(),
                "ts_last": datetime.fromtimestamp(window[-1]).isoformat(),
            }
    return list(alerts.values())


def test_q71():
    base = datetime(2026, 5, 29, 8, 0, 0)
    events = []
    # IP 5.5.5.5: 6 failures in 30 seconds
    for i in range(6):
        events.append({
            "src_ip": "5.5.5.5", "result": "failure",
            "ts": (base + timedelta(seconds=i * 5)).isoformat(),
        })
    # IP 1.1.1.1: 3 failures spread over 5 minutes (below threshold)
    for i in range(3):
        events.append({
            "src_ip": "1.1.1.1", "result": "failure",
            "ts": (base + timedelta(seconds=i * 120)).isoformat(),
        })

    results = failed_logins_in_window(events, threshold=5, window_seconds=60)
    ips = {r["src_ip"] for r in results}
    assert "5.5.5.5" in ips
    assert "1.1.1.1" not in ips
    print("Q71 tests passed")

"""
Problem:

Clarifying questions:

  2. Useful for DDoS / scraping / API abuse detection.

Security angle:

def requests_per_ip_in_window(
    events: List[Dict],
    threshold: int = 100,
    window_seconds: int = 60,
) -> List[Dict]:
    """
    Optimal: same deque sliding window as Q71 but counts all events.
    O(n) total time.
    """
    windows: Dict[str, deque] = defaultdict(deque)
    alerts: Dict[str, Dict] = {}

    for e in events:
        ip = e.get("src_ip") or "unknown"
        ts_raw = e.get("ts", "")
        try:
            ts = datetime.fromisoformat(ts_raw.rstrip("Z")).timestamp()

                "src_ip": ip,

# Q73 — API Rate Limit Simulation
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Simulate an API gateway rate limiter: given a list of API requests with
  (api_key, ts), return which requests are allowed and which are throttled
  based on a per-key limit of N requests per T seconds.

Clarifying questions:
  1. Should the window be fixed (tumbling) or sliding?
  2. Should throttled requests be queued or dropped?
  3. Should the response include retry-after time?

Security angle:
  Rate limiting is a primary control against brute force, scraping, and
  API abuse. Simulating the rate limiter logic allows security testing.
  Overly permissive rate limits let attackers enumerate users or tokens.
"""


def simulate_rate_limiter(
    requests: List[Dict],
    limit: int = 10,
    window_seconds: int = 60,
) -> List[Dict]:
    """
    Optimal: sliding window per api_key using deque.
    Returns each request annotated with allowed/throttled.
    O(n) total time.
    """
    windows: Dict[str, deque] = defaultdict(deque)
    results = []

    for req in requests:
        key = req.get("api_key") or "anonymous"
        ts_raw = req.get("ts", "")
        try:
            ts = datetime.fromisoformat(ts_raw.rstrip("Z")).timestamp()
        except (ValueError, AttributeError):
            continue

        window = windows[key]
        # Evict expired timestamps
        while window and ts - window[0] >= window_seconds:
            window.popleft()

        if len(window) < limit:
            window.append(ts)
            results.append({**req, "status": "allowed"})
        else:
            retry_after = int(window_seconds - (ts - window[0])) + 1
            results.append({**req, "status": "throttled", "retry_after_s": retry_after})

    return results


def test_q73():
    base = datetime(2026, 5, 29, 8, 0, 0)
    reqs = []
    for i in range(15):
        reqs.append({
            "api_key": "key-abc",
            "ts": (base + timedelta(seconds=i * 3)).isoformat(),
        })

    results = simulate_rate_limiter(reqs, limit=10, window_seconds=60)
    allowed = [r for r in results if r["status"] == "allowed"]
    throttled = [r for r in results if r["status"] == "throttled"]
    assert len(allowed) == 10
    assert len(throttled) == 5
    assert all("retry_after_s" in r for r in throttled)
    print("Q73 tests passed")

# Q74 — Sliding Window Rate Limiter (Class)
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Implement a SlidingWindowRateLimiter class with an is_allowed(key) method
  that uses the current wall-clock time.

Clarifying questions:
  1. Is this for unit-testing or production use?
  2. Should the state persist across requests (in-memory vs Redis)?
  3. Is per-key granularity sufficient?

Security angle:
  A class-based limiter is the pattern used in API gateway middleware,
  Flask-Limiter, and AWS API Gateway throttle settings. The sliding window
  algorithm is more accurate than a fixed (tumbling) window because it
  doesn't suffer from the burst-at-window-boundary problem.
"""


class SlidingWindowRateLimiter:
    """
    Sliding window rate limiter.
    Allows at most `limit` requests per `window_seconds` per key.
    O(1) amortised per is_allowed call (deque eviction is amortised).
    """

    def __init__(self, limit: int, window_seconds: float):
        self.limit = limit
        self.window = window_seconds
        self._windows: Dict[str, deque] = defaultdict(deque)

    def is_allowed(self, key: str, now: Optional[float] = None) -> bool:
        ts = now if now is not None else time.time()
        window = self._windows[key]
        while window and ts - window[0] >= self.window:
            window.popleft()
        if len(window) < self.limit:
            window.append(ts)
            return True
        return False

    def reset(self, key: str) -> None:
        self._windows.pop(key, None)


def test_q74():
    limiter = SlidingWindowRateLimiter(limit=3, window_seconds=10)
    base_ts = 1000.0  # arbitrary base

    assert limiter.is_allowed("u1", now=base_ts + 0) is True
    assert limiter.is_allowed("u1", now=base_ts + 1) is True
    assert limiter.is_allowed("u1", now=base_ts + 2) is True
    assert limiter.is_allowed("u1", now=base_ts + 3) is False   # limit hit

    # After window expires, should be allowed again
    assert limiter.is_allowed("u1", now=base_ts + 11) is True   # first 3 expired

    # Different key is independent
    assert limiter.is_allowed("u2", now=base_ts + 0) is True
    print("Q74 tests passed")

# Q75 — Token Bucket Rate Limiter
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Implement a TokenBucketRateLimiter that refills at `rate` tokens/second
  up to a maximum `capacity`. Each request consumes one token.

Clarifying questions:
  1. Should fractional tokens be allowed (continuous refill)?
  2. Should the bucket be per-user or global?
  3. Is this for write-heavy or read-heavy APIs?

Security angle:
  Token bucket is the algorithm used by AWS API Gateway and many CDNs.
  Key advantage over sliding window: allows short bursts up to bucket
  capacity, which accommodates legitimate bursty users while still
  bounding peak rate over time.
  Abuse pattern: if an attacker gets >1 request/token through, the bucket
  leaks security guarantees.
"""


class TokenBucketRateLimiter:
    """
    Token bucket rate limiter (per-key).
    rate: tokens added per second.
    capacity: maximum token accumulation.
    O(1) per is_allowed call.
    """

    def __init__(self, rate: float, capacity: float):
        self.rate = rate
        self.capacity = capacity
        self._tokens: Dict[str, float] = defaultdict(lambda: capacity)
        self._last_refill: Dict[str, float] = {}

    def _refill(self, key: str, now: float) -> None:
        last = self._last_refill.get(key, now)
        elapsed = now - last
        self._tokens[key] = min(self.capacity, self._tokens[key] + elapsed * self.rate)
        self._last_refill[key] = now

    def is_allowed(self, key: str, now: Optional[float] = None) -> bool:
        ts = now if now is not None else time.time()
        self._refill(key, ts)
        if self._tokens[key] >= 1:
            self._tokens[key] -= 1
            return True
        return False


def test_q75():
    limiter = TokenBucketRateLimiter(rate=2.0, capacity=5.0)
    base = 1000.0

    # Burst: 5 requests immediately (full bucket)
    for i in range(5):
        assert limiter.is_allowed("u1", now=base + i * 0.01) is True

    # Bucket empty: next request denied
    assert limiter.is_allowed("u1", now=base + 0.1) is False

    # After 1 second, 2 tokens refilled
    assert limiter.is_allowed("u1", now=base + 1.1) is True
    assert limiter.is_allowed("u1", now=base + 1.2) is True
    assert limiter.is_allowed("u1", now=base + 1.3) is False
    print("Q75 tests passed")

# Q76 — Detect Alert Storm
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given a stream of SIEM alerts, detect "alert storms" — periods where the
  alert rate exceeds a threshold (e.g., >50 alerts in 60 seconds).

Clarifying questions:
  1. Should alert storms suppress further alerts, or just be flagged?
  2. Is the storm detection per-rule, per-host, or global?
  3. How should the storm window boundaries be defined?

Security angle:
  Alert storms indicate either:
  - A real active attack generating mass events.
  - A misconfigured detection rule causing false positive floods.
  Both need to be detected and handled. During a real attack, storm
  suppression can blind the SOC — alert management requires careful tuning.
"""


def detect_alert_storm(
    alerts: List[Dict],
    threshold: int = 50,
    window_seconds: int = 60,
    group_by: Optional[str] = None,
) -> List[Dict]:
    """
    Optimal: sliding window count per (optional) group key.
    Returns list of storm events with start/end timestamps.
    O(n) total time.
    """
    windows: Dict[str, deque] = defaultdict(deque)
    storms: Dict[str, bool] = {}
    storm_alerts = []

    for alert in alerts:
        key = alert.get(group_by, "global") if group_by else "global"
        ts_raw = alert.get("ts", "")
        try:
            ts = datetime.fromisoformat(ts_raw.rstrip("Z")).timestamp()
        except (ValueError, AttributeError):
            continue

        window = windows[key]
        window.append(ts)
        while window and ts - window[0] > window_seconds:
            window.popleft()

        count = len(window)
        if count > threshold and not storms.get(key):
            storms[key] = True
            storm_alerts.append({
                "key": key,
                "alert_count_in_window": count,
                "storm_start_ts": datetime.fromtimestamp(window[0]).isoformat(),
                "storm_detected_ts": datetime.fromtimestamp(ts).isoformat(),
            })
        elif count <= threshold:
            storms[key] = False

    return storm_alerts


def test_q76():
    base = datetime(2026, 5, 29, 8, 0, 0)
    alerts = []
    # Storm: 60 alerts in 30 seconds
    for i in range(60):
        alerts.append({
            "rule": "brute_force",
            "ts": (base + timedelta(milliseconds=i * 500)).isoformat(),
        })
    # Quiet period
    for i in range(5):
        alerts.append({
            "rule": "brute_force",
            "ts": (base + timedelta(seconds=120 + i * 10)).isoformat(),
        })

    results = detect_alert_storm(alerts, threshold=50, window_seconds=60)
    assert len(results) >= 1
    print("Q76 tests passed")

# Q77 — Detect Repeated Malware Hash
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given EDR file-scan events, flag any file hash seen on MORE THAN N hosts —
  indicating possible worm/malware spreading across the environment.

Clarifying questions:
  1. Should we count distinct hosts or total occurrences?
  2. Should we filter out known-good hashes from a whitelist?
  3. What is N? (default: 2 hosts = lateral spread indicator)

Security angle:
  A file hash appearing on 3+ hosts is a strong indicator of lateral movement
  via malware self-copying, malicious share/drive mapping, or supply-chain
  compromise. This is how CrowdStrike's "prevalence" score works — rare hashes
  (seen on few hosts) are treated as higher risk.
"""


def detect_repeated_malware_hash(
    events: List[Dict],
    min_hosts: int = 2,
    whitelist: Optional[Set[str]] = None,
) -> List[Dict]:
    """
    Optimal: defaultdict of sets — hash → set of host names.
    O(n) time, O(h) space where h = unique hashes.
    """
    hash_hosts: Dict[str, Set[str]] = defaultdict(set)
    whitelist = whitelist or set()

    for e in events:
        h = (e.get("hash") or e.get("sha256") or "").lower()
        host = e.get("host")
        if h and host and h not in whitelist:
            hash_hosts[h].add(host)

    return [
        {
            "hash": h,
            "host_count": len(hosts),
            "hosts": sorted(hosts),
        }
        for h, hosts in hash_hosts.items()
        if len(hosts) >= min_hosts
    ]


def test_q77():
    events = [
        {"host": "ws-01", "hash": "deadbeefdeadbeef01"},
        {"host": "ws-02", "hash": "deadbeefdeadbeef01"},   # same hash, 2nd host
        {"host": "ws-03", "hash": "deadbeefdeadbeef01"},   # 3rd host
        {"host": "ws-04", "hash": "cafecafecafecafe01"},   # unique hash
    ]
    results = detect_repeated_malware_hash(events, min_hosts=2)
    hashes = {r["hash"] for r in results}
    assert "deadbeefdeadbeef01" in hashes
    assert "cafecafecafecafe01" not in hashes
    assert next(r for r in results if r["hash"] == "deadbeefdeadbeef01")["host_count"] == 3
    print("Q77 tests passed")

# Q78 — Detect Lateral Movement Candidate
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Detect hosts that authenticate to multiple other internal hosts within a
  short time window — a hallmark of lateral movement (e.g., pass-the-hash,
  WMI exec, PsExec across multiple targets).

Clarifying questions:
  1. What authentication protocols are included? (SMB, WinRM, RDP, SSH)
  2. What is the minimum number of distinct targets to flag? (default: 3)
  3. What time window? (default: 10 minutes)

Security angle:
  MITRE ATT&CK TA0008 (Lateral Movement). A compromised host suddenly
  authenticating to 5+ peers is the primary detection signal for worm
  propagation and attacker pivoting through an internal network.
"""


def detect_lateral_movement(
    events: List[Dict],
    min_targets: int = 3,
    window_seconds: int = 600,
) -> List[Dict]:
    """
    Optimal: (src_host) → list of (ts, dst_host) sorted by ts.
    Sliding window of distinct dst_hosts.
    O(n log n) due to sort per host.
    """
    src_events: Dict[str, List[Tuple[float, str]]] = defaultdict(list)

    for e in events:
        src = e.get("src_host") or e.get("src_ip")
        dst = e.get("dst_host") or e.get("dst_ip")
        ts_raw = e.get("ts", "")
        if not (src and dst):
            continue
        try:
            ts = datetime.fromisoformat(ts_raw.rstrip("Z")).timestamp()
        except (ValueError, AttributeError):
            continue
        src_events[src].append((ts, dst))

    alerts = []
    for src, ts_dst_list in src_events.items():
        ts_dst_list.sort()
        window: deque = deque()   # deque of (ts, dst)
        distinct_dsts: Dict[str, int] = defaultdict(int)
        max_distinct = 0

        for ts, dst in ts_dst_list:
            window.append((ts, dst))
            distinct_dsts[dst] += 1

            while window and ts - window[0][0] > window_seconds:
                old_ts, old_dst = window.popleft()
                distinct_dsts[old_dst] -= 1
                if distinct_dsts[old_dst] == 0:
                    del distinct_dsts[old_dst]

            max_distinct = max(max_distinct, len(distinct_dsts))

        if max_distinct >= min_targets:
            alerts.append({
                "src_host": src,
                "max_distinct_targets": max_distinct,
                "window_seconds": window_seconds,
            })
    return alerts


def test_q78():
    base = datetime(2026, 5, 29, 8, 0, 0)
    events = []
    # Attacker lateral movement: ws-99 connects to 5 internal hosts
    for i, dst in enumerate(["host-01", "host-02", "host-03", "host-04", "host-05"]):
        events.append({
            "src_host": "ws-99", "dst_host": dst,
            "ts": (base + timedelta(seconds=i * 60)).isoformat(),
        })
    # Normal: ws-01 connects to 2 hosts over 2 hours
    events.append({"src_host": "ws-01", "dst_host": "print-server",
                   "ts": base.isoformat()})
    events.append({"src_host": "ws-01", "dst_host": "file-server",
                   "ts": (base + timedelta(hours=2)).isoformat()})

    results = detect_lateral_movement(events, min_targets=3, window_seconds=600)
    hosts = {r["src_host"] for r in results}
    assert "ws-99" in hosts
    assert "ws-01" not in hosts
    print("Q78 tests passed")


# ─────────────────────────────────────────────────────────────────────────────
# Q79 — Detect Privilege Escalation Sequence
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given a sequence of process/command events per user, flag users who execute
  a known privilege escalation sequence: e.g., within a session:
  1. whoami or id
  2. sudo -l  (list sudo privileges)
  3. sudo <command> or su - root

Clarifying questions:
  1. Should the events be ordered by timestamp within a session?
  2. Should all three steps be required, or any two?
  3. What is the session boundary? (same host, same user, within 10 min)

Security angle:
  This sequence is the classic Linux privilege escalation enumeration pattern.
  Red teams and attackers routinely run whoami → sudo -l → sudo bash.
  Detecting the ordered sequence reduces false positives compared to
  alerting on each command individually.
"""

ESCALATION_STEPS = [
    re.compile(r"\b(whoami|id)\b", re.IGNORECASE),
    re.compile(r"\bsudo\s+-l\b", re.IGNORECASE),
    re.compile(r"\b(sudo\s+\S+|su\s+-\s*root|su\s+-)\b", re.IGNORECASE),
]


def detect_privilege_escalation(
    events: List[Dict],
    require_all_steps: bool = True,
) -> List[Dict]:
    """
    Optimal: per-user ordered scan using state machine (step index).
    O(n) time.
    """
    user_step: Dict[str, int] = defaultdict(int)
    alerts = []

    # Sort by ts to handle out-of-order logs
    sorted_events = sorted(events, key=lambda e: e.get("ts", ""))

    for e in sorted_events:
        user = e.get("user") or e.get("username") or "unknown"
        cmd = e.get("command") or e.get("cmd") or ""
        step = user_step[user]

        if step < len(ESCALATION_STEPS) and ESCALATION_STEPS[step].search(cmd):
            user_step[user] += 1
            if user_step[user] == len(ESCALATION_STEPS):
                alerts.append({
                    "user": user,
                    "steps_completed": len(ESCALATION_STEPS),
                    "ts": e.get("ts"),
                })
                user_step[user] = 0   # reset for next sequence

    return alerts


def test_q79():
    events = [
        {"user": "mallory", "cmd": "whoami",    "ts": "T1"},
        {"user": "alice",   "cmd": "ls -la",    "ts": "T2"},
        {"user": "mallory", "cmd": "sudo -l",   "ts": "T3"},
        {"user": "mallory", "cmd": "sudo bash", "ts": "T4"},
        {"user": "alice",   "cmd": "sudo -l",   "ts": "T5"},   # only step 2
    ]
    results = detect_privilege_escalation(events)
    users = {r["user"] for r in results}
    assert "mallory" in users
    assert "alice" not in users
    print("Q79 tests passed")


# ─────────────────────────────────────────────────────────────────────────────
# Q80 — Detect Cloud API Abuse Sequence
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Detect an attacker recon sequence in CloudTrail events for a given
  principal: ListBuckets → GetBucketPolicy → GetObject (exfiltration).
  Flag principals that complete this ordered sequence within a time window.

Clarifying questions:
  1. Must all three steps occur in exact order?
  2. What is the time window (default: 5 minutes)?
  3. Should we track per-bucket or global across all buckets?

Security angle:
  This is the classic AWS S3 recon → exfiltration sequence:
  1. ListBuckets — enumerate all S3 buckets in the account.
  2. GetBucketPolicy — check which buckets are accessible.
  3. GetObject — exfiltrate data.
  AWS GuardDuty "Exfiltration:S3/MaliciousIPCaller" detects related signals.
  Tracking multi-step sequences requires maintaining per-principal state,
  which is the core of SIEM correlation rules.
"""

CLOUD_ABUSE_STEPS = ["ListBuckets", "GetBucketPolicy", "GetObject"]


def detect_cloud_api_abuse_sequence(
    events: List[Dict],
    window_seconds: int = 300,
) -> List[Dict]:
    """
    Optimal: per-principal state machine (step index + first step timestamp).
    O(n) time.
    """
    principal_state: Dict[str, Dict] = defaultdict(lambda: {"step": 0, "start_ts": None})
    alerts = []

    sorted_events = sorted(events, key=lambda e: e.get("ts", ""))

    for e in sorted_events:
        principal = e.get("principal") or "unknown"
        event_name = e.get("event_name") or ""
        ts_raw = e.get("ts", "")
        try:
            ts = datetime.fromisoformat(ts_raw.rstrip("Z")).timestamp()
        except (ValueError, AttributeError):
            continue

        state = principal_state[principal]
        expected_step = state["step"]

        if event_name == CLOUD_ABUSE_STEPS[expected_step]:
            if expected_step == 0:
                state["start_ts"] = ts
            else:
                # Check window
                if ts - state["start_ts"] > window_seconds:
                    # Timed out — restart from step 1
                    state["step"] = 0
                    state["start_ts"] = ts

            state["step"] += 1

            if state["step"] == len(CLOUD_ABUSE_STEPS):
                alerts.append({
                    "principal": principal,
                    "sequence": CLOUD_ABUSE_STEPS[:],
                    "sequence_duration_s": round(ts - state["start_ts"], 1),
                    "ts_completed": ts_raw,
                })
                state["step"] = 0
                state["start_ts"] = None

    return alerts
