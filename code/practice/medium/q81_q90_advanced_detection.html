# Q81 — Detect Suspicious Kubernetes API Sequence
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Detect a Kubernetes attacker recon sequence in audit logs:
  1. list/get Pods (enumerate workloads)
  2. exec into a Pod (interactive access)
  3. create a new Pod (persistence via backdoor)
  Flag principals who complete all three within a time window.

Clarifying questions:
  1. Is this based on k8s audit log verb + resource?
  2. What time window? (default: 10 minutes)
  3. Should we alert mid-sequence or only on completion?

Security angle:
  MITRE ATT&CK for Containers: T1613 (Container Discovery),
  T1609 (Container Admin Command / exec), T1610 (Deploy Container).
  The sequence strongly indicates an attacker who compromised a k8s service
  account or stole a kubeconfig and is escalating inside the cluster.
"""

K8S_STEPS = [
    {"verb": frozenset({"list", "get"}), "resource": "pods"},
    {"verb": frozenset({"create"}),      "resource": "pods/exec"},
    {"verb": frozenset({"create"}),      "resource": "pods"},
]


def detect_k8s_attack_sequence(
    events: List[Dict],
    window_seconds: int = 600,
) -> List[Dict]:
    """
    Optimal: per-principal state machine matching ordered K8S_STEPS.
    O(n) time.
    """
    state: Dict[str, Dict] = defaultdict(lambda: {"step": 0, "start_ts": None})
    alerts = []
    sorted_events = sorted(events, key=lambda e: e.get("ts", ""))

    for e in sorted_events:
        principal = e.get("principal") or e.get("user") or "unknown"
        verb = (e.get("verb") or "").lower()
        resource = (e.get("resource") or "").lower()
        ts_raw = e.get("ts", "")
        try:
            ts = datetime.fromisoformat(ts_raw.rstrip("Z")).timestamp()
        except (ValueError, AttributeError):
            continue

        s = state[principal]
        expected = K8S_STEPS[s["step"]]

        if verb in expected["verb"] and resource == expected["resource"]:
            if s["step"] == 0:
                s["start_ts"] = ts
            else:
                if ts - s["start_ts"] > window_seconds:
                    s["step"] = 0
                    s["start_ts"] = ts
                    continue
            s["step"] += 1
            if s["step"] == len(K8S_STEPS):
                alerts.append({
                    "principal": principal,
                    "sequence": [f"{st['verb']} {st['resource']}" for st in K8S_STEPS],
                    "duration_s": round(ts - s["start_ts"], 1),
                    "ts_completed": ts_raw,
                })
                s["step"] = 0
                s["start_ts"] = None

    return alerts


def test_q81():
    base = datetime(2026, 5, 29, 8, 0, 0)
    events = [
        {"principal": "eve",   "verb": "list",   "resource": "pods",      "ts": (base + timedelta(seconds=0)).isoformat()},
        {"principal": "eve",   "verb": "create", "resource": "pods/exec", "ts": (base + timedelta(seconds=30)).isoformat()},
        {"principal": "eve",   "verb": "create", "resource": "pods",      "ts": (base + timedelta(seconds=90)).isoformat()},
        {"principal": "alice", "verb": "get",    "resource": "pods",      "ts": (base + timedelta(seconds=5)).isoformat()},
        {"principal": "alice", "verb": "list",   "resource": "services",  "ts": (base + timedelta(seconds=10)).isoformat()},
    ]
    results = detect_k8s_attack_sequence(events, window_seconds=600)
    principals = {r["principal"] for r in results}
    assert "eve" in principals
    assert "alice" not in principals
    print("Q81 tests passed")

# Q82 — Detect Repeated 401 Then 200 (Brute Force Success)
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  From HTTP access logs, detect any IP/user that had N consecutive 401
  responses followed by a 200 — indicating a successful brute force login.

Clarifying questions:
  1. Must the 401s be strictly consecutive, or just the majority in a window?
  2. Should we track per-IP, per-username, or per (IP, path)?
  3. What is N? (default: 5)

Security angle:
  "Low-and-slow" brute force often flies under rate-limit thresholds. Spotting
  5+ 401s followed by a 200 on a login endpoint is highly actionable: the
  account is now compromised. Many SIEMs call this "authentication success
  after failure" (ASAF). It's also the primary pattern for detecting
  credential-stuffing success.
"""


def detect_401_then_200(
    logs: List[Dict],
    failure_threshold: int = 5,
    group_by: str = "src_ip",
) -> List[Dict]:
    """
    Optimal: per-key counter of consecutive 401s reset on non-401.
    Flag on 200 following ≥ failure_threshold 401s.
    O(n) time.
    """
    consecutive_failures: Dict[str, int] = defaultdict(int)
    alerts = []

    for log in logs:
        key = log.get(group_by) or "unknown"
        status = log.get("status") or log.get("status_code")
        try:
            status = int(status)
        except (TypeError, ValueError):
            continue

        if status == 401:
            consecutive_failures[key] += 1
        elif status == 200:
            if consecutive_failures[key] >= failure_threshold:
                alerts.append({
                    group_by: key,
                    "failures_before_success": consecutive_failures[key],
                    "ts": log.get("ts"),
                })
            consecutive_failures[key] = 0
        else:
            consecutive_failures[key] = 0

    return alerts


def test_q82():
    logs = [
        {"src_ip": "1.2.3.4", "status": 401, "ts": "T1"},
        {"src_ip": "1.2.3.4", "status": 401, "ts": "T2"},
        {"src_ip": "1.2.3.4", "status": 401, "ts": "T3"},
        {"src_ip": "1.2.3.4", "status": 401, "ts": "T4"},
        {"src_ip": "1.2.3.4", "status": 401, "ts": "T5"},
        {"src_ip": "1.2.3.4", "status": 200, "ts": "T6"},   # brute force success
        {"src_ip": "9.9.9.9", "status": 401, "ts": "T7"},
        {"src_ip": "9.9.9.9", "status": 401, "ts": "T8"},
        {"src_ip": "9.9.9.9", "status": 200, "ts": "T9"},   # only 2 failures → OK
    ]
    results = detect_401_then_200(logs, failure_threshold=5)
    ips = {r["src_ip"] for r in results}
    assert "1.2.3.4" in ips
    assert "9.9.9.9" not in ips
    print("Q82 tests passed")


# ─────────────────────────────────────────────────────────────────────────────
# Q83 — Detect Slow Brute Force
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Detect a "slow" brute force — N failed logins from the same IP spread over
  a long time window (e.g., 24 hours) but never in a short burst.

Clarifying questions:
  1. Is this per-user or per-IP?
  2. What is the minimum failure count and window? (default: 20 in 86400s)
  3. Should we suppress if there's also a burst alert (Q71)?

Security angle:
  Attackers deliberately slow their attempts to evade short-window rate limits.
  Detecting over a 24-hour window catches these while a 60-second window misses
  them. Low-and-slow brute force is a growing technique against MFA-bypass
  endpoints like OTP windows.
"""


def detect_slow_brute_force(
    events: List[Dict],
    min_failures: int = 20,
    window_hours: int = 24,
    group_by: str = "src_ip",
) -> List[Dict]:
    """
    Optimal: deque sliding window over a long period.
    Same O(n) algorithm as Q71 but with a 24-hour window.
    """
    window_seconds = window_hours * 3600
    windows: Dict[str, deque] = defaultdict(deque)
    alerts: Dict[str, Dict] = {}

    for e in events:
        if e.get("result") != "failure":
            continue
        key = e.get(group_by) or "unknown"
        ts_raw = e.get("ts", "")
        try:
            ts = datetime.fromisoformat(ts_raw.rstrip("Z")).timestamp()
        except (ValueError, AttributeError):
            continue

        window = windows[key]
        window.append(ts)
        while window and ts - window[0] > window_seconds:
            window.popleft()

        if len(window) >= min_failures and key not in alerts:
            alerts[key] = {
                group_by: key,
                "failures_in_window": len(window),
                "window_hours": window_hours,
            }

    return list(alerts.values())


def test_q83():
    base = datetime(2026, 5, 29, 0, 0, 0)
    events = []
    # Slow brute force: 25 failures spread over 20 hours
    for i in range(25):
        events.append({
            "src_ip": "5.5.5.5", "result": "failure",
            "ts": (base + timedelta(hours=i * 0.8)).isoformat(),
        })
    # Normal: 5 failures in 24 hours
    for i in range(5):
        events.append({
            "src_ip": "1.1.1.1", "result": "failure",
            "ts": (base + timedelta(hours=i * 4)).isoformat(),
        })

    results = detect_slow_brute_force(events, min_failures=20, window_hours=24)
    ips = {r["src_ip"] for r in results}
    assert "5.5.5.5" in ips
    assert "1.1.1.1" not in ips
    print("Q83 tests passed")

# Q84 — Detect Session Hijacking Candidate
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given session activity logs (session_id, src_ip, user_agent, ts), flag
  sessions where the src_ip or user_agent changes mid-session — a candidate
  for session hijacking or cookie theft.

Clarifying questions:
  1. How many unique IPs / UAs before alerting? (default: >1 distinct IP)
  2. Should we ignore CDN/proxy IP changes?
  3. What about mobile users switching networks?

Security angle:
  Session hijacking (OWASP A07): stolen cookies used from a different IP.
  User-agent changes mid-session indicate the attacker's browser differs from
  the victim's. Both are weak signals individually; combined they're strong.
  This is how Gmail detects "sign-in from new location during session."
"""


def detect_session_hijacking(
    logs: List[Dict],
    max_ips: int = 1,
    check_user_agent: bool = True,
) -> List[Dict]:
    """
    Optimal: per-session sets of IPs and user_agents.
    O(n) time.
    """
    session_ips: Dict[str, Set[str]] = defaultdict(set)
    session_uas: Dict[str, Set[str]] = defaultdict(set)

    for log in logs:
        sid = log.get("session_id") or "unknown"
        ip = log.get("src_ip") or ""
        ua = log.get("user_agent") or ""
        if ip:
            session_ips[sid].add(ip)
        if ua:
            session_uas[sid].add(ua)

    alerts = []
    all_sids = set(session_ips) | set(session_uas)
    for sid in all_sids:
        ips = session_ips.get(sid, set())
        uas = session_uas.get(sid, set())
        reasons = []
        if len(ips) > max_ips:
            reasons.append(f"IP changed: {sorted(ips)}")
        if check_user_agent and len(uas) > 1:
            reasons.append(f"UA changed: {sorted(uas)}")
        if reasons:
            alerts.append({
                "session_id": sid,
                "distinct_ips": len(ips),
                "distinct_uas": len(uas),
                "reasons": reasons,
            })

    return alerts


def test_q84():
    logs = [
        {"session_id": "s1", "src_ip": "1.1.1.1", "user_agent": "Chrome/100"},
        {"session_id": "s1", "src_ip": "9.9.9.9", "user_agent": "Firefox/99"},  # hijack
        {"session_id": "s2", "src_ip": "2.2.2.2", "user_agent": "Chrome/100"},
        {"session_id": "s2", "src_ip": "2.2.2.2", "user_agent": "Chrome/100"},  # normal
    ]
    results = detect_session_hijacking(logs)
    sids = {r["session_id"] for r in results}
    assert "s1" in sids
    assert "s2" not in sids
    print("Q84 tests passed")

# Q85 — Detect Token Reuse After Logout
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given auth events (user, action, token_id, ts), detect if a token is used
  for API calls AFTER a logout event has been recorded for that token.

Clarifying questions:
  1. Is the token revoked server-side, or only client-side on logout?
  2. Should we track per-token or per-user?
  3. Are tokens JWTs (stateless) or opaque (server-side revocable)?

Security angle:
  JWT tokens are stateless: if the server doesn't maintain a revocation list,
  a stolen token can be reused even after the legitimate user logs out.
  Detecting post-logout token use reveals either:
  - Stolen/replayed tokens (attacker reuse).
  - Misconfigured logout (client-only invalidation).
  This maps to OWASP A07 (Auth Failures) and CWE-613 (Insufficient Session Expiration).
"""


def detect_token_reuse_after_logout(events: List[Dict]) -> List[Dict]:
    """
    Optimal: single pass — record logout timestamp per token,
    then flag any use event with ts > logout_ts.
    Requires two passes if events are unordered.
    O(n) time (two passes).
    """
    sorted_events = sorted(events, key=lambda e: e.get("ts", ""))

    logout_ts: Dict[str, float] = {}
    alerts = []

    # Pass 1: collect logout timestamps
    for e in sorted_events:
        if e.get("action") == "logout":
            token = e.get("token_id") or e.get("token") or ""
            ts_raw = e.get("ts", "")
            try:
                ts = datetime.fromisoformat(ts_raw.rstrip("Z")).timestamp()
                logout_ts[token] = ts
            except (ValueError, AttributeError):
                pass

    # Pass 2: detect use after logout
    for e in sorted_events:
        action = e.get("action") or ""
        if action in ("logout", "login"):
            continue
        token = e.get("token_id") or e.get("token") or ""
        ts_raw = e.get("ts", "")
        if token in logout_ts:
            try:
                ts = datetime.fromisoformat(ts_raw.rstrip("Z")).timestamp()
            except (ValueError, AttributeError):
                continue
            if ts > logout_ts[token]:
                alerts.append({
                    "token_id": token,
                    "user": e.get("user"),
                    "action": action,
                    "ts": ts_raw,
                    "logout_ts": datetime.fromtimestamp(logout_ts[token]).isoformat(),
                })

    return alerts


def test_q85():
    base = datetime(2026, 5, 29, 8, 0, 0)
    events = [
        {"user": "alice", "action": "login",       "token_id": "tok-1", "ts": (base + timedelta(seconds=0)).isoformat()},
        {"user": "alice", "action": "api_call",    "token_id": "tok-1", "ts": (base + timedelta(seconds=30)).isoformat()},
        {"user": "alice", "action": "logout",      "token_id": "tok-1", "ts": (base + timedelta(seconds=60)).isoformat()},
        # Attacker reuses token after logout
        {"user": "unknown", "action": "api_call",  "token_id": "tok-1", "ts": (base + timedelta(seconds=120)).isoformat()},
        # Normal: different token
        {"user": "bob",   "action": "api_call",    "token_id": "tok-2", "ts": (base + timedelta(seconds=90)).isoformat()},
    ]
    results = detect_token_reuse_after_logout(events)
    tokens = {r["token_id"] for r in results}
    assert "tok-1" in tokens
    assert "tok-2" not in tokens
    print("Q85 tests passed")

# Q86 — Detect API Key Used from New IP
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given a baseline of (api_key → known IPs) and a stream of new API events,
  flag API keys used from an IP not in their baseline.

Clarifying questions:
  1. How is the baseline established? (30-day lookback, static config)
  2. Should we alert once per new IP or every time?
  3. Should we tolerate IPs in the same /24 subnet?

Security angle:
  API keys are long-lived credentials. Unlike OAuth tokens, they rarely rotate.
  A key suddenly used from a new IP may indicate:
  - Credential leakage (GitHub secret scanning scenario).
  - Stolen from a developer laptop or CI/CD pipeline.
  This is analogous to "impossible travel" for API keys.
"""


def detect_api_key_from_new_ip(
    events: List[Dict],
    baseline: Optional[Dict[str, Set[str]]] = None,
) -> List[Dict]:
    """
    Optimal: frozenset baseline lookup.
    O(n) time per stream event.
    """
    baseline = baseline or {}
    # Convert lists to sets if needed
    frozen_baseline: Dict[str, frozenset] = {
        k: frozenset(v) for k, v in baseline.items()
    }
    seen_alerts: Set[Tuple[str, str]] = set()
    alerts = []

    for e in events:
        key = e.get("api_key") or ""
        ip = e.get("src_ip") or ""
        if not (key and ip):
            continue

        known_ips = frozen_baseline.get(key, frozenset())
        if ip not in known_ips:
            alert_key = (key, ip)
            if alert_key not in seen_alerts:
                seen_alerts.add(alert_key)
                alerts.append({
                    "api_key": key,
                    "new_ip": ip,
                    "ts": e.get("ts"),
                })

    return alerts


def test_q86():
    baseline = {
        "key-prod-1": {"10.0.0.1", "10.0.0.2"},
        "key-dev-2":  {"192.168.1.5"},
    }
    events = [
        {"api_key": "key-prod-1", "src_ip": "10.0.0.1",   "ts": "T1"},   # known
        {"api_key": "key-prod-1", "src_ip": "185.220.0.1", "ts": "T2"},   # new!
        {"api_key": "key-dev-2",  "src_ip": "192.168.1.5", "ts": "T3"},   # known
        {"api_key": "key-dev-2",  "src_ip": "8.8.8.8",    "ts": "T4"},   # new!
    ]
    results = detect_api_key_from_new_ip(events, baseline)
    keys = [r["api_key"] for r in results]
    assert "key-prod-1" in keys
    assert "key-dev-2" in keys
    assert len(results) == 2
    print("Q86 tests passed")


# ─────────────────────────────────────────────────────────────────────────────
# Q87 — Detect Sudden Increase in Error Rate
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given a stream of HTTP events grouped by minute, detect when the error rate
  (4xx + 5xx) in a minute exceeds a multiplier (e.g., 3×) compared to the
  rolling baseline of the previous N minutes.

Clarifying questions:
  1. What is N for the baseline window? (default: 5 minutes)
  2. What multiplier constitutes a "spike"? (default: 3×)
  3. Should we group by endpoint, or globally?

Security angle:
  A sudden 3× spike in error rate can indicate:
  - Active attack (scanning for valid paths → 404 flood).
  - WAF misfire (new rule causing 403 flood).
  - DDoS causing 5xx cascade.
  Automated anomaly detection on error rates is core to SRE/SoC integration.
"""


def _minute_bucket(ts_raw: str) -> Optional[str]:
    try:
        dt = datetime.fromisoformat(ts_raw.rstrip("Z"))
        return dt.strftime("%Y-%m-%dT%H:%M")
    except (ValueError, AttributeError):
        return None


def detect_error_rate_spike(
    events: List[Dict],
    baseline_minutes: int = 5,
    spike_multiplier: float = 3.0,
    min_requests: int = 10,
) -> List[Dict]:
    """
    Optimal: aggregate into minute buckets, deque baseline.
    O(n) time.
    """
    bucket_total: Dict[str, int] = defaultdict(int)
    bucket_errors: Dict[str, int] = defaultdict(int)

    for e in events:
        bucket = _minute_bucket(e.get("ts", ""))
        if not bucket:
            continue
        bucket_total[bucket] += 1
        status = e.get("status") or 0
        try:
            if int(status) >= 400:
                bucket_errors[bucket] += 1
        except (TypeError, ValueError):
            pass

    sorted_buckets = sorted(bucket_total)
    baseline: deque = deque()   # recent (total, errors) tuples
    alerts = []

    for bucket in sorted_buckets:
        total = bucket_total[bucket]
        errors = bucket_errors.get(bucket, 0)
        error_rate = errors / total if total else 0.0

        if baseline:
            baseline_total = sum(b[0] for b in baseline)
            baseline_errors = sum(b[1] for b in baseline)
            baseline_rate = baseline_errors / baseline_total if baseline_total else 0.0
            if (total >= min_requests and baseline_rate > 0
                    and error_rate >= spike_multiplier * baseline_rate):
                alerts.append({
                    "minute": bucket,
                    "error_rate": round(error_rate, 3),
                    "baseline_rate": round(baseline_rate, 3),
                    "multiplier": round(error_rate / baseline_rate, 2),
                })

        baseline.append((total, errors))
        if len(baseline) > baseline_minutes:
            baseline.popleft()

    return alerts


def test_q87():
    base = datetime(2026, 5, 29, 8, 0, 0)
    events = []
    # 5 quiet minutes: 100 requests, 5% error rate
    for minute in range(5):
        for i in range(100):
            status = 500 if i < 5 else 200
            events.append({
                "status": status,
                "ts": (base + timedelta(minutes=minute, seconds=i % 60)).isoformat(),
            })
    # Spike minute: 100 requests, 60% error rate
    for i in range(100):
        status = 500 if i < 60 else 200
        events.append({
            "status": status,
            "ts": (base + timedelta(minutes=5, seconds=i % 60)).isoformat(),
        })

    results = detect_error_rate_spike(events, baseline_minutes=5, spike_multiplier=3.0)
    assert len(results) >= 1
    spike = results[-1]
    assert spike["error_rate"] > 0.5
    print("Q87 tests passed")


# ─────────────────────────────────────────────────────────────────────────────
# Q88 — Detect Traffic Spike by Endpoint
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given HTTP logs, detect per-endpoint traffic spikes: when requests to an
  endpoint in a 1-minute window exceed N× the rolling average of previous
  M minutes for that endpoint.

Clarifying questions:
  1. Should paths be normalised (e.g., /user/123 → /user/{id})?
  2. What constitutes M and N? (default: M=5, N=5×)
  3. Should bots be excluded?

Security angle:
  Endpoint-specific spikes indicate:
  - Targeted DDoS on a specific microservice.
  - Automated enumeration of a specific resource.
  - Viral traffic to an innocuous endpoint (false positive).
  Detecting per-endpoint spikes is more precise than global rate monitoring
  and directly informs WAF rule tuning.
"""


def detect_traffic_spike_by_endpoint(
    events: List[Dict],
    baseline_minutes: int = 5,
    spike_multiplier: float = 5.0,
    min_requests: int = 5,
) -> List[Dict]:
    """
    Optimal: per-endpoint deque baseline of minute counts.
    O(n) time, O(endpoints × baseline_minutes) space.
    """
    # bucket → endpoint → count
    from collections import Counter
    bucket_endpoint: Dict[str, Counter] = defaultdict(Counter)

    for e in events:
        bucket = _minute_bucket(e.get("ts", ""))
        path = e.get("path") or "/"
        if bucket:
            bucket_endpoint[bucket][path] += 1

    sorted_buckets = sorted(bucket_endpoint)
    baselines: Dict[str, deque] = defaultdict(deque)
    alerts = []

    for bucket in sorted_buckets:
        for path, count in bucket_endpoint[bucket].items():
            baseline = baselines[path]

            if baseline:
                avg = sum(baseline) / len(baseline)
                if count >= min_requests and avg > 0 and count >= spike_multiplier * avg:
                    alerts.append({
                        "minute": bucket,
                        "path": path,
                        "request_count": count,
                        "baseline_avg": round(avg, 1),
                        "multiplier": round(count / avg, 2),
                    })

            baseline.append(count)
            if len(baseline) > baseline_minutes:
                baseline.popleft()

    return alerts


def test_q88():
    base = datetime(2026, 5, 29, 8, 0, 0)
    events = []
    # 5 baseline minutes: /api/data gets 10 req/min
    for minute in range(5):
        for i in range(10):
            events.append({
                "path": "/api/data",
                "ts": (base + timedelta(minutes=minute, seconds=i * 6)).isoformat(),
            })
    # Spike minute: /api/data gets 80 requests
    for i in range(80):
        events.append({
            "path": "/api/data",
            "ts": (base + timedelta(minutes=5, seconds=i % 60)).isoformat(),
        })

# Q89 — Detect Authentication Spike by Country
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given login events with (user, country, ts), flag any country that generates
  a sudden spike in authentication attempts compared to its baseline — could
  indicate a geo-targeted credential stuffing campaign.

Clarifying questions:
  1. Should we alert per-country globally or per-country per-user?
  2. What is the baseline period (1 week)?
  3. Should successful and failed logins both count?

Security angle:
  Credential stuffing campaigns often originate from specific geographies.
  A 10× spike from CN, RU, or BR on a Monday morning indicates an automated
  campaign. Country-level alerting is a standard feature of SIEM auth modules
  (e.g., Azure AD "sign-in risk" by location).
"""


def detect_auth_spike_by_country(
    events: List[Dict],
    baseline_minutes: int = 5,
    spike_multiplier: float = 4.0,
    min_events: int = 5,
) -> List[Dict]:
    """
    Optimal: per-country deque baseline of per-minute counts.
    O(n) time.
    """
    from collections import Counter
    bucket_country: Dict[str, Counter] = defaultdict(Counter)

    for e in events:
        bucket = _minute_bucket(e.get("ts", ""))
        country = e.get("country") or "UNKNOWN"
        if bucket:
            bucket_country[bucket][country] += 1

    sorted_buckets = sorted(bucket_country)
    baselines: Dict[str, deque] = defaultdict(deque)
    alerts = []

    for bucket in sorted_buckets:
        for country, count in bucket_country[bucket].items():
            baseline = baselines[country]
            if baseline:
                avg = sum(baseline) / len(baseline)
                if count >= min_events and avg > 0 and count >= spike_multiplier * avg:
                    alerts.append({
                        "minute": bucket,
                        "country": country,
                        "count": count,
                        "baseline_avg": round(avg, 1),
                        "multiplier": round(count / avg, 2),
                    })
            baseline.append(count)
            if len(baseline) > baseline_minutes:
                baseline.popleft()

    return alerts


def test_q89():
    base = datetime(2026, 5, 29, 8, 0, 0)
    events = []
    # Baseline: 5 minutes, CN sends 5 logins/min
    for minute in range(5):
        for i in range(5):
            events.append({"country": "CN", "ts": (base + timedelta(minutes=minute, seconds=i * 10)).isoformat()})
    # Spike minute: CN sends 50 logins
    for i in range(50):
        events.append({"country": "CN", "ts": (base + timedelta(minutes=5, seconds=i % 60)).isoformat()})
    # Normal: US steady 10/min
    for minute in range(6):
        for i in range(10):
            events.append({"country": "US", "ts": (base + timedelta(minutes=minute, seconds=i * 5)).isoformat()})

    results = detect_auth_spike_by_country(events, baseline_minutes=5, spike_multiplier=4.0)
    countries = {r["country"] for r in results}
    assert "CN" in countries
    assert "US" not in countries
    print("Q89 tests passed")

# Q90 — Detect Repeated Access Denied Events
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Given IAM / file-system audit events, find principals with more than N
  access_denied events in a rolling window — indicating either misconfiguration
  or an attacker probing authorization boundaries.

Clarifying questions:
  1. Should we alert per-principal or per (principal, resource)?
  2. What is N? (default: 10 in 10 minutes)
  3. Should we also flag the resources being denied?

Security angle:
  Repeated access-denied is the top signal for:
  - Privilege escalation attempts (trying restricted APIs).
  - Misconfigured IAM roles (breaks a workflow, generates noise).
  - Lateral movement (attacker testing each resource).
  AWS CloudTrail "unauthorized operation" alerts use this pattern.
  Differentiating attacker vs. misconfiguration requires additional context
  (new IP, off-hours, sensitive resource).
"""


def detect_repeated_access_denied(
    events: List[Dict],
    threshold: int = 10,
    window_seconds: int = 600,
    group_by: str = "principal",
) -> List[Dict]:
    """
    Optimal: deque sliding window per principal.
    Also collects the top denied resources.
    O(n) total time.
    """
    windows: Dict[str, deque] = defaultdict(deque)
    denied_resources: Dict[str, Dict[str, int]] = defaultdict(lambda: defaultdict(int))
    alerts: Dict[str, Dict] = {}

    for e in events:
        action = e.get("action") or e.get("result") or ""
        if "denied" not in action.lower() and "unauthorized" not in action.lower():
            continue
        key = e.get(group_by) or "unknown"
        resource = e.get("resource") or "unknown"
        ts_raw = e.get("ts", "")
        try:
            ts = datetime.fromisoformat(ts_raw.rstrip("Z")).timestamp()
        except (ValueError, AttributeError):
            continue

        window = windows[key]
        window.append(ts)
        denied_resources[key][resource] += 1

        while window and ts - window[0] > window_seconds:
            window.popleft()

        if len(window) > threshold and key not in alerts:
            top_resources = sorted(
                denied_resources[key].items(), key=lambda x: -x[1]
            )[:5]
            alerts[key] = {
                group_by: key,
                "denied_count_in_window": len(window),
                "window_seconds": window_seconds,
                "top_denied_resources": [r for r, _ in top_resources],
            }

    return list(alerts.values())


def test_q90():
    base = datetime(2026, 5, 29, 8, 0, 0)
    events = []
    # Attacker: 15 access_denied in 5 minutes
    for i in range(15):
        events.append({
            "principal": "eve", "action": "access_denied",
            "resource": f"s3://bucket-{i % 5}",
            "ts": (base + timedelta(seconds=i * 20)).isoformat(),
        })
    # Normal admin: 3 access_denied (misconfiguration)
    for i in range(3):
        events.append({
            "principal": "admin", "action": "access_denied",
            "resource": "ec2:DescribeInstances",
            "ts": (base + timedelta(minutes=i * 2)).isoformat(),
        })

    results = detect_repeated_access_denied(events, threshold=10, window_seconds=600)
    principals = {r["principal"] for r in results}
    assert "eve" in principals
    assert "admin" not in principals
    print("Q90 tests passed")
