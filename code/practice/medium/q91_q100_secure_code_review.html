# Q91 — Review Login Code for SQL Injection
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Write a function that analyses a Python code snippet for SQL injection
  vulnerabilities — specifically unsafe string interpolation or concatenation
  into SQL queries.

Clarifying questions:
  1. Should we flag only SELECT/INSERT/UPDATE/DELETE?
  2. Should parameterised queries be flagged as safe?
  3. Is this a static analysis tool or a runtime interceptor?

Security angle:
  SQL injection (OWASP A03) is the most commonly exploited web vulnerability.
  Classic patterns: f-string/%-format/+ concatenation into a query string.
  Safe patterns: query+params tuple passed to execute(), SQLAlchemy ORM,
  psycopg2 parameterised queries.
  False positive: raw string that is never passed to execute().
"""

# Patterns that indicate unsafe SQL construction
_SQLI_PATTERNS = [
    re.compile(r"""(SELECT|INSERT|UPDATE|DELETE|FROM|WHERE).*(%s|{|}|f["']|["']\s*\+)""",
               re.IGNORECASE | re.DOTALL),
    re.compile(r"""cursor\.execute\(\s*["'].*%.*%|cursor\.execute\(\s*f["']""",
               re.IGNORECASE),
    re.compile(r"""cursor\.execute\(\s*[^"'(]*(["'].*\+)""", re.IGNORECASE),
]

_SAFE_PARAMETERISED = re.compile(
    r"""cursor\.execute\(\s*["'][^"']*["']\s*,""", re.IGNORECASE
)


def review_sql_injection(code: str) -> Dict:
    """
    Naive: regex scan for unsafe SQL construction patterns.
    Returns a report with verdict and matched evidence.
    O(n) per pattern.
    """
    findings = []
    for pattern in _SQLI_PATTERNS:
        for match in pattern.finditer(code):
            findings.append(match.group(0).strip()[:120])

    # Parameterised queries are safe even if the SQL keyword pattern fires on %s
    safe_count = len(_SAFE_PARAMETERISED.findall(code))
    if safe_count > 0:
        verdict = "LIKELY_SAFE"
    elif findings:
        verdict = "VULNERABLE"
    else:
        verdict = "REVIEW_NEEDED"
    return {"verdict": verdict, "findings": findings}


def test_q91():
    unsafe_code = """
def login(username, password):
    query = "SELECT * FROM users WHERE username='" + username + "'"
    cursor.execute(query)
"""
    safe_code = """
def login(username, password):
    cursor.execute("SELECT * FROM users WHERE username=%s", (username,))
"""
    r_unsafe = review_sql_injection(unsafe_code)
    r_safe = review_sql_injection(safe_code)
    assert r_unsafe["verdict"] == "VULNERABLE"
    assert r_safe["verdict"] == "LIKELY_SAFE"
    print("Q91 tests passed")

# Q92 — Review Password Hashing Code
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Analyse a code snippet for insecure password hashing:
  - MD5/SHA1/SHA256 used directly without a salt.
  - Missing use of bcrypt/argon2/scrypt/pbkdf2.

Clarifying questions:
  1. Is this a static checker or a runtime validator?
  2. Should we check for salt absence or just weak algorithm?
  3. Is PBKDF2 with high iterations acceptable? (yes, with iteration count check)

Security angle:
  Weak password hashing (CWE-916) allows offline cracking via rainbow tables
  (unsalted MD5) or GPU attacks (fast hashes). NIST SP 800-63B mandates
  memory-hard KDFs (bcrypt, argon2, scrypt). SHA256(password) with no salt
  is categorically insecure regardless of algorithm strength.
"""

_WEAK_HASH_RE = re.compile(
    r"\bhashlib\.(md5|sha1|sha256|sha512)\b.*\.update\(.*password|"
    r"\bhashlib\.(md5|sha1|sha256|sha512)\(.*password",
    re.IGNORECASE,
)
_STRONG_HASH_RE = re.compile(
    r"\b(bcrypt|argon2|scrypt|pbkdf2_hmac)\b", re.IGNORECASE
)


def review_password_hashing(code: str) -> Dict:
    """
    Static analysis: check for weak hash algorithms applied to passwords.
    Returns verdict and details.
    """
    weak_matches = _WEAK_HASH_RE.findall(code)
    strong_matches = _STRONG_HASH_RE.findall(code)

    if weak_matches and not strong_matches:
        verdict = "INSECURE"
        reason = f"Weak hash algorithm used directly on password: {weak_matches}"
    elif strong_matches:
        verdict = "LIKELY_SECURE"
        reason = f"Strong KDF found: {strong_matches}"
    else:
        verdict = "REVIEW_NEEDED"
        reason = "No recognised hash pattern found; manual review required"

    return {"verdict": verdict, "reason": reason}


def test_q92():
    insecure = "h = hashlib.sha256(password.encode()).hexdigest()"
    secure = "h = bcrypt.hashpw(password.encode(), bcrypt.gensalt())"
    also_secure = "dk = hashlib.pbkdf2_hmac('sha256', password.encode(), salt, 260000)"

    assert review_password_hashing(insecure)["verdict"] == "INSECURE"
    assert review_password_hashing(secure)["verdict"] == "LIKELY_SECURE"
    assert review_password_hashing(also_secure)["verdict"] == "LIKELY_SECURE"
    print("Q92 tests passed")

# Q93 — Review Session Token Generation
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Analyse code snippets for insecure session token generation:
  - Using random.random() or uuid.uuid4() — not CSPRNG.
  - Insufficient entropy (< 128 bits).

Clarifying questions:
  1. Is uuid4() acceptable? (No: uuid4 uses os.urandom internally in Python
     but is not typically designed as a security token; secrets.token_urlsafe
     is preferred and more explicit.)
  2. Should we check token length too?
  3. How about JWT token signing validation?

Security angle:
  Predictable session tokens are CWE-330 / OWASP A07. An attacker who can
  predict session IDs can hijack accounts. Python's random module is seeded
  with time and is not cryptographically secure. The correct approach is
  secrets.token_urlsafe(32) which uses os.urandom (CSPRNG).
"""

_INSECURE_TOKEN_RE = re.compile(
    r"\brandom\.(random|randint|choice|choices|randrange|getrandbits)\b",
    re.IGNORECASE,
)
_SECURE_TOKEN_RE = re.compile(
    r"\b(secrets\.token_urlsafe|secrets\.token_hex|os\.urandom|secrets\.token_bytes)\b",
    re.IGNORECASE,
)


def review_session_token_generation(code: str) -> Dict:
    insecure = _INSECURE_TOKEN_RE.findall(code)
    secure = _SECURE_TOKEN_RE.findall(code)

    if insecure:
        return {
            "verdict": "INSECURE",
            "reason": f"Non-CSPRNG source used: {insecure}",
        }
    if secure:
        return {"verdict": "SECURE", "reason": f"CSPRNG used: {secure}"}
    return {"verdict": "REVIEW_NEEDED", "reason": "No recognised token generation pattern"}


def test_q93():
    bad = "token = str(random.random())[2:]"
    good = "token = secrets.token_urlsafe(32)"

    assert review_session_token_generation(bad)["verdict"] == "INSECURE"
    assert review_session_token_generation(good)["verdict"] == "SECURE"
    print("Q93 tests passed")


# ─────────────────────────────────────────────────────────────────────────────
# Q94 — Review Password Reset Flow
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Analyse a password reset token code snippet for vulnerabilities:
  - No expiry check on reset token.
  - Token not invalidated after first use.
  - Predictable token generation.
  - Timing-safe comparison not used.

Clarifying questions:
  1. Is this server-side validation code?
  2. Should we treat DB-fetched tokens as potentially constant-time?
  3. Do we check for old-password re-use?

Security angle:
  Password reset flaws (CWE-640) are a top account-takeover vector:
  1. No expiry → token valid indefinitely after compromise.
  2. No invalidation after use → replay attack possible.
  3. Predictable token → attacker enumerates valid tokens.
  4. == comparison → timing side-channel leaks validity.
  Correct: secrets.compare_digest(). Django uses this; many frameworks miss #2.
"""

_TIMING_SAFE_RE = re.compile(r"\bhmac\.compare_digest|secrets\.compare_digest\b")
_UNSAFE_COMPARE_RE = re.compile(r"token\s*==\s*\w+|\w+\s*==\s*token", re.IGNORECASE)
_EXPIRY_CHECK_RE = re.compile(r"expir|expires_at|created_at|timedelta|datetime\.now", re.IGNORECASE)
_INVALIDATE_RE = re.compile(r"(delete|remove|invalidate|used\s*=\s*True|used_at)", re.IGNORECASE)


def review_password_reset(code: str) -> Dict:
    issues = []
    if _UNSAFE_COMPARE_RE.search(code) and not _TIMING_SAFE_RE.search(code):
        issues.append("Timing-unsafe token comparison (use hmac.compare_digest)")
    if not _EXPIRY_CHECK_RE.search(code):
        issues.append("No expiry check found on reset token")
    if not _INVALIDATE_RE.search(code):
        issues.append("Token not invalidated after use")
    verdict = "INSECURE" if issues else "LIKELY_SECURE"
    return {"verdict": verdict, "issues": issues}


def test_q94():
    bad_code = """
def reset_password(token, new_password):
    record = db.get_token(token)
    if record.token == token:
        db.update_password(record.user_id, new_password)
"""
    good_code = """
def reset_password(token, new_password):
    record = db.get_token(token)
    if not record or record.expires_at < datetime.now():
        raise ValueError('Invalid or expired token')
    if not hmac.compare_digest(record.token, token):
        raise ValueError('Invalid token')
    db.update_password(record.user_id, new_password)
    db.delete(record)
"""
    r_bad = review_password_reset(bad_code)
    r_good = review_password_reset(good_code)
    assert r_bad["verdict"] == "INSECURE"
    assert len(r_bad["issues"]) >= 2
    assert r_good["verdict"] == "LIKELY_SECURE"
    print("Q94 tests passed")

# Q95 — Review File Upload Handler
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Analyse a file upload handler for vulnerabilities:
  - No file type validation (MIME sniffing or extension check).
  - Path traversal via filename (../../etc/passwd).
  - Files stored in web root (direct execution risk).
  - No size limit.

Clarifying questions:
  1. Is this Flask/Django/FastAPI code?
  2. Is the upload directory configurable?
  3. Should we check for virus scanning integration?

Security angle:
  File upload attacks (OWASP A04): uploading webshells, path traversal
  to overwrite system files, zip bombs for DoS. Secure upload requires:
  1. Allowlist of MIME types.
  2. Filename sanitisation (os.path.basename + uuid rename).
  3. Storage outside web root.
  4. Size limit.
"""

_FILENAME_SECURE_RE = re.compile(
    r"secure_filename|os\.path\.basename|uuid|uuid4", re.IGNORECASE
)
_MIME_CHECK_RE = re.compile(
    r"content_type|mimetype|magic\.|filetype|allowed_extensions", re.IGNORECASE
)
_SIZE_LIMIT_RE = re.compile(
    r"MAX_CONTENT_LENGTH|content_length|\.seek\(|os\.path\.getsize|max_size", re.IGNORECASE
)


def review_file_upload(code: str) -> Dict:
    issues = []
    if not _FILENAME_SECURE_RE.search(code):
        issues.append("Filename not sanitised (path traversal risk)")
    if not _MIME_CHECK_RE.search(code):
        issues.append("No MIME / extension validation")
    if not _SIZE_LIMIT_RE.search(code):
        issues.append("No file size limit enforced")
    verdict = "INSECURE" if issues else "LIKELY_SECURE"
    return {"verdict": verdict, "issues": issues}


def test_q95():
    bad = """
@app.route('/upload', methods=['POST'])
def upload():
    f = request.files['file']
    f.save('/var/uploads/' + f.filename)
"""
    good = """
@app.route('/upload', methods=['POST'])
def upload():
    f = request.files['file']
    if f.content_type not in allowed_extensions:
        abort(400)
    if f.content_length > MAX_CONTENT_LENGTH:
        abort(413)
    filename = secure_filename(f.filename)
    f.save(os.path.join(UPLOAD_DIR, filename))
"""
    r_bad = review_file_upload(bad)
    r_good = review_file_upload(good)
    assert r_bad["verdict"] == "INSECURE"
    assert len(r_bad["issues"]) >= 2
    assert r_good["verdict"] == "LIKELY_SECURE"
    print("Q95 tests passed")

# Q96 — Secure File Path Join
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Implement a safe file path join that prevents path traversal attacks.
  Given a base directory and a user-supplied filename, return the resolved
  path only if it is within the base directory.

Clarifying questions:
  1. Should symlinks be followed? (no — resolve symlinks before checking)
  2. Should null bytes be rejected? (yes — CWE-626)
  3. Should Windows paths (backslash) be normalised?

Security angle:
  Path traversal (CWE-22) lets attackers read /etc/passwd by supplying
  "../../etc/passwd" as a filename. The fix is:
  1. os.path.realpath() to resolve all ".." and symlinks.
  2. Check that the resolved path starts with the base directory.
  Never use string manipulation alone — os.path.normpath does NOT follow symlinks
  and can be fooled by symbolic links pointing outside the base.
"""


def safe_path_join(base_dir: str, filename: str) -> Optional[str]:
    """
    Returns the safe resolved path, or None if traversal is detected.
    """
    if "\x00" in filename:
        return None  # Null byte injection
    # Reject obvious traversal markers early
    if ".." in filename or filename.startswith("/") or filename.startswith("\\"):
        return None
    base_real = os.path.realpath(base_dir)
    candidate = os.path.realpath(os.path.join(base_real, filename))
    if candidate.startswith(base_real + os.sep) or candidate == base_real:
        return candidate
    return None


def test_q96():
    import tempfile, os
    with tempfile.TemporaryDirectory() as tmpdir:
        # Normal file
        result = safe_path_join(tmpdir, "report.pdf")
        assert result is not None
        assert result.startswith(os.path.realpath(tmpdir))

        # Path traversal
        assert safe_path_join(tmpdir, "../../etc/passwd") is None
        assert safe_path_join(tmpdir, "../etc/passwd") is None

        # Null byte
        assert safe_path_join(tmpdir, "file\x00.txt") is None

        # Absolute path
        assert safe_path_join(tmpdir, "/etc/passwd") is None
    print("Q96 tests passed")

# Q97 — Validate Redirect URL
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Implement a redirect URL validator that prevents open redirect attacks.
  Only allow redirects to the same origin or to an explicit allowlist of
  trusted domains.

Clarifying questions:
  1. Should relative paths (no domain) be allowed? (yes — always safe)
  2. Should sub-domains of an allowed domain be permitted?
  3. Should we handle protocol-relative URLs (//evil.com)?

Security angle:
  Open redirect (CWE-601) enables phishing: attacker crafts
  https://trusted.com/login?next=https://evil.com to steal credentials.
  The correct fix is an allowlist of trusted hosts, not a denylist.
  Protocol-relative URLs (//evil.com) bypass naive "starts with /" checks.
"""

_ALLOWED_HOSTS: frozenset = frozenset({"example.com", "app.example.com"})


def is_safe_redirect(url: str, allowed_hosts: Optional[frozenset] = None) -> bool:
    """
    Returns True only if the URL is safe to redirect to.
    """
    if allowed_hosts is None:
        allowed_hosts = _ALLOWED_HOSTS
    if not url:
        return False
    try:
        parsed = urllib.parse.urlparse(url)
    except Exception:
        return False

    # Relative URL (no scheme, no netloc) is safe
    if not parsed.scheme and not parsed.netloc:
        # But not protocol-relative: //evil.com → netloc='evil.com'
        if url.startswith("//"):
            return False
        return True

    # Must be http or https
    if parsed.scheme not in ("http", "https"):
        return False

    # Must be in the allowlist (strip port)
    host = parsed.hostname or ""
    return host in allowed_hosts


def test_q97():
    assert is_safe_redirect("/dashboard") is True
    assert is_safe_redirect("//evil.com/steal") is False
    assert is_safe_redirect("https://example.com/home") is True
    assert is_safe_redirect("https://evil.com/phish") is False
    assert is_safe_redirect("javascript:alert(1)") is False
    assert is_safe_redirect("ftp://example.com/file") is False
    assert is_safe_redirect("https://app.example.com/page") is True
    print("Q97 tests passed")

# Q98 — Sanitise HTML Output
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Implement an HTML output sanitiser that:
  1. Escapes dangerous HTML characters for plain-text contexts.
  2. Strips disallowed tags for rich-text contexts (allowlist of safe tags).

Clarifying questions:
  1. Is this for a template context or direct DOM insertion?
  2. Should we preserve allowed inline tags? (e.g., <b>, <i>, <a>)
  3. Should we strip event handlers from allowed tags?

Security angle:
  XSS (OWASP A03) via unsanitised user input rendered as HTML. The two levels:
  1. Escape: html.escape() → safe for plain-text display.
  2. Strip: regex or parser-based allowlist for user-generated HTML.
  Using a regex-only strip is fragile; production should use bleach/nh3.
  Here we implement a simple regex-based strip for demonstration.
"""

_ALLOWED_TAGS = frozenset({"b", "i", "u", "em", "strong", "p", "br", "ul", "li", "ol"})
_TAG_RE = re.compile(r"</?([a-zA-Z][a-zA-Z0-9]*)\b[^>]*>", re.IGNORECASE)
_ATTR_EVENT_RE = re.compile(r'''\bon\w+\s*=\s*(?:"[^"]*"|'[^']*'|\S+)''', re.IGNORECASE)


def sanitise_html_escape(text: str) -> str:
    """Level 1: full HTML escape — safe for plain-text contexts."""
    return html.escape(text, quote=True)


def sanitise_html_strip(html_input: str) -> str:
    """
    Level 2: strip disallowed tags; keep allowed safe tags.
    Also strips event handler attributes.
    WARNING: regex-based stripping is incomplete; prefer bleach/nh3 in production.
    """
    # Strip event handlers from any tag
    clean = _ATTR_EVENT_RE.sub("", html_input)

    def replace_tag(m: re.Match) -> str:
        tag_name = m.group(1).lower()
        if tag_name in _ALLOWED_TAGS:
            return m.group(0)
        return ""

    return _TAG_RE.sub(replace_tag, clean)


def test_q98():
    # Escape test
    xss = '<script>alert("xss")</script>'
    escaped = sanitise_html_escape(xss)
    assert "<script>" not in escaped
    assert "&lt;script&gt;" in escaped

    # Strip test
    dirty = '<b>bold</b><script>evil()</script><p onclick="bad()">text</p>'
    clean = sanitise_html_strip(dirty)
    assert "<script>" not in clean
    assert "<b>" in clean
    assert "onclick" not in clean
    assert "<p" in clean      # p is allowed, but onclick stripped
    print("Q98 tests passed")

# Q99 — Detect Unsafe Deserialization Usage
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Analyse a Python code snippet for unsafe deserialization:
  - pickle.loads / pickle.load on untrusted input.
  - yaml.load() without Loader=yaml.SafeLoader.
  - marshal.loads() on untrusted data.

Clarifying questions:
  1. Should we flag all pickle usage or only where input is external?
  2. Is yaml.safe_load() acceptable? (yes)
  3. Should we flag shelve / jsonpickle?

Security angle:
  Unsafe deserialization (OWASP A08) allows arbitrary code execution.
  pickle.loads(attacker_data) → RCE in one line. This has caused major
  incidents (CVE-2019-20907 and many ML model poisoning attacks).
  Safer alternatives: json, yaml.safe_load, msgpack (no code execution).
"""

_UNSAFE_DESER_RE = re.compile(
    r"\b(pickle|marshal|shelve)\.(loads?|load)\b|"
    r"\byaml\.load\b(?!\s*\([^)]*SafeLoader)|"
    r"\bjsonpickle\.decode\b",
    re.IGNORECASE,
)
_SAFE_YAML_RE = re.compile(r"\byaml\.safe_load\b|\byaml\.load\b.*SafeLoader", re.IGNORECASE)


def detect_unsafe_deserialization(code: str) -> Dict:
    unsafe = _UNSAFE_DESER_RE.findall(code)
    safe_yaml = _SAFE_YAML_RE.findall(code)

    # If yaml.load is flagged but SafeLoader is present, it's a false positive
    adjusted = [u for u in unsafe if not (isinstance(u, str) and "yaml" in u.lower() and safe_yaml)]

    verdict = "VULNERABLE" if adjusted else "LIKELY_SAFE"
    return {"verdict": verdict, "unsafe_calls": adjusted}


def test_q99():
    bad_pickle = "data = pickle.loads(user_input)"
    bad_yaml = "cfg = yaml.load(open('config.yaml'))"
    good_yaml = "cfg = yaml.safe_load(open('config.yaml'))"
    good_json = "data = json.loads(user_input)"

    assert detect_unsafe_deserialization(bad_pickle)["verdict"] == "VULNERABLE"
    assert detect_unsafe_deserialization(bad_yaml)["verdict"] == "VULNERABLE"
    assert detect_unsafe_deserialization(good_yaml)["verdict"] == "LIKELY_SAFE"
    assert detect_unsafe_deserialization(good_json)["verdict"] == "LIKELY_SAFE"
    print("Q99 tests passed")


# ─────────────────────────────────────────────────────────────────────────────
# Q100 — Detect Shell Injection Risk
# ─────────────────────────────────────────────────────────────────────────────
"""
Problem:
  Analyse code for shell injection vulnerabilities:
  - os.system() with user input.
  - subprocess with shell=True and interpolated input.
  - eval() / exec() on user-controlled data.

Clarifying questions:
  1. Should we flag subprocess(shell=False) with user input? (no — safe)
  2. Should we flag subprocess(shell=True) even with hardcoded strings? (warn)
  3. Is eval() always unsafe? (yes in this context — flag it)

Security angle:
  Shell injection (OWASP A03, CWE-78) allows OS command execution.
  subprocess.run(["ls", user_dir], shell=False) is safe.
  subprocess.run(f"ls {user_dir}", shell=True) is NOT safe.
  os.system(cmd) is never safe with user input.
  Detecting shell=True is the primary indicator; eval/exec are always risky.
"""

_SHELL_INJECT_RE = re.compile(
    r"\bos\.system\s*\(|"
    r"\bsubprocess\.(run|call|Popen|check_output|check_call)\s*\([^)]*shell\s*=\s*True|"
    r"\beval\s*\(|"
    r"\bexec\s*\(",
    re.IGNORECASE,
)

_SAFE_SUBPROCESS_RE = re.compile(
    r"\bsubprocess\.(run|call|Popen|check_output)\s*\(\[",   # list form → safe
    re.IGNORECASE,
)


def detect_shell_injection_risk(code: str) -> Dict:
    risky = _SHELL_INJECT_RE.findall(code)
    safe = _SAFE_SUBPROCESS_RE.findall(code)
    verdict = "RISKY" if risky else "LIKELY_SAFE"
    return {"verdict": verdict, "risky_patterns": risky, "safe_patterns": safe}


def test_q100():
    bad1 = "os.system('ls ' + user_dir)"
    bad2 = "subprocess.run(f'cat {filename}', shell=True)"
    bad3 = "eval(user_expression)"
    good = "subprocess.run(['ls', user_dir], capture_output=True)"

    assert detect_shell_injection_risk(bad1)["verdict"] == "RISKY"
    assert detect_shell_injection_risk(bad2)["verdict"] == "RISKY"
    assert detect_shell_injection_risk(bad3)["verdict"] == "RISKY"
    assert detect_shell_injection_risk(good)["verdict"] == "LIKELY_SAFE"
    print("Q100 tests passed")
