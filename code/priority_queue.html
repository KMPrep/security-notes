"""
Priority Queue (Min-Heap) — no built-in heap/sort functions.
Backed by a plain list of integers.

Heap invariant: parent <= both children  (min at index 0)

Index math (0-based):
  parent(i)      = (i - 1) // 2
  left_child(i)  = 2 * i + 1
  right_child(i) = 2 * i + 2
"""


class PriorityQueue:
    def __init__(self, data: list[int] | None = None):
        self._heap: list[int] = []
        if data:
            # Copy the input list and heapify in O(n)
            self._heap = list(data)
            self._heapify()

    # ------------------------------------------------------------------ #
    # Public API
    # ------------------------------------------------------------------ #

    def push(self, value: int) -> None:
        """Insert a value.  O(log n)"""
        self._heap.append(value)
        self._sift_up(len(self._heap) - 1)

    def pop(self) -> int:
        """Remove and return the minimum value.  O(log n)"""
        if not self._heap:
            raise IndexError("pop from empty priority queue")
        # Swap root with last element, shrink, then restore
        self._swap(0, len(self._heap) - 1)
        minimum = self._heap.pop()
        if self._heap:
            self._sift_down(0)
        return minimum

    def peek(self) -> int:
        """Return the minimum value without removing it.  O(1)"""
        if not self._heap:
            raise IndexError("peek at empty priority queue")
        return self._heap[0]

    def __len__(self) -> int:
        return len(self._heap)

    def __repr__(self) -> str:
        return f"PriorityQueue({self._heap})"

    # ------------------------------------------------------------------ #
    # Internal helpers
    # ------------------------------------------------------------------ #

    def _heapify(self) -> None:
        """Build heap from an arbitrary list in O(n) by sifting down
        every non-leaf node from the bottom up."""
        last_non_leaf = (len(self._heap) - 2) // 2
        for i in range(last_non_leaf, -1, -1):
            self._sift_down(i)

    def _sift_up(self, i: int) -> None:
        """Move element at index i up until heap property is satisfied."""
        while i > 0:
            parent = (i - 1) // 2
            if self._heap[i] < self._heap[parent]:
                self._swap(i, parent)
                i = parent
            else:
                break

    def _sift_down(self, i: int) -> None:
        """Move element at index i down until heap property is satisfied."""
        n = len(self._heap)
        while True:
            smallest = i
            left  = 2 * i + 1
            right = 2 * i + 2

            if left < n and self._heap[left] < self._heap[smallest]:
                smallest = left
            if right < n and self._heap[right] < self._heap[smallest]:
                smallest = right

            if smallest == i:
                break
            self._swap(i, smallest)
            i = smallest

    def _swap(self, a: int, b: int) -> None:
        self._heap[a], self._heap[b] = self._heap[b], self._heap[a]

if __name__ == "__main__":
    # --- Build from static list ---
    numbers = [9, 4, 7, 1, 8, 3, 6, 2, 5]
    print("Input:", numbers)

    pq = PriorityQueue(numbers)
    print("Heap internal:", pq)

    print("\nPop order (should be sorted ascending):")
    while len(pq):
        print(pq.pop(), end=" ")
    print()

    # --- Push one by one ---
    print("\nPush one by one: 15, 10, 20, 5")
    pq2 = PriorityQueue()
    for v in [15, 10, 20, 5]:
        pq2.push(v)
        print(f"  pushed {v} → heap: {pq2._heap}")

    print("peek:", pq2.peek())
    print("pop:", pq2.pop())
    print("pop:", pq2.pop())
    print("Remaining heap:", pq2._heap)
