"""
Priority Queue — Sorted Singly Linked List (no built-ins).
Backed by Node objects; no arrays.

Strategy: keep the list sorted ascending at all times.
  push()  → walk to correct position, splice in new node  O(n)
  pop()   → remove head (always minimum)                  O(1)
  peek()  → read head value                               O(1)

Trade-off vs heap: simpler logic but O(n) insert instead of O(log n).
Good for small queues or clarity.
"""


class Node:
    def __init__(self, value: int):
        self.value: int = value
        self.next: "Node | None" = None

    def __repr__(self) -> str:
        return f"Node({self.value})"


class PriorityQueueLinked:
    def __init__(self, data: list[int] | None = None):
        self._head: Node | None = None
        self._size: int = 0
        if data:
            for v in data:
                self.push(v)

    # ------------------------------------------------------------------ #
    # Public API
    # ------------------------------------------------------------------ #

    def push(self, value: int) -> None:
        """Insert value in sorted position.  O(n)"""
        new_node = Node(value)
        self._size += 1

        # Insert before head if list is empty or new value is smallest
        if self._head is None or value < self._head.value:
            new_node.next = self._head
            self._head = new_node
            return

        # Walk until we find the right gap: current.value <= value < current.next.value
        current = self._head
        while current.next is not None and current.next.value <= value:
            current = current.next

        new_node.next = current.next
        current.next = new_node

    def pop(self) -> int:
        """Remove and return the minimum (head) value.  O(1)"""
        if self._head is None:
            raise IndexError("pop from empty priority queue")
        minimum = self._head.value
        self._head = self._head.next
        self._size -= 1
        return minimum

    def peek(self) -> int:
        """Return the minimum value without removing it.  O(1)"""
        if self._head is None:
            raise IndexError("peek at empty priority queue")
        return self._head.value

    def __len__(self) -> int:
        return self._size

    def __repr__(self) -> str:
        values = []
        current = self._head
        while current is not None:
            values.append(str(current.value))
            current = current.next
        return "PriorityQueueLinked([" + " -> ".join(values) + "])"

# Demo / tests

if __name__ == "__main__":
    # --- Build from static list ---
    numbers = [9, 4, 7, 1, 8, 3, 6, 2, 5]
    print("Input:", numbers)

    pq = PriorityQueueLinked(numbers)
    print("Linked list state:", pq)

    print("\nPop order (should be sorted ascending):")
    while len(pq):
        print(pq.pop(), end=" ")
    print()

    # --- Push one by one ---
    print("\nPush one by one: 15, 10, 20, 5")
    pq2 = PriorityQueueLinked()
    for v in [15, 10, 20, 5]:
        pq2.push(v)
        print(f"  pushed {v} → {pq2}")

    print("peek:", pq2.peek())
    print("pop:", pq2.pop())
    print("pop:", pq2.pop())
    print("Remaining:", pq2)

    # --- Duplicate values ---
    print("\nDuplicate values: 3, 1, 3, 2, 1")
    pq3 = PriorityQueueLinked([3, 1, 3, 2, 1])
    print("State:", pq3)
    while len(pq3):
        print(pq3.pop(), end=" ")
    print()
