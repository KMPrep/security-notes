# Q1 - Two Sum
# Given a list of integers and a target, return the indices of the two numbers
# that add up to the target. Exactly one solution guaranteed.
#
# Approach:
#   - Naive:          O(n^2) — nested loops, check every pair.
#   - No built-ins:   O(n)   — manual dict: store value -> index, look up complement.
#   - With built-ins: O(n)   — same dict approach, idiomatic Python.
#
# Edge cases:
#   - Negative numbers         -> handled naturally
#   - Same element used twice  -> indices must differ (guaranteed by problem)
#   - No solution              -> return [] (problem guarantees one, but defensive)
#   - None / non-list input    -> TypeError
#   - Non-numeric target       -> TypeError
#   - Non-numeric element      -> TypeError
#   - Single-element list      -> return [] (can't use same index twice)
#   - Float precision          -> exact dict lookup may miss; documented, not "fixed"


# === APPROACH 1: NAIVE — O(n^2) ===
# "Check every pair. Easy to explain but too slow for large inputs."

def two_sum_naive(nums, target):
    if nums is None:
        raise TypeError("Input cannot be None")
    if not isinstance(nums, list):
        raise TypeError(f"Expected list, got {type(nums).__name__}")
    if not isinstance(target, (int, float)):
        raise TypeError(f"Expected numeric target, got {type(target).__name__}")
    for i in range(len(nums)):
        for j in range(i + 1, len(nums)):
            if nums[i] + nums[j] == target:
                return [i, j]
    return []


# === APPROACH 2: NO BUILT-INS — O(n) ===
# "One-pass: for each number store its index; check if complement was already seen."

def two_sum_no_builtins(nums, target):
    if nums is None:
        raise TypeError("Input cannot be None")
    if not isinstance(nums, list):
        raise TypeError(f"Expected list, got {type(nums).__name__}")
    if not isinstance(target, (int, float)):
        raise TypeError(f"Expected numeric target, got {type(target).__name__}")
    seen = {}
    for i in range(len(nums)):
        complement = target - nums[i]
        if complement in seen:
            return [seen[complement], i]
        seen[nums[i]] = i
    return []


# === APPROACH 3: WITH BUILT-INS — O(n) ===
# "enumerate() makes the intent cleaner."

def two_sum(nums, target):
    if nums is None:
        raise TypeError("Input cannot be None")
    if not isinstance(nums, list):
        raise TypeError(f"Expected list, got {type(nums).__name__}")
    if not isinstance(target, (int, float)):
        raise TypeError(f"Expected numeric target, got {type(target).__name__}")

    seen = {}
    for i, num in enumerate(nums):
        complement = target - num
        if complement in seen:
            return [seen[complement], i]
        seen[num] = i
    return []


# --- tests ---
assert two_sum_naive([2, 7, 11, 15], 9) == [0, 1]
assert two_sum_no_builtins([2, 7, 11, 15], 9) == [0, 1]
assert two_sum([2, 7, 11, 15], 9) == [0, 1]
assert two_sum([3, 2, 4], 6) == [1, 2]
assert two_sum([3, 3], 6) == [0, 1]
assert two_sum([], 5) == []

# single-element: no valid pair
assert two_sum([5], 5) == []

# non-numeric target -> TypeError
try:
    two_sum([1, 2], None)
    assert False, "expected TypeError"
except TypeError:
    pass

# non-numeric element -> TypeError
try:
    two_sum([1, "a", 3], 4)
    assert False, "expected TypeError"
except TypeError:
    pass

# float precision: 0.1 + 0.2 != 0.3 in floating point — returns [] by design
assert two_sum([0.1, 0.2], 0.3) == []

print(two_sum([2, 7, 11, 15], 9))   # [0, 1]
print(two_sum([3, 2, 4], 6))        # [1, 2]
print(two_sum([3, 3], 6))           # [0, 1]
