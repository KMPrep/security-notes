# Q10 - Minimum Window Substring
# Find the shortest substring of s containing all characters of t (including duplicates).
#
# Approach:
#   - Naive:          O(n^2) — for each left position, find the shortest valid right.
#   - Two pointers:   O(n)   — expand right until valid, shrink left while still valid.
#   - With built-ins: O(n)   — same, using Counter for cleaner frequency tracking.
#
# Edge cases:
#   - t longer than s          -> ""
#   - s == t                   -> s
#   - t chars not all in s     -> ""
#   - Repeated chars in t      -> must be covered that many times
#   - None input               -> TypeError


from collections import Counter


# === APPROACH 1: NAIVE — O(n^2) ===
# "For every left, advance right until all of t is covered."

def min_window_naive(s, t):
    if s is None or t is None:
        raise TypeError("Input cannot be None")
    if not t or not s:
        return ""
    best = ""
    for left in range(len(s)):
        need = {}
        for c in t:
            need[c] = need.get(c, 0) + 1
        have = 0
        required = len(need)
        for right in range(left, len(s)):
            c = s[right]
            if c in need:
                need[c] -= 1
                if need[c] == 0:
                    have += 1
            if have == required:
                window = s[left:right + 1]
                if not best or len(window) < len(best):
                    best = window
                break
    return best


# === APPROACH 2: TWO POINTERS — NO BUILT-INS — O(n) ===
# "formed tracks how many unique chars in t are fully satisfied in the window."

def min_window_two_pointer(s, t):
    if s is None or t is None:
        raise TypeError("Input cannot be None")
    if not t or not s:
        return ""
    need = {}
    for c in t:
        need[c] = need.get(c, 0) + 1
    have = {}
    formed = 0
    required = len(need)
    left = 0
    best = ""

    for right in range(len(s)):
        c = s[right]
        have[c] = have.get(c, 0) + 1
        if c in need and have[c] == need[c]:
            formed += 1

        while formed == required:
            window = s[left:right + 1]
            if not best or len(window) < len(best):
                best = window
            left_char = s[left]
            have[left_char] -= 1
            if left_char in need and have[left_char] < need[left_char]:
                formed -= 1
            left += 1

    return best


# === APPROACH 3: WITH BUILT-INS — O(n) ===
# "Counter makes need/have cleaner. Same two-pointer logic."

def min_window(s, t):
    if s is None or t is None:
        raise TypeError("Input cannot be None")
    if not isinstance(s, str) or not isinstance(t, str):
        raise TypeError("Both inputs must be strings")
    if not t or not s:
        return ""

    need = Counter(t)
    have = Counter()
    formed = 0
    required = len(need)
    left = 0
    result = ""

    for right, char in enumerate(s):
        have[char] += 1
        if char in need and have[char] == need[char]:
            formed += 1

        while formed == required:
            window = s[left:right + 1]
            if not result or len(window) < len(result):
                result = window
            have[s[left]] -= 1
            if s[left] in need and have[s[left]] < need[s[left]]:
                formed -= 1
            left += 1

    return result


# --- tests ---
assert min_window_two_pointer("ADOBECODEBANC", "ABC") == "BANC"
assert min_window("ADOBECODEBANC", "ABC") == "BANC"
assert min_window("a", "a") == "a"
assert min_window("a", "aa") == ""
assert min_window("", "a") == ""
assert min_window("aa", "aa") == "aa"
assert min_window("ab", "b") == "b"

print(min_window("ADOBECODEBANC", "ABC"))  # "BANC"
print(min_window("a", "a"))               # "a"
print(min_window("a", "aa"))              # ""
