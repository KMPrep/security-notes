# Q11 - Search in Rotated Sorted Array
# A sorted array was rotated at an unknown pivot. Return target index or -1.
#
# Approach:
#   - Naive:       O(n)     — linear scan.
#   - Binary search: O(log n) — one half of any [lo, hi] window is always sorted.
#                              Identify which half, check if target is in it, narrow.
#   - Same algorithm, clean variable names.
#
# Key insight: After rotation, split at mid: nums[lo..mid] or nums[mid..hi] is
# always sorted. Check the sorted half first to decide which side to discard.
#
# Edge cases:
#   - Target not present             -> -1
#   - Single element                 -> direct check
#   - No rotation (sorted normally)  -> works as standard binary search
#   - None input                     -> TypeError


# === APPROACH 1: NAIVE — O(n) ===

def search_rotated_naive(nums, target):
    if nums is None:
        raise TypeError("Input cannot be None")
    for i in range(len(nums)):
        if nums[i] == target:
            return i
    return -1


# === APPROACH 2: MODIFIED BINARY SEARCH — NO BUILT-INS — O(log n) ===
# "At each mid: is left half sorted? Is target in that sorted half? Pick your side."

def search_rotated_bs(nums, target):
    if nums is None:
        raise TypeError("Input cannot be None")
    lo, hi = 0, len(nums) - 1
    while lo <= hi:
        mid = (lo + hi) // 2
        if nums[mid] == target:
            return mid
        # Left half is sorted
        if nums[lo] <= nums[mid]:
            if nums[lo] <= target < nums[mid]:
                hi = mid - 1
            else:
                lo = mid + 1
        # Right half is sorted
        else:
            if nums[mid] < target <= nums[hi]:
                lo = mid + 1
            else:
                hi = mid - 1
    return -1


# === APPROACH 3: WITH CLEAN NAMING — O(log n) ===

def search_rotated(nums, target):
    if nums is None:
        raise TypeError("Input cannot be None")
    if not isinstance(nums, list):
        raise TypeError(f"Expected list, got {type(nums).__name__}")

    left, right = 0, len(nums) - 1
    while left <= right:
        mid = (left + right) // 2
        if nums[mid] == target:
            return mid

        if nums[left] <= nums[mid]:               # left half is sorted
            if nums[left] <= target < nums[mid]:
                right = mid - 1
            else:
                left = mid + 1
        else:                                      # right half is sorted
            if nums[mid] < target <= nums[right]:
                left = mid + 1
            else:
                right = mid - 1

    return -1


# --- tests ---
assert search_rotated_naive([4, 5, 6, 7, 0, 1, 2], 0) == 4
assert search_rotated_bs([4, 5, 6, 7, 0, 1, 2], 0) == 4
assert search_rotated([4, 5, 6, 7, 0, 1, 2], 0) == 4
assert search_rotated([4, 5, 6, 7, 0, 1, 2], 3) == -1
assert search_rotated([1], 0) == -1
assert search_rotated([1], 1) == 0
assert search_rotated([3, 1], 1) == 1
assert search_rotated([1, 3], 3) == 1

print(search_rotated([4, 5, 6, 7, 0, 1, 2], 0))   # 4
print(search_rotated([4, 5, 6, 7, 0, 1, 2], 3))   # -1
print(search_rotated([3, 1], 1))                    # 1
