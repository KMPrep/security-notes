# Q13 - Kth Largest Element in an Array
# Find the kth largest element (not kth distinct) in an unsorted array.
#
# Approach:
#   - Naive:        O(n log n) — sort descending, return index k-1.
#   - Min-heap:     O(n log k) — maintain min-heap of size k; root is the answer.
#   - Quickselect:  O(n) avg  — partition like quicksort but recurse into one side only.
#
# Key insight (quickselect): the kth largest is at index (n - k) in sorted order.
# Partition around a pivot; if pivot lands at that index, we're done.
#
# Edge cases:
#   - k == 1              -> return max
#   - k == len(nums)      -> return min
#   - Duplicate elements  -> handled naturally
#   - None / empty input  -> TypeError / ValueError


import heapq


# === APPROACH 1: NAIVE — O(n log n) ===

def kth_largest_naive(nums, k):
    if nums is None:
        raise TypeError("Input cannot be None")
    return sorted(nums, reverse=True)[k - 1]


# === APPROACH 2: MIN-HEAP — O(n log k) ===
# "Keep a heap of the k largest seen. When size > k, pop the minimum.
#  After full scan, the root is the kth largest."

def kth_largest_heap(nums, k):
    if nums is None:
        raise TypeError("Input cannot be None")
    heap = []
    for num in nums:
        heapq.heappush(heap, num)
        if len(heap) > k:
            heapq.heappop(heap)
    return heap[0]


# === APPROACH 3: QUICKSELECT — O(n) AVERAGE ===
# "target_idx is the index of the kth largest in sorted order.
#  Partition in place; recurse only into the relevant half."

def kth_largest(nums, k):
    if nums is None:
        raise TypeError("Input cannot be None")
    if not isinstance(nums, list) or not nums:
        raise ValueError("nums must be a non-empty list")

    nums = nums[:]                    # defensive copy — do not modify caller's list
    target_idx = len(nums) - k

    def quickselect(lo, hi):
        pivot = nums[hi]
        store = lo
        for i in range(lo, hi):
            if nums[i] <= pivot:
                nums[i], nums[store] = nums[store], nums[i]
                store += 1
        nums[store], nums[hi] = nums[hi], nums[store]

        if store == target_idx:
            return nums[store]
        elif store < target_idx:
            return quickselect(store + 1, hi)
        else:
            return quickselect(lo, store - 1)

    return quickselect(0, len(nums) - 1)


# --- tests ---
assert kth_largest_naive([3, 2, 1, 5, 6, 4], 2) == 5
assert kth_largest_heap([3, 2, 1, 5, 6, 4], 2) == 5
assert kth_largest([3, 2, 1, 5, 6, 4], 2) == 5
assert kth_largest([3, 2, 3, 1, 2, 4, 5, 5, 6], 4) == 4
assert kth_largest([1], 1) == 1
assert kth_largest([7, 6, 5, 4, 3, 2, 1], 1) == 7

print(kth_largest([3, 2, 1, 5, 6, 4], 2))          # 5
print(kth_largest([3, 2, 3, 1, 2, 4, 5, 5, 6], 4)) # 4
