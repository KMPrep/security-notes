# Q14 - Merge Intervals
# Given a list of intervals, merge all overlapping ones.
#
# Approach:
#   - Naive:          O(n^2) — repeatedly scan for any overlapping pair and merge.
#   - Sort + sweep:   O(n log n) — sort by start; greedily extend the last merged interval.
#   - Same algorithm, clean version.
#
# Edge cases:
#   - Single interval           -> return as-is
#   - No overlaps               -> return original (sorted)
#   - All merge into one        -> return single interval
#   - Contained interval        -> [1,4],[2,3] -> [1,4]
#   - None input                -> TypeError


# === APPROACH 1: NAIVE — O(n^2) ===
# "Keep looping until no merge happened in a full pass."

def merge_intervals_naive(intervals):
    if intervals is None:
        raise TypeError("Input cannot be None")
    if not intervals:
        return []
    changed = True
    merged = [list(i) for i in intervals]
    while changed:
        changed = False
        result = []
        used = [False] * len(merged)
        for i in range(len(merged)):
            if used[i]:
                continue
            current = merged[i][:]
            for j in range(i + 1, len(merged)):
                if used[j]:
                    continue
                a, b = current
                c, d = merged[j]
                if a <= d and c <= b:                  # overlap condition
                    current = [min(a, c), max(b, d)]
                    used[j] = True
                    changed = True
            result.append(current)
        merged = result
    return merged


# === APPROACH 2: SORT + SWEEP — NO BUILT-INS — O(n log n) ===
# "Sort by start time. For each interval, either extend the last result or append."

def merge_intervals_sort(intervals):
    if intervals is None:
        raise TypeError("Input cannot be None")
    if not intervals:
        return []
    sorted_ivs = sorted(intervals, key=lambda x: x[0])
    result = [list(sorted_ivs[0])]
    for i in range(1, len(sorted_ivs)):
        start, end = sorted_ivs[i]
        if start <= result[-1][1]:
            if end > result[-1][1]:
                result[-1][1] = end
        else:
            result.append([start, end])
    return result


# === APPROACH 3: WITH BUILT-INS — O(n log n) ===
# "Same sweep, using max() for the merge step."

def merge_intervals(intervals):
    if intervals is None:
        raise TypeError("Input cannot be None")
    if not isinstance(intervals, list):
        raise TypeError(f"Expected list, got {type(intervals).__name__}")
    if not intervals:
        return []

    intervals = sorted(intervals, key=lambda x: x[0])
    merged = [list(intervals[0])]

    for start, end in intervals[1:]:
        if start <= merged[-1][1]:
            merged[-1][1] = max(merged[-1][1], end)
        else:
            merged.append([start, end])

    return merged


# --- tests ---
assert merge_intervals([[1, 3], [2, 6], [8, 10], [15, 18]]) == [[1, 6], [8, 10], [15, 18]]
assert merge_intervals([[1, 4], [4, 5]]) == [[1, 5]]
assert merge_intervals([[1, 4], [2, 3]]) == [[1, 4]]
assert merge_intervals([[1, 4]]) == [[1, 4]]
assert merge_intervals([]) == []
assert merge_intervals_sort([[1, 3], [2, 6], [8, 10], [15, 18]]) == [[1, 6], [8, 10], [15, 18]]

print(merge_intervals([[1, 3], [2, 6], [8, 10], [15, 18]]))  # [[1, 6], [8, 10], [15, 18]]
print(merge_intervals([[1, 4], [4, 5]]))                      # [[1, 5]]
print(merge_intervals([[1, 4], [2, 3]]))                      # [[1, 4]]
