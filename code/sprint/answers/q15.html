# Q15 - Search a 2D Matrix
# Matrix where each row is sorted and row[i][-1] < row[i+1][0]. Search for target.
#
# Approach:
#   - Naive:         O(m*n)       — scan every cell.
#   - Staircase:     O(m+n)       — start top-right; move left if too big, down if too small.
#   - Binary search: O(log(m*n)) — treat as a virtual 1D array; map mid via mid//n, mid%n.
#
# Edge cases:
#   - Empty matrix           -> False
#   - Target below min       -> False
#   - Target above max       -> False
#   - Single cell            -> direct compare
#   - None input             -> TypeError


# === APPROACH 1: NAIVE — O(m*n) ===

def search_matrix_naive(matrix, target):
    if matrix is None:
        raise TypeError("Input cannot be None")
    for row in matrix:
        for val in row:
            if val == target:
                return True
    return False


# === APPROACH 2: STAIRCASE — O(m+n) ===
# "Start at top-right corner. Too big -> step left. Too small -> step down."

def search_matrix_staircase(matrix, target):
    if matrix is None:
        raise TypeError("Input cannot be None")
    if not matrix or not matrix[0]:
        return False
    row, col = 0, len(matrix[0]) - 1
    while row < len(matrix) and col >= 0:
        if matrix[row][col] == target:
            return True
        elif matrix[row][col] > target:
            col -= 1
        else:
            row += 1
    return False


# === APPROACH 3: BINARY SEARCH — O(log(m*n)) ===
# "Treat the matrix as a sorted 1D array. Index mid maps to [mid//n][mid%n]."

def search_matrix(matrix, target):
    if matrix is None:
        raise TypeError("Input cannot be None")
    if not isinstance(matrix, list) or not matrix or not matrix[0]:
        return False

    m, n = len(matrix), len(matrix[0])
    lo, hi = 0, m * n - 1

    while lo <= hi:
        mid = (lo + hi) // 2
        val = matrix[mid // n][mid % n]
        if val == target:
            return True
        elif val < target:
            lo = mid + 1
        else:
            hi = mid - 1

    return False


# --- tests ---
matrix = [[1, 3, 5, 7], [10, 11, 16, 20], [23, 30, 34, 60]]
assert search_matrix_naive(matrix, 3) == True
assert search_matrix_staircase(matrix, 3) == True
assert search_matrix(matrix, 3) == True
assert search_matrix(matrix, 13) == False
assert search_matrix([], 1) == False
assert search_matrix([[1]], 1) == True
assert search_matrix([[1]], 2) == False
assert search_matrix([[1, 3, 5, 7], [10, 11, 16, 20]], 11) == True

print(search_matrix([[1, 3, 5, 7], [10, 11, 16, 20], [23, 30, 34, 60]], 3))   # True
print(search_matrix([[1, 3, 5, 7], [10, 11, 16, 20], [23, 30, 34, 60]], 13))  # False
