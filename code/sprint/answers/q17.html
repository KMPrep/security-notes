# Q17 - Validate Binary Search Tree
# Determine if a binary tree is a valid BST.
# Rule: for every node, all left descendants < node.val < all right descendants.
#
# Approach:
#   - Inorder traversal:   O(n) — inorder of a valid BST must be strictly ascending.
#   - Min/max bounds:      O(n) — pass valid (min, max) range down recursively.
#   - Iterative inorder:   O(n) — avoids recursion stack; useful for very deep trees.
#
# Edge cases:
#   - None root             -> True
#   - Single node           -> True
#   - Duplicate values      -> False (strict inequality required)
#   - Node violates ancestor constraint (not just parent) -> must use bounds, not just parent check


class TreeNode:
    def __init__(self, val=0, left=None, right=None):
        self.val = val
        self.left = left
        self.right = right


# === APPROACH 1: INORDER TRAVERSAL — O(n) ===
# "Inorder of a valid BST is strictly ascending. Track previous value."

def is_valid_bst_inorder(root):
    prev = [float('-inf')]

    def inorder(node):
        if not node:
            return True
        if not inorder(node.left):
            return False
        if node.val <= prev[0]:
            return False
        prev[0] = node.val
        return inorder(node.right)

    return inorder(root)


# === APPROACH 2: MIN/MAX BOUNDS — O(n) ===
# "Every node must be strictly inside the range inherited from its ancestors."

def is_valid_bst_bounds(root, min_val=float('-inf'), max_val=float('inf')):
    if not root:
        return True
    if root.val <= min_val or root.val >= max_val:
        return False
    return (is_valid_bst_bounds(root.left, min_val, root.val) and
            is_valid_bst_bounds(root.right, root.val, max_val))


# === APPROACH 3: ITERATIVE INORDER — O(n) ===
# "Use an explicit stack to simulate inorder without recursion."

def is_valid_bst(root):
    stack = []
    prev_val = float('-inf')
    curr = root
    while curr or stack:
        while curr:
            stack.append(curr)
            curr = curr.left
        curr = stack.pop()
        if curr.val <= prev_val:
            return False
        prev_val = curr.val
        curr = curr.right
    return True


# --- tests ---
# Valid BST: [2, 1, 3]
t1 = TreeNode(2, TreeNode(1), TreeNode(3))
assert is_valid_bst_inorder(t1) == True
assert is_valid_bst_bounds(t1) == True
assert is_valid_bst(t1) == True

# Invalid: [5, 1, 4, null, null, 3, 6] — 4 is right child of 5 but 4 < 5
t2 = TreeNode(5, TreeNode(1), TreeNode(4, TreeNode(3), TreeNode(6)))
assert is_valid_bst_inorder(t2) == False
assert is_valid_bst_bounds(t2) == False
assert is_valid_bst(t2) == False

# Tricky: [10, 5, 15, null, null, 6, 20] — 6 is in right subtree but 6 < 10
t3 = TreeNode(10, TreeNode(5), TreeNode(15, TreeNode(6), TreeNode(20)))
assert is_valid_bst(t3) == False

assert is_valid_bst(None) == True
assert is_valid_bst(TreeNode(1)) == True

print(is_valid_bst(t1))  # True
print(is_valid_bst(t2))  # False
print(is_valid_bst(t3))  # False
