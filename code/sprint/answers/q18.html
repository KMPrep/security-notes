# Q18 - Lowest Common Ancestor of a Binary Search Tree
# Given a BST and two values p and q, find their lowest common ancestor (LCA).
# LCA is the deepest node that has both p and q as descendants (a node is a descendant of itself).
#
# Approach:
#   - Naive (path-based): O(n) — find root-to-node paths, return last common node.
#   - BST recursive:      O(h) — use BST property to decide left/right/here at each step.
#   - BST iterative:      O(h) — same logic without recursion stack.
#
# Key insight: If both p and q are less than current node -> LCA is in left subtree.
# If both greater -> right subtree. Otherwise, current node is the LCA.
#
# Edge cases:
#   - p or q equals root -> root is LCA
#   - p is ancestor of q -> p is LCA
#   - None root          -> None


class TreeNode:
    def __init__(self, val=0, left=None, right=None):
        self.val = val
        self.left = left
        self.right = right


# === APPROACH 1: PATH-BASED NAIVE — O(n) ===
# "Find path to each node; last node in common prefix is LCA."

def lca_naive(root, p, q):
    def find_path(node, target, path):
        if not node:
            return False
        path.append(node)
        if node.val == target:
            return True
        if find_path(node.left, target, path) or find_path(node.right, target, path):
            return True
        path.pop()
        return False

    path_p, path_q = [], []
    find_path(root, p, path_p)
    find_path(root, q, path_q)
    lca = None
    for a, b in zip(path_p, path_q):
        if a.val == b.val:
            lca = a
        else:
            break
    return lca.val if lca else None


# === APPROACH 2: BST RECURSIVE — O(h) ===
# "Both smaller -> go left. Both larger -> go right. Diverge -> current node is LCA."

def lca_recursive(root, p, q):
    if not root:
        return None
    if p < root.val and q < root.val:
        return lca_recursive(root.left, p, q)
    if p > root.val and q > root.val:
        return lca_recursive(root.right, p, q)
    return root.val


# === APPROACH 3: BST ITERATIVE — O(h) ===
# "Same BST navigation logic; no recursion needed."

def lca_bst(root, p, q):
    if not root:
        return None
    node = root
    while node:
        if p < node.val and q < node.val:
            node = node.left
        elif p > node.val and q > node.val:
            node = node.right
        else:
            return node.val
    return None


# --- tests ---
# BST: [6, 2, 8, 0, 4, 7, 9, null, null, 3, 5]
root = TreeNode(6,
    TreeNode(2, TreeNode(0), TreeNode(4, TreeNode(3), TreeNode(5))),
    TreeNode(8, TreeNode(7), TreeNode(9)))

assert lca_naive(root, 2, 8) == 6
assert lca_recursive(root, 2, 8) == 6
assert lca_bst(root, 2, 8) == 6
assert lca_bst(root, 2, 4) == 2
assert lca_bst(root, 0, 5) == 2
assert lca_bst(root, 7, 9) == 8
assert lca_bst(root, 3, 5) == 4

print(lca_bst(root, 2, 8))  # 6
print(lca_bst(root, 2, 4))  # 2
print(lca_bst(root, 7, 9))  # 8
