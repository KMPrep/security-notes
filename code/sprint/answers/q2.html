# Q2 - Longest Substring Without Repeating Characters
# Find the length of the longest substring with no repeating characters.
#
# Approach:
#   - Naive:          O(n^2) — for each start position, expand right until duplicate.
#   - No built-ins:   O(n)   — sliding window with manual dict (char -> last seen index).
#   - With built-ins: O(n)   — same, using enumerate() for cleaner iteration.
#
# Edge cases:
#   - Empty string           -> 0
#   - All same characters    -> 1
#   - All unique characters  -> len(s)
#   - None input             -> TypeError


# === APPROACH 1: NAIVE — O(n^2) ===
# "For every start, expand right and track seen chars — reset on duplicate."

def longest_no_repeat_naive(s):
    if s is None:
        raise TypeError("Input cannot be None")
    max_len = 0
    for i in range(len(s)):
        seen = {}
        for j in range(i, len(s)):
            if s[j] in seen:
                break
            seen[s[j]] = True
            max_len = max(max_len, j - i + 1)
    return max_len


# === APPROACH 2: SLIDING WINDOW — NO BUILT-INS — O(n) ===
# "Store last-seen index of each char. On duplicate, jump left pointer past it."

def longest_no_repeat_no_builtins(s):
    if s is None:
        raise TypeError("Input cannot be None")
    last_seen = {}
    max_len = 0
    left = 0
    for right in range(len(s)):
        char = s[right]
        if char in last_seen and last_seen[char] >= left:
            left = last_seen[char] + 1
        last_seen[char] = right
        window = right - left + 1
        if window > max_len:
            max_len = window
    return max_len


# === APPROACH 3: WITH BUILT-INS — O(n) ===
# "Same window logic with enumerate() and max()."

def longest_no_repeat(s):
    if s is None:
        raise TypeError("Input cannot be None")
    if not isinstance(s, str):
        raise TypeError(f"Expected str, got {type(s).__name__}")

    last_seen = {}
    left = 0
    max_len = 0
    for right, char in enumerate(s):
        if char in last_seen and last_seen[char] >= left:
            left = last_seen[char] + 1
        last_seen[char] = right
        max_len = max(max_len, right - left + 1)
    return max_len


# --- tests ---
assert longest_no_repeat_naive("abcabcbb") == 3
assert longest_no_repeat_no_builtins("abcabcbb") == 3
assert longest_no_repeat("abcabcbb") == 3
assert longest_no_repeat("bbbbb") == 1
assert longest_no_repeat("pwwkew") == 3
assert longest_no_repeat("") == 0
assert longest_no_repeat("abcd") == 4

print(longest_no_repeat("abcabcbb"))  # 3
print(longest_no_repeat("bbbbb"))     # 1
print(longest_no_repeat("pwwkew"))    # 3
