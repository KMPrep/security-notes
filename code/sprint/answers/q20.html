# Q20 - Clone Graph
# Given a node in a connected undirected graph, return a deep copy (clone) of the graph.
#
# Approach:
#   - BFS:           O(V+E) — queue-based; HashMap from original -> clone prevents revisiting.
#   - DFS recursive: O(V+E) — recursive with HashMap; cleaner but limited by recursion depth.
#   - DFS iterative: O(V+E) — explicit stack + HashMap; avoids stack overflow on large graphs.
#
# Edge cases:
#   - None input (empty graph)   -> None
#   - Single node, no neighbors  -> clone with empty neighbors list
#   - Cycles                     -> HashMap prevents infinite loop


from collections import deque


class Node:
    def __init__(self, val=0, neighbors=None):
        self.val = val
        self.neighbors = neighbors if neighbors is not None else []


# === APPROACH 1: BFS — O(V+E) ===

def clone_graph_bfs(node):
    if not node:
        return None
    cloned = {node: Node(node.val)}
    queue = deque([node])
    while queue:
        curr = queue.popleft()
        for neighbor in curr.neighbors:
            if neighbor not in cloned:
                cloned[neighbor] = Node(neighbor.val)
                queue.append(neighbor)
            cloned[curr].neighbors.append(cloned[neighbor])
    return cloned[node]


# === APPROACH 2: DFS RECURSIVE — O(V+E) ===
# "Clone each node once; HashMap lookup handles cycles and shared neighbors."

def clone_graph_dfs(node, cloned=None):
    if not node:
        return None
    if cloned is None:
        cloned = {}
    if node in cloned:
        return cloned[node]
    clone = Node(node.val)
    cloned[node] = clone
    for neighbor in node.neighbors:
        clone.neighbors.append(clone_graph_dfs(neighbor, cloned))
    return clone


# === APPROACH 3: DFS ITERATIVE — O(V+E) ===
# "Same HashMap approach with an explicit stack instead of the call stack."

def clone_graph(node):
    if not node:
        return None
    cloned = {node: Node(node.val)}
    stack = [node]
    while stack:
        curr = stack.pop()
        for neighbor in curr.neighbors:
            if neighbor not in cloned:
                cloned[neighbor] = Node(neighbor.val)
                stack.append(neighbor)
            cloned[curr].neighbors.append(cloned[neighbor])
    return cloned[node]


# --- tests ---
# Graph: 1 -- 2 -- 3 -- 4 -- 1  (cycle)
n1, n2, n3, n4 = Node(1), Node(2), Node(3), Node(4)
n1.neighbors = [n2, n4]
n2.neighbors = [n1, n3]
n3.neighbors = [n2, n4]
n4.neighbors = [n1, n3]

for clone_fn in [clone_graph_bfs, clone_graph_dfs, clone_graph]:
    c = clone_fn(n1)
    assert c is not n1                            # must be a new object
    assert c.val == 1
    assert len(c.neighbors) == 2
    neighbor_vals = sorted(nb.val for nb in c.neighbors)
    assert neighbor_vals == [2, 4]

assert clone_graph(None) is None

single = Node(1)
sc = clone_graph(single)
assert sc is not single
assert sc.val == 1
assert sc.neighbors == []

print("All clone_graph tests passed")
print(f"Clone val: {clone_graph(n1).val}, neighbors: {sorted(nb.val for nb in clone_graph(n1).neighbors)}")
# Clone val: 1, neighbors: [2, 4]
