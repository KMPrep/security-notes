# Q21 - Climbing Stairs
# You can climb 1 or 2 steps at a time. How many distinct ways to reach step n?
#
# Approach:
#   - Naive recursive:   O(2^n) — each call branches into 1-step and 2-step choices.
#   - Memoized:          O(n)   — top-down DP; cache each subproblem.
#   - Bottom-up O(1):    O(n)   — only keep the last two values (Fibonacci sliding window).
#
# Key insight: ways(n) = ways(n-1) + ways(n-2). Identical to Fibonacci.
# n-1: took 1 step to arrive here. n-2: took 2 steps to arrive here.
#
# Edge cases:
#   - n == 0    -> 1 (one way: do nothing)
#   - n == 1    -> 1
#   - n == 2    -> 2
#   - Negative n -> ValueError


# === APPROACH 1: NAIVE RECURSIVE — O(2^n) ===
# "Correct but exponential — only explain this to introduce memoization."

def climb_naive(n):
    if n < 0:
        raise ValueError("n must be non-negative")
    if n <= 1:
        return 1
    return climb_naive(n - 1) + climb_naive(n - 2)


# === APPROACH 2: MEMOIZED TOP-DOWN — O(n) ===
# "Cache already-computed subproblems to avoid redundant work."

def climb_memo(n, memo=None):
    if n < 0:
        raise ValueError("n must be non-negative")
    if memo is None:
        memo = {}
    if n <= 1:
        return 1
    if n in memo:
        return memo[n]
    memo[n] = climb_memo(n - 1, memo) + climb_memo(n - 2, memo)
    return memo[n]


# === APPROACH 3: BOTTOM-UP O(1) SPACE ===
# "Slide a two-variable window forward — no array needed."

def climb_stairs(n):
    if n is None:
        raise TypeError("Input cannot be None")
    if not isinstance(n, int) or n < 0:
        raise ValueError("n must be a non-negative integer")
    if n <= 1:
        return 1
    prev2, prev1 = 1, 1
    for _ in range(2, n + 1):
        curr = prev1 + prev2
        prev2 = prev1
        prev1 = curr
    return prev1


# --- tests ---
assert climb_naive(1) == 1
assert climb_naive(2) == 2
assert climb_naive(5) == 8
assert climb_memo(5) == 8
assert climb_stairs(0) == 1
assert climb_stairs(1) == 1
assert climb_stairs(2) == 2
assert climb_stairs(5) == 8
assert climb_stairs(10) == 89

for i in range(8):
    print(f"climb_stairs({i}) = {climb_stairs(i)}")
