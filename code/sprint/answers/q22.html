# Q22 - Coin Change (Minimum Coins)
# Given coin denominations and a target amount, find the fewest coins to make that amount.
# Return -1 if the amount cannot be made.
#
# Approach:
#   - Memoized:      O(amount * len(coins)) — top-down DP with cache.
#   - Bottom-up DP:  O(amount * len(coins)) — iterative; dp[i] = min coins for amount i.
#   - With path:     O(amount * len(coins)) — bottom-up + reconstruct which coins were used.
#
# Key recurrence: dp[i] = min(dp[i - coin] + 1) for each coin <= i
#
# Edge cases:
#   - amount == 0        -> 0 coins
#   - No valid combo     -> -1
#   - Coin > amount      -> skip that coin
#   - None / negative    -> TypeError / ValueError


# === APPROACH 1: MEMOIZED RECURSIVE — O(amount * len(coins)) ===

def coin_change_memo(coins, amount):
    if amount < 0:
        raise ValueError("amount must be non-negative")
    memo = {}
    def dp(remaining):
        if remaining == 0:
            return 0
        if remaining < 0:
            return float('inf')
        if remaining in memo:
            return memo[remaining]
        memo[remaining] = min(dp(remaining - c) + 1 for c in coins)
        return memo[remaining]
    result = dp(amount)
    return result if result != float('inf') else -1


# === APPROACH 2: BOTTOM-UP DP — O(amount * len(coins)) ===
# "Build dp from 0 to amount. dp[i] = min coins to make exactly i."

def coin_change_dp(coins, amount):
    if amount < 0:
        raise ValueError("amount must be non-negative")
    dp = [float('inf')] * (amount + 1)
    dp[0] = 0
    for i in range(1, amount + 1):
        for coin in coins:
            if coin <= i:
                dp[i] = min(dp[i], dp[i - coin] + 1)
    return dp[amount] if dp[amount] != float('inf') else -1


# === APPROACH 3: BOTTOM-UP WITH PATH RECONSTRUCTION ===
# "Google follow-up: which coins were used?"

def coin_change(coins, amount):
    if coins is None or amount is None:
        raise TypeError("Input cannot be None")
    if not isinstance(amount, int) or amount < 0:
        raise ValueError("amount must be a non-negative integer")

    dp = [float('inf')] * (amount + 1)
    parent = [-1] * (amount + 1)
    dp[0] = 0

    for i in range(1, amount + 1):
        for coin in coins:
            if coin <= i and dp[i - coin] + 1 < dp[i]:
                dp[i] = dp[i - coin] + 1
                parent[i] = coin

    if dp[amount] == float('inf'):
        return -1, []

    path = []
    rem = amount
    while rem > 0:
        path.append(parent[rem])
        rem -= parent[rem]
    return dp[amount], path


# --- tests ---
assert coin_change_memo([1, 5, 11], 15) == 3
assert coin_change_dp([1, 5, 11], 15) == 3
assert coin_change([1, 5, 11], 15) == (3, [5, 5, 5])
assert coin_change([2], 3) == (-1, [])
assert coin_change([1], 0) == (0, [])
assert coin_change([1, 2, 5], 11) == (3, [5, 5, 1])
assert coin_change_dp([1, 2, 5], 11) == 3

count, used = coin_change([1, 5, 11], 15)
print(f"Min coins: {count}, used: {used}")      # Min coins: 3, used: [5, 5, 5]

count2, used2 = coin_change([1, 2, 5], 11)
print(f"Min coins: {count2}, used: {used2}")    # Min coins: 3, used: [5, 5, 1]
