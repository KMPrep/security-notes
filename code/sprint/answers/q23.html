# Q23 - Longest Increasing Subsequence (LIS)
# Find the length of the longest strictly increasing subsequence.
#
# Approach:
#   - DP O(n^2):          dp[i] = length of LIS ending at index i.
#   - DP with sequence:   same, but also reconstruct the actual subsequence.
#   - Binary search O(n log n): maintain a sorted `tails` array; binary search to place each element.
#
# Key insight (binary search): tails[i] = the smallest tail of any increasing subsequence of length i+1.
# tails is always sorted. For each num: find its position in tails with bisect_left.
# Replace or append. Length of tails at end = LIS length.
#
# Edge cases:
#   - Empty array        -> 0
#   - All same values    -> 1
#   - Already sorted     -> n
#   - None input         -> TypeError


import bisect


# === APPROACH 1: DP — O(n^2) ===
# "For each i, check every j < i: if nums[j] < nums[i], we can extend."

def lis_dp(nums):
    if nums is None:
        raise TypeError("Input cannot be None")
    if not nums:
        return 0
    n = len(nums)
    dp = [1] * n
    for i in range(1, n):
        for j in range(i):
            if nums[j] < nums[i]:
                dp[i] = max(dp[i], dp[j] + 1)
    return max(dp)


# === APPROACH 2: DP WITH SEQUENCE RECONSTRUCTION — O(n^2) ===

def lis_dp_with_sequence(nums):
    if not nums:
        return 0, []
    n = len(nums)
    dp = [1] * n
    parent = [-1] * n
    for i in range(1, n):
        for j in range(i):
            if nums[j] < nums[i] and dp[j] + 1 > dp[i]:
                dp[i] = dp[j] + 1
                parent[i] = j
    max_len = max(dp)
    idx = dp.index(max_len)
    seq = []
    while idx != -1:
        seq.append(nums[idx])
        idx = parent[idx]
    return max_len, seq[::-1]


# === APPROACH 3: BINARY SEARCH — O(n log n) ===
# "tails[i] = smallest possible tail of an IS of length i+1.
#  bisect_left finds where num fits in tails; replace or extend."

def lis(nums):
    if nums is None:
        raise TypeError("Input cannot be None")
    if not isinstance(nums, list):
        raise TypeError(f"Expected list, got {type(nums).__name__}")
    if not nums:
        return 0

    tails = []
    for num in nums:
        pos = bisect.bisect_left(tails, num)
        if pos == len(tails):
            tails.append(num)
        else:
            tails[pos] = num

    return len(tails)


# --- tests ---
assert lis_dp([10, 9, 2, 5, 3, 7, 101, 18]) == 4
assert lis([10, 9, 2, 5, 3, 7, 101, 18]) == 4
assert lis([0, 1, 0, 3, 2, 3]) == 4
assert lis([7, 7, 7, 7]) == 1
assert lis([1, 2, 3, 4, 5]) == 5
assert lis([]) == 0
assert lis([3, 10, 2, 1, 20]) == 3

length, seq = lis_dp_with_sequence([10, 9, 2, 5, 3, 7, 101, 18])
print(f"LIS length: {length}, one valid sequence: {seq}")  # 4, e.g. [2, 3, 7, 101]
print(f"Fast LIS: {lis([10, 9, 2, 5, 3, 7, 101, 18])}")   # 4
