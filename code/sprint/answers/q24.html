# Q24 - Word Break
# Given string s and a dictionary of words, determine if s can be segmented
# into a space-separated sequence of one or more dictionary words.
#
# Approach:
#   - Naive recursive:  O(2^n)    — try every split point, recursing on both halves.
#   - Memoized:         O(n^2)    — cache results for each suffix starting at index i.
#   - Bottom-up DP:     O(n^2)    — dp[i] = True if s[:i] can be fully segmented.
#
# Key recurrence: dp[i] = any(dp[j] and s[j:i] in word_set) for j in range(i)
#
# Edge cases:
#   - Empty string         -> True (empty segmentation)
#   - Word not in dict     -> False
#   - Single char match    -> True
#   - None input           -> TypeError


# === APPROACH 1: NAIVE RECURSIVE — O(2^n) ===
# "Try every split — exponential without memoization."

def word_break_naive(s, word_dict):
    if s is None or word_dict is None:
        raise TypeError("Input cannot be None")
    if not s:
        return True
    for i in range(1, len(s) + 1):
        if s[:i] in word_dict and word_break_naive(s[i:], word_dict):
            return True
    return False


# === APPROACH 2: MEMOIZED — O(n^2) ===
# "Cache by the current remaining string to avoid recomputing same suffixes."

def word_break_memo(s, word_dict, memo=None):
    if memo is None:
        memo = {}
    if s in memo:
        return memo[s]
    if not s:
        return True
    for i in range(1, len(s) + 1):
        if s[:i] in word_dict and word_break_memo(s[i:], word_dict, memo):
            memo[s] = True
            return True
    memo[s] = False
    return False


# === APPROACH 3: BOTTOM-UP DP — O(n^2) ===
# "dp[i] = can s[:i] be segmented?
#  For each i, check all j < i where dp[j] is True and s[j:i] is a dict word."

def word_break(s, word_dict):
    if s is None or word_dict is None:
        raise TypeError("Input cannot be None")
    if not isinstance(s, str):
        raise TypeError(f"Expected str, got {type(s).__name__}")

    word_set = set(word_dict)
    n = len(s)
    dp = [False] * (n + 1)
    dp[0] = True

    for i in range(1, n + 1):
        for j in range(i):
            if dp[j] and s[j:i] in word_set:
                dp[i] = True
                break

    return dp[n]


# --- tests ---
assert word_break_naive("leetcode", ["leet", "code"]) == True
assert word_break_memo("leetcode", ["leet", "code"]) == True
assert word_break("leetcode", ["leet", "code"]) == True
assert word_break("applepenapple", ["apple", "pen"]) == True
assert word_break("catsandog", ["cats", "dog", "sand", "and", "cat"]) == False
assert word_break("", ["a"]) == True
assert word_break("a", ["b"]) == False
assert word_break("aaaaaaa", ["aaaa", "aaa"]) == True

print(word_break("leetcode", ["leet", "code"]))                           # True
print(word_break("applepenapple", ["apple", "pen"]))                      # True
print(word_break("catsandog", ["cats", "dog", "sand", "and", "cat"]))    # False
