# Q25 - House Robber
# Cannot rob two adjacent houses. Find the maximum amount you can rob.
#
# Approach:
#   - Naive recursive:   O(2^n) — try all non-adjacent subsets.
#   - Memoized:          O(n)   — top-down DP; cache results per index.
#   - Bottom-up O(1):    O(n)   — slide two variables forward (no array needed).
#
# Key recurrence: dp[i] = max(dp[i-1], dp[i-2] + nums[i])
# At each house: either skip it (take dp[i-1]) or rob it (dp[i-2] + nums[i]).
#
# Edge cases:
#   - Empty array   -> 0
#   - Single house  -> that house's value
#   - Two houses    -> max of the two
#   - All zeros     -> 0
#   - None input    -> TypeError


# === APPROACH 1: NAIVE RECURSIVE — O(2^n) ===
# "At each index: rob this house (skip next) or skip this house."

def rob_naive(nums, i=None):
    if nums is None:
        raise TypeError("Input cannot be None")
    if i is None:
        i = len(nums) - 1
    if i < 0:
        return 0
    return max(rob_naive(nums, i - 1), nums[i] + rob_naive(nums, i - 2))


# === APPROACH 2: MEMOIZED — O(n) ===
# "Cache each house index to avoid recomputing the same subproblem."

def rob_memo(nums):
    if nums is None:
        raise TypeError("Input cannot be None")
    memo = {}
    def dp(i):
        if i < 0:
            return 0
        if i in memo:
            return memo[i]
        memo[i] = max(dp(i - 1), nums[i] + dp(i - 2))
        return memo[i]
    return dp(len(nums) - 1)


# === APPROACH 3: BOTTOM-UP O(1) SPACE ===
# "prev2 = best loot up to i-2; prev1 = best loot up to i-1.
#  Roll them forward one house at a time."

def rob(nums):
    if nums is None:
        raise TypeError("Input cannot be None")
    if not isinstance(nums, list):
        raise TypeError(f"Expected list, got {type(nums).__name__}")
    if not nums:
        return 0

    prev2, prev1 = 0, 0
    for amount in nums:
        curr = max(prev1, prev2 + amount)
        prev2 = prev1
        prev1 = curr
    return prev1


# --- tests ---
assert rob_naive([1, 2, 3, 1]) == 4
assert rob_memo([1, 2, 3, 1]) == 4
assert rob([1, 2, 3, 1]) == 4
assert rob([2, 7, 9, 3, 1]) == 12
assert rob([]) == 0
assert rob([5]) == 5
assert rob([2, 1]) == 2
assert rob([0, 0, 0]) == 0
assert rob([1, 2]) == 2

print(rob([2, 7, 9, 3, 1]))  # 12
print(rob([1, 2, 3, 1]))     # 4
print(rob([]))                # 0
