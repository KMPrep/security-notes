# Q3 - Maximum Subarray Sum (Kadane's Algorithm)
# Find the contiguous subarray with the largest sum.
#
# Approach:
#   - Naive:           O(n^2) — try all subarrays with a nested loop.
#   - Kadane's:        O(n)   — at each position, extend current run or start fresh.
#   - With indices:    O(n)   — Kadane's extended to return the actual subarray.
#
# Key insight: current_sum = max(nums[i], current_sum + nums[i])
# If current_sum ever goes negative, starting fresh from nums[i] is strictly better.
#
# Edge cases:
#   - All negative numbers  -> return the single least-negative element
#   - Single element        -> return that element
#   - All positive          -> sum of entire array
#   - None / non-list input -> TypeError


# === APPROACH 1: NAIVE — O(n^2) ===
# "Sum every possible subarray, track the maximum."

def max_subarray_naive(nums):
    if nums is None:
        raise TypeError("Input cannot be None")
    if not nums:
        return 0
    max_sum = nums[0]
    for i in range(len(nums)):
        current = 0
        for j in range(i, len(nums)):
            current += nums[j]
            if current > max_sum:
                max_sum = current
    return max_sum


# === APPROACH 2: KADANE'S — NO BUILT-INS — O(n) ===
# "Decide at every step: extend the current run, or restart from here."

def max_subarray_kadane(nums):
    if nums is None:
        raise TypeError("Input cannot be None")
    if not isinstance(nums, list):
        raise TypeError(f"Expected list, got {type(nums).__name__}")
    if not nums:
        return 0

    max_sum = nums[0]
    current_sum = nums[0]
    for i in range(1, len(nums)):
        extended = current_sum + nums[i]
        current_sum = nums[i] if nums[i] > extended else extended
        if current_sum > max_sum:
            max_sum = current_sum
    return max_sum


# === APPROACH 3: WITH SUBARRAY INDICES — O(n) ===
# "Google follow-up: return the actual subarray, not just the sum."

def max_subarray(nums):
    if nums is None:
        raise TypeError("Input cannot be None")
    if not isinstance(nums, list):
        raise TypeError(f"Expected list, got {type(nums).__name__}")
    if not nums:
        return 0, []

    max_sum = current_sum = nums[0]
    start = end = temp_start = 0

    for i in range(1, len(nums)):
        if nums[i] > current_sum + nums[i]:
            current_sum = nums[i]
            temp_start = i
        else:
            current_sum += nums[i]

        if current_sum > max_sum:
            max_sum = current_sum
            start = temp_start
            end = i

    return max_sum, nums[start:end + 1]


# --- tests ---
assert max_subarray_naive([-2, 1, -3, 4, -1, 2, 1, -5, 4]) == 6
assert max_subarray_kadane([-2, 1, -3, 4, -1, 2, 1, -5, 4]) == 6
assert max_subarray([-2, 1, -3, 4, -1, 2, 1, -5, 4]) == (6, [4, -1, 2, 1])
assert max_subarray_kadane([-1, -2, -3]) == -1
assert max_subarray_kadane([1]) == 1
assert max_subarray_kadane([1, 2, 3]) == 6

print(max_subarray([-2, 1, -3, 4, -1, 2, 1, -5, 4]))  # (6, [4, -1, 2, 1])
print(max_subarray([-1, -2, -3]))                       # (-1, [-1])
print(max_subarray([1, 2, 3]))                          # (6, [1, 2, 3])
