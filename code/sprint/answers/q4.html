# Q4 - Container With Most Water
# Given an array of heights, find two lines forming a container with maximum water.
# Water held = min(height[left], height[right]) * (right - left).
#
# Approach:
#   - Naive:      O(n^2) — try all pairs.
#   - Two pointer: O(n)  — shrink from both ends; always move the shorter bar inward.
#   - Same algorithm, clean version with built-ins.
#
# Key insight: Moving the taller pointer inward can only decrease or maintain height
# while also decreasing width — so it can never increase area. Always move the shorter.
#
# Edge cases:
#   - Less than 2 heights  -> 0
#   - All equal heights    -> handled naturally
#   - None input           -> TypeError


# === APPROACH 1: NAIVE — O(n^2) ===
# "Check every pair (i, j). Correct but O(n^2) — too slow for large inputs."

def max_water_naive(height):
    if height is None:
        raise TypeError("Input cannot be None")
    max_area = 0
    for i in range(len(height)):
        for j in range(i + 1, len(height)):
            area = min(height[i], height[j]) * (j - i)
            if area > max_area:
                max_area = area
    return max_area


# === APPROACH 2: TWO POINTERS — NO BUILT-INS — O(n) ===
# "Advance whichever pointer has the shorter bar; wider window cannot compensate a shorter bar."

def max_water_two_pointer(height):
    if height is None:
        raise TypeError("Input cannot be None")
    if not isinstance(height, list):
        raise TypeError(f"Expected list, got {type(height).__name__}")
    if len(height) < 2:
        return 0

    left, right = 0, len(height) - 1
    max_area = 0

    while left < right:
        h = height[left] if height[left] < height[right] else height[right]
        area = h * (right - left)
        if area > max_area:
            max_area = area
        if height[left] < height[right]:
            left += 1
        else:
            right -= 1

    return max_area


# === APPROACH 3: WITH BUILT-INS — O(n) ===
# "Same logic using min() and max() for readability."

def max_water(height):
    if height is None:
        raise TypeError("Input cannot be None")
    if not isinstance(height, list):
        raise TypeError(f"Expected list, got {type(height).__name__}")
    if len(height) < 2:
        return 0

    left, right = 0, len(height) - 1
    max_area = 0

    while left < right:
        area = min(height[left], height[right]) * (right - left)
        max_area = max(max_area, area)
        if height[left] < height[right]:
            left += 1
        else:
            right -= 1

    return max_area


# --- tests ---
assert max_water_naive([1, 8, 6, 2, 5, 4, 8, 3, 7]) == 49
assert max_water_two_pointer([1, 8, 6, 2, 5, 4, 8, 3, 7]) == 49
assert max_water([1, 8, 6, 2, 5, 4, 8, 3, 7]) == 49
assert max_water([1, 1]) == 1
assert max_water([4, 3, 2, 1, 4]) == 16
assert max_water([1, 2, 1]) == 2
assert max_water([]) == 0

print(max_water([1, 8, 6, 2, 5, 4, 8, 3, 7]))  # 49
print(max_water([1, 1]))                          # 1
print(max_water([4, 3, 2, 1, 4]))                # 16
