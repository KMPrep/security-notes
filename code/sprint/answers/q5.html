# Q5 - Find All Anagrams in a String
# Return start indices of all anagrams of p in s.
#
# Approach:
#   - Naive:          O(n * k log k) — sort every k-length window and compare to sorted(p).
#   - No built-ins:   O(n)           — sliding window with two int[26] freq arrays.
#   - With built-ins: O(n)           — sliding window with Counter objects.
#
# Edge cases:
#   - p longer than s    -> []
#   - Empty s or p       -> []
#   - No anagrams found  -> []
#   - None input         -> TypeError


from collections import Counter


# === APPROACH 1: NAIVE — O(n * k log k) ===
# "Sort every window — simple but sort overhead per window."

def find_anagrams_naive(s, p):
    if s is None or p is None:
        raise TypeError("Input cannot be None")
    result = []
    k = len(p)
    p_sorted = sorted(p)
    for i in range(len(s) - k + 1):
        if sorted(s[i:i + k]) == p_sorted:
            result.append(i)
    return result


# === APPROACH 2: FREQ ARRAY — NO BUILT-INS — O(n) ===
# "Fixed-size window: increment incoming char, decrement outgoing char, compare arrays."

def find_anagrams_no_builtins(s, p):
    if s is None or p is None:
        raise TypeError("Input cannot be None")
    result = []
    k = len(p)
    if k > len(s):
        return result

    p_freq = [0] * 26
    s_freq = [0] * 26
    a = ord('a')

    for c in p:
        p_freq[ord(c) - a] += 1
    for c in s[:k]:
        s_freq[ord(c) - a] += 1

    if s_freq == p_freq:
        result.append(0)

    for i in range(k, len(s)):
        s_freq[ord(s[i]) - a] += 1
        s_freq[ord(s[i - k]) - a] -= 1
        if s_freq == p_freq:
            result.append(i - k + 1)

    return result


# === APPROACH 3: WITH BUILT-INS — O(n) ===
# "Counter sliding window — add incoming, remove outgoing, delete zero-count keys."

def find_anagrams(s, p):
    if s is None or p is None:
        raise TypeError("Input cannot be None")
    if not isinstance(s, str) or not isinstance(p, str):
        raise TypeError("Both inputs must be strings")

    k = len(p)
    if k > len(s):
        return []

    result = []
    p_count = Counter(p)
    window = Counter(s[:k])

    if window == p_count:
        result.append(0)

    for i in range(k, len(s)):
        char_in = s[i]
        char_out = s[i - k]
        window[char_in] += 1
        window[char_out] -= 1
        if window[char_out] == 0:
            del window[char_out]
        if window == p_count:
            result.append(i - k + 1)

    return result


# --- tests ---
assert find_anagrams_naive("cbaebabacd", "abc") == [0, 6]
assert find_anagrams_no_builtins("cbaebabacd", "abc") == [0, 6]
assert find_anagrams("cbaebabacd", "abc") == [0, 6]
assert find_anagrams("abab", "ab") == [0, 1, 2]
assert find_anagrams("aa", "bb") == []
assert find_anagrams("", "abc") == []
assert find_anagrams("a", "a") == [0]

print(find_anagrams("cbaebabacd", "abc"))  # [0, 6]
print(find_anagrams("abab", "ab"))          # [0, 1, 2]
print(find_anagrams("aa", "bb"))            # []
