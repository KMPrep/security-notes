# Q6 - Valid Parentheses
# Given a string of brackets ()[]{}}, return True if every open bracket is
# closed in the correct order with no unmatched brackets.
#
# Approach:
#   - Naive:          O(n^2) — repeatedly collapse matching adjacent pairs until none remain.
#   - No built-ins:   O(n)   — stack (list) with a manual close->open mapping dict.
#   - With built-ins: O(n)   — same, idiomatic Python with dict.values().
#
# Edge cases:
#   - Empty string          -> True
#   - Only open brackets    -> False
#   - Only close brackets   -> False
#   - Single char           -> False
#   - None input            -> TypeError


# === APPROACH 1: NAIVE — O(n^2) ===
# "Collapse '()', '[]', '{}' repeatedly until nothing collapses."

def is_valid_naive(s):
    if s is None:
        raise TypeError("Input cannot be None")
    while "()" in s or "[]" in s or "{}" in s:
        s = s.replace("()", "").replace("[]", "").replace("{}", "")
    return s == ""


# === APPROACH 2: STACK — NO BUILT-INS — O(n) ===
# "Push on open bracket; on close bracket, pop and verify it matches."

def is_valid_no_builtins(s):
    if s is None:
        raise TypeError("Input cannot be None")
    stack = []
    matching = {')': '(', ']': '[', '}': '{'}
    for char in s:
        if char in ('(', '[', '{'):
            stack.append(char)
        elif char in (')', ']', '}'):
            if not stack or stack[-1] != matching[char]:
                return False
            stack.pop()
    return len(stack) == 0


# === APPROACH 3: WITH BUILT-INS — O(n) ===
# "Same stack approach; use dict.values() to identify open brackets."

def is_valid(s):
    if s is None:
        raise TypeError("Input cannot be None")
    if not isinstance(s, str):
        raise TypeError(f"Expected str, got {type(s).__name__}")

    stack = []
    close_to_open = {')': '(', ']': '[', '}': '{'}

    for char in s:
        if char in close_to_open:
            if not stack or stack[-1] != close_to_open[char]:
                return False
            stack.pop()
        elif char in close_to_open.values():
            stack.append(char)

    return not stack


# --- tests ---
assert is_valid_naive("()") == True
assert is_valid_naive("()[]{}") == True
assert is_valid_naive("(]") == False
assert is_valid_no_builtins("({[]})") == True
assert is_valid_no_builtins("([)]") == False
assert is_valid("") == True
assert is_valid("({[]})") == True
assert is_valid("([)]") == False
assert is_valid("{[]") == False
assert is_valid(")") == False

print(is_valid("({[]})"))   # True
print(is_valid("([)]"))     # False
print(is_valid(""))         # True
print(is_valid(")"))        # False
