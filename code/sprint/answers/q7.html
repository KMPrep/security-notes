# Q7 - Group Anagrams
# Given a list of strings, group them by their anagram family.
#
# Approach:
#   - Naive:           O(n^2 * k) — compare every pair of strings using sorted().
#   - Freq-tuple key:  O(n * k)   — 26-int frequency tuple as dict key (no sorting).
#   - Sorted key:      O(n * k log k) — sorted string as dict key (most common standard answer).
#
# Edge cases:
#   - Empty list         -> []
#   - Single string      -> [[that string]]
#   - Empty strings      -> [""] is its own group
#   - None input         -> TypeError


from collections import defaultdict


# === APPROACH 1: NAIVE — O(n^2 * k) ===
# "Mark each string as assigned once grouped. Compare via sorted()."

def group_anagrams_naive(strs):
    if strs is None:
        raise TypeError("Input cannot be None")
    if not strs:
        return []
    groups = []
    assigned = [False] * len(strs)
    for i in range(len(strs)):
        if assigned[i]:
            continue
        group = [strs[i]]
        assigned[i] = True
        for j in range(i + 1, len(strs)):
            if not assigned[j] and sorted(strs[i]) == sorted(strs[j]):
                group.append(strs[j])
                assigned[j] = True
        groups.append(group)
    return groups


# === APPROACH 2: FREQ-TUPLE KEY — O(n * k) ===
# "Count char frequencies into a 26-int tuple — avoids sorting entirely."

def group_anagrams_freq(strs):
    if strs is None:
        raise TypeError("Input cannot be None")
    if not isinstance(strs, list):
        raise TypeError(f"Expected list, got {type(strs).__name__}")
    groups = {}
    for s in strs:
        freq = [0] * 26
        for c in s:
            freq[ord(c) - ord('a')] += 1
        key = tuple(freq)
        if key not in groups:
            groups[key] = []
        groups[key].append(s)
    return list(groups.values())


# === APPROACH 3: SORTED KEY — O(n * k log k) ===
# "Sort each string to get its canonical anagram key."

def group_anagrams(strs):
    if strs is None:
        raise TypeError("Input cannot be None")
    if not isinstance(strs, list):
        raise TypeError(f"Expected list, got {type(strs).__name__}")
    groups = defaultdict(list)
    for s in strs:
        key = tuple(sorted(s))
        groups[key].append(s)
    return list(groups.values())


# --- helper for order-agnostic comparison ---
def same_groups(result, expected):
    r = sorted([sorted(g) for g in result])
    e = sorted([sorted(g) for g in expected])
    return r == e


# --- tests ---
inp = ["eat", "tea", "tan", "ate", "nat", "bat"]
expected = [["eat", "tea", "ate"], ["tan", "nat"], ["bat"]]

assert same_groups(group_anagrams_naive(inp), expected)
assert same_groups(group_anagrams_freq(inp), expected)
assert same_groups(group_anagrams(inp), expected)
assert group_anagrams([]) == []
assert same_groups(group_anagrams([""]), [[""]])
assert same_groups(group_anagrams(["a"]), [["a"]])

print(group_anagrams(["eat", "tea", "tan", "ate", "nat", "bat"]))
# e.g. [['eat', 'tea', 'ate'], ['tan', 'nat'], ['bat']]
