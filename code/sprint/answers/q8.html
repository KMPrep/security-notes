# Q8 - Top K Frequent Elements
# Return the k most frequent elements in the array.
#
# Approach:
#   - Naive:          O(n log n) — count frequencies, sort by frequency descending.
#   - Min-heap:       O(n log k) — maintain a min-heap of size k; evict least frequent.
#   - Bucket sort:    O(n)       — bucket by frequency (index = count), scan from top.
#
# Edge cases:
#   - k == number of unique elements  -> return all unique elements
#   - k == 1                          -> return single most frequent element
#   - All same frequency              -> any k elements valid
#   - None input                      -> TypeError


import heapq
from collections import Counter


# === APPROACH 1: NAIVE — O(n log n) ===
# "Count frequencies, sort descending, slice top k."

def top_k_naive(nums, k):
    if nums is None:
        raise TypeError("Input cannot be None")
    count = {}
    for n in nums:
        count[n] = count.get(n, 0) + 1
    sorted_items = sorted(count.items(), key=lambda x: x[1], reverse=True)
    return [item[0] for item in sorted_items[:k]]


# === APPROACH 2: MIN-HEAP — O(n log k) ===
# "Push each (freq, num) onto a min-heap; pop when size exceeds k.
#  The heap always holds the k most frequent seen so far."

def top_k_heap(nums, k):
    if nums is None:
        raise TypeError("Input cannot be None")
    count = {}
    for n in nums:
        count[n] = count.get(n, 0) + 1
    heap = []
    for num, freq in count.items():
        heapq.heappush(heap, (freq, num))
        if len(heap) > k:
            heapq.heappop(heap)
    return [num for freq, num in heap]


# === APPROACH 3: BUCKET SORT — O(n) ===
# "Bucket index = frequency. Max possible freq = n. Scan buckets from high to low."

def top_k_frequent(nums, k):
    if nums is None:
        raise TypeError("Input cannot be None")
    if not isinstance(nums, list):
        raise TypeError(f"Expected list, got {type(nums).__name__}")

    count = Counter(nums)
    buckets = [[] for _ in range(len(nums) + 1)]
    for num, freq in count.items():
        buckets[freq].append(num)

    result = []
    for i in range(len(buckets) - 1, 0, -1):
        for num in buckets[i]:
            result.append(num)
            if len(result) == k:
                return result
    return result


# --- tests ---
assert set(top_k_naive([1, 1, 1, 2, 2, 3], 2)) == {1, 2}
assert set(top_k_heap([1, 1, 1, 2, 2, 3], 2)) == {1, 2}
assert set(top_k_frequent([1, 1, 1, 2, 2, 3], 2)) == {1, 2}
assert top_k_frequent([1], 1) == [1]
assert set(top_k_frequent([4, 1, -1, 2, -1, 2, 3], 2)) == {-1, 2}

print(top_k_frequent([1, 1, 1, 2, 2, 3], 2))       # [1, 2]
print(top_k_frequent([4, 1, -1, 2, -1, 2, 3], 2))  # [-1, 2]
