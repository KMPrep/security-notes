# Q9 - Encode and Decode Strings
# Design encode/decode so a list of strings survives a round-trip through one string.
#
# Approach:
#   - Naive (delimiter):   O(n) — join with a rare delimiter (chr(0)).
#                          Fragile: fails if that character appears in a string.
#   - Length-prefix:       O(n) — "<len>#<string>" per chunk. Safe for any characters.
#   - Length-prefix clean: O(n) — same, using generator expression in encode.
#
# Edge cases:
#   - Strings containing '#'   -> length-prefix handles correctly
#   - Empty string in list     -> encoded as "0#"
#   - Empty list               -> encoded as "" -> decoded as []
#   - None input               -> TypeError


# === APPROACH 1: NAIVE (DELIMITER) — O(n) — FRAGILE ===
# "Works only if strings never contain the delimiter character."

DELIM = chr(0)  # null byte — rare but not guaranteed safe

def encode_naive(strings):
    return DELIM.join(strings)

def decode_naive(s):
    return s.split(DELIM)


# === APPROACH 2: LENGTH-PREFIX — NO BUILT-INS — O(n) ===
# "Prepend each string with its byte-length and '#'. Parsing reads the length
#  to know exactly how many characters to consume — no delimiter ambiguity."

def encode_no_builtins(strings):
    if strings is None:
        raise TypeError("Input cannot be None")
    result = ""
    for s in strings:
        result += str(len(s)) + "#" + s
    return result

def decode_no_builtins(s):
    if s is None:
        raise TypeError("Input cannot be None")
    result = []
    i = 0
    while i < len(s):
        j = i
        while s[j] != "#":
            j += 1
        length = int(s[i:j])
        result.append(s[j + 1: j + 1 + length])
        i = j + 1 + length
    return result


# === APPROACH 3: WITH BUILT-INS — O(n) ===
# "Generator expression in encode; str.index() to locate '#' in decode."

def encode(strings):
    if strings is None:
        raise TypeError("Input cannot be None")
    if not isinstance(strings, list):
        raise TypeError(f"Expected list, got {type(strings).__name__}")
    return "".join(f"{len(s)}#{s}" for s in strings)

def decode(s):
    if s is None:
        raise TypeError("Input cannot be None")
    if not isinstance(s, str):
        raise TypeError(f"Expected str, got {type(s).__name__}")
    result = []
    i = 0
    while i < len(s):
        j = s.index("#", i)
        length = int(s[i:j])
        result.append(s[j + 1: j + 1 + length])
        i = j + 1 + length
    return result


# --- tests ---
words = ["hello", "world", "foo"]
assert decode(encode(words)) == words
assert decode(encode(["", "a", "b#c"])) == ["", "a", "b#c"]
assert decode(encode([])) == []
assert decode_no_builtins(encode_no_builtins(words)) == words

enc = encode(["hello", "world", "foo"])
print(enc)                                    # "5#hello5#world3#foo"
print(decode(enc))                            # ['hello', 'world', 'foo']
print(decode(encode(["", "a#b", "c"])))       # ['', 'a#b', 'c']
